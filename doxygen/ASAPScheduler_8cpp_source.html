20 LogicalResult scheduling::scheduleASAP(Problem &prob) {
21  auto allOps = prob.getOperations();
22  // Keep track of ops that don't have a start time yet
23  llvm::SmallVector<Operation *> unscheduledOps;
24  unscheduledOps.insert(unscheduledOps.begin(), allOps.begin(), allOps.end());
25 
26  // We may need multiple attempts to schedule all operations in case the
27  // problem's operation list is not in a topological order w.r.t. the
28  // dependence graph.
29  while (!unscheduledOps.empty()) {
30  // Remember how many unscheduled operations we have at the beginning of this
31  // attempt. This is a fail-safe for cyclic dependence graphs: If we do not
32  // schedule at least one operation per attempt, we have encountered a cycle.
33  unsigned numUnscheduledBefore = unscheduledOps.size();
34 
35  // Set up the worklist for this attempt, and initialize it in reverse order
36  // so that we can pop off its back later.
37  llvm::SmallVector<Operation *> worklist;
38  worklist.insert(worklist.begin(), unscheduledOps.rbegin(),
39  unscheduledOps.rend());
40  unscheduledOps.clear();
41 
42  while (!worklist.empty()) {
43  Operation *op = worklist.pop_back_val();
44 
45  // Operations with no predecessors are scheduled at time step 0
46  if (prob.getDependences(op).empty()) {
47  prob.setStartTime(op, 0);
48  continue;
49  }
50 
51  // op has at least one predecessor. Compute start time as:
52  // max_{p : preds} startTime[p] + latency[linkedOpr[p]]
53  unsigned startTime = 0;
54  bool startTimeIsValid = true;
55  for (auto &dep : prob.getDependences(op)) {
56  Operation *pred = dep.getSource();
57  if (auto predStart = prob.getStartTime(pred)) {
58  // pred is already scheduled
59  auto predOpr = *prob.getLinkedOperatorType(pred);
60  unsigned predLatency = *prob.getLatency(predOpr);
61  startTime = std::max(startTime, *predStart + predLatency);
62  } else {
63  // pred is not yet scheduled, give up and try again later
64  startTimeIsValid = false;
65  break;
66  }
67  }
68 
69  if (startTimeIsValid)
70  prob.setStartTime(op, startTime);
71  else
72  unscheduledOps.push_back(op);
73  }
74 
75  // Fail if no progress was made during this attempt
76  if (numUnscheduledBefore == unscheduledOps.size())
77  return prob.getContainingOp()->emitError() << "dependence cycle detected";
78  }
79 
80  return success();
81 }
