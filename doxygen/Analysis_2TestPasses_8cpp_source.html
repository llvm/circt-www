43 void TestDependenceAnalysisPass::runOnOperation() {
44  MLIRContext *context = &getContext();
45 
46  MemoryDependenceAnalysis analysis(getOperation());
47 
48  getOperation().walk([&](Operation *op) {
49  if (!isa<AffineReadOpInterface, AffineWriteOpInterface>(op))
50  return;
51 
52  SmallVector<Attribute> deps;
53 
54  for (auto dep : analysis.getDependences(op)) {
55  if (dep.dependenceType != mlir::DependenceResult::HasDependence)
56  continue;
57 
58  SmallVector<Attribute> comps;
59  for (auto comp : dep.dependenceComponents) {
60  SmallVector<Attribute> vector;
61  vector.push_back(IntegerAttr::get(IntegerType::get(context, 64),
62  comp.lb.getValue()));
63  vector.push_back(IntegerAttr::get(IntegerType::get(context, 64),
64  comp.ub.getValue()));
65  comps.push_back(ArrayAttr::get(context, vector));
66  }
67 
68  deps.push_back(ArrayAttr::get(context, comps));
69  }
70 
71  auto dependences = ArrayAttr::get(context, deps);
72  op->setAttr("dependences", dependences);
73  });
74 }

94 void TestSchedulingAnalysisPass::runOnOperation() {
95  MLIRContext *context = &getContext();
96 
97  CyclicSchedulingAnalysis analysis = getAnalysis<CyclicSchedulingAnalysis>();
98 
99  getOperation().walk([&](AffineForOp forOp) {
100  if (isa<AffineForOp>(forOp.getBody()->front()))
101  return;
102  CyclicProblem problem = analysis.getProblem(forOp);
103  forOp.getBody()->walk([&](Operation *op) {
104  for (auto dep : problem.getDependences(op)) {
105  assert(!dep.isInvalid());
106  if (dep.isAuxiliary())
107  op->setAttr("dependence", UnitAttr::get(context));
108  }
109  });
110  });
111 }
