32 LogicalResult BackedgeBuilder::clearOrEmitError() {
33  unsigned numInUse = 0;
34  for (Operation *op : edges) {
35  if (!op->use_empty()) {
36  auto diag = op->emitError("backedge of type `")
37  << op->getResult(0).getType() << "`still in use";
38  for (auto user : op->getUsers())
39  diag.attachNote(user->getLoc()) << "used by " << *user;
40  ++numInUse;
41  continue;
42  }
43  if (rewriter)
44  rewriter->eraseOp(op);
45  else
46  op->erase();
47  }
48  edges.clear();
49  if (numInUse > 0)
50  mlir::emitRemark(loc, "abandoned ") << numInUse << " backedges";
51  return success(numInUse == 0);
52 }
