25 /// Compute a hash code for a MemOp. This specialized hash ignores the naming of
26 /// the memory and ports.
27 llvm::hash_code computeHash(MemOp op) {
28  // MemOp attributes
29  llvm::hash_code hash =
30  llvm::hash_combine(op.readLatencyAttr(), op.writeLatencyAttr());
31  hash = llvm::hash_combine(hash, op.depthAttr());
32  hash = llvm::hash_combine(hash, op.ruwAttr());
33 
34  // Result Types
35  return llvm::hash_combine(hash, op->getResultTypes());
36 }

38 /// Compare memory operations for equivalence. Only compares the types of the
39 /// memory and not the name or the memory, or the ports.
40 static bool isEquivalentTo(MemOp lhs, MemOp rhs) {
41  if (lhs == rhs)
42  return true;
43  // Compare attributes.
44  if (lhs.readLatencyAttr() != rhs.readLatencyAttr())
45  return false;
46  if (lhs.writeLatencyAttr() != rhs.writeLatencyAttr())
47  return false;
48  if (lhs.depthAttr() != rhs.depthAttr())
49  return false;
50  if (lhs.ruwAttr() != rhs.ruwAttr())
51  return false;
52  // Compare result types. Taken from operation equivalence.
53  if (lhs->getResultTypes() != rhs->getResultTypes())
54  return false;
55  return true;
56 }

59 struct MemOpInfo : public llvm::DenseMapInfo<MemOp> {
60  static MemOp getEmptyKey() {
61  auto pointer = llvm::DenseMapInfo<void *>::getEmptyKey();
62  return MemOp::getFromOpaquePointer(pointer);
63  }
64  static MemOp getTombstoneKey() {
65  auto pointer = llvm::DenseMapInfo<void *>::getTombstoneKey();
66  return MemOp::getFromOpaquePointer(pointer);
67  }
68  static unsigned getHashValue(MemOp op) { return computeHash(op); }
69  static bool isEqual(MemOp lhs, MemOp rhs) {
70  if (lhs == rhs)
71  return true;
72  if (lhs == getTombstoneKey() || lhs == getEmptyKey() ||
73  rhs == getTombstoneKey() || rhs == getEmptyKey())
74  return false;
75  return isEquivalentTo(lhs, rhs);
76  }
77 };

81 static InstanceOp createInstance(OpBuilder builder, Location loc,
82  StringRef moduleName, StringAttr instanceName,
83  ArrayRef<ModulePortInfo> modulePorts) {
84  // Make a bundle of the inputs and outputs of the specified module.
85  SmallVector<Type, 4> resultTypes;
86  resultTypes.reserve(modulePorts.size());
87  for (auto port : modulePorts)
88  resultTypes.push_back(port.type);

94 /// Get the pohwist for an external module representing a blackbox memory. This
95 /// external module must be compatible with the modules which are generated for
96 /// `vlsi_mem_gen`, which can be found in the rocketchip project.
97 static void
98 getBlackBoxPortsForMemOp(MemOp op, ArrayRef<MemOp::NamedPort> memPorts,
99  SmallVectorImpl<ModulePortInfo> &extPorts) {
100  OpBuilder builder(op);
101  unsigned readPorts = 0;
102  unsigned writePorts = 0;
103  unsigned readWritePorts = 0;
104  for (size_t i = 0, e = memPorts.size(); i != e; ++i) {
105  // Calculate the naming prefix to use based on the kind of the port.
106  std::string prefix;
107  switch (memPorts[i].second) {
108  case MemOp::PortKind::Read:
109  prefix = ("R" + Twine(readPorts++) + "_").str();
110  break;
111  case MemOp::PortKind::Write:
112  prefix = ("W" + Twine(writePorts++) + "_").str();
113  break;
114  case MemOp::PortKind::ReadWrite:
115  prefix = ("RW" + Twine(readWritePorts++) + "_").str();
116  break;
117  }
118  // Flatten the bundle representing a memory port, name-mangling and adding
119  // every field in the bundle to the exter module's port list. All memory
120  // ports have an outer flip, so we just strip this.
121  auto type = op.getResult(i).getType();
122  for (auto bundleElement : type.cast<BundleType>().getElements()) {
123  auto name = (prefix + bundleElement.name.getValue()).str();
124  auto type = bundleElement.type;
125  auto direction = Direction::Input;
126  if (bundleElement.isFlip)
127  direction = Direction::Output;
128  extPorts.push_back(
129  {builder.getStringAttr(name), type, direction, op.getLoc()});
130  }
131  }
132 }

134 /// Create an external module black box representing the memory operation.
135 /// Returns the port list of the external module.
136 static FExtModuleOp
137 createBlackboxModuleForMem(MemOp op, ArrayRef<ModulePortInfo> extPorts) {
138 
139  OpBuilder builder(op->getContext());
140 
141  // The module's name is the name of the memory postfixed with "_ext".
142  auto memName = op.name().str();
143  if (memName.empty())
144  memName = "mem";
145  std::string extName = memName + "_ext";
146 
147  // Create the black box external module.
148  auto extModuleOp = builder.create<FExtModuleOp>(
149  op.getLoc(), builder.getStringAttr(extName), extPorts);
150 
151  // Insert the external module into the circuit. This will rename the
152  // external module if there is a conflict with another module name.
153  auto circuitOp = op->getParentOfType<CircuitOp>();
154  SymbolTable symbolTable(circuitOp);
155  symbolTable.insert(extModuleOp);
156 
157  // Move the external module to the beginning of the circuitOp. Inserting into
158  // the symbol table may have moved the operation to the end of the circuit.
159  extModuleOp->moveBefore(&circuitOp.front());
160 
161  // Add the memory parameters as attributes on the external module. These will
162  // be used by the generator tools to create memory configuration files, or
163  // create the memory itself. We use a named attr list to avoid creating lots
164  // of intermediate dictionary attributes.
165  NamedAttrList attrs(extModuleOp->getAttrDictionary());
166  attrs.set("generator", builder.getStringAttr("FIRRTLMemory"));
167  attrs.set("readLatency", op.readLatencyAttr());
168  attrs.set("writeLatency", op.writeLatencyAttr());
169  attrs.set("depth", op.depthAttr());
170  attrs.set("ruw", op.ruwAttr());
171  extModuleOp->setAttrs(attrs.getDictionary(builder.getContext()));
172 
173  return extModuleOp;
174 }

176 /// Create a regular module to wrap the external module. The wrapper module
177 /// instantiates the external module, and connects all the inputs and outputs
178 /// together. This module will match the bundle return type of the memory op,
179 /// and connects to the flattened parameters of the external module. This is
180 /// done for compatibility with the Scala FIRRTL compiler and it is unclear if
181 /// this will be needed in the long run.
182 static FModuleOp
183 createWrapperModule(MemOp op, ArrayRef<MemOp::NamedPort> memPorts,
184  FExtModuleOp extModuleOp, ArrayRef<ModulePortInfo> extPorts,
185  SmallVectorImpl<ModulePortInfo> &modPorts) {
186  OpBuilder builder(op->getContext());
187 
188  // The wrapper module's name is the name of the memory.
189  auto memName = op.name();
190  if (memName.empty())
191  memName = "mem";
192 
193  // Create a wrapper module with the same type as the memory
194  modPorts.reserve(op.getResults().size());
195  for (size_t i = 0, e = memPorts.size(); i != e; ++i) {
196  auto name = op.getPortName(i);
197  auto type = op.getPortType(i);
198  modPorts.push_back({name, type, Direction::Input, op.getLoc()});
199  }
200  auto moduleOp = builder.create<FModuleOp>(
201  op.getLoc(), builder.getStringAttr(memName), modPorts);
202 
203  // Insert the externaml module into the circuit. This will rename the module
204  // if there is a conflict with another module name.
205  auto circuitOp = op->getParentOfType<CircuitOp>();
206  SymbolTable symbolTable(circuitOp);
207  symbolTable.insert(moduleOp);
208 
209  // Move the module right after the external module, for readability purposes.
210  moduleOp->moveAfter(extModuleOp);
211 
212  // Create the module
213  builder.setInsertionPointToStart(moduleOp.getBodyBlock());
214  auto instanceOp = createInstance(builder, op.getLoc(), extModuleOp.getName(),
215  op.nameAttr(), extPorts);
216 
217  // Connect the ports between the memory module and the instance of the black
218  // box memory module. The outer module has a single bundle representing each
219  // Memory port, while the inner module has a separate field for each memory
220  // port bundle in the memory bundle.
221  auto extResultIt = instanceOp.result_begin();
222  for (auto memPort : moduleOp.getArguments()) {
223  auto memPortType = memPort.getType().cast<FIRRTLType>();
224  for (auto field :
225  llvm::enumerate(memPortType.cast<BundleType>().getElements())) {
226  auto fieldValue =
227  builder.create<SubfieldOp>(op.getLoc(), memPort, field.index());
228  // Create the connection between module arguments and the external module,
229  // making sure that sinks are on the LHS
230  if (!field.value().isFlip)
231  builder.create<ConnectOp>(op.getLoc(), *extResultIt, fieldValue);
232  else
233  builder.create<ConnectOp>(op.getLoc(), fieldValue, *extResultIt);
234  // advance the external module field iterator
235  ++extResultIt;
236  }
237  }
238 
239  return moduleOp;
240 }

242 /// Create a bundle wire for each memory port. This takes all the individual
243 /// fields returned from instantiating the external module, and wraps them in to
244 /// bundles to match the memory return type.
245 static void createWiresForMemoryPorts(OpBuilder builder, Location loc, MemOp op,
246  InstanceOp instanceOp,
247  ArrayRef<ModulePortInfo> extPorts,
248  SmallVectorImpl<Value> &results) {
249 
250  auto extResultIt = instanceOp.result_begin();
251 
252  for (auto memPort : op.getResults()) {
253  // Create a wire bundle for each memory port
254  auto wireOp = builder.create<WireOp>(loc, memPort.getType());
255  results.push_back(wireOp.getResult());
256 
257  // Connect each wire to the corresponding ports in the external module
258  auto wireBundle = memPort.getType().cast<FIRRTLType>();
259  for (auto field :
260  llvm::enumerate(wireBundle.cast<BundleType>().getElements())) {
261  auto fieldValue =
262  builder.create<SubfieldOp>(op.getLoc(), wireOp, field.index());
263  // Create the connection between module arguments and the external module,
264  // making sure that sinks are on the LHS
265  if (field.value().isFlip)
266  builder.create<ConnectOp>(op.getLoc(), fieldValue, *extResultIt);
267  else
268  builder.create<ConnectOp>(op.getLoc(), *extResultIt, fieldValue);
269  // advance the external module field iterator
270  ++extResultIt;
271  }
272  }
273 }

275 static void
276 replaceMemWithWrapperModule(DenseMap<MemOp, FModuleOp, MemOpInfo> &knownMems,
277  MemOp memOp) {
278 
279  // The module we will be replacing the MemOp with. If we don't have a
280  // suitable memory module already created, a new one representing the memory
281  // will be created.
282  FModuleOp moduleOp;
283  SmallVector<ModulePortInfo, 2> modPorts;
284 
285  // Check if we have already created a suitable wrapper module. If we have not
286  // seen a similar memory, create a new wrapper module.
287  auto it = knownMems.find(memOp);
288  auto found = it != knownMems.end();
289  if (found) {
290  // Create an instance of the wrapping module. We have to retrieve the
291  // module port information back from the module.
292  moduleOp = it->second;
293  modPorts = getModulePortInfo(moduleOp);
294  } else {
295  // Get the memory port descriptors. This gives us the name and kind of each
296  // memory port created by the MemOp.
297  auto memPorts = memOp.getPorts();
298 
299  // Get the pohwist for a module which represents the black box memory.
300  // Typically has 1R + 1W memory port, which has 4+5=9 fields.
301  SmallVector<ModulePortInfo, 9> extPortList;
302  getBlackBoxPortsForMemOp(memOp, memPorts, extPortList);
303  auto extModuleOp = createBlackboxModuleForMem(memOp, extPortList);
304  moduleOp = createWrapperModule(memOp, memPorts, extModuleOp, extPortList,
305  modPorts);
306  knownMems[memOp] = moduleOp;
307  }
308 
309  // Create an instance of the wrapping module
310  auto instanceOp =
311  createInstance(OpBuilder(memOp), memOp.getLoc(), moduleOp.getName(),
312  memOp.nameAttr(), modPorts);
313 
314  // Replace the memory operation with the module instance
315  memOp.replaceAllUsesWith(instanceOp.getResults());
316 
317  // If we are using the memory operation as a key in the map, we cannot delete
318  // it yet.
319  if (found)
320  memOp->erase();
321 }

323 /// Replaces all memories which match the predicate function with external
324 /// modules and simple wrapper module which instantiates it. Returns true if
325 /// any memories were replaced.
326 static bool
327 replaceMemsWithWrapperModules(CircuitOp circuit,
328  function_ref<bool(MemOp)> shouldReplace) {
329  /// A set of replaced memory operations. When two memory operations
330  /// share the same types, they can share the same modules.
331  DenseMap<MemOp, FModuleOp, MemOpInfo> knownMems;
332  for (auto fmodule : circuit.getOps<FModuleOp>()) {
333  for (auto memOp : llvm::make_early_inc_range(fmodule.getOps<MemOp>())) {
334  if (shouldReplace(memOp)) {
335  replaceMemWithWrapperModule(knownMems, memOp);
336  }
337  }
338  }
339  // Erase any of the remaining memory operations. These were kept around
340  // to check if other memory operations were equivalent.
341  for (auto it : knownMems)
342  it.first.erase();
343 
344  // Return true if something changed.
345  return !knownMems.empty();
346 }

420 struct BlackBoxMemoryPass : public BlackBoxMemoryBase<BlackBoxMemoryPass> {
421  void runOnOperation() override {
422  // A memory must have read and write latencies of 1 in order to be
423  // blackboxed. In the future this will probably be configurable.
424  auto shouldReplace = [](MemOp memOp) -> bool {
425  return memOp.readLatency() == 1 && memOp.writeLatency() == 1;
426  };
427  auto anythingChanged = false;
428  if (emitWrapper)
429  anythingChanged =
430  replaceMemsWithWrapperModules(getOperation(), shouldReplace);
431  else
432  anythingChanged =
433  replaceMemsWithExtModules(getOperation(), shouldReplace);
434 
435  if (!anythingChanged)
436  markAllAnalysesPreserved();
437  }
438 };
