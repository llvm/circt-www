90 /// Emit the annotated source code for black boxes in a circuit.
91 void BlackBoxReaderPass::runOnOperation() {
92  CircuitOp circuitOp = getOperation();
93  instanceGraph = &getAnalysis<InstanceGraph>();
94  auto context = &getContext();
95 
96  // If this pass has changed anything.
97  bool anythingChanged = false;
98 
99  // Internalize some string attributes for easy reference later.
100 
101  // Determine the target directory and resource file name from the
102  // annotations present on the circuit operation.
103  targetDir = ".";
104  resourceFileName = "firrtl_black_box_resource_files.f";
105 
106  // Process black box annotations on the circuit. Some of these annotations
107  // will affect how the rest of the annotations are resolved.
108  SmallVector<Attribute, 4> filteredAnnos;
109  for (auto annot : AnnotationSet(circuitOp)) {
110  // Handle resource file name annotation.
111  if (annot.isClass(blackBoxResourceFileNameAnnoClass)) {
112  if (auto resourceFN = annot.getMember<StringAttr>("resourceFileName")) {
113  resourceFileName = resourceFN.getValue();
114  continue;
115  }
116 
117  circuitOp->emitError(blackBoxResourceFileNameAnnoClass)
118  << " annotation missing \"resourceFileName\" attribute";
119  signalPassFailure();
120  continue;
121  }
122  filteredAnnos.push_back(annot.getDict());
123 
124  // Get the testbench and cover directories.
125  if (annot.isClass(extractCoverageAnnoClass))
126  if (auto dir = annot.getMember<StringAttr>("directory")) {
127  coverDir = dir.getValue();
128  continue;
129  }
130 
131  if (annot.isClass(testBenchDirAnnoClass))
132  if (auto dir = annot.getMember<StringAttr>("dirname")) {
133  testBenchDir = dir.getValue();
134  continue;
135  }
136 
137  // Handle target dir annotation.
138  if (annot.isClass(blackBoxTargetDirAnnoClass)) {
139  if (auto target = annot.getMember<StringAttr>("targetDir")) {
140  targetDir = target.getValue();
141  continue;
142  }
143  circuitOp->emitError(blackBoxTargetDirAnnoClass)
144  << " annotation missing \"targetDir\" attribute";
145  signalPassFailure();
146  continue;
147  }
148  }
149  // Apply the filtered annotations to the circuit. If we updated the circuit
150  // and record that they changed.
151  anythingChanged |=
152  AnnotationSet(filteredAnnos, context).applyToOperation(circuitOp);
153 
154  LLVM_DEBUG(llvm::dbgs() << "Black box target directory: " << targetDir << "\n"
155  << "Black box resource file name: "
156  << resourceFileName << "\n");
157 
158  // Newly generated IR will be placed at the end of the circuit.
159  auto builder = OpBuilder::atBlockEnd(circuitOp->getBlock());
160 
161  // Gather the relevant annotations on all modules in the circuit.
162  for (auto &op : *circuitOp.getBody()) {
163  if (!isa<FModuleOp>(op) && !isa<FExtModuleOp>(op))
164  continue;
165 
166  StringRef verifBBClass =
167  "freechips.rocketchip.annotations.InternalVerifBlackBoxAnnotation";
168  SmallVector<Attribute, 4> filteredAnnos;
169  auto annos = AnnotationSet(&op);
170  // If the cover directory is set and it has the verifBBClass annotation,
171  // then output directory should be cover dir.
172  auto isCover = !coverDir.empty() && annos.hasAnnotation(verifBBClass);
173  for (auto anno : annos) {
174  if (runOnAnnotation(&op, anno, builder, isCover))
175  // Since the annotation was consumed, add a `BlackBox` annotation to
176  // indicate that this extmodule was provided by one of the black box
177  // annotations. This is useful for metadata generation.
178  filteredAnnos.push_back(builder.getDictionaryAttr(
179  {{builder.getStringAttr("class"),
180  builder.getStringAttr("firrtl.transforms.BlackBox")}}));
181  else
182  filteredAnnos.push_back(anno.getDict());
183  }
184 
185  // Update the operation annotations to exclude the ones we have consumed.
186  anythingChanged |=
187  AnnotationSet(filteredAnnos, context).applyToOperation(&op);
188  }
189 
190  // If we have emitted any files, generate a file list operation that
191  // documents the additional annotation-controlled file listing to be
192  // created.
193  if (!fileListFiles.empty()) {
194  // Output the file list in sorted order.
195  llvm::sort(fileListFiles.begin(), fileListFiles.end());
196 
197  // Create the file list contents by prepending the file name with the target
198  // directory, and putting each file on its own line.
199  std::string output;
200  llvm::raw_string_ostream os(output);
201  llvm::interleave(
202  fileListFiles, os,
203  [&](StringRef fileName) {
204  SmallString<32> filePath(targetDir);
205  llvm::sys::path::append(filePath, fileName);
206  llvm::sys::path::remove_dots(filePath);
207  os << filePath;
208  },
209  "\n");
210 
211  // Put the file list in to a verbatim op. Use "unknown location" so that no
212  // file info will unnecessarily print.
213  auto op =
214  builder.create<VerbatimOp>(builder.getUnknownLoc(), std::move(output));
215 
216  // Attach the output file information to the verbatim op.
217  op->setAttr("output_file", hw::OutputFileAttr::getFromFilename(
218  context, resourceFileName,
219  /*excludeFromFileList=*/true));
220  }
221 
222  // If nothing has changed we can preseve the analysis.
223  if (!anythingChanged)
224  markAllAnalysesPreserved();
225  markAnalysesPreserved<InstanceGraph>();
226 
227  // Clean up.
228  emittedFiles.clear();
229  fileListFiles.clear();
230 }

232 /// Run on an operation-annotation pair. The annotation need not be a black box
233 /// annotation. Returns `true` if the annotation was indeed a black box
234 /// annotation (even if it was incomplete) and should be removed from the op.
235 bool BlackBoxReaderPass::runOnAnnotation(Operation *op, Annotation anno,
236  OpBuilder &builder, bool isCover) {
237  // Handle inline annotation.
238  if (anno.isClass(blackBoxInlineAnnoClass)) {
239  auto name = anno.getMember<StringAttr>("name");
240  auto text = anno.getMember<StringAttr>("text");
241  if (!name || !text) {
242  op->emitError(blackBoxInlineAnnoClass)
243  << " annotation missing \"name\" or \"text\" attribute";
244  signalPassFailure();
245  return true;
246  }
247 
248  LLVM_DEBUG(llvm::dbgs()
249  << "Add black box source `" << name.getValue() << "` inline\n");
250 
251  // Skip this inline annotation if the target is already generated.
252  if (emittedFiles.count(name))
253  return true;
254 
255  // Create an IR node to hold the contents. Use "unknown location" so that
256  // no file info will unnecessarily print.
257  auto verbatim = builder.create<VerbatimOp>(builder.getUnknownLoc(), text);
258  setOutputFile(verbatim, op, name, isDut(op), isCover);
259  return true;
260  }
261 
262  // Handle path annotation.
263  if (anno.isClass(blackBoxPathAnnoClass)) {
264  auto path = anno.getMember<StringAttr>("path");
265  if (!path) {
266  op->emitError(blackBoxPathAnnoClass)
267  << " annotation missing \"path\" attribute";
268  signalPassFailure();
269  return true;
270  }
271  SmallString<128> inputPath(inputPrefix);
272  appendPossiblyAbsolutePath(inputPath, path.getValue());
273  auto verbatim = loadFile(op, inputPath, builder);
274  if (!verbatim) {
275  op->emitError("Cannot find file ") << inputPath;
276  signalPassFailure();
277  return false;
278  }
279  auto name = builder.getStringAttr(llvm::sys::path::filename(path));
280  setOutputFile(verbatim, op, name, isDut(op), isCover);
281  return true;
282  }
283 
284  // Annotation was not concerned with black boxes.
285  return false;
286 }

288 /// Copies a black box source file to the appropriate location in the target
289 /// directory.
290 VerbatimOp BlackBoxReaderPass::loadFile(Operation *op, StringRef inputPath,
291  OpBuilder &builder) {
292  auto fileName = llvm::sys::path::filename(inputPath);
293  LLVM_DEBUG(llvm::dbgs() << "Add black box source `" << fileName << "` from `"
294  << inputPath << "`\n");
295 
296  // Skip this annotation if the target is already loaded.
297  auto fileNameAttr = builder.getStringAttr(fileName);
298  if (emittedFiles.count(fileNameAttr))
299  return {};
300 
301  // Open and read the input file.
302  std::string errorMessage;
303  auto input = mlir::openInputFile(inputPath, &errorMessage);
304  if (!input)
305  return {};
306 
307  // Create an IR node to hold the contents. Use "unknown location" so that no
308  // file info will unnecessarily print.
309  return builder.create<VerbatimOp>(builder.getUnknownLoc(),
310  input->getBuffer());
311 }

313 /// This function is called for every file generated. It does the following
314 /// things:
315 /// 1. Attaches the output file attribute to the VerbatimOp.
316 /// 2. Record that the file has been generated to avoid duplicates.
317 /// 3. Add each file name to the generated "file list" file.
318 void BlackBoxReaderPass::setOutputFile(VerbatimOp op, Operation *origOp,
319  StringAttr fileNameAttr, bool isDut,
320  bool isCover) {
321  // If the output file was set on the original operation then either: (1) copy
322  // this to the new op if it is a filename or (2) use this directory (since it
323  // is a directory) as the lowest priority directory to put this file.
324  auto outputFile = origOp->getAttrOfType<OutputFileAttr>("output_file");
325  if (outputFile && !outputFile.isDirectory()) {
326  op->setAttr("output_file", outputFile);
327  if (!outputFile.getExcludeFromFilelist().getValue())
328  fileListFiles.push_back(outputFile.getFilename());
329  return;
330  }
331 
332  // Exclude Verilog header files since we expect them to be included
333  // explicitly by compiler directives in other source files.
334  auto *context = &getContext();
335  auto fileName = fileNameAttr.getValue();
336  auto ext = llvm::sys::path::extension(fileName);
337  bool exclude = (ext == ".h" || ext == ".vh" || ext == ".svh");
338  auto outDir = targetDir;
339  if (!testBenchDir.empty() && targetDir.equals(".") && !isDut)
340  outDir = testBenchDir;
341  else if (isCover)
342  outDir = coverDir;
343  else if (outputFile)
344  outDir = outputFile.getFilename();
345 
346  // If targetDir is not set explicitly and this is a testbench module, then
347  // update the targetDir to be the "../testbench".
348  auto outFileAttr = OutputFileAttr::getFromDirectoryAndFilename(
349  context, outDir, fileName,
350  /*excludeFromFileList=*/exclude);
351  op->setAttr("output_file", outFileAttr);
352 
353  // Record that this file has been generated.
354  assert(!emittedFiles.count(fileNameAttr) &&
355  "Can't generate the same file twice.");
356  emittedFiles.insert(fileNameAttr);
357 
358  // Append this file to the file list if its not excluded.
359  if (!exclude)
360  fileListFiles.push_back(outFileAttr.getFilename());
361 }

363 /// Return true if module is in the DUT hierarchy.
364 bool BlackBoxReaderPass::isDut(Operation *module) {
365  // Check if result already cached.
366  auto iter = dutModuleMap.find(module);
367  if (iter != dutModuleMap.end())
368  return iter->getSecond();
369  const StringRef dutAnno = "sifive.enterprise.firrtl.MarkDUTAnnotation";
370  AnnotationSet annos(module);
371  // Any module with the dutAnno, is the DUT.
372  if (annos.hasAnnotation(dutAnno)) {
373  dutModuleMap[module] = true;
374  return true;
375  }
376  auto *node = instanceGraph->lookup(module);
377  bool anyParentIsDut = false;
378  if (node)
379  for (auto *u : node->uses()) {
380  // Recursively check the parents.
381  auto dut = isDut(u->getInstance()->getParentOfType<FModuleOp>());
382  // Cache the result.
383  dutModuleMap[module] = dut;
384  anyParentIsDut |= dut;
385  }
386  dutModuleMap[module] = anyParentIsDut;
387  return anyParentIsDut;
388 }
