53  // Returns true if a block argument should have buffers added to its uses.
54  static bool shouldBufferArgument(BlockArgument arg) {
55  // At the moment, buffers only make sense on arguments which we know
56  // will lower down to a handshake bundle.
57  return arg.getType().isIntOrFloat() || arg.getType().isa<NoneType>();
58  }
59 
60  static bool isUnbufferedChannel(Operation *definingOp, Operation *usingOp) {
61  return !isa_and_nonnull<BufferOp>(definingOp) && !isa<BufferOp>(usingOp);
62  }

64  void insertBuffer(Location loc, Value operand, OpBuilder &builder,
65  unsigned numSlots, BufferTypeEnum bufferType) {
66  auto ip = builder.saveInsertionPoint();
67  builder.setInsertionPointAfterValue(operand);
68  auto bufferOp = builder.create<handshake::BufferOp>(
69  loc, operand.getType(), numSlots, operand, bufferType);
70  operand.replaceUsesWithIf(
71  bufferOp,
72  function_ref<bool(OpOperand &)>([](OpOperand &operand) -> bool {
73  return !isa<handshake::BufferOp>(operand.getOwner());
74  }));
75  builder.restoreInsertionPoint(ip);
76  }

78  // Inserts a buffer at a specific operand use.
79  void bufferOperand(OpOperand &use, OpBuilder &builder, size_t numSlots,
80  BufferTypeEnum bufferType) {
81  auto *usingOp = use.getOwner();
82  Value usingValue = use.get();
83 
84  builder.setInsertionPoint(usingOp);
85  auto buffer = builder.create<handshake::BufferOp>(
86  usingValue.getLoc(), usingValue.getType(),
87  /*slots=*/numSlots, usingValue,
88  /*bufferType=*/bufferType);
89  usingOp->setOperand(use.getOperandNumber(), buffer);
90  }
91 
92  // Inserts buffers at all operands of an operation.
93  void bufferOperands(Operation *op, OpBuilder builder, size_t numSlots,
94  BufferTypeEnum bufferType) {
95  for (auto &use : op->getOpOperands()) {
96  auto *srcOp = use.get().getDefiningOp();
97  if (isa_and_nonnull<handshake::BufferOp>(srcOp))
98  continue;
99  bufferOperand(use, builder, numSlots, bufferType);
100  }
101  }
102 
103  // Inserts buffers at all results of an operation
104  void bufferResults(OpBuilder &builder, Operation *op, unsigned numSlots,
105  BufferTypeEnum bufferType) {
106  for (auto res : op->getResults()) {
107  Operation *user = *res.getUsers().begin();
108  if (isa<handshake::BufferOp>(user))
109  continue;
110  insertBuffer(op->getLoc(), res, builder, numSlots, bufferType);
111  }
112  };

114  // Add a buffer to any un-buffered channel.
115  void bufferAllStrategy(handshake::FuncOp f, OpBuilder &builder,
116  unsigned numSlots,
117  BufferTypeEnum bufferType = BufferTypeEnum::seq) {
118 
119  for (auto &arg : f.getArguments()) {
120  if (!shouldBufferArgument(arg))
121  continue;
122  insertBuffer(arg.getLoc(), arg, builder, numSlots, bufferType);
123  }
124 
125  for (auto &defOp : f.getOps()) {
126  for (auto res : defOp.getResults()) {
127  for (auto *useOp : res.getUsers()) {
128  if (!isUnbufferedChannel(&defOp, useOp))
129  continue;
130  insertBuffer(res.getLoc(), res, builder, numSlots, bufferType);
131  }
132  }
133  }
134  }

136  // Combination of bufferCyclesStrategy and bufferAllStrategy, where we add a
137  // sequential buffer on graph cycles, and add FIFO buffers on all other
138  // connections.
139  void bufferAllFIFOStrategy(handshake::FuncOp f, OpBuilder &builder) {
140  // First, buffer cycles with sequential buffers
141  bufferCyclesStrategy(f, builder, /*numSlots=*/2,
142  /*bufferType=*/BufferTypeEnum::seq);
143  // Then, buffer remaining channels with transparent FIFO buffers
144  bufferAllStrategy(f, builder, bufferSize,
145  /*bufferType=*/BufferTypeEnum::fifo);
146  }

148  // Perform a depth first search and insert buffers when cycles are detected.
149  void
150  bufferCyclesStrategy(handshake::FuncOp f, OpBuilder &builder,
151  unsigned numSlots,
152  BufferTypeEnum /*bufferType*/ = BufferTypeEnum::seq) {
153  // Cycles can only occur at merge-like operations so those are our buffering
154  // targets. Placing the buffer at the output of the merge-like op,
155  // as opposed to naivly placing buffers *whenever* cycles are detected
156  // ensures that we don't place a bunch of buffers on each input of the
157  // merge-like op.
158  auto isSeqBuffer = [](auto op) {
159  auto bufferOp = dyn_cast<handshake::BufferOp>(op);
160  return bufferOp && bufferOp.isSequential();
161  };
162 
163  for (auto mergeOp : f.getOps<MergeLikeOpInterface>()) {
164  // We insert a sequential buffer whenever the op is determined to be
165  // within a cycle (to break combinational cycles). Else, place a FIFO
166  // buffer.
167  bool sequential = inCycle(mergeOp, isSeqBuffer);
168  bufferResults(builder, mergeOp, numSlots,
169  sequential ? BufferTypeEnum::seq : BufferTypeEnum::fifo);
170  }
171  }
172 
173  // Returns true if 'src' is within a cycle. 'breaksCycle' is a function which
174  // determines whether an operation breaks a cycle.
175  bool inCycle(Operation *src,
176  llvm::function_ref<bool(Operation *)> breaksCycle,
177  Operation *curr = nullptr, SetVector<Operation *> path = {}) {
178  // If visiting the source node, then we're in a cycle.
179  if (curr == src)
180  return true;
181 
182  // Initial case; set current node to source node
183  if (curr == nullptr) {
184  curr = src;
185  }
186 
187  path.insert(curr);
188  for (auto &operand : curr->getUses()) {
189  auto *user = operand.getOwner();
190 
191  // We might encounter a cycle, but we only care about the case when such
192  // cycles include 'src'.
193  if (path.count(user) && user != src)
194  continue;
195  if (breaksCycle(curr))
196  continue;
197  if (inCycle(src, breaksCycle, user, path))
198  return true;
199  }
200  return false;
201  }

203  void runOnOperation() override {
204  auto f = getOperation();
205  auto builder = OpBuilder(f.getContext());
206 
207  if (strategy == "cycles")
208  bufferCyclesStrategy(f, builder, bufferSize);
209  else if (strategy == "all")
210  bufferAllStrategy(f, builder, bufferSize);
211  else if (strategy == "allFIFO")
212  bufferAllFIFOStrategy(f, builder);
213  else {
214  getOperation().emitOpError() << "Unknown buffer strategy: " << strategy;
215  signalPassFailure();
216  return;
217  }
218  }
