85 /// A tracker to determine which libraries should be imported for a given
86 /// program.
87 struct ImportTracker {
88 public:
89  /// Returns the list of library names used for in this program.
90  /// E.g. if `primitives/core.futil` is used, returns { "core" }.
91  FailureOr<llvm::SmallSet<StringRef, 4>> getLibraryNames(ProgramOp program) {
92  auto walkRes = program.walk([&](ComponentOp component) {
93  for (auto &op : *component.getBody()) {
94  if (!isa<CellInterface>(op) || isa<InstanceOp>(op))
95  // It is not a primitive.
96  continue;
97  auto libraryName = getLibraryFor(&op);
98  if (failed(libraryName))
99  return WalkResult::interrupt();
100  usedLibraries.insert(libraryName.getValue());
101  }
102  return WalkResult::advance();
103  });
104  if (walkRes.wasInterrupted())
105  return failure();
106  return usedLibraries;
107  }
108 
109 private:
110  /// Returns the library name for a given Operation Type.
111  FailureOr<StringRef> getLibraryFor(Operation *op) {
112  return TypeSwitch<Operation *, FailureOr<StringRef>>(op)
113  .Case<MemoryOp, RegisterOp, NotLibOp, AndLibOp, OrLibOp, XorLibOp,
114  AddLibOp, SubLibOp, GtLibOp, LtLibOp, EqLibOp, NeqLibOp, GeLibOp,
115  LeLibOp, LshLibOp, RshLibOp, SliceLibOp, PadLibOp, WireLibOp>(
116  [&](auto op) -> FailureOr<StringRef> {
117  static constexpr std::string_view sCore = "core";
118  return {sCore};
119  })
120  .Case<SgtLibOp, SltLibOp, SeqLibOp, SneqLibOp, SgeLibOp, SleLibOp,
121  SrshLibOp, MultPipeLibOp, RemUPipeLibOp, RemSPipeLibOp,
122  DivUPipeLibOp, DivSPipeLibOp>(
123  [&](auto op) -> FailureOr<StringRef> {
124  static constexpr std::string_view sBinaryOperators =
125  "binary_operators";
126  return {sBinaryOperators};
127  })
128  /*.Case<>([&](auto op) { library = "math"; })*/
129  .Default([&](auto op) {
130  auto diag = op->emitOpError() << "not supported for emission";
131  auto note = unsupportedOpInfo(op);
132  if (note)
133  diag.attachNote() << *note;
134  return diag;
135  });
136  }
137  /// Maintains a unique list of libraries used throughout the lifetime of the
138  /// tracker.
139  llvm::SmallSet<StringRef, 4> usedLibraries;
140 };

269  /// Calyx attributes are emitted in one of the two following formats:
270  /// (1) @<attribute-name>(<attribute-value>), e.g. `@go`, `@bound(5)`.
271  /// (2) <"<attribute-name>"=<attribute-value>>, e.g. `<"static"=1>`.
272  ///
273  /// Since ports are structural in nature and not operations, an
274  /// extra boolean value is added to determine whether this is a port of the
275  /// given operation.
276  std::string getAttribute(Operation *op, StringRef identifier, Attribute attr,
277  bool isPort) {
278  // Verify this attribute is supported for emission.
279  if (!isValidCalyxAttribute(identifier))
280  return "";
281 
282  // Determines whether the attribute should follow format (2).
283  bool isGroupOrComponentAttr = isa<GroupOp, ComponentOp>(op) && !isPort;
284 
285  std::string output;
286  llvm::raw_string_ostream buffer(output);
287  buffer.reserveExtraSpace(16);
288 
289  bool isBooleanAttribute =
290  llvm::find(booleanAttributes, identifier) != booleanAttributes.end();
291  if (attr.isa<UnitAttr>()) {
292  assert(isBooleanAttribute &&
293  "Non-boolean attributes must provide an integer value.");
294  if (isGroupOrComponentAttr) {
295  buffer << LAngleBracket() << delimiter() << identifier << delimiter()
296  << equals() << "1" << RAngleBracket();
297  } else {
298  buffer << addressSymbol() << identifier << space();
299  }
300  } else if (auto intAttr = attr.dyn_cast<IntegerAttr>()) {
301  APInt value = intAttr.getValue();
302  if (isGroupOrComponentAttr) {
303  buffer << LAngleBracket() << delimiter() << identifier << delimiter()
304  << equals() << value << RAngleBracket();
305  } else {
306  buffer << addressSymbol() << identifier;
307  // The only time we may omit the value is when it is a Boolean attribute
308  // with value 1.
309  if (!isBooleanAttribute || intAttr.getValue() != 1) {
310  // Retrieve the unsigned representation of the value.
311  SmallVector<char, 4> s;
312  value.toStringUnsigned(s, /*Radix=*/10);
313  buffer << LParen() << s << RParen();
314  }
315  buffer << space();
316  }
317  }
318  return buffer.str();
319  }

321  /// Emits the attributes of a dictionary. If the `attributes` dictionary is
322  /// not nullptr, we assume this is for a port.
323  std::string getAttributes(Operation *op,
324  DictionaryAttr attributes = nullptr) {
325  bool isPort = attributes != nullptr;
326  if (!isPort)
327  attributes = op->getAttrDictionary();
328 
329  SmallString<16> calyxAttributes;
330  for (auto &attr : attributes)
331  calyxAttributes.append(
332  getAttribute(op, attr.getName(), attr.getValue(), isPort));
333  return calyxAttributes.c_str();
334  }

360  /// Helper function for emitting combinational operations.
361  template <typename CombinationalOp>
362  void emitCombinationalValue(CombinationalOp op, StringRef logicalSymbol) {
363  auto inputs = op.inputs();
364  os << LParen();
365  for (size_t i = 0, e = inputs.size(); i != e; ++i) {
366  emitValue(inputs[i], /*isIndented=*/false);
367  if (i + 1 == e)
368  continue;
369  os << space() << logicalSymbol << space();
370  }
371  os << RParen();
372  }

374  /// Emits the value of a guard or assignment.
375  void emitValue(Value value, bool isIndented) {
376  if (auto blockArg = value.dyn_cast<BlockArgument>()) {
377  // Emit component block argument.
378  StringAttr portName = getPortInfo(blockArg).name;
379  (isIndented ? indent() : os) << portName.getValue();
380  return;
381  }
382 
383  auto definingOp = value.getDefiningOp();
384  assert(definingOp && "Value does not have a defining operation.");
385 
386  TypeSwitch<Operation *>(definingOp)
387  .Case<CellInterface>([&](auto cell) {
388  // A cell port should be defined as <instance-name>.<port-name>
389  (isIndented ? indent() : os)
390  << cell.instanceName() << period() << cell.portName(value);
391  })
392  .Case<hw::ConstantOp>([&](auto op) {
393  // A constant is defined as <bit-width>'<base><value>, where the base
394  // is `b` (binary), `o` (octal), `h` hexadecimal, or `d` (decimal).
395  APInt value = op.value();
396 
397  (isIndented ? indent() : os)
398  << std::to_string(value.getBitWidth()) << apostrophe() << "d";
399  // We currently default to the decimal representation.
400  value.print(os, /*isSigned=*/false);
401  })
402  .Case<comb::AndOp>([&](auto op) { emitCombinationalValue(op, "&"); })
403  .Case<comb::OrOp>([&](auto op) { emitCombinationalValue(op, "|"); })
404  .Case<comb::XorOp>([&](auto op) {
405  // The XorOp is a bit different, since the Combinational dialect
406  // uses it to represent binary not.
407  if (!op.isBinaryNot()) {
408  emitOpError(op, "Only supporting Binary Not for XOR.");
409  return;
410  }
411  // The LHS is the value to be negated, and the RHS is a constant with
412  // all ones (guaranteed by isBinaryNot).
413  os << exclamationMark();
414  emitValue(op.inputs()[0], /*isIndented=*/false);
415  })
416  .Default(
417  [&](auto op) { emitOpError(op, "not supported for emission"); });
418  }

420  /// Emits a port for a Group.
421  template <typename OpTy>
422  void emitGroupPort(GroupInterface group, OpTy op, StringRef portHole) {
423  assert((isa<GroupGoOp>(op) || isa<GroupDoneOp>(op)) &&
424  "Required to be a group port.");
425  indent() << group.symName().getValue() << LSquare() << portHole << RSquare()
426  << space() << equals() << space();
427  if (op.guard()) {
428  emitValue(op.guard(), /*isIndented=*/false);
429  os << questionMark();
430  }
431  emitValue(op.src(), /*isIndented=*/false);
432  os << semicolonEndL();

435  /// Recursively emits the Calyx control.
436  void emitCalyxControl(Block *body) {
437  Operation *parent = body->getParentOp();
438  assert((isa<ControlOp>(parent) || parent->hasTrait<ControlLike>()) &&
439  "This should only be used to emit Calyx Control structures.");
440 
441  // Check to see if this is a stand-alone EnableOp, i.e.
442  // calyx.control { calyx.enable @G }
443  if (auto enable = dyn_cast<EnableOp>(parent)) {
444  emitEnable(enable);
445  // Early return since an EnableOp has no body.
446  return;
447  }
448  // Attribute dictionary is always prepended for a control operation.
449  auto prependAttributes = [&](Operation *op, StringRef sym) {
450  return (getAttributes(op) + sym).str();
451  };
452 
453  for (auto &&op : *body) {
454 
455  TypeSwitch<Operation *>(&op)
456  .Case<SeqOp>([&](auto op) {
457  emitCalyxSection(prependAttributes(op, "seq"),
458  [&]() { emitCalyxControl(op.getBody()); });
459  })
460  .Case<ParOp>([&](auto op) {
461  emitCalyxSection(prependAttributes(op, "par"),
462  [&]() { emitCalyxControl(op.getBody()); });
463  })
464  .Case<WhileOp>([&](auto op) {
465  indent() << prependAttributes(op, "while ");
466  emitValue(op.cond(), /*isIndented=*/false);
467 
468  if (auto groupName = op.groupName(); groupName.hasValue())
469  os << " with " << groupName.getValue();
470 
471  emitCalyxBody([&]() { emitCalyxControl(op.getBody()); });
472  })
473  .Case<IfOp>([&](auto op) {
474  indent() << prependAttributes(op, "if ");
475  emitValue(op.cond(), /*isIndented=*/false);
476 
477  if (auto groupName = op.groupName(); groupName.hasValue())
478  os << " with " << groupName.getValue();
479 
480  emitCalyxBody([&]() { emitCalyxControl(op.getThenBody()); });
481  if (op.elseBodyExists())
482  emitCalyxSection("else",
483  [&]() { emitCalyxControl(op.getElseBody()); });
484  })
485  .Case<EnableOp>([&](auto op) { emitEnable(op); })
486  .Default([&](auto op) {
487  emitOpError(op, "not supported for emission inside control.");
488  });
489  }
490  }
491 
492  /// The stream we are emitting into.
493  llvm::raw_ostream &os;
494 
495  /// Whether we have encountered any errors during emission.
496  bool encounteredError = false;
497 
498  /// Current level of indentation. See `indent()` and
499  /// `addIndent()`/`reduceIndent()`.
500  unsigned currentIndent = 0;
501 };

517 /// Emit a component.
518 void Emitter::emitComponent(ComponentOp op) {
519  indent() << "component " << op.getName() << getAttributes(op);
520  // Emit the ports.
521  emitComponentPorts(op);
522  os << space() << LBraceEndL();
523  addIndent();
524  WiresOp wires;
525  ControlOp control;
526 
527  // Emit cells.
528  emitCalyxSection("cells", [&]() {
529  for (auto &&bodyOp : *op.getBody()) {
530  TypeSwitch<Operation *>(&bodyOp)
531  .Case<WiresOp>([&](auto op) { wires = op; })
532  .Case<ControlOp>([&](auto op) { control = op; })
533  .Case<InstanceOp>([&](auto op) { emitInstance(op); })
534  .Case<RegisterOp>([&](auto op) { emitRegister(op); })
535  .Case<MemoryOp>([&](auto op) { emitMemory(op); })
536  .Case<hw::ConstantOp>([&](auto op) { /*Do nothing*/ })
537  .Case<SliceLibOp, PadLibOp>(
538  [&](auto op) { emitLibraryPrimTypedByAllPorts(op); })
539  .Case<LtLibOp, GtLibOp, EqLibOp, NeqLibOp, GeLibOp, LeLibOp, SltLibOp,
540  SgtLibOp, SeqLibOp, SneqLibOp, SgeLibOp, SleLibOp, AddLibOp,
541  SubLibOp, ShruLibOp, RshLibOp, SrshLibOp, LshLibOp, AndLibOp,
542  NotLibOp, OrLibOp, XorLibOp, WireLibOp>(
543  [&](auto op) { emitLibraryPrimTypedByFirstInputPort(op); })
544  .Case<MultPipeLibOp>(
545  [&](auto op) { emitLibraryPrimTypedByFirstOutputPort(op); })
546  .Case<RemUPipeLibOp, DivUPipeLibOp>([&](auto op) {
547  emitLibraryPrimTypedByFirstOutputPort(
548  op, /*calyxLibName=*/{"std_div_pipe"});
549  })
550  .Case<RemSPipeLibOp, DivSPipeLibOp>([&](auto op) {
551  emitLibraryPrimTypedByFirstOutputPort(
552  op, /*calyxLibName=*/{"std_sdiv_pipe"});
553  })
554  .Default([&](auto op) {
555  emitOpError(op, "not supported for emission inside component");
556  });
557  }
558  });
559 
560  emitWires(wires);
561  emitControl(control);
562  reduceIndent();
563  os << RBraceEndL();
564 }

566 /// Emit the ports of a component.
567 void Emitter::emitComponentPorts(ComponentOp op) {
568  auto emitPorts = [&](auto ports) {
569  os << LParen();
570  for (size_t i = 0, e = ports.size(); i < e; ++i) {
571  const PortInfo &port = ports[i];
572 
573  // We only care about the bit width in the emitted .futil file.
574  auto bitWidth = port.type.getIntOrFloatBitWidth();
575  os << getAttributes(op, port.attributes) << port.name.getValue()
576  << colon() << bitWidth;
577 
578  if (i + 1 < e)
579  os << comma();
580  }
581  os << RParen();
582  };
583  emitPorts(op.getInputPortInfo());
584  os << arrow();
585  emitPorts(op.getOutputPortInfo());
586 }

601 void Emitter::emitMemory(MemoryOp memory) {
602  size_t dimension = memory.sizes().size();
603  if (dimension < 1 || dimension > 4) {
604  emitOpError(memory, "Only memories with dimensionality in range [1, 4] are "
605  "supported by the native Calyx compiler.");
606  return;
607  }
608  indent() << getAttributes(memory) << memory.instanceName() << space()
609  << equals() << space() << "std_mem_d" << std::to_string(dimension)
610  << LParen() << memory.width() << comma();
611  for (Attribute size : memory.sizes()) {
612  APInt memSize = size.cast<IntegerAttr>().getValue();
613  memSize.print(os, /*isSigned=*/false);
614  os << comma();
615  }
