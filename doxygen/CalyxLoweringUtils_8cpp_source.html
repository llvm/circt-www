31 void appendPortsForExternalMemref(PatternRewriter &rewriter, StringRef memName,
32  Value memref, unsigned memoryID,
33  SmallVectorImpl<calyx::PortInfo> &inPorts,
34  SmallVectorImpl<calyx::PortInfo> &outPorts) {
35  MemRefType memrefType = memref.getType().cast<MemRefType>();
36 
37  // Ports constituting a memory interface are added a set of attributes under
38  // a "mem : {...}" dictionary. These attributes allows for deducing which
39  // top-level I/O signals constitutes a unique memory interface.
40  auto getMemoryInterfaceAttr = [&](StringRef tag,
41  Optional<unsigned> addrIdx = {}) {
42  auto attrs = SmallVector<NamedAttribute>{
43  // "id" denotes a unique memory interface.
44  rewriter.getNamedAttr("id", rewriter.getI32IntegerAttr(memoryID)),
45  // "tag" denotes the function of this signal.
46  rewriter.getNamedAttr("tag", rewriter.getStringAttr(tag))};
47  if (addrIdx.hasValue())
48  // "addr_idx" denotes the address index of this signal, for
49  // multi-dimensional memory interfaces.
50  attrs.push_back(rewriter.getNamedAttr(
51  "addr_idx", rewriter.getI32IntegerAttr(addrIdx.getValue())));
52 
53  return rewriter.getNamedAttr("mem", rewriter.getDictionaryAttr(attrs));
54  };
55 
56  // Read data
57  inPorts.push_back(calyx::PortInfo{
58  rewriter.getStringAttr(memName + "_read_data"),
59  memrefType.getElementType(), calyx::Direction::Input,
60  DictionaryAttr::get(rewriter.getContext(),
61  {getMemoryInterfaceAttr("read_data")})});
62 
63  // Done
64  inPorts.push_back(
65  calyx::PortInfo{rewriter.getStringAttr(memName + "_done"),
66  rewriter.getI1Type(), calyx::Direction::Input,
67  DictionaryAttr::get(rewriter.getContext(),
68  {getMemoryInterfaceAttr("done")})});
69 
70  // Write data
71  outPorts.push_back(calyx::PortInfo{
72  rewriter.getStringAttr(memName + "_write_data"),
73  memrefType.getElementType(), calyx::Direction::Output,
74  DictionaryAttr::get(rewriter.getContext(),
75  {getMemoryInterfaceAttr("write_data")})});
76 
77  // Memory address outputs
78  for (auto dim : enumerate(memrefType.getShape())) {
79  outPorts.push_back(calyx::PortInfo{
80  rewriter.getStringAttr(memName + "_addr" + std::to_string(dim.index())),
81  rewriter.getIntegerType(calyx::handleZeroWidth(dim.value())),
82  calyx::Direction::Output,
83  DictionaryAttr::get(rewriter.getContext(),
84  {getMemoryInterfaceAttr("addr", dim.index())})});
85  }
86 
87  // Write enable
88  outPorts.push_back(calyx::PortInfo{
89  rewriter.getStringAttr(memName + "_write_en"), rewriter.getI1Type(),
90  calyx::Direction::Output,
91  DictionaryAttr::get(rewriter.getContext(),
92  {getMemoryInterfaceAttr("write_en")})});
93 }

359 LogicalResult
360 ModuleOpConversion::matchAndRewrite(mlir::ModuleOp moduleOp,
361  PatternRewriter &rewriter) const {
362  if (!moduleOp.getOps<calyx::ProgramOp>().empty())
363  return failure();
364 
365  rewriter.updateRootInPlace(moduleOp, [&] {
366  // Create ProgramOp
367  rewriter.setInsertionPointAfter(moduleOp);
368  auto programOp = rewriter.create<calyx::ProgramOp>(
369  moduleOp.getLoc(), StringAttr::get(getContext(), topLevelFunction));
370 
371  // Inline the module body region
372  rewriter.inlineRegionBefore(moduleOp.getBodyRegion(),
373  programOp.getBodyRegion(),
374  programOp.getBodyRegion().end());
375 
376  // Inlining the body region also removes ^bb0 from the module body
377  // region, so recreate that, before finally inserting the programOp
378  auto *moduleBlock = rewriter.createBlock(&moduleOp.getBodyRegion());
379  rewriter.setInsertionPointToStart(moduleBlock);
380  rewriter.insert(programOp);
381  *programOpOutput = programOp;
382  });
383  return success();
384 }

426 LogicalResult
427 ConvertIndexTypes::partiallyLowerFuncToComp(mlir::func::FuncOp funcOp,
428  PatternRewriter &rewriter) const {
429  funcOp.walk([&](Block *block) {
430  for (Value arg : block->getArguments())
431  arg.setType(calyx::convIndexType(rewriter, arg.getType()));
432  });
433 
434  funcOp.walk([&](Operation *op) {
435  for (Value result : op->getResults()) {
436  Type resType = result.getType();
437  if (!resType.isIndex())
438  continue;
439 
440  result.setType(calyx::convIndexType(rewriter, resType));
441  auto constant = dyn_cast<mlir::arith::ConstantOp>(op);
442  if (!constant)
443  continue;
444 
445  APInt value;
446  calyx::matchConstantOp(constant, value);
447  rewriter.setInsertionPoint(constant);
448  rewriter.replaceOpWithNewOp<mlir::arith::ConstantOp>(
449  constant, rewriter.getI32IntegerAttr(value.getSExtValue()));
450  }
451  });
452  return success();
453 }

475 LogicalResult
476 MultipleGroupDonePattern::matchAndRewrite(calyx::GroupOp groupOp,
477  PatternRewriter &rewriter) const {
478  auto groupDoneOps = SmallVector<calyx::GroupDoneOp>(
479  groupOp.getBody()->getOps<calyx::GroupDoneOp>());
480 
481  if (groupDoneOps.size() <= 1)
482  return failure();
483 
484  /// 'and' all of the calyx::GroupDoneOp's.
485  rewriter.setInsertionPointToEnd(groupDoneOps[0]->getBlock());
486  SmallVector<Value> doneOpSrcs;
487  llvm::transform(groupDoneOps, std::back_inserter(doneOpSrcs),
488  [](calyx::GroupDoneOp op) { return op.src(); });
489  Value allDone =
490  rewriter.create<comb::AndOp>(groupDoneOps.front().getLoc(), doneOpSrcs);
491 
492  /// Create a group done op with the complex expression as a guard.
493  rewriter.create<calyx::GroupDoneOp>(
494  groupOp.getLoc(),
495  rewriter.create<hw::ConstantOp>(groupOp.getLoc(), APInt(1, 1)), allDone);
496  for (auto groupDoneOp : groupDoneOps)
497  rewriter.eraseOp(groupDoneOp);
498 
499  return success();
500 }

526 LogicalResult
527 InlineCombGroups::partiallyLower(calyx::GroupInterface originGroup,
528  PatternRewriter &rewriter) const {
529  auto component = originGroup->getParentOfType<calyx::ComponentOp>();
530  ComponentLoweringStateInterface *state = pls.getState(component);
531 
532  // Filter groups which are not part of the control schedule.
533  if (SymbolTable::symbolKnownUseEmpty(originGroup.symName(),
534  component.getControlOp()))
535  return success();
536 
537  // Maintain a set of the groups which we've inlined so far. The group
538  // itself is implicitly inlined.
539  llvm::SmallSetVector<Operation *, 8> inlinedGroups;
540  inlinedGroups.insert(originGroup);
541 
542  // Starting from the matched originGroup, we traverse use-def chains of
543  // combinational logic, and inline assignments from the defining
544  // combinational groups.
545  recurseInlineCombGroups(rewriter, *state, inlinedGroups, originGroup,
546  originGroup,
547  /*doInline=*/false);
548  return success();
549 }
550 
551 void InlineCombGroups::recurseInlineCombGroups(
552  PatternRewriter &rewriter, ComponentLoweringStateInterface &state,
553  llvm::SmallSetVector<Operation *, 8> &inlinedGroups,
554  calyx::GroupInterface originGroup, calyx::GroupInterface recGroup,
555  bool doInline) const {
556  inlinedGroups.insert(recGroup);
557  for (auto assignOp : recGroup.getBody()->getOps<calyx::AssignOp>()) {
558  if (doInline) {
559  /// Inline the assignment into the originGroup.
560  auto *clonedAssignOp = rewriter.clone(*assignOp.getOperation());
561  clonedAssignOp->moveBefore(originGroup.getBody(),
562  originGroup.getBody()->end());
563  }
564  Value src = assignOp.src();
565 
566  // Things which stop recursive inlining (or in other words, what
567  // breaks combinational paths).
568  // - Component inputs
569  // - Register and memory reads
570  // - Constant ops (constant ops are not evaluated by any group)
571  // - Multiplication pipelines are sequential.
572  // - 'While' return values (these are registers, however, 'while'
573  // return values have at the current point of conversion not yet
574  // been rewritten to their register outputs, see comment in
575  // LateSSAReplacement)
576  if (src.isa<BlockArgument>() ||
577  isa<calyx::RegisterOp, calyx::MemoryOp, hw::ConstantOp,
578  mlir::arith::ConstantOp, calyx::MultPipeLibOp, calyx::DivUPipeLibOp,
579  calyx::DivSPipeLibOp, calyx::RemSPipeLibOp, calyx::RemUPipeLibOp,
580  mlir::scf::WhileOp>(src.getDefiningOp()))
581  continue;
582 
583  auto srcCombGroup = dyn_cast<calyx::CombGroupOp>(
584  state.getEvaluatingGroup(src).getOperation());
585  if (!srcCombGroup)
586  continue;
587  if (inlinedGroups.count(srcCombGroup))
588  continue;
589 
590  recurseInlineCombGroups(rewriter, state, inlinedGroups, originGroup,
591  srcCombGroup, /*doInline=*/true);
592  }
593 }

597 //===----------------------------------------------------------------------===//
598 
599 LogicalResult
600 RewriteMemoryAccesses::partiallyLower(calyx::AssignOp assignOp,
601  PatternRewriter &rewriter) const {
602  auto *state = pls.getState(assignOp->getParentOfType<calyx::ComponentOp>());
603 
604  Value dest = assignOp.dest();
605  if (!state->isInputPortOfMemory(dest).hasValue())
606  return success();
607 
608  Value src = assignOp.src();
609  unsigned srcBits = src.getType().getIntOrFloatBitWidth();
610  unsigned dstBits = dest.getType().getIntOrFloatBitWidth();
611  if (srcBits == dstBits)
612  return success();
613 
614  SmallVector<Type> types = {
615  rewriter.getIntegerType(srcBits),
616  rewriter.getIntegerType(dstBits),
617  };
618  mlir::Location loc = assignOp.getLoc();
619  Operation *newOp;
620  if (srcBits > dstBits)
621  newOp =
622  state->getNewLibraryOpInstance<calyx::SliceLibOp>(rewriter, loc, types);
623  else
624  newOp =
625  state->getNewLibraryOpInstance<calyx::PadLibOp>(rewriter, loc, types);
626 
627  rewriter.setInsertionPoint(assignOp->getBlock(),
628  assignOp->getBlock()->begin());
629  rewriter.create<calyx::AssignOp>(assignOp->getLoc(), newOp->getResult(0),
630  src);
631  assignOp.setOperand(1, newOp->getResult(1));
632 
633  return success();
634 }

640 LogicalResult
641 BuildBasicBlockRegs::partiallyLowerFuncToComp(mlir::func::FuncOp funcOp,
642  PatternRewriter &rewriter) const {
643  funcOp.walk([&](Block *block) {
644  /// Do not register component input values.
645  if (block == &block->getParent()->front())
646  return;
647 
648  for (auto arg : enumerate(block->getArguments())) {
649  Type argType = arg.value().getType();
650  assert(argType.isa<IntegerType>() && "unsupported block argument type");
651  unsigned width = argType.getIntOrFloatBitWidth();
652  std::string index = std::to_string(arg.index());
653  std::string name = programState().blockName(block) + "_arg" + index;
654  auto reg = createRegister(arg.value().getLoc(), rewriter, getComponent(),
