157 /// Holds common utilities used for scheduling when lowering to Calyx.
158 template <typename T>
159 class SchedulerInterface {
160 public:
161  /// Register 'scheduleable' as being generated through lowering 'block'.
162  ///
163  /// TODO(mortbopet): Add a post-insertion check to ensure that the use-def
164  /// ordering invariant holds for the groups. When the control schedule is
165  /// generated, scheduleables within a block are emitted sequentially based on
166  /// the order that this function was called during conversion.
167  ///
168  /// Currently, we assume this to always be true. Walking the FuncOp IR implies
169  /// sequential iteration over operations within basic blocks.
170  void addBlockScheduleable(mlir::Block *block, const T &scheduleable) {
171  blockScheduleables[block].push_back(scheduleable);
172  }
173 
174  /// Returns an ordered list of schedulables which registered themselves to be
175  /// a result of lowering the block in the source program. The list order
176  /// follows def-use chains between the scheduleables in the block.
177  SmallVector<T> getBlockScheduleables(mlir::Block *block) {
178  if (auto it = blockScheduleables.find(block);
179  it != blockScheduleables.end())
180  return it->second;
181  /// In cases of a block resulting in purely combinational logic, no
182  /// scheduleables registered themselves with the block.
183  return {};
184  }
185 
186 private:
187  /// BlockScheduleables is a list of scheduleables that should be
188  /// sequentially executed when executing the associated basic block.
189  DenseMap<mlir::Block *, SmallVector<T>> blockScheduleables;
190 };

196 // Handles state during the lowering of a loop. It will be used for
197 // several lowering patterns.
198 template <typename Loop>
199 class LoopLoweringStateInterface {
200  static_assert(std::is_base_of_v<LoopInterface, Loop>);
201 
202 public:
203  ~LoopLoweringStateInterface() = default;
204 
205  /// Register reg as being the idx'th iter_args register for 'op'.
206  void addLoopIterReg(Loop op, calyx::RegisterOp reg, unsigned idx) {
207  assert(loopIterRegs[op.getOperation()].count(idx) == 0 &&
208  "A register was already registered for the given loop iter_arg "
209  "index");
210  assert(idx < op.getBodyArgs().size());
211  loopIterRegs[op.getOperation()][idx] = reg;
212  }
213 
214  /// Return a mapping of block argument indices to block argument.
215  calyx::RegisterOp getLoopIterReg(Loop op, unsigned idx) {
216  auto iterRegs = getLoopIterRegs(op);
217  auto it = iterRegs.find(idx);
218  assert(it != iterRegs.end() &&
219  "No iter arg register set for the provided index");
220  return it->second;
221  }
222 
223  /// Return a mapping of block argument indices to block argument.
224  const DenseMap<unsigned, calyx::RegisterOp> &getLoopIterRegs(Loop op) {
225  return loopIterRegs[op.getOperation()];
226  }
227 
228  /// Registers grp to be the loop latch group of `op`.
229  void setLoopLatchGroup(Loop op, calyx::GroupOp group) {
230  Operation *operation = op.getOperation();
231  assert(loopLatchGroups.count(operation) == 0 &&
232  "A latch group was already set for this loopOp");
233  loopLatchGroups[operation] = group;
234  }
235 
236  /// Retrieve the loop latch group registered for `op`.
237  calyx::GroupOp getLoopLatchGroup(Loop op) {
238  auto it = loopLatchGroups.find(op.getOperation());
239  assert(it != loopLatchGroups.end() &&
240  "No loop latch group was set for this loopOp");
241  return it->second;
242  }
243 
244  /// Creates a new group that assigns the 'ops' values to the iter arg
245  /// registers of the loop operation.
246  calyx::GroupOp buildLoopIterArgAssignments(OpBuilder &builder, Loop op,
247  calyx::ComponentOp componentOp,
248  Twine uniqueSuffix,
249  MutableArrayRef<OpOperand> ops) {
250  /// Pass iteration arguments through registers. This follows closely
251  /// to what is done for branch ops.
252  std::string groupName = "assign_" + uniqueSuffix.str();
253  auto groupOp = calyx::createGroup<calyx::GroupOp>(builder, componentOp,
254  op.getLoc(), groupName);
255  /// Create register assignment for each iter_arg. a calyx::GroupDone signal
256  /// is created for each register. These will be &'ed together in
257  /// MultipleGroupDonePattern.
258  for (OpOperand &arg : ops) {
259  auto reg = getLoopIterReg(op, arg.getOperandNumber());
260  buildAssignmentsForRegisterWrite(builder, groupOp, componentOp, reg,
261  arg.get());
262  }
263  return groupOp;
264  }
265 
266 private:
267  /// A mapping from loop ops to iteration argument registers.
268  DenseMap<Operation *, DenseMap<unsigned, calyx::RegisterOp>> loopIterRegs;
269 
270  /// A loop latch group is a group that should be sequentially executed when
271  /// finishing a loop body. The execution of this group will write the
272  /// yield'ed loop body values to the iteration argument registers.
273  DenseMap<Operation *, calyx::GroupOp> loopLatchGroups;
274 };

418  /// Returns the component lowering state associated with `op`. If not found
419  /// already found, a new mapping is added for this ComponentOp. Different
420  /// conversions may have different derived classes of the interface, so we
421  /// provided a template.
422  template <typename T = calyx::ComponentLoweringStateInterface>
423  T *getState(calyx::ComponentOp op) {
424  static_assert(std::is_convertible_v<T, ComponentLoweringStateInterface>);
425  auto it = componentStates.find(op);
426  if (it == componentStates.end()) {
427  // Create a new ComponentLoweringState for the compOp.
428  bool success;
429  std::tie(it, success) =
430  componentStates.try_emplace(op, std::make_unique<T>(op));
431  }
432 
433  return static_cast<T *>(it->second.get());
434  }

456 /// Base class for partial lowering passes. A partial lowering pass
457 /// modifies the root operation in place, but does not replace the root
458 /// operation.
459 /// The RewritePatternType template parameter allows for using both
460 /// OpRewritePattern (default) or OpInterfaceRewritePattern.
461 template <class OpType,
462  template <class> class RewritePatternType = OpRewritePattern>
463 class PartialLoweringPattern : public RewritePatternType<OpType> {
464 public:
465  using RewritePatternType<OpType>::RewritePatternType;
466  PartialLoweringPattern(MLIRContext *ctx, LogicalResult &resRef)
467  : RewritePatternType<OpType>(ctx), partialPatternRes(resRef) {}
468 
469  LogicalResult matchAndRewrite(OpType op,
470  PatternRewriter &rewriter) const override {
471  rewriter.updateRootInPlace(
472  op, [&] { partialPatternRes = partiallyLower(op, rewriter); });
473  return partialPatternRes;
474  }
475 
476  virtual LogicalResult partiallyLower(OpType op,
477  PatternRewriter &rewriter) const = 0;
478 
479 private:
480  LogicalResult &partialPatternRes;
481 };
