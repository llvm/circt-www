31 LogicalResult calyx::verifyControlLikeOp(Operation *op) {
32  auto parent = op->getParentOp();
33  // Operations that may parent other ControlLike operations.
34  auto isValidParent = [](Operation *operation) {
35  return isa<ControlOp, SeqOp>(operation);
36  };
37  if (!isValidParent(parent))
38  return op->emitOpError()
39  << "has parent: " << parent
40  << ", which is not allowed for a control-like operation.";
41 
42  if (op->getNumRegions() == 0)
43  return success();
44 
45  auto &region = op->getRegion(0);
46  // Operations that are allowed in the body of a ControlLike op.
47  auto isValidBodyOp = [](Operation *operation) {
48  return isa<EnableOp, SeqOp>(operation);
49  };
50  for (auto &&bodyOp : region.front()) {
51  if (isValidBodyOp(&bodyOp))
52  continue;
53 
54  return op->emitOpError()
55  << "has operation: " << bodyOp.getName()
56  << ", which is not allowed in this control-like operation";
57  }
58  return success();
59 }

78 /// This is a helper function that should only be used to get the WiresOp or
79 /// ControlOp of a ComponentOp, which are guaranteed to exist and generally at
80 /// the end of a component's body. In the worst case, this will run in linear
81 /// time with respect to the number of instances within the cell.
82 template <typename Op>
83 static Op getControlOrWiresFrom(ComponentOp op) {
84  auto body = op.getBody();
85  // We verify there is a single WiresOp and ControlOp,
86  // so this is safe.
87  auto opIt = body->getOps<Op>().begin();
88  return *opIt;
89 }

115 /// Returns the port information for the given component.
116 SmallVector<ComponentPortInfo> calyx::getComponentPortInfo(Operation *op) {
117  assert(isa<ComponentOp>(op) &&
118  "Can only get port information from a component.");
119  auto component = dyn_cast<ComponentOp>(op);
120 
121  auto functionType = getComponentType(component);
122  auto inPortTypes = functionType.getInputs();
123  auto outPortTypes = functionType.getResults();
124  auto inPortNamesAttr = getComponentPortNames(component, PortDirection::INPUT);
125  auto outPortNamesAttr =
126  getComponentPortNames(component, PortDirection::OUTPUT);
127 
128  SmallVector<ComponentPortInfo> results;
129  for (size_t i = 0, e = inPortTypes.size(); i != e; ++i) {
130  results.push_back({inPortNamesAttr[i].cast<StringAttr>(), inPortTypes[i],
131  PortDirection::INPUT});
132  }
133  for (size_t i = 0, e = outPortTypes.size(); i != e; ++i) {
134  results.push_back({outPortNamesAttr[i].cast<StringAttr>(), outPortTypes[i],
135  PortDirection::OUTPUT});
136  }
137  return results;
138 }

165  printPortDefList(p, inputPortTypes, inputPortNames);

175 /// Parses the ports of a Calyx component signature, and adds the corresponding
176 /// port names to `attrName`.
177 static ParseResult
178 parsePortDefList(OpAsmParser &parser, MLIRContext *context,
179  OperationState &result,
180  SmallVectorImpl<OpAsmParser::OperandType> &ports,
181  SmallVectorImpl<Type> &portTypes, StringRef attrName) {
182  if (parser.parseLParen())
183  return failure();
184 
185  do {
186  OpAsmParser::OperandType port;
187  Type portType;
188  if (failed(parser.parseOptionalRegionArgument(port)) ||
189  failed(parser.parseOptionalColon()) ||
190  failed(parser.parseType(portType)))
191  continue;
192  ports.push_back(port);
193  portTypes.push_back(portType);
194  } while (succeeded(parser.parseOptionalComma()));
195 
196  // Add attribute for port names; these are currently
197  // just inferred from the arguments of the component.
198  SmallVector<Attribute> portNames(ports.size());
199  llvm::transform(ports, portNames.begin(), [&](auto port) -> StringAttr {
200  return StringAttr::get(context, port.name);
201  });
202  result.addAttribute(attrName, ArrayAttr::get(context, portNames));
203 
204  return (parser.parseRParen());
205 }

207 /// Parses the signature of a Calyx component.
208 static ParseResult
209 parseComponentSignature(OpAsmParser &parser, OperationState &result,
210  SmallVectorImpl<OpAsmParser::OperandType> &inPorts,
211  SmallVectorImpl<Type> &inPortTypes,
212  SmallVectorImpl<OpAsmParser::OperandType> &outPorts,
213  SmallVectorImpl<Type> &outPortTypes) {
214  auto *context = parser.getBuilder().getContext();
215  if (parsePortDefList(parser, context, result, inPorts, inPortTypes,
216  "inPortNames") ||
217  parser.parseArrow() ||
218  parsePortDefList(parser, context, result, outPorts, outPortTypes,
219  "outPortNames"))
220  return failure();
221 
222  return success();
223 }

225 static ParseResult parseComponentOp(OpAsmParser &parser,
226  OperationState &result) {
227  using namespace mlir::function_like_impl;
228 
229  StringAttr componentName;
230  if (parser.parseSymbolName(componentName, SymbolTable::getSymbolAttrName(),
231  result.attributes))
232  return failure();
233 
234  SmallVector<OpAsmParser::OperandType> inPorts, outPorts;
235  SmallVector<Type> inPortTypes, outPortTypes;
236  if (parseComponentSignature(parser, result, inPorts, inPortTypes, outPorts,
237  outPortTypes))
238  return failure();
239 
240  // Build the component's type for FunctionLike trait.
241  auto &builder = parser.getBuilder();
242  auto type = builder.getFunctionType(inPortTypes, outPortTypes);
243  result.addAttribute(ComponentOp::getTypeAttrName(), TypeAttr::get(type));
244 
245  // The entry block needs to have same number of
246  // input port definitions as the component.
247  auto *body = result.addRegion();
248  if (parser.parseRegion(*body, inPorts, inPortTypes))
249  return failure();
250 
251  if (body->empty())
252  body->push_back(new Block());
253 
254  return success();
255 }

257 static LogicalResult verifyComponentOp(ComponentOp op) {
258  // Verify there is exactly one of each section:
259  // calyx.wires, and calyx.control.
260  uint32_t numWires = 0, numControl = 0;
261  for (auto &bodyOp : *op.getBody()) {
262  if (isa<WiresOp>(bodyOp))
263  ++numWires;
264  else if (isa<ControlOp>(bodyOp))
265  ++numControl;
266  }
267  if (numWires == 1 && numControl == 1)
268  return success();
269 
270  return op.emitOpError() << "requires exactly one of each: "
271  "'calyx.wires', 'calyx.control'.";
272 }

277 static LogicalResult verifyControlOp(ControlOp control) {
278  auto body = control.getBody();
279 
280  // A control operation may have a single EnableOp within it. However,
281  // that must be the only operation. E.g.
282  // Allowed: calyx.control { calyx.enable @A }
283  // Not Allowed: calyx.control { calyx.enable @A calyx.seq { ... } }
284  if (llvm::any_of(*body, [](auto &&op) { return isa<EnableOp>(op); }) &&
285  body->getOperations().size() > 1)
286  return control->emitOpError(
287  "EnableOp is not a composition operator. It should be nested "
288  "in a control flow operation, such as \"calyx.seq\"");
289 
290  return success();
291 }

296 static LogicalResult verifyWiresOp(WiresOp wires) {
297  auto component = wires->getParentOfType<ComponentOp>();
298  auto control = component.getControlOp();
299 
300  // Verify each group is referenced in the control section.
301  for (auto &&op : *wires.getBody()) {
302  if (!isa<GroupOp>(op))
303  continue;
304  auto group = cast<GroupOp>(op);
305  StringRef groupName = group.sym_name();
306  if (SymbolTable::symbolKnownUseEmpty(groupName, control))
307  return op.emitOpError() << "with name: " << groupName
308  << " is unused in the control execution schedule";
309  }
310  return success();
311 }

327 static LogicalResult verifyCellOp(CellOp cell) {
328  if (cell.componentName() == "main")
329  return cell.emitOpError("cannot reference the entry point.");
330 
331  // Verify the referenced component exists in this program.
332  ComponentOp referencedComponent = cell.getReferencedComponent();
333  if (!referencedComponent)
334  return cell.emitOpError()
335  << "is referencing component: " << cell.componentName()
336  << ", which does not exist.";
337 
338  // Verify the referenced component is not instantiating itself.
339  auto parentComponent = cell->getParentOfType<ComponentOp>();
340  if (parentComponent == referencedComponent)
341  return cell.emitOpError()
342  << "is a recursive instantiation of its parent component: "
343  << cell.componentName();
344 
345  // Verify the instance result ports with those of its referenced component.
346  SmallVector<ComponentPortInfo> componentPorts =
347  getComponentPortInfo(referencedComponent);
348 
349  size_t numResults = cell.getNumResults();
350  if (numResults != componentPorts.size())
351  return cell.emitOpError()
352  << "has a wrong number of results; expected: "
353  << componentPorts.size() << " but got " << numResults;
354 
355  for (size_t i = 0; i != numResults; ++i) {
356  auto resultType = cell.getResult(i).getType();
357  auto expectedType = componentPorts[i].type;
358  if (resultType == expectedType)
359  continue;
360  return cell.emitOpError()
361  << "result type for " << componentPorts[i].name << " must be "
362  << expectedType << ", but got " << resultType;
363  }
364  return success();
365 }
