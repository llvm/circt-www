59 /// This pattern collapses a calyx.seq or calyx.par operation when it
60 /// contains exactly one calyx.enable operation.
61 template <typename CtrlOp>
62 struct CollapseUnaryControl : mlir::OpRewritePattern<CtrlOp> {
63  using mlir::OpRewritePattern<CtrlOp>::OpRewritePattern;
64  LogicalResult matchAndRewrite(CtrlOp ctrlOp,
65  PatternRewriter &rewriter) const override {
66  auto &ops = ctrlOp.getBody()->getOperations();
67  bool isUnaryControl = (ops.size() == 1) && isa<EnableOp>(ops.front()) &&
68  isa<SeqOp, ParOp>(ctrlOp->getParentOp());

70  return failure();

73  rewriter.eraseOp(ctrlOp);
74  return success();

79 /// of an AssignOp should be a constant or port, e.g.
80 /// %and = comb.and %a, %b : i1
81 /// calyx.assign %port = %c1_i1 ? %and : i1 // Incorrect

88  Operation *definingOp = op.src().getDefiningOp();

137 /// Verifies the body of a ControlLikeOp.
138 static LogicalResult verifyControlBody(Operation *op) {
139  if (isa<SeqOp, ParOp>(op))
140  // This does not apply to sequential and parallel regions.
141  return success();
142 
143  // Some ControlLike operations have (possibly) multiple regions, e.g. IfOp.
144  for (auto &region : op->getRegions()) {
145  auto opsIt = region.getOps();
146  size_t numOperations = std::distance(opsIt.begin(), opsIt.end());
147  // A body of a ControlLike operation may have a single EnableOp within it.
148  // However, that must be the only operation.
149  // E.g. Allowed: calyx.control { calyx.enable @A }
150  // Not Allowed: calyx.control { calyx.enable @A calyx.seq { ... } }
151  bool usesEnableAsCompositionOperator =
152  numOperations > 1 && llvm::any_of(region.front(), [](auto &&bodyOp) {
153  return isa<EnableOp>(bodyOp);
154  });
155  if (usesEnableAsCompositionOperator)
156  return op->emitOpError(
157  "EnableOp is not a composition operator. It should be nested "
158  "in a control flow operation, such as \"calyx.seq\"");
159 
160  // Verify that multiple control flow operations are nested inside a single
161  // control operator. See: https://github.com/llvm/circt/issues/1723
162  size_t numControlFlowRegions =
163  llvm::count_if(opsIt, [](auto &&op) { return hasControlRegion(&op); });
164  if (numControlFlowRegions > 1)
165  return op->emitOpError(
166  "has an invalid control sequence. Multiple control flow operations "
167  "must all be nested in a single calyx.seq or calyx.par");
168  }
169  return success();
170 }

180 LogicalResult calyx::verifyControlLikeOp(Operation *op) {
181  auto parent = op->getParentOp();
182 
183  if (isa<calyx::EnableOp>(op) &&
184  !isa<calyx::CalyxDialect>(parent->getDialect())) {
185  // Allow embedding calyx.enable ops within other dialects. This is motivated
186  // by allowing experimentation with new styles of Calyx lowering. For more
187  // info and the historical discussion, see:
188  // https://github.com/llvm/circt/pull/3211
189  return success();
190  }
191 
192  if (!hasControlRegion(parent))
193  return op->emitOpError()
194  << "has parent: " << parent
195  << ", which is not allowed for a control-like operation.";
196 
197  if (op->getNumRegions() == 0)
198  return success();
199 
200  auto &region = op->getRegion(0);
201  // Operations that are allowed in the body of a ControlLike op.
202  auto isValidBodyOp = [](Operation *operation) {
203  return isa<EnableOp, SeqOp, IfOp, WhileOp, ParOp>(operation);
204  };
205  for (auto &&bodyOp : region.front()) {
206  if (isValidBodyOp(&bodyOp))
207  continue;
208 
209  return op->emitOpError()
210  << "has operation: " << bodyOp.getName()
211  << ", which is not allowed in this control-like operation";
212  }
213  return verifyControlBody(op);
214 }

216 // Helper function for parsing a group port operation, i.e. GroupDoneOp and
217 // GroupPortOp. These may take one of two different forms:
218 // (1) %<guard> ? %<src> : i1
219 // (2) %<src> : i1
220 static ParseResult parseGroupPort(OpAsmParser &parser, OperationState &result) {
221  SmallVector<OpAsmParser::UnresolvedOperand, 2> operandInfos;
222  OpAsmParser::UnresolvedOperand guardOrSource;
223  if (parser.parseOperand(guardOrSource))
224  return failure();
225 
226  if (succeeded(parser.parseOptionalQuestion())) {
227  OpAsmParser::UnresolvedOperand source;
228  // The guard exists.
229  if (parser.parseOperand(source))
230  return failure();
231  operandInfos.push_back(source);
232  }
233  // No matter if this is the source or guard, it should be last.
234  operandInfos.push_back(guardOrSource);
235 
236  Type type;
237  // Resolving the operands with the same type works here since the source and
238  // guard of a group port is always i1.
239  if (parser.parseColonType(type) ||
240  parser.resolveOperands(operandInfos, type, result.operands))
241  return failure();
242 
243  return success();
244 }

290 /// A helper function to check whether the conditional and group (if it exists)
291 /// needs to be erased to maintain a valid state of a Calyx program. If these
292 /// have no more uses, they will be erased.
293 template <typename OpTy>
294 static void eraseControlWithGroupAndConditional(OpTy op,
295  PatternRewriter &rewriter) {
296  static_assert(std::is_same<OpTy, IfOp>() || std::is_same<OpTy, WhileOp>(),
297  "This is only applicable to WhileOp and IfOp.");
298 
299  // Save information about the operation, and erase it.
300  Value cond = op.cond();
301  Optional<StringRef> groupName = op.groupName();
302  auto component = op->template getParentOfType<ComponentOp>();
303  rewriter.eraseOp(op);
304 
305  // Clean up the attached conditional and combinational group (if it exists).
306  if (groupName.hasValue()) {
307  auto group = component.getWiresOp().template lookupSymbol<GroupInterface>(
308  *groupName);
309  if (SymbolTable::symbolKnownUseEmpty(group, component.getRegion()))
310  rewriter.eraseOp(group);
311  }
312  // Check the conditional after the Group, since it will be driven within.
313  if (!cond.isa<BlockArgument>() && cond.getDefiningOp()->use_empty())
314  rewriter.eraseOp(cond.getDefiningOp());
315 }

480 /// Parses the signature of a Calyx component.
481 static ParseResult
482 parseComponentSignature(OpAsmParser &parser, OperationState &result,
483  SmallVectorImpl<OpAsmParser::Argument> &ports,
484  SmallVectorImpl<Type> &portTypes) {
485  SmallVector<OpAsmParser::Argument> inPorts, outPorts;
486  SmallVector<Type> inPortTypes, outPortTypes;
487  SmallVector<NamedAttrList> portAttributes;
488 
489  if (parsePortDefList(parser, result, inPorts, inPortTypes, portAttributes))
490  return failure();
491 
492  if (parser.parseArrow() ||
493  parsePortDefList(parser, result, outPorts, outPortTypes, portAttributes))
494  return failure();
495 
496  auto *context = parser.getBuilder().getContext();
497  // Add attribute for port names; these are currently
498  // just inferred from the SSA names of the component.
499  SmallVector<Attribute> portNames;
500  auto getPortName = [context](const auto &port) -> StringAttr {
501  StringRef name = port.ssaName.name;
502  if (name.startswith("%"))
503  name = name.drop_front();
504  return StringAttr::get(context, name);
505  };
506  llvm::transform(inPorts, std::back_inserter(portNames), getPortName);
507  llvm::transform(outPorts, std::back_inserter(portNames), getPortName);
508 
509  result.addAttribute("portNames", ArrayAttr::get(context, portNames));
510  result.addAttribute(
511  "portDirections",
512  direction::packAttribute(context, inPorts.size(), outPorts.size()));
513 
514  ports.append(inPorts);
515  ports.append(outPorts);
516  portTypes.append(inPortTypes);
517  portTypes.append(outPortTypes);
518 
519  SmallVector<Attribute> portAttrs;
520  llvm::transform(portAttributes, std::back_inserter(portAttrs),
521  [&](auto attr) { return attr.getDictionary(context); });
522  result.addAttribute("portAttributes", ArrayAttr::get(context, portAttrs));
523 
524  return success();
525 }

528  using namespace mlir::function_interface_impl;
529 
530  StringAttr componentName;

539  if (parseComponentSignature(parser, result, ports, portTypes))
540  return failure();

544  auto type =

556  return failure();
557

561 /// Determines whether the given ComponentOp has all the required ports.
562 static LogicalResult hasRequiredPorts(ComponentOp op) {
563  // Get all identifiers from the component ports.
564  llvm::SmallVector<StringRef, 4> identifiers;
565  for (PortInfo &port : op.getPortInfo()) {
566  auto portIds = port.getAllIdentifiers();
567  identifiers.append(portIds.begin(), portIds.end());
568  }
569  // Sort the identifiers: a pre-condition for std::set_intersection.
570  std::sort(identifiers.begin(), identifiers.end());
571 
572  llvm::SmallVector<StringRef, 4> intersection,
573  interfacePorts{"clk", "done", "go", "reset"};
574  // Find the intersection between all identifiers and required ports.
575  std::set_intersection(interfacePorts.begin(), interfacePorts.end(),
576  identifiers.begin(), identifiers.end(),
577  std::back_inserter(intersection));
578 
579  if (intersection.size() == interfacePorts.size())
580  return success();
581 
582  SmallVector<StringRef, 4> difference;
583  std::set_difference(interfacePorts.begin(), interfacePorts.end(),
584  intersection.begin(), intersection.end(),
585  std::back_inserter(difference));
586  return op->emitOpError()
587  << "is missing the following required port attribute identifiers: "
588  << difference;
589 }

591 LogicalResult ComponentOp::verify() {
592  // Verify there is exactly one of each the wires and control operations.
593  auto wIt = getBody()->getOps<WiresOp>();
594  auto cIt = getBody()->getOps<ControlOp>();
595  if (std::distance(wIt.begin(), wIt.end()) +
596  std::distance(cIt.begin(), cIt.end()) !=
597  2)
598  return emitOpError(
599  "requires exactly one of each: 'calyx.wires', 'calyx.control'.");
600 
601  if (failed(hasRequiredPorts(*this)))
602  return failure();
603 
604  // Verify the component actually does something: has a non-empty Control
605  // region, or continuous assignments.
606  bool hasNoControlConstructs =
607  getControlOp().getBody()->getOperations().empty();
608  bool hasNoAssignments = getWiresOp().getBody()->getOps<AssignOp>().empty();
609  if (hasNoControlConstructs && hasNoAssignments)
610  return emitOpError(
611  "The component currently does nothing. It needs to either have "
612  "continuous assignments in the Wires region or control constructs in "
613  "the Control region.");
614 
615  return success();
616 }

631 
632  result.addAttribute(::mlir::SymbolTable::getSymbolAttrName(), name);

636  std::pair<SmallVector<Attribute, 8>, SmallVector<Attribute, 8>>
637  portIOAttributes;
638  SmallVector<Direction, 8> portDirections;

644  (isInput ? portIONames.first : portIONames.second).push_back(port.name);

650  auto portAttributes = concat(portIOAttributes.first, portIOAttributes.second);

659  direction::packAttribute(builder.getContext(),
660  portIOTypes.first.size(),
661  portIOTypes.second.size()));
662  // Record the attributes of the ports.
663  result.addAttribute("portAttributes", builder.getArrayAttr(portAttributes));

782 /// Verifies the defining operation of a value is combinational.
783 static LogicalResult isCombinational(Value value, GroupInterface group) {
784  Operation *definingOp = value.getDefiningOp();
785  if (definingOp == nullptr || definingOp->hasTrait<Combinational>())
786  // This is a port of the parent component or combinational.
787  return success();
788 
789  // For now, assumes all component instances are combinational. Once
790  // combinational components are supported, this can be strictly enforced.
791  if (isa<InstanceOp>(definingOp))
792  return success();
793 
794  // Constants and logical operations are OK.
795  if (isa<comb::CombDialect, hw::HWDialect>(definingOp->getDialect()))
796  return success();

805  "combinational.";

807  auto writePorts = {m.writeData(), m.writeEn()};

809  ? success()
810  : group->emitOpError()
811  << "with memory: \"" << m.instanceName()
812  << "\" is conducting a memory store. This "
813  "is not combinational.";
814  }

816  std::string portName =
817  valueName(group->getParentOfType<ComponentOp>(), value);
818  return group->emitOpError() << "with port: " << portName
819  << ". This operation is not combinational.";
820 }

855 /// Determines whether the given port is used in the group. Its use depends on
856 /// the `isDriven` value; if true, then the port should be a destination in an
857 /// AssignOp. Otherwise, it should be the source, i.e. a read.
858 static bool portIsUsedInGroup(GroupInterface group, Value port, bool isDriven) {
859  return llvm::any_of(port.getUses(), [&](auto &&use) {
860  auto assignOp = dyn_cast<AssignOp>(use.getOwner());
861  if (assignOp == nullptr)
862  return false;
863

865  if (isa<WiresOp>(parent))
866  // This is a continuous assignment.
867  return false;

870  // (1) it is a {source, destination} of an assignment.
871  // (2) that assignment is found in the provided group.
872 
873  // If not driven, then read.
874  Value expected = isDriven ? assignOp.dest() : assignOp.src();
875  return expected == port && group == parent;
876  });

950 /// Verifies that certain ports of primitives are either driven or read
951 /// together.
952 static LogicalResult verifyPrimitivePortDriving(AssignOp assign,
953  GroupInterface group) {
954  Operation *destDefiningOp = assign.dest().getDefiningOp();
955  if (destDefiningOp == nullptr)
956  return success();
957  auto destCell = dyn_cast<CellInterface>(destDefiningOp);
958  if (destCell == nullptr)
959  return success();
960 
961  LogicalResult verifyWrites =
962  TypeSwitch<Operation *, LogicalResult>(destCell)
963  .Case<RegisterOp>([&](auto op) {
964  // We only want to verify this is written to if the {write enable,
965  // in} port is driven.
966  return succeeded(group.drivesAnyPort({op.write_en(), op.in()}))
967  ? group.drivesAllPorts({op.write_en(), op.in()})
968  : success();
969  })
970  .Case<MemoryOp>([&](auto op) {
971  SmallVector<Value> requiredWritePorts;
972  // If writing to memory, write_en, write_data, and all address ports
973  // should be driven.
974  requiredWritePorts.push_back(op.writeEn());
975  requiredWritePorts.push_back(op.writeData());
976  for (Value address : op.addrPorts())
977  requiredWritePorts.push_back(address);
978 
979  // We only want to verify the write ports if either write_data or
980  // write_en is driven.
981  return succeeded(
982  group.drivesAnyPort({op.writeData(), op.writeEn()}))
983  ? group.drivesAllPorts(requiredWritePorts)
984  : success();
985  })
986  .Case<AndLibOp, OrLibOp, XorLibOp, AddLibOp, SubLibOp, GtLibOp,
987  LtLibOp, EqLibOp, NeqLibOp, GeLibOp, LeLibOp, LshLibOp,
988  RshLibOp, SgtLibOp, SltLibOp, SeqLibOp, SneqLibOp, SgeLibOp,
989  SleLibOp, SrshLibOp>([&](auto op) {
990  Value lhs = op.left(), rhs = op.right();
991  return succeeded(group.drivesAnyPort({lhs, rhs}))
992  ? group.drivesAllPorts({lhs, rhs})
993  : success();
994  })
995  .Default([&](auto op) { return success(); });
996 
997  if (failed(verifyWrites))
998  return group->emitOpError()
999  << "with cell: " << destCell->getName() << " \""
1000  << destCell.instanceName()
1001  << "\" is performing a write and failed to drive all necessary "
1002  "ports.";
1003 
1004  Operation *srcDefiningOp = assign.src().getDefiningOp();
1005  if (srcDefiningOp == nullptr)
1006  return success();
1007  auto srcCell = dyn_cast<CellInterface>(srcDefiningOp);
1008  if (srcCell == nullptr)
1009  return success();
1010 
1011  LogicalResult verifyReads =
1012  TypeSwitch<Operation *, LogicalResult>(srcCell)
1013  .Case<MemoryOp>([&](auto op) {
1014  // If reading memory, all address ports should be driven. Note that
1015  // we only want to verify the read ports if read_data is used in the
1016  // group.
1017  return succeeded(group.readsAnyPort({op.readData()}))
1018  ? group.drivesAllPorts(op.addrPorts())
1019  : success();
1020  })
1021  .Default([&](auto op) { return success(); });
1022 
1023  if (failed(verifyReads))
1024  return group->emitOpError() << "with cell: " << srcCell->getName() << " \""
1025  << srcCell.instanceName()
1026  << "\" is having a read performed upon it, and "
1027  "failed to drive all necessary ports.";
1028 
1029  return success();
1030 }

1127 ParseResult AssignOp::parse(OpAsmParser &parser, OperationState &result) {
1128  OpAsmParser::UnresolvedOperand destination;
1129  if (parser.parseOperand(destination) || parser.parseEqual())
1130  return failure();
1131 
1132  // An AssignOp takes one of the two following forms:
1133  // (1) %<dest> = %<src> : <type>
1134  // (2) %<dest> = %<guard> ? %<src> : <type>
1135  OpAsmParser::UnresolvedOperand guardOrSource;
1136  if (parser.parseOperand(guardOrSource))
1137  return failure();
1138 
1139  // Since the guard is optional, we need to check if there is an accompanying
1140  // `?` symbol.
1141  OpAsmParser::UnresolvedOperand source;
1142  bool hasGuard = succeeded(parser.parseOptionalQuestion());
1143  if (hasGuard) {
1144  // The guard exists. Parse the source.
1145  if (parser.parseOperand(source))
1146  return failure();
1147  }
1148 
1149  Type type;
1150  if (parser.parseColonType(type) ||
1151  parser.resolveOperand(destination, type, result.operands))
1152  return failure();
1153 
1154  if (hasGuard) {
1155  Type i1Type = parser.getBuilder().getI1Type();
1156  // Since the guard is optional, it is listed last in the arguments of the
1157  // AssignOp. Therefore, we must parse the source first.
1158  if (parser.resolveOperand(source, type, result.operands) ||
1159  parser.resolveOperand(guardOrSource, i1Type, result.operands))
1160  return failure();
1161  } else {
1162  // This is actually a source.
1163  if (parser.resolveOperand(guardOrSource, type, result.operands))
1164  return failure();
1165  }
1166 
1167  return success();
1168 }

1542 /// Checks preconditions for the common tail pattern. This canonicalization is
1543 /// stringent about not entering nested control operations, as this may cause
1544 /// unintentional changes in behavior.
1545 /// We only look for two cases: (1) both regions are ParOps, and
1546 /// (2) both regions are SeqOps. The case when these are different, e.g. ParOp
1547 /// and SeqOp, will only produce less optimal code, or even worse, change the
1548 /// behavior.
1549 template <typename OpTy>
1550 static bool hasCommonTailPatternPreConditions(IfOp op) {
1551  static_assert(std::is_same<SeqOp, OpTy>() || std::is_same<ParOp, OpTy>(),
1552  "Should be a SeqOp or ParOp.");
1553 
1554  if (!op.thenBodyExists() || !op.elseBodyExists())
1555  return false;
1556  if (op.getThenBody()->empty() || op.getElseBody()->empty())
1557  return false;
1558 
1559  Block *thenBody = op.getThenBody(), *elseBody = op.getElseBody();
1560  return isa<OpTy>(thenBody->front()) && isa<OpTy>(elseBody->front());
1561 }
1562 
1563 /// seq {
1564 /// if %a with @G { if %a with @G {
1565 /// seq { ... calyx.enable @A } seq { ... }
1566 /// else { -> } else {
1567 /// seq { ... calyx.enable @A } seq { ... }
1568 /// } }
1569 /// calyx.enable @A
1570 /// }
1571 struct CommonTailPatternWithSeq : mlir::OpRewritePattern<IfOp> {
1572  using mlir::OpRewritePattern<IfOp>::OpRewritePattern;
1573 
1574  LogicalResult matchAndRewrite(IfOp ifOp,
1575  PatternRewriter &rewriter) const override {
1576  if (!hasCommonTailPatternPreConditions<SeqOp>(ifOp))
1577  return failure();
1578 
1579  auto thenControl = cast<SeqOp>(ifOp.getThenBody()->front()),
1580  elseControl = cast<SeqOp>(ifOp.getElseBody()->front());
1581  Optional<EnableOp> lastThenEnableOp = getLastEnableOp(thenControl),
1582  lastElseEnableOp = getLastEnableOp(elseControl);
1583 
1584  if (!lastThenEnableOp.hasValue() || !lastElseEnableOp.hasValue())
1585  return failure();
1586  if (lastThenEnableOp->groupName() != lastElseEnableOp->groupName())
1587  return failure();
1588 
1589  // Place the IfOp and pulled EnableOp inside a sequential region, in case
1590  // this IfOp is nested in a ParOp. This avoids unintentionally
1591  // parallelizing the pulled out EnableOps.
1592  rewriter.setInsertionPointAfter(ifOp);
1593  SeqOp seqOp = rewriter.create<SeqOp>(ifOp.getLoc());
1594  Block *body = seqOp.getBody();
1595  ifOp->remove();
1596  body->push_back(ifOp);
1597  rewriter.setInsertionPointToEnd(body);
1598  rewriter.create<EnableOp>(seqOp.getLoc(), lastThenEnableOp->groupName());
1599 
1600  // Erase the common EnableOp from the Then and Else regions.
1601  rewriter.eraseOp(*lastThenEnableOp);
1602  rewriter.eraseOp(*lastElseEnableOp);
1603  return success();
1604  }
1605 };

1607 /// if %a with @G { par {
1608 /// par { if %a with @G {
1609 /// ... par { ... }
1610 /// calyx.enable @A } else {
1611 /// calyx.enable @B -> par { ... }
1612 /// } }
1613 /// } else { calyx.enable @A
1614 /// par { calyx.enable @B
1615 /// ... }
1616 /// calyx.enable @A
1617 /// calyx.enable @B
1618 /// }
1619 /// }
1620 struct CommonTailPatternWithPar : mlir::OpRewritePattern<IfOp> {
1621  using mlir::OpRewritePattern<IfOp>::OpRewritePattern;
1622 
1623  LogicalResult matchAndRewrite(IfOp ifOp,
1624  PatternRewriter &rewriter) const override {
1625  if (!hasCommonTailPatternPreConditions<ParOp>(ifOp))
1626  return failure();
1627  auto thenControl = cast<ParOp>(ifOp.getThenBody()->front()),
1628  elseControl = cast<ParOp>(ifOp.getElseBody()->front());
1629 
1630  llvm::StringMap<EnableOp> A = getAllEnableOpsInImmediateBody(thenControl),
1631  B = getAllEnableOpsInImmediateBody(elseControl);
1632 
1633  // Compute the intersection between `A` and `B`.
1634  SmallVector<StringRef> groupNames;
1635  for (auto a = A.begin(); a != A.end(); ++a) {
1636  StringRef groupName = a->getKey();
1637  auto b = B.find(groupName);
1638  if (b == B.end())
1639  continue;
1640  // This is also an element in B.
1641  groupNames.push_back(groupName);
1642  // Since these are being pulled out, erase them.
1643  rewriter.eraseOp(a->getValue());
1644  rewriter.eraseOp(b->getValue());
1645  }
1646  // Place the IfOp and EnableOp(s) inside a parallel region, in case this
1647  // IfOp is nested in a SeqOp. This avoids unintentionally sequentializing
1648  // the pulled out EnableOps.
1649  rewriter.setInsertionPointAfter(ifOp);
1650  ParOp parOp = rewriter.create<ParOp>(ifOp.getLoc());
1651  Block *body = parOp.getBody();
1652  ifOp->remove();
1653  body->push_back(ifOp);
1654 
1655  // Pull out the intersection between these two sets, and erase their
1656  // counterparts in the Then and Else regions.
1657  rewriter.setInsertionPointToEnd(body);
1658  for (StringRef groupName : groupNames)
1659  rewriter.create<EnableOp>(parOp.getLoc(), groupName);
1660 
1661  return success();
1662  }
1663 };
