43 /// Signals that the following operation is combinational.
44 template <typename ConcreteType>
45 class Combinational
46  : public mlir::OpTrait::TraitBase<ConcreteType, Combinational> {
47 public:
48  static LogicalResult verifyTrait(Operation *op) {
49  Attribute staticAttribute = op->getAttr("static");
50  if (staticAttribute == nullptr)
51  return success();
52 
53  // If the operation has the static attribute, verify it is zero.
54  APInt staticValue = staticAttribute.cast<IntegerAttr>().getValue();
55  assert(staticValue == 0 && "If combinational, it should take 0 cycles.");
56 
57  return success();
58  }
59 };

75 /// This holds information about the port for either a Component or Cell.
76 struct PortInfo {
77  StringAttr name;
78  Type type;
79  Direction direction;
80  DictionaryAttr attributes;
81 
82  /// Returns whether the given port has attribute with Identifier `name`.
83  bool hasAttribute(StringRef identifier) const {
84  assert(attributes && "PortInfo::attributes should be instantiated.");
85  return llvm::any_of(attributes, [&](auto idToAttribute) {
86  return identifier == idToAttribute.getName();
87  });
88  }
89 
90  /// Returns the attribute associated with the given name if it exists,
91  /// otherwise std::nullopt.
92  llvm::Optional<Attribute> getAttribute(StringRef identifier) const {
93  assert(attributes && "PortInfo::attributes should be instantiated.");
94  auto it = llvm::find_if(attributes, [&](auto idToAttribute) {
95  return identifier == idToAttribute.getName();
96  });
97  if (it == attributes.end())
98  return None;
99  return it->getValue();
100  }
101 
102  /// Returns all identifiers for this dictionary attribute.
103  SmallVector<StringRef> getAllIdentifiers() const {
104  assert(attributes && "PortInfo::attributes should be instantiated.");
105  SmallVector<StringRef> identifiers;
106  llvm::transform(attributes, std::back_inserter(identifiers),
107  [](auto idToAttribute) { return idToAttribute.getName(); });
108  return identifiers;
109  }
110 };
