38 struct ConvertProgramOp : public OpConversionPattern<ProgramOp> {
39  using OpConversionPattern::OpConversionPattern;
40 
41  LogicalResult
42  matchAndRewrite(ProgramOp program, OpAdaptor adaptor,
43  ConversionPatternRewriter &rewriter) const override {
44  ModuleOp mod = program->getParentOfType<ModuleOp>();
45  rewriter.inlineRegionBefore(program.body(), &mod.getBodyRegion().front());

47  return success();

51 struct ConvertComponentOp : public OpConversionPattern<ComponentOp> {
52  using OpConversionPattern::OpConversionPattern;
53 
54  LogicalResult
55  matchAndRewrite(ComponentOp component, OpAdaptor adaptor,
56  ConversionPatternRewriter &rewriter) const override {
57  SmallVector<hw::PortInfo> hwInputInfo;
58  for (auto [name, type, direction, _] : component.getPortInfo())
59  hwInputInfo.push_back({name, hwDirection(direction), type});
60 
61  auto hwMod = rewriter.create<HWModuleOp>(
62  component.getLoc(), component.getNameAttr(), hwInputInfo);
63 
64  rewriter.eraseOp(hwMod.getBodyBlock()->getTerminator());
65  ConversionPatternRewriter::InsertionGuard g(rewriter);
66  rewriter.setInsertionPointToEnd(hwMod.getBodyBlock());
67 
68  SmallVector<Value> argValues;
69  SmallVector<Value> outputWires;
70  size_t portIdx = 0;
71  for (auto [name, type, direction, _] : component.getPortInfo()) {
72  switch (direction) {
73  case calyx::Direction::Input:
74  assert(hwMod.getArgument(portIdx).getType() == type);
75  argValues.push_back(hwMod.getArgument(portIdx));
76  break;
77  case calyx::Direction::Output:
78  auto wire = rewriter.create<sv::WireOp>(component.getLoc(), type, name);
79  auto wireRead =
80  rewriter.create<sv::ReadInOutOp>(component.getLoc(), wire);
81  argValues.push_back(wireRead);
82  outputWires.push_back(wireRead);
83  break;
84  }
85  ++portIdx;
86  }
87 
88  rewriter.mergeBlocks(component.getBody(), hwMod.getBodyBlock(), argValues);
89  rewriter.create<OutputOp>(component.getLoc(), outputWires);
90  rewriter.eraseOp(component);
91 
92  return success();
93  }
94 
95 private:
96  hw::PortDirection hwDirection(calyx::Direction dir) const {
97  switch (dir) {
98  case calyx::Direction::Input:
99  return hw::PortDirection::INPUT;
100  case calyx::Direction::Output:
101  return hw::PortDirection::OUTPUT;
102  }
103  }
104 };

135 struct ConvertAssignOp : public OpConversionPattern<calyx::AssignOp> {
136  using OpConversionPattern::OpConversionPattern;
137 
138  LogicalResult
139  matchAndRewrite(calyx::AssignOp assign, OpAdaptor adaptor,
140  ConversionPatternRewriter &rewriter) const override {
141  Value dest = adaptor.dest();
142 
143  // To make life easy in ConvertComponentOp, we read from the output wires so
144  // the dialect conversion block argument mapping would work without a type
145  // converter. This means assigns to ComponentOp outputs will try to assign
146  // to a read from a wire, so we need to map to the wire.
147  if (auto readInOut = dyn_cast<ReadInOutOp>(adaptor.dest().getDefiningOp()))
148  dest = readInOut.input();
149 
150  Value src = adaptor.src();
151  if (auto guard = adaptor.guard()) {
152  auto zero =
153  rewriter.create<hw::ConstantOp>(assign.getLoc(), src.getType(), 0);
154  src = rewriter.create<MuxOp>(assign.getLoc(), guard, src, zero);
155  }
156 
157  rewriter.replaceOpWithNewOp<sv::AssignOp>(assign, dest, src);
158 
159  return success();
160  }
161 };

163 struct ConvertCellOp : public OpInterfaceConversionPattern<CellInterface> {
164  using OpInterfaceConversionPattern::OpInterfaceConversionPattern;
165 
166  LogicalResult
167  matchAndRewrite(CellInterface cell, ArrayRef<Value> operands,
168  ConversionPatternRewriter &rewriter) const override {
169  assert(operands.empty() && "calyx cells do not have operands");
170 
171  SmallVector<Value> wires;
172  ImplicitLocOpBuilder builder(cell.getLoc(), rewriter);
173  convertPrimitiveOp(cell, wires, builder);
174  if (wires.size() != cell.getPortInfo().size()) {
175  auto diag = cell.emitOpError("couldn't convert to core primitive");
176  for (Value wire : wires)
177  diag.attachNote() << "with wire: " << wire;
178  return diag;
179  }
180 
181  rewriter.replaceOp(cell, wires);
182 
183  return success();
184  }

187  void convertPrimitiveOp(Operation *op, SmallVectorImpl<Value> &wires,
188  ImplicitLocOpBuilder &b) const {
189  TypeSwitch<Operation *>(op)
190  // Comparison operations.
191  .Case([&](EqLibOp op) {
192  convertCompareBinaryOp(op, ICmpPredicate::eq, wires, b);
193  })
194  .Case([&](NeqLibOp op) {
195  convertCompareBinaryOp(op, ICmpPredicate::ne, wires, b);
196  })
197  .Case([&](LtLibOp op) {
198  convertCompareBinaryOp(op, ICmpPredicate::ult, wires, b);
199  })
200  .Case([&](LeLibOp op) {
201  convertCompareBinaryOp(op, ICmpPredicate::ule, wires, b);
202  })
203  .Case([&](GtLibOp op) {
204  convertCompareBinaryOp(op, ICmpPredicate::ugt, wires, b);
205  })
206  .Case([&](GeLibOp op) {
207  convertCompareBinaryOp(op, ICmpPredicate::uge, wires, b);
208  })
209  .Case([&](SltLibOp op) {
210  convertCompareBinaryOp(op, ICmpPredicate::slt, wires, b);
211  })
212  .Case([&](SleLibOp op) {
213  convertCompareBinaryOp(op, ICmpPredicate::sle, wires, b);
214  })
215  .Case([&](SgtLibOp op) {
216  convertCompareBinaryOp(op, ICmpPredicate::sgt, wires, b);
217  })
218  .Case([&](SgeLibOp op) {
219  convertCompareBinaryOp(op, ICmpPredicate::sge, wires, b);
220  })
221  // Combinational arithmetic and logical operations.
222  .Case([&](AddLibOp op) {
223  convertArithBinaryOp<AddLibOp, AddOp>(op, wires, b);
224  })
225  .Case([&](SubLibOp op) {
226  convertArithBinaryOp<SubLibOp, SubOp>(op, wires, b);
227  })
228  .Case([&](RshLibOp op) {
229  convertArithBinaryOp<RshLibOp, ShrUOp>(op, wires, b);
230  })
231  .Case([&](SrshLibOp op) {
232  convertArithBinaryOp<SrshLibOp, ShrSOp>(op, wires, b);
233  })
234  .Case([&](LshLibOp op) {
235  convertArithBinaryOp<LshLibOp, ShlOp>(op, wires, b);
236  })
237  .Case([&](AndLibOp op) {
238  convertArithBinaryOp<AndLibOp, AndOp>(op, wires, b);
239  })
240  .Case([&](OrLibOp op) {
241  convertArithBinaryOp<OrLibOp, OrOp>(op, wires, b);
242  })
243  .Case([&](XorLibOp op) {
244  convertArithBinaryOp<XorLibOp, XorOp>(op, wires, b);
245  })
246  // Pipelined arithmetic operations.
247  .Case([&](MultPipeLibOp op) {
248  convertPipelineOp<MultPipeLibOp, comb::MulOp>(op, wires, b);
249  })
250  .Case([&](DivUPipeLibOp op) {
251  convertPipelineOp<DivUPipeLibOp, comb::DivUOp>(op, wires, b);
252  })
253  .Case([&](DivSPipeLibOp op) {
254  convertPipelineOp<DivSPipeLibOp, comb::DivSOp>(op, wires, b);
255  })
256  .Case([&](RemSPipeLibOp op) {
257  convertPipelineOp<RemSPipeLibOp, comb::ModSOp>(op, wires, b);
258  })
259  .Case([&](RemUPipeLibOp op) {
260  convertPipelineOp<RemUPipeLibOp, comb::ModUOp>(op, wires, b);
261  })
262  // Sequential operations.
263  .Case([&](RegisterOp op) {
264  auto in = wireIn(op.in(), op.instanceName(), op.portName(op.in()), b);
265  auto writeEn = wireIn(op.write_en(), op.instanceName(),
266  op.portName(op.write_en()), b);
267  auto clk =
268  wireIn(op.clk(), op.instanceName(), op.portName(op.clk()), b);
269  auto reset =
270  wireIn(op.reset(), op.instanceName(), op.portName(op.reset()), b);
271 
272  auto outReg = reg(in, clk, reset, op.instanceName() + "_reg", b);
273  auto doneReg =
274  reg(writeEn, clk, reset, op.instanceName() + "_done_reg", b);
275 
276  auto out = wireOut(outReg, op.instanceName(), "", b);
277  auto done =
278  wireOut(doneReg, op.instanceName(), op.portName(op.done()), b);
279  wires.append({in.input(), writeEn.input(), clk.input(), reset.input(),
280  out, done});
281  })
282  // Unary operqations.
283  .Case([&](SliceLibOp op) {
284  auto in = wireIn(op.in(), op.instanceName(), op.portName(op.in()), b);
285  auto outWidth = op.out().getType().getIntOrFloatBitWidth();
286 
287  auto extract = b.create<ExtractOp>(in, 0, outWidth);
288 
289  auto out =
290  wireOut(extract, op.instanceName(), op.portName(op.out()), b);
291  wires.append({in.input(), out});
292  })
293  .Case([&](NotLibOp op) {
294  auto in = wireIn(op.in(), op.instanceName(), op.portName(op.in()), b);
295  auto one = b.create<hw::ConstantOp>(op.in().getType(), 0);
296 
297  auto xorOp = b.create<XorOp>(in, one);
298 
299  auto out =
300  wireOut(xorOp, op.instanceName(), op.portName(op.out()), b);
301  wires.append({in.input(), out});
302  })
303  .Case([&](WireLibOp op) {
304  auto wire = wireIn(op.in(), op.instanceName(), "", b);
305  wires.append({wire.input(), wire});
306  })
307  .Case([&](PadLibOp op) {
308  auto in = wireIn(op.in(), op.instanceName(), op.portName(op.in()), b);
309  auto srcWidth = in.getType().getIntOrFloatBitWidth();
310  auto destWidth = op.out().getType().getIntOrFloatBitWidth();
311  auto zero = b.create<hw::ConstantOp>(op.getLoc(),
312  APInt(destWidth - srcWidth, 0));
313  auto padded = wireOut(b.createOrFold<comb::ConcatOp>(zero, in),
314  op.instanceName(), op.portName(op.out()), b);
315  wires.append({in.input(), padded});
316  })
317  .Case([&](ExtSILibOp op) {
318  auto in = wireIn(op.in(), op.instanceName(), op.portName(op.in()), b);
319  auto extsi =
320  wireOut(createOrFoldSExt(op.getLoc(), in, op.out().getType(), b),
321  op.instanceName(), op.portName(op.out()), b);
322  wires.append({in.input(), extsi});
323  })
324  .Default([](Operation *) { return SmallVector<Value>(); });
325  }

327  template <typename OpTy, typename ResultTy>
328  void convertArithBinaryOp(OpTy op, SmallVectorImpl<Value> &wires,
329  ImplicitLocOpBuilder &b) const {
330  auto left = wireIn(op.left(), op.instanceName(), op.portName(op.left()), b);
331  auto right =
332  wireIn(op.right(), op.instanceName(), op.portName(op.right()), b);
333 
334  auto add = b.create<ResultTy>(left, right);
335 
336  auto out = wireOut(add, op.instanceName(), op.portName(op.out()), b);
337  wires.append({left.input(), right.input(), out});
338  }

340  template <typename OpTy>
341  void convertCompareBinaryOp(OpTy op, ICmpPredicate pred,
342  SmallVectorImpl<Value> &wires,
343  ImplicitLocOpBuilder &b) const {
344  auto left = wireIn(op.left(), op.instanceName(), op.portName(op.left()), b);
345  auto right =
346  wireIn(op.right(), op.instanceName(), op.portName(op.right()), b);
347 
348  auto add = b.create<ICmpOp>(pred, left, right);
349 
350  auto out = wireOut(add, op.instanceName(), op.portName(op.out()), b);
351  wires.append({left.input(), right.input(), out});
352  }

354  template <typename SrcOpTy, typename TargetOpTy>
355  void convertPipelineOp(SrcOpTy op, SmallVectorImpl<Value> &wires,
356  ImplicitLocOpBuilder &b) const {
357  auto clk = wireIn(op.clk(), op.instanceName(), op.portName(op.clk()), b);
358  auto reset =
359  wireIn(op.reset(), op.instanceName(), op.portName(op.reset()), b);
360  auto go = wireIn(op.go(), op.instanceName(), op.portName(op.go()), b);
361  auto left = wireIn(op.left(), op.instanceName(), op.portName(op.left()), b);
362  auto right =
363  wireIn(op.right(), op.instanceName(), op.portName(op.right()), b);
364  wires.append(
365  {clk.input(), reset.input(), go.input(), left.input(), right.input()});
366 
367  auto targetOp = b.create<TargetOpTy>(left, right);
368  for (auto &&[targetRes, sourceRes] :
369  llvm::zip(targetOp->getResults(), op.getOutputPorts())) {
370  auto portName = op.portName(sourceRes);
371  auto resReg = reg(targetRes, clk, reset,
372  createName(op.instanceName(), portName), b);
373  wires.push_back(wireOut(resReg, op.instanceName(), portName, b));
374  }
375 
376  auto doneReg = reg(go, clk, reset,
377  op.instanceName() + "_" + op.portName(op.done()), b);
378  auto done = wireOut(doneReg, op.instanceName(), op.portName(op.done()), b);
379  wires.push_back(done);
380  }

432 LogicalResult CalyxToHWPass::runOnProgram(ProgramOp program) {
433  MLIRContext &context = getContext();
434 
435  ConversionTarget target(context);
436  target.addIllegalDialect<CalyxDialect>();
437  target.addLegalDialect<HWDialect>();
438  target.addLegalDialect<CombDialect>();
439  target.addLegalDialect<SeqDialect>();
440  target.addLegalDialect<SVDialect>();
441 
442  RewritePatternSet patterns(&context);
443  patterns.add<ConvertProgramOp>(&context);
444  patterns.add<ConvertComponentOp>(&context);
445  patterns.add<ConvertWiresOp>(&context);
446  patterns.add<ConvertControlOp>(&context);
447  patterns.add<ConvertCellOp>(&context);
448  patterns.add<ConvertAssignOp>(&context);
449 
450  return applyPartialConversion(program, target, std::move(patterns));
451 }
