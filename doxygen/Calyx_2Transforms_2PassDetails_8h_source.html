29 /// Updates the guard of each assignment within a group with `op`.
30 template <typename Op>
31 static void updateGroupAssignmentGuards(OpBuilder &builder, GroupOp &group,
32  Op &op) {
33  group.walk([&](AssignOp assign) {
34  if (assign.guard())
35  // If the assignment is guarded already, take the bitwise & of the current
36  // guard and the group's go signal.
37  assign->setOperand(
38  2, builder.create<comb::AndOp>(group.getLoc(), assign.guard(), op));
39  else
40  // Otherwise, just insert it as the guard.
41  assign->insertOperands(2, {op});
42  });
43 }
