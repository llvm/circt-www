70 namespace {
71 /// This is a child iterator class for FIRRTL values. The base iterator is the
72 /// `use_iterator` of a value. On top of the `use_iterator`, uses which are
73 /// 'dest`s of connect ops and uses whose owner has no results are filtered out.
74 class ChildIterator
75  : llvm::iterator_facade_base<ChildIterator, std::forward_iterator_tag,
76  Value> {
77  void skipToNextValidChild() {
78  auto isChild = [&]() {
79  if (auto connect = dyn_cast<FConnectLike>(childIt->getOwner()))
80  return childIt->get() == connect.src();
81  return childIt->getOwner()->getNumResults() > 0;
82  };
83  while (childIt != childEnd && !isChild())
84  ++childIt;
85  }
86 
87 public:
88  ChildIterator() = default;
89  explicit ChildIterator(Value v, bool end = false)
90  : childEnd(v.use_end()), childIt(end ? childEnd : v.use_begin()) {
91  skipToNextValidChild();
92  }
93 
94  /// The iterator is empty or at the end.
95  bool isAtEnd() { return childIt == nullptr || childIt == childEnd; }
96 
97  using llvm::iterator_facade_base<ChildIterator, std::forward_iterator_tag,
98  Value>::operator++;
99  ChildIterator &operator++() {
100  assert(!isAtEnd() && "incrementing the end iterator");
101  ++childIt;
102  skipToNextValidChild();
103  return *this;
104  }
105 
106  Value operator*() {
107  assert(!isAtEnd() && "dereferencing the end iterator");
108  if (auto connect = dyn_cast<FConnectLike>(childIt->getOwner()))
109  return connect.dest();
110  return childIt->getOwner()->getResult(0);
111  }
112 
113  bool operator==(const ChildIterator &rhs) const {
114  return childIt == rhs.childIt;
115  }
116  bool operator!=(const ChildIterator &rhs) const { return !(*this == rhs); }
117 
118 private:
119  Value::use_iterator childEnd;
120  Value::use_iterator childIt;
121 };

168 class InstanceNodeIterator : public NodeIterator {
169  /// Skip instance ports with not child.
170  void skipToNextValidPort() {
171  ChildIterator newChild;
172  while (portIt != portEnd) {
173  newChild = ChildIterator(instance.getResult(*portIt));
174  if (newChild.isAtEnd())
175  ++portIt;
176  else
177  break;
178  }
179  child = portIt == portEnd ? ChildIterator() : newChild;
180  }
181 
182 public:
183  explicit InstanceNodeIterator(InstanceOp instance, Node node,
184  bool end = false)
185  : NodeIterator(node, true), instance(instance) {
186  assert(instance == getValue().getDefiningOp<InstanceOp>() &&
187  "instance must be the defining op of the node value");
188  if (end)
189  return;
190 
191  // Query the combinational paths between IOs of the current instance.
192  auto module = getInstanceGraph()->getReferencedModule(instance);
193  auto &combPaths = getCombPathsMap()->FindAndConstruct(module).second;
194  auto &ports = combPaths[getValue().cast<OpResult>().getResultNumber()];
195 
196  portEnd = ports.end();
197  portIt = ports.begin();
198  skipToNextValidPort();
199  }
200 
201  InstanceNodeIterator &operator++() {
202  if (!child.isAtEnd())
203  ++child;
204  if (child.isAtEnd()) {
205  ++portIt;
206  skipToNextValidPort();
207  }
208  return *this;
209  }
210 
211  unsigned getPortNumber() const { return *portIt; }
212 
213 private:
214  InstanceOp instance;
215  SmallVectorImpl<size_t>::iterator portEnd;
216  SmallVectorImpl<size_t>::iterator portIt;
217 };

224 namespace {
225 class SubfieldNodeIterator : public NodeIterator {
226 public:
227  explicit SubfieldNodeIterator(SubfieldOp subfield, Node node,
228  bool end = false)
229  : NodeIterator(node, true) {
230  assert(subfield == node.value.getDefiningOp<SubfieldOp>() &&
231  "subfield must be the defining op of the node value");
232  if (end)
233  return;
234 
235  auto memory = subfield.input().getDefiningOp<MemOp>();
236  if (!memory) {
237  subfield->emitOpError("input must be a port of a MemOp, please run "
238  "-firrtl-lower-types first");
239  return;
240  }
241 
242  if (memory.readLatency() != 0)
243  return;
244 
245  auto portKind =
246  memory.getPortKind(subfield.input().cast<OpResult>().getResultNumber());
247  auto subfieldIndex = subfield.fieldIndex();
248  // Combinational path exists only when the current subfield is `addr`.
249  if (!(portKind == MemOp::PortKind::Read &&
250  subfieldIndex == (unsigned)ReadPortSubfield::addr) &&
251  !(portKind == MemOp::PortKind::ReadWrite &&
252  subfieldIndex == (unsigned)ReadWritePortSubfield::addr))
253  return;
254 
255  // Only `data` or `rdata` subfield is combinationally connected to `addr`
256  // subfield. Find the corresponding subfield op.
257  for (auto user : subfield.input().getUsers()) {
258  auto currentSubfield = dyn_cast<SubfieldOp>(user);
259  if (!currentSubfield) {
260  user->emitOpError("MemOp must be used by SubfieldOp, please run "
261  "-firrtl-lower-types first");
262  return;
263  }
264 
265  auto index = currentSubfield.fieldIndex();
266  if ((portKind == MemOp::PortKind::Read &&
267  index == (unsigned)ReadPortSubfield::data) ||
268  (portKind == MemOp::PortKind::ReadWrite &&
269  index == (unsigned)ReadWritePortSubfield::rdata)) {
270  child = ChildIterator(currentSubfield.result());
271  dataPort = currentSubfield.result();
272  return;
273  }
274  }
275  }
276 
277  SubfieldNodeIterator &operator++() {
278  if (!child.isAtEnd())
279  ++child;
280  if (child.isAtEnd())
281  child = ChildIterator();
282  return *this;
283  }
284 
285  Value getDataPort() { return dataPort; }
286 
287 private:
288  Value dataPort;
289 };

482 // Sample a cycle from SCC.
483 SmallVector<Node> sampleCycle(SCCIterator &scc) {
484  llvm::SmallDenseSet<Node, 4> sccNodes;
485  for (auto node : *scc)
486  sccNodes.insert(node);
487 
488  auto current = *(*scc).begin();
489  SmallVector<Node> path;
490  SmallDenseMap<Node, unsigned> visitOrder;
491  while (true) {
492  for (auto child :
493  llvm::make_range(GT::child_begin(current), GT::child_end(current))) {
494  // If the child is out of SCC, we don't explore.
495  if (!sccNodes.contains(child))
496  continue;
497 
498  auto it = visitOrder.find(child);
499  // If the child is visited before, path[visitedOrder[c]] ... path.end()
500  // forms a cycle.
501  if (it != visitOrder.end())
502  return SmallVector<Node>(path.begin() + it->second, path.end());
503 
504  visitOrder.insert({child, path.size()});
505  path.push_back(child);
506  current = child;
507  break;
508  }
509  }
510  llvm_unreachable("a cycle must be found in SCC");
511 }

520 // Dump a path to a diagnostic.
521 void dumpPath(SmallVector<Node> &path, SmallString<16> &instancePath,
522  FModuleOp module, bool isCycle, mlir::InFlightDiagnostic &diag) {
523  unsigned pathSize = isCycle ? path.size() + 1 : path.size();
524  for (unsigned i = 0, e = pathSize; i < e; ++i) {
525  Node current = path[i % path.size()];
526  Value currentValue = current.value;
527  bool isCycleEnd = i == path.size();
528  auto attachInfo = [&]() -> mlir::Diagnostic & {
529  return diag.attachNote(currentValue.getLoc()) << instancePath << ".";
530  };
531 
532  // If the currentValue is port, emit its name.
533  if (auto arg = currentValue.dyn_cast<BlockArgument>()) {
534  attachInfo() << module.getPortName(arg.getArgNumber());
535  continue;
536  }
537 
538  TypeSwitch<Operation *>(currentValue.getDefiningOp())
539  .Case<WireOp, RegOp, RegResetOp>(
540  // For operations which declare signals, we simply print signal
541  // names.
542  [&](auto op) { attachInfo() << op.name(); })
543  .Case<InstanceOp, SubfieldOp>([&](auto op) {
544  // If the op is InstanceOp or SubfieldOp, it is necessary to
545  // investigate the next value since output values do not expilicty
546  // appear in the cycle.
547  Node next = path[(i + 1) % path.size()];
548  for (auto iter = GT::child_begin(current),
549  end = GT::child_end(current);
550  iter != end; ++iter) {
551  if ((*iter).value != next.value)
552  continue;
553 
554  auto iterImpl = iter.getIteratorImpl();
555  if (std::holds_alternative<InstanceNodeIterator>(iterImpl)) {
556  // Instance. Print names of input and output ports.
557  auto instance = currentValue.getDefiningOp<InstanceOp>();
558  auto inputPort = currentValue.cast<OpResult>().getResultNumber();
559  auto outputPort =
560  std::get<InstanceNodeIterator>(iterImpl).getPortNumber();
561  if (isCycleEnd)
562  attachInfo() << instance.name() << "."
563  << instance.getPortName(inputPort).str();
564  else
565  dumpPathBetweenModulePorts(instancePath, instance, inputPort,
566  outputPort, current.context, diag);
567 
568  } else if (std::holds_alternative<SubfieldNodeIterator>(iterImpl)) {
569  // SubfieldNodeIterator represents the connection between addr
570  // port and data port.
571  auto subfieldAddr = currentValue.getDefiningOp<SubfieldOp>();
572  auto subfieldData =
573  std::get<SubfieldNodeIterator>(iterImpl).getDataPort();
574 
575  attachInfo() << getFieldName(getFieldRefFromValue(subfieldAddr));
576  if (!isCycleEnd)
577  diag.attachNote(subfieldData.getLoc())
578  << module.getName().str() << "."
579  << getFieldName(getFieldRefFromValue(subfieldData));
580  }
581  break;
582  }
583  })
584  .Default([&](auto op) {});
585  }

588 // This function dumps a (shortest) path from `inputPort` to `outputPort` by
589 // looking at a module referred by `instance`.
590 void dumpPathBetweenModulePorts(SmallString<16> &instancePath,
591  InstanceOp instance, unsigned inputPort,
592  unsigned outputPort, NodeContext *context,
593  mlir::InFlightDiagnostic &diag) {
594  auto module = dyn_cast<FModuleOp>(*instance.getReferencedModule());
595  if (!module)
596  return;
597  unsigned instancePathSize = instancePath.size();
598  instancePath += ".";
599  instancePath += instance.name();
600  auto start = Node(module.getArgument(inputPort), context);
601  auto end = Node(module.getArgument(outputPort), context);
602  llvm::DenseMap<Node, Node> previousNode;
603  // Find a shortest path from input port to output port by BFS.
604  std::queue<Node> que;
605  que.push(start);
606  while (!que.empty()) {
607  Node current = que.front();
608  if (current == end)
609  break;
610  for (auto child :
611  llvm::make_range(GT::child_begin(current), GT::child_end(current))) {
612  // If a child is marked before, skip it.
613  if (previousNode.count(child))
614  continue;
615  // Record a previous node.
616  previousNode.insert({child, current});
617  que.push(child);
618  }
619 
620  que.pop();
621  }
622 
623  SmallVector<Node> path;
624  Node current = end;
625  // Reconstruct a path backwardly.
626  while (current.value) {
627  path.push_back(current);
628  current = previousNode[current];
629  }
630  std::reverse(path.begin(), path.end());
631 
632  diag.attachNote(instance.getLoc())
633  << "Instance " << instancePath << ": "
634  << instance.getPortName(inputPort).str() << " ---> "
635  << instance.getPortName(outputPort).str();
636  dumpPath(path, instancePath, module, /*isCycle=*/false, diag);
637  instancePath.resize(instancePathSize);
638 }

649 /// This pass constructs a local graph for each module to detect combinational
650 /// cycles. To capture the cross-module combinational cycles, this pass inlines
651 /// the combinational paths between IOs of its subinstances into a subgraph and
652 /// encodes them in a `combPathsMap`.
653 class CheckCombCyclesPass : public CheckCombCyclesBase<CheckCombCyclesPass> {
654  void runOnOperation() override {
655  auto &instanceGraph = getAnalysis<InstanceGraph>();
656  bool detectedCycle = false;
657 
658  // Traverse modules in a post order to make sure the combinational paths
659  // between IOs of a module have been detected and recorded in `combPathsMap`
660  // before we handle its parent modules.
661  for (auto node : llvm::post_order<InstanceGraph *>(&instanceGraph)) {
662  if (auto module = dyn_cast<FModuleOp>(*node->getModule())) {
663  NodeContext context(&map, &instanceGraph,
664  module.getOps<FConnectLike>());
665  auto dummyNode = Node(nullptr, &context);
666 
667  // Traversing SCCs in the combinational graph to detect cycles. As
668  // FIRRTL module is an SSA region, all cycles must contain at least one
669  // connect op. Thus we introduce a dummy source node to iterate on the
670  // `dest`s of all connect ops in the module.
671  for (auto combSCC = SCCIterator::begin(dummyNode); !combSCC.isAtEnd();
672  ++combSCC) {
673  if (combSCC.hasCycle()) {
674  detectedCycle = true;
675  auto errorDiag = mlir::emitError(
676  module.getLoc(),
677  "detected combinational cycle in a FIRRTL module");
678  if (printSimpleCycle)
679  dumpSimpleCycle(combSCC, module, errorDiag);
680  else {
681  for (auto node : *combSCC) {
682  auto &noteDiag = errorDiag.attachNote(node.value.getLoc());
683  noteDiag << "this operation is part of the combinational cycle";
684  }
685  }
686  }
687  }
688 
689  SmallVector<bool, 8> directionVec;
690  for (auto &port : module.getPorts())
691  directionVec.push_back(port.isOutput());
692 
693  auto &combPaths = map[module];
694  NodeDenseSet nodeSet;
695  SmallVector<size_t, 2> outputVec;
696  unsigned index = 0;
697 
698  // Record all combinational paths.
699  for (auto &port : module.getPorts()) {
700  nodeSet.clear();
701  outputVec.clear();
702  auto arg = module.getArgument(index++);
703  if (port.isOutput()) {
704  combPaths.push_back(outputVec);
705  continue;
706  }
707  Node inputNode(arg, &context);
708  for (auto node : llvm::depth_first_ext<Node>(inputNode, nodeSet)) {
709  if (auto output = node.value.dyn_cast<BlockArgument>())
710  if (directionVec[output.getArgNumber()])
711  outputVec.push_back(output.getArgNumber());
712  }
713  combPaths.push_back(outputVec);
714  }
715  continue;
716  }
717  if (auto extModule = dyn_cast<FExtModuleOp>(*node->getModule())) {
718  // TODO: Handle FExtModuleOp with `ExtModulePathAnnotation`s.
719  auto &combPaths = map[extModule];
720  combPaths.resize(extModule.getNumPorts());
721  continue;
722  }
723  llvm_unreachable("invalid instance graph node");
724  }
725 
726  if (detectedCycle)
727  signalPassFailure();
728  markAllAnalysesPreserved();
729  }
730 
731 private:
732  /// A global map from FIRRTL modules to their combinational paths between IOs.
733  CombPathsMap map;
734 };
