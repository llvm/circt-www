26 /// A simple pass that emits errors for any occurrences of `uint`, `sint`, or
27 /// `analog` without a known width.
28 class CheckWidthsPass : public CheckWidthsBase<CheckWidthsPass> {
29  void runOnOperation() override;
30  void checkType(Type type, Type reportType, Location loc,
31  llvm::function_ref<void(InFlightDiagnostic &)> addContext);
32 
33  InFlightDiagnostic emitError(Location loc, const Twine &message) {
34  anyFailed = true;
35  signalPassFailure();
36  return mlir::emitError(loc, message);
37  }
38 
39  /// A set of already-checked types, to avoid unnecessary work.
40  llvm::DenseSet<Type> checkedTypes;
41 
42  /// Whether any checks have failed.
43  bool anyFailed;
44 };

47 void CheckWidthsPass::runOnOperation() {
48  FModuleOp module = getOperation();
49  anyFailed = false;
50 
51  // Check the port types. Unfortunately we don't have an exact location of the
52  // port name, so we just point at the module with the port name mentioned in
53  // the error.
54  for (auto &port : module.getPorts()) {
55  checkType(
56  port.type, port.type, module.getLoc(), [&](InFlightDiagnostic &error) {
57  error.attachNote() << "in port `" << port.getName() << "` of module `"
58  << module.getName() << "`";
59  });
60  if (anyFailed)
61  break;
62  }
63 
64  // Check the results of each operation.
65  module->walk([&](Operation *op) {
66  for (auto type : op->getResultTypes())
67  checkType(type, type, op->getLoc(), [&](InFlightDiagnostic &error) {
68  error.attachNote() << "in result of `" << op->getName() << "`";
69  });
70  if (anyFailed)
71  return WalkResult::interrupt();
72  return WalkResult::advance();
73  });
74 
75  // Cleanup.
76  checkedTypes.clear();
77  markAllAnalysesPreserved();
78 }

80 /// Check a type and its children for any occurrences of uninferred widths.
81 /// Specifically look for `uint`, `sint`, and `analog` types. The argument
82 /// `type` is the one being checked, `reportType` is the one included in any
83 /// errors, `loc` is the location to report errors to, and `addContext` gives
84 /// the caller the chance to add additional context information to the error.
85 void CheckWidthsPass::checkType(
86  Type type, Type reportType, Location loc,
87  llvm::function_ref<void(InFlightDiagnostic &)> addContext) {
88  // Add the type to the already-checked set. If an entry was already there,
89  // don't do the work twice.
90  if (!checkedTypes.insert(type).second)
91  return;
92 
93  TypeSwitch<Type>(type)
94  // Primarily complain about integer and analog types not having a proper
95  // width.
96  .template Case<IntType, AnalogType>([&](auto type) {
97  if (!type.hasWidth()) {
98  auto error = emitError(loc, "uninferred width: type ")
99  << reportType << " has no known width";
100  addContext(error);
101  }
102  })
103  // Look into vector types, but report the entire
104  // vector type upon failure. So `vector<uint>` will have the error mention
105  // `vector<uint>` instead of just `uint`.
106  .template Case<FVectorType>([&](auto type) {
107  checkType(type.getElementType(), reportType, loc, addContext);
108  })
109  // Look into bundle types. Report only the type of the offending field
110  // upon error, and add a note indicating which field was problematic.
111  .template Case<BundleType>([&](auto type) {
112  for (auto &bundleElement : type.getElements())
113  checkType(bundleElement.type, bundleElement.type, loc,
114  [&](InFlightDiagnostic &error) {
115  addContext(error);
116  error.attachNote()
117  << "in bundle field `"
118  << bundleElement.name.getValue() << "`";
119  });
120  });
121 }
