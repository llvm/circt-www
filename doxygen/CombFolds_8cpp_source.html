120 /// Flattens a single input in `op` if `hasOneUse` is true and it can be defined
121 /// as an Op. Returns true if successful, and false otherwise.
122 ///
123 /// Example: op(1, 2, op(3, 4), 5) -> op(1, 2, 3, 4, 5) // returns true
124 ///
125 static bool tryFlatteningOperands(Operation *op, PatternRewriter &rewriter) {
126  auto inputs = op->getOperands();
127 
128  for (size_t i = 0, size = inputs.size(); i != size; ++i) {
129  Operation *flattenOp = inputs[i].getDefiningOp();
130  if (!flattenOp || flattenOp->getName() != op->getName())
131  continue;
132 
133  // Check for loops
134  if (flattenOp == op)
135  continue;
136 
137  // Don't duplicate logic when it has multiple uses.
138  if (!inputs[i].hasOneUse()) {
139  // We can fold a multi-use binary operation into this one if this allows a
140  // constant to fold though. For example, fold
141  // (or a, b, c, (or d, cst1), cst2) --> (or a, b, c, d, cst1, cst2)
142  // since the constants will both fold and we end up with the equiv cost.
143  //
144  // We don't do this for add/mul because the hardware won't be shared
145  // between the two ops if duplicated.
146  if (flattenOp->getNumOperands() != 2 || !isa<AndOp, OrOp, XorOp>(op) ||
147  !flattenOp->getOperand(1).getDefiningOp<hw::ConstantOp>() ||
148  !inputs.back().getDefiningOp<hw::ConstantOp>())
149  continue;
150  }
151 
152  // Otherwise, flatten away.
153  auto flattenOpInputs = flattenOp->getOperands();
154 
155  SmallVector<Value, 4> newOperands;
156  newOperands.reserve(size + flattenOpInputs.size());
157 
158  auto flattenOpIndex = inputs.begin() + i;
159  newOperands.append(inputs.begin(), flattenOpIndex);
160  newOperands.append(flattenOpInputs.begin(), flattenOpInputs.end());
161  newOperands.append(flattenOpIndex + 1, inputs.end());
162 
163  Value result =
164  createGenericOp(op->getLoc(), op->getName(), newOperands, rewriter);
165  replaceOpAndCopyName(rewriter, op, result);
166  return true;
167  }
168  return false;
169 }

171 // Given a range of uses of an operation, find the lowest and highest bits
172 // inclusive that are ever referenced. The range of uses must not be empty.
173 static std::pair<size_t, size_t>
174 getLowestBitAndHighestBitRequired(Operation *op, bool narrowTrailingBits,
175  size_t originalOpWidth) {
176  auto users = op->getUsers();
177  assert(!users.empty() &&
178  "getLowestBitAndHighestBitRequired cannot operate on "
179  "a empty list of uses.");
180 
181  // when we don't want to narrowTrailingBits (namely in arithmetic
182  // operations), forcing lowestBitRequired = 0
183  size_t lowestBitRequired = narrowTrailingBits ? originalOpWidth - 1 : 0;
184  size_t highestBitRequired = 0;
185 
186  for (auto *user : users) {
187  if (auto extractOp = dyn_cast<ExtractOp>(user)) {
188  size_t lowBit = extractOp.lowBit();
189  size_t highBit = extractOp.getType().getWidth() + lowBit - 1;
190  highestBitRequired = std::max(highestBitRequired, highBit);
191  lowestBitRequired = std::min(lowestBitRequired, lowBit);
192  continue;
193  }
194 
195  highestBitRequired = originalOpWidth - 1;
196  lowestBitRequired = 0;
197  break;
198  }
199 
200  return {lowestBitRequired, highestBitRequired};
201 }

203 template <class OpTy>
204 static bool narrowOperationWidth(OpTy op, bool narrowTrailingBits,
205  PatternRewriter &rewriter) {
206  IntegerType opType =
207  op.getResult().getType().template dyn_cast<IntegerType>();
208  if (!opType)
209  return false;
210 
211  auto range = getLowestBitAndHighestBitRequired(op, narrowTrailingBits,
212  opType.getWidth());
213  if (range.second + 1 == opType.getWidth() && range.first == 0)
214  return false;
215 
216  SmallVector<Value> args;
217  auto newType = rewriter.getIntegerType(range.second - range.first + 1);
218  for (auto inop : op.getOperands()) {
219  // deal with muxes here
220  if (inop.getType() != op.getType())
221  args.push_back(inop);
222  else
223  args.push_back(rewriter.createOrFold<ExtractOp>(inop.getLoc(), newType,
224  inop, range.first));
225  }
226  Value newop = rewriter.createOrFold<OpTy>(op.getLoc(), newType, args);
227  if (range.first)
228  newop = rewriter.createOrFold<ConcatOp>(
229  op.getLoc(), newop,
230  rewriter.create<hw::ConstantOp>(op.getLoc(),
231  APInt::getZero(range.first)));
232  if (range.second + 1 < opType.getWidth())
233  newop = rewriter.createOrFold<ConcatOp>(
234  op.getLoc(),
235  rewriter.create<hw::ConstantOp>(
236  op.getLoc(), APInt::getZero(opType.getWidth() - range.second - 1)),
237  newop);
238  replaceOpAndCopyName(rewriter, op, newop);
239  return true;
240 }

247  // Replicate one time -> noop.
248  if (getType().getWidth() == input().getType().getIntOrFloatBitWidth())
249  return input();
250 
251  // Constant fold.
252  if (auto input = constants[0].dyn_cast_or_null<IntegerAttr>()) {
253  if (input.getValue().getBitWidth() == 1) {
254  if (input.getValue().isZero())
255  return getIntAttr(APInt::getZero(getType().getWidth()), getContext());
256  return getIntAttr(APInt::getAllOnes(getType().getWidth()), getContext());
257  }
258 
259  APInt result = APInt::getZeroWidth();
260  for (auto i = getMultiple(); i != 0; --i)
261  result = result.concat(input.getValue());
262  return getIntAttr(result, getContext());
263  }
264 
265  return {};
266 }

314  // ShlOp(x, cst) -> Concat(Extract(x), zeros)
315  APInt value;

317  return failure();

425 // Transforms extract(lo, cat(a, b, c, d, e)) into
426 // cat(extract(lo1, b), c, extract(lo2, d)).
427 // innerCat must be the argument of the provided ExtractOp.
428 static LogicalResult extractConcatToConcatExtract(ExtractOp op,
429  ConcatOp innerCat,
430  PatternRewriter &rewriter) {
431  auto reversedConcatArgs = llvm::reverse(innerCat.inputs());
432  size_t beginOfFirstRelevantElement = 0;
433  auto it = reversedConcatArgs.begin();
434  size_t lowBit = op.lowBit();
435 
436  // This loop finds the first concatArg that is covered by the ExtractOp
437  for (; it != reversedConcatArgs.end(); it++) {
438  assert(beginOfFirstRelevantElement <= lowBit &&
439  "incorrectly moved past an element that lowBit has coverage over");
440  auto operand = *it;
441 
442  size_t operandWidth = operand.getType().getIntOrFloatBitWidth();
443  if (lowBit < beginOfFirstRelevantElement + operandWidth) {
444  // A bit other than the first bit will be used in this element.
445  // ...... ........ ...
446  // ^---lowBit
447  // ^---beginOfFirstRelevantElement
448  //
449  // Edge-case close to the end of the range.
450  // ...... ........ ...
451  // ^---(position + operandWidth)
452  // ^---lowBit
453  // ^---beginOfFirstRelevantElement
454  //
455  // Edge-case close to the beginning of the rang
456  // ...... ........ ...
457  // ^---lowBit
458  // ^---beginOfFirstRelevantElement
459  //
460  break;
461  }
462 
463  // extraction discards this element.
464  // ...... ........ ...
465  // | ^---lowBit
466  // ^---beginOfFirstRelevantElement
467  beginOfFirstRelevantElement += operandWidth;
468  }
469  assert(it != reversedConcatArgs.end() &&
470  "incorrectly failed to find an element which contains coverage of "
471  "lowBit");
472 
473  SmallVector<Value> reverseConcatArgs;
474  size_t widthRemaining = op.getType().getWidth();
475  size_t extractLo = lowBit - beginOfFirstRelevantElement;
476 
477  // Transform individual arguments of innerCat(..., a, b, c,) into
478  // [ extract(a), b, extract(c) ], skipping an extract operation where
479  // possible.
480  for (; widthRemaining != 0 && it != reversedConcatArgs.end(); it++) {
481  auto concatArg = *it;
482  size_t operandWidth = concatArg.getType().getIntOrFloatBitWidth();
483  size_t widthToConsume = std::min(widthRemaining, operandWidth - extractLo);
484 
485  if (widthToConsume == operandWidth && extractLo == 0) {
486  reverseConcatArgs.push_back(concatArg);
487  } else {
488  auto resultType = IntegerType::get(rewriter.getContext(), widthToConsume);
489  reverseConcatArgs.push_back(
490  rewriter.create<ExtractOp>(op.getLoc(), resultType, *it, extractLo));
491  }
492 
493  widthRemaining -= widthToConsume;
494 
495  // Beyond the first element, all elements are extracted from position 0.
496  extractLo = 0;
497  }
498 
499  if (reverseConcatArgs.size() == 1) {
500  replaceOpAndCopyName(rewriter, op, reverseConcatArgs[0]);
501  } else {
502  replaceOpWithNewOpAndCopyName<ConcatOp>(
503  rewriter, op, SmallVector<Value>(llvm::reverse(reverseConcatArgs)));
504  }
505  return success();
506 }

508 // Transforms extract(lo, replicate(a, N)) into replicate(a, N-c).
509 static bool extractFromReplicate(ExtractOp op, ReplicateOp replicate,
510  PatternRewriter &rewriter) {
511  auto extractResultWidth = op.getType().getWidth();
512  auto replicateEltWidth =
513  replicate.getOperand().getType().getIntOrFloatBitWidth();
514 
515  // If the extract starts at the base of an element and is an even multiple,
516  // we can replace the extract with a smaller replicate.
517  if (op.lowBit() % replicateEltWidth == 0 &&
518  extractResultWidth % replicateEltWidth == 0) {
519  replaceOpWithNewOpAndCopyName<ReplicateOp>(rewriter, op, op.getType(),
520  replicate.getOperand());
521  return true;
522  }
523 
524  // If the extract is completely contained in one element, extract from the
525  // element.
526  if (op.lowBit() % replicateEltWidth + extractResultWidth <=
527  replicateEltWidth) {
528  replaceOpWithNewOpAndCopyName<ExtractOp>(rewriter, op, op.getType(),
529  replicate.getOperand(),
530  op.lowBit() % replicateEltWidth);
531  return true;
532  }
533 
534  // We don't currently handle the case of extracting from non-whole elements,
535  // e.g. `extract (replicate 2-bit-thing, N), 1`.
536  return false;
537 }

539 LogicalResult ExtractOp::canonicalize(ExtractOp op, PatternRewriter &rewriter) {
540  auto *inputOp = op.input().getDefiningOp();
541 
542  // If the extracted bits are all known, then return the result.
543  auto knownBits = computeKnownBits(op.input())
544  .extractBits(op.getType().getWidth(), op.lowBit());
545  if (knownBits.isConstant()) {
546  replaceOpWithNewOpAndCopyName<hw::ConstantOp>(rewriter, op,
547  knownBits.getConstant());
548  return success();
549  }
550 
551  // extract(olo, extract(ilo, x)) = extract(olo + ilo, x)
552  if (auto innerExtract = dyn_cast_or_null<ExtractOp>(inputOp)) {
553  replaceOpWithNewOpAndCopyName<ExtractOp>(
554  rewriter, op, op.getType(), innerExtract.input(),
555  innerExtract.lowBit() + op.lowBit());
556  return success();
557  }
558 
559  // extract(lo, cat(a, b, c, d, e)) = cat(extract(lo1, b), c, extract(lo2, d))
560  if (auto innerCat = dyn_cast_or_null<ConcatOp>(inputOp))
561  return extractConcatToConcatExtract(op, innerCat, rewriter);
562 
563  // extract(lo, replicate(a))
564  if (auto replicate = dyn_cast_or_null<ReplicateOp>(inputOp))
565  if (extractFromReplicate(op, replicate, rewriter))
566  return success();
567 
568  // `extract(and(a, cst))` -> `extract(a)` when the relevant bits of the
569  // and/or/xor are not modifying the extracted bits.
570  if (inputOp && inputOp->getNumOperands() == 2 &&
571  isa<AndOp, OrOp, XorOp>(inputOp)) {
572  if (auto cstRHS = inputOp->getOperand(1).getDefiningOp<hw::ConstantOp>()) {
573  auto extractedCst =
574  cstRHS.getValue().extractBits(op.getType().getWidth(), op.lowBit());
575  if (isa<OrOp, XorOp>(inputOp) && extractedCst.isZero()) {
576  replaceOpWithNewOpAndCopyName<ExtractOp>(
577  rewriter, op, op.getType(), inputOp->getOperand(0), op.lowBit());
578  return success();
579  }
580 
581  // `extract(and(a, cst))` -> `concat(extract(a), 0)` if we only need one
582  // extract to represent the result. Turning it into a pile of extracts is
583  // always fine by our cost model, but we don't want to explode things into
584  // a ton of bits because it will bloat the IR and generated Verilog.
585  if (isa<AndOp>(inputOp)) {
586  // For our cost model, we only do this if the bit pattern is a
587  // contiguous series of ones.
588  unsigned lz = extractedCst.countLeadingZeros();
589  unsigned tz = extractedCst.countTrailingZeros();
590  unsigned pop = extractedCst.countPopulation();
591  if (extractedCst.getBitWidth() - lz - tz == pop) {
592  auto resultTy = rewriter.getIntegerType(pop);
593  SmallVector<Value> resultElts;
594  if (lz)
595  resultElts.push_back(rewriter.create<hw::ConstantOp>(
596  op.getLoc(), APInt::getZero(lz)));
597  resultElts.push_back(rewriter.createOrFold<ExtractOp>(
598  op.getLoc(), resultTy, inputOp->getOperand(0), op.lowBit() + tz));
599  if (tz)
600  resultElts.push_back(rewriter.create<hw::ConstantOp>(
601  op.getLoc(), APInt::getZero(tz)));
602  replaceOpWithNewOpAndCopyName<ConcatOp>(rewriter, op, resultElts);

647 /// When we find a logical operation (and, or, xor) with a constant e.g.
648 /// `X & 42`, we want to push the constant into the computation of X if it leads
649 /// to simplification.
650 ///
651 /// This function handles the case where the logical operation has a concat
652 /// operand. We check to see if we can simplify the concat, e.g. when it has
653 /// constant operands.
654 ///
655 /// This returns true when a simplification happens.
656 static bool canonicalizeLogicalCstWithConcat(Operation *logicalOp,
657  size_t concatIdx, const APInt &cst,
658  PatternRewriter &rewriter) {
659  auto concatOp = logicalOp->getOperand(concatIdx).getDefiningOp<ConcatOp>();
660  assert((isa<AndOp, OrOp, XorOp>(logicalOp) && concatOp));
661 
662  // Check to see if any operands can be simplified by pushing the logical op
663  // into all parts of the concat.
664  bool canSimplify =
665  llvm::any_of(concatOp->getOperands(), [&](Value operand) -> bool {
666  auto *operandOp = operand.getDefiningOp();
667  if (!operandOp)
668  return false;
669 
670  // If the concat has a constant operand then we can transform this.
671  if (isa<hw::ConstantOp>(operandOp))
672  return true;
673  // If the concat has the same logical operation and that operation has
674  // a constant operation than we can fold it into that suboperation.
675  return operandOp->getName() == logicalOp->getName() &&
676  operandOp->hasOneUse() && operandOp->getNumOperands() != 0 &&
677  operandOp->getOperands().back().getDefiningOp<hw::ConstantOp>();
678  });
679 
680  if (!canSimplify)
681  return false;
682 
683  // Create a new instance of the logical operation. We have to do this the
684  // hard way since we're generic across a family of different ops.
685  auto createLogicalOp = [&](ArrayRef<Value> operands) -> Value {
686  return createGenericOp(logicalOp->getLoc(), logicalOp->getName(), operands,
687  rewriter);
688  };
689 
690  // Ok, let's do the transformation. We do this by slicing up the constant
691  // for each unit of the concat and duplicate the operation into the
692  // sub-operand.
693  SmallVector<Value> newConcatOperands;
694  newConcatOperands.reserve(concatOp->getNumOperands());
695 
696  // Work from MSB to LSB.
697  size_t nextOperandBit = concatOp.getType().getIntOrFloatBitWidth();
698  for (Value operand : concatOp->getOperands()) {
699  size_t operandWidth = operand.getType().getIntOrFloatBitWidth();
700  nextOperandBit -= operandWidth;
701  // Take a slice of the constant.
702  auto eltCst = rewriter.create<hw::ConstantOp>(
703  logicalOp->getLoc(), cst.lshr(nextOperandBit).trunc(operandWidth));
704 
705  newConcatOperands.push_back(createLogicalOp({operand, eltCst}));
706  }
707 
708  // Create the concat, and the rest of the logical op if we need it.
709  Value newResult =
710  rewriter.create<ConcatOp>(concatOp.getLoc(), newConcatOperands);
711 
712  // If we had a variadic logical op on the top level, then recreate it with the
713  // new concat and without the constant operand.
714  if (logicalOp->getNumOperands() > 2) {
715  auto origOperands = logicalOp->getOperands();
716  SmallVector<Value> operands;
717  // Take any stuff before the concat.
718  operands.append(origOperands.begin(), origOperands.begin() + concatIdx);
719  // Take any stuff after the concat but before the constant.
720  operands.append(origOperands.begin() + concatIdx + 1,
721  origOperands.begin() + (origOperands.size() - 1));
722  // Include the new concat.
723  operands.push_back(newResult);
724  newResult = createLogicalOp(operands);
725  }
726 
727  replaceOpAndCopyName(rewriter, logicalOp, newResult);
728  return true;
729 }

735  for (auto operand : constants) {

787 LogicalResult AndOp::canonicalize(AndOp op, PatternRewriter &rewriter) {
788  auto inputs = op.inputs();
789  auto size = inputs.size();
790  assert(size > 1 && "expected 2 or more operands, `fold` should handle this");
791 
792  // and(..., x, ..., x) -> and(..., x, ...) -- idempotent
793  // Trivial and(x), and(x, x) cases are handled by [AndOp::fold] above.
794  if (size > 2 && canonicalizeIdempotentInputs(op, rewriter))
795  return success();
796 
797  // Patterns for and with a constant on RHS.
798  APInt value;
799  if (matchPattern(inputs.back(), m_RConstant(value))) {
800  // and(..., '1) -> and(...) -- identity
801  if (value.isAllOnes()) {
802  replaceOpWithNewOpAndCopyName<AndOp>(rewriter, op, op.getType(),
803  inputs.drop_back());
804  return success();
805  }
806 
807  // TODO: Combine multiple constants together even if they aren't at the
808  // end. and(..., c1, c2) -> and(..., c3) where c3 = c1 & c2 -- constant
809  // folding
810  APInt value2;
811  if (matchPattern(inputs[size - 2], m_RConstant(value2))) {
812  auto cst = rewriter.create<hw::ConstantOp>(op.getLoc(), value & value2);
813  SmallVector<Value, 4> newOperands(inputs.drop_back(/*n=*/2));
814  newOperands.push_back(cst);
815  replaceOpWithNewOpAndCopyName<AndOp>(rewriter, op, op.getType(),
816  newOperands);
817  return success();
818  }
819 
820  // Handle 'and' with a single bit constant on the RHS.
821  if (size == 2 && value.isPowerOf2()) {
822  // If the LHS is a replicate from a single bit, we can 'concat' it
823  // into place. e.g.:
824  // `replicate(x) & 4` -> `concat(zeros, x, zeros)`
825  // TODO: Generalize this for non-single-bit operands.
826  if (auto replicate = inputs[0].getDefiningOp<ReplicateOp>()) {
827  auto replicateOperand = replicate.getOperand();
828  if (replicateOperand.getType().isInteger(1)) {
829  unsigned resultWidth = op.getType().getIntOrFloatBitWidth();
830  auto trailingZeros = value.countTrailingZeros();
831 
832  // Don't add zero bit constants unnecessarily.
833  SmallVector<Value, 3> concatOperands;
834  if (trailingZeros != resultWidth - 1) {
835  auto highZeros = rewriter.create<hw::ConstantOp>(
836  op.getLoc(), APInt::getZero(resultWidth - trailingZeros - 1));
837  concatOperands.push_back(highZeros);
838  }
839  concatOperands.push_back(replicateOperand);
840  if (trailingZeros != 0) {
841  auto lowZeros = rewriter.create<hw::ConstantOp>(
842  op.getLoc(), APInt::getZero(trailingZeros));
843  concatOperands.push_back(lowZeros);
844  }
845  replaceOpWithNewOpAndCopyName<ConcatOp>(rewriter, op, op.getType(),
846  concatOperands);
847  return success();
848  }
849  }
850  }
851 
852  // If this is an and from an extract op, try shrinking the extract.
853  if (auto extractOp = inputs[0].getDefiningOp<ExtractOp>()) {
854  if (size == 2 &&
855  // We can shrink it if the mask has leading or trailing zeros.
856  (value.countLeadingZeros() || value.countTrailingZeros())) {
857  unsigned lz = value.countLeadingZeros();
858  unsigned tz = value.countTrailingZeros();
859 
860  // Start by extracting the smaller number of bits.
861  auto smallTy = rewriter.getIntegerType(value.getBitWidth() - lz - tz);
862  Value smallElt = rewriter.createOrFold<ExtractOp>(
863  extractOp.getLoc(), smallTy, extractOp->getOperand(0),
864  extractOp.lowBit() + tz);
865  // Apply the 'and' mask if needed.
866  APInt smallMask = value.extractBits(smallTy.getWidth(), tz);
867  if (!smallMask.isAllOnes()) {
868  auto loc = inputs.back().getLoc();
869  smallElt = rewriter.createOrFold<AndOp>(
870  loc, smallElt, rewriter.create<hw::ConstantOp>(loc, smallMask));
871  }
872 
873  // The final replacement will be a concat of the leading/trailing zeros
874  // along with the smaller extracted value.
875  SmallVector<Value> resultElts;
876  if (lz)
877  resultElts.push_back(
878  rewriter.create<hw::ConstantOp>(op.getLoc(), APInt::getZero(lz)));
879  resultElts.push_back(smallElt);
880  if (tz)
881  resultElts.push_back(
882  rewriter.create<hw::ConstantOp>(op.getLoc(), APInt::getZero(tz)));
883  replaceOpWithNewOpAndCopyName<ConcatOp>(rewriter, op, resultElts);
884  return success();
885  }
886  }
887 
888  // and(concat(x, cst1), a, b, c, cst2)
889  // ==> and(a, b, c, concat(and(x,cst2'), and(cst1,cst2'')).
890  // We do this for even more multi-use concats since they are "just wiring".
891  for (size_t i = 0; i < size - 1; ++i) {
892  if (auto concat = inputs[i].getDefiningOp<ConcatOp>())
893  if (canonicalizeLogicalCstWithConcat(op, i, value, rewriter))
894  return success();
895  }
896  }
897 
898  // and(x, and(...)) -> and(x, ...) -- flatten
899  if (tryFlatteningOperands(op, rewriter))
900  return success();
901 
902  // extracts only of and(...) -> and(extract()...)
903  if (narrowOperationWidth(op, true, rewriter))
904  return success();
905 
906  /// TODO: and(..., x, not(x)) -> and(..., 0) -- complement
907  return failure();
908 }

946 /// Simplify concat ops in an or op when a constant operand is present in either
947 /// concat.
948 ///
949 /// This will invert an or(concat, concat) into concat(or, or, ...), which can
950 /// often be further simplified due to the smaller or ops being easier to fold.
951 ///
952 /// For example:
953 ///
954 /// or(..., concat(x, 0), concat(0, y))
955 /// ==> or(..., concat(x, 0, y)), when x and y don't overlap.
956 ///
957 /// or(..., concat(x: i2, cst1: i4), concat(cst2: i5, y: i1))
958 /// ==> or(..., concat(or(x: i2, extract(cst2, 4..3)),
959 /// or(extract(cst1, 3..1), extract(cst2, 2..0)),
960 /// or(extract(cst1, 0..0), y: i1))
961 static bool canonicalizeOrOfConcatsWithCstOperands(OrOp op, size_t concatIdx1,
962  size_t concatIdx2,
963  PatternRewriter &rewriter) {
964  assert(concatIdx1 < concatIdx2 && "concatIdx1 must be < concatIdx2");
965 
966  auto inputs = op.inputs();
967  auto concat1 = inputs[concatIdx1].getDefiningOp<ConcatOp>();
968  auto concat2 = inputs[concatIdx2].getDefiningOp<ConcatOp>();
969 
970  assert(concat1 && concat2 && "expected indexes to point to ConcatOps");
971 
972  // We can simplify as long as a constant is present in either concat.
973  bool hasConstantOp1 =
974  llvm::any_of(concat1->getOperands(), [&](Value operand) -> bool {
975  return operand.getDefiningOp<hw::ConstantOp>();
976  });
977  if (!hasConstantOp1) {
978  bool hasConstantOp2 =
979  llvm::any_of(concat2->getOperands(), [&](Value operand) -> bool {
980  return operand.getDefiningOp<hw::ConstantOp>();
981  });
982  if (!hasConstantOp2)
983  return false;
984  }
985 
986  SmallVector<Value> newConcatOperands;
987 
988  // Simultaneously iterate over the operands of both concat ops, from MSB to
989  // LSB, pushing out or's of overlapping ranges of the operands. When operands
990  // span different bit ranges, we extract only the maximum overlap.
991  auto operands1 = concat1->getOperands();
992  auto operands2 = concat2->getOperands();
993  // Number of bits already consumed from operands 1 and 2, respectively.
994  unsigned consumedWidth1 = 0;
995  unsigned consumedWidth2 = 0;
996  for (auto it1 = operands1.begin(), end1 = operands1.end(),
997  it2 = operands2.begin(), end2 = operands2.end();
998  it1 != end1 && it2 != end2;) {
999  auto operand1 = *it1;
1000  auto operand2 = *it2;
1001 
1002  unsigned remainingWidth1 =
1003  hw::getBitWidth(operand1.getType()) - consumedWidth1;
1004  unsigned remainingWidth2 =
1005  hw::getBitWidth(operand2.getType()) - consumedWidth2;
1006  unsigned widthToConsume = std::min(remainingWidth1, remainingWidth2);
1007  auto narrowedType = rewriter.getIntegerType(widthToConsume);
1008 
1009  auto extract1 = rewriter.createOrFold<ExtractOp>(
1010  op.getLoc(), narrowedType, operand1, remainingWidth1 - widthToConsume);
1011  auto extract2 = rewriter.createOrFold<ExtractOp>(
1012  op.getLoc(), narrowedType, operand2, remainingWidth2 - widthToConsume);
1013 
1014  newConcatOperands.push_back(
1015  rewriter.createOrFold<OrOp>(op.getLoc(), extract1, extract2));
1016 
1017  consumedWidth1 += widthToConsume;
1018  consumedWidth2 += widthToConsume;
1019 
1020  if (widthToConsume == remainingWidth1) {
1021  ++it1;
1022  consumedWidth1 = 0;
1023  }
1024  if (widthToConsume == remainingWidth2) {
1025  ++it2;
1026  consumedWidth2 = 0;
1027  }
1028  }
1029 
1030  ConcatOp newOp = rewriter.create<ConcatOp>(op.getLoc(), newConcatOperands);
1031 
1032  // Copy the old operands except for concatIdx1 and concatIdx2, and append the
1033  // new ConcatOp to the end.
1034  SmallVector<Value> newOrOperands;
1035  newOrOperands.append(inputs.begin(), inputs.begin() + concatIdx1);
1036  newOrOperands.append(inputs.begin() + concatIdx1 + 1,
1037  inputs.begin() + concatIdx2);
1038  newOrOperands.append(inputs.begin() + concatIdx2 + 1,
1039  inputs.begin() + inputs.size());
1040  newOrOperands.push_back(newOp);
1041 
1042  replaceOpWithNewOpAndCopyName<OrOp>(rewriter, op, op.getType(),
1043  newOrOperands);
1044  return true;
1045 }

1523 LogicalResult ConcatOp::canonicalize(ConcatOp op, PatternRewriter &rewriter) {
1524  auto inputs = op.inputs();
1525  auto size = inputs.size();
1526  assert(size > 1 && "expected 2 or more operands");
1527 
1528  // This function is used when we flatten neighboring operands of a
1529  // (variadic) concat into a new vesion of the concat. first/last indices
1530  // are inclusive.
1531  auto flattenConcat = [&](size_t firstOpIndex, size_t lastOpIndex,
1532  ValueRange replacements) -> LogicalResult {
1533  SmallVector<Value, 4> newOperands;
1534  newOperands.append(inputs.begin(), inputs.begin() + firstOpIndex);
1535  newOperands.append(replacements.begin(), replacements.end());
1536  newOperands.append(inputs.begin() + lastOpIndex + 1, inputs.end());
1537  if (newOperands.size() == 1)
1538  replaceOpAndCopyName(rewriter, op, newOperands[0]);
1539  else
1540  replaceOpWithNewOpAndCopyName<ConcatOp>(rewriter, op, op.getType(),
1541  newOperands);
1542  return success();
1543  };
1544 
1545  Value commonOperand = inputs[0];
1546  for (size_t i = 0; i != size; ++i) {
1547  // Check to see if all operands are the same.
1548  if (inputs[i] != commonOperand)
1549  commonOperand = Value();
1550 
1551  // If an operand to the concat is itself a concat, then we can fold them
1552  // together.
1553  if (auto subConcat = inputs[i].getDefiningOp<ConcatOp>())
1554  return flattenConcat(i, i, subConcat->getOperands());
1555

1557  if (i != 0) {
1558  // Merge neighboring constants.
1559  if (auto cst = inputs[i].getDefiningOp<hw::ConstantOp>()) {
1560  if (auto prevCst = inputs[i - 1].getDefiningOp<hw::ConstantOp>()) {
1561  unsigned prevWidth = prevCst.getValue().getBitWidth();
1562  unsigned thisWidth = cst.getValue().getBitWidth();
1563  auto resultCst = cst.getValue().zext(prevWidth + thisWidth);
1564  resultCst |= prevCst.getValue().zext(prevWidth + thisWidth)
1565  << thisWidth;

1567  rewriter.create<hw::ConstantOp>(op.getLoc(), resultCst);
1568  return flattenConcat(i - 1, i, replacement);
1569  }
1570  }

1573  if (inputs[i] == inputs[i - 1]) {
1574  Value replacement =
1575  rewriter.createOrFold<ReplicateOp>(op.getLoc(), inputs[i], 2);
1576  return flattenConcat(i - 1, i, replacement);
1577  }
1578

1580  // one.

1582  // ... x, repl(x, n), ... ==> ..., repl(x, n+1), ...
1583  if (repl.getOperand() == inputs[i - 1]) {
1584  Value replacement = rewriter.createOrFold<ReplicateOp>(
1585  op.getLoc(), repl.getOperand(), repl.getMultiple() + 1);

1593  repl.getMultiple() + prevRepl.getMultiple());

1595  }
1596  }
1597  }

1599  // ... repl(x, n), x, ... ==> ..., repl(x, n+1), ...
1600  if (auto repl = inputs[i - 1].getDefiningOp<ReplicateOp>()) {
1601  if (repl.getOperand() == inputs[i]) {
1602  Value replacement = rewriter.createOrFold<ReplicateOp>(
1603  op.getLoc(), inputs[i], repl.getMultiple() + 1);
1604  return flattenConcat(i - 1, i, replacement);
1605  }
1606  }

1609  // {A[3], A[2]} -> A[3:2]
1610  if (auto extract = inputs[i].getDefiningOp<ExtractOp>()) {
1611  if (auto prevExtract = inputs[i - 1].getDefiningOp<ExtractOp>()) {
1612  if (extract.input() == prevExtract.input()) {

1614  if (prevExtract.lowBit() == extract.lowBit() + thisWidth) {
1615  auto prevWidth = prevExtract.getType().getIntOrFloatBitWidth();
1616  auto resType = rewriter.getIntegerType(thisWidth + prevWidth);
1617  Value replacement = rewriter.create<ExtractOp>(
1618  op.getLoc(), resType, extract.input(), extract.lowBit());

1621  }
1622  }
1623  }

1709 /// This returns false if we cannot form the mux tree (or do not want to) and
1710 /// returns true if the mux was replaced.
1711 static bool foldMuxChain(MuxOp rootMux, bool isFalseSide,
1712  PatternRewriter &rewriter) {
1713  // Get the index value being compared. Later we check to see if it is
1714  // compared to a constant with the right predicate.
1715  auto rootCmp = rootMux.cond().getDefiningOp<ICmpOp>();
1716  if (!rootCmp)
1717  return false;
1718  Value indexValue = rootCmp.lhs();
1719 
1720  // Return the value to use if the equality match succeeds.
1721  auto getCaseValue = [&](MuxOp mux) -> Value {
1722  return mux.getOperand(1 + unsigned(!isFalseSide));
1723  };
1724 
1725  // Return the value to use if the equality match fails. This is the next
1726  // mux in the sequence or the "otherwise" value.
1727  auto getTreeValue = [&](MuxOp mux) -> Value {
1728  return mux.getOperand(1 + unsigned(isFalseSide));
1729  };
1730 
1731  // Start scanning the mux tree to see what we've got. Keep track of the
1732  // constant comparison value and the SSA value to use when equal to it.
1733  SmallVector<Location> locationsFound;
1734  SmallVector<std::pair<hw::ConstantOp, Value>, 4> valuesFound;
1735 
1736  /// Extract constants and values into `valuesFound` and return true if this is
1737  /// part of the mux tree, otherwise return false.
1738  auto collectConstantValues = [&](MuxOp mux) -> bool {
1739  return getMuxChainCondConstant(
1740  mux.cond(), indexValue, isFalseSide, [&](hw::ConstantOp cst) {
1741  valuesFound.push_back({cst, getCaseValue(mux)});
1742  locationsFound.push_back(mux.cond().getLoc());
1743  locationsFound.push_back(mux->getLoc());
1744  });
1745  };
1746 
1747  // Make sure the root is a correct comparison with a constant.
1748  if (!collectConstantValues(rootMux))
1749  return false;
1750 
1751  // Make sure that we're not looking at the intermediate node in a mux tree.
1752  if (rootMux->hasOneUse()) {
1753  if (auto userMux = dyn_cast<MuxOp>(*rootMux->user_begin())) {
1754  if (getTreeValue(userMux) == rootMux.getResult() &&
1755  getMuxChainCondConstant(userMux.cond(), indexValue, isFalseSide,
1756  [&](hw::ConstantOp cst) {}))
1757  return false;
1758  }
1759  }
1760 
1761  // Scan up the tree linearly.
1762  auto nextTreeValue = getTreeValue(rootMux);
1763  while (1) {
1764  auto nextMux = nextTreeValue.getDefiningOp<MuxOp>();
1765  if (!nextMux || !nextMux->hasOneUse())
1766  break;
1767  if (!collectConstantValues(nextMux))
1768  break;
1769  nextTreeValue = getTreeValue(nextMux);
1770  }
1771 
1772  // We need to have more than three values to create an array. This is an
1773  // arbitrary threshold which is saying that one or two muxes together is ok,
1774  // but three should be folded.
1775  if (valuesFound.size() < 3)
1776  return false;
1777 
1778  // If the array is greater that 9 bits, it will take over 512 elements and
1779  // it will be too large for a single expression.
1780  auto indexWidth = indexValue.getType().cast<IntegerType>().getWidth();
1781  if (indexWidth >= 9)
1782  return false;
1783 
1784  // Next we need to see if the values are dense-ish. We don't want to have
1785  // a tremendous number of replicated entries in the array. Some sparsity is
1786  // ok though, so we require the table to be at least 5/8 utilized.
1787  uint64_t tableSize = 1ULL << indexWidth;
1788  if (valuesFound.size() < (tableSize * 5) / 8)
1789  return false; // Not dense enough.
1790 
1791  // Ok, we're going to do the transformation, start by building the table
1792  // filled with the "otherwise" value.
1793  SmallVector<Value, 8> table(tableSize, nextTreeValue);
1794 
1795  // Fill in entries in the table from the leaf to the root of the expression.
1796  // This ensures that any duplicate matches end up with the ultimate value,
1797  // which is the one closer to the root.
1798  for (auto &elt : llvm::reverse(valuesFound)) {
1799  uint64_t idx = elt.first.getValue().getZExtValue();
1800  assert(idx < table.size() && "constant should be same bitwidth as index");
1801  table[idx] = elt.second;
1802  }
1803 
1804  // The hw.array_create operation has the operand list in unintuitive order
1805  // with a[0] stored as the last element, not the first.
1806  std::reverse(table.begin(), table.end());
1807 
1808  // Build the array_create and the array_get.
1809  auto fusedLoc = rewriter.getFusedLoc(locationsFound);
1810  auto array = rewriter.create<hw::ArrayCreateOp>(fusedLoc, table);
1811  replaceOpWithNewOpAndCopyName<hw::ArrayGetOp>(rewriter, rootMux, array,
1812  indexValue);
1813  return true;
1814 }

1816 /// Given a fully associative variadic operation like (a+b+c+d), break the
1817 /// expression into two parts, one without the specified operand (e.g.
1818 /// `tmp = a+b+d`) and one that combines that into the full expression (e.g.
1819 /// `tmp+c`), and return the inner expression.
1820 ///
1821 /// NOTE: This mutates the operation in place if it only has a single user,
1822 /// which assumes that user will be removed.
1823 ///
1824 static Value extractOperandFromFullyAssociative(Operation *fullyAssoc,
1825  size_t operandNo,
1826  PatternRewriter &rewriter) {
1827  assert(fullyAssoc->getNumOperands() >= 2 && "cannot split up unary ops");
1828  assert(operandNo < fullyAssoc->getNumOperands() && "Invalid operand #");
1829 
1830  // If this expression already has two operands (the common case) no splitting
1831  // is necessary.
1832  if (fullyAssoc->getNumOperands() == 2)
1833  return fullyAssoc->getOperand(operandNo ^ 1);
1834 
1835  // If the operation has a single use, mutate it in place.
1836  if (fullyAssoc->hasOneUse()) {
1837  fullyAssoc->eraseOperand(operandNo);
1838  return fullyAssoc->getResult(0);
1839  }
1840 
1841  // Form the new operation with the operands that remain.
1842  SmallVector<Value> operands;
1843  operands.append(fullyAssoc->getOperands().begin(),
1844  fullyAssoc->getOperands().begin() + operandNo);
1845  operands.append(fullyAssoc->getOperands().begin() + operandNo + 1,
1846  fullyAssoc->getOperands().end());
1847  Value opWithoutExcluded = createGenericOp(
1848  fullyAssoc->getLoc(), fullyAssoc->getName(), operands, rewriter);
1849  Value excluded = fullyAssoc->getOperand(operandNo);
1850 
1851  Value fullResult =
1852  createGenericOp(fullyAssoc->getLoc(), fullyAssoc->getName(),
1853  ArrayRef<Value>{opWithoutExcluded, excluded}, rewriter);
1854  replaceOpAndCopyName(rewriter, fullyAssoc, fullResult);
1855  return opWithoutExcluded;

1858 /// Fold things like `mux(cond, x|y|z|a, a)` -> `(x|y|z)&replicate(cond)|a` and
1859 /// `mux(cond, a, x|y|z|a) -> `(x|y|z)&replicate(~cond) | a` (when isTrueOperand
1860 /// is true. Return true on successful transformation, false if not.
1861 ///
1862 /// These are various forms of "predicated ops" that can be handled with a
1863 /// replicate/and combination.
1864 static bool foldCommonMuxValue(MuxOp op, bool isTrueOperand,
1865  PatternRewriter &rewriter) {
1866  // Check to see the operand in question is an operation. If it is a port,
1867  // we can't simplify it.
1868  Operation *subExpr =
1869  (isTrueOperand ? op.falseValue() : op.trueValue()).getDefiningOp();
1870  if (!subExpr || subExpr->getNumOperands() < 2)
1871  return false;
1872 
1873  // If this isn't an operation we can handle, don't spend energy on it.
1874  if (!isa<AndOp, XorOp, OrOp, MuxOp>(subExpr))
1875  return false;
1876 
1877  // Check to see if the common value occurs in the operand list for the
1878  // subexpression op. If so, then we can simplify it.
1879  Value commonValue = isTrueOperand ? op.trueValue() : op.falseValue();
1880  size_t opNo = 0, e = subExpr->getNumOperands();
1881  while (opNo != e && subExpr->getOperand(opNo) != commonValue)
1882  ++opNo;
1883  if (opNo == e)
1884  return false;
1885 
1886  // If we got a hit, then go ahead and simplify it!
1887  Value cond = op.cond();
1888 
1889  // `mux(cond, a, mux(cond2, a, b))` -> `mux(cond|cond2, a, b)`
1890  // `mux(cond, a, mux(cond2, b, a))` -> `mux(cond|~cond2, a, b)`
1891  // `mux(cond, mux(cond2, a, b), a)` -> `mux(~cond|cond2, a, b)`
1892  // `mux(cond, mux(cond2, b, a), a)` -> `mux(~cond|~cond2, a, b)`
1893  if (auto subMux = dyn_cast<MuxOp>(subExpr)) {
1894  Value otherValue;
1895  Value subCond = subMux.cond();
1896 
1897  // Invert th subCond if needed and dig out the 'b' value.
1898  if (subMux.trueValue() == commonValue)
1899  otherValue = subMux.falseValue();
1900  else if (subMux.falseValue() == commonValue) {
1901  otherValue = subMux.trueValue();
1902  subCond = createOrFoldNot(op.getLoc(), subCond, rewriter);
1903  } else {
1904  // We can't fold `mux(cond, a, mux(a, x, y))`.
1905  return false;
1906  }
1907 
1908  // Invert the outer cond if needed, and combine the mux conditions.
1909  if (!isTrueOperand)
1910  cond = createOrFoldNot(op.getLoc(), cond, rewriter);
1911  cond = rewriter.createOrFold<OrOp>(op.getLoc(), cond, subCond);
1912  replaceOpWithNewOpAndCopyName<MuxOp>(rewriter, op, cond, commonValue,
1913  otherValue);
1914  return true;
1915  }
1916 
1917  // Invert the condition if needed. Or/Xor invert when dealing with
1918  // TrueOperand, And inverts for False operand.
1919  bool isaAndOp = isa<AndOp>(subExpr);
1920  if (isTrueOperand ^ isaAndOp)
1921  cond = createOrFoldNot(op.getLoc(), cond, rewriter);
1922 
1923  auto extendedCond =
1924  rewriter.createOrFold<ReplicateOp>(op.getLoc(), op.getType(), cond);
1925 
1926  // Cache this information before subExpr is erased by extraction below.
1927  bool isaXorOp = isa<XorOp>(subExpr);
1928  bool isaOrOp = isa<OrOp>(subExpr);
1929 
1930  // Handle the fully associative ops, start by pulling out the subexpression
1931  // from a many operand version of the op.
1932  auto restOfAssoc =
1933  extractOperandFromFullyAssociative(subExpr, opNo, rewriter);
1934 
1935  // `mux(cond, x|y|z|a, a)` -> `(x|y|z)&replicate(cond) | a`
1936  // `mux(cond, x^y^z^a, a)` -> `(x^y^z)&replicate(cond) ^ a`
1937  if (isaOrOp || isaXorOp) {
1938  auto masked =
1939  rewriter.createOrFold<AndOp>(op.getLoc(), extendedCond, restOfAssoc);
1940  if (isaXorOp)
1941  replaceOpWithNewOpAndCopyName<XorOp>(rewriter, op, masked, commonValue);
1942  else
1943  replaceOpWithNewOpAndCopyName<OrOp>(rewriter, op, masked, commonValue);
1944  return true;
1945  }
1946 
1947  // `mux(cond, a, x&y&z&a)` -> `((x&y&z)|replicate(cond)) & a`
1948  assert(isaAndOp && "unexpected operation here");
1949  auto masked =
1950  rewriter.createOrFold<OrOp>(op.getLoc(), extendedCond, restOfAssoc);
1951  replaceOpWithNewOpAndCopyName<AndOp>(rewriter, op, masked, commonValue);
1952  return true;
1953 }

1955 /// This function is invoke when we find a mux with true/false operations that
1956 /// have the same opcode. Check to see if we can strength reduce the mux by
1957 /// applying it to less data by applying this transformation:
1958 /// `mux(cond, op(a, b), op(a, c))` -> `op(a, mux(cond, b, c))`
1959 static bool foldCommonMuxOperation(MuxOp mux, Operation *trueOp,
1960  Operation *falseOp,
1961  PatternRewriter &rewriter) {
1962  // Right now we only apply to concat.
1963  // TODO: Generalize this to and, or, xor, icmp(!), which all occur in practice
1964  if (!isa<ConcatOp>(trueOp))
1965  return false;
1966 
1967  // Decode the operands, looking through recursive concats and replicates.
1968  SmallVector<Value> trueOperands, falseOperands;
1969  getConcatOperands(trueOp->getResult(0), trueOperands);
1970  getConcatOperands(falseOp->getResult(0), falseOperands);
1971 
1972  size_t numTrueOperands = trueOperands.size();
1973  size_t numFalseOperands = falseOperands.size();
1974 
1975  if (!numTrueOperands || !numFalseOperands ||
1976  (trueOperands.front() != falseOperands.front() &&
1977  trueOperands.back() != falseOperands.back()))
1978  return false;
1979 
1980  // Pull all leading shared operands out into their own op if any are common.
1981  if (trueOperands.front() == falseOperands.front()) {
1982  SmallVector<Value> operands;
1983  size_t i;
1984  for (i = 0; i < numTrueOperands; ++i) {
1985  Value trueOperand = trueOperands[i];
1986  if (trueOperand == falseOperands[i])
1987  operands.push_back(trueOperand);
1988  else
1989  break;
1990  }
1991  if (i == numTrueOperands) {
1992  // Selecting between distinct, but lexically identical, concats.
1993  replaceOpAndCopyName(rewriter, mux, trueOp->getResult(0));
1994  return true;
1995  }
1996 
1997  Value sharedMSB;
1998  if (llvm::all_of(operands, [&](Value v) { return v == operands.front(); }))
1999  sharedMSB = rewriter.createOrFold<ReplicateOp>(
2000  mux->getLoc(), operands.front(), operands.size());
2001  else
2002  sharedMSB = rewriter.createOrFold<ConcatOp>(mux->getLoc(), operands);
2003  operands.clear();
2004 
2005  // Get a concat of the LSB's on each side.
2006  operands.append(trueOperands.begin() + i, trueOperands.end());
2007  Value trueLSB = rewriter.createOrFold<ConcatOp>(trueOp->getLoc(), operands);
2008  operands.clear();
2009  operands.append(falseOperands.begin() + i, falseOperands.end());
2010  Value falseLSB =
2011  rewriter.createOrFold<ConcatOp>(falseOp->getLoc(), operands);
2012  // Merge the LSBs with a new mux and concat the MSB with the LSB to be
2013  // done.
2014  Value lsb = rewriter.createOrFold<MuxOp>(mux->getLoc(), mux.cond(), trueLSB,
2015  falseLSB);
2016  replaceOpWithNewOpAndCopyName<ConcatOp>(rewriter, mux, sharedMSB, lsb);
2017  return true;
2018  }
2019 
2020  // If trailing operands match, try to commonize them.
2021  if (trueOperands.back() == falseOperands.back()) {
2022  SmallVector<Value> operands;
2023  size_t i;
2024  for (i = 0;; ++i) {
2025  Value trueOperand = trueOperands[numTrueOperands - i - 1];
2026  if (trueOperand == falseOperands[numFalseOperands - i - 1])
2027  operands.push_back(trueOperand);
2028  else
2029  break;
2030  }
2031  std::reverse(operands.begin(), operands.end());
2032  Value sharedLSB = rewriter.createOrFold<ConcatOp>(mux->getLoc(), operands);
2033  operands.clear();
2034 
2035  // Get a concat of the MSB's on each side.
2036  operands.append(trueOperands.begin(), trueOperands.end() - i);
2037  Value trueMSB = rewriter.createOrFold<ConcatOp>(trueOp->getLoc(), operands);
2038  operands.clear();
2039  operands.append(falseOperands.begin(), falseOperands.end() - i);
2040  Value falseMSB =
2041  rewriter.createOrFold<ConcatOp>(falseOp->getLoc(), operands);
2042  // Merge the MSBs with a new mux and concat the MSB with the LSB to be done.
2043  Value msb = rewriter.createOrFold<MuxOp>(mux->getLoc(), mux.cond(), trueMSB,
2044  falseMSB);
2045  replaceOpWithNewOpAndCopyName<ConcatOp>(rewriter, mux, msb, sharedLSB);
2046  return true;
2047  }
2048 
2049  return false;
2050 }

2056  if (value.getBitWidth() == 1) {
2057  // mux(a, 0, b) -> and(~a, b) for single-bit values.
2058  if (value.isZero()) {
2059  auto notCond = createOrFoldNot(op.getLoc(), op.cond(), rewriter);
2060  replaceOpWithNewOpAndCopyName<AndOp>(rewriter, op, notCond,
2061  op.falseValue());

2063  }
2064 
2065  // mux(a, 1, b) -> or(a, b) for single-bit values.
2066  replaceOpWithNewOpAndCopyName<OrOp>(rewriter, op, op.cond(),
2067  op.falseValue());

2069  }
2070 
2071  // Check for mux of two constants. There are many ways to simplify them.
2072  APInt value2;
2073  if (matchPattern(op.falseValue(), m_RConstant(value2))) {
2074  // When both inputs are constants and differ by only one bit, we can
2075  // simplify by splitting the mux into up to three contiguous chunks: one
2076  // for the differing bit and up to two for the bits that are the same.
2077  // E.g. mux(a, 3'h2, 0) -> concat(0, mux(a, 1, 0), 0) -> concat(0, a, 0)
2078  APInt xorValue = value ^ value2;
2079  if (xorValue.isPowerOf2()) {
2080  unsigned leadingZeros = xorValue.countLeadingZeros();
2081  unsigned trailingZeros = value.getBitWidth() - leadingZeros - 1;

2083

2085  // order of bit indexes.

2087  // xorValue), we can extract directly from either input value, and we
2088  // arbitrarily pick the trueValue().

2091  operands.push_back(rewriter.createOrFold<ExtractOp>(

2095  // ~cond.

2097  op.getLoc(), op.cond(),

2100  rewriter.createOrFold<ExtractOp>(op.getLoc(), op.falseValue(),
2101  trailingZeros, 1)));

2105  op.getLoc(), op.trueValue(), 0, trailingZeros));
