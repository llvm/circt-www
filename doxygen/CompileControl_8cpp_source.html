27 /// Given some number of states, returns the necessary bit width
28 /// TODO(Calyx): Probably a better built-in operation?
29 static size_t getNecessaryBitWidth(size_t numStates) {
30  APInt apNumStates(64, numStates);
31  size_t log2 = apNumStates.ceilLogBase2();
32  return log2 > 1 ? log2 : 1;
33 }

54 /// Generates a latency-insensitive FSM to realize a sequential operation.
55 /// This is done by initializing GroupGoOp values for the enabled groups in
56 /// the SeqOp, and then creating a new Seq GroupOp with the given FSM. Each
57 /// step in the FSM is guarded by the done operation of the group currently
58 /// being executed. After the group is complete, the FSM is incremented. This
59 /// SeqOp is then replaced in the control with an Enable statement referring
60 /// to the new Seq GroupOp.
61 void CompileControlVisitor::visit(SeqOp seq, ComponentOp &component) {
62  auto wires = component.getWiresOp();
63  Block *wiresBody = wires.getBody();
64 
65  auto &seqOps = seq.getBody()->getOperations();
66  if (!llvm::all_of(seqOps, [](auto &&op) { return isa<EnableOp>(op); })) {
67  seq.emitOpError("should only contain EnableOps in this pass.");
68  return;
69  }
70 
71  // This should be the number of enable statements + 1 since this is the
72  // maximum value the FSM register will reach.
73  size_t fsmBitWidth = getNecessaryBitWidth(seqOps.size() + 1);
74 
75  OpBuilder builder(component->getRegion(0));
76  auto fsmRegister =
77  createRegister(seq.getLoc(), builder, component, fsmBitWidth, "fsm");
78  Value fsmIn = fsmRegister.in();
79  Value fsmWriteEn = fsmRegister.write_en();
80  Value fsmOut = fsmRegister.out();
81 
82  builder.setInsertionPointToStart(wiresBody);
83  auto oneConstant = createConstant(wires.getLoc(), builder, component, 1, 1);
84 
85  // Create the new compilation group to replace this SeqOp.
86  builder.setInsertionPointToEnd(wiresBody);
87  auto seqGroup =
88  builder.create<GroupOp>(wires->getLoc(), builder.getStringAttr("seq"));
89 
90  // Guarantees a unique SymbolName for the group.
91  SymbolTable symTable(wires);
92  symTable.insert(seqGroup);
93 
94  size_t fsmIndex = 0;
95  SmallVector<Attribute, 8> compiledGroups;
96  Value fsmNextState;
97  seq.walk([&](EnableOp enable) {
98  StringRef groupName = enable.groupName();
99  compiledGroups.push_back(
100  SymbolRefAttr::get(builder.getContext(), groupName));
101  auto groupOp = symTable.lookup<GroupOp>(groupName);
102 
103  builder.setInsertionPoint(groupOp);
104  auto fsmCurrentState = createConstant(wires->getLoc(), builder, component,
105  fsmBitWidth, fsmIndex);
106 
107  // TODO(Calyx): Eventually, we should canonicalize the GroupDoneOp's guard
108  // and source.
109  auto guard = groupOp.getDoneOp().guard();
110  auto source = groupOp.getDoneOp().src();
111  auto doneOpValue =
112  !guard ? source
113  : builder.create<comb::AndOp>(wires->getLoc(), guard, source);
114 
115  // Build the Guard for the `go` signal of the current group being walked.
116  // The group should begin when:
117  // (1) the current step in the fsm is reached, and
118  // (2) the done signal of this group is not high.
119  auto eqCmp = builder.create<comb::ICmpOp>(
120  wires->getLoc(), comb::ICmpPredicate::eq, fsmOut, fsmCurrentState);
121  auto notDone = comb::createOrFoldNot(wires->getLoc(), doneOpValue, builder);
122  auto groupGoGuard =
123  builder.create<comb::AndOp>(wires->getLoc(), eqCmp, notDone);
124 
125  // Guard for the `in` and `write_en` signal of the fsm register. These are
126  // driven when the group has completed.
127  builder.setInsertionPoint(seqGroup);
128  auto groupDoneGuard =
129  builder.create<comb::AndOp>(wires->getLoc(), eqCmp, doneOpValue);
130 
131  // Directly update the GroupGoOp of the current group being walked.
132  auto goOp = groupOp.getGoOp();
133  assert(goOp && "The Go Insertion pass should be run before this.");
134  goOp->setOperands({oneConstant, groupGoGuard});
135 
136  // Add guarded assignments to the fsm register `in` and `write_en` ports.
137  fsmNextState = createConstant(wires->getLoc(), builder, component,
138  fsmBitWidth, fsmIndex + 1);
139  builder.setInsertionPointToEnd(seqGroup.getBody());
140  builder.create<AssignOp>(wires->getLoc(), fsmIn, fsmNextState,
141  groupDoneGuard);
142  builder.create<AssignOp>(wires->getLoc(), fsmWriteEn, oneConstant,
143  groupDoneGuard);
144  // Increment the fsm index for the next group.
145  ++fsmIndex;
146  });
147 
148  // Build the final guard for the new Seq group's GroupDoneOp. This is
149  // defined by the fsm's final state.
150  builder.setInsertionPoint(seqGroup);
151  auto isFinalState = builder.create<comb::ICmpOp>(
152  wires->getLoc(), comb::ICmpPredicate::eq, fsmOut, fsmNextState);
153 
154  // Insert the respective GroupDoneOp.
155  builder.setInsertionPointToEnd(seqGroup.getBody());
156  builder.create<GroupDoneOp>(seqGroup->getLoc(), oneConstant, isFinalState);
157 
158  // Add continuous wires to reset the `in` and `write_en` ports of the fsm
159  // when the SeqGroup is finished executing.
160  builder.setInsertionPointToEnd(wiresBody);
161  auto zeroConstant =
162  createConstant(wires->getLoc(), builder, component, fsmBitWidth, 0);
163  builder.create<AssignOp>(wires->getLoc(), fsmIn, zeroConstant, isFinalState);
164  builder.create<AssignOp>(wires->getLoc(), fsmWriteEn, oneConstant,
165  isFinalState);
166 
167  // Replace the SeqOp with an EnableOp.
168  builder.setInsertionPoint(seq);
169  builder.create<EnableOp>(
170  seq->getLoc(), seqGroup.sym_name(),
171  ArrayAttr::get(builder.getContext(), compiledGroups));
172 
173  seq->erase();
174 }

184 void CompileControlPass::runOnOperation() {
185  ComponentOp component = getOperation();
186  CompileControlVisitor CompileControlVisitor;
187  component.getControlOp().walk(
188  [&](Operation *op) { CompileControlVisitor.dispatch(op, component); });
189 
190  // A post-condition of this pass is that all undefined GroupGoOps, created
191  // in the Go Insertion pass, are now defined.
192  component.getWiresOp().walk([&](UndefinedOp op) { op->erase(); });
193 }
