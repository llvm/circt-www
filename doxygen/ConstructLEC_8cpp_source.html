39 static Value lookupOrCreateStringGlobal(OpBuilder &builder, ModuleOp moduleOp,
40  StringRef str) {
41  Location loc = moduleOp.getLoc();
42  auto global = moduleOp.lookupSymbol<LLVM::GlobalOp>(str);
43  if (!global) {
44  OpBuilder b = OpBuilder::atBlockEnd(moduleOp.getBody());
45  auto arrayTy = LLVM::LLVMArrayType::get(b.getI8Type(), str.size() + 1);
46  global = b.create<LLVM::GlobalOp>(
47  loc, arrayTy, /*isConstant=*/true, LLVM::linkage::Linkage::Private, str,
48  StringAttr::get(b.getContext(), Twine(str).concat(Twine('\00'))));
49  }
50 
51  // FIXME: sanity check the fetched global: do all the attributes match what
52  // we expect?
53 
54  return builder.create<LLVM::AddressOfOp>(loc, global);
55 }

57 hw::HWModuleOp ConstructLECPass::lookupModule(StringRef name) {
58  Operation *expectedModule = SymbolTable::lookupNearestSymbolFrom(
59  getOperation(), StringAttr::get(&getContext(), name));
60  if (!expectedModule || !isa<hw::HWModuleOp>(expectedModule)) {
61  getOperation().emitError("module named '") << name << "' not found";
62  return {};
63  }
64  return cast<hw::HWModuleOp>(expectedModule);
65 }

67 void ConstructLECPass::runOnOperation() {
68  // Create necessary function declarations and globals
69  OpBuilder builder = OpBuilder::atBlockEnd(getOperation().getBody());
70  Location loc = getOperation()->getLoc();
71  auto ptrTy = LLVM::LLVMPointerType::get(builder.getContext());
72  auto voidTy = LLVM::LLVMVoidType::get(&getContext());
73 
74  // Lookup or declare printf function.
75  auto printfFunc =
76  LLVM::lookupOrCreateFn(getOperation(), "printf", ptrTy, voidTy, true);
77 
78  // Lookup the modules.
79  auto moduleA = lookupModule(firstModule);
80  if (!moduleA)
81  return signalPassFailure();
82  auto moduleB = lookupModule(secondModule);
83  if (!moduleB)
84  return signalPassFailure();
85 
86  if (moduleA.getModuleType() != moduleB.getModuleType()) {
87  moduleA.emitError("module's IO types don't match second modules: ")
88  << moduleA.getModuleType() << " vs " << moduleB.getModuleType();
89  return signalPassFailure();
90  }
91 
92  // Reuse the name of the first module for the entry function, so we don't have
93  // to do any uniquing and the LEC driver also already knows this name.
94  FunctionType functionType = FunctionType::get(&getContext(), {}, {});
95  func::FuncOp entryFunc =
96  builder.create<func::FuncOp>(loc, firstModule, functionType);
97 
98  if (insertMainFunc) {
99  OpBuilder::InsertionGuard guard(builder);
100  auto i32Ty = builder.getI32Type();
101  auto mainFunc = builder.create<func::FuncOp>(
102  loc, "main", builder.getFunctionType({i32Ty, ptrTy}, {i32Ty}));
103  builder.createBlock(&mainFunc.getBody(), {}, {i32Ty, ptrTy}, {loc, loc});
104  builder.create<func::CallOp>(loc, entryFunc, ValueRange{});
105  // TODO: don't use LLVM here
106  Value constZero = builder.create<LLVM::ConstantOp>(loc, i32Ty, 0);
107  builder.create<func::ReturnOp>(loc, constZero);
108  }
109 
110  builder.createBlock(&entryFunc.getBody());
111  auto lecOp = builder.create<verif::LogicEquivalenceCheckingOp>(loc);
112  auto *outputOpA = moduleA.getBodyBlock()->getTerminator();
113  auto *outputOpB = moduleB.getBodyBlock()->getTerminator();
114  lecOp.getFirstCircuit().takeBody(moduleA.getBody());
115  lecOp.getSecondCircuit().takeBody(moduleB.getBody());
116  moduleA->erase();
117  moduleB->erase();
118 
119  {
120  OpBuilder::InsertionGuard guard(builder);
121  builder.setInsertionPoint(outputOpA);
122  builder.create<verif::YieldOp>(loc, outputOpA->getOperands());
123  outputOpA->erase();
124  builder.setInsertionPoint(outputOpB);
125  builder.create<verif::YieldOp>(loc, outputOpB->getOperands());
126  outputOpB->erase();
127  }
128 
129  // TODO: we should find a more elegant way of reporting the result than
130  // already inserting some LLVM here
131  Value eqFormatString =
132  lookupOrCreateStringGlobal(builder, getOperation(), "c1 == c2\n");
133  Value neqFormatString =
134  lookupOrCreateStringGlobal(builder, getOperation(), "c1 != c2\n");
135  Value formatString = builder.create<LLVM::SelectOp>(
136  loc, lecOp.getAreEquivalent(), eqFormatString, neqFormatString);
137  builder.create<LLVM::CallOp>(loc, printfFunc, ValueRange{formatString});
138 
139  builder.create<func::ReturnOp>(loc, ValueRange{});
140 }
