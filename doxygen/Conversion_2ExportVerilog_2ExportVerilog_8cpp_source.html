63 /// where a lower number binds tighter.
64 enum VerilogPrecedence {
65  // Normal precedence levels.
66  Symbol, // Atomic symbol like "foo" and {a,b}
67  Selection, // () , [] , :: , ., $signed()
68  Unary, // Unary operators like ~foo
69  Multiply, // * , / , %
70  Addition, // + , -
71  Shift, // << , >>, <<<, >>>
72  Comparison, // > , >= , < , <=
73  Equality, // == , !=
74  And, // &

77  AndShortCircuit, // &&
78  Conditional, // ? :

115 /// Return true for nullary operations that are better emitted multiple
116 /// times as inline expression (when they have multiple uses) rather than having
117 /// a temporary wire.
118 ///
119 /// This can only handle nullary expressions, because we don't want to replicate
120 /// subtrees arbitrarily.
121 static bool isDuplicatableNullaryExpression(Operation *op) {
122  // We don't want wires that are just constants aesthetically.
123  if (isConstantExpression(op))
124  return true;
125 
126  // If this is a small verbatim expression with no side effects, duplicate it
127  // inline.
128  if (isa<VerbatimExprOp>(op)) {
129  if (op->getNumOperands() == 0 &&
130  op->getAttrOfType<StringAttr>("string").getValue().size() <= 32)
131  return true;
132  }
133 
134  // If this is a macro reference without side effects, allow duplication.
135  if (isa<MacroRefExprOp>(op))
136  return true;
137 
138  return false;
139 }

141 // Return true if the expression can be inlined even when the op has multiple
142 // uses. Be careful to add operations here since it might cause exponential
143 // emission without proper restrictions.
144 static bool isDuplicatableExpression(Operation *op) {
145  if (op->getNumOperands() == 0)
146  return isDuplicatableNullaryExpression(op);

156  return false;

190 StringRef getPortVerilogName(Operation *module, ssize_t portArgNum) {
191  auto numInputs = hw::getModuleNumInputs(module);
192  // portArgNum is the index into the result of getAllModulePortInfos.
193  // Also ensure the correct index into the input/output list is computed.
194  ssize_t portId = portArgNum;
195  char verilogNameAttr[] = "hw.verilogName";
196  // Check for input ports.

200  if (auto argDict = argAttr[portArgNum].cast<DictionaryAttr>())
201  if (auto updatedName = argDict.get(verilogNameAttr))
202  return updatedName.cast<StringAttr>().getValue();
203  // Get the original name of input port if no renaming.
204  return module->getAttrOfType<ArrayAttr>("argNames")[portArgNum]
205  .cast<StringAttr>()
206  .getValue();
207  }
208

210  portId = portArgNum - numInputs;

212  mlir::function_interface_impl::getResultDictAttrName()))

214  if (auto updatedName = argDict.get(verilogNameAttr))
215  return updatedName.cast<StringAttr>().getValue();
216  // Get the original name of output port if no renaming.
217  return module->getAttrOfType<ArrayAttr>("resultNames")[portId]
218  .cast<StringAttr>()
219  .getValue();

247  return TypeSwitch<Type, int>(type)

262 /// Push this type's dimension into a vector.
263 static void getTypeDims(SmallVectorImpl<Attribute> &dims, Type type,
264  Location loc) {
265  if (auto integer = hw::type_dyn_cast<IntegerType>(type)) {
266  if (integer.getWidth() != 1)
267  dims.push_back(getInt32Attr(type.getContext(), integer.getWidth()));
268  return;
269  }
270  if (auto array = hw::type_dyn_cast<ArrayType>(type)) {
271  dims.push_back(getInt32Attr(type.getContext(), array.getSize()));
272  getTypeDims(dims, array.getElementType(), loc);
273 
274  return;
275  }
276  if (auto intType = hw::type_dyn_cast<IntType>(type)) {
277  dims.push_back(intType.getWidth());
278  return;
279  }
280 
281  if (auto inout = hw::type_dyn_cast<InOutType>(type))
282  return getTypeDims(dims, inout.getElementType(), loc);
283  if (auto uarray = hw::type_dyn_cast<hw::UnpackedArrayType>(type))
284  return getTypeDims(dims, uarray.getElementType(), loc);
285  if (hw::type_isa<InterfaceType, StructType, EnumType>(type))
286  return;
287 
288  mlir::emitError(loc, "value has an unsupported verilog type ") << type;
289 }

302 /// Return true if this is a zero bit type, e.g. a zero bit integer or array
303 /// thereof.
304 static bool isZeroBitType(Type type) {
305  if (auto intType = type.dyn_cast<IntegerType>())
306  return intType.getWidth() == 0;
307  if (auto inout = type.dyn_cast<hw::InOutType>())
308  return isZeroBitType(inout.getElementType());
309  if (auto uarray = type.dyn_cast<hw::UnpackedArrayType>())
310  return isZeroBitType(uarray.getElementType());
311  if (auto array = type.dyn_cast<hw::ArrayType>())
312  return isZeroBitType(array.getElementType());
313  if (auto structType = type.dyn_cast<hw::StructType>())
314  return llvm::all_of(structType.getElements(),
315  [](auto elem) { return isZeroBitType(elem.type); });
316 
317  // We have an open type system, so assume it is ok.
318  return false;
319 }

345 /// Return the word (e.g. "reg") in Verilog to declare the specified thing.
346 static StringRef getVerilogDeclWord(Operation *op,
347  const LoweringOptions &options) {
348  if (isa<RegOp>(op)) {
349  // Check if the type stored in this register is a struct or array of
350  // structs. In this case, according to spec section 6.8, the "reg" prefix
351  // should be left off.
352  auto elementType =
353  op->getResult(0).getType().cast<InOutType>().getElementType();
354  if (elementType.isa<StructType>())
355  return "";
356  if (elementType.isa<EnumType>())
357  return "";
358  if (auto innerType = elementType.dyn_cast<ArrayType>()) {
359  while (innerType.getElementType().isa<ArrayType>())
360  innerType = innerType.getElementType().cast<ArrayType>();
361  if (innerType.getElementType().isa<StructType>() ||
362  innerType.getElementType().isa<TypeAliasType>())
363  return "";
364  }
365  if (elementType.isa<TypeAliasType>())
366  return "";
367 
368  return "reg";
369  }
370  if (isa<WireOp>(op))
371  return "wire";
372  if (isa<ConstantOp, LocalParamOp, ParamValueOp>(op))
373  return "localparam";
374 
375  // Interfaces instances use the name of the declared interface.
376  if (auto interface = dyn_cast<InterfaceInstanceOp>(op))
377  return interface.getInterfaceType().getInterface().getValue();
378 
379  // If 'op' is in a module, output 'wire'. If 'op' is in a procedural block,
380  // fall through to default.
381  bool isProcedural = op->getParentOp()->hasTrait<ProceduralRegion>();
382 
383  if (isa<LogicOp>(op)) {
384  // If the logic op is defined in a procedural region, add 'automatic'
385  // keyword. If the op has a struct type, 'logic' keyword is already emitted
386  // within a struct type definition (e.g. struct packed {logic foo;}). So we
387  // should not emit extra 'logic'.
388  bool hasStruct = hasStructType(op->getResult(0).getType());
389  if (isProcedural)
390  return hasStruct ? "automatic" : "automatic logic";
391  return hasStruct ? "" : "logic";
392  }
393 
394  if (!isProcedural)
395  return "wire";
396 
397  // "automatic" values aren't allowed in disallowLocalVariables mode.
398  assert(!options.disallowLocalVariables && "automatic variables not allowed");
399 
400  // If the type contains a struct type, we have to use only "automatic" because
401  // "automatic struct" is syntactically correct.
402  return hasStructType(op->getResult(0).getType()) ? "automatic"
403  : "automatic logic";
404 }

418 /// Return the location information as a (potentially empty) string.
419 static std::string
420 getLocationInfoAsStringImpl(const SmallPtrSet<Operation *, 8> &ops) {
421  std::string resultStr;
422  llvm::raw_string_ostream sstr(resultStr);
423 
424  // Multiple operations may come from the same location or may not have useful
425  // location info. Unique it now.
426  SmallPtrSet<Attribute, 8> locationSet;
427  for (auto *op : ops)
428  collectFileLineColLocs(op->getLoc(), locationSet);
429 
430  auto printLoc = [&](FileLineColLoc loc) {
431  sstr << loc.getFilename().getValue();
432  if (auto line = loc.getLine()) {
433  sstr << ':' << line;
434  if (auto col = loc.getColumn())
435  sstr << ':' << col;
436  }
437  };
438 
439  // Fast pass some common cases.
440  switch (locationSet.size()) {
441  case 1:
442  printLoc((*locationSet.begin()).cast<FileLineColLoc>());
443  LLVM_FALLTHROUGH;
444  case 0:
445  return sstr.str();
446  default:
447  break;
448  }
449 
450  // Sort the entries.
451  SmallVector<FileLineColLoc, 8> locVector;
452  locVector.reserve(locationSet.size());
453  for (auto loc : locationSet)
454  locVector.push_back(loc.cast<FileLineColLoc>());
455 
456  llvm::array_pod_sort(
457  locVector.begin(), locVector.end(),
458  [](const FileLineColLoc *lhs, const FileLineColLoc *rhs) -> int {
459  if (auto fn = lhs->getFilename().compare(rhs->getFilename()))
460  return fn;
461  if (lhs->getLine() != rhs->getLine())
462  return lhs->getLine() < rhs->getLine() ? -1 : 1;
463  return lhs->getColumn() < rhs->getColumn() ? -1 : 1;
464  });
465 
466  // The entries are sorted by filename, line, col. Try to merge together
467  // entries to reduce verbosity on the column info.
468  StringRef lastFileName;
469  for (size_t i = 0, e = locVector.size(); i != e;) {
470  if (i != 0)
471  sstr << ", ";
472 
473  // Print the filename if it changed.
474  auto first = locVector[i];
475  if (first.getFilename() != lastFileName) {
476  lastFileName = first.getFilename();
477  sstr << lastFileName;
478  }
479 
480  // Scan for entries with the same file/line.
481  size_t end = i + 1;
482  while (end != e && first.getFilename() == locVector[end].getFilename() &&
483  first.getLine() == locVector[end].getLine())
484  ++end;
485 
486  // If we have one entry, print it normally.
487  if (end == i + 1) {
488  if (auto line = first.getLine()) {
489  sstr << ':' << line;
490  if (auto col = first.getColumn())
491  sstr << ':' << col;
492  }
493  ++i;
494  continue;
495  }
496 
497  // Otherwise print a brace enclosed list.
498  sstr << ':' << first.getLine() << ":{";
499  while (i != end) {
500  sstr << locVector[i++].getColumn();
501 
502  if (i != end)
503  sstr << ',';
504  }
505  sstr << '}';
506  }
507  return sstr.str();
508 }

535 /// Most expressions are invalid to bit-select from in Verilog, but some
536 /// things are ok. Return true if it is ok to inline bitselect from the
537 /// result of this expression. It is conservatively correct to return false.
538 static bool isOkToBitSelectFrom(Value v) {
539  // Module ports are always ok to bit select from.
540  if (v.isa<BlockArgument>())
541  return true;
542 
543  // Uses of a wire or register can be done inline.
544  if (auto read = v.getDefiningOp<ReadInOutOp>()) {
545  if (read.input().getDefiningOp<WireOp>() ||
546  read.input().getDefiningOp<RegOp>() ||
547  read.input().getDefiningOp<LogicOp>())
548  return true;
549  }
550 
551  // Aggregate access can be inlined.
552  if (v.getDefiningOp<StructExtractOp>())
553  return true;
554 
555  // Interface signal can be inlined.
556  if (v.getDefiningOp<ReadInterfaceSignalOp>())
557  return true;
558 
559  // TODO: We could handle concat and other operators here.
560  return false;
561 }

563 /// Return true if we are unable to ever inline the specified operation. This
564 /// happens because not all Verilog expressions are composable, notably you
565 /// can only use bit selects like x[4:6] on simple expressions, you cannot use
566 /// expressions in the sensitivity list of always blocks, etc.
567 static bool isExpressionUnableToInline(Operation *op) {
568  if (auto cast = dyn_cast<BitcastOp>(op))
569  if (!haveMatchingDims(cast.input().getType(), cast.result().getType(),
570  op->getLoc()))
571  // Bitcasts rely on the type being assigned to, so we cannot inline.
572  return true;
573 
574  // StructCreateOp needs to be assigning to a named temporary so that types
575  // are inferred properly by verilog
576  if (isa<StructCreateOp>(op))
577  return true;
578 
579  // Verbatim with a long string should be emitted as an out-of-line declration.
580  if (auto verbatim = dyn_cast<VerbatimExprOp>(op))
581  if (verbatim.string().size() > 32)
582  return true;
583 
584  // Scan the users of the operation to see if any of them need this to be
585  // emitted out-of-line.
586  for (auto *user : op->getUsers()) {
587  // Verilog bit selection is required by the standard to be:
588  // "a vector, packed array, packed structure, parameter or concatenation".
589  //
590  // It cannot be an arbitrary expression, e.g. this is invalid:
591  // assign bar = {{a}, {b}, {c}, {d}}[idx];
592  //
593  // To handle these, we push the subexpression into a temporary.
594  if (isa<ExtractOp, ArraySliceOp, ArrayGetOp, StructExtractOp>(user))
595  if (op->getResult(0) == user->getOperand(0) && // ignore index operands.
596  !isOkToBitSelectFrom(op->getResult(0)))
597  return true;
598 
599  // Always blocks must have a name in their sensitivity list, not an expr.
600  if (isa<AlwaysOp>(user) || isa<AlwaysFFOp>(user)) {
601  // Anything other than a read of a wire must be out of line.
602  if (auto read = dyn_cast<ReadInOutOp>(op))
603  if (read.input().getDefiningOp<WireOp>() ||
604  read.input().getDefiningOp<RegOp>())
605  continue;
606  return true;
607  }
608  }
609  return false;

851 void EmitterBase::emitTextWithSubstitutions(
852  StringRef string, Operation *op, std::function<void(Value)> operandEmitter,
853  ArrayAttr symAttrs, ModuleNameManager &names) {
854 
855  // Perform operand substitions as we emit the line string. We turn {{42}}
856  // into the value of operand 42.
857  auto namify = [&](Attribute sym, HWSymbolCache::Item item) {
858  // CAVEAT: These accesses can reach into other modules through inner name
859  // references, which are currently being processed. Do not add those remote
860  // operations to this module's `names`, which is reserved for things named
861  // *within* this module. Instead, you have to rely on those remote
862  // operations to have been named inside the global names table. If they
863  // haven't, take a look at name name legalization first.
864  if (auto itemOp = item.getOp()) {
865  if (item.hasPort()) {
866  return getPortVerilogName(itemOp, item.getPort());
867  }
868  StringRef symOpName = getSymOpName(itemOp);
869  if (!symOpName.empty())
870  return symOpName;
871  emitError(itemOp, "cannot get name for symbol ") << sym;
872  } else {
873  emitError(op, "cannot get name for symbol ") << sym;
874  }
875  return StringRef("<INVALID>");
876  };
877 
878  // Scan 'line' for a substitution, emitting any non-substitution prefix,
879  // then the mentioned operand, chopping the relevant text off 'line' and
880  // returning true. This returns false if no substitution is found.
881  unsigned numSymOps = symAttrs.size();
882  auto emitUntilSubstitution = [&](size_t next = 0) -> bool {
883  size_t start = 0;
884  while (1) {
885  next = string.find("{{", next);
886  if (next == StringRef::npos)
887  return false;
888 
889  // Check to make sure we have a number followed by }}. If not, we
890  // ignore the {{ sequence as something that could happen in Verilog.
891  next += 2;
892  start = next;
893  while (next < string.size() && isdigit(string[next]))
894  ++next;
895  // We need at least one digit.
896  if (start == next) {
897  next--;
898  continue;
899  }
900 
901  // We must have a }} right after the digits.
902  if (!string.substr(next).startswith("}}"))
903  continue;
904 
905  // We must be able to decode the integer into an unsigned.
906  unsigned operandNo = 0;
907  if (string.drop_front(start)
908  .take_front(next - start)
909  .getAsInteger(10, operandNo)) {
910  emitError(op, "operand substitution too large");
911  continue;
912  }
913  next += 2;
914 
915  // Emit any text before the substitution.
916  os << string.take_front(start - 2);
917 
918  // operandNo can either refer to Operands or symOps. symOps are
919  // numbered after the operands.
920  if (operandNo < op->getNumOperands())
921  // Emit the operand.
922  operandEmitter(op->getOperand(operandNo));
923  else if ((operandNo - op->getNumOperands()) < numSymOps) {
924  unsigned symOpNum = operandNo - op->getNumOperands();
925  auto sym = symAttrs[symOpNum];
926  StringRef symVerilogName;
927  if (auto fsym = sym.dyn_cast<FlatSymbolRefAttr>()) {
928  if (auto *symOp = state.symbolCache.getDefinition(fsym))
929  symVerilogName = namify(sym, symOp);
930  } else if (auto isym = sym.dyn_cast<InnerRefAttr>()) {
931  auto symOp = state.symbolCache.getInnerDefinition(isym.getModule(),
932  isym.getName());
933  symVerilogName = namify(sym, symOp);
934  }
935  os << symVerilogName;
936  } else {
937  emitError(op, "operand " + llvm::utostr(operandNo) + " isn't valid");
938  continue;
939  }
940  // Forget about the part we emitted.
941  string = string.drop_front(next);
942  return true;
943  }
944  };
945 
946  // Emit all the substitutions.
947  while (emitUntilSubstitution())
948  ;
949 
950  // Emit any text after the last substitution.
951  os << string;
952 }

954 void EmitterBase::emitComment(StringAttr comment) {
955  if (!comment)
956  return;
957 
958  // Set a line length for the comment. Subtract off the leading comment and
959  // space ("// ") as well as the current indent level to simplify later
960  // arithmetic. Ensure that this line length doesn't go below zero.
961  auto lineLength = state.options.emittedLineLength - state.currentIndent - 3;
962  if (lineLength > state.options.emittedLineLength)
963  lineLength = 0;
964 
965  // Process the comment in line chunks extracted from manually specified line
966  // breaks. This is done to preserve user-specified line breaking if used.
967  auto ref = comment.getValue();
968  StringRef line;
969  while (!ref.empty()) {
970  std::tie(line, ref) = ref.split("\n");
971  // Emit each comment line breaking it if it exceeds the emittedLineLength.
972  for (;;) {
973  indent();
974  os << "// ";
975 
976  // Base case 1: the entire comment fits on one line.
977  if (line.size() <= lineLength) {
978  os << line << "\n";
979  break;
980  }
981 
982  // The comment does NOT fit on one line. Use a simple algorithm to find
983  // a position to break the line:
984  // 1) Search backwards for whitespace and break there if you find it.
985  // 2) If no whitespace exists in (1), search forward for whitespace
986  // and break there.
987  // This algorithm violates the emittedLineLength if (2) ever occurrs,
988  // but it's dead simple.
989  auto breakPos = line.rfind(' ', lineLength);
990  // No whitespace exists looking backwards.
991  if (breakPos == StringRef::npos) {
992  breakPos = line.find(' ', lineLength);
993  // No whitespace exists looking forward (you hit the end of the
994  // string).
995  if (breakPos == StringRef::npos)
996  breakPos = line.size();
997  }
998 
999  // Emit up to the break position. Trim any whitespace after the break
1000  // position. Exit if nothing is left to emit. Otherwise, update the
1001  // comment ref and continue;
1002  os << line.take_front(breakPos) << "\n";
1003  breakPos = line.find_first_not_of(' ', breakPos);
1004  // Base Case 2: nothing left except whitespace.
1005  if (breakPos == StringRef::npos)
1006  break;
1007 
1008  line = line.drop_front(breakPos);
1009  }
1010  }
1011 }

1013 /// Given an expression that is spilled into a temporary wire, try to synthesize
1014 /// a better name than "_T_42" based on the structure of the expression.
1015 StringAttr EmitterBase::inferStructuralNameForTemporary(Value expr) {
1016  StringAttr result;
1017  bool addPrefixUnderScore = true;
1018 
1019  // Look through read_inout.
1020  if (auto read = expr.getDefiningOp<ReadInOutOp>())
1021  return inferStructuralNameForTemporary(read.input());
1022 
1023  // Module ports carry names!
1024  if (auto blockArg = expr.dyn_cast<BlockArgument>()) {
1025  auto moduleOp = cast<HWModuleOp>(blockArg.getOwner()->getParentOp());
1026  StringRef name = getPortVerilogName(moduleOp, blockArg.getArgNumber());
1027  result = StringAttr::get(expr.getContext(), name);
1028 
1029  } else if (auto *op = expr.getDefiningOp()) {
1030  // Uses of a wire, register or logic can be done inline.
1031  if (isa<WireOp, RegOp, LogicOp>(op)) {
1032  StringRef name = getSymOpName(op);
1033  result = StringAttr::get(expr.getContext(), name);
1034 
1035  } else if (auto nameHint = op->getAttrOfType<StringAttr>("sv.namehint")) {
1036  // Use a dialect (sv) attribute to get a hint for the name if the op
1037  // doesn't explicitly specify it. Do this last
1038  result = nameHint;
1039 
1040  // If there is a namehint, don't add underscores to the name.
1041  addPrefixUnderScore = false;
1042  } else {
1043  TypeSwitch<Operation *>(op)
1044  // Generate a pretty name for VerbatimExpr's that look macro-like
1045  // using the same logic that generates the MLIR syntax name.
1046  .Case([&result](VerbatimExprOp verbatim) {
1047  verbatim.getAsmResultNames([&](Value, StringRef name) {
1048  result = StringAttr::get(verbatim.getContext(), name);
1049  });
1050  })
1051  .Case([&result](VerbatimExprSEOp verbatim) {
1052  verbatim.getAsmResultNames([&](Value, StringRef name) {
1053  result = StringAttr::get(verbatim.getContext(), name);
1054  });
1055  })
1056 
1057  // If this is an extract from a namable object, derive a name from it.
1058  .Case([&result, this](ExtractOp extract) {
1059  if (auto operandName =
1060  inferStructuralNameForTemporary(extract.input())) {
1061  unsigned numBits = extract.getType().getWidth();
1062  if (numBits == 1)
1063  result = StringAttr::get(extract.getContext(),
1064  operandName.strref() + "_" +
1065  Twine(extract.lowBit()));
1066  else
1067  result =
1068  StringAttr::get(extract.getContext(),
1069  operandName.strref() + "_" +
1070  Twine(extract.lowBit() + numBits - 1) +
1071  "to" + Twine(extract.lowBit()));
1072  }
1073  });
1074  // TODO: handle other common patterns.
1075  }
1076  }
1077 
1078  // Make sure any synthesized name starts with an _.
1079  if (!result || result.strref().empty())
1080  return {};
1081 
1082  // Make sure that all temporary names start with an underscore.
1083  if (addPrefixUnderScore && result.strref().front() != '_')
1084  result = StringAttr::get(expr.getContext(), "_" + result.strref());
1085 
1086  return result;
1087 }

1363 /// Helper that prints a parameter constant value in a Verilog compatible way.
1364 /// This returns the precedence of the generated string.
1365 SubExprInfo
1366 ModuleEmitter::printParamValue(Attribute value, raw_ostream &os,
1367  VerilogPrecedence parenthesizeIfLooserThan,
1368  function_ref<InFlightDiagnostic()> emitError) {
1369  if (auto intAttr = value.dyn_cast<IntegerAttr>()) {
1370  IntegerType intTy = intAttr.getType().cast<IntegerType>();
1371  APInt value = intAttr.getValue();
1372 
1373  // We omit the width specifier if the value is <= 32-bits in size, which
1374  // makes this more compatible with unknown width extmodules.
1375  if (intTy.getWidth() > 32) {
1376  // Sign comes out before any width specifier.
1377  if (value.isNegative() && (intTy.isSigned() || intTy.isSignless())) {
1378  os << '-';
1379  value = -value;
1380  }
1381  if (intTy.isSigned())
1382  os << intTy.getWidth() << "'sd";
1383  else
1384  os << intTy.getWidth() << "'d";
1385  }
1386  value.print(os, intTy.isSigned());
1387  return {Symbol, intTy.isSigned() ? IsSigned : IsUnsigned};
1388  }
1389  if (auto strAttr = value.dyn_cast<StringAttr>()) {
1390  os << '"';
1391  os.write_escaped(strAttr.getValue());
1392  os << '"';
1393  return {Symbol, IsUnsigned};
1394  }
1395  if (auto fpAttr = value.dyn_cast<FloatAttr>()) {
1396  // TODO: relying on float printing to be precise is not a good idea.
1397  os << fpAttr.getValueAsDouble();
1398  return {Symbol, IsUnsigned};
1399  }
1400  if (auto verbatimParam = value.dyn_cast<ParamVerbatimAttr>()) {
1401  os << verbatimParam.getValue().getValue();
1402  return {Symbol, IsUnsigned};
1403  }
1404  if (auto parameterRef = value.dyn_cast<ParamDeclRefAttr>()) {
1405  // Get the name of this parameter (in case it got renamed).
1406  os << state.globalNames.getParameterVerilogName(currentModuleOp,
1407  parameterRef.getName());
1408 
1409  // TODO: Should we support signed parameters?
1410  return {Symbol, IsUnsigned};
1411  }
1412 
1413  // Handle nested expressions.
1414  auto expr = value.dyn_cast<ParamExprAttr>();
1415  if (!expr) {
1416  os << "<<UNKNOWN MLIRATTR: " << value << ">>";
1417  emitError() << " = " << value;
1418  return {LowestPrecedence, IsUnsigned};
1419  }
1420 
1421  StringRef operatorStr;
1422  VerilogPrecedence subprecedence = ForceEmitMultiUse;
1423  Optional<SubExprSignResult> operandSign;
1424  bool isUnary = false;
1425 
1426  switch (expr.getOpcode()) {
1427  case PEO::Add:
1428  operatorStr = " + ";
1429  subprecedence = Addition;
1430  break;
1431  case PEO::Mul:
1432  operatorStr = " * ";
1433  subprecedence = Multiply;
1434  break;
1435  case PEO::And:
1436  operatorStr = " & ";
1437  subprecedence = And;
1438  break;
1439  case PEO::Or:
1440  operatorStr = " | ";
1441  subprecedence = Or;
1442  break;
1443  case PEO::Xor:
1444  operatorStr = " ^ ";
1445  subprecedence = Xor;
1446  break;
1447  case PEO::Shl:
1448  operatorStr = " << ";
1449  subprecedence = Shift;
1450  break;
1451  case PEO::ShrU:
1452  // >> in verilog is always a logical shift even if operands are signed.
1453  operatorStr = " >> ";
1454  subprecedence = Shift;
1455  break;
1456  case PEO::ShrS:
1457  // >>> in verilog is an arithmetic shift if both operands are signed.
1458  operatorStr = " >>> ";
1459  subprecedence = Shift;
1460  operandSign = IsSigned;
1461  break;
1462  case PEO::DivU:
1463  operatorStr = " / ";
1464  subprecedence = Multiply;
1465  operandSign = IsUnsigned;
1466  break;
1467  case PEO::DivS:
1468  operatorStr = " / ";
1469  subprecedence = Multiply;
1470  operandSign = IsSigned;
1471  break;
1472  case PEO::ModU:
1473  operatorStr = " % ";
1474  subprecedence = Multiply;
1475  operandSign = IsUnsigned;
1476  break;
1477  case PEO::ModS:
1478  operatorStr = " % ";
1479  subprecedence = Multiply;
1480  operandSign = IsSigned;
1481  break;
1482  case PEO::CLog2:
1483  operatorStr = "$clog2";
1484  operandSign = IsUnsigned;
1485  isUnary = true;
1486  break;
1487  case PEO::StrConcat:
1488  operatorStr = ", ";
1489  subprecedence = Symbol;
1490  isUnary = false;
1491  break;
1492  }
1493 
1494  // Emit the specified operand with a $signed() or $unsigned() wrapper around
1495  // it if context requires a specific signedness to compute the right value.
1496  // This returns true if the operand is signed.
1497  // TODO: This could try harder to omit redundant casts like the mainline
1498  // expression emitter.
1499  auto emitOperand = [&](Attribute operand) -> bool {
1500  if (operandSign.hasValue())
1501  os << (operandSign.getValue() == IsSigned ? "$signed(" : "$unsigned(");
1502  auto signedness =
1503  printParamValue(operand, os, subprecedence, emitError).signedness;
1504  if (operandSign.hasValue()) {
1505  os << ')';
1506  signedness = operandSign.getValue();
1507  }
1508  return signedness == IsSigned;
1509  };
1510 
1511  if (isUnary)
1512  os << operatorStr;
1513 
1514  if (subprecedence > parenthesizeIfLooserThan)
1515  os << '(';
1516  if (expr.getOpcode() == PEO::StrConcat)
1517  os << '{';
1518  bool allOperandsSigned = emitOperand(expr.getOperands()[0]);
1519  for (auto op : ArrayRef(expr.getOperands()).drop_front()) {
1520  // Handle the special case of (a + b + -42) as (a + b - 42).
1521  // TODO: Also handle (a + b + x*-1).
1522  if (expr.getOpcode() == PEO::Add) {
1523  if (auto integer = op.dyn_cast<IntegerAttr>()) {
1524  const APInt &value = integer.getValue();
1525  if (value.isNegative() && !value.isMinSignedValue()) {
1526  os << " - ";
1527  allOperandsSigned &=
1528  emitOperand(IntegerAttr::get(op.getType(), -value));
1529  continue;
1530  }
1531  }
1532  }
1533 
1534  os << operatorStr;
1535  allOperandsSigned &= emitOperand(op);
1536  }
1537  if (expr.getOpcode() == PEO::StrConcat)
1538  os << '}';
1539  if (subprecedence > parenthesizeIfLooserThan) {
1540  os << ')';
1541  subprecedence = Symbol;
1542  }
1543  return {subprecedence, allOperandsSigned ? IsSigned : IsUnsigned};
1544 }

1771 SubExprInfo ExprEmitter::emitBinary(Operation *op, VerilogPrecedence prec,
1772  const char *syntax,
1773  unsigned emitBinaryFlags) {
1774  if (emitBinaryFlags & EB_ForceResultSigned)
1775  os << "$signed(";
1776  auto operandSignReq =
1777  SubExprSignRequirement(emitBinaryFlags & EB_OperandSignRequirementMask);
1778  auto lhsInfo = emitSubExpr(op->getOperand(0), prec, operandSignReq);
1779  os << ' ' << syntax << ' ';
1780 
1781  // Right associative operators are already generally variadic, we need to
1782  // handle things like: (a<4> == b<4>) == (c<3> == d<3>). When processing the
1783  // top operation of the tree, the rhs needs parens. When processing
1784  // known-reassociative operators like +, ^, etc we don't need parens.
1785  // TODO: MLIR should have general "Associative" trait.
1786  auto rhsPrec = prec;
1787  if (!isa<AddOp, MulOp, AndOp, OrOp, XorOp>(op))
1788  rhsPrec = VerilogPrecedence(prec - 1);
1789 
1790  // Introduce extra parentheses to specific patterns of expressions.
1791  // If op is "AndOp", and rhs is Reduction And, the output is like `a & &b`.
1792  // This is syntactically valid but some tool produces LINT warnings. Also it
1793  // would be confusing for users to read such expressions.
1794  bool emitRhsParentheses = false;
1795  if (auto rhsICmp = op->getOperand(1).getDefiningOp<ICmpOp>()) {
1796  if ((rhsICmp.isEqualAllOnes() && isa<AndOp>(op)) ||
1797  (rhsICmp.isNotEqualZero() && isa<OrOp>(op))) {
1798  if (isExpressionEmittedInline(rhsICmp)) {
1799  os << '(';
1800  emitRhsParentheses = true;
1801  rhsPrec = LowestPrecedence;
1802  }
1803  }
1804  }
1805 
1806  // If the RHS operand has self-determined width and always treated as
1807  // unsigned, inform emitSubExpr of this. This is true for the shift amount in
1808  // a shift operation.
1809  bool rhsIsUnsignedValueWithSelfDeterminedWidth = false;
1810  if (emitBinaryFlags & EB_RHS_UnsignedWithSelfDeterminedWidth) {
1811  rhsIsUnsignedValueWithSelfDeterminedWidth = true;
1812  operandSignReq = NoRequirement;
1813  }
1814 
1815  auto rhsInfo = emitSubExpr(op->getOperand(1), rhsPrec, operandSignReq,
1816  rhsIsUnsignedValueWithSelfDeterminedWidth);
1817  if (emitRhsParentheses)

1842 /// This function split the output buffer into multiple lines if the emitted
1843 /// length is larger than the constraint.
1844 void ExprEmitter::formatOutBuffer() {
1845  // If the output already satisfies the constraint, skip here.
1846  if (outBuffer.size() <= state.options.emittedLineLength)
1847  return;
1848 
1849  SmallVector<char> tmpOutBuffer;
1850  llvm::raw_svector_ostream tmpOs(tmpOutBuffer);
1851  auto it = outBuffer.begin();
1852  unsigned currentIndex = 0;
1853 
1854  while (it != outBuffer.end()) {
1855  // Split by a white space.
1856  auto next = std::find(it, outBuffer.end(), ' ');
1857  unsigned tokenLength = std::distance(it, next);
1858 
1859  if (!tmpOutBuffer.empty() &&
1860  currentIndex + tokenLength > state.options.emittedLineLength) {
1861  // It breaks the line constraint, so insert a newline and indent.
1862  tmpOs << '\n';
1863  tmpOs.indent(state.currentIndent *
1864  SPACE_PER_INDENT_IN_EXPRESSION_FORMATTING);
1865  currentIndex = tokenLength;
1866  tmpOutBuffer.insert(tmpOutBuffer.end(), it, next);
1867  } else {
1868  currentIndex += tokenLength;
1869  // If `tmpOutBuffer` is not empty, there exists a token before the
1870  // current token so insert a white space.
1871  if (!tmpOutBuffer.empty()) {
1872  currentIndex += 1;
1873  tmpOs << ' ';
1874  }
1875  tmpOutBuffer.insert(tmpOutBuffer.end(), it, next);
1876  }
1877 
1878  if (next == outBuffer.end())
1879  break;
1880  it = next + 1;
1881  }
1882  outBuffer = std::move(tmpOutBuffer);
1883 }

1898 /// Emit the specified value `exp` as a subexpression to the stream. The
1899 /// `parenthesizeIfLooserThan` parameter indicates when parentheses should be
1900 /// added aroun the subexpression. The `signReq` flag can cause emitSubExpr
1901 /// to emit a subexpression that is guaranteed to be signed or unsigned, and
1902 /// the `isSelfDeterminedUnsignedValue` flag indicates whether the value is
1903 /// known to be have "self determined" width, allowing us to omit extensions.
1904 SubExprInfo ExprEmitter::emitSubExpr(Value exp,
1905  VerilogPrecedence parenthesizeIfLooserThan,
1906  SubExprSignRequirement signRequirement,
1907  bool isSelfDeterminedUnsignedValue) {
1908  // If this is a self-determined unsigned value, look through any inline zero
1909  // extensions. This occurs on the RHS of a shift operation for example.
1910  if (isSelfDeterminedUnsignedValue && exp.hasOneUse()) {
1911  if (auto smaller = isZeroExtension(exp))
1912  exp = smaller;
1913  }
1914 
1915  auto *op = exp.getDefiningOp();
1916  bool shouldEmitInlineExpr = op && isVerilogExpression(op);
1917 
1918  // Don't emit this expression inline if it has multiple uses.
1919  if (shouldEmitInlineExpr && parenthesizeIfLooserThan != ForceEmitMultiUse &&
1920  emitter.outOfLineExpressions.count(op))
1921  shouldEmitInlineExpr = false;
1922 
1923  // If this is a non-expr or shouldn't be done inline, just refer to its name.
1924  if (!shouldEmitInlineExpr) {
1925  // All wires are declared as unsigned, so if the client needed it signed,
1926  // emit a conversion.
1927  if (signRequirement == RequireSigned) {
1928  os << "$signed(" << names.getName(exp) << ')';
1929  return {Symbol, IsSigned};
1930  }
1931 
1932  os << names.getName(exp);
1933  return {Symbol, IsUnsigned};
1934  }
1935 
1936  unsigned subExprStartIndex = outBuffer.size();
1937 
1938  // Inform the visit method about the preferred sign we want from the result.
1939  // It may choose to ignore this, but some emitters can change behavior based
1940  // on contextual desired sign.
1941  signPreference = signRequirement;
1942 
1943  bool bitCastAdded = false;
1944  if (state.options.explicitBitcast && isa<AddOp, MulOp, SubOp>(op))
1945  if (auto inType =
1946  (op->getResult(0).getType().dyn_cast_or_null<IntegerType>())) {
1947  os << inType.getWidth() << "'(";
1948  bitCastAdded = true;
1949  }
1950  // Okay, this is an expression we should emit inline. Do this through our
1951  // visitor.
1952  auto expInfo = dispatchCombinationalVisitor(exp.getDefiningOp());
1953 
1954  // Check cases where we have to insert things before the expression now that
1955  // we know things about it.
1956  auto addPrefix = [&](StringRef prefix) {
1957  outBuffer.insert(outBuffer.begin() + subExprStartIndex, prefix.begin(),
1958  prefix.end());
1959  };
1960  if (signRequirement == RequireSigned && expInfo.signedness == IsUnsigned) {
1961  addPrefix("$signed(");
1962  os << ')';
1963  expInfo.signedness = IsSigned;
1964  expInfo.precedence = Selection;
1965  } else if (signRequirement == RequireUnsigned &&
1966  expInfo.signedness == IsSigned) {
1967  addPrefix("$unsigned(");
1968  os << ')';
1969  expInfo.signedness = IsUnsigned;
1970  expInfo.precedence = Selection;
1971  } else if (expInfo.precedence > parenthesizeIfLooserThan) {
1972  // If this subexpression would bind looser than the expression it is bound
1973  // into, then we need to parenthesize it. Insert the parentheses
1974  // retroactively.
1975  addPrefix("(");
1976  os << ')';
1977  // Reset the precedence to the () level.
1978  expInfo.precedence = Selection;
1979  }
1980  if (bitCastAdded) {
1981  os << ')';
1982  }
1983 
1984  // Remember that we emitted this.
1985  emittedExprs.insert(exp.getDefiningOp());
1986  return expInfo;
1987 }

2365 void NameCollector::collectNames(Block &block) {
2366  bool isBlockProcedural = block.getParentOp()->hasTrait<ProceduralRegion>();
2367 
2368  SmallString<32> nameTmp;
2369 
2370  // Pre-pass loop to first add any names that could be the result of re-naming.
2371  // These constructs will have their names added regardless, and handling them
2372  // first ensures any out of line expressions won't trample on names selected
2373  // by re-naming. This could be combined into one pass through the IR that
2374  // collects a worklist of exprs to re-visit instead of the double traversal.
2375  for (auto &op : block) {
2376  if (auto instance = dyn_cast<InstanceOp>(op)) {
2377  names.addName(&op, getSymOpName(instance));
2378  continue;
2379  }
2380 
2381  if (auto interface = dyn_cast<InterfaceInstanceOp>(op)) {
2382  names.addName(interface.getResult(), getSymOpName(interface));
2383  continue;
2384  }
2385 
2386  if (isa<WireOp, RegOp, LogicOp, LocalParamOp>(op)) {
2387  names.addName(op.getResult(0), getSymOpName(&op));
2388  continue;
2389  }
2390  }
2391 
2392  // Loop over all of the results of all of the ops. Anything that defines a
2393  // value needs to be noticed.
2394  for (auto &op : block) {
2395  // Instances have an instance name to recognize but we don't need to look
2396  // at the result values and don't need to schedule them as valuesToEmit.
2397  // They already had their names added in the first loop, and can be skipped.
2398  if (isa<InstanceOp, InterfaceInstanceOp>(op))
2399  continue;
2400 
2401  bool isExpr = isVerilogExpression(&op);
2402  bool isInlineExpr = isExpr && isExpressionEmittedInline(&op);
2403  for (auto result : op.getResults()) {
2404  // If this is an expression emitted inline or unused, it doesn't need a
2405  // name.
2406  if (isExpr) {
2407  // If this expression is dead, or can be emitted inline, ignore it.
2408  if (result.use_empty() || isInlineExpr)
2409  continue;
2410 
2411  // Remember that this expression should be emitted out of line.
2412  moduleEmitter.outOfLineExpressions.insert(&op);
2413 
2414  // Get an explicitly set name or try to infer a name from the structure
2415  // of the expression.
2416  names.addName(result,
2417  moduleEmitter.inferStructuralNameForTemporary(result));
2418 
2419  // Don't measure or emit wires that are emitted inline (i.e. the wire
2420  // definition is emitted on the line of the expression instead of a
2421  // block at the top of the module).
2422  // Procedural blocks always emit out of line variable declarations,
2423  // because Verilog requires that they all be at the top of a block.

3702 /// Given an operation corresponding to a VerilogExpression, determine whether
3703 /// it is safe to emit inline into a 'localparam' or 'automatic logic' varaible
3704 /// initializer in a procedural region.
3705 ///
3706 /// We can't emit exprs inline when they refer to something else that can't be
3707 /// emitted inline, when they're in a general #ifdef region,
3708 static bool
3709 isExpressionEmittedInlineIntoProceduralDeclaration(Operation *op,
3710  StmtEmitter &stmtEmitter) {
3711  if (!isVerilogExpression(op))
3712  return false;
3713 
3714  // If the expression exists in an #ifdef region, then bail. Emitting it
3715  // inline would cause it to be executed unconditionally, because the
3716  // declarations are outside the #ifdef.
3717  if (isa<IfDefProceduralOp>(op->getParentOp()))
3718  return false;
3719 
3720  // This expression tree can be emitted into the initializer if all leaf
3721  // references are safe to refer to from here. They are only safe if they are
3722  // defined in an enclosing scope (guaranteed to already be live by now) or if
3723  // they are defined in this block and already emitted to an inline automatic
3724  // logic variable.
3725  SmallVector<Value, 8> exprsToScan(op->getOperands());
3726 
3727  // This loop is guaranteed to terminate because we're only scanning up
3728  // single-use expressions and other things that 'isExpressionEmittedInline'
3729  // returns success for. Cycles won't get in here.
3730  while (!exprsToScan.empty()) {
3731  Operation *expr = exprsToScan.pop_back_val().getDefiningOp();
3732  if (!expr)
3733  continue; // Ports are always safe to reference.
3734 
3735  // If this is an internal node in the expression tree, process its operands.
3736  if (isExpressionEmittedInline(expr)) {
3737  exprsToScan.append(expr->getOperands().begin(),
3738  expr->getOperands().end());
3739  continue;
3740  }
3741 
3742  // Otherwise, this isn't an inlinable expression. If it is defined outside
3743  // this block, then it is live-in.
3744  if (expr->getBlock() != op->getBlock())
3745  continue;

3747  // Otherwise, if it is defined in this block then it is only ok to reference
3748  // if it has already been emitted into an automatic logic.
3749  if (!stmtEmitter.emitter.expressionsEmittedIntoDecl.count(expr))
3750  return false;
3751  }
3752

4534 /// Actually emit the collected list of operations and strings to the
4535 /// specified file.
4536 void SharedEmitterState::emitOps(EmissionList &thingsToEmit, raw_ostream &os,
4537  bool parallelize) {
4538  MLIRContext *context = designOp->getContext();
4539 
4540  // Disable parallelization overhead if MLIR threading is disabled.
4541  if (parallelize)
4542  parallelize &= context->isMultithreadingEnabled();
4543 
4544  // If we aren't parallelizing output, directly output each operation to the
4545  // specified stream.
4546  if (!parallelize) {
4547  VerilogEmitterState state(designOp, *this, options, symbolCache,
4548  globalNames, os);
4549  for (auto &entry : thingsToEmit) {
4550  if (auto *op = entry.getOperation())
4551  emitOperation(state, op);
4552  else
4553  os << entry.getStringData();
4554  }
4555 
4556  if (state.encounteredError)
4557  encounteredError = true;
4558  return;
4559  }
4560 
4561  // If we are parallelizing emission, we emit each independent operation to a
4562  // string buffer in parallel, then concat at the end.
4563  parallelForEach(context, thingsToEmit, [&](StringOrOpToEmit &stringOrOp) {
4564  auto *op = stringOrOp.getOperation();
4565  if (!op)
4566  return; // Ignore things that are already strings.
4567 
4568  // BindOp emission reaches into the hw.module of the instance, and that
4569  // body may be being transformed by its own emission. Defer their
4570  // emission to the serial phase. They are speedy to emit anyway.
4571  if (isa<BindOp>(op) || modulesContainingBinds.count(op))
4572  return;
4573 
4574  SmallString<256> buffer;
4575  llvm::raw_svector_ostream tmpStream(buffer);
4576  VerilogEmitterState state(designOp, *this, options, symbolCache,
4577  globalNames, tmpStream);
4578  emitOperation(state, op);
4579  stringOrOp.setString(buffer);
4580  });
4581 
4582  // Finally emit each entry now that we know it is a string.
4583  for (auto &entry : thingsToEmit) {
4584  // Almost everything is lowered to a string, just concat the strings onto
4585  // the output stream.
4586  auto *op = entry.getOperation();
4587  if (!op) {
4588  os << entry.getStringData();
4589  continue;
4590  }
4591 
4592  // If this wasn't emitted to a string (e.g. it is a bind) do so now.
4593  VerilogEmitterState state(designOp, *this, options, symbolCache,
4594  globalNames, os);
4595  emitOperation(state, op);
4596  }
4597 }
