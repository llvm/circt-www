58 /// This function collects all the firrtl.mem ops and creates a verbatim op with
59 /// the relevant memory attributes.
60 LogicalResult CreateSiFiveMetadataPass::emitMemoryMetadata() {
61  if (!replSeqMem)
62  return success();
63 
64  CircuitOp circuitOp = getOperation();
65  // The instance graph analysis will be required to print the hierarchy names
66  // of the memory.
67  auto instancePathCache = InstancePathCache(getAnalysis<InstanceGraph>());
68 
69  // This lambda, writes to the given Json stream all the relevant memory
70  // attributes. Also adds the memory attrbutes to the string for creating the
71  // memmory conf file.
72  auto createMemMetadata = [&](FMemModuleOp mem,
73  llvm::json::OStream &jsonStream,
74  std::string &seqMemConfStr) {
75  // Get the memory data width.
76  auto width = mem.dataWidth();
77  // Metadata needs to be printed for memories which are candidates for
78  // macro replacement. The requirements for macro replacement::
79  // 1. read latency and write latency of one.
80  // 2. only one readwrite port or write port.
81  // 3. zero or one read port.
82  // 4. undefined read-under-write behavior.
83  if (!((mem.readLatency() == 1 && mem.writeLatency() == 1) &&
84  (mem.numWritePorts() + mem.numReadWritePorts() == 1) &&
85  (mem.numReadPorts() <= 1) && width > 0))
86  return;
87 
88  // Compute the mask granularity.
89  auto isMasked = mem.isMasked();
90  auto maskGran = width / mem.maskBits();
91  // Now create the config string for the memory.
92  std::string portStr;
93  if (mem.numWritePorts() && isMasked)
94  portStr += "mwrite";
95  else if (mem.numWritePorts())
96  portStr += "write";
97  if (mem.numReadPorts()) {
98  if (!portStr.empty())
99  portStr += ",";
100  portStr += "read";
101  }
102  if (mem.numReadWritePorts() && isMasked)
103  portStr = "mrw";
104  else if (mem.numReadWritePorts())
105  portStr = "rw";
106  auto memExtName = mem.getName();
107  auto maskGranStr =
108  !isMasked ? "" : " mask_gran " + std::to_string(maskGran);
109  seqMemConfStr = (StringRef(seqMemConfStr) + "name " + memExtName +
110  " depth " + Twine(mem.depth()) + " width " + Twine(width) +
111  " ports " + portStr + maskGranStr + "\n")
112  .str();
113  // This adds a Json array element entry corresponding to this memory.
114  jsonStream.object([&] {
115  jsonStream.attribute("module_name", memExtName);
116  jsonStream.attribute("depth", (int64_t)mem.depth());
117  jsonStream.attribute("width", (int64_t)width);
118  jsonStream.attribute("masked", isMasked);
119  jsonStream.attribute("read", mem.numReadPorts() > 0);
120  jsonStream.attribute("write", mem.numWritePorts() > 0);
121  jsonStream.attribute("readwrite", mem.numReadWritePorts() > 0);
122  if (isMasked)
123  jsonStream.attribute("mask_granularity", (int64_t)maskGran);
124  jsonStream.attributeArray("extra_ports", [&] {
125  for (auto attr : mem.extraPorts()) {
126  jsonStream.object([&] {
127  auto port = attr.cast<DictionaryAttr>();
128  auto name = port.getAs<StringAttr>("name").getValue();
129  jsonStream.attribute("name", name);
130  auto direction = port.getAs<StringAttr>("direction").getValue();
131  jsonStream.attribute("direction", direction);
132  auto width = port.getAs<IntegerAttr>("width").getUInt();
133  jsonStream.attribute("width", width);
134  });
135  }
136  });
137  // Record all the hierarchy names.
138  SmallVector<std::string> hierNames;
139  jsonStream.attributeArray("hierarchy", [&] {
140  // Get the absolute path for the parent memory, to create the
141  // hierarchy names.
142  auto paths = instancePathCache.getAbsolutePaths(mem);
143  for (auto p : paths) {
144  if (p.empty())
145  continue;
146  const InstanceOp &inst = p.front();
147  std::string hierName =
148  inst->getParentOfType<FModuleOp>().getName().str();
149  for (InstanceOp inst : p) {
150  auto parentModule = inst->getParentOfType<FModuleOp>();
151  if (dutMod == parentModule)
152  hierName = parentModule.getName().str();
153  hierName = (Twine(hierName) + "." + inst.name()).str();
154  }
155  hierNames.push_back(hierName);
156  jsonStream.value(hierName);
157  }
158  });
159  });
160  };
161 
162  SmallVector<FMemModuleOp> dutMems;
163  SmallVector<FMemModuleOp> tbMems;
164  for (auto mod : circuitOp.getOps<FMemModuleOp>()) {
165  if (dutModuleSet.contains(mod))
166  dutMems.push_back(mod);
167  else
168  tbMems.push_back(mod);
169  }
170 
171  std::string testBenchJsonBuffer;
172  llvm::raw_string_ostream testBenchOs(testBenchJsonBuffer);
173  llvm::json::OStream testBenchJson(testBenchOs, 2);
174  std::string dutJsonBuffer;
175  llvm::raw_string_ostream dutOs(dutJsonBuffer);
176  llvm::json::OStream dutJson(dutOs, 2);
177 
178  std::string seqMemConfStr;
179  dutJson.array([&] {
180  for (auto &dutM : dutMems)
181  createMemMetadata(dutM, dutJson, seqMemConfStr);
182  });
183  testBenchJson.array([&] {
184  // The tbConfStr is populated here, but unused, it will not be printed to
185  // file.
186  for (auto &tbM : tbMems)
187  createMemMetadata(tbM, testBenchJson, seqMemConfStr);
188  });
189 
190  auto *context = &getContext();
191  auto builder = OpBuilder::atBlockEnd(circuitOp.getBody());
192  AnnotationSet annos(circuitOp);
193  auto dirAnno = annos.getAnnotation(metadataDirectoryAnnoClass);
194  StringRef metadataDir = "metadata";
195  if (dirAnno)
196  if (auto dir = dirAnno.getMember<StringAttr>("dirname"))
197  metadataDir = dir.getValue();
198 
199  // Use unknown loc to avoid printing the location in the metadata files.
200  auto tbVerbatimOp = builder.create<sv::VerbatimOp>(builder.getUnknownLoc(),
201  testBenchJsonBuffer);
202  auto fileAttr = hw::OutputFileAttr::getFromDirectoryAndFilename(
203  context, metadataDir, "tb_seq_mems.json", /*excludeFromFilelist=*/true);
204  tbVerbatimOp->setAttr("output_file", fileAttr);
205  auto dutVerbatimOp =
206  builder.create<sv::VerbatimOp>(builder.getUnknownLoc(), dutJsonBuffer);
207  fileAttr = hw::OutputFileAttr::getFromDirectoryAndFilename(
208  context, metadataDir, "seq_mems.json", /*excludeFromFilelist=*/true);
209  dutVerbatimOp->setAttr("output_file", fileAttr);
210 
211  auto confVerbatimOp =
212  builder.create<sv::VerbatimOp>(builder.getUnknownLoc(), seqMemConfStr);
213  if (replSeqMemFile.empty()) {
214  circuitOp->emitError("metadata emission failed, the option "
215  "`-repl-seq-mem-file=<filename>` is mandatory for "
216  "specifying a valid seq mem metadata file");
217  return failure();
218  }
219 
220  fileAttr = hw::OutputFileAttr::getFromFilename(context, replSeqMemFile,
221  /*excludeFromFilelist=*/true);
222  confVerbatimOp->setAttr("output_file", fileAttr);
223 
224  return success();
225 }

227 /// This will search for a target annotation and remove it from the operation.
228 /// If the annotation has a filename, it will be returned in the output
229 /// argument. If the annotation is missing the filename member, or if more than
230 /// one matching annotation is attached, it will print an error and return
231 /// failure.
232 static LogicalResult removeAnnotationWithFilename(Operation *op,
233  StringRef annoClass,
234  StringRef &filename) {
235  filename = "";
236  bool error = false;
237  AnnotationSet::removeAnnotations(op, [&](Annotation anno) {
238  // If there was a previous error or its not a match, continue.
239  if (error || !anno.isClass(annoClass))
240  return false;
241 
242  // If we have already found a matching annotation, error.
243  if (!filename.empty()) {
244  op->emitError("more than one ") << annoClass << " annotation attached";
245  error = true;
246  return false;
247  }
248 
249  // Get the filename from the annotation.
250  auto filenameAttr = anno.getMember<StringAttr>("filename");
251  if (!filenameAttr) {
252  op->emitError(annoClass) << " requires a filename";
253  error = true;
254  return false;
255  }
256 
257  // Require a non-empty filename.
258  filename = filenameAttr.getValue();
259  if (filename.empty()) {
260  op->emitError(annoClass) << " requires a non-empty filename";
261  error = true;
262  return false;
263  }
264 
265  return true;
266  });
267 
268  // If there was a problem above, return failure.
269  return failure(error);
270 }

272 /// This function collects the name of each module annotated and prints them
273 /// all as a JSON array.
274 LogicalResult CreateSiFiveMetadataPass::emitRetimeModulesMetadata() {
275 
276  // Circuit level annotation.
277  auto *outputFileNameAnnotation =
278  "sifive.enterprise.firrtl.RetimeModulesAnnotation";
279  // Per module annotation.
280  auto *retimeModuleAnnoClass =
281  "freechips.rocketchip.util.RetimeModuleAnnotation";
282 
283  auto *context = &getContext();
284  auto circuitOp = getOperation();
285 
286  // Get the filename, removing the annotation from the circuit.
287  StringRef filename;
288  if (failed(removeAnnotationWithFilename(circuitOp, outputFileNameAnnotation,
289  filename)))
290  return failure();
291 
292  if (filename.empty())
293  return success();
294 
295  // Create a string buffer for the json data.
296  std::string buffer;
297  llvm::raw_string_ostream os(buffer);
298  llvm::json::OStream j(os, 2);
299 
300  // The output is a json array with each element a module name.
301  unsigned index = 0;
302  SmallVector<Attribute> symbols;
303  SmallString<3> placeholder;
304  j.array([&] {
305  for (auto module : circuitOp.getBody()->getOps<FModuleLike>()) {
306  // The annotation has no supplemental information, just remove it.
307  if (!AnnotationSet::removeAnnotations(module, retimeModuleAnnoClass))
308  continue;
309 
310  // We use symbol substitution to make sure we output the correct thing
311  // when the module goes through renaming.
312  j.value(("{{" + Twine(index++) + "}}").str());
313  symbols.push_back(SymbolRefAttr::get(module.moduleNameAttr()));
314  }
315  });
316 
317  // Put the retime information in a verbatim operation.
318  auto builder = OpBuilder::atBlockEnd(circuitOp.getBody());
319  auto verbatimOp = builder.create<sv::VerbatimOp>(
320  builder.getUnknownLoc(), buffer, ValueRange(),
321  builder.getArrayAttr(symbols));
322  auto fileAttr = hw::OutputFileAttr::getFromFilename(
323  context, filename, /*excludeFromFilelist=*/true);
324  verbatimOp->setAttr("output_file", fileAttr);
325  return success();
326 }

328 /// This function finds all external modules which will need to be generated for
329 /// the test harness to run.
330 LogicalResult CreateSiFiveMetadataPass::emitSitestBlackboxMetadata() {
331  auto *dutBlackboxAnnoClass =
332  "sifive.enterprise.firrtl.SitestBlackBoxAnnotation";
333  auto *testBlackboxAnnoClass =
334  "sifive.enterprise.firrtl.SitestTestHarnessBlackBoxAnnotation";
335 
336  // Any extmodule with these annotations or one of these ScalaClass classes
337  // should be excluded from the blackbox list.
338  auto *scalaClassAnnoClass = "sifive.enterprise.firrtl.ScalaClassAnnotation";
339  std::array<StringRef, 3> classBlackList = {
340  "freechips.rocketchip.util.BlackBoxedROM", "chisel3.shim.CloneModule",
341  "sifive.enterprise.grandcentral.MemTap"};
342  std::array<StringRef, 6> blackListedAnnos = {
343  "firrtl.transforms.BlackBox",
344  blackBoxInlineAnnoClass,
345  blackBoxPathAnnoClass,
346  dataTapsBlackboxClass,
347  memTapBlackboxClass,
348  "sifive.enterprise.grandcentral.transforms.SignalMappingAnnotation"};
349 
350  auto *context = &getContext();
351  auto circuitOp = getOperation();
352 
353  // Get the filenames from the annotations.
354  StringRef dutFilename, testFilename;
355  if (failed(removeAnnotationWithFilename(circuitOp, dutBlackboxAnnoClass,
356  dutFilename)) ||
357  failed(removeAnnotationWithFilename(circuitOp, testBlackboxAnnoClass,
358  testFilename)))
359  return failure();
360 
361  // If we don't have either annotation, no need to run this pass.
362  if (dutFilename.empty() && testFilename.empty())
363  return success();
364 
365  // Find all extmodules in the circuit. Check if they are black-listed from
366  // being included in the list. If they are not, separate them into two groups
367  // depending on if theyre in the DUT or the test harness.
368  SmallVector<StringRef> dutModules;
369  SmallVector<StringRef> testModules;
370  for (auto extModule : circuitOp.getBody()->getOps<FExtModuleOp>()) {
371  // If the module doesn't have a defname, then we can't record it properly.
372  // Just skip it.
373  if (!extModule.defname())
374  continue;
375 
376  // If its a generated blackbox, skip it.
377  AnnotationSet annos(extModule);
378  if (llvm::any_of(blackListedAnnos, [&](auto blackListedAnno) {
379  return annos.hasAnnotation(blackListedAnno);
380  }))
381  continue;
382 
383  // If its a blacklisted scala class, skip it.
384  if (auto scalaAnno = annos.getAnnotation(scalaClassAnnoClass)) {
385  auto scalaClass = scalaAnno.getMember<StringAttr>("className");
386  if (scalaClass &&
387  llvm::is_contained(classBlackList, scalaClass.getValue()))
388  continue;
389  }
390 
391  // Record the defname of the module.
392  if (dutModuleSet.contains(extModule)) {
393  dutModules.push_back(*extModule.defname());
394  } else {
395  testModules.push_back(*extModule.defname());
396  }
397  }
398 
399  // This is a helper to create the verbatim output operation.
400  auto createOutput = [&](SmallVectorImpl<StringRef> &names,
401  StringRef filename) {
402  if (filename.empty())
403  return;
404 
405  // Sort and remove duplicates.
406  std::sort(names.begin(), names.end());
407  names.erase(std::unique(names.begin(), names.end()), names.end());
408 
409  // The output is a json array with each element a module name. The
410  // defname of a module can't change so we can output them verbatim.
411  std::string buffer;
412  llvm::raw_string_ostream os(buffer);
413  llvm::json::OStream j(os, 2);
414  j.array([&] {
415  for (auto &name : names)
416  j.value(name);
417  });
418 
419  auto *body = circuitOp.getBody();
420  // Put the information in a verbatim operation.
421  auto builder = OpBuilder::atBlockEnd(body);
422  auto verbatimOp =
423  builder.create<sv::VerbatimOp>(builder.getUnknownLoc(), buffer);
424  auto fileAttr = hw::OutputFileAttr::getFromFilename(
425  context, filename, /*excludeFromFilelist=*/true);
426  verbatimOp->setAttr("output_file", fileAttr);
427  };
428 
429  createOutput(testModules, testFilename);
430  createOutput(dutModules, dutFilename);
431  return success();
432 }

440 void CreateSiFiveMetadataPass::runOnOperation() {
441  auto circuitOp = getOperation();
442  auto *body = circuitOp.getBody();
443 
444  // Find the device under test and create a set of all modules underneath it.
445  auto it = llvm::find_if(*body, [&](Operation &op) -> bool {
446  return AnnotationSet(&op).hasAnnotation(dutAnnoClass);
447  });
448  if (it != body->end()) {
449  dutMod = dyn_cast<FModuleOp>(*it);
450  auto &instanceGraph = getAnalysis<InstanceGraph>();
451  auto *node = instanceGraph.lookup(&(*it));
452  llvm::for_each(llvm::depth_first(node), [&](hw::InstanceGraphNode *node) {
453  dutModuleSet.insert(node->getModule());
454  });
455  }
456 
457  if (failed(emitRetimeModulesMetadata()) ||
458  failed(emitSitestBlackboxMetadata()) || failed(emitMemoryMetadata()))
459  return signalPassFailure();
460 
461  // This pass does not modify the hierarchy.
462  markAnalysesPreserved<InstanceGraph>();
463 }
