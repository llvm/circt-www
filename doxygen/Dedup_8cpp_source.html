128  void update(DictionaryAttr dict) {
129  for (auto namedAttr : dict) {
130  auto name = namedAttr.getName();
131  auto value = namedAttr.getValue();
132  // Skip names and annotations.
133  if (nonessentialAttributes.contains(name))
134  continue;
135  // Hash the port types.
136  if (name == portTypesAttr) {
137  auto portTypes = value.cast<ArrayAttr>().getAsValueRange<TypeAttr>();
138  for (auto type : portTypes)
139  update(type);
140  continue;
141  }
142  // Hash the interned pointer.
143  update(name.getAsOpaquePointer());
144  update(value.getAsOpaquePointer());
145  }
146  }

148  void update(Block &block) {
149  // Hash the block arguments.
150  for (auto arg : block.getArguments())
151  update(arg);
152  // Hash the operations in the block.
153  for (auto &op : block)
154  update(&op);
155  }

162  // NOLINTNEXTLINE(misc-no-recursion)
163  void update(Operation *op) {
164  update(op->getName());
165  update(op->getAttrDictionary());
166  // Hash the operands.
167  for (auto &operand : op->getOpOperands())
168  update(operand);
169  // Hash the regions. We need to make sure an empty region doesn't hash the
170  // same as no region, so we include the number of regions.
171  update(op->getNumRegions());
172  for (auto &region : op->getRegions())
173  for (auto &block : region.getBlocks())
174  update(block);
175  // Record any op results.
176  for (auto result : op->getResults())
177  update(result);
178  }
179 
180  // Every value is assigned a unique id based on their order of appearance.
181  unsigned currentIndex = 0;

186  // This is a cached "portTypes" string attr.
187  StringAttr portTypesAttr;
188 
189  // This is the actual running hash calculation. This is a stateful element
190  // that should be reinitialized after each hash is produced.
191  llvm::SHA256 sha;

258  LogicalResult check(InFlightDiagnostic &diag, BlockAndValueMapping &map,
259  Operation *a, Block &aBlock, Operation *b,
260  Block &bBlock) {
261 
262  // Block Arguments.
263  if (aBlock.getNumArguments() != bBlock.getNumArguments()) {
264  diag.attachNote(a->getLoc())
265  << "modules have a different number of ports";
266  diag.attachNote(b->getLoc()) << "second module here";
267  return failure();
268  }
269 
270  // Block argument types.
271  auto portNames = a->getAttrOfType<ArrayAttr>("portNames");
272  auto portNo = 0;
273  for (auto argPair :
274  llvm::zip(aBlock.getArguments(), bBlock.getArguments())) {
275  auto &aArg = std::get<0>(argPair);
276  auto &bArg = std::get<1>(argPair);
277  // TODO: we should print the port number if there are no port names, but
278  // there are always port names ;).
279  StringRef portName;
280  if (portNames) {
281  if (auto portNameAttr = portNames[portNo].dyn_cast<StringAttr>())
282  portName = portNameAttr.getValue();
283  }
284  // Assumption here that block arguments correspond to ports.
285  if (failed(check(diag, "module port '" + portName + "'", a,
286  aArg.getType(), b, bArg.getType())))
287  return failure();
288  map.map(aArg, bArg);
289  portNo++;
290  }
291 
292  // Blocks operations.
293  auto aIt = aBlock.begin();
294  auto aEnd = aBlock.end();
295  auto bIt = bBlock.begin();
296  auto bEnd = bBlock.end();
297  while (aIt != aEnd && bIt != bEnd)
298  if (failed(check(diag, map, &*aIt++, &*bIt++)))
299  return failure();
300  if (aIt != aEnd) {
301  diag.attachNote(aIt->getLoc()) << "first block has more operations";
302  diag.attachNote(b->getLoc()) << "second block here";
303  return failure();
304  }
305  if (bIt != bEnd) {
306  diag.attachNote(bIt->getLoc()) << "second block has more operations";
307  diag.attachNote(a->getLoc()) << "first block here";
308  return failure();
309  }
310  return success();
311  }

360  LogicalResult check(InFlightDiagnostic &diag, BlockAndValueMapping &map,
361  Operation *a, DictionaryAttr aDict, Operation *b,
362  DictionaryAttr bDict) {
363  // Fast path.
364  if (aDict == bDict)
365  return success();
366 
367  DenseSet<Attribute> seenAttrs;
368  for (auto namedAttr : aDict) {
369  auto attrName = namedAttr.getName();
370  if (nonessentialAttributes.contains(attrName))
371  continue;
372 
373  auto aAttr = namedAttr.getValue();
374  auto bAttr = bDict.get(attrName);
375  if (!bAttr) {
376  diag.attachNote(a->getLoc())
377  << "second operation is missing attribute " << attrName;
378  diag.attachNote(b->getLoc()) << "second operation here";
379  return diag;
380  }
381 
382  if (attrName == "portDirections") {
383  // Special handling for the port directions attribute for better
384  // error messages.
385  if (failed(check(diag, a, aAttr.cast<IntegerAttr>(), b,
386  bAttr.cast<IntegerAttr>())))
387  return failure();
388  } else if (aAttr != bAttr) {
389  diag.attachNote(a->getLoc())
390  << "first operation has attribute '" << attrName.getValue()
391  << "' with value " << aAttr;
392  diag.attachNote(b->getLoc()) << "second operation has value " << bAttr;
393  return failure();
394  }
395  seenAttrs.insert(attrName);
396  }
397  if (aDict.getValue().size() != bDict.getValue().size()) {
398  for (auto namedAttr : bDict) {
399  auto attrName = namedAttr.getName();
400  // Skip the attribute if we don't care about this particular one or it
401  // is one that is known to be in both dictionaries.
402  if (nonessentialAttributes.contains(attrName) ||
403  seenAttrs.contains(attrName))
404  continue;
405  // We have found an attribute that is only in the second operation.
406  diag.attachNote(a->getLoc())
407  << "first operation is missing attribute " << attrName;
408  diag.attachNote(b->getLoc()) << "second operation here";
409  return failure();
410  }
411  }
412  return success();
413  }

559  /// Remove the "fromModule", and replace all references to it with the
560  /// "toModule". Modules should be deduplicated in a bottom-up order. Any
561  /// module which is not deduplicated needs to be recorded with the `record`
562  /// call.
563  void dedup(FModuleLike toModule, FModuleLike fromModule) {
564  // A map of operation (e.g. wires, nodes) names which are changed, which is
565  // used to update NLAs that reference the "fromModule".
566  RenameMap renameMap;
567 
568  // This is a lazily constructed set of NLAs used to turn a local
569  // annotation into non-local annotations.
570  SmallVector<FlatSymbolRefAttr> toNLAs;
571  SmallVector<FlatSymbolRefAttr> fromNLAs;
572 
573  // Merge the two modules.
574  mergeOps(renameMap, toModule, toNLAs, toModule, fromModule, fromNLAs,
575  fromModule);
576 
577  // Rewrite NLAs pathing through these modules to refer to the to module.
578  if (auto to = dyn_cast<FModuleOp>(*toModule))
579  rewriteModuleNLAs(renameMap, to, cast<FModuleOp>(*fromModule));
580  else
581  rewriteExtModuleNLAs(renameMap, toModule.moduleNameAttr(),

586  /// Record the usages of any NLA's in this module, so that we may update the
587  /// annotation if the parent module is deduped with another module.
588  void record(FModuleLike module) {
589  // Record any annotations on the module.
590  recordAnnotations(module);
591  // Record port annotations.
592  for (const auto &pair : llvm::enumerate(module.getPortAnnotations()))
593  for (auto anno : AnnotationSet(pair.value().cast<ArrayAttr>()))
594  if (auto nlaRef = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal"))
595  targetMap[nlaRef.getAttr()].push_back(
596  PortAnnoTarget(module, pair.index()));
597  // Record any annotations in the module body.
598  module->walk([&](Operation *op) { recordAnnotations(op); });

716  /// Process all NLAs referencing the "from" module to point to the "to"
717  /// module. This is used after merging two modules together.
718  void addAnnotationContext(RenameMap &renameMap, FModuleOp toModule,
719  FModuleOp fromModule) {
720  auto toName = toModule.getNameAttr();
721  auto fromName = fromModule.getNameAttr();
722  // Create a copy of the current NLAs. We will be pushing and removing
723  // NLAs from this op as we go.
724  auto moduleNLAs = nlaTable->lookup(fromModule.getNameAttr()).vec();
725  // Change the NLA to target the toModule.
726  nlaTable->renameModuleAndInnerRef(toName, fromName, renameMap);
727  for (auto nla : moduleNLAs) {
728  auto elements = nla.namepath().getValue();
729  // If we don't need to add more context, we're done here.
730  if (nla.root() != toName)
731  continue;
732  // Create the replacement NLAs.
733  SmallVector<Attribute> namepath(elements.begin(), elements.end());
734  auto nlaRefs = createNLAs(toName, fromModule, namepath);
735  // Replace the uses of the old NLA with the new NLAs.
736  for (auto target : targetMap[nla.sym_nameAttr()]) {
737  // We have to clone any annotation which uses the old NLA for each new
738  // NLA. This array collects the new set of annotations.
739  SmallVector<Annotation> newAnnotations;
740  for (auto anno : target.getAnnotations()) {
741  // Find the non-local field of the annotation.
742  auto [it, found] = mlir::impl::findAttrSorted(
743  anno.begin(), anno.end(), nonLocalString);
744  // If this annotation doesn't use the target NLA, copy it with no
745  // changes.
746  if (!found || it->getValue().cast<FlatSymbolRefAttr>().getAttr() !=
747  nla.sym_nameAttr()) {
748  newAnnotations.push_back(anno);
749  continue;
750  }
751  auto nonLocalIndex = std::distance(anno.begin(), it);
752  // Clone the annotation and add it to the list of new annotations.
753  cloneAnnotation(nlaRefs, anno, ArrayRef(anno.begin(), anno.end()),
754  nonLocalIndex, newAnnotations);
755  }
756  // Apply the new annotations to the operation.
757  AnnotationSet annotations(newAnnotations, context);
758  target.setAnnotations(annotations);
759  // Record that target uses the NLA.
760  for (auto nla : nlaRefs)
761  targetMap[nla.getAttr()].push_back(target);
762  }
763 
764  // Erase the old NLA and remove it from all breadcrumbs.
765  eraseNLA(nla);
766  }
767  }

785  /// Take an annotation, and update it to be a non-local annotation. If the
786  /// annotation is already non-local and has enough context, it will be skipped
787  /// for now.
788  void makeAnnotationNonLocal(SmallVectorImpl<FlatSymbolRefAttr> &nlaRefs,
789  FModuleLike toModule, StringAttr toModuleName,
790  AnnoTarget to, FModuleLike fromModule,
791  AnnoTarget from, Annotation anno,
792  SmallVector<Annotation> &newAnnotations) {
793  // Start constructing a new annotation, pushing a "circt.nonLocal" field
794  // into the correct spot if its not already a non-local annotation.
795  SmallVector<NamedAttribute> attributes;
796  int nonLocalIndex = -1;
797  for (const auto &val : llvm::enumerate(anno)) {
798  auto attr = val.value();
799  // Is this field "circt.nonlocal"?
800  auto compare = attr.getName().compare(nonLocalString);
801  if (compare == 0) {
802  // This annotation is already a non-local annotation. Record that this
803  // operation uses that NLA and stop processing this annotation.
804  auto nlaName = attr.getValue().cast<FlatSymbolRefAttr>().getAttr();
805  targetMap[nlaName].push_back(from);
806  newAnnotations.push_back(anno);
807  return;
808  }
809  if (compare == 1) {
810  // Push an empty place holder for the non-local annotation.
811  nonLocalIndex = val.index();
812  attributes.push_back(NamedAttribute(nonLocalString, nonLocalString));
813  break;
814  }
815  attributes.push_back(attr);
816  }
817  if (nonLocalIndex == -1) {
818  // Push the "circt.nonlocal" to the last slot.
819  nonLocalIndex = attributes.size();
820  attributes.push_back(NamedAttribute(nonLocalString, nonLocalString));
821  } else {
822  // Copy the remaining annotation fields in.
823  attributes.append(anno.begin() + nonLocalIndex, anno.end());
824  }
825 
826  // Construct the NLAs if we don't have any yet.
827  if (nlaRefs.empty())
828  nlaRefs = createNLAs(toModule, toModuleName, fromModule);
829 
830  // Clone the annotation for each new NLA.
831  cloneAnnotation(nlaRefs, anno, attributes, nonLocalIndex, newAnnotations);
832  for (auto nla : nlaRefs)
833  targetMap[nla.getAttr()].push_back(to);
834  }

836  /// Merge the annotations of a specific target, either a operation or a port
837  /// on an operation.
838  void mergeAnnotations(FModuleLike toModule,
839  SmallVectorImpl<FlatSymbolRefAttr> &toNLAs,
840  AnnoTarget to, AnnotationSet toAnnos,
841  FModuleLike fromModule,
842  SmallVectorImpl<FlatSymbolRefAttr> &fromNLAs,
843  AnnoTarget from, AnnotationSet fromAnnos) {
844  // We want to make sure all NLAs are put right above the first module.
845  SmallVector<Annotation> newAnnotations;
846  SmallVector<unsigned> alreadyHandled;
847  for (auto anno : fromAnnos) {
848  // If the ops have the same annotation, we don't have to turn it into a
849  // non-local annotation.
850  auto found = llvm::find(toAnnos, anno);
851  if (found != toAnnos.end()) {
852  alreadyHandled.push_back(std::distance(toAnnos.begin(), found));
853  newAnnotations.push_back(anno);
854  continue;
855  }
856  makeAnnotationNonLocal(toNLAs, toModule, toModule.moduleNameAttr(), to,
857  fromModule, from, anno, newAnnotations);
858  }
859 
860  // This is a helper to skip already handled annotations.
861  auto *it = alreadyHandled.begin();
862  auto *end = alreadyHandled.end();
863  auto getNextHandledIndex = [&]() -> unsigned {
864  if (it == end)
865  return -1;
866  return *(it++);
867  };
868  auto index = getNextHandledIndex();
869 
870  // Merge annotations from the other op, skipping the ones already handled.
871  for (const auto &pair : llvm::enumerate(toAnnos)) {
872  // If its already handled, skip it.
873  if (pair.index() == index) {
874  index = getNextHandledIndex();
875  continue;
876  }
877  auto anno = pair.value();
878  makeAnnotationNonLocal(fromNLAs, toModule, toModule.moduleNameAttr(), to,
879  toModule, to, anno, newAnnotations);
880  }
881 
882  // Copy over all the new annotations.
883  if (!newAnnotations.empty())
884  to.setAnnotations(AnnotationSet(newAnnotations, context));
885  }

918  // merging two operations. The renamed symbols are used to update the

920  void recordSymRenames(RenameMap &renameMap, FModuleLike toModule,
921  Operation *to, FModuleLike fromModule,
922  Operation *from) {
923  // If the "from" operation has an inner_sym, we need to make sure the

925  if (auto fromSym = getInnerSymName(from)) {
926  auto toSym = OpAnnoTarget(to).getInnerSym(getNamespace(toModule));
927  renameMap[fromSym] = toSym;

930  // If there are no port symbols on the "from" operation, we are done here.
931  auto fromPortSyms = from->getAttrOfType<ArrayAttr>("portSyms");
932  if (!fromPortSyms || fromPortSyms.empty())
933  return;

935  auto &moduleNamespace = getNamespace(toModule);
936  auto portCount = fromPortSyms.size();
937  auto portNames = to->getAttrOfType<ArrayAttr>("portNames");
938  auto toPortSyms = to->getAttrOfType<ArrayAttr>("portSyms");

946  else
947  newPortSyms.assign(toPortSyms.begin(), toPortSyms.end());

950  // If this fromPort doesn't have a symbol, move on to the next one.

953  continue;
954

956  auto toSym = newPortSyms[portNo].cast<StringAttr>();
957  if (toSym == emptyString) {
958  // Get a reasonable base name for the port.
959  StringRef symName = "inner_sym";
960  if (portNames)
961  symName = portNames[portNo].cast<StringAttr>().getValue();

969  }
970 
971  // Commit the new symbol attribute.
972  to->setAttr("portSyms", ArrayAttr::get(context, newPortSyms));
973  }

1044 /// This fixes up connects when the field names of a bundle type changes. It
1045 /// finds all fields which were previously bulk connected and legalizes it
1046 /// into a connect for each field.
1047 template <typename T>
1048 void fixupConnect(ImplicitLocOpBuilder &builder, Value dst, Value src) {
1049  // If the types already match we can emit a connect.
1050  auto dstType = dst.getType();
1051  auto srcType = src.getType();
1052  if (dstType == srcType) {
1053  builder.create<T>(dst, src);
1054  return;
1055  }
1056  // It must be a bundle type and the field name has changed. We have to
1057  // manually decompose the bulk connect into a connect for each field.
1058  auto dstBundle = dstType.cast<BundleType>();
1059  auto srcBundle = srcType.cast<BundleType>();
1060  for (unsigned i = 0; i < dstBundle.getNumElements(); ++i) {
1061  auto dstField = builder.create<SubfieldOp>(dst, i);
1062  auto srcField = builder.create<SubfieldOp>(src, i);
1063  if (dstBundle.getElement(i).isFlip) {
1064  std::swap(srcBundle, dstBundle);
1065  std::swap(srcField, dstField);
1066  }
1067  fixupConnect<T>(builder, dstField, srcField);
1068  }
1069 }

1079 /// When we replace a bundle type with a similar bundle with different field
1080 /// names, we have to rewrite all the code to use the new field names. This
1081 /// mostly affects subfield result types and any bulk connects.
1082 void fixupReferences(Value oldValue, Type newType) {
1083  SmallVector<std::pair<Value, Type>> workList;
1084  workList.emplace_back(oldValue, newType);
1085  while (!workList.empty()) {
1086  auto [oldValue, newType] = workList.pop_back_val();
1087  auto oldType = oldValue.getType();
1088  // If the two types are identical, we don't need to do anything, otherwise
1089  // update the type in place.
1090  if (oldType == newType)
1091  continue;
1092  oldValue.setType(newType);
1093  for (auto *op : llvm::make_early_inc_range(oldValue.getUsers())) {
1094  if (auto subfield = dyn_cast<SubfieldOp>(op)) {
1095  // Rewrite a subfield op to return the correct type.
1096  auto index = subfield.fieldIndex();
1097  auto result = subfield.getResult();
1098  auto newResultType = newType.cast<BundleType>().getElementType(index);
1099  workList.emplace_back(result, newResultType);
1100  continue;
1101  }
1102  if (auto connect = dyn_cast<ConnectOp>(op)) {
1103  fixupConnect<ConnectOp>(connect);
1104  continue;
1105  }
1106  if (auto strict = dyn_cast<StrictConnectOp>(op)) {
1107  fixupConnect<StrictConnectOp>(strict);
1108  continue;
1109  }
1110  }
1111  }
