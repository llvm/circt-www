27 /// Helper to iterate through memory operation pairs and check for dependences
28 /// at a given loop nesting depth.
29 static void checkMemrefDependence(SmallVectorImpl<Operation *> &memoryOps,
30  unsigned depth,
31  MemoryDependenceResult &results) {
32  for (auto *source : memoryOps) {
33  for (auto *destination : memoryOps) {
34  if (source == destination)
35  continue;
36 
37  // Initialize the dependence list for this destination.
38  if (results.count(destination) == 0)
39  results[destination] = SmallVector<MemoryDependence>();
40 
41  // Look for for inter-iteration dependences on the same memory location.
42  MemRefAccess src(source);
43  MemRefAccess dst(destination);
44  FlatAffineValueConstraints dependenceConstraints;
45  SmallVector<DependenceComponent, 2> depComps;
46  DependenceResult result = checkMemrefAccessDependence(
47  src, dst, depth, &dependenceConstraints, &depComps, true);
48 
49  results[destination].emplace_back(source, result.value, depComps);
50 
51  // Also consider intra-iteration dependences on the same memory location.
52  // This currently does not consider aliasing.
53  if (src != dst)
54  continue;
55 
56  // Collect surrounding loops to use in dependence components. Only proceed
57  // if we are in the innermost loop.
58  SmallVector<AffineForOp> enclosingLoops;
59  getLoopIVs(*destination, &enclosingLoops);
60  if (enclosingLoops.size() != depth)
61  continue;
62 
63  // Look for the common parent that src and dst share. If there is none,
64  // there is nothing more to do.
65  SmallVector<Operation *> srcParents;
66  getEnclosingAffineForAndIfOps(*source, &srcParents);
67  SmallVector<Operation *> dstParents;
68  getEnclosingAffineForAndIfOps(*destination, &dstParents);
69 
70  Operation *commonParent = nullptr;
71  for (auto *srcParent : llvm::reverse(srcParents)) {
72  for (auto *dstParent : llvm::reverse(dstParents)) {
73  if (srcParent == dstParent)
74  commonParent = srcParent;
75  if (commonParent != nullptr)
76  break;
77  }
78  if (commonParent != nullptr)
79  break;
80  }
81 
82  if (commonParent == nullptr)
83  continue;
84 
85  // Check the common parent's regions.
86  for (auto &commonRegion : commonParent->getRegions()) {
87  if (commonRegion.empty())
88  continue;
89 
90  // Only support structured constructs with single-block regions for now.
91  assert(commonRegion.hasOneBlock() &&
92  "only single-block regions are supported");
93 
94  Block &commonBlock = commonRegion.front();
95 
96  // Find the src and dst ancestor in the common block, if any.
97  Operation *srcOrAncestor = commonBlock.findAncestorOpInBlock(*source);
98  Operation *dstOrAncestor =
99  commonBlock.findAncestorOpInBlock(*destination);
100  if (srcOrAncestor == nullptr || dstOrAncestor == nullptr)
101  continue;
102 
103  // Check if the src or its ancestor is before the dst or its ancestor.
104  if (srcOrAncestor->isBeforeInBlock(dstOrAncestor)) {
105  // Build dependence components for each loop depth.
106  SmallVector<DependenceComponent> intraDeps;
107  for (size_t i = 0; i < depth; ++i) {
108  DependenceComponent depComp;
109  depComp.op = enclosingLoops[i];
110  depComp.lb = 0;
111  depComp.ub = 0;
112  intraDeps.push_back(depComp);
113  }
114 
115  results[destination].emplace_back(
116  source, DependenceResult::HasDependence, intraDeps);
117  }
118  }
119  }
120  }
121 }

123 /// MemoryDependenceAnalysis traverses any AffineForOps in the FuncOp body and
124 /// checks for memory access dependences. Results are captured in a
125 /// MemoryDependenceResult, which can by queried by Operation.
126 circt::analysis::MemoryDependenceAnalysis::MemoryDependenceAnalysis(
127  Operation *op) {
128  auto funcOp = cast<func::FuncOp>(op);
129 
130  // Collect affine loops grouped by nesting depth.
131  std::vector<SmallVector<AffineForOp, 2>> depthToLoops;
132  mlir::gatherLoops(funcOp, depthToLoops);
133 
134  // Collect load and store operations to check.
135  SmallVector<Operation *> memoryOps;
136  funcOp.walk([&](Operation *op) {
137  if (isa<AffineReadOpInterface, AffineWriteOpInterface>(op))
138  memoryOps.push_back(op);
139  });
140 
141  // For each depth, check memref accesses.
142  for (unsigned depth = 1, e = depthToLoops.size(); depth <= e; ++depth)
143  checkMemrefDependence(memoryOps, depth, results);
144 }

152 /// Replaces the dependences, if any, from the oldOp to the newOp.
153 void circt::analysis::MemoryDependenceAnalysis::replaceOp(Operation *oldOp,
154  Operation *newOp) {
155  // If oldOp had any dependences.
156  auto it = results.find(oldOp);
157  if (it != results.end())
158  // Move the dependences to newOp.
159  it->first = newOp;
160 
161  // Find any dependences originating from oldOp and make newOp the source.
162  // TODO(mikeurbach): consider adding an inverted index to avoid this scan.
163  for (auto &it : results)
164  for (auto &dep : it.second)
165  if (dep.source == oldOp)
166  dep.source = newOp;
167 }
