31 /// MemoryDependence captures a dependence from one memory operation to another.
32 /// It represents the destination of the dependence edge, the type of the
33 /// dependence, and the components associated with each enclosing loop.
34 struct MemoryDependence {
35  MemoryDependence(Operation *source,
36  mlir::DependenceResult::ResultEnum dependenceType,
37  ArrayRef<mlir::DependenceComponent> dependenceComponents)
38  : source(source), dependenceType(dependenceType),
39  dependenceComponents(dependenceComponents.begin(),
40  dependenceComponents.end()) {}
41 
42  // The source Operation where this dependence originates.
43  Operation *source;
44 
45  // The dependence type denotes whether or not there is a dependence.
46  mlir::DependenceResult::ResultEnum dependenceType;
47 
48  // The dependence components include lower and upper bounds for each loop.
49  SmallVector<mlir::DependenceComponent> dependenceComponents;
50 };
51 
52 /// MemoryDependenceResult captures a set of memory dependences. The map key is
53 /// the operation to which the dependences exist, and the map value is zero or
54 /// more MemoryDependences for that operation.
55 using MemoryDependenceResult =
56  DenseMap<Operation *, SmallVector<MemoryDependence>>;

58 /// MemoryDependenceAnalysis traverses any AffineForOps in the FuncOp body and
59 /// checks for affine memory access dependences. Non-affine memory dependences
60 /// are currently not supported. Results are captured in a
61 /// MemoryDependenceResult, and an API is exposed to query dependences of a
62 /// given Operation.
63 /// TODO(mikeurbach): consider upstreaming this to MLIR's AffineAnalysis.
64 struct MemoryDependenceAnalysis {
65  // Construct the analysis from a FuncOp.
66  MemoryDependenceAnalysis(Operation *funcOp);
67 
68  // Returns the dependences, if any, that the given Operation depends on.
69  ArrayRef<MemoryDependence> getDependences(Operation *);
70 
71  // Replaces the dependences, if any, from the oldOp to the newOp.
72  void replaceOp(Operation *oldOp, Operation *newOp);
73
