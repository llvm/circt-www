23 /// The namespace of a `CircuitOp`, generally inhabited by modules.
24 struct CircuitNamespace : public Namespace {
25  CircuitNamespace() {}
26  CircuitNamespace(CircuitOp circuit) { add(circuit); }
27 
28  /// Populate the namespace from a circuit operation. This namespace will be
29  /// composed of any operation in the first level of the circuit that contains
30  /// a symbol.
31  void add(CircuitOp circuit) {
32  for (auto &op : *circuit.getBody())
33  if (auto symbol = op.getAttrOfType<mlir::StringAttr>(
34  SymbolTable::getSymbolAttrName()))
35  nextIndex.insert({symbol.getValue(), 0});
36  }
37 };
38 
39 /// The namespace of a `FModuleLike` operation, generally inhabited by its ports
40 /// and declarations.
41 struct ModuleNamespace : public Namespace {
42  ModuleNamespace() {}
43  ModuleNamespace(FModuleLike module) : module(module) { add(module); }
44 
45  /// Populate the namespace from a module-like operation. This namespace will
46  /// be composed of the `inner_sym`s of the module's ports and declarations.
47  void add(FModuleLike module) {
48  addPorts(module);
49  addBody(module);
50  }
51 
52  /// Populate the namespace with the ports of a module-like operation.
53  void addPorts(FModuleLike module) {
54  for (auto portSymbol : module.getPortSymbolsAttr().getAsRange<StringAttr>())
55  if (!portSymbol.getValue().empty())
56  nextIndex.insert({portSymbol.getValue(), 0});
57  }
58 
59  void addPorts(ArrayRef<PortInfo> ports) {
60  for (auto port : ports)
61  if (port.sym)
62  nextIndex.insert({port.sym.getValue(), 0});
63  }
64 
65  /// Populate the namespace with the body of a module-like operation.
66  void addBody(FModuleLike module) {
67  module.walk([&](Operation *op) {
68  auto attr = getInnerSymName(op);
69  if (attr)
70  nextIndex.insert({attr.getValue(), 0});
71  });
72  }
73 
74  /// The module associated with this namespace.
75  FModuleLike module;
76 };
