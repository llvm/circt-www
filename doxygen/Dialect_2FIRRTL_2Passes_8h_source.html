152 /// Configure which values will be explicitly preserved by the DropNames pass.
153 namespace PreserveValues {
154 enum PreserveMode {
155  /// Strip all names. No name on declaration is preserved.
156  Strip,
157  /// Don't explicitly preserve any named values. Every named operation could
158  /// be optimized away by the compiler. Unlike `Strip` names could be preserved
159  /// until the end.
160  None,
161  // Explicitly preserved values with meaningful names. If a name begins with
162  // an "_" it is not considered meaningful.
163  Named,
164  // Explicitly preserve all values. No named operation should be optimized
165  // away by the compiler.
166  All,
167 };
168 }
