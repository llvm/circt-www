27  /// Populate the namespace from a module-like operation. This namespace will
28  /// be composed of the `inner_sym`s of the module's ports and declarations.
29  void add(hw::HWModuleOp module) {
30  for (auto port : module.getAllPorts())
31  if (port.sym && !port.sym.getValue().empty())
32  nextIndex.insert({port.sym.getValue(), 0});
33  module.walk([&](Operation *op) {
34  auto attr = op->getAttrOfType<StringAttr>("inner_sym");
35  if (attr)
36  nextIndex.insert({attr.getValue(), 0});
37  });
38  }
39 };
