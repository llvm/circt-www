34 /// Emit the contents of 'msg' to the log file in hex.
35 static void log(int epId, bool toClient, Endpoint::BlobPtr msg) {
36  std::lock_guard<std::mutex> g(serverMutex);
37  if (!logFile)
38  return;
39 
40  fprintf(logFile, "[ep: %4x to: %4s]", epId, toClient ? "host" : "sim");
41  size_t msgSize = msg->size();
42  for (size_t i = 0; i < msgSize; ++i) {
43  auto b = (*msg)[i];
44  // Separate 32-bit words.
45  if (i % 4 == 0 && i > 0)
46  fprintf(logFile, " ");
47  // Separate 64-bit words (capnp word size)
48  if (i % 8 == 0 && i > 0)
49  fprintf(logFile, " ");
50  fprintf(logFile, " %02x", b);
51  }
52  fprintf(logFile, "\n");
53 }

67 /// Check that an array is an array of bytes and has some size.
68 // NOLINTNEXTLINE(misc-misplaced-const)
69 static int validateSvOpenArray(const svOpenArrayHandle data,
70  int expectedElemSize) {
71  if (svDimensions(data) != 1) {
72  printf("DPI-C: ERROR passed array argument that doesn't have expected 1D "
73  "dimensions\n");
74  return -1;
75  }
76  if (svGetArrayPtr(data) == NULL) {
77  printf("DPI-C: ERROR passed array argument that doesn't have C layout "
78  "(ptr==NULL)\n");
79  return -2;
80  }
81  int totalBytes = svSizeOfArray(data);
82  if (totalBytes == 0) {
83  printf("DPI-C: ERROR passed array argument that doesn't have C layout "
84  "(total_bytes==0)\n");
85  return -3;
86  }
87  int numElems = svSize(data, 1);
88  int elemSize = numElems == 0 ? 0 : (totalBytes / numElems);
89  if (numElems * expectedElemSize != totalBytes) {
90  printf("DPI-C: ERROR: passed array argument that doesn't have expected "
91  "element-size: expected=%d actual=%d numElems=%d totalBytes=%d\n",
92  expectedElemSize, elemSize, numElems, totalBytes);
93  return -4;
94  }
95  return 0;
96 }

114 // Attempt to recieve data from a client.
115 // - Returns negative when call failed (e.g. EP not registered).
116 // - If no message, return 0 with dataSize == 0.
117 // - Assumes buffer is large enough to contain entire message. Fails if not
118 // large enough. (In the future, will add support for getting the message
119 // into a fixed-size buffer over multiple calls.)
120 DPI int sv2cCosimserverEpTryGet(unsigned int endpointId,
121  // NOLINTNEXTLINE(misc-misplaced-const)
122  const svOpenArrayHandle data,
123  unsigned int *dataSize) {
124  if (server == nullptr)
125  return -1;
126 
127  Endpoint *ep = server->endpoints[endpointId];
128  if (!ep) {
129  fprintf(stderr, "Endpoint not found in registry!\n");
130  return -4;
131  }
132 
133  Endpoint::BlobPtr msg;
134  // Poll for a message.
135  if (!ep->getMessageToSim(msg)) {
136  // No message.
137  *dataSize = 0;
138  return 0;
139  }
140  // Do the validation only if there's a message available. Since the
141  // simulator is going to poll up to every tick and there's not going to be
142  // a message most of the time, this is important for performance.
143 
144  log(endpointId, false, msg);
145 
146  if (validateSvOpenArray(data, sizeof(int8_t)) != 0) {
147  printf("ERROR: DPI-func=%s line=%d event=invalid-sv-array\n", __func__,
148  __LINE__);
149  return -2;
150  }
151 
152  // Detect or verify size of buffer.
153  if (*dataSize == ~0u) {
154  *dataSize = svSizeOfArray(data);
155  } else if (*dataSize > (unsigned)svSizeOfArray(data)) {
156  printf("ERROR: DPI-func=%s line %d event=invalid-size (max %d)\n", __func__,
157  __LINE__, (unsigned)svSizeOfArray(data));
158  return -3;
159  }
160  // Verify it'll fit.
161  size_t msgSize = msg->size();
162  if (msgSize > *dataSize) {
163  printf("ERROR: Message size too big to fit in HW buffer\n");
164  return -5;
165  }
166 
167  // Copy the message data.
168  size_t i;
169  for (i = 0; i < msgSize; ++i) {
170  auto b = (*msg)[i];
171  *(char *)svGetArrElemPtr1(data, i) = b;
172  }
173  // Zero out the rest of the buffer.
174  for (; i < *dataSize; ++i) {
175  *(char *)svGetArrElemPtr1(data, i) = 0;
176  }
177  // Set the output data size.
178  *dataSize = msg->size();
179  return 0;
180 }

182 // Attempt to send data to a client.
183 // - return 0 on success, negative on failure (unregistered EP).
184 // - if dataSize is negative, attempt to dynamically determine the size of
185 // 'data'.
186 DPI int sv2cCosimserverEpTryPut(unsigned int endpointId,
187  // NOLINTNEXTLINE(misc-misplaced-const)
188  const svOpenArrayHandle data, int dataSize) {
189  if (server == nullptr)
190  return -1;
191 
192  if (validateSvOpenArray(data, sizeof(int8_t)) != 0) {
193  printf("ERROR: DPI-func=%s line=%d event=invalid-sv-array\n", __func__,
194  __LINE__);
195  return -2;
196  }
197 
198  // Detect or verify size.
199  if (dataSize < 0) {
200  dataSize = svSizeOfArray(data);
201  } else if (dataSize > svSizeOfArray(data)) { // not enough data
202  printf("ERROR: DPI-func=%s line %d event=invalid-size limit %d array %d\n",
203  __func__, __LINE__, dataSize, svSizeOfArray(data));
204  return -3;
205  }
206 
207  Endpoint::BlobPtr blob = std::make_shared<Endpoint::Blob>(dataSize);
208  // Copy the message data into 'blob'.
209  for (int i = 0; i < dataSize; ++i) {
210  (*blob)[i] = *(char *)svGetArrElemPtr1(data, i);
211  }
212  // Queue the blob.
213  Endpoint *ep = server->endpoints[endpointId];
214  if (!ep) {
215  fprintf(stderr, "Endpoint not found in registry!\n");
216  return -4;
217  }
218  log(endpointId, true, blob);
219  ep->pushMessageToClient(blob);
220  return 0;
221 }

238 // connections from new SW-clients).

240  std::lock_guard<std::mutex> g(serverMutex);
