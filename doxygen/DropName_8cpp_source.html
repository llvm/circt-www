22 struct DropNamesPass : public DropNameBase<DropNamesPass> {
23  DropNamesPass(PreserveValues::PreserveMode preserveMode) {
24  this->preserveMode = preserveMode;
25  }
26 
27  void runOnOperation() override {
28  if (preserveMode == PreserveValues::None) {
29  // Drop all names.
30  numNamesConverted += dropNamesIf([](FNamableOp) { return true; });
31  } else if (preserveMode == PreserveValues::Named) {
32  // Drop the name if it isn't considered meaningful.
33  numNamesConverted += dropNamesIf(
34  [](FNamableOp op) { return isUselessName(op.getName()); });
35  }
36  }
37 
38 private:
39  size_t dropNamesIf(llvm::function_ref<bool(FNamableOp)> pred) {
40  size_t changedNames = 0;
41  getOperation()->walk([&](FNamableOp op) {
42  if (pred(op)) {
43  ++changedNames;
44  op.dropName();
45  }
46  });
47  return changedNames;
48  }
49 };
