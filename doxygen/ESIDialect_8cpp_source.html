50 /// Try to find a valid/ready port for the specified data port. If found, append
51 /// to 'names'.
52 static void findValidReady(Operation *modOp,
53  const llvm::StringMap<hw::PortInfo> &nameMap,
54  hw::PortInfo dataPort, bool trimName, bool warn,
55  SmallVectorImpl<ESIPortValidReadyMapping> &names) {
56  if (dataPort.direction == hw::PortDirection::INOUT) {
57  if (warn)
58  modOp->emitWarning("Data port '")
59  << dataPort.getName() << "' cannot be inout direction.";
60  return;
61  }
62 
63  SmallString<64> name = dataPort.getName();
64  size_t nameLen = name.size();
65  if (trimName && name.endswith("_data")) { // Detect both `foo` and `foo_data`.
66  nameLen -= 5;
67  name.resize(nameLen);
68  }
69 
70  // Look for a 'valid' port.
71  name.append("_valid");
72  auto valid = nameMap.find(name);
73  if (valid == nameMap.end() || valid->second.direction != dataPort.direction ||
74  !valid->second.type.isSignlessInteger(1)) {
75  if (warn)
76  modOp->emitWarning("Could not find appropriate valid port for '")
77  << name << "'.";
78  return;
79  }
80 
81  // Try to find a corresponding 'ready' port.
82  name.resize(nameLen);
83  name.append("_ready");
84  hw::PortDirection readyDir = dataPort.direction == hw::PortDirection::INPUT
85  ? hw::PortDirection::OUTPUT
86  : hw::PortDirection::INPUT;
87  auto ready = nameMap.find(name);
88  if (ready == nameMap.end() || ready->second.direction != readyDir ||
89  !ready->second.type.isSignlessInteger(1)) {
90  if (warn)
91  modOp->emitWarning("Could not find appropriate ready port for '")
92  << name << "'.";
93  return;
94  }
95 
96  // Found one.
97  names.push_back(
98  ESIPortValidReadyMapping{dataPort, valid->second, ready->second});
99 }

101 /// Find all the port triples on a module which fit the
102 /// <name>[_data]/<name>_valid/<name>_ready pattern. Ready must be the opposite
103 /// direction of the other two.
104 void circt::esi::findValidReadySignals(
105  Operation *modOp, SmallVectorImpl<ESIPortValidReadyMapping> &names) {
106 
107  SmallVector<hw::PortInfo> ports = hw::getAllModulePortInfos(modOp);
108  llvm::StringMap<hw::PortInfo> nameMap(ports.size());
109  for (auto port : ports)
110  nameMap[port.getName()] = port;
111  for (auto port : ports)
112  findValidReady(modOp, nameMap, port, /*trimName=*/true, /*warn=*/false,
113  names);
114 }

116 /// Given a list of logical port names, find the data/valid/ready port triples.
117 void circt::esi::resolvePortNames(
118  Operation *modOp, ArrayRef<StringRef> portNames,
119  SmallVectorImpl<ESIPortValidReadyMapping> &names) {
120 
121  SmallVector<hw::PortInfo> ports = hw::getAllModulePortInfos(modOp);
122  llvm::StringMap<hw::PortInfo> nameMap(ports.size());
123  for (auto port : ports)
124  nameMap[port.getName()] = port;
125 
126  SmallString<64> nameBuffer;
127  for (auto name : portNames) {
128  nameBuffer = name;
129 
130  // Look for a 'data' port.
131  auto it = nameMap.find(nameBuffer);
132  if (it == nameMap.end()) {
133  nameBuffer.append("_data");
134  it = nameMap.find(nameBuffer);
135  if (it == nameMap.end())
136  modOp->emitWarning("Could not find data port '") << name << "'.";
137  else
138  findValidReady(modOp, nameMap, it->second, /*trimName=*/true,
139  /*warn=*/true, names);
140  } else {
141  findValidReady(modOp, nameMap, it->second, /*trimName=*/false,
142  /*warn=*/true, names);
143  }
144  }
145 }

147 /// Build an ESI module wrapper, converting the wires with latency-insensitive
148 /// semantics to ESI channels and passing through the rest.
149 Operation *
150 circt::esi::buildESIWrapper(OpBuilder &b, Operation *pearl,
151  ArrayRef<ESIPortValidReadyMapping> portsToConvert) {
152  // In order to avoid the similar sounding and looking "wrapped" and "wrapper"
153  // names or the ambiguous "module", we use "pearl" for the module _being
154  // wrapped_ and "shell" for the _wrapper_ modules which is being created
155  // (terms typically used in latency insensitive design papers).
156 
157  auto *ctxt = b.getContext();
158  Location loc = pearl->getLoc();
159  FunctionType modType = hw::getModuleType(pearl);
160 
161  // -----
162  // First, build up a set of data structures to use throughout this function.
163 
164  StringSet<> controlPorts; // Memoize the list of ready/valid ports to ignore.
165  llvm::StringMap<ESIPortValidReadyMapping>
166  dataPortMap; // Store a lookup table of ports to convert indexed on the
167  // data port name.
168  // Validate input and assemble lookup structures.
169  for (const auto &esiPort : portsToConvert) {
170  if (esiPort.data.direction == hw::PortDirection::INOUT) {
171  pearl->emitError("Data signal '")
172  << esiPort.data.name << "' must not be INOUT";
173  return nullptr;
174  }
175  dataPortMap[esiPort.data.name.getValue()] = esiPort;
176 
177  if (esiPort.valid.direction != esiPort.data.direction) {
178  pearl->emitError("Valid port '")
179  << esiPort.valid.name << "' direction must match data port.";
180  return nullptr;
181  }
182  if (esiPort.valid.type != b.getI1Type()) {
183  pearl->emitError("Valid signal '")
184  << esiPort.valid.name << "' must be i1 type";
185  return nullptr;
186  }
187  controlPorts.insert(esiPort.valid.name.getValue());
188 
189  if (esiPort.ready.direction != (esiPort.data.isOutput()
190  ? hw::PortDirection::INPUT
191  : hw::PortDirection::OUTPUT)) {
192  pearl->emitError("Ready port '")
193  << esiPort.ready.name
194  << "' must be opposite direction to data signal.";
195  return nullptr;
196  }
197  if (esiPort.ready.type != b.getI1Type()) {
198  pearl->emitError("Ready signal '")
199  << esiPort.ready.name << "' must be i1 type";
200  return nullptr;
201  }
202  controlPorts.insert(esiPort.ready.name.getValue());
203  }
204 
205  // -----
206  // Second, build a list of ports for the shell module, skipping the
207  // valid/ready, and converting the ESI data ports to the ESI channel port
208  // type. Store some bookkeeping information.
209 
210  SmallVector<hw::PortInfo, 64> shellPorts;
211  // Map the shell operand to the pearl port.
212  SmallVector<hw::PortInfo, 64> inputPortMap;
213  // Map the shell result to the pearl port.
214  SmallVector<hw::PortInfo, 64> outputPortMap;
215 
216  SmallVector<hw::PortInfo> pearlPorts = hw::getAllModulePortInfos(pearl);
217  for (const auto &port : pearlPorts) {
218  if (controlPorts.contains(port.name.getValue()))
219  continue;
220 
221  hw::PortInfo newPort = port;
222  if (dataPortMap.find(port.name.getValue()) != dataPortMap.end())
223  newPort.type = esi::ChannelPort::get(ctxt, port.type);
224 
225  if (port.isOutput()) {
226  newPort.argNum = outputPortMap.size();
227  outputPortMap.push_back(port);
228  } else {
229  newPort.argNum = inputPortMap.size();
230  inputPortMap.push_back(port);
231  }
232  shellPorts.push_back(newPort);
233  }
234 
235  // -----
236  // Third, create the shell module and also some builders for the inside.
237 
238  SmallString<64> shellNameBuf;
239  StringAttr shellName =
240  b.getStringAttr((SymbolTable::getSymbolName(pearl).getValue() + "_esi")
241  .toStringRef(shellNameBuf));
242  auto shell = b.create<hw::HWModuleOp>(loc, shellName, shellPorts);
243  shell.getBodyBlock()->clear(); // Erase the terminator.
244  auto modBuilder =
245  ImplicitLocOpBuilder::atBlockBegin(loc, shell.getBodyBlock());
246  BackedgeBuilder bb(modBuilder, modBuilder.getLoc());
247 
248  // Hold the operands for `hw.output` here.
249  SmallVector<Value, 64> outputs(shell.getNumResults());
250 
251  // -----
252  // Fourth, assemble the inputs for the pearl module AND build all the ESI wrap
253  // and unwrap ops for both the input and output channels.
254 
255  SmallVector<Value, 64> pearlOperands(modType.getNumInputs());
256 
257  // Since we build all the ESI wrap and unwrap operations before pearl
258  // instantiation, we only need backedges from the pearl instance result. Index
259  // the backedges by the pearl modules result number.
260  llvm::DenseMap<size_t, Backedge> backedges;
261 
262  // Go through the shell input ports, either tunneling them through or
263  // unwrapping the ESI channels. We'll need backedges for the ready signals
264  // since they are results from the upcoming pearl instance.
265  for (const auto &port : shellPorts) {
266  if (port.isOutput())
267  continue;
268 
269  Value arg = shell.getArgument(port.argNum);
270  auto esiPort = dataPortMap.find(port.name.getValue());
271  if (esiPort == dataPortMap.end()) {
272  // If it's just a regular port, it just gets passed through.
273  size_t pearlOpNum = inputPortMap[port.argNum].argNum;
274  pearlOperands[pearlOpNum] = arg;
275  continue;
276  }
277 
278  Backedge ready = bb.get(modBuilder.getI1Type());
279  backedges.insert(std::make_pair(esiPort->second.ready.argNum, ready));
280  auto unwrap = modBuilder.create<UnwrapValidReady>(arg, ready);
281  pearlOperands[esiPort->second.data.argNum] = unwrap.rawOutput();
282  pearlOperands[esiPort->second.valid.argNum] = unwrap.valid();
283  }
284 
285  // Iterate through the shell output ports, identify the ESI channels, and
286  // build ESI wrapper ops for signals being output from the pearl. The data and
287  // valid for these wrap ops will need to be backedges.
288  for (const auto &port : shellPorts) {
289  if (!port.isOutput())
290  continue;
291  auto esiPort = dataPortMap.find(port.name.getValue());
292  if (esiPort == dataPortMap.end())
293  continue;
294 
295  Backedge data = bb.get(esiPort->second.data.type);
296  Backedge valid = bb.get(modBuilder.getI1Type());
297  auto wrap = modBuilder.create<WrapValidReady>(data, valid);
298  backedges.insert(std::make_pair(esiPort->second.data.argNum, data));
299  backedges.insert(std::make_pair(esiPort->second.valid.argNum, valid));
300  outputs[port.argNum] = wrap.chanOutput();
301  pearlOperands[esiPort->second.ready.argNum] = wrap.ready();
302  }
303 
304  // -----
305  // Fifth, instantiate the pearl module.
306 
307  auto pearlInst =
308  modBuilder.create<hw::InstanceOp>(pearl, "pearl", pearlOperands);
309 
310  // Hookup all the backedges.
311  for (size_t i = 0, e = pearlInst.getNumResults(); i < e; ++i) {
312  auto backedge = backedges.find(i);
313  if (backedge != backedges.end())
314  backedge->second.setValue(pearlInst.getResult(i));
315  }
316 
317  // -----
318  // Finally, find all the regular outputs and either tunnel them through.
319  for (const auto &port : shellPorts) {
320  if (!port.isOutput())
321  continue;
322  auto esiPort = dataPortMap.find(port.name.getValue());
323  if (esiPort != dataPortMap.end())
324  continue;
325  size_t pearlResNum = outputPortMap[port.argNum].argNum;
326  outputs[port.argNum] = pearlInst.getResult(pearlResNum);
327  }
328 
329  modBuilder.create<hw::OutputOp>(outputs);
330  return shell;
331 }
