38 static MlirOperation pyWrapModule(MlirOperation cModOp,
39  std::vector<std::string> portNames) {
40  mlir::Operation *modOp = unwrap(cModOp);
41  SmallVector<StringRef, 8> portNamesRefs;
42  for (auto name : portNames)
43  portNamesRefs.push_back(name);
44  SmallVector<ESIPortValidReadyMapping, 8> portTriples;
45  resolvePortNames(modOp, portNamesRefs, portTriples);
46  OpBuilder b(modOp);
47  Operation *wrapper = buildESIWrapper(b, modOp, portTriples);
48  return wrap(wrapper);
49 }

68  /// Load the contents of an MLIR asm file into the system module.
69  void loadMlir(std::string filename) {
70  auto loadedMod = mlir::parseSourceFile(filename, ctxt());
71  Block *loadedBlock = loadedMod->getBody();
72  ModuleOp modOp = mod();
73  assert(!modOp->getRegions().empty());
74  if (modOp.body().empty()) {
75  modOp.body().push_back(loadedBlock);
76  return;
77  }
78  auto &ops = modOp.getBody()->getOperations();
79  ops.splice(ops.end(), loadedBlock->getOperations());
80  }
