29 ParseResult ChannelBuffer::parse(OpAsmParser &parser, OperationState &result) {
30  llvm::SMLoc inputOperandsLoc = parser.getCurrentLocation();
31 
32  llvm::SmallVector<OpAsmParser::UnresolvedOperand, 4> operands;
33  if (parser.parseOperandList(operands, /*requiredOperandCount=*/3,
34  /*delimiter=*/OpAsmParser::Delimiter::None))
35  return failure();
36 
37  Type innerOutputType;
38  if (parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
39  parser.parseType(innerOutputType))
40  return failure();
41  auto outputType =
42  ChannelPort::get(parser.getBuilder().getContext(), innerOutputType);
43  result.addTypes({outputType});
44 
45  auto i1 = IntegerType::get(result.getContext(), 1);
46  if (parser.resolveOperands(operands, {i1, i1, outputType}, inputOperandsLoc,
47  result.operands))
48  return failure();
49  return success();
50 }

159 /// If 'iface' looks like an ESI interface, return the inner data type.
160 static Type getEsiDataType(circt::sv::InterfaceOp iface) {
161  using namespace circt::sv;
162  if (!iface.lookupSymbol<InterfaceSignalOp>("valid"))
163  return Type();
164  if (!iface.lookupSymbol<InterfaceSignalOp>("ready"))
165  return Type();
166  auto dataSig = iface.lookupSymbol<InterfaceSignalOp>("data");
167  if (!dataSig)
168  return Type();
169  return dataSig.type();
170 }

172 /// Verify that the modport type of 'modportArg' points to an interface which
173 /// looks like an ESI interface and the inner data from said interface matches
174 /// the chan type's inner data type.
175 static LogicalResult verifySVInterface(Operation *op,
176  circt::sv::ModportType modportType,
177  ChannelPort chanType) {
178  auto modport =
179  SymbolTable::lookupNearestSymbolFrom<circt::sv::InterfaceModportOp>(
180  op, modportType.getModport());
181  if (!modport)
182  return op->emitError("Could not find modport ")
183  << modportType.getModport() << " in symbol table.";
184  auto iface = cast<circt::sv::InterfaceOp>(modport->getParentOp());
185  Type esiDataType = getEsiDataType(iface);
186  if (!esiDataType)
187  return op->emitOpError("Interface is not a valid ESI interface.");
188  if (esiDataType != chanType.getInner())
189  return op->emitOpError("Operation specifies ")
190  << chanType << " but type inside doesn't match interface data type "
191  << esiDataType << ".";
192  return success();
193 }
