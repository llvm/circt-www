125 static StringAttr constructUniqueSymbol(Operation *tableOp,
126  StringRef proposedNameRef) {
127  SmallString<64> proposedName = proposedNameRef;
128 
129  // Normalize the type name.
130  for (char &ch : proposedName) {
131  if (isalpha(ch) || isdigit(ch) || ch == '_')
132  continue;
133  ch = '_';
134  }
135 
136  // Make sure that this symbol isn't taken. If it is, append a number and try
137  // again.
138  size_t baseLength = proposedName.size();
139  size_t tries = 0;
140  while (SymbolTable::lookupSymbolIn(tableOp, proposedName)) {
141  proposedName.resize(baseLength);
142  proposedName.append(llvm::utostr(++tries));
143  }
144 
145  return StringAttr::get(tableOp->getContext(), proposedName);
146 }

148 StringAttr ESIHWBuilder::constructInterfaceName(ChannelPort port) {
149  Operation *tableOp =
150  getInsertionPoint()->getParentWithTrait<mlir::OpTrait::SymbolTable>();
151 
152  // Get a name based on the type.
153  std::string portTypeName;
154  llvm::raw_string_ostream nameOS(portTypeName);
155  TypeSwitch<Type>(port.getInner())
156  .Case([&](hw::ArrayType arr) {
157  nameOS << "ArrayOf" << arr.getSize() << 'x' << arr.getElementType();
158  })
159  .Case([&](hw::StructType t) { nameOS << "Struct"; })
160  .Default([&](Type t) { nameOS << port.getInner(); });
161 
162  // Don't allow the name to end with '_'.
163  ssize_t i = portTypeName.size() - 1;
164  while (i >= 0 && portTypeName[i] == '_') {
165  --i;
166  }
167  portTypeName = portTypeName.substr(0, i + 1);
168 
169  // All stage names start with this.
170  SmallString<64> proposedName("IValidReady_");
171  proposedName.append(portTypeName);
172  return constructUniqueSymbol(tableOp, proposedName);
173 }

284 public:
285  using OpConversionPattern::OpConversionPattern;

287  LogicalResult
288  matchAndRewrite(ChannelBuffer buffer, OpAdaptor adaptor,
289  ConversionPatternRewriter &rewriter) const final;

293 LogicalResult ChannelBufferLowering::matchAndRewrite(
294  ChannelBuffer buffer, OpAdaptor adaptor,
295  ConversionPatternRewriter &rewriter) const {
296  auto loc = buffer.getLoc();
297 
298  auto type = buffer.getType();
299 
300  // Expand 'abstract' buffer into 'physical' stages.
301  auto stages = buffer.stagesAttr();
302  uint64_t numStages = 1;
303  if (stages) {
304  // Guaranteed positive by the parser.
305  numStages = stages.getValue().getLimitedValue();
306  }
307  Value input = buffer.input();
308  StringAttr bufferName = buffer.nameAttr();
309  for (uint64_t i = 0; i < numStages; ++i) {
310  // Create the stages, connecting them up as we build.
311  auto stage = rewriter.create<PipelineStage>(loc, type, buffer.clk(),
312  buffer.rstn(), input);
313  if (bufferName) {
314  SmallString<64> stageName(
315  {bufferName.getValue(), "_stage", std::to_string(i)});
316  stage->setAttr("name", StringAttr::get(rewriter.getContext(), stageName));
317  }
318  input = stage;
319  }
320 
321  // Replace the buffer.
322  rewriter.replaceOp(buffer, input);
323  return success();
324 }

333 void ESIToPhysicalPass::runOnOperation() {
334  // Set up a conversion and give it a set of laws.
335  ConversionTarget target(getContext());
336  target.addLegalDialect<ESIDialect>();
337  target.addIllegalOp<ChannelBuffer>();
338 
339  // Add all the conversion patterns.
340  RewritePatternSet patterns(&getContext());
341  patterns.insert<ChannelBufferLowering>(&getContext());
342 
343  // Run the conversion.
344  if (failed(
345  applyPartialConversion(getOperation(), target, std::move(patterns))))
346  signalPassFailure();
347 }

370 /// Iterate through the `hw.module[.extern]`s and lower their ports.
371 void ESIPortsPass::runOnOperation() {
372  ModuleOp top = getOperation();
373  ESIHWBuilder b(top);
374  build = &b;
375 
376  // Find all externmodules and try to modify them. Remember the modified ones.
377  DenseMap<StringRef, HWModuleExternOp> externModsMutated;
378  for (auto mod : top.getOps<HWModuleExternOp>())
379  if (updateFunc(mod))
380  externModsMutated[mod.getName()] = mod;
381 
382  // Find all instances and update them.
383  top.walk([&externModsMutated, this](InstanceOp inst) {
384  auto mapIter = externModsMutated.find(inst.moduleName());
385  if (mapIter != externModsMutated.end())
386  updateInstance(mapIter->second, inst);
387  });
388 
389  // Find all modules and try to modify them to have wires with valid/ready
390  // semantics. Remember the modified ones.
391  DenseMap<StringRef, HWModuleOp> modsMutated;
392  for (auto mod : top.getOps<HWModuleOp>())
393  if (updateFunc(mod))
394  modsMutated[mod.getName()] = mod;
395 
396  // Find all instances and update them.
397  top.walk([&modsMutated, this](InstanceOp inst) {
398  auto mapIter = modsMutated.find(inst.moduleName());
399  if (mapIter != modsMutated.end())
400  updateInstance(mapIter->second, inst);
401  });
402 
403  build = nullptr;
404 }

412 /// Convert all input and output ChannelPorts into valid/ready wires. Try not to
413 /// change the order and materialize ops in reasonably intuitive locations.
414 bool ESIPortsPass::updateFunc(HWModuleOp mod) {
415  auto funcType = mod.getFunctionType();
416  // Build ops in the module.
417  ImplicitLocOpBuilder modBuilder(mod.getLoc(), mod.getBody());
418  Type i1 = modBuilder.getI1Type();
419 
420  // Get information to be used later on.
421  hw::OutputOp outOp = cast<hw::OutputOp>(mod.getBodyBlock()->getTerminator());
422 
423  bool updated = false;
424 
425  // Reconstruct the list of operand types, changing the type whenever an ESI
426  // port is found. Keep the argument attributes, apply the ESI ports attributes
427  // to the data port only.
428  SmallVector<Type, 16> newArgTypes;
429 
430  // 'Ready' signals are outputs. Remember them for later when we deal with the
431  // returns.
432  SmallVector<std::pair<Value, StringAttr>, 8> newReadySignals;
433  SmallVector<Attribute> newArgNames;
434 
435  for (size_t argNum = 0, blockArgNum = 0, e = mod.getNumArguments();
436  argNum < e; ++argNum, ++blockArgNum) {
437  Type argTy = funcType.getInput(argNum);
438  auto argNameAttr = getModuleArgumentNameAttr(mod, argNum);
439 
440  auto chanTy = argTy.dyn_cast<ChannelPort>();
441  if (!chanTy) {
442  // If not ESI, pass through.
443  newArgTypes.push_back(argTy);
444  newArgNames.push_back(argNameAttr);
445  continue;
446  }
447 
448  // When we find one, add a data and valid signal to the new args.
449  newArgTypes.push_back(chanTy.getInner());
450  newArgTypes.push_back(i1);
451  newArgNames.push_back(argNameAttr);
452  newArgNames.push_back(appendToRtlName(argNameAttr, "_valid"));
453  // Add the BlockArguments.
454  Value data = mod.front().insertArgument(blockArgNum, chanTy.getInner(),
455  mod.getArgument(argNum).getLoc());
456  Value valid = mod.front().insertArgument(blockArgNum + 1, i1,
457  mod.getArgument(argNum).getLoc());
458  // Build the ESI wrap operation to translate the lowered signals to what
459  // they were. (A later pass takes care of eliminating the ESI ops.)
460  auto wrap = modBuilder.create<WrapValidReady>(data, valid);
461  // Replace uses of the old ESI port argument with the new one from the wrap.
462  mod.front()
463  .getArgument(blockArgNum + 2)
464  .replaceAllUsesWith(wrap.chanOutput());
465  // Delete the ESI port block argument.
466  mod.front().eraseArgument(blockArgNum + 2);
467  newReadySignals.push_back(
468  std::make_pair(wrap.ready(), appendToRtlName(argNameAttr, "_ready")));
469 
470  // Since we added 2 block args but erased one, there's a net increase of 1.
471  blockArgNum += 1;
472  updated = true;
473  }
474 
475  // Iterate through the outputs, appending to all of the next three lists.
476  // Lower the ESI ports.
477  SmallVector<Type, 8> newResultTypes;
478  SmallVector<Value, 8> newOutputOperands;
479  SmallVector<Attribute> newResultNames;
480 
481  modBuilder.setInsertionPointToEnd(mod.getBodyBlock());
482  for (size_t resNum = 0, numRes = funcType.getNumResults(); resNum < numRes;
483  ++resNum) {
484  Type resTy = funcType.getResult(resNum);
485  auto chanTy = resTy.dyn_cast<ChannelPort>();
486  Value oldOutputValue = outOp.getOperand(resNum);
487  auto oldResultName = getModuleResultNameAttr(mod, resNum);
488  if (!chanTy) {
489  // If not ESI, pass through.
490  newResultTypes.push_back(resTy);
491  newResultNames.push_back(oldResultName);
492  newOutputOperands.push_back(oldOutputValue);
493  continue;
494  }
495 
496  // Lower the output, adding ready signals directly to the arg list.
497  Value ready = mod.front().addArgument(
498  i1, modBuilder.getUnknownLoc()); // Ready block arg.
499  auto unwrap = modBuilder.create<UnwrapValidReady>(oldOutputValue, ready);
500  newOutputOperands.push_back(unwrap.rawOutput());
501  newOutputOperands.push_back(unwrap.valid());
502 
503  newResultTypes.push_back(chanTy.getInner()); // Raw data.
504  newResultTypes.push_back(i1); // Valid.
505  newResultNames.push_back(oldResultName);
506  newResultNames.push_back(appendToRtlName(oldResultName, "_valid"));
507 
508  newArgTypes.push_back(i1); // Ready func arg.
509  newArgNames.push_back(appendToRtlName(oldResultName, "_ready"));
510  updated = true;
511  }
512 
513  // Append the ready list signals we remembered above.
514  for (const auto &readySig : newReadySignals) {
515  newResultTypes.push_back(i1);
516  newResultNames.push_back(readySig.second);
517  newOutputOperands.push_back(readySig.first);
518  }
519 
520  if (!updated)
521  return false;
522 
523  // A new output op is necessary.
524  outOp.erase();
525  modBuilder.create<hw::OutputOp>(newOutputOperands);
526 
527  // Set the new types.
528  auto newFuncType = modBuilder.getFunctionType(newArgTypes, newResultTypes);
529  mod.setType(newFuncType);
530  setModuleArgumentNames(mod, newArgNames);
531  setModuleResultNames(mod, newResultNames);
532  return true;
533 }

535 /// Update an instance of an updated module by adding `esi.[un]wrap.vr`
536 /// ops around the instance. Lowering or folding away `[un]wrap` ops is another
537 /// pass.
538 void ESIPortsPass::updateInstance(HWModuleOp mod, InstanceOp inst) {
539  ImplicitLocOpBuilder b(inst.getLoc(), inst);
540  BackedgeBuilder beb(b, inst.getLoc());
541  Type i1 = b.getI1Type();
542 
543  // -----
544  // Lower the operands.
545 
546  SmallVector<Value, 16> newOperands;
547  // Store the 'ready' operands from the unwrap as a list of backedges. This
548  // doubles as a count of `i1`s to append to the existing results.
549  SmallVector<Backedge, 8> inputReadysToConnect;
550  for (auto operand : inst.getOperands()) {
551  if (!operand.getType().isa<ChannelPort>()) {
552  newOperands.push_back(operand);
553  continue;
554  }
555 
556  auto ready = beb.get(i1);
557  inputReadysToConnect.push_back(ready);
558  auto unwrap = b.create<UnwrapValidReady>(operand, ready);
559  newOperands.push_back(unwrap.rawOutput());
560  newOperands.push_back(unwrap.valid());
561  }
562 
563  // -----
564  // Lower the result types.
565 
566  SmallVector<Type, 16> resTypes;
567  // Backedges which we add as operands to be used later when we build the
568  // 'wrap' ops.
569  SmallVector<Backedge, 8> outputReadysToConnect;
570  for (auto resTy : inst.getResultTypes()) {
571  auto cpTy = resTy.dyn_cast<ChannelPort>();
572  if (!cpTy) {
573  resTypes.push_back(resTy);
574  continue;
575  }
576  resTypes.push_back(cpTy.getInner());
577  resTypes.push_back(i1);
578  Backedge ready = beb.get(i1);
579  newOperands.push_back(ready);
580  outputReadysToConnect.push_back(ready);
581  }
582  resTypes.append(inputReadysToConnect.size(), i1);
583 
584  // -----
585  // Clone the instance.
586  b.setInsertionPointAfter(inst);
587  auto newInst = b.create<InstanceOp>(mod, inst.instanceNameAttr(), newOperands,
588  inst.parameters(), inst.inner_symAttr());
589 
590  // -----
591  // Wrap the results back into ESI channels and connect up all the ready
592  // signals.
593 
594  size_t newInstResNum = 0;
595  size_t readyIdx = 0;
596  for (auto res : inst.getResults()) {
597  auto cpTy = res.getType().dyn_cast<ChannelPort>();
598  if (!cpTy) {
599  res.replaceAllUsesWith(newInst.getResult(newInstResNum));
600  newInstResNum++;
601  continue;
602  }
603 
604  auto wrap = b.create<WrapValidReady>(newInst.getResult(newInstResNum),
605  newInst.getResult(newInstResNum + 1));
606  newInstResNum += 2;
607  res.replaceAllUsesWith(wrap.chanOutput());
608  outputReadysToConnect[readyIdx].setValue(wrap.ready());
609  readyIdx++;
610  }
611 
612  for (auto inputReady : inputReadysToConnect) {
613  inputReady.setValue(newInst.getResult(newInstResNum));
614  newInstResNum++;
615  }
616 
617  // Erase the old instance.
618  inst.erase();
619 }

621 /// Convert all input and output ChannelPorts into SV Interfaces. For inputs,
622 /// just switch the type to `ModportType`. For outputs, append a `ModportType`
623 /// to the inputs and remove the output channel from the results. Returns true
624 /// if 'mod' was updated. Delay updating the instances to amortize the IR walk
625 /// over all the module updates.
626 bool ESIPortsPass::updateFunc(HWModuleExternOp mod) {
627  auto *ctxt = &getContext();
628 
629  bool updated = false;
630 
631  SmallVector<Attribute> newArgNames, newResultNames;
632 
633  // Reconstruct the list of operand types, changing the type whenever an ESI
634  // port is found.
635  SmallVector<Type, 16> newArgTypes;
636  size_t nextArgNo = 0;
637  for (auto argTy : mod.getArgumentTypes()) {
638  auto chanTy = argTy.dyn_cast<ChannelPort>();
639  newArgNames.push_back(getModuleArgumentNameAttr(mod, nextArgNo++));
640 
641  if (!chanTy) {
642  newArgTypes.push_back(argTy);
643  continue;
644  }
645 
646  // When we find one, construct an interface, and add the 'source' modport to
647  // the type list.
648  auto iface = build->getOrConstructInterface(chanTy);
649  newArgTypes.push_back(iface.getModportType(ESIHWBuilder::sourceStr));
650  updated = true;
651  }
652 
653  // Iterate through the results and append to one of the two below lists. The
654  // first for non-ESI-ports. The second, ports which have been re-located to an
655  // operand.
656  SmallVector<Type, 8> newResultTypes;
657  SmallVector<DictionaryAttr, 4> newResultAttrs;
658  auto funcType = mod.getFunctionType();
659  for (size_t resNum = 0, numRes = mod.getNumResults(); resNum < numRes;
660  ++resNum) {
661  Type resTy = funcType.getResult(resNum);
662  auto chanTy = resTy.dyn_cast<ChannelPort>();
663  auto resNameAttr = getModuleResultNameAttr(mod, resNum);
664  if (!chanTy) {
665  newResultTypes.push_back(resTy);
666  newResultNames.push_back(resNameAttr);
667  continue;
668  }
669 
670  // When we find one, construct an interface, and add the 'sink' modport to
671  // the type list.
672  InterfaceOp iface = build->getOrConstructInterface(chanTy);
673  ModportType sinkPort = iface.getModportType(ESIHWBuilder::sinkStr);
674  newArgTypes.push_back(sinkPort);
675  newArgNames.push_back(resNameAttr);
676  updated = true;
677  }
678 
679  if (!updated)
680  return false;
681 
682  // Set the new types.
683  auto newFuncType = FunctionType::get(ctxt, newArgTypes, newResultTypes);
684  mod.setType(newFuncType);
685  setModuleArgumentNames(mod, newArgNames);
686  setModuleResultNames(mod, newResultNames);
687  return true;
688 }

690 static StringRef getOperandName(Value operand) {
691  if (BlockArgument arg = operand.dyn_cast<BlockArgument>()) {
692  auto op = arg.getParentBlock()->getParentOp();
693  if (op && hw::isAnyModule(op))
694  return hw::getModuleArgumentName(op, arg.getArgNumber());
695  } else {
696  auto srcOp = operand.getDefiningOp();
697  if (auto instOp = dyn_cast<InstanceOp>(srcOp))
698  return instOp.instanceName();
699 
700  if (auto srcName = srcOp->getAttrOfType<StringAttr>("name"))
701  return srcName.getValue();
702  }
703  return "";
704 }

706 /// Create a reasonable name for a SV interface instance.
707 static std::string &constructInstanceName(Value operand, InterfaceOp iface,
708  std::string &name) {
709  llvm::raw_string_ostream s(name);
710  // Drop the "IValidReady_" part of the interface name.
711  s << llvm::toLower(iface.sym_name()[12]) << iface.sym_name().substr(13);
712 
713  // Indicate to where the source is connected.
714  if (operand.hasOneUse()) {
715  Operation *dstOp = *operand.getUsers().begin();
716  if (auto instOp = dyn_cast<InstanceOp>(dstOp))
717  s << "To" << llvm::toUpper(instOp.instanceName()[0])
718  << instOp.instanceName().substr(1);
719  else if (auto dstName = dstOp->getAttrOfType<StringAttr>("name"))
720  s << "To" << dstName.getValue();
721  }
722 
723  // Indicate to where the sink is connected.
724  StringRef operName = getOperandName(operand);
725  if (!operName.empty())
726  s << "From" << llvm::toUpper(operName[0]) << operName.substr(1);
727  return s.str();
728 }

730 /// Update an instance of an updated module by adding `esi.(un)wrap.iface`
731 /// around the instance. Create a new instance at the end from the lists built
732 /// up before.
733 void ESIPortsPass::updateInstance(HWModuleExternOp mod, InstanceOp inst) {
734  using namespace circt::sv;
735  circt::ImplicitLocOpBuilder instBuilder(inst.getLoc(), inst);
736  FunctionType funcTy = mod.getFunctionType();
737 
738  // op counter for error reporting purposes.
739  size_t opNum = 0;
740  // List of new operands.
741  SmallVector<Value, 16> newOperands;
742 
743  // Fill the new operand list with old plain operands and mutated ones.
744  std::string nameStringBuffer; // raw_string_ostream uses std::string.
745  for (auto op : inst.getOperands()) {
746  auto instChanTy = op.getType().dyn_cast<ChannelPort>();
747  if (!instChanTy) {
748  newOperands.push_back(op);
749  ++opNum;
750  continue;
751  }
752 
753  // Get the interface from the cache, and make sure it's the same one as
754  // being used in the module.
755  auto iface = build->getOrConstructInterface(instChanTy);
756  if (iface.getModportType(ESIHWBuilder::sourceStr) !=
757  funcTy.getInput(opNum)) {
758  inst.emitOpError("ESI ChannelPort (operand #")
759  << opNum << ") doesn't match module!";
760  ++opNum;
761  newOperands.push_back(op);
762  continue;
763  }
764  ++opNum;
765 
766  // Build a gasket by instantiating an interface, connecting one end to an
767  // `esi.unwrap.iface` and the other end to the instance.
768  auto ifaceInst =
769  instBuilder.create<InterfaceInstanceOp>(iface.getInterfaceType());
770  nameStringBuffer.clear();
771  ifaceInst->setAttr(
772  "name",
773  StringAttr::get(mod.getContext(),
774  constructInstanceName(op, iface, nameStringBuffer)));
775  GetModportOp sinkModport =
776  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sinkStr);
777  instBuilder.create<UnwrapSVInterface>(op, sinkModport);
778  GetModportOp sourceModport =
779  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sourceStr);
780  // Finally, add the correct modport to the list of operands.
781  newOperands.push_back(sourceModport);
782  }
783 
784  // Go through the results and get both a list of the plain old values being
785  // produced and their types.
786  SmallVector<Value, 8> newResults;
787  SmallVector<Type, 8> newResultTypes;
788  for (size_t resNum = 0, numRes = inst.getNumResults(); resNum < numRes;
789  ++resNum) {
790  Value res = inst.getResult(resNum);
791  auto instChanTy = res.getType().dyn_cast<ChannelPort>();
792  if (!instChanTy) {
793  newResults.push_back(res);
794  newResultTypes.push_back(res.getType());
795  continue;
796  }
797 
798  // Get the interface from the cache, and make sure it's the same one as
799  // being used in the module.
800  auto iface = build->getOrConstructInterface(instChanTy);
801  if (iface.getModportType(ESIHWBuilder::sinkStr) != funcTy.getInput(opNum)) {
802  inst.emitOpError("ESI ChannelPort (result #")
803  << resNum << ", operand #" << opNum << ") doesn't match module!";
804  ++opNum;
805  newResults.push_back(res);
806  newResultTypes.push_back(res.getType());
807  continue;
808  }
809  ++opNum;
810 
811  // Build a gasket by instantiating an interface, connecting one end to an
812  // `esi.wrap.iface` and the other end to the instance. Append it to the
813  // operand list.
814  auto ifaceInst =
815  instBuilder.create<InterfaceInstanceOp>(iface.getInterfaceType());
816  nameStringBuffer.clear();
817  ifaceInst->setAttr(
818  "name",
819  StringAttr::get(mod.getContext(),
820  constructInstanceName(res, iface, nameStringBuffer)));
821  GetModportOp sourceModport =
822  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sourceStr);
823  auto newChannel =
824  instBuilder.create<WrapSVInterface>(res.getType(), sourceModport);
825  // Connect all the old users of the output channel with the newly
826  // wrapped replacement channel.
827  res.replaceAllUsesWith(newChannel);
828  GetModportOp sinkModport =
829  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sinkStr);
830  // And add the modport on the other side to the new operand list.
831  newOperands.push_back(sinkModport);
832  }
833 
834  // Create the new instance!
835  InstanceOp newInst =
836  instBuilder.create<InstanceOp>(mod, inst.instanceNameAttr(), newOperands,
837  inst.parameters(), inst.inner_symAttr());
838 
839  // Go through the old list of non-ESI result values, and replace them with the
840  // new non-ESI results.
841  for (size_t resNum = 0, numRes = newResults.size(); resNum < numRes;
842  ++resNum) {
843  newResults[resNum].replaceAllUsesWith(newInst.getResult(resNum));
844  }
845  // Erase the old instance!
846  inst.erase();
847 }

872 LogicalResult PipelineStageLowering::matchAndRewrite(
873  PipelineStage stage, OpAdaptor adaptor,
874  ConversionPatternRewriter &rewriter) const {
875  auto loc = stage.getLoc();
876  auto chPort = stage.input().getType().dyn_cast<ChannelPort>();
877  if (!chPort)
878  return failure();
879  Operation *symTable = stage->getParentWithTrait<OpTrait::SymbolTable>();
880  auto stageModule = builder.declareStage(symTable, chPort.getInner());
881 
882  size_t width = circt::hw::getBitWidth(chPort.getInner());
883 
884  ArrayAttr stageParams =
885  builder.getStageParameterList(rewriter.getUI32IntegerAttr(width));
886 
887  // Unwrap the channel. The ready signal is a Value we haven't created yet, so
888  // create a temp value and replace it later. Give this constant an odd-looking
889  // type to make debugging easier.
890  circt::BackedgeBuilder back(rewriter, loc);
891  circt::Backedge wrapReady = back.get(rewriter.getI1Type());
892  auto unwrap =
893  rewriter.create<UnwrapValidReady>(loc, stage.input(), wrapReady);
894 
895  StringRef pipeStageName = "pipelineStage";
896  if (auto name = stage->getAttrOfType<StringAttr>("name"))
897  pipeStageName = name.getValue();
898 
899  // Instantiate the "ESI_PipelineStage" external module.
900  circt::Backedge stageReady = back.get(rewriter.getI1Type());
901  Value operands[] = {stage.clk(), stage.rstn(), unwrap.rawOutput(),
902  unwrap.valid(), stageReady};
903  auto stageInst = rewriter.create<InstanceOp>(loc, stageModule, pipeStageName,
904  operands, stageParams);
905  auto stageInstResults = stageInst.getResults();
906 
907  // Set a_ready (from the unwrap) back edge correctly to its output from stage.
908  wrapReady.setValue(stageInstResults[0]);
909 
910  Value x = stageInstResults[1];
911  Value xValid = stageInstResults[2];
912 
913  // Wrap up the output of the HW stage module.
914  auto wrap = rewriter.create<WrapValidReady>(loc, chPort, rewriter.getI1Type(),
915  x, xValid);
916  // Set the stages x_ready backedge correctly.
917  stageReady.setValue(wrap.ready());
918 
919  rewriter.replaceOp(stage, wrap.chanOutput());
920  return success();
921 }

957 struct RemoveWrapUnwrap : public ConversionPattern {
958 public:
959  RemoveWrapUnwrap(MLIRContext *context)
960  : ConversionPattern(MatchAnyOpTypeTag(), /*benefit=*/1, context) {}
961 
962  virtual LogicalResult
963  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
964  ConversionPatternRewriter &rewriter) const override {
965  Value valid, ready, data;

967  UnwrapValidReady unwrap = dyn_cast<UnwrapValidReady>(op);
968  if (wrap) {
969  if (!wrap.chanOutput().hasOneUse() ||
970  !(unwrap = dyn_cast<UnwrapValidReady>(
971  wrap.chanOutput().use_begin()->getOwner())))
972  return rewriter.notifyMatchFailure(
973  wrap, "This conversion only supports wrap-unwrap back-to-back. "
974  "Could not find 'unwrap'.");

976  valid = operands[1];
977  ready = unwrap.ready();
978  } else if (unwrap) {
979  wrap = dyn_cast<WrapValidReady>(operands[0].getDefiningOp());
980  if (!wrap)
981  return rewriter.notifyMatchFailure(
982  operands[0].getDefiningOp(),
983  "This conversion only supports wrap-unwrap back-to-back. "
984  "Could not find 'wrap'.");

986  data = wrap.rawInput();
987  ready = operands[1];
988  } else {
989  return failure();
990  }
991

993  return rewriter.notifyMatchFailure(wrap, [](Diagnostic &d) {
994  d << "This conversion only supports wrap-unwrap back-to-back. "
995  "Wrap didn't have exactly one use.";
996  });
997  rewriter.replaceOp(wrap, {nullptr, ready});
998  rewriter.replaceOp(unwrap, {data, valid});
999  return success();
1000  }

1114 
1115 LogicalResult
1116 CosimLowering::matchAndRewrite(CosimEndpoint ep, OpAdaptor adaptor,
1117  ConversionPatternRewriter &rewriter) const {
1118 #ifndef CAPNP
1119  (void)builder;
1120  return rewriter.notifyMatchFailure(
1121  ep, "Cosim lowering requires the ESI capnp plugin, which was disabled.");
1122 #else
1123  auto loc = ep.getLoc();
1124  auto *ctxt = rewriter.getContext();
1125  auto operands = adaptor.getOperands();
1126  Value clk = operands[0];
1127  Value rstn = operands[1];
1128  Value send = operands[2];
1129 
1130  circt::BackedgeBuilder bb(rewriter, loc);
1131  Type ui64Type =
1132  IntegerType::get(ctxt, 64, IntegerType::SignednessSemantics::Unsigned);
1133  capnp::TypeSchema sendTypeSchema(send.getType());
1134  if (!sendTypeSchema.isSupported())
1135  return rewriter.notifyMatchFailure(ep, "Send type not supported yet");
1136  capnp::TypeSchema recvTypeSchema(ep.recv().getType());
1137  if (!recvTypeSchema.isSupported())
1138  return rewriter.notifyMatchFailure(ep, "Recv type not supported yet");
1139 
1140  // Set all the parameters.
1141  SmallVector<Attribute, 8> params;
1142  params.push_back(ParamDeclAttr::get(
1143  "ENDPOINT_ID", rewriter.getI32IntegerAttr(ep.endpointID())));
1144  params.push_back(ParamDeclAttr::get(
1145  "SEND_TYPE_ID",
1146  IntegerAttr::get(ui64Type, sendTypeSchema.capnpTypeID())));
1147  params.push_back(
1148  ParamDeclAttr::get("SEND_TYPE_SIZE_BITS",
1149  rewriter.getI32IntegerAttr(sendTypeSchema.size())));
1150  params.push_back(ParamDeclAttr::get(
1151  "RECV_TYPE_ID",
1152  IntegerAttr::get(ui64Type, recvTypeSchema.capnpTypeID())));
1153  params.push_back(
1154  ParamDeclAttr::get("RECV_TYPE_SIZE_BITS",
1155  rewriter.getI32IntegerAttr(recvTypeSchema.size())));
1156 
1157  // Set up the egest route to drive the EP's send ports.
1158  ArrayType egestBitArrayType =
1159  ArrayType::get(rewriter.getI1Type(), sendTypeSchema.size());
1160  auto sendReady = bb.get(rewriter.getI1Type());
1161  UnwrapValidReady unwrapSend =
1162  rewriter.create<UnwrapValidReady>(loc, send, sendReady);
1163  auto encodeData = rewriter.create<CapnpEncode>(
1164  loc, egestBitArrayType, clk, unwrapSend.valid(), unwrapSend.rawOutput());
1165 
1166  // Get information necessary for injest path.
1167  auto recvReady = bb.get(rewriter.getI1Type());
1168  ArrayType ingestBitArrayType =
1169  ArrayType::get(rewriter.getI1Type(), recvTypeSchema.size());
1170 
1171  // Build or get the cached Cosim Endpoint module parameterization.
1172  Operation *symTable = ep->getParentWithTrait<OpTrait::SymbolTable>();
1173  HWModuleExternOp endpoint = builder.declareCosimEndpoint(
1174  symTable, egestBitArrayType, ingestBitArrayType);
1175 
1176  // Create replacement Cosim_Endpoint instance.
1177  StringAttr nameAttr = ep->getAttr("name").dyn_cast_or_null<StringAttr>();
1178  StringRef name = nameAttr ? nameAttr.getValue() : "cosimEndpoint";
1179  Value epInstInputs[] = {
1180  clk, rstn, recvReady, unwrapSend.valid(), encodeData.capnpBits(),
1181  };
1182 
1183  auto cosimEpModule =
1184  rewriter.create<InstanceOp>(loc, endpoint, name, epInstInputs,
1185  ArrayAttr::get(ctxt, params), StringAttr());
