76 LogicalResult ExportCosimSchema::visitEndpoint(CosimEndpoint ep) {
77  capnp::TypeSchema sendTypeSchema(ep.send().getType());
78  if (!sendTypeSchema.isSupported())
79  return ep.emitOpError("Type ") << ep.send().getType() << " not supported.";
80  types.push_back(sendTypeSchema);
81 
82  capnp::TypeSchema recvTypeSchema(ep.recv().getType());
83  if (!recvTypeSchema.isSupported())
84  return ep.emitOpError("Type '")
85  << ep.recv().getType() << "' not supported.";
86  types.push_back(recvTypeSchema);
87 
88  os << "# Endpoint ";
89  StringAttr epName = ep->getAttrOfType<StringAttr>("name");
90  if (epName)
91  os << epName << " is endpoint ";
92  os << "#" << ep.endpointID() << " at " << ep.getLoc() << ":\n";
93  os << "# Send type: ";
94  sendTypeSchema.writeMetadata(os);
95  os << "\n";
96 
97  os << "# Recv type: ";
98  recvTypeSchema.writeMetadata(os);
99  os << "\n";
100 
101  return success();
102 }

116 LogicalResult ExportCosimSchema::emit() {
117  os << "#########################################################\n"
118  << "## ESI generated schema.\n"
119  << "#########################################################\n";
120 
121  // Walk and collect the type data.
122  auto walkResult = module.walk([this](CosimEndpoint ep) {
123  if (failed(visitEndpoint(ep)))
124  return mlir::WalkResult::interrupt();
125  return mlir::WalkResult::advance();
126  });
127  if (walkResult.wasInterrupted())
128  return failure();
129  os << "#########################################################\n";
130 
131  // We need a sorted list to ensure determinism.
132  llvm::sort(types.begin(), types.end(),
133  [](capnp::TypeSchema &a, capnp::TypeSchema &b) {
134  return a.capnpTypeID() > b.capnpTypeID();
135  });
136 
137  // Compute and emit the capnp file id.
138  uint64_t fileHash = 2544816649379317016; // Some random number.
139  for (capnp::TypeSchema &schema : types)
140  fileHash =
141  llvm::hashing::detail::hash_16_bytes(fileHash, schema.capnpTypeID());
142  // Capnp IDs always have a '1' high bit.
143  fileHash |= 0x8000000000000000;
144  emitId(fileHash) << ";\n\n";
145 
146  os << "#########################################################\n"
147  << "## Types for your design.\n"
148  << "#########################################################\n\n";
149  // Iterate through the various types and emit their schemas.
150  auto end = std::unique(types.begin(), types.end());
151  for (auto typeIter = types.begin(); typeIter < end; ++typeIter) {
152  if (failed(typeIter->write(os)))
153  // If we fail during an emission, dump out early since the output may be
154  // corrupted.
155  return failure();
156  }
157 
158  // Include the RPC schema in each generated file.
159  emitCosimSchemaBody(os);
160 
161  return errorCount == 0 ? success() : failure();
162 }
