145 /// Check if an `OMNode` is an `OMSRAM` and requires special treatment of its
146 /// instance path field. This returns the ID of the tracker stored in the
147 /// `instancePath` or `finalPath` field if the node has an array field `omType`
148 /// that contains a `OMString:OMSRAM` entry.
149 static IntegerAttr isOMSRAM(Attribute &node) {
150  auto dict = node.dyn_cast<DictionaryAttr>();
151  if (!dict)
152  return {};
153  auto idAttr = dict.getAs<StringAttr>("id");
154  if (!idAttr)
155  return {};
156  IntegerAttr id;
157  if (auto infoAttr = dict.getAs<DictionaryAttr>("fields")) {
158  auto finalPath = infoAttr.getAs<DictionaryAttr>("finalPath");
159  // The following is used prior to an upstream bump in Chisel.
160  if (!finalPath)
161  finalPath = infoAttr.getAs<DictionaryAttr>("instancePath");
162  if (finalPath)
163  if (auto v = finalPath.getAs<DictionaryAttr>("value"))
164  if (v.getAs<UnitAttr>("omir.tracker"))
165  id = v.getAs<IntegerAttr>("id");
166  if (auto omTy = infoAttr.getAs<DictionaryAttr>("omType"))
167  if (auto valueArr = omTy.getAs<ArrayAttr>("value"))
168  for (auto attr : valueArr)
169  if (auto str = attr.dyn_cast<StringAttr>())
170  if (str.getValue().equals("OMString:OMSRAM"))
171  return id;
172  }
173  return {};
174 }

180 /// Recursively walk Object Model IR and convert FIRRTL targets to identifiers
181 /// while scattering trackers into the newAnnotations argument.
182 ///
183 /// Object Model IR consists of a type hierarchy built around recursive arrays
184 /// and dictionaries whose leaves are "string-encoded types". This is an Object
185 /// Model-specific construct that puts type information alongside a value.
186 /// Concretely, these look like:
187 ///
188 /// 'OM' type ':' value
189 ///
190 /// This function is only concerned with unpacking types whose values are FIRRTL
191 /// targets. This is because these need to be kept up-to-date with
192 /// modifications made to the circuit whereas other types are just passing
193 /// through CIRCT.
194 ///
195 /// At a later time this understanding may be expanded or Object Model IR may
196 /// become its own Dialect. At this time, this function is trying to do as
197 /// minimal work as possible to just validate that the OMIR looks okay without
198 /// doing lots of unnecessary unpacking/repacking of string-encoded types.
199 static Optional<Attribute> scatterOMIR(Attribute original, ApplyState &state) {
200  auto *ctx = original.getContext();
201 
202  // Convert a string-encoded type to a dictionary that includes the type
203  // information and an identifier derived from the current annotationID. Then
204  // increment the annotationID. Return the constructed dictionary.
205  auto addID = [&](StringRef tpe, StringRef path,
206  IntegerAttr id) -> DictionaryAttr {
207  NamedAttrList fields;
208  fields.append("id", id);
209  fields.append("omir.tracker", UnitAttr::get(ctx));
210  fields.append("path", StringAttr::get(ctx, path));
211  fields.append("type", StringAttr::get(ctx, tpe));
212  return DictionaryAttr::getWithSorted(ctx, fields);
213  };
214 
215  return TypeSwitch<Attribute, Optional<Attribute>>(original)
216  // Most strings in the Object Model are actually string-encoded types.
217  // These are types which look like: "<type>:<value>". This code will
218  // examine all strings, parse them into type and value, and then either
219  // store them in their unpacked state (and possibly scatter trackers into
220  // the circuit), store them in their packed state (because CIRCT is not
221  // expected to care about them right now), or error if we see them
222  // (because they should not exist and are expected to serialize to a
223  // different format).
224  .Case<StringAttr>([&](StringAttr str) -> Optional<Attribute> {
225  // Unpack the string into type and value.
226  StringRef tpe, value;
227  std::tie(tpe, value) = str.getValue().split(":");
228 
229  // These are string-encoded types that are targets in the circuit.
230  // These require annotations to be scattered for them. Replace their
231  // target with an ID and scatter a tracker.
232  if (tpe == "OMReferenceTarget" || tpe == "OMMemberReferenceTarget" ||
233  tpe == "OMMemberInstanceTarget" || tpe == "OMInstanceTarget" ||
234  tpe == "OMDontTouchedReferenceTarget") {
235  auto idAttr = state.newID();
236  NamedAttrList tracker;
237  tracker.append("class", StringAttr::get(ctx, omirTrackerAnnoClass));
238  tracker.append("id", idAttr);
239  tracker.append("target", StringAttr::get(ctx, value));
240 
241  state.addToWorklistFn(DictionaryAttr::get(ctx, tracker));
242 
243  return addID(tpe, value, idAttr);
244  }
245 
246  // The following are types that may exist, but we do not unbox them. At
247  // a later time, we may want to change this behavior and unbox these if
248  // we wind up building out an Object Model dialect:
249  if (isOMIRStringEncodedPassthrough(tpe))
250  return str;
251 
252  // The following types are not expected to exist because they have
253  // serializations to JSON types or are removed during serialization.
254  // Hence, any of the following types are NOT expected to exist and we
255  // error if we see them. These are explicitly specified as opposed to
256  // being handled in the "unknown" catch-all case below because we want
257  // to provide a good error message that a user may be doing something
258  // very weird.
259  if (tpe == "OMMap" || tpe == "OMArray" || tpe == "OMBoolean" ||
260  tpe == "OMInt" || tpe == "OMDouble" || tpe == "OMFrozenTarget") {
261  auto diag =
262  mlir::emitError(state.circuit.getLoc())
263  << "found known string-encoded OMIR type \"" << tpe
264  << "\", but this type should not be seen as it has a defined "
265  "serialization format that does NOT use a string-encoded type";
266  diag.attachNote()
267  << "the problematic OMIR is reproduced here: " << original;
268  return None;
269  }
270 
271  // This is a catch-all for any unknown types.
272  auto diag = mlir::emitError(state.circuit.getLoc())
273  << "found unknown string-encoded OMIR type \"" << tpe
274  << "\" (Did you misspell it? Is CIRCT missing an Object "
275  "Model OMIR type?)";
276  diag.attachNote() << "the problematic OMIR is reproduced here: "
277  << original;
278  return None;
279  })
280  // For an array, just recurse into each element and rewrite the array with
281  // the results.
282  .Case<ArrayAttr>([&](ArrayAttr arr) -> Optional<Attribute> {
283  SmallVector<Attribute> newArr;
284  for (auto element : arr) {
285  auto newElement = scatterOMIR(element, state);
286  if (!newElement)
287  return None;
288  newArr.push_back(newElement.getValue());
289  }
290  return ArrayAttr::get(ctx, newArr);
291  })
292  // For a dictionary, recurse into each value and rewrite the key/value
293  // pairs.
294  .Case<DictionaryAttr>([&](DictionaryAttr dict) -> Optional<Attribute> {
295  NamedAttrList newAttrs;
296  for (auto pairs : dict) {
297  auto maybeValue = scatterOMIR(pairs.getValue(), state);
298  if (!maybeValue)
299  return None;
300  newAttrs.append(pairs.getName(), maybeValue.getValue());
301  }
302  return DictionaryAttr::get(ctx, newAttrs);
303  })
304  // These attributes are all expected. They are OMIR types, but do not
305  // have string-encodings (hence why these should error if we see them as
306  // strings).
307  .Case</* OMBoolean */ BoolAttr, /* OMDouble */ FloatAttr,
308  /* OMInt */ IntegerAttr>(
309  [](auto passThrough) { return passThrough; })
310  // Error if we see anything else.
311  .Default([&](auto) -> Optional<Attribute> {
312  auto diag = mlir::emitError(state.circuit.getLoc())
313  << "found unexpected MLIR attribute \"" << original
314  << "\" while trying to scatter OMIR";
315  return None;
316  });
317 }

319 /// Convert an Object Model Field into an optional pair of a string key and a
320 /// dictionary attribute. Expand internal source locator strings to location
321 /// attributes. Scatter any FIRRTL targets into the circuit. If this is an
322 /// illegal Object Model Field return None.
323 ///
324 /// Each Object Model Field consists of three mandatory members with
325 /// the following names and types:
326 ///
327 /// - "info": Source Locator String
328 /// - "name": String
329 /// - "value": Object Model IR
330 ///
331 /// The key is the "name" and the dictionary consists of the "info" and "value"
332 /// members. Each value is recursively traversed to scatter any FIRRTL targets
333 /// that may be used inside it.
334 ///
335 /// This conversion from an object (dictionary) to key--value pair is safe
336 /// because each Object Model Field in an Object Model Node must have a unique
337 /// "name". Anything else is illegal Object Model.
338 static Optional<std::pair<StringRef, DictionaryAttr>>
339 scatterOMField(Attribute original, const Attribute root, unsigned index,
340  ApplyState &state) {
341  // The input attribute must be a dictionary.
342  DictionaryAttr dict = original.dyn_cast<DictionaryAttr>();
343  if (!dict) {
344  llvm::errs() << "OMField is not a dictionary, but should be: " << original
345  << "\n";
346  return None;
347  }
348 
349  auto loc = state.circuit.getLoc();
350  auto *ctx = state.circuit.getContext();
351 
352  // Generate an arbitrary identifier to use for caching when using
353  // `maybeStringToLocation`.
354  StringAttr locatorFilenameCache = StringAttr::get(ctx, ".");
355  FileLineColLoc fileLineColLocCache;
356 
357  // Convert location from a string to a location attribute.
358  auto infoAttr = tryGetAs<StringAttr>(dict, root, "info", loc, omirAnnoClass);
359  if (!infoAttr)
360  return None;
361  auto maybeLoc =
362  maybeStringToLocation(infoAttr.getValue(), false, locatorFilenameCache,
363  fileLineColLocCache, ctx);
364  mlir::LocationAttr infoLoc;
365  if (maybeLoc.first)
366  infoLoc = maybeLoc.second.getValue();
367  else
368  infoLoc = UnknownLoc::get(ctx);
369 
370  // Extract the name attribute.
371  auto nameAttr = tryGetAs<StringAttr>(dict, root, "name", loc, omirAnnoClass);
372  if (!nameAttr)
373  return None;
374 
375  // The value attribute is unstructured and just copied over.
376  auto valueAttr = tryGetAs<Attribute>(dict, root, "value", loc, omirAnnoClass);
377  if (!valueAttr)
378  return None;
379  auto newValue = scatterOMIR(valueAttr, state);
380  if (!newValue)
381  return None;
382 
383  NamedAttrList values;
384  // We add the index if one was provided. This can be used later to
385  // reconstruct the order of the original array.
386  values.append("index", IntegerAttr::get(IntegerType::get(ctx, 64), index));
387  values.append("info", infoLoc);
388  values.append("value", newValue.getValue());
389 
390  return {{nameAttr.getValue(), DictionaryAttr::getWithSorted(ctx, values)}};
391 }

393 /// Convert an Object Model Node to an optional dictionary, convert source
394 /// locator strings to location attributes, and scatter FIRRTL targets into the
395 /// circuit. If this is an illegal Object Model Node, then return None.
396 ///
397 /// An Object Model Node is expected to look like:
398 ///
399 /// - "info": Source Locator String
400 /// - "id": String-encoded integer ('OMID' ':' Integer)
401 /// - "fields": Array<Object>
402 ///
403 /// The "fields" member may be absent. If so, then construct an empty array.
404 static Optional<DictionaryAttr>
405 scatterOMNode(Attribute original, const Attribute root, ApplyState &state) {
406 
407  auto loc = state.circuit.getLoc();
408 
409  /// The input attribute must be a dictionary.
410  DictionaryAttr dict = original.dyn_cast<DictionaryAttr>();
411  if (!dict) {
412  llvm::errs() << "OMNode is not a dictionary, but should be: " << original
413  << "\n";
414  return None;
415  }
416 
417  NamedAttrList omnode;
418  auto *ctx = state.circuit.getContext();
419 
420  // Generate an arbitrary identifier to use for caching when using
421  // `maybeStringToLocation`.
422  StringAttr locatorFilenameCache = StringAttr::get(ctx, ".");
423  FileLineColLoc fileLineColLocCache;
424 
425  // Convert the location from a string to a location attribute.
426  auto infoAttr = tryGetAs<StringAttr>(dict, root, "info", loc, omirAnnoClass);
427  if (!infoAttr)
428  return None;
429  auto maybeLoc =
430  maybeStringToLocation(infoAttr.getValue(), false, locatorFilenameCache,
431  fileLineColLocCache, ctx);
432  mlir::LocationAttr infoLoc;
433  if (maybeLoc.first)
434  infoLoc = maybeLoc.second.getValue();
435  else
436  infoLoc = UnknownLoc::get(ctx);
437 
438  // Extract the OMID. Don't parse this, just leave it as a string.
439  auto idAttr = tryGetAs<StringAttr>(dict, root, "id", loc, omirAnnoClass);
440  if (!idAttr)
441  return None;
442 
443  // Convert the fields from an ArrayAttr to a DictionaryAttr keyed by their
444  // "name". If no fields member exists, then just create an empty dictionary.
445  // Note that this is safe to construct because all fields must have unique
446  // "name" members relative to each other.
447  auto maybeFields = dict.getAs<ArrayAttr>("fields");
448  DictionaryAttr fields;
449  if (!maybeFields)
450  fields = DictionaryAttr::get(ctx);
451  else {
452  auto fieldAttr = maybeFields.getValue();
453  NamedAttrList fieldAttrs;
454  for (size_t i = 0, e = fieldAttr.size(); i != e; ++i) {
455  auto field = fieldAttr[i];
456  if (auto newField = scatterOMField(field, root, i, state)) {
457  fieldAttrs.append(newField.getValue().first,
458  newField.getValue().second);
459  continue;
460  }
461  return None;
462  }
463  fields = DictionaryAttr::get(ctx, fieldAttrs);
464  }
465 
466  omnode.append("fields", fields);
467  omnode.append("id", idAttr);
468  omnode.append("info", infoLoc);
469 
470  return DictionaryAttr::getWithSorted(ctx, omnode);
471 }

473 /// Main entry point to handle scattering of an OMIRAnnotation. Return the
474 /// modified optional attribute on success and None on failure. Any scattered
475 /// annotations will be added to the reference argument `newAnnotations`.
476 LogicalResult circt::firrtl::applyOMIR(const AnnoPathValue &target,
477  DictionaryAttr anno, ApplyState &state) {
478 
479  auto loc = state.circuit.getLoc();
480 
481  auto nodes = tryGetAs<ArrayAttr>(anno, anno, "nodes", loc, omirAnnoClass);
482  if (!nodes)
483  return failure();
484 
485  SmallVector<Attribute> newNodes;
486  for (auto node : nodes) {
487  auto newNode = scatterOMNode(node, anno, state);
488  if (!newNode)
489  return failure();
490  newNodes.push_back(newNode.getValue());
491  }
492 
493  auto *ctx = state.circuit.getContext();
494 
495  NamedAttrList newAnnotation;
496  newAnnotation.append("class", StringAttr::get(ctx, omirAnnoClass));
497  newAnnotation.append("nodes", ArrayAttr::get(ctx, newNodes));
498 
499  AnnotationSet annotations(state.circuit);
500  annotations.addAnnotations(DictionaryAttr::get(ctx, newAnnotation));
501  annotations.applyToOperation(state.circuit);
502 
503  return success();
504 }

510 void EmitOMIRPass::runOnOperation() {
511  MLIRContext *context = &getContext();
512  anyFailures = false;
513  circuitNamespace = nullptr;
514  instanceGraph = nullptr;
515  instancePaths = nullptr;
516  trackers.clear();
517  symbols.clear();
518  symbolIndices.clear();
519  removeTempNLAs.clear();
520  moduleNamespaces.clear();
521  instancesByName.clear();
522  CircuitOp circuitOp = getOperation();
523 
524  // Gather the relevant annotations from the circuit. On the one hand these are
525  // all the actual `OMIRAnnotation`s that need processing and emission, as well
526  // as an optional `OMIRFileAnnotation` that overrides the default OMIR output
527  // file. Also while we're at it, keep track of all OMIR nodes that qualify as
528  // an SRAM and that require their trackers to be turned into NLAs starting at
529  // the root of the hierarchy.
530  SmallVector<ArrayRef<Attribute>> annoNodes;
531  DenseSet<Attribute> sramIDs;
532  Optional<StringRef> outputFilename = {};
533 
534  AnnotationSet::removeAnnotations(circuitOp, [&](Annotation anno) {
535  if (anno.isClass(omirFileAnnoClass)) {
536  auto pathAttr = anno.getMember<StringAttr>("filename");
537  if (!pathAttr) {
538  circuitOp.emitError(omirFileAnnoClass)
539  << " annotation missing `filename` string attribute";
540  anyFailures = true;
541  return true;
542  }
543  LLVM_DEBUG(llvm::dbgs() << "- OMIR path: " << pathAttr << "\n");
544  outputFilename = pathAttr.getValue();
545  return true;
546  }
547  if (anno.isClass(omirAnnoClass)) {
548  auto nodesAttr = anno.getMember<ArrayAttr>("nodes");
549  if (!nodesAttr) {
550  circuitOp.emitError(omirAnnoClass)
551  << " annotation missing `nodes` array attribute";
552  anyFailures = true;
553  return true;
554  }
555  LLVM_DEBUG(llvm::dbgs() << "- OMIR: " << nodesAttr << "\n");
556  annoNodes.push_back(nodesAttr.getValue());
557  for (auto node : nodesAttr) {
558  if (auto id = isOMSRAM(node)) {
559  LLVM_DEBUG(llvm::dbgs() << " - is SRAM with tracker " << id << "\n");
560  sramIDs.insert(id);
561  }
562  }
563  return true;
564  }
565  return false;
566  });
567  if (anyFailures)
568  return signalPassFailure();
569 
570  // If an OMIR output filename has been specified as a pass parameter, override
571  // whatever the annotations have configured. If neither are specified we just
572  // bail.
573  if (!this->outputFilename.empty())
574  outputFilename = this->outputFilename;
575  if (!outputFilename) {
576  LLVM_DEBUG(llvm::dbgs() << "Not emitting OMIR because no annotation or "
577  "pass parameter specified an output file\n");
578  markAllAnalysesPreserved();
579  return;
580  }
581  // Establish some of the analyses we need throughout the pass.
582  CircuitNamespace currentCircuitNamespace(circuitOp);
583  InstanceGraph &currentInstanceGraph = getAnalysis<InstanceGraph>();
584  nlaTable = &getAnalysis<NLATable>();
585  InstancePathCache currentInstancePaths(currentInstanceGraph);
586  circuitNamespace = &currentCircuitNamespace;
587  instanceGraph = &currentInstanceGraph;
588  instancePaths = &currentInstancePaths;
589  dutModuleName = {};
590 
591  // Traverse the IR and collect all tracker annotations that were previously
592  // scattered into the circuit.
593  circuitOp.walk([&](Operation *op) {
594  if (auto instOp = dyn_cast<InstanceOp>(op)) {
595  // This instance does not have a symbol, but we are adding one. Remove it
596  // after the pass.
597  if (!op->getAttr(InnerSymbolTable::getInnerSymbolAttrName()))
598  tempSymInstances.insert(instOp);
599 
600  instancesByName.insert({getInnerRefTo(op), instOp});
601  }
602  auto setTracker = [&](int portNo, Annotation anno) {
603  if (!anno.isClass(omirTrackerAnnoClass))
604  return false;
605  Tracker tracker;
606  tracker.op = op;
607  tracker.id = anno.getMember<IntegerAttr>("id");
608  tracker.portNo = portNo;
609  if (!tracker.id) {
610  op->emitError(omirTrackerAnnoClass)
611  << " annotation missing `id` integer attribute";
612  anyFailures = true;
613  return true;
614  }
615  if (auto nlaSym = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal")) {
616  auto tmp = nlaTable->getNLA(nlaSym.getAttr());
617  if (!tmp) {
618  op->emitError("missing annotation ") << nlaSym.getValue();
619  anyFailures = true;
620  return true;
621  }
622  tracker.nla = cast<HierPathOp>(tmp);
623  }
624  if (sramIDs.erase(tracker.id))
625  makeTrackerAbsolute(tracker);
626  trackers.insert({tracker.id, tracker});
627  return true;
628  };
629  AnnotationSet::removePortAnnotations(op, setTracker);
630  AnnotationSet::removeAnnotations(
631  op, std::bind(setTracker, -1, std::placeholders::_1));
632  if (auto modOp = dyn_cast<FModuleOp>(op)) {
633  AnnotationSet annos(modOp.annotations());
634  if (!annos.hasAnnotation("sifive.enterprise.firrtl.MarkDUTAnnotation"))
635  return;
636  dutModuleName = modOp.getNameAttr();
637  }
638  });
639 
640  // Build the output JSON.
641  std::string jsonBuffer;
642  llvm::raw_string_ostream jsonOs(jsonBuffer);
643  llvm::json::OStream json(jsonOs, 2);
644  json.array([&] {
645  for (auto nodes : annoNodes) {
646  for (auto node : nodes) {
647  emitOMNode(node, json);
648  if (anyFailures)
649  return;
650  }
651  }
652  });
653  if (anyFailures)
654  return signalPassFailure();
655 
656  // Drop temporary (and sometimes invalid) NLA's created during the pass:
657  for (auto nla : removeTempNLAs) {
658  LLVM_DEBUG(llvm::dbgs() << "Removing '" << nla << "'\n");
659  nlaTable->erase(nla);
660  nla.erase();
661  }
662  removeTempNLAs.clear();
663 
664  // Remove the temp symbol from instances.
665  for (auto *op : tempSymInstances)
666  cast<InstanceOp>(op)->removeAttr("inner_sym");
667  tempSymInstances.clear();
668 
669  // Emit the OMIR JSON as a verbatim op.
670  auto builder = OpBuilder(circuitOp);
671  builder.setInsertionPointAfter(circuitOp);
672  auto verbatimOp =
673  builder.create<sv::VerbatimOp>(builder.getUnknownLoc(), jsonBuffer);
674  auto fileAttr = hw::OutputFileAttr::getFromFilename(
675  context, *outputFilename, /*excludeFromFilelist=*/true, false);
676  verbatimOp->setAttr("output_file", fileAttr);
677  verbatimOp.symbolsAttr(ArrayAttr::get(context, symbols));
678 
679  markAnalysesPreserved<NLATable>();
680 }

682 /// Make a tracker absolute by adding an NLA to it which starts at the root
683 /// module of the circuit. Generates an error if any module along the path is
684 /// instantiated multiple times.
685 void EmitOMIRPass::makeTrackerAbsolute(Tracker &tracker) {
686  auto builder = OpBuilder::atBlockBegin(getOperation().getBody());
687 
688  // Pick a name for the NLA that doesn't collide with anything.
689  StringAttr opName;
690  if (auto module = dyn_cast<FModuleLike>(tracker.op))
691  opName = module.moduleNameAttr();
692  else
693  opName = tracker.op->getAttrOfType<StringAttr>("name");
694  auto nlaName = circuitNamespace->newName("omir_nla_" + opName.getValue());
695 
696  // Get all the paths instantiating this module. If there is an NLA already
697  // attached to this tracker, we use it as a base to disambiguate the path to
698  // the memory.
699  Operation *mod;
700  if (tracker.nla)
701  mod = instanceGraph->lookup(tracker.nla.root())->getModule();
702  else
703  mod = tracker.op->getParentOfType<FModuleOp>();
704 
705  // Get all the paths instantiating this module.
706  auto paths = instancePaths->getAbsolutePaths(mod);
707  if (paths.empty()) {
708  tracker.op->emitError("OMIR node targets uninstantiated component `")
709  << opName.getValue() << "`";
710  anyFailures = true;
711  return;
712  }
713  if (paths.size() > 1) {
714  auto diag = tracker.op->emitError("OMIR node targets ambiguous component `")
715  << opName.getValue() << "`";
716  diag.attachNote(tracker.op->getLoc())
717  << "may refer to the following paths:";
718  for (auto path : paths)
719  formatInstancePath(diag.attachNote(tracker.op->getLoc()) << "- ", path);
720  anyFailures = true;
721  return;
722  }
723 
724  // Assemble the module and name path for the NLA. Also attach an NLA reference
725  // annotation to each instance participating in the path.
726  SmallVector<Attribute> namepath;
727  auto addToPath = [&](Operation *op, StringAttr name) {
728  namepath.push_back(getInnerRefTo(op));
729  };
730  // Add the path up to where the NLA starts.
731  for (InstanceOp inst : paths[0])
732  addToPath(inst, inst.nameAttr());
733  // Add the path from the NLA to the op.
734  if (tracker.nla) {
735  auto path = tracker.nla.namepath().getValue();
736  for (auto attr : path.drop_back()) {
737  auto ref = attr.cast<hw::InnerRefAttr>();
738  // Find the instance referenced by the NLA.
739  auto *node = instanceGraph->lookup(ref.getModule());
740  auto it = llvm::find_if(*node, [&](hw::InstanceRecord *record) {
741  return getInnerSymName(cast<InstanceOp>(*record->getInstance())) ==
742  ref.getName();
743  });
744  assert(it != node->end() &&
745  "Instance referenced by NLA does not exist in module");
746  addToPath((*it)->getInstance(), ref.getName());
747  }
748  }
749 
750  // TODO: Don't create NLA if namepath is empty
751  // (care needed to ensure this will be handled correctly elsewhere)
752 
753  // Add the op itself.
754  if (auto module = dyn_cast<FModuleLike>(tracker.op))
755  namepath.push_back(FlatSymbolRefAttr::get(module.moduleNameAttr()));
756  else
757  namepath.push_back(getInnerRefTo(tracker.op));
758 
759  // Add the NLA to the tracker and mark it to be deleted later.
760  tracker.nla = builder.create<HierPathOp>(builder.getUnknownLoc(),
761  builder.getStringAttr(nlaName),
762  builder.getArrayAttr(namepath));
763  nlaTable->addNLA(tracker.nla);
764 
765  removeTempNLAs.push_back(tracker.nla);
766 }

768 /// Emit a source locator into a string, for inclusion in the `info` field of
769 /// `OMNode` and `OMField`.
770 void EmitOMIRPass::emitSourceInfo(Location input, SmallString<64> &into) {
771  into.clear();
772  input->walk([&](Location loc) {
773  if (FileLineColLoc fileLoc = loc.dyn_cast<FileLineColLoc>()) {
774  into.append(into.empty() ? "@[" : " ");
775  (Twine(fileLoc.getFilename()) + " " + Twine(fileLoc.getLine()) + ":" +
776  Twine(fileLoc.getColumn()))
777  .toVector(into);
778  }
779  return WalkResult::advance();
780  });
781  if (!into.empty())
782  into.append("]");
783  else
784  into.append("UnlocatableSourceInfo");

787 /// Emit an entire `OMNode` as JSON.
788 void EmitOMIRPass::emitOMNode(Attribute node, llvm::json::OStream &jsonStream) {
789  auto dict = node.dyn_cast<DictionaryAttr>();
790  if (!dict) {
791  getOperation()
792  .emitError("OMNode must be a dictionary")
793  .attachNote(getOperation().getLoc())
794  << node;
795  anyFailures = true;
796  return;
797  }
798 
799  // Extract the `info` field and serialize the location.
800  SmallString<64> info;
801  if (auto infoAttr = dict.getAs<LocationAttr>("info"))
802  emitSourceInfo(infoAttr, info);
803  if (anyFailures)
804  return;
805 
806  // Extract the `id` field.
807  auto idAttr = dict.getAs<StringAttr>("id");
808  if (!idAttr) {
809  getOperation()
810  .emitError("OMNode missing `id` string field")
811  .attachNote(getOperation().getLoc())
812  << dict;
813  anyFailures = true;
814  return;
815  }
816 
817  // Extract and order the fields of this node.
818  SmallVector<std::tuple<unsigned, StringAttr, DictionaryAttr>> orderedFields;
819  auto fieldsDict = dict.getAs<DictionaryAttr>("fields");
820  if (fieldsDict) {
821  for (auto nameAndField : fieldsDict.getValue()) {
822  auto fieldDict = nameAndField.getValue().dyn_cast<DictionaryAttr>();
823  if (!fieldDict) {
824  getOperation()
825  .emitError("OMField must be a dictionary")
826  .attachNote(getOperation().getLoc())
827  << nameAndField.getValue();
828  anyFailures = true;
829  return;
830  }
831 
832  unsigned index = 0;
833  if (auto indexAttr = fieldDict.getAs<IntegerAttr>("index"))
834  index = indexAttr.getValue().getLimitedValue();
835 
836  orderedFields.push_back({index, nameAndField.getName(), fieldDict});
837  }
838  llvm::sort(orderedFields,
839  [](auto a, auto b) { return std::get<0>(a) < std::get<0>(b); });
840  }
841 
842  jsonStream.object([&] {
843  jsonStream.attribute("info", info);
844  jsonStream.attribute("id", idAttr.getValue());
845  jsonStream.attributeArray("fields", [&] {
846  for (auto &orderedField : orderedFields) {
847  emitOMField(std::get<1>(orderedField), std::get<2>(orderedField),
848  jsonStream);
849  if (anyFailures)
850  return;
851  }
852  if (auto node = fieldsDict.getAs<DictionaryAttr>("containingModule"))
853  if (auto value = node.getAs<DictionaryAttr>("value"))
854  emitOptionalRTLPorts(value, jsonStream);
855  });
856  });
857 }

859 /// Emit a single `OMField` as JSON. This expects the field's name to be
860 /// provided from the outside, for example as the field name that this attribute
861 /// has in the surrounding dictionary.
862 void EmitOMIRPass::emitOMField(StringAttr fieldName, DictionaryAttr field,
863  llvm::json::OStream &jsonStream) {

865  auto infoAttr = field.getAs<LocationAttr>("info");
866  SmallString<64> info;
867  if (infoAttr)
868  emitSourceInfo(infoAttr, info);
869  if (anyFailures)
870  return;
871 
872  jsonStream.object([&] {
873  jsonStream.attribute("info", info);
874  jsonStream.attribute("name", fieldName.strref());
875  jsonStream.attributeBegin("value");
876  emitValue(field.get("value"), jsonStream,
877  fieldName.strref().equals("dutInstance"));
878  jsonStream.attributeEnd();

882 // If the given `node` refers to a valid tracker in the IR, gather the
883 // additional port metadata of the module it refers to. Then emit this port
884 // metadata as a `ports` array field for the surrounding `OMNode`.
885 void EmitOMIRPass::emitOptionalRTLPorts(DictionaryAttr node,
886  llvm::json::OStream &jsonStream) {
887  // First make sure we actually have a valid tracker. If not, just silently
888  // abort and don't emit any port metadata.
889  auto idAttr = node.getAs<IntegerAttr>("id");
890  auto trackerIt = trackers.find(idAttr);
891  if (!idAttr || !node.getAs<UnitAttr>("omir.tracker") ||
892  trackerIt == trackers.end())
893  return;
894  auto tracker = trackerIt->second;
895 
896  // Lookup the module the tracker refers to. If it points at something *within*
897  // a module, go dig up the surrounding module. This is roughly what
898  // `Target.referringModule(...)` does on the Scala side.
899  auto module = dyn_cast<FModuleLike>(tracker.op);
900  if (!module)
901  module = tracker.op->getParentOfType<FModuleLike>();
902  if (!module) {
903  LLVM_DEBUG(llvm::dbgs() << "Not emitting RTL ports since tracked operation "
904  "does not have a FModuleLike parent: "
905  << *tracker.op << "\n");
906  return;
907  }
908  LLVM_DEBUG(llvm::dbgs() << "Emitting RTL ports for module `"
909  << module.moduleName() << "`\n");
910 
911  // Emit the JSON.
912  SmallString<64> buf;
913  jsonStream.object([&] {
914  buf.clear();
915  emitSourceInfo(module.getLoc(), buf);
916  jsonStream.attribute("info", buf);
917  jsonStream.attribute("name", "ports");
918  jsonStream.attributeArray("value", [&] {
919  for (const auto &port : llvm::enumerate(module.getPorts())) {
920  if (port.value().type.getBitWidthOrSentinel() == 0)
921  continue;
922  jsonStream.object([&] {
923  // Emit the `ref` field.
924  buf.assign("OMDontTouchedReferenceTarget:~");
925  if (module.moduleNameAttr() == dutModuleName) {
926  // If module is DUT, then root the target relative to the DUT.
927  buf.append(module.moduleName());
928  } else {
929  buf.append(getOperation().name());
930  }
931  buf.push_back('|');
932  buf.append(addSymbol(module));
933  buf.push_back('>');
934  buf.append(addSymbol(getInnerRefTo(module, port.index())));
935  jsonStream.attribute("ref", buf);
936 
937  // Emit the `direction` field.
938  buf.assign("OMString:");
939  buf.append(port.value().isOutput() ? "Output" : "Input");
940  jsonStream.attribute("direction", buf);
941 
942  // Emit the `width` field.
943  buf.assign("OMBigInt:");
944  Twine::utohexstr(port.value().type.getBitWidthOrSentinel())
945  .toVector(buf);
946  jsonStream.attribute("width", buf);
947  });
948  }
949  });
950  });
951 }

953 void EmitOMIRPass::emitValue(Attribute node, llvm::json::OStream &jsonStream,
954  bool dutInstance) {
955  // Handle the null case.
956  if (!node || node.isa<UnitAttr>())
957  return jsonStream.value(nullptr);
958 
959  // Handle the trivial cases where the OMIR serialization simply uses the
960  // builtin JSON types.
961  if (auto attr = node.dyn_cast<BoolAttr>())
962  return jsonStream.value(attr.getValue()); // OMBoolean
963  if (auto attr = node.dyn_cast<IntegerAttr>()) {
964  // CAVEAT: We expect these integers to come from an OMIR file that is
965  // initially read in from JSON, where they are i32 or i64, so this should
966  // yield a valid value. However, a user could cook up an arbitrary precision
967  // integer attr in MLIR input and then subtly break the JSON spec.
968  SmallString<16> val;
969  attr.getValue().toStringSigned(val);
970  return jsonStream.rawValue(val); // OMInt
971  }
972  if (auto attr = node.dyn_cast<FloatAttr>()) {
973  // CAVEAT: We expect these floats to come from an OMIR file that is
974  // initially read in from JSON, where they are f32 or f64, so this should
975  // yield a valid value. However, a user could cook up an arbitrary precision
976  // float attr in MLIR input and then subtly break the JSON spec.
977  SmallString<16> val;
978  attr.getValue().toString(val);
979  return jsonStream.rawValue(val); // OMDouble
980  }
981 
982  // Handle aggregate types.
983  if (auto attr = node.dyn_cast<ArrayAttr>()) {
984  jsonStream.array([&] {
985  for (auto element : attr.getValue()) {
986  emitValue(element, jsonStream, dutInstance);
987  if (anyFailures)
988  return;
989  }
990  });
991  return;
992  }
993  if (auto attr = node.dyn_cast<DictionaryAttr>()) {
994  // Handle targets that have a corresponding tracker annotation in the IR.
995  if (attr.getAs<UnitAttr>("omir.tracker"))
996  return emitTrackedTarget(attr, jsonStream, dutInstance);
997 
998  // Handle regular dictionaries.
999  jsonStream.object([&] {
1000  for (auto field : attr.getValue()) {
1001  jsonStream.attributeBegin(field.getName());
1002  emitValue(field.getValue(), jsonStream, dutInstance);
1003  jsonStream.attributeEnd();
1004  if (anyFailures)
1005  return;
1006  }
1007  });
1008  return;
1009  }
1010 
1011  // The remaining types are all simple string-encoded pass-through cases.
1012  if (auto attr = node.dyn_cast<StringAttr>()) {
1013  StringRef val = attr.getValue();
1014  if (isOMIRStringEncodedPassthrough(val.split(":").first))
1015  return jsonStream.value(val);
1016  }
1017 
1018  // If we get here, we don't know how to serialize the given MLIR attribute as
1019  // a OMIR value.
1020  jsonStream.value("<unsupported value>");
1021  getOperation().emitError("unsupported attribute for OMIR serialization: `")
1022  << node << "`";
1023  anyFailures = true;
1024 }

1026 void EmitOMIRPass::emitTrackedTarget(DictionaryAttr node,
1027  llvm::json::OStream &jsonStream,
1028  bool dutInstance) {
1029  // Extract the `id` field.
1030  auto idAttr = node.getAs<IntegerAttr>("id");
1031  if (!idAttr) {
1032  getOperation()
1033  .emitError("tracked OMIR target missing `id` string field")
1034  .attachNote(getOperation().getLoc())
1035  << node;
1036  anyFailures = true;
1037  return jsonStream.value("<error>");
1038  }
1039 
1040  // Extract the `type` field.
1041  auto typeAttr = node.getAs<StringAttr>("type");
1042  if (!typeAttr) {
1043  getOperation()
1044  .emitError("tracked OMIR target missing `type` string field")
1045  .attachNote(getOperation().getLoc())
1046  << node;
1047  anyFailures = true;
1048  return jsonStream.value("<error>");
1049  }
1050  StringRef type = typeAttr.getValue();
1051 
1052  // Find the tracker for this target, and handle the case where the tracker has
1053  // been deleted.
1054  auto trackerIt = trackers.find(idAttr);
1055  if (trackerIt == trackers.end()) {
1056  // Some of the target types indicate removal of the target through an
1057  // `OMDeleted` node.
1058  if (type == "OMReferenceTarget" || type == "OMMemberReferenceTarget" ||
1059  type == "OMMemberInstanceTarget")
1060  return jsonStream.value("OMDeleted:");
1061 
1062  // The remaining types produce an error upon removal of the target.
1063  auto diag = getOperation().emitError("tracked OMIR target of type `")
1064  << type << "` was deleted";
1065  diag.attachNote(getOperation().getLoc())
1066  << "`" << type << "` should never be deleted";
1067  if (auto path = node.getAs<StringAttr>("path"))
1068  diag.attachNote(getOperation().getLoc())
1069  << "original path: `" << path.getValue() << "`";
1070  anyFailures = true;
1071  return jsonStream.value("<error>");
1072  }
1073  auto tracker = trackerIt->second;
1074 
1075  // In case this is an `OMMemberTarget`, handle the case where the component
1076  // used to be a "reference target" (wire, register, memory, node) when the
1077  // OMIR was read in, but has been change to an "instance target" during the
1078  // execution of the compiler. This mainly occurs during mapping of
1079  // `firrtl.mem` operations to a corresponding `firrtl.instance`.
1080  if (type == "OMMemberReferenceTarget" && isa<InstanceOp, MemOp>(tracker.op))
1081  type = "OMMemberInstanceTarget";
1082 
1083  // Serialize the target circuit first.
1084  SmallString<64> target(type);
1085  target.append(":~");
1086  target.append(getOperation().name());
1087  target.push_back('|');
1088 
1089  // Serialize the local or non-local module/instance hierarchy path.
1090  if (tracker.nla) {
1091  bool notFirst = false;
1092  hw::InnerRefAttr instName;
1093  for (auto nameRef : tracker.nla.namepath()) {
1094  StringAttr modName;
1095  if (auto innerRef = nameRef.dyn_cast<hw::InnerRefAttr>())
1096  modName = innerRef.getModule();
1097  else if (auto ref = nameRef.dyn_cast<FlatSymbolRefAttr>())
1098  modName = ref.getAttr();
1099  if (!dutInstance && modName == dutModuleName) {
1100  // Check if the DUT module occurs in the instance path.
1101  // Print the path relative to the DUT, if the nla is inside the DUT.
1102  // Keep the path for dutInstance relative to test harness. (SFC
1103  // implementation in TestHarnessOMPhase.scala)
1104  target = type;
1105  target.append(":~");
1106  target.append(dutModuleName);
1107  target.push_back('|');
1108  notFirst = false;
1109  instName = {};
1110  }
1111 
1112  Operation *module = nlaTable->getModule(modName);
1113  assert(module);
1114  if (notFirst)
1115  target.push_back('/');
1116  notFirst = true;
1117  if (instName) {
1118  target.append(addSymbol(instName));
1119  target.push_back(':');
1120  }
1121  target.append(addSymbol(module));
1122 
1123  if (auto innerRef = nameRef.dyn_cast<hw::InnerRefAttr>()) {
1124  // Find an instance with the given name in this module. Ensure it has a
1125  // symbol that we can refer to.
1126  auto instOp = instancesByName.lookup(innerRef);
1127  if (!instOp)
1128  continue;
1129  LLVM_DEBUG(llvm::dbgs() << "Marking NLA-participating instance "
1130  << innerRef.getName() << " in module "
1131  << modName << " as dont-touch\n");
1132  tempSymInstances.erase(instOp);
1133  instName = getInnerRefTo(instOp);
1134  }
1135  }
1136  } else {
1137  FModuleOp module = dyn_cast<FModuleOp>(tracker.op);
1138  if (!module)
1139  module = tracker.op->getParentOfType<FModuleOp>();
1140  assert(module);
1141  if (module.getNameAttr() == dutModuleName) {
1142  // If module is DUT, then root the target relative to the DUT.
1143  target = type;
1144  target.append(":~");
1145  target.append(dutModuleName);
1146  target.push_back('|');
1147  }
1148  target.append(addSymbol(module));
1149  }
1150 
1151  // Serialize any potential component *inside* the module that this target may
1152  // specifically refer to.
1153  hw::InnerRefAttr componentName;
1154  if (isa<WireOp, RegOp, RegResetOp, InstanceOp, NodeOp, MemOp>(tracker.op)) {
1155  tempSymInstances.erase(tracker.op);
1156  componentName = getInnerRefTo(tracker.op);
1157  LLVM_DEBUG(llvm::dbgs() << "Marking OMIR-targeted " << componentName
1158  << " as dont-touch\n");
1159  } else if (auto mod = dyn_cast<FModuleLike>(tracker.op)) {
1160  if (tracker.portNo >= 0)
1161  componentName = getInnerRefTo(mod, tracker.portNo);
1162  } else if (!isa<FModuleLike>(tracker.op)) {
1163  tracker.op->emitError("invalid target for `") << type << "` OMIR";
1164  anyFailures = true;
1165  return jsonStream.value("<error>");
1166  }
1167  if (componentName) {
1168  // Check if the targeted component is going to be emitted as an instance.
1169  // This is trivially the case for `InstanceOp`s, but also for `MemOp`s that
1170  // get converted to an instance during lowering to HW dialect and generator
1171  // callout.
1172  [&] {
1173  if (type == "OMMemberInstanceTarget") {
1174  if (auto instOp = dyn_cast<InstanceOp>(tracker.op)) {
1175  target.push_back('/');
1176  target.append(addSymbol(componentName));
1177  target.push_back(':');
1178  target.append(addSymbol(instOp.moduleNameAttr()));
1179  return;
1180  }
1181  if (auto memOp = dyn_cast<MemOp>(tracker.op)) {
1182  target.push_back('/');
1183  target.append(addSymbol(componentName));
1184  target.push_back(':');
1185  target.append(memOp.getSummary().getFirMemoryName());
1186  return;
1187  }
1188  }
1189  target.push_back('>');
1190  target.append(addSymbol(componentName));
1191  }();
1192  }
1193 
1194  jsonStream.value(target);
1195 }
