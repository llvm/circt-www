38  int sendTypeMaxSize,

41  Lock g(m);
42  if (endpoints.find(epId) != endpoints.end()) {
43  fprintf(stderr, "Endpoint ID already exists!\n");
44  return false;
45  }
46  // The following ugliness adds an Endpoint to the map of Endpoints. The
47  // Endpoint class has its copy constructor deleted, thus the metaprogramming.
48  endpoints.emplace(std::piecewise_construct,
49  // Map key.
50  std::forward_as_tuple(epId),
51  // Endpoint constructor args.
52  std::forward_as_tuple(sendTypeId, sendTypeMaxSize,
53  recvTypeId, recvTypeMaxSize));

55 }
