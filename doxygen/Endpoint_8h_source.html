26 /// Implements a bi-directional, thread-safe bridge between the RPC server and
27 /// DPI functions.
28 ///
29 /// Several of the methods below are inline with the declaration to make them
30 /// candidates for inlining during compilation. This is particularly important
31 /// on the simulation side since polling happens at each clock and we do not
32 /// want to slow down the simulation any more than necessary.
33 class Endpoint {
34 public:
35  /// Representing messages as shared pointers to vectors may be a performance
36  /// issue in the future but it is the easiest way to ensure memory
37  /// correctness.
38  using Blob = std::vector<uint8_t>;
39  using BlobPtr = std::shared_ptr<Blob>;
40 
41  /// Construct an endpoint which knows and the type IDs in both directions.
42  Endpoint(uint64_t sendTypeId, int sendTypeMaxSize, uint64_t recvTypeId,
43  int recvTypeMaxSize);
44  ~Endpoint();
45  /// Disallow copying. There is only ONE endpoint object per logical endpoint
46  /// so copying is almost always a bug.
47  Endpoint(const Endpoint &) = delete;
48 
49  uint64_t getSendTypeId() const { return sendTypeId; }
50  uint64_t getRecvTypeId() const { return recvTypeId; }
51 
52  /// These two are used to set and unset the inUse flag, to ensure that an open
53  /// endpoint is not opened again.
54  bool setInUse();
55  void returnForUse();
56 
57  /// Queue message to the simulation.
58  void pushMessageToSim(BlobPtr msg) {
59  Lock g(m);
60  toCosim.push(msg);
61  }
62 
63  /// Pop from the to-simulator queue. Return true if there was a message in the
64  /// queue.
65  bool getMessageToSim(BlobPtr &msg) {
66  Lock g(m);
67  if (toCosim.empty())
68  return false;
69  msg = toCosim.front();
70  toCosim.pop();
71  return true;
72  }
73 
74  /// Queue message to the RPC client.
75  void pushMessageToClient(BlobPtr msg) {
76  Lock g(m);
77  toClient.push(msg);
78  }
79 
80  /// Pop from the to-RPC-client queue. Return true if there was a message in
81  /// the queue.
82  bool getMessageToClient(BlobPtr &msg) {
83  Lock g(m);
84  if (toClient.empty())
85  return false;
86  msg = toClient.front();
87  toClient.pop();
88  return true;
89  }
90 
91 private:
92  const uint64_t sendTypeId;
93  const uint64_t recvTypeId;
94  bool inUse;
95 
96  using Lock = std::lock_guard<std::mutex>;
97 
98  /// This class needs to be thread-safe. All of the mutable member variables
99  /// are protected with this object-wide lock. This may be a performance issue
100  /// in the future.
101  std::mutex m;
102  /// Message queue from RPC client to the simulation.
103  std::queue<BlobPtr> toCosim;
104  /// Message queue to RPC client from the simulation.
105  std::queue<BlobPtr> toClient;
106 };

108 /// The Endpoint registry is where Endpoints report their existence (register)
109 /// and they are looked up by RPC clients.
110 class EndpointRegistry {
111 public:
112  /// Register an Endpoint. Creates the Endpoint object and owns it. Returns
113  /// false if unsuccessful.
114  bool registerEndpoint(int epId, uint64_t sendTypeId, int sendTypeMaxSize,
115  uint64_t recvTypeId, int recvTypeMaxSize);
116 
117  /// Get the specified endpoint. Return nullptr if it does not exist. This
118  /// method is defined inline so it can be inlined at compile time. Performance
119  /// is important here since this method is used in the polling call from the
120  /// simulator. Returns nullptr if the endpoint cannot be found.
121  Endpoint *operator[](int epId) {
122  Lock g(m);
123  auto it = endpoints.find(epId);
124  if (it == endpoints.end())
125  return nullptr;
126  return &it->second;
127  }
128 
129  /// Iterate over the list of endpoints, calling the provided function for each
130  /// endpoint.
131  void iterateEndpoints(std::function<void(int id, const Endpoint &)> f) const;
132  /// Return the number of endpoints.
133  size_t size() const;
134 
135 private:
136  using Lock = std::lock_guard<std::mutex>;
137 
138  /// This object needs to be thread-safe. An object-wide mutex is sufficient.
139  std::mutex m;
140 
141  /// Endpoint ID to object pointer mapping.
142  std::map<int, Endpoint> endpoints;
143 };
