67 int Engine::simulate(int n, uint64_t maxTime) {
68  assert(engine && "engine not found");
69  assert(state && "state not found");
70 
71  auto tm = static_cast<TraceMode>(traceMode);
72  Trace trace(state, out, tm);
73 
74  SmallVector<void *, 1> arg({&state});
75  // Initialize tbe simulation state.
76  auto invocationResult = engine->invokePacked("llhd_init", arg);
77  if (invocationResult) {
78  llvm::errs() << "Failed invocation of llhd_init: " << invocationResult;
79  return -1;
80  }
81 
82  if (traceMode != TraceMode::None) {
83  // Add changes for all the signals' initial values.
84  for (size_t i = 0, e = state->signals.size(); i < e; ++i) {
85  trace.addChange(i);
86  }
87  }
88 
89  // Add a dummy event to get the simulation started.
90  state->queue.push_back(Slot(Time()));
91  ++state->queue.events;
92 
93  // Keep track of the instances that need to wakeup.
94  llvm::SmallVector<unsigned, 8> wakeupQueue;
95 
96  // Add all instances to the wakeup queue for the first run and add the jitted
97  // function pointers to all of the instances to make them readily available.
98  for (size_t i = 0, e = state->instances.size(); i < e; ++i) {
99  wakeupQueue.push_back(i);
100  auto &inst = state->instances[i];
101  auto expectedFPtr = engine->lookupPacked(inst.unit);
102  if (!expectedFPtr) {
103  llvm::errs() << "Could not lookup " << inst.unit << "!\n";
104  return -1;
105  }
106  inst.unitFPtr = *expectedFPtr;
107  }
108 
109  int cycle = 0;
110  while (state->queue.events > 0) {
111  const auto &pop = state->queue.top();
112 
113  // Interrupt the simulation if a stop condition is met.
114  if ((n > 0 && cycle >= n) ||
115  (maxTime > 0 && pop.time.getTime() > maxTime)) {
116  break;
117  }
118 
119  // Update the simulation time.
120  state->time = pop.time;
121 
122  if (traceMode != TraceMode::None)
123  trace.flush();
124 
125  // Process signal changes.
126  size_t i = 0, e = pop.changesSize;
127  while (i < e) {
128  const auto sigIndex = pop.changes[i].first;
129  auto &curr = state->signals[sigIndex];
130  APInt buff(
131  curr.getSize() * 8,
132  llvm::makeArrayRef(reinterpret_cast<uint64_t *>(curr.getValue()),
133  llvm::divideCeil(curr.getSize(), 8)));
134 
135  // Apply the changes to the buffer until we reach the next signal.
136  while (i < e && pop.changes[i].first == sigIndex) {
137  const auto &change = pop.buffers[pop.changes[i].second];
138  const auto offset = change.first;
139  const auto &drive = change.second;
140  if (drive.getBitWidth() < buff.getBitWidth())
141  buff.insertBits(drive, offset);
142  else
143  buff = drive;
144 
145  ++i;
146  }
147 
148  if (!curr.updateWhenChanged(buff.getRawData()))
149  continue;
150 
151  // Add sensitive instances.
152  for (auto inst : curr.getTriggeredInstanceIndices()) {
153  // Skip if the process is not currently sensible to the signal.
154  if (!state->instances[inst].isEntity) {
155  const auto &sensList = state->instances[inst].sensitivityList;
156  auto it = std::find_if(sensList.begin(), sensList.end(),
157  [sigIndex](const SignalDetail &sig) {
158  return sig.globalIndex == sigIndex;
159  });
160  if (sensList.end() != it &&
161  state->instances[inst].procState->senses[it - sensList.begin()] ==
162  0)
163  continue;
164 
165  // Invalidate scheduled wakeup
166  state->instances[inst].expectedWakeup = Time();
167  }
168  wakeupQueue.push_back(inst);
169  }
170 
171  // Dump the updated signal.
172  if (traceMode != TraceMode::None)
173  trace.addChange(sigIndex);
174  }
175 
176  // Add scheduled process resumes to the wakeup queue.
177  for (auto inst : pop.scheduled) {
178  if (state->time == state->instances[inst].expectedWakeup)
179  wakeupQueue.push_back(inst);
180  }
181 
182  state->queue.pop();
183 
184  std::sort(wakeupQueue.begin(), wakeupQueue.end());
185  wakeupQueue.erase(std::unique(wakeupQueue.begin(), wakeupQueue.end()),
186  wakeupQueue.end());
187 
188  // Run the instances present in the wakeup queue.
189  for (auto i : wakeupQueue) {
190  auto &inst = state->instances[i];
191  auto signalTable = inst.sensitivityList.data();
192 
193  // Gather the instance arguments for unit invocation.
194  SmallVector<void *, 3> args;
195  if (inst.isEntity)
196  args.assign({&state, &inst.entityState, &signalTable});
197  else {
198  args.assign({&state, &inst.procState, &signalTable});
199  }
200  // Run the unit.
201  (*inst.unitFPtr)(args.data());
202  }
203 
204  // Clear wakeup queue.
205  wakeupQueue.clear();
206  ++cycle;
207  }
208 
209  if (traceMode != TraceMode::None) {
210  // Flush any remainign changes
211  trace.flush(/*force=*/true);
212  }
213 
214  llvm::errs() << "Finished at " << state->time.toString() << " (" << cycle
215  << " cycles)\n";
216  return 0;
217 }

219 void Engine::buildLayout(ModuleOp module) {
220  // Start from the root entity.
221  auto rootEntity = module.lookupSymbol<EntityOp>(root);
222  assert(rootEntity && "root entity not found!");
223 
224  // Build root instance, the parent and instance names are the same for the
225  // root.
226  Instance rootInst(state->root);
227  rootInst.unit = root;
228  rootInst.path = root;
229 
230  // Recursively walk the units starting at root.
231  walkEntity(rootEntity, rootInst);
232 
233  // The root is always an instance.
234  rootInst.isEntity = true;
235  // Store the root instance.
236  state->instances.push_back(std::move(rootInst));
237 
238  // Add triggers to signals.
239  for (size_t i = 0, e = state->instances.size(); i < e; ++i) {
240  auto &inst = state->instances[i];
241  for (auto trigger : inst.sensitivityList) {
242  state->signals[trigger.globalIndex].pushInstanceIndex(i);
243  }
244  }
245 }

247 void Engine::walkEntity(EntityOp entity, Instance &child) {
248  entity.walk([&](Operation *op) {
249  assert(op);
250 
251  // Add a signal to the signal table.
252  if (auto sig = dyn_cast<SigOp>(op)) {
253  uint64_t index = state->addSignal(sig.name().str(), child.name);
254  child.sensitivityList.push_back(
255  SignalDetail({nullptr, 0, child.sensitivityList.size(), index}));
256  }
257 
258  // Build (recursive) instance layout.
259  if (auto inst = dyn_cast<InstOp>(op)) {
260  // Skip self-recursion.
261  if (inst.callee() == child.name)
262  return;
263  if (auto e =
264  op->getParentOfType<ModuleOp>().lookupSymbol(inst.callee())) {
265  Instance newChild(child.unit + '.' + inst.name().str());
266  newChild.unit = inst.callee().str();
267  newChild.nArgs = inst.getNumOperands();
268  newChild.path = child.path + "/" + inst.name().str();
269 
270  // Add instance arguments to sensitivity list. The first nArgs signals
271  // in the sensitivity list represent the unit's arguments, while the
272  // following ones represent the unit-defined signals.
273  llvm::SmallVector<Value, 8> args;
274  args.insert(args.end(), inst.inputs().begin(), inst.inputs().end());
275  args.insert(args.end(), inst.outputs().begin(), inst.outputs().end());
276 
277  for (size_t i = 0, e = args.size(); i < e; ++i) {
278  // The signal comes from an instance's argument.
279  if (auto blockArg = args[i].dyn_cast<BlockArgument>()) {
280  auto detail = child.sensitivityList[blockArg.getArgNumber()];
281  detail.instIndex = i;
282  newChild.sensitivityList.push_back(detail);
283  } else if (auto sigOp = dyn_cast<SigOp>(args[i].getDefiningOp())) {
284  // The signal comes from one of the instance's owned signals.
285  auto it = std::find_if(
286  child.sensitivityList.begin(), child.sensitivityList.end(),
287  [&](SignalDetail &detail) {
288  return state->signals[detail.globalIndex].getName() ==
289  sigOp.name() &&
290  state->signals[detail.globalIndex].getOwner() ==
291  child.name;
292  });
293  if (it != child.sensitivityList.end()) {
294  auto detail = *it;
295  detail.instIndex = i;
296  newChild.sensitivityList.push_back(detail);
297  }
298  }
299  }
300 
301  // Recursively walk a new entity, otherwise it is a process and cannot
302  // define new signals or instances.
303  if (auto ent = dyn_cast<EntityOp>(e)) {
304  newChild.isEntity = true;
305  walkEntity(ent, newChild);
306  } else {
307  newChild.isEntity = false;
308  }
309 
310  // Store the created instance.
311  state->instances.push_back(std::move(newChild));
312  }
313  }
314  });
315 }
