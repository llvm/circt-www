33 /// This is a stack of hashtables, if lookup fails in the top-most hashtable,
34 /// it will attempt to lookup in lower hashtables. This class is used instead
35 /// of a ScopedHashTable so we can manually pop off a scope and keep it around.
36 ///
37 /// This only allows inserting into the outermost scope.
38 template <typename KeyT, typename ValueT>
39 struct HashTableStack {
40  using ScopeT = typename llvm::MapVector<KeyT, ValueT>;
41  using StackT = typename llvm::SmallVector<ScopeT, 3>;
42 
43  HashTableStack() {
44  // We require at least one scope.
45  pushScope();
46  }
47 
48  // TODO: This class will need its own iterator eventually.
49  using iterator = typename ScopeT::iterator;
50 
51  iterator end() { return iterator(); }
52 
53  iterator find(const KeyT &key) {
54  // Try to find a hashtable with the missing value.
55  for (auto &map : llvm::reverse(mapStack)) {
56  auto it = map.find(key);
57  if (it != map.end())
58  return it;
59  }
60  return iterator();
61  }
62 
63  ScopeT &getLastScope() { return mapStack.back(); }
64 
65  void pushScope() { mapStack.emplace_back(); }
66 
67  ScopeT popScope() {
68  assert(mapStack.size() > 1 && "Cannot pop the last scope");
69  return mapStack.pop_back_val();
70  }
71 
72  // This class lets you insert into the top scope.
73  ValueT &operator[](const KeyT &key) { return mapStack.back()[key]; }
74 
75 private:
76  StackT mapStack;
77 };

92 class LastConnectResolver : public FIRRTLVisitor<ConcreteT> {

106  /// Records a connection to a destination in the current scope. This will
107  /// delete a previous connection to a destination if there was one. Returns
108  /// true if an old connect was erased.
109  bool setLastConnect(FieldRef dest, Operation *connection) {
110  // Try to insert, if it doesn't insert, replace the previous value.
111  auto itAndInserted = driverMap.getLastScope().insert({dest, connection});
112  if (!std::get<1>(itAndInserted)) {
113  auto iterator = std::get<0>(itAndInserted);
114  auto changed = false;
115  // Delete the old connection if it exists. Null connections are inserted
116  // on declarations.
117  if (auto *oldConnect = iterator->second) {
118  oldConnect->erase();
119  changed = true;
120  }
121  iterator->second = connection;
122  return changed;
123  }
124  return false;
125  }

139  /// For every leaf field in the sink, record that it exists and should be
140  /// initialized.
141  void declareSinks(Value value, Flow flow) {
142  auto type = value.getType();
143  unsigned id = 0;
144 
145  // Recurse through a bundle and declare each leaf sink node.
146  std::function<void(Type, Flow)> declare = [&](Type type, Flow flow) {
147  // If this is a bundle type, recurse to each of the fields.
148  if (auto bundleType = type.dyn_cast<BundleType>()) {
149  for (auto &element : bundleType.getElements()) {
150  id++;
151  if (element.isFlip)
152  declare(element.type, swapFlow(flow));
153  else
154  declare(element.type, flow);
155  }
156  return;
157  }
158 
159  // If this is a vector type, recurse to each of the elements.
160  if (auto vectorType = type.dyn_cast<FVectorType>()) {
161  for (unsigned i = 0; i < vectorType.getNumElements(); ++i) {
162  id++;
163  declare(vectorType.getElementType(), flow);
164  }
165  return;
166  }
167 
168  // If this is an analog type, it does not need to be tracked.
169  if (auto analogType = type.dyn_cast<AnalogType>())
170  return;
171 
172  // If it is a leaf node with Flow::Sink or Flow::Duplex, it must be
173  // initialized.
174  if (flow != Flow::Source)
175  driverMap[{value, id}] = nullptr;
176  };
177  declare(type, flow);
178  }

180  /// Take two connection operations and merge them into a new connect under a
181  /// condition. Destination of both connects should be `dest`.
182  ConnectOp flattenConditionalConnections(OpBuilder &b, Location loc,
183  Value dest, Value cond,
184  Operation *whenTrueConn,
185  Operation *whenFalseConn) {
186  auto fusedLoc =
187  b.getFusedLoc({loc, whenTrueConn->getLoc(), whenFalseConn->getLoc()});
188  auto whenTrue = getConnectedValue(whenTrueConn);
189  auto trueIsInvalid =
190  isa_and_nonnull<InvalidValueOp>(whenTrue.getDefiningOp());
191  auto whenFalse = getConnectedValue(whenFalseConn);
192  auto falseIsInvalid =
193  isa_and_nonnull<InvalidValueOp>(whenFalse.getDefiningOp());
194  // If one of the branches of the mux is an invalid value, we optimize the
195  // mux to be the non-invalid value. This optimization can only be
196  // performed while lowering when-ops into muxes, and would not be legal as
197  // a more general mux folder.
198  // mux(cond, invalid, x) -> x
199  // mux(cond, x, invalid) -> x
200  Value newValue = whenTrue;
201  if (trueIsInvalid == falseIsInvalid)
202  newValue = b.createOrFold<MuxPrimOp>(fusedLoc, cond, whenTrue, whenFalse);
203  else if (trueIsInvalid)
204  newValue = whenFalse;
205  return b.create<ConnectOp>(loc, dest, newValue);
206  }

210  /// Take an aggregate value and construct ground subelements recursively.
211  /// And then apply function `fn`.
212  void foreachSubelement(OpBuilder &builder, Value value,
213  llvm::function_ref<void(Value)> fn) {
214  TypeSwitch<Type>(value.getType())
215  .template Case<BundleType>([&](BundleType bundle) {
216  for (auto i : llvm::seq(0u, (unsigned)bundle.getNumElements())) {
217  auto subfield =
218  builder.create<SubfieldOp>(value.getLoc(), value, i);
219  foreachSubelement(builder, subfield, fn);
220  }
221  })
222  .template Case<FVectorType>([&](FVectorType vector) {
223  for (auto i : llvm::seq((size_t)0, vector.getNumElements())) {
224  auto subindex =
225  builder.create<SubindexOp>(value.getLoc(), value, i);
226  foreachSubelement(builder, subindex, fn);
227  }
228  })
229  .Default([&](auto) { fn(value); });
230  }

278  void processWhenOp(WhenOp whenOp, Value outerCondition);

280  /// Combine the connect statements from each side of the block. There are 5
281  /// cases to consider. If all are set, last connect semantics dictate that it
282  /// is actually the third case.
283  ///
284  /// Prev | Then | Else | Outcome
285  /// -----|------|------|-------
286  /// | set | | then
287  /// | | set | else
288  /// set | set | set | mux(p, then, else)
289  /// | set | set | impossible
290  /// set | set | | mux(p, then, prev)
291  /// set | | set | mux(p, prev, else)
292  ///
293  /// If the value was declared in the block, then it does not need to have been
294  /// assigned a previous value. If the value was declared before the block,
295  /// then there is an incomplete initialization error.
296  void mergeScopes(Location loc, DriverMap &thenScope, DriverMap &elseScope,
297  Value thenCondition) {
298 
299  // Process all connects in the `then` block.
300  for (auto &destAndConnect : thenScope) {
301  auto dest = std::get<0>(destAndConnect);
302  auto thenConnect = std::get<1>(destAndConnect);
303 
304  auto outerIt = driverMap.find(dest);
305  if (outerIt == driverMap.end()) {
306  // `dest` is set in `then` only. This indicates it was created in the
307  // `then` block, so just copy it into the outer scope.
308  driverMap[dest] = thenConnect;
309  continue;
310  }
311 
312  auto elseIt = elseScope.find(dest);
313  if (elseIt != elseScope.end()) {
314  // `dest` is set in `then` and `else`. We need to combine them into and
315  // delete any previous connect.
316 
317  // Create a new connect with `mux(p, then, else)`.
318  auto &elseConnect = std::get<1>(*elseIt);
319  OpBuilder connectBuilder(elseConnect);
320  auto newConnect = flattenConditionalConnections(
321  connectBuilder, loc, getDestinationValue(thenConnect),
322  thenCondition, thenConnect, elseConnect);
323 
324  // Delete all old connections.
325  thenConnect->erase();
326  elseConnect->erase();
327  setLastConnect(dest, newConnect);
328 
329  // Do not process connect in the else scope.
330  elseScope.erase(dest);
331  continue;
332  }
333 
334  auto &outerConnect = std::get<1>(*outerIt);
335  if (!outerConnect) {
336  // `dest` is null in the outer scope. This indicate an initialization
337  // problem: `mux(p, then, nullptr)`. Just delete the broken connect.
338  thenConnect->erase();
339  continue;
340  }
341 
342  // `dest` is set in `then` and the outer scope. Create a new connect with
343  // `mux(p, then, outer)`.
344  OpBuilder connectBuilder(thenConnect);
345  auto newConnect = flattenConditionalConnections(
346  connectBuilder, loc, getDestinationValue(thenConnect), thenCondition,
347  thenConnect, outerConnect);
348 
349  // Delete all old connections.
350  thenConnect->erase();
351  setLastConnect(dest, newConnect);
352  }
353 
354  // Process all connects in the `else` block.
355  for (auto &destAndConnect : elseScope) {
356  auto dest = std::get<0>(destAndConnect);
357  auto elseConnect = std::get<1>(destAndConnect);
358 
359  auto outerIt = driverMap.find(dest);
360  if (outerIt == driverMap.end()) {
361  // `dest` is set in `else` only. This indicates it was created in the
362  // `else` block, so just copy it into the outer scope.
363  driverMap[dest] = elseConnect;
364  continue;
365  }
366 
367  auto &outerConnect = std::get<1>(*outerIt);
368  if (!outerConnect) {
369  // `dest` is null in the outer scope. This indicate an initialization
370  // problem: `mux(p, null, else)`. Just delete the broken connect.
371  elseConnect->erase();
372  continue;
373  }
374 
375  // `dest` is set in the `else` and outer scope. Create a new connect with
376  // `mux(p, outer, else)`.
377  OpBuilder connectBuilder(elseConnect);
378  auto newConnect = flattenConditionalConnections(
379  connectBuilder, loc, getDestinationValue(outerConnect), thenCondition,
380  outerConnect, elseConnect);
381 
382  // Delete all old connections.
383  elseConnect->erase();
384  setLastConnect(dest, newConnect);
385  }
386  }

465 /// This is a common helper that is dispatched to by the concrete visitors.
466 /// This condition should be the conjunction of all surrounding WhenOp
467 /// condititions.
468 ///
469 /// This requires WhenOpVisitor to be fully defined.
470 template <typename ConcreteT>
471 void LastConnectResolver<ConcreteT>::processWhenOp(WhenOp whenOp,
472  Value outerCondition) {
473  OpBuilder b(whenOp);
474  auto loc = whenOp.getLoc();
475  Block *parentBlock = whenOp->getBlock();
476  auto condition = whenOp.condition();
477  auto ui1Type = condition.getType();
478 
479  // Process both sides of the the WhenOp, fixing up all simulation
480  // contructs, and resolving last connect semantics in each block. This
481  // process returns the set of connects in each side of the when op.
482 
483  // Process the `then` block. If we are already in a whenblock, the we need to
484  // conjoin ('and') the outer conditions.
485  auto thenCondition = whenOp.condition();
486  if (outerCondition)
487  thenCondition =
488  b.createOrFold<AndPrimOp>(loc, ui1Type, outerCondition, thenCondition);
489 
490  auto &thenBlock = whenOp.getThenBlock();
491  driverMap.pushScope();
492  WhenOpVisitor(driverMap, thenCondition).process(thenBlock);
493  mergeBlock(*parentBlock, Block::iterator(whenOp), thenBlock);
494  auto thenScope = driverMap.popScope();
495 
496  // Process the `else` block.
497  DriverMap elseScope;
498  if (whenOp.hasElseRegion()) {
499  // Else condition is the compliment of the then condition.
500  auto elseCondition =
501  b.createOrFold<NotPrimOp>(loc, condition.getType(), condition);
502  // Conjoin the when condition with the outer condition.
503  if (outerCondition)
504  elseCondition = b.createOrFold<AndPrimOp>(loc, ui1Type, outerCondition,
505  elseCondition);
506  auto &elseBlock = whenOp.getElseBlock();
507  driverMap.pushScope();
508  WhenOpVisitor(driverMap, elseCondition).process(elseBlock);
509  mergeBlock(*parentBlock, Block::iterator(whenOp), elseBlock);
510  elseScope = driverMap.popScope();
511  }
512 
513  mergeScopes(loc, thenScope, elseScope, condition);
514 
515  // Delete the now empty WhenOp.
516  whenOp.erase();
517 }

548 /// Run expand whens on the Module. This will emit an error for each
549 /// incomplete initialization found. If an initialiazation error was detected,
550 /// this will return failure and leave the IR in an inconsistent state.
551 bool ModuleVisitor::run(FModuleOp module) {
552  // Track any results (flipped arguments) of the module for init coverage.
553  for (const auto &it : llvm::enumerate(module.getArguments())) {
554  auto flow = module.getPortDirection(it.index()) == Direction::In
555  ? Flow::Source
556  : Flow::Sink;
557  declareSinks(it.value(), flow);
558  }
559 
560  // Process the body of the module.
561  for (auto &op : llvm::make_early_inc_range(*module.getBody())) {
562  dispatchVisitor(&op);
563  }
564  return anythingChanged;
565 }

581 /// Perform initialization checking. This uses the built up state from
582 /// running on a module. Returns failure in the event of bad initialization.
583 LogicalResult ModuleVisitor::checkInitialization() {
584  for (auto destAndConnect : driverMap.getLastScope()) {
585  // If there is valid connection to this destination, everything is good.
586  auto *connect = std::get<1>(destAndConnect);
587  if (connect)
588  continue;
589 
590  // Get the op which defines the sink, and emit an error.
591  FieldRef dest = std::get<0>(destAndConnect);
592  auto *definingOp = dest.getDefiningOp();
593  if (auto mod = dyn_cast<FModuleLike>(definingOp))
594  mlir::emitError(definingOp->getLoc())
595  << "port \"" + getFieldName(dest) +
596  "\" not fully initialized in module \""
597  << mod.moduleName() << "\"";
598  else
599  definingOp->emitError("sink \"" + getFieldName(dest) +
600  "\" not fully initialized");
601  return failure();
602  }
603  return success();
604 }
