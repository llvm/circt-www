29 /// Move all operations from a source block in to a destination block. Leaves
30 /// the source block empty.
31 static void mergeBlock(Block &destination, Block::iterator insertPoint,
32  Block &source) {
33  destination.getOperations().splice(insertPoint, source.getOperations());
34 }

42 /// This visitor visits process a block resolving last connect semantics
43 /// and expanding WhenOps.
44 template <typename ConcreteT>
45 class LastConnectResolver : public FIRRTLVisitor<ConcreteT> {
46 protected:
47  bool changed = false;
48 
49  /// Map of destinations and the operation which is driving a value to it in
50  /// the current scope. This is used for resolving last connect semantics, and
51  /// for retrieving the responsible connect operation.
52  ScopeMap &scope;
53 
54 public:
55  LastConnectResolver(ScopeMap &scope) : scope(scope) {}
56 
57  using FIRRTLVisitor<ConcreteT>::visitExpr;
58  using FIRRTLVisitor<ConcreteT>::visitDecl;
59  using FIRRTLVisitor<ConcreteT>::visitStmt;
60 
61  /// Records a connection to a destination. This will delete a previous
62  /// connection to a destination if there was one. Returns true if an old
63  /// connect was erased.
64  bool setLastConnect(FieldRef dest, Operation *connection) {
65  // Try to insert, if it doesn't insert, replace the previous value.
66  auto itAndInserted = scope.insert({dest, connection});
67  if (!std::get<1>(itAndInserted)) {
68  auto iterator = std::get<0>(itAndInserted);
69  auto changed = false;
70  // Delete the old connection if it exists. Null connections are inserted
71  // on declarations.
72  if (auto *oldConnect = iterator->second) {
73  oldConnect->erase();
74  changed = true;
75  }
76  iterator->second = connection;
77  return changed;
78  }
79  return false;
80  }
81 
82  /// Get the destination value from a connection. This supports any operation
83  /// which is capable of driving a value.
84  static Value getDestinationValue(Operation *op) {
85  return cast<ConnectOp>(op).dest();
86  }
87 
88  /// Get the source value from a connection. This supports any operation which
89  /// is capable of driving a value.
90  static Value getConnectedValue(Operation *op) {
91  return cast<ConnectOp>(op).src();
92  }
93 
94  /// For every leaf field in the sink, record that it exists and should be
95  /// initialized.
96  void declareSinks(Value value, Flow flow) {
97  auto type = value.getType();
98  unsigned id = 0;
99 
100  // Recurse through a bundle and declare each leaf sink node.
101  std::function<void(Type, Flow)> declare = [&](Type type, Flow flow) {
102  // If this is a bundle type, recurse to each of the fields.
103  if (auto bundleType = type.dyn_cast<BundleType>()) {
104  for (auto &element : bundleType.getElements()) {
105  id++;
106  if (element.isFlip)
107  declare(element.type, swapFlow(flow));
108  else
109  declare(element.type, flow);
110  }
111  return;
112  }
113 
114  // If this is an analog type, it does not need to be tracked.
115  if (auto analogType = type.dyn_cast<AnalogType>())
116  return;
117 
118  // If it is a leaf node with Flow::Sink or Flow::Duplex, it must be
119  // initialized.
120  if (flow != Flow::Source)
121  scope[{value, id}] = nullptr;
122  };
123  declare(type, flow);
124  }
125 
126  /// If a value has an outer flip, convert the value to passive.
127  Value convertToPassive(OpBuilder &builder, Location loc, Value input) {
128  auto inType = input.getType().cast<FIRRTLType>();
129  return builder.createOrFold<AsPassivePrimOp>(loc, inType.getPassiveType(),
130  input);
131  }
132 
133  /// Take two connection operations and merge them in to a new connect under a
134  /// condition. Destination of both connects should be `dest`.
135  ConnectOp flattenConditionalConnections(OpBuilder &b, Location loc,
136  Value dest, Value cond,
137  Operation *whenTrueConn,
138  Operation *whenFalseConn) {
139  auto whenTrue = convertToPassive(b, loc, getConnectedValue(whenTrueConn));
140  auto whenFalse = convertToPassive(b, loc, getConnectedValue(whenFalseConn));
141  auto newValue = b.createOrFold<MuxPrimOp>(loc, cond, whenTrue, whenFalse);
142  auto newConnect = b.create<ConnectOp>(loc, dest, newValue);
143  whenTrueConn->erase();
144  whenFalseConn->erase();
145  return newConnect;
146  }

148  void visitDecl(WireOp op) { declareSinks(op.result(), Flow::Duplex); }
149 
150  void visitDecl(RegOp op) {
151  // Registers are initialized to themselves.
152  // TODO: register of bundle type are not supported.
153  auto connect = OpBuilder(op->getBlock(), ++Block::iterator(op))
154  .create<ConnectOp>(op.getLoc(), op, op);
155  scope[getFieldRefFromValue(op.result())] = connect;
156  }
157 
158  void visitDecl(RegResetOp op) {
159  // Registers are initialized to themselves.
160  // TODO: register of bundle type are not supported.
161  assert(!op.result().getType().isa<BundleType>() &&
162  "registers can't be bundle type");
163  auto connect = OpBuilder(op->getBlock(), ++Block::iterator(op))
164  .create<ConnectOp>(op.getLoc(), op, op);
165  scope[getFieldRefFromValue(op.result())] = connect;
166  }
167 
168  void visitDecl(InstanceOp op) {
169  // Track any instance inputs which need to be connected to for init
170  // coverage.
171  for (auto result : llvm::enumerate(op.results()))
172  if (getModulePortDirection(op.getReferencedModule(), result.index()) ==
173  Direction::Output)
174  declareSinks(result.value(), Flow::Source);
175  else
176  declareSinks(result.value(), Flow::Sink);
177  }
178 
179  void visitDecl(MemOp op) {
180  // Track any memory inputs which require connections.
181  for (auto result : op.results())
182  declareSinks(result, Flow::Sink);
183  }
184 
185  void visitStmt(PartialConnectOp op) {
186  llvm_unreachable("PartialConnectOps should have been removed.");
187  }
188 
189  void visitStmt(ConnectOp op) {
190  setLastConnect(getFieldRefFromValue(op.dest()), op);
191  }
192 
193  /// Combine the connect statements from each side of the block. There are 5
194  /// cases to consider. If all are set, last connect semantics dictate that it
195  /// is actually the third case.
196  ///
197  /// Prev | Then | Else | Outcome
198  /// -----|------|------|-------
199  /// | set | | then
200  /// | | set | else
201  /// | set | set | mux(p, then, else)
202  /// set | set | | mux(p, then, prev)
203  /// set | | set | mux(p, prev, else)
204  ///
205  /// If the value was declared in the block, then it does not need to have been
206  /// assigned a previous value. If the value was declared before the block,
207  /// then there is an incomplete initialization error.
208  void mergeScopes(ScopeMap &thenScope, ScopeMap &elseScope,
209  Value thenCondition) {
210 
211  // Process all connects in the `then` block.
212  for (auto &destAndConnect : thenScope) {
213  auto dest = std::get<0>(destAndConnect);
214  auto thenConnect = std::get<1>(destAndConnect);
215 
216  // `dest` is set in `then` only.
217  auto itAndInserted = scope.insert({dest, thenConnect});
218  if (std::get<1>(itAndInserted))
219  continue;
220  auto outerIt = std::get<0>(itAndInserted);
221 
222  // `dest` is set in `then` and `else`.
223  auto elseIt = elseScope.find(dest);
224  if (elseIt != elseScope.end()) {
225  auto &elseConnect = std::get<1>(*elseIt);
226  // Create a new connect with `mux(p, then, else)`.
227  OpBuilder connectBuilder(elseConnect);
228  auto newConnect = flattenConditionalConnections(
229  connectBuilder, elseConnect->getLoc(),
230  getDestinationValue(thenConnect), thenCondition, thenConnect,
231  elseConnect);
232  setLastConnect(dest, newConnect);
233  // Do not process connect in the else scope.
234  elseScope.erase(dest);
235  continue;
236  }
237 
238  // `dest` is null in the outer scope.
239  auto &outerConnect = std::get<1>(*outerIt);
240  if (!outerConnect) {
241  thenConnect->erase();
242  continue;
243  }
244 
245  // `dest` is set in the outer scope.
246  // Create a new connect with mux(p, then, outer)
247  OpBuilder connectBuilder(thenConnect);
248  auto newConnect = flattenConditionalConnections(
249  connectBuilder, thenConnect->getLoc(),
250  getDestinationValue(thenConnect), thenCondition, thenConnect,
251  outerConnect);
252  outerIt->second = newConnect;
253  }
254 
255  // Process all connects in the `else` block.
256  for (auto &destAndConnect : elseScope) {
257  auto dest = std::get<0>(destAndConnect);
258  auto elseConnect = std::get<1>(destAndConnect);
259 
260  // `dest` is set in `then` only.
261  auto itAndInserted = scope.insert({dest, elseConnect});
262  if (std::get<1>(itAndInserted))
263  continue;
264 
265  // `dest` is null in the outer scope.
266  auto outerIt = std::get<0>(itAndInserted);
267  auto &outerConnect = std::get<1>(*outerIt);
268  if (!outerConnect) {
269  elseConnect->erase();
270  continue;
271  }
272 
273  // `dest` is set in the outer scope.
274  // Create a new connect with mux(p, outer, else).
275  OpBuilder connectBuilder(elseConnect);
276  auto newConnect = flattenConditionalConnections(
277  connectBuilder, elseConnect->getLoc(),
278  getDestinationValue(outerConnect), thenCondition, outerConnect,
279  elseConnect);
280  outerIt->second = newConnect;
281  }
282  }
283 };

290 /// This extends the ModuleVisitor with additional funcationality that is only
291 /// required in side a WhenOp. This visitor handles all Simulation constructs.
292 namespace {
293 class WhenOpVisitor : public LastConnectResolver<WhenOpVisitor> {
294 
295 public:
296  WhenOpVisitor(ScopeMap &scope, Value condition)
297  : LastConnectResolver<WhenOpVisitor>(scope), condition(condition) {}
298 
299  using LastConnectResolver<WhenOpVisitor>::visitExpr;
300  using LastConnectResolver<WhenOpVisitor>::visitDecl;
301  using LastConnectResolver<WhenOpVisitor>::visitStmt;
302 
303  /// Process a block, recording each declaration, and expanding all whens.
304  void process(Block &block);
305 
306  /// Simulation Constructs.
307  void visitStmt(AssertOp op);
308  void visitStmt(AssumeOp op);
309  void visitStmt(CoverOp op);
310  void visitStmt(ModuleOp op);
311  void visitStmt(PrintFOp op);
312  void visitStmt(StopOp op);
313  void visitStmt(WhenOp op);
314 
315 private:
316  /// And a 1-bit value with the current condition. If we are in the outer
317  /// scope, i.e. not in a WhenOp region, then there is no condition.
318  Value andWithCondition(Operation *op, Value value) {
319  // 'and' the value with the current condition.
320  return OpBuilder(op).createOrFold<AndPrimOp>(
321  condition.getLoc(), condition.getType(), condition, value);
322  }
323 
324 private:
325  /// The current wrapping condition. If null, we are in the outer scope.
326  Value condition;
327 };

357 void WhenOpVisitor::visitStmt(WhenOp whenOp) {
358  OpBuilder b(whenOp);
359  Block *parentBlock = whenOp->getBlock();
360  auto condition = whenOp.condition();
361 
362  // Process both sides of the the WhenOp, fixing up all simulation
363  // contructs, and resolving last connect semantics in each block. This process
364  // returns the set of connects in each side of the when op.
365 
366  // Process the `then` block.
367  ScopeMap thenScope;
368  auto thenCondition = andWithCondition(whenOp, condition);
369  auto &thenBlock = whenOp.getThenBlock();
370  WhenOpVisitor(thenScope, thenCondition).process(thenBlock);
371  mergeBlock(*parentBlock, Block::iterator(whenOp), thenBlock);
372 
373  // Process the `else` block.
374  ScopeMap elseScope;
375  if (whenOp.hasElseRegion()) {
376  auto notOp = b.createOrFold<NotPrimOp>(whenOp.getLoc(), condition.getType(),
377  condition);
378  Value elseCondition = andWithCondition(whenOp, notOp);
379  auto &elseBlock = whenOp.getElseBlock();
380  WhenOpVisitor(elseScope, elseCondition).process(elseBlock);
381  mergeBlock(*parentBlock, Block::iterator(whenOp), elseBlock);
382  }
383 
384  mergeScopes(thenScope, elseScope, thenCondition);
385 
386  // Delete the now empty WhenOp.
387  whenOp.erase();
388 }

422 mlir::FailureOr<bool> ModuleVisitor::run(FModuleOp module) {
423  // Track any results (flipped arguments) of the module for init coverage.
424  for (auto it : llvm::enumerate(module.getArguments())) {
425  auto flow = getModulePortDirection(module, it.index()) == Direction::Input
426  ? Flow::Source
427  : Flow::Sink;
428  declareSinks(it.value(), flow);
429  }
430 
431  // Process the body of the module.
432  for (auto &op : llvm::make_early_inc_range(*module.getBodyBlock())) {
433  dispatchVisitor(&op);
434  }
435 
436  // Check for any incomplete initialization.
437  for (auto destAndConnect : outerScope) {
438  // If there is valid connection to this destination, everything is good.
439  auto *connect = std::get<1>(destAndConnect);
440  if (connect)
441  continue;
442  // Get the op which defines the sink, and emit an error.
443  auto dest = std::get<0>(destAndConnect);
444  dest.getDefiningOp()->emitError("sink \"" + getFieldName(dest) +
445  "\" not fully initialized");
446  return failure();
447  }
448  return mlir::FailureOr<bool>(anythingChanged);
449 }

455 void ModuleVisitor::visitStmt(WhenOp whenOp) {
456  Block *parentBlock = whenOp->getBlock();
457  auto condition = whenOp.condition();
458 
459  // Process both sides of the the WhenOp, fixing up all simulation contructs,
460  // and resolving last connect semantics in each block. This process returns
461  // the set of connects in each side of the when op.
462 
463  // Process the `then` block.
464  ScopeMap thenScope;
465  auto &thenBlock = whenOp.getThenBlock();
466  WhenOpVisitor(thenScope, condition).process(thenBlock);
467  mergeBlock(*parentBlock, Block::iterator(whenOp), thenBlock);
468 
469  // Process the `else` block.
470  ScopeMap elseScope;
471  if (whenOp.hasElseRegion()) {
472  OpBuilder b(whenOp);
473  auto notCondition = b.createOrFold<NotPrimOp>(
474  whenOp.getLoc(), condition.getType(), condition);
475  auto &elseBlock = whenOp.getElseBlock();
476  WhenOpVisitor(elseScope, notCondition).process(elseBlock);
477  mergeBlock(*parentBlock, Block::iterator(whenOp), elseBlock);
478  }
479 
480  mergeScopes(thenScope, elseScope, condition);
481 
482  // If we are deleting a WhenOp something definitely changed.
483  anythingChanged = true;
484 
485  // Delete the now empty WhenOp.
486  whenOp.erase();
487 }
