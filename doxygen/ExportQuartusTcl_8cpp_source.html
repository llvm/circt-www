37 LogicalResult TclEmitter::populate() {
38  if (populated)
39  return success();
40 
41  // Populated the symbol cache.
42  for (auto symOp : topLevel.getOps<mlir::SymbolOpInterface>())
43  if (auto name = symOp.getNameAttr())
44  topLevelSymbols.addDefinition(name, symOp);
45  topLevelSymbols.freeze();
46  populated = true;
47 
48  // Bin any operations we may need to emit based on the root module in the
49  // instance hierarchy path and the potential instance name.
50 
51  // Look in InstanceHierarchyOps to get the instance named ones.
52  for (auto hier : topLevel.getOps<InstanceHierarchyOp>()) {
53  Operation *mod = topLevelSymbols.getDefinition(hier.topModuleRefAttr());
54  auto &tclOps = tclOpsForModInstance[mod][hier.instNameAttr()];
55  for (auto tclOp : hier.getOps<DynInstDataOpInterface>()) {
56  assert(tclOp.getTopModule(topLevelSymbols) == mod &&
57  "Referenced mod does does not match");
58  tclOps.push_back(tclOp);
59  }
60  }
61 
62  // Locations at the global scope are assumed to refer to the module without an
63  // instance.
64  for (auto tclOp : topLevel.getOps<DynInstDataOpInterface>()) {
65  Operation *mod = tclOp.getTopModule(topLevelSymbols);
66  assert(mod && "Must be able to resolve top module");
67  tclOpsForModInstance[mod][{}].push_back(tclOp);
68  }
69 
70  return success();
71 }

76  return topLevelSymbols.getDefinition(sym);

133 void TclOutputState::emitPath(hw::GlobalRefOp ref,
134  Optional<StringRef> subpath) {
135  // Traverse each part of the path.
136  auto parts = ref.namepathAttr().getAsRange<hw::InnerRefAttr>();
137  auto lastPart = std::prev(parts.end());
138  for (auto part : parts) {
139  emitInnerRefPart(part);
140  if (part != *lastPart)
141  os << '|';
142  }
143 
144  // Some placements don't require subpaths.
145  if (subpath)
146  os << subpath;
147 }

149 void TclOutputState::emit(PhysLocationAttr pla) {
150  // Different devices have different 'number' letters (the 'N' in 'N0'). M20Ks
151  // and DSPs happen to have the same one, probably because they never co-exist
152  // at the same location.
153  char numCharacter;
154  switch (pla.getPrimitiveType().getValue()) {
155  case PrimitiveType::M20K:
156  os << "M20K";
157  numCharacter = 'N';
158  break;
159  case PrimitiveType::DSP:
160  os << "MPDSP";
161  numCharacter = 'N';
162  break;
163  case PrimitiveType::FF:
164  os << "FF";
165  numCharacter = 'N';
166  break;
167  }
168 
169  // Write out the rest of the location info.
170  os << "_X" << pla.getX() << "_Y" << pla.getY() << "_" << numCharacter
171  << pla.getNum();
172 }

225 /// Emit tcl in the form of:
226 /// set_instance_assignment -name PLACE_REGION "X1 Y1 X20 Y20" -to $parent|a|b|c
227 /// set_instance_assignment -name RESERVE_PLACE_REGION OFF -to $parent|a|b|c
228 /// set_instance_assignment -name CORE_ONLY_PLACE_REGION ON -to $parent|a|b|c
229 /// set_instance_assignment -name REGION_NAME test_region -to $parent|a|b|c
230 LogicalResult TclOutputState::emit(PDPhysRegionOp region) {
231  GlobalRefOp ref = getRefOp(region);
232 
233  auto physicalRegion = dyn_cast_or_null<DeclPhysicalRegionOp>(
234  emitter.getDefinition(region.physRegionRefAttr()));
235  if (!physicalRegion)
236  return region.emitOpError(
237  "could not find physical region declaration named ")
238  << region.physRegionRefAttr();
239 
240  // PLACE_REGION directive.
241  indent() << "set_instance_assignment -name PLACE_REGION \"";
242  auto physicalBounds =
243  physicalRegion.bounds().getAsRange<PhysicalBoundsAttr>();
244  llvm::interleave(
245  physicalBounds, os,
246  [&](PhysicalBoundsAttr bounds) {
247  os << 'X' << bounds.getXMin() << ' ';
248  os << 'Y' << bounds.getYMin() << ' ';
249  os << 'X' << bounds.getXMax() << ' ';
250  os << 'Y' << bounds.getYMax();
251  },
252  ";");
253  os << '"';
254 
255  os << " -to $parent|";
256  emitPath(ref, region.subPath());
257  os << '\n';
258 
259  // RESERVE_PLACE_REGION directive.
260  indent() << "set_instance_assignment -name RESERVE_PLACE_REGION OFF";
261  os << " -to $parent|";
262  emitPath(ref, region.subPath());
263  os << '\n';
264 
265  // CORE_ONLY_PLACE_REGION directive.
266  indent() << "set_instance_assignment -name CORE_ONLY_PLACE_REGION ON";
267  os << " -to $parent|";
268  emitPath(ref, region.subPath());
269  os << '\n';
270 
271  // REGION_NAME directive.
272  indent() << "set_instance_assignment -name REGION_NAME ";
273  os << physicalRegion.getName();
274  os << " -to $parent|";
275  emitPath(ref, region.subPath());
276  os << '\n';
277  return success();
278 }

280 /// Write out all the relevant tcl commands. Create one 'proc' per module which
281 /// takes the parent entity name since we don't assume that the created module
282 /// is the top level for the entire design.
283 LogicalResult TclEmitter::emit(Operation *hwMod, StringRef outputFile) {
284  if (failed(populate()))
285  return failure();
286 
287  // Build up the output Tcl, tracking symbol references in state.
288  std::string s;
289  llvm::raw_string_ostream os(s);
290  TclOutputState state(*this, os);
291 
292  // Iterate through all the "instances" for 'hwMod' and produce a tcl proc for
293  // each one.
294  for (auto tclOpsForInstancesKV : tclOpsForModInstance[hwMod]) {
295  StringAttr instName = tclOpsForInstancesKV.first;
296  os << "proc {{" << state.symbolRefs.size() << "}}";
297  if (instName)
298  os << '_' << instName.getValue();
299  os << "_config { parent } {\n";
300  state.symbolRefs.push_back(SymbolRefAttr::get(hwMod));
301 
302  // Loop through the ops relevant to the specified root module "instance".
303  LogicalResult ret = success();
304  auto &tclOpsForMod = tclOpsForInstancesKV.second;
305  for (Operation *tclOp : tclOpsForMod) {
306  LogicalResult rc =
307  TypeSwitch<Operation *, LogicalResult>(tclOp)
308  .Case([&](PDPhysLocationOp op) { return state.emit(op); })
309  .Case([&](PDRegPhysLocationOp op) { return state.emit(op); })
310  .Case([&](PDPhysRegionOp op) { return state.emit(op); })
311  .Case([&](DynamicInstanceVerbatimAttrOp op) {
312  return state.emit(op);
313  })
314  .Default([](Operation *op) {
315  return op->emitOpError("could not determine how to output tcl");
316  });
317  if (failed(rc))
318  ret = failure();
319  }
320  os << "}\n\n";
321  }
322 
323  // Create a verbatim op containing the Tcl and symbol references.
324  OpBuilder builder = OpBuilder::atBlockEnd(hwMod->getBlock());
325  auto verbatim = builder.create<sv::VerbatimOp>(
326  builder.getUnknownLoc(), os.str(), ValueRange{},
327  builder.getArrayAttr(state.symbolRefs));
328 
329  // When requested, give the verbatim op an output file.
330  if (!outputFile.empty()) {
331  auto outputFileAttr =
332  OutputFileAttr::getFromFilename(builder.getContext(), outputFile);
333  verbatim->setAttr("output_file", outputFileAttr);
334  }
335 
336  return success();
337 }
