73 Optional<Entity> Entity::enter(InstanceOp inst) {
74  auto mod = dyn_cast_or_null<hw::HWModuleOp>(inst.getReferencedModule());
75  if (!mod) // Could be an extern module, which we should ignore.
76  return {};
77  bool modEmitted = insideEmittedModule ||
78  s.modulesEmitted.find(mod) != s.modulesEmitted.end();
79  s.modulesEmitted.insert(mod);
80 
81  return Entity(this, inst.instanceName(), modEmitted);
82 }

84 /// Emit tcl in the form of:
85 /// "set_location_assignment MPDSP_X34_Y285_N0 -to $parent|fooInst|entityName"
86 LogicalResult Entity::emit(Operation *op, StringRef attrKey, StringRef instName,
87  PhysLocationAttr pla) {
88  if (insideEmittedModule)
89  op->emitWarning(
90  "The placement information for this module has already been emitted. "
91  "Modules are required to only be instantiated once.");
92 
93  if (!attrKey.startswith_insensitive("loc:"))
94  return op->emitError("Error in '")
95  << attrKey << "' PhysLocation attribute. Expected loc:<entityName>.";
96 
97  StringRef childEntity = attrKey.substr(4);
98  if (childEntity.empty())
99  return op->emitError("Entity name cannot be empty in 'loc:<entityName>'");
100 
101  s.indent() << "set_location_assignment ";
102 
103  // Different devices have different 'number' letters (the 'N' in 'N0'). M20Ks
104  // and DSPs happen to have the same one, probably because they never co-exist
105  // at the same location.
106  char numCharacter;
107  switch (pla.getDevType().getValue()) {
108  case DeviceType::M20K:
109  s.os << "M20K";
110  numCharacter = 'N';
111  break;
112  case DeviceType::DSP:
113  s.os << "MPDSP";
114  numCharacter = 'N';
115  break;
116  }
117 
118  // Write out the rest of the location info.
119  s.os << "_X" << pla.getX() << "_Y" << pla.getY() << "_" << numCharacter
120  << pla.getNum();
121 
122  // To which entity does this apply?
123  s.os << " -to ";
124  emitPath();
125  // If instance name is specified, add it in between the parent entity path and
126  // the child entity patch.
127  if (!instName.empty())
128  s.os << instName << '|';
129  else if (auto name = op->getAttrOfType<StringAttr>("name"))
130  s.os << name.getValue() << '|';
131  s.os << childEntity << '\n';
132 
133  emittedAttrKeys.insert(attrKey);
134  return success();
135 }

144 /// Export the TCL for a particular entity, corresponding to op. Do this
145 /// recusively, assume that all descendants are in the same entity. When this is
146 /// no longer a sound assuption, we'll have to refactor this code. For now, only
147 /// HWModule instances create a new entity.
148 static LogicalResult exportTcl(Entity &entity, Operation *op) {
149  // Instances require a new child entity and trigger a descent of the
150  // instantiated module in the new entity.
151  StringRef instName;
152  if (auto inst = dyn_cast<hw::InstanceOp>(op)) {
153  instName = inst.instanceName();
154  Optional<Entity> inModule = entity.enter(inst);
155  if (inModule && failed(exportTcl(*inModule, inst.getReferencedModule())))
156  return failure();
157  }
158 
159  // Iterate through 'op's attributes, looking for attributes which we
160  // recognize.
161  for (NamedAttribute attr : op->getAttrs()) {
162  if (entity.emittedAttrKeys.find(attr.first) != entity.emittedAttrKeys.end())
163  op->emitWarning("Attribute has already been emitted: '")
164  << attr.first << "'";
165 
166  if (auto loc = attr.second.dyn_cast<PhysLocationAttr>())
167  if (failed(entity.emit(op, attr.first, instName, loc)))
168  return failure();
169  }
170 
171  auto result = op->walk([&](Operation *innerOp) {
172  if (innerOp != op && failed(exportTcl(entity, innerOp)))
173  return mlir::WalkResult::interrupt();
174  return mlir::WalkResult::advance();
175  });
176  return failure(result.wasInterrupted());
177 }

179 /// Write out all the relevant tcl commands. Create one 'proc' per module (since
180 /// we don't know which one will be the 'top' module). Said procedure takes the
181 /// parent entity name since we don't assume that the created module is the top
182 /// level for the entire design.
183 LogicalResult circt::msft::exportQuartusTcl(ModuleOp module,
184  llvm::raw_ostream &os) {
185  TclOutputState state(os);
186 
187  for (auto &op : module.getBody()->getOperations()) {
188  auto hwMod = dyn_cast<HWModuleOp>(op);
189  if (!hwMod)
190  continue;
191  os << "proc " << hwMod.getName() << "_config { parent } {\n";
192  Entity entity(state);
193  if (failed(exportTcl(entity, hwMod)))
194  return failure();
195  os << "}\n\n";
196  }
197  return success();
198 }
