33 /// This class keeps track of global names at the module/interface level.
34 /// It is built in a global pass over the entire design and then frozen to allow
35 /// concurrent accesses.
36 struct GlobalNameTable {
37  GlobalNameTable(GlobalNameTable &&) = default;
38 
39  /// Return the string to use for the specified parameter name in the specified
40  /// module. Parameters may be renamed for a variety of reasons (e.g.
41  /// conflicting with ports or Verilog keywords), and this returns the
42  /// legalized name to use.
43  StringRef getParameterVerilogName(Operation *module,
44  StringAttr paramName) const {
45  auto it = renamedParams.find(std::make_pair(module, paramName));
46  return (it != renamedParams.end() ? it->second : paramName).getValue();
47  }
48 
49 private:
50  friend class GlobalNameResolver;
51  GlobalNameTable() {}
52  GlobalNameTable(const GlobalNameTable &) = delete;
53  void operator=(const GlobalNameTable &) = delete;
54 
55  void addRenamedParam(Operation *module, StringAttr oldName,
56  StringRef newName) {
57  renamedParams[{module, oldName}] =
58  StringAttr::get(oldName.getContext(), newName);
59  }
60 
61  /// This contains entries for any parameters that got renamed. The key is a
62  /// moduleop/paramName tuple, the value is the name to use.
63  DenseMap<std::pair<Operation *, Attribute>, StringAttr> renamedParams;
64 };

153 /// This class wraps an operation or a fixed string that should be emitted.
154 class StringOrOpToEmit {
155 public:
156  explicit StringOrOpToEmit(Operation *op) : pointerData(op), length(~0ULL) {}
157 
158  explicit StringOrOpToEmit(StringRef string) {
159  pointerData = (Operation *)nullptr;
160  setString(string);
161  }
162 
163  ~StringOrOpToEmit() {
164  if (const void *ptr = pointerData.dyn_cast<const void *>())
165  free(const_cast<void *>(ptr));
166  }
167 
168  /// If the value is an Operation*, return it. Otherwise return null.
169  Operation *getOperation() const {
170  return pointerData.dyn_cast<Operation *>();
171  }
172 
173  /// If the value wraps a string, return it. Otherwise return null.
174  StringRef getStringData() const {
175  if (const void *ptr = pointerData.dyn_cast<const void *>())
176  return StringRef((const char *)ptr, length);
177  return StringRef();
178  }
179 
180  /// This method transforms the entry from an operation to a string value.
181  void setString(StringRef value) {
182  assert(pointerData.is<Operation *>() && "shouldn't already be a string");
183  length = value.size();
184  void *data = malloc(length);
185  memcpy(data, value.data(), length);
186  pointerData = (const void *)data;
187  }
188 
189  // These move just fine.
190  StringOrOpToEmit(StringOrOpToEmit &&rhs)
191  : pointerData(rhs.pointerData), length(rhs.length) {
192  rhs.pointerData = (Operation *)nullptr;
193  }
194 
195 private:
196  StringOrOpToEmit(const StringOrOpToEmit &) = delete;
197  void operator=(const StringOrOpToEmit &) = delete;
198  PointerUnion<Operation *, const void *> pointerData;
199  size_t length;
200 };

257 /// Return true for operations that must always be inlined into a containing
258 /// expression for correctness.
259 static inline bool isExpressionAlwaysInline(Operation *op) {
260  // We need to emit array indexes inline per verilog "lvalue" semantics.
261  if (isa<sv::ArrayIndexInOutOp>(op) || isa<sv::StructFieldInOutOp>(op) ||
262  isa<sv::IndexedPartSelectInOutOp>(op) || isa<sv::ReadInOutOp>(op))
263  return true;
264 
265  // An SV interface modport is a symbolic name that is always inlined.
266  if (isa<sv::GetModportOp>(op) || isa<sv::ReadInterfaceSignalOp>(op))
267  return true;
268 
269  // XMRs can't be spilled if they are on the lhs. Conservatively never spill
270  // them.
271  if (isa<sv::XMROp>(op))
272  return true;
273 
274  if (isa<sv::SampledOp>(op))
275  return true;
276 
277  return false;
278 }
