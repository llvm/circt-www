25  /// Helper function to convert a value to its simple bit vector
26  /// representation, if it has one. Otherwise returns null.
27  Value convertToSimpleBitVector(Value value) {
28  if (!value)
29  return {};
30  if (auto type = dyn_cast_or_null<moore::UnpackedType>(value.getType())) {
31  if (type.isSimpleBitVector())
32  return value;
33  if (auto sbvt = type.castToSimpleBitVectorOrNull())
34  return builder.create<moore::ConversionOp>(
35  loc, sbvt.getType(builder.getContext()), value);
36  }
37  mlir::emitError(loc, "expression of type ")
38  << value.getType() << " cannot be cast to a simple bit vector";
39  return {};
40  }

42  /// Helper function to convert a value to its "truthy" boolean value.
43  Value convertToBool(Value value) {
44  if (!value)
45  return {};
46  if (auto type = dyn_cast_or_null<moore::IntType>(value.getType()))
47  if (type.getBitSize() == 1)
48  return value;
49  if (auto type = dyn_cast_or_null<moore::UnpackedType>(value.getType()))
50  return builder.create<moore::BoolCastOp>(loc, value);
51  mlir::emitError(loc, "expression of type ")
52  << value.getType() << " cannot be cast to a boolean";
53  return {};
54  }

84  // Handle blocking and non-blocking assignments.
85  Value visit(const slang::ast::AssignmentExpression &expr) {
86  auto lhs = context.convertExpression(expr.left());
87  context.lvalueStack.push_back(lhs);
88  auto rhs = context.convertExpression(expr.right());
89  context.lvalueStack.pop_back();
90  if (!lhs || !rhs)
91  return {};
92 
93  if (lhs.getType() != rhs.getType())
94  rhs = builder.create<moore::ConversionOp>(loc, lhs.getType(), rhs);
95 
96  if (expr.timingControl) {
97  auto loc = context.convertLocation(expr.timingControl->sourceRange);
98  mlir::emitError(loc, "delayed assignments not supported");
99  return {};
100  }
101 
102  if (expr.isNonBlocking())
103  builder.create<moore::NonBlockingAssignOp>(loc, lhs, rhs);
104  else
105  builder.create<moore::BlockingAssignOp>(loc, lhs, rhs);
106  return rhs;
107  }

109  // Helper function to convert an argument to a simple bit vector type, pass it
110  // to a reduction op, and optionally invert the result.
111  template <class ConcreteOp>
112  Value createReduction(Value arg, bool invert) {
113  arg = convertToSimpleBitVector(arg);
114  if (!arg)
115  return {};
116  Value result = builder.create<ConcreteOp>(loc, arg);
117  if (invert)
118  result = builder.create<moore::NotOp>(loc, result);
119  return result;
120  }

122  // Helper function to create pre and post increments and decrements.
123  Value createIncrement(Value arg, bool isInc, bool isPost) {
124  auto preValue = convertToSimpleBitVector(arg);
125  if (!preValue)
126  return {};
127  preValue = builder.create<moore::ReadLValueOp>(loc, preValue);
128  auto one = builder.create<moore::ConstantOp>(loc, preValue.getType(), 1);
129  auto postValue =
130  isInc ? builder.create<moore::AddOp>(loc, preValue, one).getResult()
131  : builder.create<moore::SubOp>(loc, preValue, one).getResult();
132  builder.create<moore::BlockingAssignOp>(loc, arg, postValue);
133  return isPost ? preValue : postValue;
134  }

136  // Handle unary operators.
137  Value visit(const slang::ast::UnaryExpression &expr) {
138  auto arg = context.convertExpression(expr.operand());
139  if (!arg)
140  return {};
141 
142  using slang::ast::UnaryOperator;
143  switch (expr.op) {
144  // `+a` is simply `a`, but converted to a simple bit vector type since
145  // this is technically an arithmetic operation.
146  case UnaryOperator::Plus:
147  return convertToSimpleBitVector(arg);
148 
149  case UnaryOperator::Minus:
150  arg = convertToSimpleBitVector(arg);
151  if (!arg)
152  return {};
153  return builder.create<moore::NegOp>(loc, arg);
154 
155  case UnaryOperator::BitwiseNot:
156  arg = convertToSimpleBitVector(arg);
157  if (!arg)
158  return {};
159  return builder.create<moore::NotOp>(loc, arg);
160 
161  case UnaryOperator::BitwiseAnd:
162  return createReduction<moore::ReduceAndOp>(arg, false);
163  case UnaryOperator::BitwiseOr:
164  return createReduction<moore::ReduceOrOp>(arg, false);
165  case UnaryOperator::BitwiseXor:
166  return createReduction<moore::ReduceXorOp>(arg, false);
167  case UnaryOperator::BitwiseNand:
168  return createReduction<moore::ReduceAndOp>(arg, true);
169  case UnaryOperator::BitwiseNor:
170  return createReduction<moore::ReduceOrOp>(arg, true);
171  case UnaryOperator::BitwiseXnor:
172  return createReduction<moore::ReduceXorOp>(arg, true);
173 
174  case UnaryOperator::LogicalNot:
175  arg = convertToBool(arg);
176  if (!arg)
177  return {};
178  return builder.create<moore::NotOp>(loc, arg);
179 
180  case UnaryOperator::Preincrement:
181  return createIncrement(arg, true, false);
182  case UnaryOperator::Predecrement:
183  return createIncrement(arg, false, false);
184  case UnaryOperator::Postincrement:
185  return createIncrement(arg, true, true);
186  case UnaryOperator::Postdecrement:
187  return createIncrement(arg, false, true);
188  }
189 
190  mlir::emitError(loc, "unsupported unary operator");
191  return {};
192  }

194  // Helper function to convert two arguments to a simple bit vector type and
195  // pass them into a binary op.
196  template <class ConcreteOp>
197  Value createBinary(Value lhs, Value rhs) {
198  lhs = convertToSimpleBitVector(lhs);
199  rhs = convertToSimpleBitVector(rhs);
200  if (!lhs || !rhs)
201  return {};
202  return builder.create<ConcreteOp>(loc, lhs, rhs);

205  // Handle binary operators.
206  Value visit(const slang::ast::BinaryExpression &expr) {
207  auto lhs = context.convertExpression(expr.left());
208  auto rhs = context.convertExpression(expr.right());
209  if (!lhs || !rhs)
210  return {};
211 
212  using slang::ast::BinaryOperator;
213  switch (expr.op) {
214  case BinaryOperator::Add:
215  return createBinary<moore::AddOp>(lhs, rhs);
216  case BinaryOperator::Subtract:
217  return createBinary<moore::SubOp>(lhs, rhs);
218  case BinaryOperator::Multiply:
219  return createBinary<moore::MulOp>(lhs, rhs);
220  case BinaryOperator::Divide:
221  return createBinary<moore::DivOp>(lhs, rhs);
222  case BinaryOperator::Mod:
223  return createBinary<moore::ModOp>(lhs, rhs);
224 
225  case BinaryOperator::BinaryAnd:
226  return createBinary<moore::AndOp>(lhs, rhs);
227  case BinaryOperator::BinaryOr:
228  return createBinary<moore::OrOp>(lhs, rhs);
229  case BinaryOperator::BinaryXor:
230  return createBinary<moore::XorOp>(lhs, rhs);
231  case BinaryOperator::BinaryXnor: {
232  auto result = createBinary<moore::XorOp>(lhs, rhs);
233  if (!result)
234  return {};
235  return builder.create<moore::NotOp>(loc, result);
236  }
237 
238  case BinaryOperator::Equality:
239  return createBinary<moore::EqOp>(lhs, rhs);
240  case BinaryOperator::Inequality:
241  return createBinary<moore::NeOp>(lhs, rhs);
242  case BinaryOperator::CaseEquality:
243  return createBinary<moore::CaseEqOp>(lhs, rhs);
244  case BinaryOperator::CaseInequality:
245  return createBinary<moore::CaseNeOp>(lhs, rhs);
246  case BinaryOperator::WildcardEquality:
247  return createBinary<moore::WildcardEqOp>(lhs, rhs);
248  case BinaryOperator::WildcardInequality:
249  return createBinary<moore::WildcardNeOp>(lhs, rhs);
250 
251  case BinaryOperator::GreaterThanEqual:
252  return createBinary<moore::GeOp>(lhs, rhs);
253  case BinaryOperator::GreaterThan:
254  return createBinary<moore::GtOp>(lhs, rhs);
255  case BinaryOperator::LessThanEqual:
256  return createBinary<moore::LeOp>(lhs, rhs);
257  case BinaryOperator::LessThan:
258  return createBinary<moore::LtOp>(lhs, rhs);
259 
260  // See IEEE 1800-2017 ยง 11.4.7 "Logical operators".
261  case BinaryOperator::LogicalAnd: {
262  // TODO: This should short-circuit. Put the RHS code into an scf.if.
263  lhs = convertToBool(lhs);
264  rhs = convertToBool(rhs);
265  if (!lhs || !rhs)
266  return {};
267  return builder.create<moore::AndOp>(loc, lhs, rhs);
268  }
269  case BinaryOperator::LogicalOr: {
270  // TODO: This should short-circuit. Put the RHS code into an scf.if.
271  lhs = convertToBool(lhs);
272  rhs = convertToBool(rhs);
273  if (!lhs || !rhs)
274  return {};
275  return builder.create<moore::OrOp>(loc, lhs, rhs);
276  }
277  case BinaryOperator::LogicalImplication: {
278  // `(lhs -> rhs)` equivalent to `(!lhs || rhs)`.
279  lhs = convertToBool(lhs);
280  rhs = convertToBool(rhs);
281  if (!lhs || !rhs)
282  return {};
283  auto notLHS = builder.create<moore::NotOp>(loc, lhs);
284  return builder.create<moore::OrOp>(loc, notLHS, rhs);
285  }
286  case BinaryOperator::LogicalEquivalence: {
287  // `(lhs <-> rhs)` equivalent to `(lhs && rhs) || (!lhs && !rhs)`.
288  lhs = convertToBool(lhs);
289  rhs = convertToBool(rhs);
290  if (!lhs || !rhs)
291  return {};
292  auto notLHS = builder.create<moore::NotOp>(loc, lhs);
293  auto notRHS = builder.create<moore::NotOp>(loc, rhs);
294  auto both = builder.create<moore::AndOp>(loc, lhs, rhs);
295  auto notBoth = builder.create<moore::AndOp>(loc, notLHS, notRHS);
296  return builder.create<moore::OrOp>(loc, both, notBoth);
297  }
298 
299  case BinaryOperator::LogicalShiftLeft:
300  return createBinary<moore::ShlOp>(lhs, rhs);
301  case BinaryOperator::LogicalShiftRight:
302  return createBinary<moore::ShrOp>(lhs, rhs);
303  case BinaryOperator::ArithmeticShiftLeft:
304  return createBinary<moore::ShlOp>(lhs, rhs);
305  case BinaryOperator::ArithmeticShiftRight: {
306  // The `>>>` operator is an arithmetic right shift if the LHS operand is
307  // signed, or a logical right shift if the operand is unsigned.
308  lhs = convertToSimpleBitVector(lhs);
309  rhs = convertToSimpleBitVector(rhs);
310  if (!lhs || !rhs)
311  return {};
312  if (cast<moore::PackedType>(lhs.getType())
313  .getSimpleBitVector()
314  .isSigned())
315  return builder.create<moore::AShrOp>(loc, lhs, rhs);
316  return builder.create<moore::ShrOp>(loc, lhs, rhs);
317  }
318 
319  case BinaryOperator::Power:
320  break;
321  }
322 
323  mlir::emitError(loc, "unsupported binary operator");
324  return {};
325  }

327  // Materialize a Slang integer literal as a constant op.
328  Value convertSVInt(const slang::SVInt &value, Type type) {
329  if (value.hasUnknown()) {
330  mlir::emitError(loc, "literals with X or Z bits not supported");
331  return {};
332  }
333  if (value.getBitWidth() > 64) {
334  mlir::emitError(loc, "unsupported bit width: literal is ")
335  << value.getBitWidth() << " bits wide; only 64 supported";
336  return {};
337  }
338  auto truncValue = value.as<uint64_t>().value();
339  return builder.create<moore::ConstantOp>(loc, type, truncValue);
340  }
