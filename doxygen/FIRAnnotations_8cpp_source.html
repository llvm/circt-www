35 /// Split a target into a base target (including a reference if one exists) and
36 /// an optional array of subfield/subindex tokens.
37 static std::pair<StringRef, llvm::Optional<ArrayAttr>>
38 splitTarget(StringRef target, MLIRContext *context) {
39  if (target.empty())
40  return {target, None};
41 
42  // Find the string index where the target can be partitioned into the "base
43  // target" and the "target". The "base target" is the module or instance and
44  // the "target" is everything else. This has two variants that need to be
45  // considered:
46  //
47  // 1) A Local target, e.g., ~Foo|Foo>bar.baz
48  // 2) An instance target, e.g., ~Foo|Foo/bar:Bar>baz.qux
49  //
50  // In (1), this should be partitioned into ["~Foo|Foo>bar", ".baz"]. In (2),
51  // this should be partitioned into ["~Foo|Foo/bar:Bar", ">baz.qux"].
52  bool isInstance = false;
53  size_t fieldBegin = target.find_if_not([&isInstance](char c) {
54  switch (c) {
55  case '/':
56  return isInstance = true;
57  case '>':
58  return !isInstance;
59  case '[':
60  case '.':
61  return false;
62  default:
63  return true;
64  };
65  });
66 
67  // Exit if the target does not contain a subfield or subindex.
68  if (fieldBegin == StringRef::npos)
69  return {target, None};
70 
71  auto targetBase = target.take_front(fieldBegin);
72  target = target.substr(fieldBegin);
73  SmallVector<Attribute> annotationVec;
74  SmallString<16> temp;
75  for (auto c : target.drop_front()) {
76  if (c == ']') {
77  // Create a IntegerAttr with the previous sub-index token.
78  APInt subIndex;
79  if (!temp.str().getAsInteger(10, subIndex))
80  annotationVec.push_back(IntegerAttr::get(IntegerType::get(context, 64),
81  subIndex.getZExtValue()));
82  else
83  // We don't have a good way to emit error here. This will be reported as
84  // an error in the FIRRTL parser.
85  annotationVec.push_back(StringAttr::get(context, temp));
86  temp.clear();
87  } else if (c == '[' || c == '.') {
88  // Create a StringAttr with the previous token.
89  if (!temp.empty())
90  annotationVec.push_back(StringAttr::get(context, temp));
91  temp.clear();
92  } else
93  temp.push_back(c);
94  }
95  // Save the last token.
96  if (!temp.empty())
97  annotationVec.push_back(StringAttr::get(context, temp));
98 
99  return {targetBase, ArrayAttr::get(context, annotationVec)};
100 }

102 /// Split out non-local paths. This will return a set of target strings for
103 /// each named entity along the path.
104 /// c|c:ai/Am:bi/Bm>d.agg[3] ->
105 /// c|c>ai, c|Am>bi, c|Bm>d.agg[2]
106 static SmallVector<std::tuple<std::string, StringRef, StringRef>>
107 expandNonLocal(StringRef target) {
108  SmallVector<std::tuple<std::string, StringRef, StringRef>> retval;
109  StringRef circuit;
110  std::tie(circuit, target) = target.split('|');
111  while (target.count(':')) {
112  StringRef nla;
113  std::tie(nla, target) = target.split(':');
114  StringRef inst, mod;
115  std::tie(mod, inst) = nla.split('/');
116  retval.emplace_back((circuit + "|" + mod + ">" + inst).str(), mod, inst);
117  }
118  if (target.empty()) {
119  retval.emplace_back(circuit.str(), "", "");
120  } else {
121 
122  StringRef mod, name;
123  // remove aggregate
124  auto targetBase =
125  target.take_until([](char c) { return c == '.' || c == '['; });
126  std::tie(mod, name) = targetBase.split('>');
127  retval.emplace_back((circuit + "|" + target).str(), mod, name);
128  }
129  return retval;
130 }

132 /// Make an anchor for a non-local annotation. Use the expanded path to build
133 /// the module and name list in the anchor.
134 static FlatSymbolRefAttr
135 buildNLA(CircuitOp circuit, size_t nlaSuffix,
136  SmallVectorImpl<std::tuple<std::string, StringRef, StringRef>> &nlas) {
137  OpBuilder b(circuit.getBodyRegion());
138  MLIRContext *ctxt = circuit.getContext();
139  SmallVector<Attribute> insts;
140  for (size_t i = 0, e = nlas.size() - 1; i < e; ++i) {
141  auto &nla = nlas[i];
142  // Assumption: Symbol name = Operation name.
143  auto module = std::get<1>(nla);
144  auto inst = std::get<2>(nla);
145  insts.push_back(hw::InnerRefAttr::get(StringAttr::get(ctxt, module),
146  StringAttr::get(ctxt, inst)));
147  }
148  insts.push_back(FlatSymbolRefAttr::get(ctxt, std::get<1>(nlas.back())));
149 
150  auto instAttr = ArrayAttr::get(ctxt, insts);
151  auto nla = b.create<HierPathOp>(circuit.getLoc(),
152  "nla_" + std::to_string(nlaSuffix), instAttr);
153  return FlatSymbolRefAttr::get(nla);
154 }

161 /// Mutably update a prototype Annotation (stored as a `NamedAttrList`) with
162 /// subfield/subindex information from a Target string. Subfield/subindex
163 /// information will be placed in the key "target" at the back of the
164 /// Annotation. If no subfield/subindex information, the Annotation is
165 /// unmodified. Return the split input target as a base target (include a
166 /// reference if one exists) and an optional array containing subfield/subindex
167 /// tokens.
168 static std::pair<StringRef, llvm::Optional<ArrayAttr>>
169 splitAndAppendTarget(NamedAttrList &annotation, StringRef target,
170  MLIRContext *context) {
171  auto targetPair = splitTarget(target, context);
172  if (targetPair.second.hasValue())
173  appendTarget(annotation, targetPair.second.getValue());
174 
175  return targetPair;
176 }

178 /// Return an input \p target string in canonical form. This converts a Legacy
179 /// Annotation (e.g., A.B.C) into a modern annotation (e.g., ~A|B>C). Trailing
180 /// subfield/subindex references are preserved.
181 static llvm::Optional<std::string> oldCanonicalizeTarget(StringRef target) {
182 
183  if (target.empty())
184  return {};
185 
186  // If this is a normal Target (not a Named), erase that field in the JSON
187  // object and return that Target.
188  if (target[0] == '~')
189  return target.str();
190 
191  // This is a legacy target using the firrtl.annotations.Named type. This
192  // can be trivially canonicalized to a non-legacy target, so we do it with
193  // the following three mappings:
194  // 1. CircuitName => CircuitTarget, e.g., A -> ~A
195  // 2. ModuleName => ModuleTarget, e.g., A.B -> ~A|B
196  // 3. ComponentName => ReferenceTarget, e.g., A.B.C -> ~A|B>C
197  std::string newTarget = "~";
198  llvm::raw_string_ostream s(newTarget);
199  unsigned tokenIdx = 0;
200  for (auto a : target) {
201  if (a == '.') {
202  switch (tokenIdx) {
203  case 0:
204  s << "|";
205  break;
206  case 1:
207  s << ">";
208  break;
209  default:
210  s << ".";
211  break;
212  }
213  ++tokenIdx;
214  } else
215  s << a;
216  }
217  return llvm::Optional<std::string>(newTarget);
218 }

220 /// Convert arbitrary JSON to an MLIR Attribute.
221 static Attribute convertJSONToAttribute(MLIRContext *context,
222  json::Value &value, json::Path p) {
223  // String or quoted JSON
224  if (auto a = value.getAsString()) {
225  // Test to see if this might be quoted JSON (a string that is actually
226  // JSON). Sometimes FIRRTL developers will do this to serialize objects
227  // that the Scala FIRRTL Compiler doesn't know about.
228  auto unquotedValue = json::parse(a.getValue());
229  auto err = unquotedValue.takeError();
230  // If this parsed without an error, then it's more JSON and recurse on
231  // that.
232  if (!err)
233  return convertJSONToAttribute(context, unquotedValue.get(), p);
234  // If there was an error, then swallow it and handle this as a string.
235  handleAllErrors(std::move(err), [&](const json::ParseError &a) {});
236  return StringAttr::get(context, a.getValue());
237  }
238 
239  // Integer
240  if (auto a = value.getAsInteger())
241  return IntegerAttr::get(IntegerType::get(context, 64), a.getValue());
242 
243  // Float
244  if (auto a = value.getAsNumber())
245  return FloatAttr::get(mlir::FloatType::getF64(context), a.getValue());
246 
247  // Boolean
248  if (auto a = value.getAsBoolean())
249  return BoolAttr::get(context, a.getValue());
250 
251  // Null
252  if (auto a = value.getAsNull())
253  return mlir::UnitAttr::get(context);
254 
255  // Object
256  if (auto a = value.getAsObject()) {
257  NamedAttrList metadata;
258  for (auto b : *a)
259  metadata.append(
260  b.first, convertJSONToAttribute(context, b.second, p.field(b.first)));
261  return DictionaryAttr::get(context, metadata);
262  }
263 
264  // Array
265  if (auto a = value.getAsArray()) {
266  SmallVector<Attribute> metadata;
267  for (size_t i = 0, e = (*a).size(); i != e; ++i)
268  metadata.push_back(convertJSONToAttribute(context, (*a)[i], p.index(i)));
269  return ArrayAttr::get(context, metadata);
270  }
271 
272  llvm_unreachable("Impossible unhandled JSON type");
273 }

275 static std::string addNLATargets(
276  MLIRContext *context, StringRef targetStrRef, CircuitOp circuit,
277  size_t &nlaNumber, NamedAttrList &metadata,
278  llvm::StringMap<llvm::SmallVector<Attribute>> &mutableAnnotationMap) {
279 
280  auto nlaTargets = expandNonLocal(targetStrRef);
281 
282  FlatSymbolRefAttr nlaSym;
283  if (nlaTargets.size() > 1) {
284  nlaSym = buildNLA(circuit, ++nlaNumber, nlaTargets);
285  metadata.append("circt.nonlocal", nlaSym);
286  }
287 
288  for (int i = 0, e = nlaTargets.size() - 1; i < e; ++i) {
289  NamedAttrList dontTouch;
290  dontTouch.append(
291  "class",
292  StringAttr::get(context, "firrtl.transforms.DontTouchAnnotation"));
293  // Every op with nonlocal anchor must have a symbol. Hence add the
294  // dontTouch.
295  mutableAnnotationMap[std::get<0>(nlaTargets[i])].push_back(
296  DictionaryAttr::get(context, dontTouch));
297  }
298 
299  // Annotations on the element instance.
300  auto leafTarget =
301  splitAndAppendTarget(metadata, std::get<0>(nlaTargets.back()), context)
302  .first;
303 
304  return leafTarget.str();
305 }

307 /// Deserialize a JSON value into FIRRTL Annotations. Annotations are
308 /// represented as a Target-keyed arrays of attributes. The input JSON value is
309 /// checked, at runtime, to be an array of objects. Returns true if successful,
310 /// false if unsuccessful.
311 bool circt::firrtl::fromJSON(json::Value &value, StringRef circuitTarget,
312  llvm::StringMap<ArrayAttr> &annotationMap,
313  json::Path path, CircuitOp circuit,
314  size_t &nlaNumber) {
315  auto context = circuit.getContext();
316 
317  /// Examine an Annotation JSON object and return an optional string indicating
318  /// the target associated with this annotation. Erase the target from the
319  /// JSON object if a target was found. Automatically convert any legacy Named
320  /// targets to actual Targets. Note: it is expected that a target may not
321  /// exist, e.g., any subclass of firrtl.annotations.NoTargetAnnotation will
322  /// not have a target.
323  auto findAndEraseTarget = [](json::Object *object,
324  json::Path p) -> llvm::Optional<std::string> {
325  // If no "target" field exists, then promote the annotation to a
326  // CircuitTarget annotation by returning a target of "~".
327  auto maybeTarget = object->get("target");
328  if (!maybeTarget)
329  return llvm::Optional<std::string>("~");
330 
331  // Find the target.
332  auto maybeTargetStr = maybeTarget->getAsString();
333  if (!maybeTargetStr) {
334  p.field("target").report("target must be a string type");
335  return {};
336  }
337  auto canonTargetStr = oldCanonicalizeTarget(maybeTargetStr.getValue());
338  if (!canonTargetStr) {
339  p.field("target").report("invalid target string");
340  return {};
341  }
342 
343  auto target = canonTargetStr.getValue();
344 
345  // Remove the target field from the annotation and return the target.
346  object->erase("target");
347  return llvm::Optional<std::string>(target);
348  };
349 
350  // The JSON value must be an array of objects. Anything else is reported as
351  // invalid.
352  auto array = value.getAsArray();
353  if (!array) {
354  path.report(
355  "Expected annotations to be an array, but found something else.");
356  return false;
357  }
358 
359  // Build a mutable map of Target to Annotation.
360  llvm::StringMap<llvm::SmallVector<Attribute>> mutableAnnotationMap;
361  for (size_t i = 0, e = (*array).size(); i != e; ++i) {
362  auto object = (*array)[i].getAsObject();
363  auto p = path.index(i);
364  if (!object) {
365  p.report("Expected annotations to be an array of objects, but found an "
366  "array of something else.");
367  return false;
368  }
369 
370  // If the annotation has a class name which matches an annotation which the
371  // LowerAnnotations pass knows about, then defer its processing.
372  if (auto *clazz = object->get("class")) {
373  auto classString = clazz->getAsString();
374  if (classString && isAnnoClassLowered(classString.getValue())) {
375  NamedAttrList metadata;
376  for (auto field : *object) {
377  if (auto value = convertJSONToAttribute(context, field.second, p)) {
378  metadata.append(field.first, value);
379  continue;
380  }
381  return false;
382  }
383  mutableAnnotationMap[rawAnnotations].push_back(
384  DictionaryAttr::get(context, metadata));
385  continue;
386  }
387  }
388 
389  // Find and remove the "target" field from the Annotation object if it
390  // exists. In the FIRRTL Dialect, the target will be implicitly specified
391  // based on where the attribute is applied.
392  auto optTarget = findAndEraseTarget(object, p);
393  if (!optTarget)
394  return false;
395  StringRef targetStrRef = optTarget.getValue();
396 
397  if (targetStrRef != "~") {
398  auto circuitFieldEnd = targetStrRef.find_first_of('|');
399  if (circuitTarget != targetStrRef.take_front(circuitFieldEnd)) {
400  p.report("annotation has invalid circuit name");
401  return false;
402  }
403  }
404 
405  // Build up the Attribute to represent the Annotation and store it in the
406  // global Target -> Attribute mapping.
407  NamedAttrList metadata;
408  for (auto field : *object) {
409  if (auto value = convertJSONToAttribute(context, field.second, p)) {
410  metadata.append(field.first, value);
411  continue;
412  }
413  return false;
414  }
415 
416  auto leafTarget = addNLATargets(context, targetStrRef, circuit, nlaNumber,
417  metadata, mutableAnnotationMap);
418 
419  mutableAnnotationMap[leafTarget].push_back(
420  DictionaryAttr::get(context, metadata));
421  }
422 
423  // Convert the mutable Annotation map to a SmallVector<ArrayAttr>.
424  for (auto a : mutableAnnotationMap.keys()) {
425  // If multiple annotations on a single object, then append it.
426  if (annotationMap.count(a))
427  for (auto attr : annotationMap[a])
428  mutableAnnotationMap[a].push_back(attr);
429 
430  annotationMap[a] = ArrayAttr::get(context, mutableAnnotationMap[a]);
431  }
432 
433  return true;
434 }

436 /// Convert a JSON value containing OMIR JSON (an array of OMNodes), convert
437 /// this to an OMIRAnnotation, and add it to a mutable `annotationMap` argument.
438 bool circt::firrtl::fromOMIRJSON(json::Value &value, StringRef circuitTarget,
439  llvm::StringMap<ArrayAttr> &annotationMap,
440  json::Path path, MLIRContext *context) {
441  // The JSON value must be an array of objects. Anything else is reported as
442  // invalid.
443  auto *array = value.getAsArray();
444  if (!array) {
445  path.report(
446  "Expected OMIR to be an array of nodes, but found something else.");
447  return false;
448  }
449 
450  // Build a mutable map of Target to Annotation.
451  SmallVector<Attribute> omnodes;
452  for (size_t i = 0, e = (*array).size(); i != e; ++i) {
453  auto *object = (*array)[i].getAsObject();
454  auto p = path.index(i);
455  if (!object) {
456  p.report("Expected OMIR to be an array of objects, but found an array of "
457  "something else.");
458  return false;
459  }
460 
461  // Manually built up OMNode.
462  NamedAttrList omnode;
463 
464  // Validate that this looks like an OMNode. This should have three fields:
465  // - "info": String
466  // - "id": String that starts with "OMID:"
467  // - "fields": Array<Object>
468  // Fields is optional and is a dictionary encoded as an array of objects:
469  // - "info": String
470  // - "name": String
471  // - "value": JSON
472  // The dictionary is keyed by the "name" member and the array of fields is
473  // guaranteed to not have collisions of the "name" key.
474  auto maybeInfo = object->getString("info");
475  if (!maybeInfo) {
476  p.report("OMNode missing mandatory member \"info\" with type \"string\"");
477  return false;
478  }
479  auto maybeID = object->getString("id");
480  if (!maybeID || !maybeID.getValue().startswith("OMID:")) {
481  p.report("OMNode missing mandatory member \"id\" with type \"string\" "
482  "that starts with \"OMID:\"");
483  return false;
484  }
485  auto *maybeFields = object->get("fields");
486  if (maybeFields && !maybeFields->getAsArray()) {
487  p.report("OMNode has \"fields\" member with incorrect type (expected "
488  "\"array\")");
489  return false;
490  }
491  Attribute fields;
492  if (!maybeFields)
493  fields = DictionaryAttr::get(context, {});
494  else {
495  auto array = *maybeFields->getAsArray();
496  NamedAttrList fieldAttrs;
497  for (size_t i = 0, e = array.size(); i != e; ++i) {
498  auto *field = array[i].getAsObject();
499  auto pI = p.field("fields").index(i);
500  if (!field) {
501  pI.report("OMNode has field that is not an \"object\"");
502  return false;
503  }
504  auto maybeInfo = field->getString("info");
505  if (!maybeInfo) {
506  pI.report(
507  "OMField missing mandatory member \"info\" with type \"string\"");
508  return false;
509  }
510  auto maybeName = field->getString("name");
511  if (!maybeName) {
512  pI.report(
513  "OMField missing mandatory member \"name\" with type \"string\"");
514  return false;
515  }
516  auto *maybeValue = field->get("value");
517  if (!maybeValue) {
518  pI.report("OMField missing mandatory member \"value\"");
519  return false;
520  }
521  NamedAttrList values;
522  values.append("info", StringAttr::get(context, maybeInfo.getValue()));
523  values.append("value", convertJSONToAttribute(context, *maybeValue,
524  pI.field("value")));
525  fieldAttrs.append(maybeName.getValue(),
526  DictionaryAttr::get(context, values));
527  }
528  fields = DictionaryAttr::get(context, fieldAttrs);
529  }
530 
531  omnode.append("info", StringAttr::get(context, maybeInfo.getValue()));
532  omnode.append("id", convertJSONToAttribute(context, *object->get("id"),
533  p.field("id")));
534  omnode.append("fields", fields);
535  omnodes.push_back(DictionaryAttr::get(context, omnode));
536  }
537 
538  NamedAttrList omirAnnoFields;
539  omirAnnoFields.append("class", StringAttr::get(context, omirAnnoClass));
540  omirAnnoFields.append("nodes", convertJSONToAttribute(context, value, path));
541 
542  DictionaryAttr omirAnno = DictionaryAttr::get(context, omirAnnoFields);
543 
544  // If no circuit annotations exist, just insert the OMIRAnnotation.
545  auto &oldAnnotations = annotationMap[rawAnnotations];
546  if (!oldAnnotations) {
547  oldAnnotations = ArrayAttr::get(context, {omirAnno});
548  return true;
549  }
550 
551  // Rewrite the ArrayAttr for the circuit.
552  SmallVector<Attribute> newAnnotations(oldAnnotations.begin(),
553  oldAnnotations.end());
554  newAnnotations.push_back(omirAnno);
555  oldAnnotations = ArrayAttr::get(context, newAnnotations);
556 
557  return true;
558 }

560 /// Deserialize a JSON value into FIRRTL Annotations. Annotations are
561 /// represented as a Target-keyed arrays of attributes. The input JSON value is
562 /// checked, at runtime, to be an array of objects. Returns true if successful,
563 /// false if unsuccessful.
564 bool circt::firrtl::fromJSONRaw(json::Value &value, StringRef circuitTarget,
565  SmallVectorImpl<Attribute> &attrs,
566  json::Path path, MLIRContext *context) {
567 
568  // The JSON value must be an array of objects. Anything else is reported as
569  // invalid.
570  auto array = value.getAsArray();
571  if (!array) {
572  path.report(
573  "Expected annotations to be an array, but found something else.");
574  return false;
575  }
576 
577  // Build an array of annotations.
578  for (size_t i = 0, e = (*array).size(); i != e; ++i) {
579  auto object = (*array)[i].getAsObject();
580  auto p = path.index(i);
581  if (!object) {
582  p.report("Expected annotations to be an array of objects, but found an "
583  "array of something else.");
584  return false;
585  }
586 
587  // Build up the Attribute to represent the Annotation
588  NamedAttrList metadata;
589 
590  for (auto field : *object) {
591  if (auto value = convertJSONToAttribute(context, field.second, p)) {
592  metadata.append(field.first, value);
593  continue;
594  }
595  return false;
596  }
597 
598  attrs.push_back(DictionaryAttr::get(context, metadata));
599  }
600 
601  return true;
602 }
