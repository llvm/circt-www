208 /// Emit an entire circuit.
209 void Emitter::emitCircuit(CircuitOp op) {
210  circuitNamespace.add(op);
211  indent() << "circuit " << op.name() << " :\n";
212  addIndent();
213  for (auto &bodyOp : *op.getBody()) {
214  if (encounteredError)
215  break;
216  TypeSwitch<Operation *>(&bodyOp)
217  .Case<FModuleOp, FExtModuleOp>([&](auto op) {
218  emitModule(op);
219  os << "\n";
220  })
221  .Default([&](auto op) {
222  emitOpError(op, "not supported for emission inside circuit");
223  });
224  }
225  reduceIndent();
226  circuitNamespace.clear();
227 }

229 /// Emit an entire module.
230 void Emitter::emitModule(FModuleOp op) {
231  indent() << "module " << op.getName() << " :\n";
232  addIndent();
233 
234  // Emit the ports.
235  auto ports = op.getPorts();
236  emitModulePorts(ports, op.getArguments());
237  if (!ports.empty() && !op.getBody()->empty())
238  os << "\n";
239 
240  // Emit the module body.
241  emitStatementsInBlock(*op.getBody());
242 
243  reduceIndent();
244  valueNames.clear();
245  valueNamesStorage.clear();
246 }

248 /// Emit an external module.
249 void Emitter::emitModule(FExtModuleOp op) {
250  indent() << "extmodule " << op.getName() << " :\n";
251  addIndent();
252 
253  // Emit the ports.
254  auto ports = op.getPorts();
255  emitModulePorts(ports);
256 
257  // Emit the optional `defname`.
258  if (op.defname() && !op.defname()->empty())
259  indent() << "defname = " << op.defname() << "\n";
260 
261  // Emit the parameters.
262  for (auto param : llvm::map_range(op.parameters(), [](Attribute attr) {
263  return attr.cast<ParamDeclAttr>();
264  })) {
265  indent() << "parameter " << param.getName().getValue() << " = ";
266  TypeSwitch<Attribute>(param.getValue())
267  .Case<IntegerAttr>([&](auto attr) { os << attr.getValue(); })
268  .Case<FloatAttr>([&](auto attr) {
269  SmallString<16> str;
270  attr.getValue().toString(str);
271  os << str;
272  })
273  .Case<StringAttr>([&](auto attr) {
274  os << "\"";
275  os.write_escaped(attr.getValue());
276  os << "\"";
277  })
278  .Default([&](auto attr) {
279  emitOpError(op, "with unsupported parameter attribute: ") << attr;
280  os << "<unsupported-attr " << attr << ">";
281  });
282  os << "\n";
283  }
284 
285  reduceIndent();
286 }

288 /// Emit the ports of a module or extmodule. If the `arguments` array is
289 /// non-empty, it is used to populate `emittedNames` with the port names for use
290 /// during expression emission.
291 void Emitter::emitModulePorts(ArrayRef<PortInfo> ports,
292  Block::BlockArgListType arguments) {
293  for (unsigned i = 0, e = ports.size(); i < e; ++i) {
294  const auto &port = ports[i];

296  if (!arguments.empty())

310 void Emitter::emitStatementsInBlock(Block &block) {
311  for (auto &bodyOp : block) {
312  if (encounteredError)
313  return;
314  if (isEmittedInline(&bodyOp))
315  continue;
316  TypeSwitch<Operation *>(&bodyOp)
317  .Case<WhenOp, WireOp, RegOp, RegResetOp, NodeOp, StopOp, SkipOp,
318  PrintFOp, AssertOp, AssumeOp, CoverOp, ConnectOp, StrictConnectOp,
319  InstanceOp, AttachOp, MemOp, InvalidValueOp, SeqMemOp, CombMemOp,
320  MemoryPortOp, MemoryPortAccessOp>(
321  [&](auto op) { emitStatement(op); })
322  .Default([&](auto op) {
323  indent() << "// operation " << op->getName() << "\n";
324  emitOpError(op, "not supported as statement");
325  });
326  }
327 }

329 void Emitter::emitStatement(WhenOp op, bool noIndent) {
330  (noIndent ? os : indent()) << "when ";
331  emitExpression(op.condition());
332  os << " :";
333  emitLocationAndNewLine(op);
334  addIndent();
335  emitStatementsInBlock(op.getThenBlock());
336  reduceIndent();
337  if (!op.hasElseRegion())
338  return;
339 
340  indent() << "else ";
341  // Sugar to `else when ...` if there's only a single when statement in the
342  // else block.
343  auto &elseBlock = op.getElseBlock();
344  if (!elseBlock.empty() && &elseBlock.front() == &elseBlock.back()) {
345  if (auto whenOp = dyn_cast<WhenOp>(&elseBlock.front())) {
346  emitStatement(whenOp, true);
347  return;
348  }
349  }
350  // Otherwise print the block as `else :`.
351  os << ":\n";
352  addIndent();
353  emitStatementsInBlock(elseBlock);
354  reduceIndent();
355 }

497 void Emitter::emitStatement(MemOp op) {
498  SmallString<16> portName(op.name());
499  portName.push_back('.');
500  auto portNameBaseLen = portName.size();
501  for (auto result : llvm::zip(op.getResults(), op.portNames())) {
502  portName.resize(portNameBaseLen);
503  portName.append(std::get<1>(result).cast<StringAttr>().getValue());
504  addValueName(std::get<0>(result), portName);
505  }
506 
507  indent() << "mem " << op.name() << " :";
508  emitLocationAndNewLine(op);
509  addIndent();
510 
511  indent() << "data-type => ";
512  emitType(op.getDataType());
513  os << "\n";
514  indent() << "depth => " << op.depth() << "\n";
515  indent() << "read-latency => " << op.readLatency() << "\n";
516  indent() << "write-latency => " << op.writeLatency() << "\n";
517 
518  SmallString<16> reader, writer, readwriter;
519  for (std::pair<StringAttr, MemOp::PortKind> port : op.getPorts()) {
520  auto add = [&](SmallString<16> &to, StringAttr name) {
521  if (!to.empty())
522  to.push_back(' ');
523  to.append(name.getValue());
524  };
525  switch (port.second) {
526  case MemOp::PortKind::Read:
527  add(reader, port.first);
528  break;
529  case MemOp::PortKind::Write:
530  add(writer, port.first);
531  break;
532  case MemOp::PortKind::ReadWrite:
533  add(readwriter, port.first);
534  break;
535  }
536  }
537  if (!reader.empty())
538  indent() << "reader => " << reader << "\n";
539  if (!writer.empty())
540  indent() << "writer => " << writer << "\n";
541  if (!readwriter.empty())
542  indent() << "readwriter => " << readwriter << "\n";
543 
544  indent() << "read-under-write => ";
545  emitAttribute(op.ruw());
546  os << "\n";
547 
548  reduceIndent();
549 }

614 void Emitter::emitExpression(Value value) {
615  // Handle the trivial case where we already have a name for this value which
616  // we can use.
617  if (auto name = lookupEmittedName(value)) {
618  os << name;
619  return;
620  }
621 
622  auto op = value.getDefiningOp();
623  assert(op && "value must either be a block arg or the result of an op");
624  TypeSwitch<Operation *>(op)
625  .Case<
626  // Basic expressions
627  ConstantOp, SpecialConstantOp, SubfieldOp, SubindexOp, SubaccessOp,
628  // Binary
629  AddPrimOp, SubPrimOp, MulPrimOp, DivPrimOp, RemPrimOp, AndPrimOp,
630  OrPrimOp, XorPrimOp, LEQPrimOp, LTPrimOp, GEQPrimOp, GTPrimOp,
631  EQPrimOp, NEQPrimOp, CatPrimOp, DShlPrimOp, DShlwPrimOp, DShrPrimOp,
632  // Unary

634  CvtPrimOp, NegPrimOp, NotPrimOp, AndRPrimOp, OrRPrimOp, XorRPrimOp,
635  // Miscellaneous
636  BitsPrimOp, HeadPrimOp, TailPrimOp, PadPrimOp, MuxPrimOp, ShlPrimOp,
637  ShrPrimOp>([&](auto op) { emitExpression(op); })
638  .Default([&](auto op) {

641  });

725 /// Emit a FIRRTL type into the output.
726 void Emitter::emitType(Type type) {
727  auto emitWidth = [&](Optional<int32_t> width) {
728  if (!width.hasValue())
729  return;
730  os << "<" << *width << ">";
731  };
732  TypeSwitch<Type>(type)
733  .Case<ClockType>([&](auto) { os << "Clock"; })
734  .Case<ResetType>([&](auto) { os << "Reset"; })
735  .Case<AsyncResetType>([&](auto) { os << "AsyncReset"; })
736  .Case<UIntType>([&](auto type) {
737  os << "UInt";
738  emitWidth(type.getWidth());
739  })
740  .Case<SIntType>([&](auto type) {
741  os << "SInt";
742  emitWidth(type.getWidth());
743  })
744  .Case<AnalogType>([&](auto type) {
745  os << "Analog";
746  emitWidth(type.getWidth());
747  })
748  .Case<BundleType>([&](auto type) {
749  os << "{";
750  bool anyEmitted = false;
751  for (auto &element : type.getElements()) {
752  os << (anyEmitted ? ", " : " ");
753  if (element.isFlip)
754  os << "flip ";
755  os << element.name.getValue() << " : ";
756  emitType(element.type);
757  anyEmitted = true;
758  }
759  if (anyEmitted)
760  os << " ";
761  os << "}";
762  })
763  .Case<FVectorType>([&](auto type) {
764  emitType(type.getElementType());
765  os << "[" << type.getNumElements() << "]";
766  })
767  .Case<CMemoryType>([&](auto type) {
768  emitType(type.getElementType());
769  os << "[" << type.getNumElements() << "]";
770  })
771  .Default([&](auto type) {
772  llvm_unreachable("all types should be implemented");
773  });
774 }
