377 /// Emit an entire circuit.
378 void Emitter::emitCircuit(CircuitOp op) {
379  circuitNamespace.add(op);
380  SymInfos circuitSymInfos(op);
381  symInfos = circuitSymInfos;
382  startStatement();
383  ps << "FIRRTL version ";
384  ps.addAsString(version.major);
385  ps << ".";
386  ps.addAsString(version.minor);
387  ps << ".";
388  ps.addAsString(version.patch);
389  ps << PP::newline;
390  ps << "circuit " << PPExtString(legalize(op.getNameAttr())) << " :";
391  setPendingNewline();
392  ps.scopedBox(PP::bbox2, [&]() {
393  for (auto &bodyOp : *op.getBodyBlock()) {
394  if (encounteredError)
395  break;
396  TypeSwitch<Operation *>(&bodyOp)
397  .Case<FModuleOp, FExtModuleOp, FIntModuleOp>([&](auto op) {
398  emitModule(op);
399  ps << PP::newline;
400  })
401  .Case<LayerOp>([&](auto op) { emitDeclaration(op); })
402  .Case<OptionOp>([&](auto op) { emitDeclaration(op); })
403  .Default([&](auto op) {
404  emitOpError(op, "not supported for emission inside circuit");
405  });
406  }
407  });
408  circuitNamespace.clear();
409  symInfos = std::nullopt;
410 }

609 void Emitter::emitStatement(WhenOp op) {
610  startStatement();
611  ps << "when ";
612  emitExpression(op.getCondition());
613  ps << " :";
614  emitLocationAndNewLine(op);
615  ps.scopedBox(PP::bbox2, [&]() { emitStatementsInBlock(op.getThenBlock()); });
616  // emitStatementsInBlock(op.getThenBlock());
617  if (!op.hasElseRegion())
618  return;
619 
620  startStatement();
621  ps << "else ";
622  // Sugar to `else when ...` if there's only a single when statement in the
623  // else block.
624  auto &elseBlock = op.getElseBlock();
625  if (!elseBlock.empty() && &elseBlock.front() == &elseBlock.back()) {
626  if (auto whenOp = dyn_cast<WhenOp>(&elseBlock.front())) {
627  emitStatement(whenOp);
628  return;
629  }
630  }
631  // Otherwise print the block as `else :`.
632  ps << ":";
633  setPendingNewline();
634  ps.scopedBox(PP::bbox2, [&]() { emitStatementsInBlock(elseBlock); });
635 }
