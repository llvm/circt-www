63 std::string FIRToken::getStringValue(StringRef spelling) {
64  // Start by dropping the quotes.
65  StringRef bytes = spelling.drop_front().drop_back();
66 
67  std::string result;
68  result.reserve(bytes.size());
69  for (size_t i = 0, e = bytes.size(); i != e;) {
70  auto c = bytes[i++];
71  if (c != '\\') {
72  result.push_back(c);
73  continue;
74  }
75 
76  assert(i + 1 <= e && "invalid string should be caught by lexer");
77  auto c1 = bytes[i++];
78  switch (c1) {
79  case '\\':
80  case '"':
81  case '\'':
82  result.push_back(c1);
83  continue;
84  case 'b':
85  result.push_back('\b');
86  continue;
87  case 'n':
88  result.push_back('\n');
89  continue;
90  case 't':
91  result.push_back('\t');
92  continue;
93  case 'f':
94  result.push_back('\f');
95  continue;
96  case 'r':
97  result.push_back('\r');
98  continue;
99  // TODO: Handle the rest of the escapes (octal and unicode).
100  default:
101  break;
102  }
103 
104  assert(i + 1 <= e && "invalid string should be caught by lexer");
105  auto c2 = bytes[i++];
106 
107  assert(llvm::isHexDigit(c1) && llvm::isHexDigit(c2) && "invalid escape");
108  result.push_back((llvm::hexDigitValue(c1) << 4) | llvm::hexDigitValue(c2));
109  }
110 
111  return result;
112 }

183 /// Return the indentation level of the specified token.
184 Optional<unsigned> FIRLexer::getIndentation(const FIRToken &tok) const {
185  // Count the number of horizontal whitespace characters before the token.
186  auto *bufStart = curBuffer.begin();
187 
188  auto isHorizontalWS = [](char c) -> bool {
189  return c == ' ' || c == '\t' || c == ',';
190  };
191  auto isVerticalWS = [](char c) -> bool {
192  return c == '\n' || c == '\r' || c == '\f' || c == '\v';
193  };
194 
195  unsigned indent = 0;
196  const auto *ptr = (const char *)tok.getSpelling().data();
197  while (ptr != bufStart && isHorizontalWS(ptr[-1]))
198  --ptr, ++indent;
199 
200  // If the character we stopped at isn't the start of line, then return none.
201  if (ptr != bufStart && !isVerticalWS(ptr[-1]))
202  return None;
203 
204  return indent;
205 }

211 FIRToken FIRLexer::lexTokenImpl() {
212  while (true) {
213  const char *tokStart = curPtr;
214  switch (*curPtr++) {
215  default:
216  // Handle identifiers.
217  if (llvm::isAlpha(curPtr[-1]))
218  return lexIdentifierOrKeyword(tokStart);
219 
220  // Unknown character, emit an error.
221  return emitError(tokStart, "unexpected character");
222 
223  case 0:
224  // This may either be a nul character in the source file or may be the EOF
225  // marker that llvm::MemoryBuffer guarantees will be there.
226  if (curPtr - 1 == curBuffer.end())
227  return formToken(FIRToken::eof, tokStart);
228 
229  LLVM_FALLTHROUGH; // Treat as whitespace.
230 
231  case ' ':
232  case '\t':
233  case '\n':
234  case '\r':
235  case ',':
236  // Handle whitespace.
237  continue;
238 
239  case '_':
240  // Handle identifiers.
241  return lexIdentifierOrKeyword(tokStart);
242 
243  case '.':
244  return formToken(FIRToken::period, tokStart);
245  case ':':
246  return formToken(FIRToken::colon, tokStart);
247  case '(':
248  return formToken(FIRToken::l_paren, tokStart);
249  case ')':
250  return formToken(FIRToken::r_paren, tokStart);
251  case '{':
252  return formToken(FIRToken::l_brace, tokStart);
253  case '}':
254  return formToken(FIRToken::r_brace, tokStart);
255  case '[':
256  return formToken(FIRToken::l_square, tokStart);
257  case ']':
258  return formToken(FIRToken::r_square, tokStart);
259  case '<':
260  if (*curPtr == '-')
261  return ++curPtr, formToken(FIRToken::less_minus, tokStart);
262  if (*curPtr == '=')
263  return ++curPtr, formToken(FIRToken::less_equal, tokStart);
264  return formToken(FIRToken::less, tokStart);
265  case '>':
266  return formToken(FIRToken::greater, tokStart);
267  case '=':
268  if (*curPtr == '>')
269  return ++curPtr, formToken(FIRToken::equal_greater, tokStart);
270  return formToken(FIRToken::equal, tokStart);
271  case '?':
272  return formToken(FIRToken::question, tokStart);
273  case '@':
274  if (*curPtr == '[')
275  return lexFileInfo(tokStart);
276  // Unknown character, emit an error.
277  return emitError(tokStart, "unexpected character");
278  case '%':
279  if (*curPtr == '[')
280  return lexInlineAnnotation(tokStart);
281  return emitError(tokStart, "unexpected character following '%'");
282 
283  case ';':
284  skipComment();
285  continue;
286 
287  case '"':
288  return lexString(tokStart, /*isRaw=*/false);
289  case '\'':
290  return lexString(tokStart, /*isRaw=*/true);
291 
292  case '+':
293  case '-':
294  case '0':
295  case '1':
296  case '2':
297  case '3':
298  case '4':
299  case '5':
300  case '6':
301  case '7':
302  case '8':
303  case '9':
304  return lexNumber(tokStart);
305  }
306  }
307 }

309 /// Lex a file info specifier.
310 ///
311 /// FileInfo ::= '@[' ('\]'|.)* ']'
312 ///
313 FIRToken FIRLexer::lexFileInfo(const char *tokStart) {
314  while (1) {
315  switch (*curPtr++) {
316  case ']': // This is the end of the fileinfo literal.
317  return formToken(FIRToken::fileinfo, tokStart);
318  case '\\':
319  // Ignore escaped ']'
320  if (*curPtr == ']')
321  ++curPtr;
322  break;
323  case 0:
324  // This could be the end of file in the middle of the fileinfo. If so
325  // emit an error.
326  if (curPtr - 1 != curBuffer.end())
327  break;
328  LLVM_FALLTHROUGH;
329  case '\n': // Vertical whitespace isn't allowed in a fileinfo.
330  case '\v':
331  case '\f':
332  return emitError(tokStart, "unterminated file info specifier");
333  default:
334  // Skip over other characters.
335  break;
336  }
337  }
338 }

340 /// Lex a non-standard inline Annotation file.
341 ///
342 /// InlineAnnotation ::= '%[' (.)* ']'
343 ///
344 FIRToken FIRLexer::lexInlineAnnotation(const char *tokStart) {
345  size_t depth = 0;
346  bool stringMode = false;
347  while (1) {
348  switch (*curPtr++) {
349  case '\\':
350  ++curPtr;
351  break;
352  case '"':
353  stringMode = !stringMode;
354  break;
355  case ']':
356  if (stringMode)
357  break;
358  if (depth == 1)
359  return formToken(FIRToken::inlineannotation, tokStart);
360  --depth;
361  break;
362  case '[':
363  if (stringMode)
364  break;
365  ++depth;
366  break;
367  case 0:
368  if (curPtr - 1 != curBuffer.end())
369  break;
370  return emitError(tokStart, "unterminated inline annotation");
371  default:
372  break;
373  }
374  }
375 }

377 /// Lex an identifier or keyword that starts with a letter.
378 ///
379 /// LegalStartChar ::= [a-zA-Z_]
380 /// LegalIdChar ::= LegalStartChar | [0-9] | '$'
381 //
382 /// Id ::= LegalStartChar (LegalIdChar)*
383 ///
384 FIRToken FIRLexer::lexIdentifierOrKeyword(const char *tokStart) {
385  // Match the rest of the identifier regex: [0-9a-zA-Z_$-]*
386  while (llvm::isAlpha(*curPtr) || llvm::isDigit(*curPtr) || *curPtr == '_' ||
387  *curPtr == '$' || *curPtr == '-')
388  ++curPtr;
389 
390  StringRef spelling(tokStart, curPtr - tokStart);
391 
392  // Check to see if this is a 'primop', which is an identifier juxtaposed with
393  // a '(' character.
394  if (*curPtr == '(') {
395  FIRToken::Kind kind = llvm::StringSwitch<FIRToken::Kind>(spelling)
396 #define TOK_LPKEYWORD(SPELLING) .Case(#SPELLING, FIRToken::lp_##SPELLING)
397 #include "FIRTokenKinds.def"
398  .Default(FIRToken::identifier);
399  if (kind != FIRToken::identifier) {
400  ++curPtr;
401  return formToken(kind, tokStart);
402  }
403  }
404 
405  // Check to see if this identifier is a keyword.
406  FIRToken::Kind kind = llvm::StringSwitch<FIRToken::Kind>(spelling)
407 #define TOK_KEYWORD(SPELLING) .Case(#SPELLING, FIRToken::kw_##SPELLING)
408 #include "FIRTokenKinds.def"
409  .Default(FIRToken::identifier);
410 
411  return FIRToken(kind, spelling);
412 }

414 /// Skip a comment line, starting with a ';' and going to end of line.
415 void FIRLexer::skipComment() {
416  while (true) {
417  switch (*curPtr++) {
418  case '\n':
419  case '\r':
420  // Newline is end of comment.
421  return;
422  case 0:
423  // If this is the end of the buffer, end the comment.
424  if (curPtr - 1 == curBuffer.end()) {
425  --curPtr;
426  return;
427  }
428  LLVM_FALLTHROUGH;
429  default:
430  // Skip over other characters.
431  break;
432  }
433  }

437 /// RawString ::= '\'' UnquotedString? '\''
438 /// UnquotedString ::= ( '\\\'' | '\\"' | ~[\r\n] )+?

440 FIRToken FIRLexer::lexString(const char *tokStart, bool isRaw) {
441  while (1) {
442  switch (*curPtr++) {
443  case '"': // This is the end of the string literal.
444  if (isRaw)
445  break;
446  return formToken(FIRToken::string, tokStart);
447  case '\'': // This is the end of the raw string.
448  if (!isRaw)
449  break;
450  return formToken(FIRToken::raw_string, tokStart);
451  case '\\':
452  // Ignore escaped '\'' or '"'
453  if (*curPtr == '\'' || *curPtr == '"')
454  ++curPtr;
455  break;
456  case 0:
457  // This could be the end of file in the middle of the string. If so
458  // emit an error.
459  if (curPtr - 1 != curBuffer.end())
460  break;
461  LLVM_FALLTHROUGH;
462  case '\n': // Vertical whitespace isn't allowed in a string.
463  case '\r':
464  case '\v':
465  case '\f':
466  return emitError(tokStart, "unterminated string");
467  default:
468  // Skip over other characters.
469  break;
470  }
471  }
472 }

474 /// Lex a number literal.
475 ///
476 /// UnsignedInt ::= '0' | PosInt
477 /// PosInt ::= [1-9] ([0-9])*
478 /// DoubleLit ::=
479 /// ( '+' | '-' )? Digit+ '.' Digit+ ( 'E' ( '+' | '-' )? Digit+ )?
480 ///
481 FIRToken FIRLexer::lexNumber(const char *tokStart) {
482  assert(llvm::isDigit(curPtr[-1]) || curPtr[-1] == '+' || curPtr[-1] == '-');
483 
484  // There needs to be at least one digit.
485  if (!llvm::isDigit(*curPtr) && !llvm::isDigit(curPtr[-1]))
486  return emitError(tokStart, "unexpected character after sign");
487 
488  while (llvm::isDigit(*curPtr))
489  ++curPtr;
490 
491  // If we encounter a '.' followed by a digit, then this is a floating point
492  // literal, otherwise this is an integer or negative integer.
493  if (*curPtr != '.' || !llvm::isDigit(curPtr[1])) {
494  if (*tokStart == '-' || *tokStart == '+')
495  return formToken(FIRToken::signed_integer, tokStart);
496  return formToken(FIRToken::integer, tokStart);
497  }
498 
499  // Lex a floating point literal.
500  curPtr += 2;
501  while (llvm::isDigit(*curPtr))
502  ++curPtr;
503 
504  if (*curPtr == 'E') {
505  ++curPtr;
506  if (*curPtr == '+' || *curPtr == '-')
507  ++curPtr;
508  while (llvm::isDigit(*curPtr))
509  ++curPtr;
510  }
511  return formToken(FIRToken::floatingpoint, tokStart);
512 }
