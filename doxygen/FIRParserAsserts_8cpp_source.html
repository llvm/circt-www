211 /// Chisel has a tendency to emit complex assert/assume/cover statements encoded
212 /// as print operations with special formatting and metadata embedded in the
213 /// message literal. These always reside in a when block of the following form:
214 ///
215 /// when invCond:
216 /// printf(clock, UInt<1>(1), "...[verif-library-assert]...")
217 /// stop(clock, UInt<1>(1), 1)
218 ///
219 /// Depending on the nature the verification operation, the `stop` may be
220 /// optional. The Scala implementation simply removes all `stop`s that have the
221 /// same condition as the printf.
222 ParseResult circt::firrtl::foldWhenEncodedVerifOp(PrintFOp printOp) {
223  auto *context = printOp.getContext();
224  auto whenStmt = dyn_cast<WhenOp>(printOp->getParentOp());
225 
226  // If the parent of printOp is not when, printOp doesn't encode a
227  // verification.
228  if (!whenStmt)
229  return success();
230 
231  // The when blocks we're interested in don't have an else region.
232  if (whenStmt.hasElseRegion())
233  return success();
234 
235  // The when blocks we're interested in contain a `PrintFOp` and an optional
236  // `StopOp` with the same clock and condition as the print.
237  Block &thenBlock = whenStmt.getThenBlock();
238  auto opIt = std::next(printOp->getIterator());
239  auto opEnd = thenBlock.end();
240 
241  // optional `stop(clock, enable, ...)`
242  //
243  // FIXME: Currently, we can't detetct stopOp in the following IR:
244  // when invCond:
245  // printf(io.clock, UInt<1>(1), "assert: ..")
246  // stop(io.clock, UInt<1>(1), 1)
247  // It is because `io.clock` will create another subfield op so StopOp is not
248  // the next operation. Also, we will have to modify `stopOp.clock() !=
249  // printOp.clock()` below since they are not CSEd.
250  if (opIt != opEnd) {
251  auto stopOp = dyn_cast<StopOp>(*opIt++);
252  if (!stopOp || opIt != opEnd || stopOp.clock() != printOp.clock() ||
253  stopOp.cond() != printOp.cond())
254  return success();
255  stopOp.erase();
256  }
257 
258  // Detect if we're dealing with a verification statement, and what flavor of
259  // statement it is.
260  auto fmt = printOp.formatString();
261  VerifFlavor flavor;
262  if (fmt.contains("[verif-library-assert]"))
263  flavor = VerifFlavor::VerifLibAssert;
264  else if (fmt.contains("[verif-library-assume]"))
265  flavor = VerifFlavor::VerifLibAssume;
266  else if (fmt.consume_front("assert:"))
267  flavor = VerifFlavor::Assert;
268  else if (fmt.consume_front("assume:"))
269  flavor = VerifFlavor::Assume;
270  else if (fmt.consume_front("cover:"))
271  flavor = VerifFlavor::Cover;
272  else if (fmt.consume_front("assertNotX:"))
273  flavor = VerifFlavor::AssertNotX;
274  else if (fmt.startswith("Assertion failed"))
275  flavor = VerifFlavor::ChiselAssert;
276  else
277  return success();
278 
279  // Check if the condition of the `WhenOp` is a trivial inversion operation,
280  // and remove any immediately preceding verification ops that ensure this
281  // condition. This caters to the following pattern emitted by Chisel:
282  //
283  // assert(clock, cond, enable, ...)
284  // node N = eq(cond, UInt<1>(0))
285  // when N:
286  // printf(clock, enable, ...)
287  Value flippedCond = whenStmt.condition();
288  if (auto node = flippedCond.getDefiningOp<NodeOp>())
289  flippedCond = node.input();
290  if (auto notOp = flippedCond.getDefiningOp<NotPrimOp>()) {
291  flippedCond = notOp.input();
292  } else if (auto eqOp = flippedCond.getDefiningOp<EQPrimOp>()) {
293  auto isConst0 = [](Value v) {
294  if (auto constOp = v.getDefiningOp<ConstantOp>())
295  return constOp.value().isZero();
296  return false;
297  };
298  if (isConst0(eqOp.lhs()))
299  flippedCond = eqOp.rhs();
300  else if (isConst0(eqOp.rhs()))
301  flippedCond = eqOp.lhs();
302  else
303  flippedCond = {};
304  } else {
305  flippedCond = {};
306  }
307 
308  // If we have found such a condition, erase any verification ops that use it
309  // and that match the op we are about to assemble. This is necessary since the
310  // `printf` op actually carries all the information we need for the assert,
311  // while the actual `assert` has none of it. This makes me sad.
312  if (flippedCond) {
313  // Use a set to catch cases where a verification op is a double user of the
314  // flipped condition.
315  SmallPtrSet<Operation *, 1> opsToErase;
316  for (const auto &user : flippedCond.getUsers()) {
317  TypeSwitch<Operation *>(user).Case<AssertOp, AssumeOp, CoverOp>(
318  [&](auto op) {
319  if (op.clock() == printOp.clock() &&
320  op.enable() == printOp.cond() &&
321  op.predicate() == flippedCond && !op.isConcurrent())
322  opsToErase.insert(op);
323  });
324  }
325  for (auto op : opsToErase)
326  op->erase();
327  }
328 
329  // A recursive function to move all the dependency in `printOp` operands.
330  std::function<void(Operation *)> moveOperationsToParent = [&](Operation *op) {
331  // If operation is not defined in when, it's ok.
332  if (!op || op->getBlock() != &whenStmt.getThenBlock())
333  return;
334 
335  llvm::for_each(op->getOperands(), [&](Value value) {
336  moveOperationsToParent(value.getDefiningOp());
337  });
338 
339  // `op` might be already moved to parent op by the previous recursive calls,
340  // so check again.
341  if (op->getBlock() != &whenStmt.getThenBlock())
342  return;
343 
344  op->moveBefore(whenStmt);
345  };
346 
347  // Move all operands in printOp to the parent block.
348  llvm::for_each(printOp->getOperands(), [&](Value value) {
349  moveOperationsToParent(value.getDefiningOp());
350  });
351 
352  ImplicitLocOpBuilder builder(whenStmt.getLoc(), whenStmt);
353  builder.setInsertionPointAfter(whenStmt);
354  // CAREFUL: Since the assertions are encoded as "when something wrong, then
355  // print" an error message, we're actually asserting that something is *not*
356  // going wrong.
357  //
358  // In code:
359  //
360  // when somethingGoingWrong:
361  // printf("oops")
362  //
363  // Actually expresses:
364  //
365  // assert(not(somethingGoingWrong), "oops")
366  //
367  // So you'll find quite a few inversions of the when condition below to
368  // represent this.
369 
370  // TODO: None of the following ops preserve interpolated operands in the
371  // format string. SV allows this, and we might want to extend the
372  // `firrtl.{assert,assume,cover}` operations to deal with this.
373 
374  switch (flavor) {
375  // Handle the case of property verification encoded as `<assert>:<msg>` or
376  // `<assert>:<label>:<msg>`.
377  case VerifFlavor::Assert:
378  case VerifFlavor::Assume:
379  case VerifFlavor::Cover:
380  case VerifFlavor::AssertNotX: {
381  // Extract label and message from the format string.
382  StringRef label;
383  StringRef message = fmt;
384  auto index = fmt.find(':');
385  if (index != StringRef::npos) {
386  label = fmt.slice(0, index);
387  message = fmt.slice(index + 1, StringRef::npos);
388  }
389 
390  // AssertNotX has the special format `assertNotX:%d:msg`, where the `%d`
391  // would theoretically interpolate the value being check for X, but in
392  // practice the Scala impl of ExtractTestCode just discards that `%d` label
393  // and replaces it with `notX`. Also prepare the condition to be checked
394  // here.
395  Value predicate = whenStmt.condition();
396  if (flavor != VerifFlavor::Cover)
397  predicate = builder.create<NotPrimOp>(predicate);
398  if (flavor == VerifFlavor::AssertNotX) {
399  label = "notX";
400  if (printOp.operands().size() != 1) {
401  printOp.emitError("printf-encoded assertNotX requires one operand");
402  return failure();
403  }
404  // Construct a `!whenCond | (value !== 1'bx)` predicate.
405  Value notCond = predicate;
406  predicate = builder.create<XorRPrimOp>(printOp.operands()[0]);
407  predicate = builder.create<VerbatimExprOp>(UIntType::get(context, 1),
408  "{{0}} !== 1'bx", predicate);
409  predicate = builder.create<OrPrimOp>(notCond, predicate);
410  }
411 
412  // CAVEAT: The Scala impl of ExtractTestCode explicitly sets `emitSVA` to
413  // false for these flavors of verification operations. I think it's a bad
414  // idea to decide at parse time if something becomes an SVA or a print+fatal
415  // process, so I'm not doing this here. If we ever come across this need, we
416  // may want to annotate the operation with an attribute to indicate that it
417  // wants to explicitly *not* be an SVA.
418 
419  // TODO: Sanitize the label by replacing whitespace with "_" as done in the
420  // Scala impl of ExtractTestCode.
421  ValueRange args;
422  if (printOp.operands().size())
423  args = printOp.operands().drop_front();
424  if (args.size())
425  printOp.emitWarning()
426  << "printf-encoded assertion has format string arguments which may "
427  "cause lint warnings";
428  if (flavor == VerifFlavor::Assert || flavor == VerifFlavor::AssertNotX)
429  builder.create<AssertOp>(printOp.clock(), predicate, printOp.cond(),
430  message, args, label, true);
431  else if (flavor == VerifFlavor::Assume)
432  builder.create<AssumeOp>(printOp.clock(), predicate, printOp.cond(),
433  message, args, label, true);
434  else // VerifFlavor::Cover
435  builder.create<CoverOp>(printOp.clock(), predicate, printOp.cond(),
436  message, args, label, true);
437  printOp.erase();
438  break;
439  }
440 
441  // Handle the case of builtin Chisel assertions.
442  case VerifFlavor::ChiselAssert: {
443  auto op = builder.create<AssertOp>(
444  printOp.clock(), builder.create<NotPrimOp>(whenStmt.condition()),
445  printOp.cond(), fmt, printOp.operands(), "chisel3_builtin", true);
446  op->setAttr("format", StringAttr::get(context, "ifElseFatal"));
447  printOp.erase();
448  break;
449  }
450 
451  // Handle the SiFive verification library asserts/assumes, which contain
452  // additional configuration attributes for the verification op, serialized
453  // to JSON and embedded in the print message within `<extraction-summary>`
454  // XML tags.
455  case VerifFlavor::VerifLibAssert:
456  case VerifFlavor::VerifLibAssume: {
457  // Isolate the JSON text in the `<extraction-summary>` XML tag.
458  StringRef prefix, exStr, suffix;
459  std::tie(prefix, exStr) = fmt.split("<extraction-summary>");
460  std::tie(exStr, suffix) = exStr.split("</extraction-summary>");
461 
462  // The extraction summary is necessary for this kind of assertion, so we
463  // throw an error if it is missing.
464  if (exStr.empty()) {
465  auto diag = printOp.emitError(
466  "printf-encoded assert/assume requires extraction summary");
467  diag.attachNote(printOp.getLoc())
468  << "because printf message contains "
469  "`[verif-library-{assert,assume}]` tag";
470  diag.attachNote(printOp.getLoc())
471  << "expected JSON-encoded extraction summary in "
472  "`<extraction-summary>` XML tag";
473  return failure();
474  }
475 
476  // Parse the extraction summary, which contains additional parameters for
477  // the assertion.
478  auto ex = json::parse(exStr);
479  if (auto err = ex.takeError()) {
480  handleAllErrors(std::move(err), [&](const json::ParseError &a) {
481  printOp.emitError("failed to parse JSON extraction summary")
482  .attachNote()
483  << a.message();
484  });
485  return failure();
486  }
487 
488  // The extraction summary must be an object.
489  auto exObj = ex->getAsObject();
490  if (!exObj) {
491  printOp.emitError("extraction summary must be a JSON object");
492  return failure();
493  }
494 
495  // Extract and apply any predicate modifier.
496  PredicateModifier predMod;
497  if (parseJson(printOp.getLoc(), exObj, [&](const auto &ex) {
498  return ex.withObjectField("predicateModifier", [&](const auto &ex) {
499  return ex.withStringField("type", [&](const auto &ex) {
500  if (auto pm = parsePredicateModifier(ex)) {
501  predMod = *pm;
502  return success();
503  }
504  return failure();
505  });
506  });
507  }))
508  return failure();
509 
510  Value predicate = whenStmt.condition();
511  predicate = builder.create<NotPrimOp>(
512  predicate); // assertion triggers when predicate fails
513  switch (predMod) {
514  case PredicateModifier::NoMod:
515  // Leave the predicate unmodified.
516  break;
517  case PredicateModifier::TrueOrIsX:
518  // Construct a `predicate | (^predicate === 1'bx)`.
519  Value orX = builder.create<XorRPrimOp>(predicate);
520  orX = builder.create<VerbatimExprOp>(UIntType::get(context, 1),
521  "{{0}} === 1'bx", orX);
522  predicate = builder.create<OrPrimOp>(predicate, orX);
523  break;
524  }
525 
526  // Extract the preprocessor macro names that should guard this assertion.
527  SmallVector<Attribute> guards;
528  if (parseJson(printOp.getLoc(), exObj, [&](const auto &ex) {
529  return ex.withArrayField(
530  "conditionalCompileToggles", [&](const auto &ex) {
531  return ex.withObject([&](const auto &ex) {
532  return ex.withStringField("type", [&](const auto &ex) {
533  if (auto guard = parseConditionalCompileToggle(ex)) {
534  guards.push_back(
535  StringAttr::get(builder.getContext(), *guard));
536  return success();
537  }
538  return failure();
539  });
540  });
541  });
542  }))
543  return failure();
544 
545  // Extract label additions and the message.
546  SmallString<32> label("verif_library");
547  if (parseJson(printOp.getLoc(), exObj, [&](const auto &ex) {
548  return ex.withArrayField("labelExts", [&](const auto &ex) {
549  return ex.withString([&](const auto &ex) {
550  label.push_back('_');
551  label.append(ex.value);
552  return success();
553  });
554  });
555  }))
556  return failure();
557 
558  StringRef message = fmt;
559  if (parseJson(printOp.getLoc(), exObj, [&](const auto &ex) {
560  return ex.withStringField("baseMsg", [&](const auto &ex) {
561  message = ex.value;
562  return success();
563  });
564  }))
565  return failure();
566 
567  // Assertions carry an additional `format` field.
568  Optional<StringRef> format;
569  if (flavor == VerifFlavor::VerifLibAssert) {
570  if (parseJson(printOp.getLoc(), exObj, [&](const auto &ex) {
571  return ex.withObjectField("format", [&](const auto &ex) {
572  return ex.withStringField("type", [&](const auto &ex) {
573  if (auto f = parseAssertionFormat(ex)) {
574  format = *f;
575  return success();
576  }
577  return failure();
578  });
579  });
580  }))
581  return failure();
582  }
583 
584  // Build the verification op itself.
585  Operation *op;
586  // TODO: The "ifElseFatal" variant isn't actually a concurrent assertion,
587  // but downstream logic assumes that isConcurrent is set.
588  if (flavor == VerifFlavor::VerifLibAssert)
589  op = builder.create<AssertOp>(printOp.clock(), predicate, printOp.cond(),
590  message, printOp.operands(), label, true);
591  else // VerifFlavor::VerifLibAssume
592  op = builder.create<AssumeOp>(printOp.clock(), predicate, printOp.cond(),
593  message, printOp.operands(), label, true);
594  printOp.erase();
595 
596  // Attach additional attributes extracted from the JSON object.
597  op->setAttr("guards", ArrayAttr::get(context, guards));
598  if (format)
599  op->setAttr("format", StringAttr::get(context, *format));
600 
601  break;
602  }
603  }
604 
605  // Clean up the `WhenOp` if it has become completely empty.
606  if (thenBlock.empty())
607  whenStmt.erase();
608  return success();
609 }
