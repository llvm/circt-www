390 /// Parse an @info marker if present. If so, fill in the specified Location,
391 /// if not, ignore it.
392 ParseResult FIRParser::parseOptionalInfoLocator(LocationAttr &result) {
393  if (getToken().isNot(FIRToken::fileinfo))
394  return success();
395 
396  auto loc = getToken().getLoc();
397 
398  auto spelling = getTokenSpelling();
399  consumeToken(FIRToken::fileinfo);
400 
401  auto locationPair = maybeStringToLocation(
402  spelling, constants.options.ignoreInfoLocators, locatorFilenameCache,
403  fileLineColLocCache, getContext());
404 
405  // If parsing failed, then indicate that a weird info was found.
406  if (!locationPair.first) {
407  mlir::emitWarning(translateLocation(loc),
408  "ignoring unknown @ info record format");
409  return success();
410  }
411 
412  // If the parsing succeeded, but we are supposed to drop locators, then just
413  // return.
414  if (locationPair.first && constants.options.ignoreInfoLocators)
415  return success();
416 
417  // Otherwise, set the location attribute and return.
418  result = locationPair.second.getValue();
419  return success();
420 }

466 /// intLit ::= UnsignedInt
467 /// ::= SignedInt
468 /// ::= HexLit
469 /// ::= OctalLit
470 /// ::= BinaryLit
471 /// HexLit ::= '"' 'h' ( '+' | '-' )? ( HexDigit )+ '"'
472 /// OctalLit ::= '"' 'o' ( '+' | '-' )? ( OctalDigit )+ '"'
473 /// BinaryLit ::= '"' 'b' ( '+' | '-' )? ( BinaryDigit )+ '"'
474 ///
475 ParseResult FIRParser::parseIntLit(APInt &result, const Twine &message) {
476  auto spelling = getTokenSpelling();
477  bool isNegative = false;
478  switch (getToken().getKind()) {
479  case FIRToken::signed_integer:
480  isNegative = spelling[0] == '-';
481  assert(spelling[0] == '+' || spelling[0] == '-');
482  spelling = spelling.drop_front();
483  LLVM_FALLTHROUGH;
484  case FIRToken::integer:
485  if (spelling.getAsInteger(10, result))
486  return emitError(message), failure();
487 
488  // Make sure that the returned APInt has a zero at the top so clients don't
489  // confuse it with a negative number.
490  if (result.isNegative())
491  result = result.zext(result.getBitWidth() + 1);
492 
493  if (isNegative)
494  result = -result;
495 
496  // If this was parsed as >32 bits, but can be represented in 32 bits,
497  // truncate off the extra width. This is important for extmodules which
498  // like parameters to be 32-bits, and insulates us from some arbitraryness
499  // in StringRef::getAsInteger.
500  if (result.getBitWidth() > 32 && result.getMinSignedBits() <= 32)
501  result = result.trunc(32);
502 
503  consumeToken();
504  return success();
505  case FIRToken::string: {
506  // Drop the quotes.
507  assert(spelling.front() == '"' && spelling.back() == '"');
508  spelling = spelling.drop_back().drop_front();
509 
510  // Decode the base.
511  unsigned base;
512  switch (spelling.empty() ? ' ' : spelling.front()) {
513  case 'h':
514  base = 16;
515  break;
516  case 'o':
517  base = 8;
518  break;
519  case 'b':
520  base = 2;
521  break;
522  default:
523  return emitError("expected base specifier (h/o/b) in integer literal"),
524  failure();
525  }
526  spelling = spelling.drop_front();
527 
528  // Handle the optional sign.
529  bool isNegative = false;
530  if (!spelling.empty() && spelling.front() == '+')
531  spelling = spelling.drop_front();
532  else if (!spelling.empty() && spelling.front() == '-') {
533  isNegative = true;
534  spelling = spelling.drop_front();
535  }
536 
537  // Parse the digits.
538  if (spelling.empty())
539  return emitError("expected digits in integer literal"), failure();
540 
541  if (spelling.getAsInteger(base, result))
542  return emitError("invalid character in integer literal"), failure();
543 
544  // We just parsed the positive version of this number. Make sure it has
545  // a zero at the top so clients don't confuse it with a negative number and
546  // so the negation (in the case of a negative sign) doesn't overflow.
547  if (result.isNegative())
548  result = result.zext(result.getBitWidth() + 1);
549 
550  if (isNegative)
551  result = -result;
552 
553  consumeToken(FIRToken::string);
554  return success();
555  }
556 
557  default:
558  return emitError("expected integer literal"), failure();
559  }

699 
700  if (kind == FIRToken::kw_SInt)
701  result = SIntType::get(getContext(), width);
702  else if (kind == FIRToken::kw_UInt)
703  result = UIntType::get(getContext(), width);
704  else {
705  assert(kind == FIRToken::kw_Analog);
706  result = AnalogType::get(getContext(), width);
707  }
708  break;
709  }
710 
711  case FIRToken::l_brace: {
712  consumeToken(FIRToken::l_brace);
713 
714  SmallVector<BundleType::BundleElement, 4> elements;
715  if (parseListUntil(FIRToken::r_brace, [&]() -> ParseResult {
716  bool isFlipped = consumeIf(FIRToken::kw_flip);
717 
718  StringRef fieldName;
719  FIRRTLType type;
720  if (parseFieldId(fieldName, "expected bundle field name") ||
721  parseToken(FIRToken::colon, "expected ':' in bundle") ||
722  parseType(type, "expected bundle field type"))
723  return failure();
724 
725  elements.push_back(
726  {StringAttr::get(getContext(), fieldName), isFlipped, type});
727  return success();
728  }))
729  return failure();
730  result = BundleType::get(elements, getContext());
731  break;
732  }
733  }
734 
735  // Handle postfix vector sizes.
736  while (consumeIf(FIRToken::l_square)) {
737  auto sizeLoc = getToken().getLoc();
738  int64_t size;
739  if (parseIntLit(size, "expected width") ||
740  parseToken(FIRToken::r_square, "expected ]"))
741  return failure();
742 
743  if (size < 0)
744  return emitError(sizeLoc, "invalid size specifier"), failure();
745 
746  result = FVectorType::get(result, size);
747  }
748 
749  return success();

775 /// Convert the input "tokens" to a range of field IDs. Considering a FIRRTL
776 /// aggregate type, "{foo: UInt<1>, bar: UInt<1>}[2]", tokens "[0].foo" will be
777 /// converted to a field ID range of [2, 2]; tokens "[1]" will be converted to
778 /// [4, 6]. The generated field ID range will then be attached to the annotation
779 /// in a "circt.fieldID" field to indicate the applicable fields of an
780 /// annotation.
781 Optional<unsigned> FIRParser::getFieldIDFromTokens(ArrayAttr tokens, SMLoc loc,
782  Type type) {
783  if (!type)
784  return None;
785  if (tokens.empty())
786  return 0;
787 
788  auto currentType = type.cast<FIRRTLType>();
789  unsigned id = 0;
790 
791  auto getMessage = [&](unsigned tokenIdx) {
792  // Construct a string for error emission.
793  SmallString<16> message("the " + std::to_string(tokenIdx) +
794  "-th token of ");
795  for (auto token : tokens) {
796  if (auto intAttr = token.dyn_cast<IntegerAttr>()) {
797  message.push_back('[');
798  intAttr.getValue().toStringSigned(message);
799  message.push_back(']');
800  } else if (auto strAttr = token.dyn_cast<StringAttr>()) {
801  message.push_back('.');
802  message.append(strAttr.getValue());
803  }
804  }
805  return message;
806  };
807 
808  unsigned tokenIdx = 0;
809  for (auto token : tokens) {
810  ++tokenIdx;
811  if (auto bundleType = currentType.dyn_cast<BundleType>()) {
812  auto subFieldAttr = token.dyn_cast<StringAttr>();
813  if (!subFieldAttr) {
814  emitError(loc, "expect a string for " + getMessage(tokenIdx));
815  return None;
816  }
817 
818  // Token should point to valid sub-field.
819  auto index = bundleType.getElementIndex(subFieldAttr);
820  if (!index) {
821  emitError(loc, getMessage(tokenIdx) + " is not found in the bundle");
822  return None;
823  }
824 
825  id += bundleType.getFieldID(index.getValue());
826  currentType = bundleType.getElementType(index.getValue());
827  continue;
828  }
829 
830  if (auto vectorType = currentType.dyn_cast<FVectorType>()) {
831  auto subIndexAttr = token.dyn_cast<IntegerAttr>();
832  if (!subIndexAttr) {
833  emitError(loc, "expect an integer for " + getMessage(tokenIdx));
834  return None;
835  }
836 
837  // Token should be a valid index of the vector.
838  auto subIndex = subIndexAttr.getValue().getZExtValue();
839  if (subIndex >= vectorType.getNumElements()) {
840  emitError(loc, getMessage(tokenIdx) + " is out of range in the vector");
841  return None;
842  }
843 
844  id += vectorType.getFieldID(subIndex);
845  currentType = vectorType.getElementType();
846  continue;
847  }
848 
849  emitError(loc, getMessage(tokenIdx) + " expects an aggregate type");
850  return None;
851  }
852 
853  return id;
854 }

856 /// In the input "annotations", an annotation will have a "target" entry when it
857 /// is only applicable to part of what it is attached to. In this method, we
858 /// convert the tokens contained by the "target" entry to a range of field IDs
859 /// and return the converted annotations.
860 ArrayAttr FIRParser::convertSubAnnotations(ArrayRef<Attribute> annotations,
861  SMLoc loc, Type type = nullptr) {
862  // This stores the annotations applicable to all fields of the op result.
863  SmallVector<Attribute, 8> annotationVec;
864 
865  for (auto a : annotations) {
866  // If the annotation does not contain a "target" entry, it is applicable to
867  // all fields of the op result.
868  auto dict = a.cast<DictionaryAttr>();
869  auto targetAttr = dict.get("target");
870  if (!targetAttr) {
871  annotationVec.push_back(a);
872  continue;
873  }
874 
875  // Otherwise, the annotation is only applicable to part of the op result.
876  // Get a range of field IDs to indicate the applicable fields of the
877  // annotation.
878  auto fieldID =
879  getFieldIDFromTokens(targetAttr.cast<ArrayAttr>(), loc, type);
880  if (!fieldID)
881  continue;
882 
883  // Remove the "target" entry from the annotation.
884  NamedAttrList modAttr;
885  for (auto attr : dict.getValue()) {
886  // Ignore the actual target annotation, but copy the rest of annotations.
887  if (attr.getName() == "target")
888  continue;
889  modAttr.push_back(attr);
890  }
891 
892  // Add a "circt.fieldID" field with the fieldID.
893  modAttr.append("circt.fieldID",
894  IntegerAttr::get(IntegerType::get(constants.context, 64,
895  IntegerType::Signless),
896  fieldID.getValue()));
897 
898  annotationVec.push_back(DictionaryAttr::get(constants.context, modAttr));
899  }
900 
901  return ArrayAttr::get(constants.context, annotationVec);
902 }

904 /// Split the input "annotations" into two parts: (1) Annotations applicable to
905 /// all op results; (2) Annotations only applicable to one specific result. The
906 /// second kind of annotations are store densely in an ArrayAttr and can be
907 /// accessed using the result index.
908 std::pair<ArrayAttr, ArrayAttr>
909 FIRParser::splitAnnotations(ArrayRef<Attribute> annotations, SMLoc loc,
910  ArrayRef<std::pair<StringAttr, Type>> ports) {
911  // This stores the annotations applicable to all op ports.
912  SmallVector<Attribute, 8> annotationVec;
913  // This stores the mapping between port names and port-specific annotations.
914  llvm::StringMap<SmallVector<Attribute, 4>> portAnnotationMap;
915 
916  for (auto a : annotations) {
917  // If the annotation does not contain a "target" entry, it is applicable to
918  // all op ports.
919  auto dict = a.cast<DictionaryAttr>();
920  auto targetAttr = dict.get("target");
921  if (!targetAttr) {
922  annotationVec.push_back(a);
923  continue;
924  }
925 
926  // The first token should always indicate the port name.
927  auto targetTokens = targetAttr.cast<ArrayAttr>().getValue();
928  auto portName = targetTokens[0].dyn_cast<StringAttr>();
929  if (!portName) {
930  emitError(loc, "the first token is invalid");
931  continue;
932  }
933 
934  // If there's more than one tokens, the remaining tokens should be added
935  // into the annotation as a "target" field.
936  NamedAttrList modAttr;
937  for (auto attr : dict.getValue()) {
938  if (attr.getName().str() == "target") {
939  if (targetTokens.size() > 1) {
940  auto newTargetAttr =
941  ArrayAttr::get(constants.context, targetTokens.drop_front());
942  modAttr.push_back(NamedAttribute(attr.getName(), newTargetAttr));
943  }
944  continue;
945  }
946  modAttr.push_back(attr);
947  }
948 
949  portAnnotationMap[portName.getValue()].push_back(
950  DictionaryAttr::get(constants.context, modAttr));
951  }
952 
953  SmallVector<Attribute, 16> portAnnotationVec;
954  for (auto port : ports) {
955  auto portAnnotation = portAnnotationMap.lookup(port.first.getValue());
956  portAnnotationVec.push_back(
957  convertSubAnnotations(portAnnotation, loc, port.second));
958  }
959 
960  return {ArrayAttr::get(constants.context, annotationVec),
961  ArrayAttr::get(constants.context, portAnnotationVec)};
962 }

1512 /// Attach invalid values to every element of the value.
1513 void FIRStmtParser::emitInvalidate(Value val, Flow flow) {
1514  auto tpe = val.getType().cast<FIRRTLType>();
1515 
1516  auto props = tpe.getRecursiveTypeProperties();
1517  if (props.isPassive && !props.containsAnalog) {
1518  if (flow == Flow::Source)
1519  return;
1520  if (props.hasUninferredWidth)
1521  builder.create<ConnectOp>(val, builder.create<InvalidValueOp>(tpe));
1522  else
1523  builder.create<StrictConnectOp>(val, builder.create<InvalidValueOp>(tpe));
1524  return;
1525  }
1526 
1527  // Recurse until we hit passive leaves. Connect any leaves which have sink or
1528  // duplex flow.
1529  //
1530  // TODO: This is very similar to connect expansion in the LowerTypes pass
1531  // works. Find a way to unify this with methods common to LowerTypes or to
1532  // have LowerTypes to the actual work here, e.g., emitting a partial connect
1533  // to only the leaf sources.
1534  TypeSwitch<FIRRTLType>(tpe)
1535  .Case<BundleType>([&](auto tpe) {
1536  for (size_t i = 0, e = tpe.getNumElements(); i < e; ++i) {
1537  auto &subfield = moduleContext.getCachedSubaccess(val, i);
1538  if (!subfield) {
1539  OpBuilder::InsertionGuard guard(builder);
1540  builder.setInsertionPointAfterValue(val);
1541  subfield = builder.create<SubfieldOp>(val, i);
1542  }
1543  emitInvalidate(subfield,
1544  tpe.getElement(i).isFlip ? swapFlow(flow) : flow);
1545  }
1546  })
1547  .Case<FVectorType>([&](auto tpe) {
1548  auto tpex = tpe.getElementType();
1549  for (size_t i = 0, e = tpe.getNumElements(); i != e; ++i) {
1550  auto &subindex = moduleContext.getCachedSubaccess(val, i);
1551  if (!subindex) {
1552  OpBuilder::InsertionGuard guard(builder);
1553  builder.setInsertionPointAfterValue(val);
1554  subindex = builder.create<SubindexOp>(tpex, val, i);
1555  }
1556  emitInvalidate(subindex, flow);
1557  }
1558  });
1559 }

1561 void FIRStmtParser::emitPartialConnect(ImplicitLocOpBuilder &builder, Value dst,
1562  Value src) {
1563  auto dstType = dst.getType().cast<FIRRTLType>();
1564  auto srcType = src.getType().cast<FIRRTLType>();
1565  if (dstType.isa<AnalogType>()) {
1566  builder.create<AttachOp>(SmallVector{dst, src});
1567  } else if (dstType == srcType && !dstType.containsAnalog()) {
1568  emitConnect(builder, dst, src);
1569  } else if (auto dstBundle = dstType.dyn_cast<BundleType>()) {
1570  auto srcBundle = srcType.cast<BundleType>();
1571  auto numElements = dstBundle.getNumElements();
1572  for (size_t dstIndex = 0; dstIndex < numElements; ++dstIndex) {
1573  // Find a matching field by name in the other bundle.
1574  auto &dstElement = dstBundle.getElements()[dstIndex];
1575  auto name = dstElement.name;
1576  auto maybe = srcBundle.getElementIndex(name);
1577  // If there was no matching field name, don't connect this one.
1578  if (!maybe)
1579  continue;
1580  auto dstRef = moduleContext.getCachedSubaccess(dst, dstIndex);
1581  if (!dstRef) {
1582  OpBuilder::InsertionGuard guard(builder);
1583  builder.setInsertionPointAfterValue(dst);
1584  dstRef = builder.create<SubfieldOp>(dst, dstIndex);
1585  }
1586  // We are pulling two fields from the cache. If the dstField was a
1587  // pointer into the cache, then the lookup for srcField might invalidate
1588  // it. So, we just copy dstField into a local.
1589  auto dstField = dstRef;
1590  auto srcIndex = *maybe;
1591  auto &srcField = moduleContext.getCachedSubaccess(src, srcIndex);
1592  if (!srcField) {
1593  OpBuilder::InsertionGuard guard(builder);
1594  builder.setInsertionPointAfterValue(src);
1595  srcField = builder.create<SubfieldOp>(src, srcIndex);
1596  }
1597  if (!dstElement.isFlip)
1598  emitPartialConnect(builder, dstField, srcField);
1599  else
1600  emitPartialConnect(builder, srcField, dstField);
1601  }
1602  } else if (auto dstVector = dstType.dyn_cast<FVectorType>()) {
1603  auto srcVector = srcType.cast<FVectorType>();
1604  auto dstNumElements = dstVector.getNumElements();
1605  auto srcNumEelemnts = srcVector.getNumElements();
1606  // Partial connect will connect all elements up to the end of the array.
1607  auto numElements = std::min(dstNumElements, srcNumEelemnts);
1608  for (size_t i = 0; i != numElements; ++i) {
1609  auto &dstRef = moduleContext.getCachedSubaccess(dst, i);
1610  if (!dstRef) {
1611  OpBuilder::InsertionGuard guard(builder);
1612  builder.setInsertionPointAfterValue(dst);
1613  dstRef = builder.create<SubindexOp>(dst, i);
1614  }
1615  auto dstField = dstRef; // copy to ensure not invalidated
1616  auto &srcField = moduleContext.getCachedSubaccess(src, i);
1617  if (!srcField) {
1618  OpBuilder::InsertionGuard guard(builder);
1619  builder.setInsertionPointAfterValue(src);
1620  srcField = builder.create<SubindexOp>(src, i);
1621  }
1622  emitPartialConnect(builder, dstField, srcField);
1623  }
1624  } else {
1625  emitConnect(builder, dst, src);
1626  }
1627 }

1632 /// Parse the 'exp' grammar, returning all of the suboperations in the
1633 /// specified vector, and the ultimate SSA value in value.
1634 ///
1635 /// exp ::= id // Ref
1636 /// ::= prim
1637 /// ::= integer-literal-exp
1638 /// ::= exp '.' fieldId
1639 /// ::= exp '[' intLit ']'
1640 /// XX ::= exp '.' DoubleLit // TODO Workaround for #470
1641 /// ::= exp '[' exp ']'
1642 ///
1643 ///
1644 /// If 'isLeadingStmt' is true, then this is being called to parse the first
1645 /// expression in a statement. We can handle some weird cases due to this if
1646 /// we end up parsing the whole statement. In that case we return success, but
1647 /// set the 'result' value to null.
1648 ParseResult FIRStmtParser::parseExpImpl(Value &result, const Twine &message,
1649  bool isLeadingStmt) {
1650  switch (getToken().getKind()) {
1651 
1652  // Handle all the primitive ops: primop exp* intLit* ')'. There is a
1653  // redundant definition of TOK_LPKEYWORD_PRIM which is needed to to get
1654  // around a bug in the MSVC preprocessor to properly paste together the
1655  // tokens lp_##SPELLING.
1656 #define TOK_LPKEYWORD(SPELLING) case FIRToken::lp_##SPELLING:
1657 #define TOK_LPKEYWORD_PRIM(SPELLING, CLASS, NUMOPERANDS) \
1658  case FIRToken::lp_##SPELLING:
1659 #include "FIRTokenKinds.def"
1660  if (parsePrimExp(result))
1661  return failure();
1662  break;
1663 
1664  case FIRToken::kw_UInt:
1665  case FIRToken::kw_SInt:
1666  if (parseIntegerLiteralExp(result))
1667  return failure();
1668  break;
1669 
1670  // Otherwise there are a bunch of keywords that are treated as identifiers
1671  // try them.
1672  case FIRToken::identifier: // exp ::= id
1673  default: {
1674  StringRef name;
1675  auto loc = getToken().getLoc();
1676  SymbolValueEntry symtabEntry;
1677  if (parseId(name, message) ||
1678  moduleContext.lookupSymbolEntry(symtabEntry, name, loc))
1679  return failure();
1680 
1681  // If we looked up a normal value, then we're done.
1682  if (!moduleContext.resolveSymbolEntry(result, symtabEntry, loc, false))
1683  break;
1684 
1685  assert(symtabEntry.is<UnbundledID>() && "should be an instance");
1686 
1687  // Otherwise we referred to an implicitly bundled value. We *must* be in
1688  // the midst of processing a field ID reference or 'is invalid'. If not,
1689  // this is an error.
1690  if (isLeadingStmt && consumeIf(FIRToken::kw_is)) {
1691  if (parseToken(FIRToken::kw_invalid, "expected 'invalid'") ||
1692  parseOptionalInfo())
1693  return failure();
1694 
1695  locationProcessor.setLoc(loc);
1696  // Invalidate all of the results of the bundled value.
1697  unsigned unbundledId = symtabEntry.get<UnbundledID>() - 1;
1698  UnbundledValueEntry &ubEntry =
1699  moduleContext.getUnbundledEntry(unbundledId);
1700  for (auto elt : ubEntry)
1701  emitInvalidate(elt.second);
1702 
1703  // Signify that we parsed the whole statement.
1704  result = Value();
1705  return success();
1706  }
1707 
1708  // Handle the normal "instance.x" reference.
1709  StringRef fieldName;
1710  if (parseToken(FIRToken::period, "expected '.' in field reference") ||
1711  parseFieldId(fieldName, "expected field name") ||
1712  moduleContext.resolveSymbolEntry(result, symtabEntry, fieldName, loc))
1713  return failure();
1714  break;
1715  }
1716  }
1717 
1718  return parseOptionalExpPostscript(result);
1719 }
1720 
1721 /// Parse the postfix productions of expression after the leading expression
1722 /// has been parsed.
1723 ///
1724 /// exp ::= exp '.' fieldId
1725 /// ::= exp '[' intLit ']'
1726 /// XX ::= exp '.' DoubleLit // TODO Workaround for #470
1727 /// ::= exp '[' exp ']'
1728 ParseResult FIRStmtParser::parseOptionalExpPostscript(Value &result) {
1729

1731  while (1) {
1732  // Subfield: exp ::= exp '.' fieldId
1733  if (consumeIf(FIRToken::period)) {
1734  if (parsePostFixFieldId(result))
1735  return failure();
1736 
1737  continue;
1738  }
1739 
1740  // Subindex: exp ::= exp '[' intLit ']' | exp '[' exp ']'
1741  if (consumeIf(FIRToken::l_square)) {
1742  if (getToken().isAny(FIRToken::integer, FIRToken::string)) {
1743  if (parsePostFixIntSubscript(result))
1744  return failure();
1745  continue;
1746  }
1747  if (parsePostFixDynamicSubscript(result))
1748  return failure();
1749 
1750  continue;
1751  }
1752 
1753  return success();
1754  }
1755 }

1798  auto op = builder.create<SubfieldOp>(resultType, result, attrs);

1887 /// prim ::= primop exp* intLit* ')'
1888 ParseResult FIRStmtParser::parsePrimExp(Value &result) {
1889  auto kind = getToken().getKind();
1890  auto loc = getToken().getLoc();
1891  consumeToken();
1892 
1893  // Parse the operands and constant integer arguments.
1894  SmallVector<Value, 3> operands;
1895  SmallVector<int64_t, 3> integers;
1896  if (parseListUntil(FIRToken::r_paren, [&]() -> ParseResult {
1897  // Handle the integer constant case if present.
1898  if (getToken().isAny(FIRToken::integer, FIRToken::signed_integer,
1899  FIRToken::string)) {
1900  integers.push_back(0);
1901  return parseIntLit(integers.back(), "expected integer");
1902  }
1903 
1904  // Otherwise it must be a value operand. These must all come before the
1905  // integers.
1906  if (!integers.empty())
1907  return emitError("expected more integer constants"), failure();
1908 
1909  Value operand;
1910  if (parseExp(operand, "expected expression in primitive operand"))
1911  return failure();
1912 
1913  locationProcessor.setLoc(loc);
1914 
1915  operands.push_back(operand);
1916  return success();
1917  }))
1918  return failure();
1919 
1920  locationProcessor.setLoc(loc);
1921 
1922  SmallVector<FIRRTLType, 3> opTypes;
1923  for (auto v : operands)
1924  opTypes.push_back(v.getType().cast<FIRRTLType>());
1925 
1926  unsigned numOperandsExpected;
1927  SmallVector<StringAttr, 2> attrNames;
1928 
1929  // Get information about the primitive in question.
1930  switch (kind) {
1931  default:
1932  emitError(loc, "primitive not supported yet");
1933  return failure();
1934  case FIRToken::lp_validif:
1935  numOperandsExpected = 2;
1936  break;
1937 #define TOK_LPKEYWORD_PRIM(SPELLING, CLASS, NUMOPERANDS) \
1938  case FIRToken::lp_##SPELLING: \
1939  numOperandsExpected = NUMOPERANDS; \
1940  break;
1941 #include "FIRTokenKinds.def"
1942  }
1943  // Don't add code here, we want these two switch statements to be fused by
