23 // Some types have been expanded so the first layer of aggregate path is
24 // a return value.
25 static LogicalResult updateExpandedPort(StringRef field, AnnoTarget &ref) {
26  if (auto mem = dyn_cast<MemOp>(ref.getOp()))
27  for (size_t p = 0, pe = mem.portNames().size(); p < pe; ++p)
28  if (mem.getPortNameStr(p) == field) {
29  ref = PortAnnoTarget(mem, p);
30  return success();
31  }
32  ref.getOp()->emitError("Cannot find port with name ") << field;
33  return failure();
34 }

36 /// Try to resolve an non-array aggregate name from a target given the type and
37 /// operation of the resolved target. This needs to deal with places where we
38 /// represent bundle returns as split into constituent parts.
39 static FailureOr<unsigned> findBundleElement(Operation *op, Type type,
40  StringRef field) {
41  auto bundle = type.dyn_cast<BundleType>();
42  if (!bundle) {
43  op->emitError("field access '")
44  << field << "' into non-bundle type '" << bundle << "'";
45  return failure();
46  }
47  auto idx = bundle.getElementIndex(field);
48  if (!idx) {
49  op->emitError("cannot resolve field '")
50  << field << "' in subtype '" << bundle << "'";
51  return failure();
52  }
53  return *idx;
54 }

56 /// Try to resolve an array index from a target given the type of the resolved
57 /// target.
58 static FailureOr<unsigned> findVectorElement(Operation *op, Type type,
59  StringRef indexStr) {
60  size_t index;
61  if (indexStr.getAsInteger(10, index)) {
62  op->emitError("Cannot convert '") << indexStr << "' to an integer";
63  return failure();
64  }
65  auto vec = type.dyn_cast<FVectorType>();
66  if (!vec) {
67  op->emitError("index access '")
68  << index << "' into non-vector type '" << type << "'";
69  return failure();
70  }
71  return index;
72 }

74 static FailureOr<unsigned> findFieldID(AnnoTarget &ref,
75  ArrayRef<TargetToken> tokens) {
76  if (tokens.empty())
77  return 0;
78 
79  auto *op = ref.getOp();
80  auto fieldIdx = 0;
81  // The first field for some ops refers to expanded return values.
82  if (isa<MemOp>(ref.getOp())) {
83  if (failed(updateExpandedPort(tokens.front().name, ref)))
84  return {};
85  tokens = tokens.drop_front();
86  }
87 
88  auto type = ref.getType();
89  for (auto token : tokens) {
90  if (token.isIndex) {
91  auto result = findVectorElement(op, type, token.name);
92  if (failed(result))
93  return failure();
94  auto vector = type.cast<FVectorType>();
95  type = vector.getElementType();
96  fieldIdx += vector.getFieldID(*result);
97  } else {
98  auto result = findBundleElement(op, type, token.name);
99  if (failed(result))
100  return failure();
101  auto bundle = type.cast<BundleType>();
102  type = bundle.getElementType(*result);
103  fieldIdx += bundle.getFieldID(*result);
104  }
105  }
106  return fieldIdx;
107 }

109 void TokenAnnoTarget::toVector(SmallVectorImpl<char> &out) const {
110  out.push_back('~');
111  out.append(circuit.begin(), circuit.end());
112  out.push_back('|');
113  for (auto modInstPair : instances) {
114  out.append(modInstPair.first.begin(), modInstPair.first.end());
115  out.push_back('/');
116  out.append(modInstPair.second.begin(), modInstPair.second.end());
117  out.push_back(':');
118  }
119  out.append(module.begin(), module.end());
120  if (name.empty())
121  return;
122  out.push_back('>');
123  out.append(name.begin(), name.end());
124  for (auto comp : component) {
125  out.push_back(comp.isIndex ? '[' : '.');
126  out.append(comp.name.begin(), comp.name.end());
127  if (comp.isIndex)
128  out.push_back(']');
129  }
130 }

132 std::string firrtl::canonicalizeTarget(StringRef target) {
133 
134  if (target.empty())
135  return target.str();
136 
137  // If this is a normal Target (not a Named), erase that field in the JSON
138  // object and return that Target.
139  if (target[0] == '~')
140  return target.str();
141 
142  // This is a legacy target using the firrtl.annotations.Named type. This
143  // can be trivially canonicalized to a non-legacy target, so we do it with
144  // the following three mappings:
145  // 1. CircuitName => CircuitTarget, e.g., A -> ~A
146  // 2. ModuleName => ModuleTarget, e.g., A.B -> ~A|B
147  // 3. ComponentName => ReferenceTarget, e.g., A.B.C -> ~A|B>C
148  std::string newTarget = ("~" + target).str();
149  auto n = newTarget.find('.');
150  if (n != std::string::npos)
151  newTarget[n] = '|';
152  n = newTarget.find('.');
153  if (n != std::string::npos)
154  newTarget[n] = '>';
155  return newTarget;
156 }

158 Optional<AnnoPathValue> firrtl::resolveEntities(TokenAnnoTarget path,
159  CircuitOp circuit,
160  SymbolTable &symTbl,
161  CircuitTargetCache &cache) {
162  // Validate circuit name.
163  if (!path.circuit.empty() && circuit.name() != path.circuit) {
164  circuit->emitError("circuit name doesn't match annotation '")
165  << path.circuit << '\'';
166  return {};
167  }
168  // Circuit only target.
169  if (path.module.empty()) {
170  assert(path.name.empty() && path.instances.empty() &&
171  path.component.empty());
172  return AnnoPathValue(circuit);
173  }
174 
175  // Resolve all instances for non-local paths.
176  SmallVector<InstanceOp> instances;
177  for (auto p : path.instances) {
178  auto mod = symTbl.lookup<FModuleOp>(p.first);
179  if (!mod) {
180  circuit->emitError("module doesn't exist '") << p.first << '\'';
181  return {};
182  }
183  auto resolved = cache.lookup(mod, p.second);
184  if (!resolved || !isa<InstanceOp>(resolved.getOp())) {
185  circuit.emitError("cannot find instance '")
186  << p.second << "' in '" << mod.getName() << "'";
187  return {};
188  }
189  instances.push_back(cast<InstanceOp>(resolved.getOp()));
190  }
191  // The final module is where the named target is (or is the named target).
192  auto mod = symTbl.lookup<FModuleLike>(path.module);
193  if (!mod) {
194  circuit->emitError("module doesn't exist '") << path.module << '\'';
195  return {};
196  }
197  AnnoTarget ref;
198  if (path.name.empty()) {
199  assert(path.component.empty());
200  ref = OpAnnoTarget(mod);
201  } else {
202  ref = cache.lookup(mod, path.name);
203  if (!ref) {
204  circuit->emitError("cannot find name '")
205  << path.name << "' in " << mod.moduleName();
206  return {};
207  }
208  }
209 
210  // If the reference is pointing to an instance op, we have to move the target
211  // to the module. This is done both because it is logical to have one
212  // representation (this effectively canonicalizes a reference target on an
213  // instance into an instance target) and because the SFC has a pass that does
214  // this conversion. E.g., this is converting (where "bar" is an instance):
215  // ~Foo|Foo>bar
216  // Into:
217  // ~Foo|Foo/bar:Bar
218  ArrayRef<TargetToken> component(path.component);
219  if (auto instance = dyn_cast<InstanceOp>(ref.getOp())) {
220  instances.push_back(instance);
221  auto target = instance.getReferencedModule(symTbl);
222  if (component.empty()) {
223  ref = OpAnnoTarget(instance.getReferencedModule(symTbl));
224  } else if (component.front().isIndex) {
225  circuit->emitError() << "illegal target '" << path.str()
226  << "' indexes into an instance";
227  return {};
228  } else {
229  auto field = component.front().name;
230  ref = AnnoTarget();
231  for (size_t p = 0, pe = target.getNumPorts(); p < pe; ++p)
232  if (target.getPortName(p) == field) {
233  ref = PortAnnoTarget(target, p);
234  break;
235  }
236  if (!ref) {
237  circuit->emitError("!cannot find port '")
238  << field << "' in module " << target.moduleName();
239  return {};
240  }
241  component = component.drop_front();
242  }
243  }
244 
245  // If we have aggregate specifiers, resolve those now. This call can update
246  // the ref to target a port of a memory.
247  auto result = findFieldID(ref, component);
248  if (failed(result))
249  return {};
250  auto fieldIdx = *result;
251 
252  return AnnoPathValue(instances, ref, fieldIdx);
253 }

255 /// split a target string into it constituent parts. This is the primary parser
256 /// for targets.
257 Optional<TokenAnnoTarget> firrtl::tokenizePath(StringRef origTarget) {
258  // An empty string is not a legal target.
259  if (origTarget.empty())
260  return {};
261  StringRef target = origTarget;
262  TokenAnnoTarget retval;
263  std::tie(retval.circuit, target) = target.split('|');
264  if (!retval.circuit.empty() && retval.circuit[0] == '~')
265  retval.circuit = retval.circuit.drop_front();
266  while (target.count(':')) {
267  StringRef nla;
268  std::tie(nla, target) = target.split(':');
269  StringRef inst, mod;
270  std::tie(mod, inst) = nla.split('/');
271  retval.instances.emplace_back(mod, inst);
272  }
273  // remove aggregate
274  auto targetBase =
275  target.take_until([](char c) { return c == '.' || c == '['; });
276  auto aggBase = target.drop_front(targetBase.size());
277  std::tie(retval.module, retval.name) = targetBase.split('>');
278  while (!aggBase.empty()) {
279  if (aggBase[0] == '.') {
280  aggBase = aggBase.drop_front();
281  StringRef field = aggBase.take_front(aggBase.find_first_of("[."));
282  aggBase = aggBase.drop_front(field.size());
283  retval.component.push_back({field, false});
284  } else if (aggBase[0] == '[') {
285  aggBase = aggBase.drop_front();
286  StringRef index = aggBase.take_front(aggBase.find_first_of(']'));
287  aggBase = aggBase.drop_front(index.size() + 1);
288  retval.component.push_back({index, true});
289  } else {
290  return {};
291  }
292  }
293 
294  return retval;
295 }

297 Optional<AnnoPathValue> firrtl::resolvePath(StringRef rawPath,
298  CircuitOp circuit,
299  SymbolTable &symTbl,
300  CircuitTargetCache &cache) {
301  auto pathStr = canonicalizeTarget(rawPath);
302  StringRef path{pathStr};
303 
304  auto tokens = tokenizePath(path);
305  if (!tokens) {
306  circuit->emitError("Cannot tokenize annotation path ") << rawPath;
307  return {};
308  }
309 
310  return resolveEntities(*tokens, circuit, symTbl, cache);
311 }

318  // Add ports
319  for (const auto &p : llvm::enumerate(mod.getPorts()))
320  targets.insert({p.value().name, PortAnnoTarget(mod, p.index())});
321 
322  // And named things

326  SeqMemOp, MemoryPortOp>([&](auto op) {
327  // To be safe, check attribute and non-empty name before adding.
328  if (auto name = op.nameAttr(); name && !name.getValue().empty())
329  targets.insert({name, OpAnnoTarget(op)});
330  });
331  });
