49 // The potentially non-local resolved annotation.
50 struct AnnoPathValue {
51  SmallVector<InstanceOp> instances;
52  AnnoTarget ref;

54 
55  AnnoPathValue() = default;
56  AnnoPathValue(CircuitOp op) : ref(OpAnnoTarget(op)) {}
57  AnnoPathValue(Operation *op) : ref(OpAnnoTarget(op)) {}
58  AnnoPathValue(const SmallVectorImpl<InstanceOp> &insts, AnnoTarget b,
59  unsigned fieldIdx)
60  : instances(insts.begin(), insts.end()), ref(b), fieldIdx(fieldIdx) {}
61 
62  bool isLocal() const { return instances.empty(); }
63 
64  template <typename... T>
65  bool isOpOfType() const {
66  if (auto opRef = ref.dyn_cast<OpAnnoTarget>())
67  return isa<T...>(opRef.getOp());
68  return false;
69  }

81  /// Lookup the target for 'name', empty if not found.
82  /// (check for validity using operator bool()).
83  AnnoTarget getTargetForName(StringRef name) const {
84  auto it = targets.find(name);
85  if (it == targets.end())
86  return {};
87  return it->second;
88  }
89 
90 private:

97 /// Cache AnnoTargets for a circuit's modules, walked as needed.
98 struct CircuitTargetCache {
99  /// Get cache for specified module, creating it as needed.
100  /// Returned reference may become invalidated by future calls.
101  const AnnoTargetCache &getOrCreateCacheFor(FModuleLike module) {
102  auto it = targetCaches.find(module);
103  if (it == targetCaches.end())
104  it = targetCaches.try_emplace(module, module).first;
105  return it->second;
106  }
107 
108  /// Lookup the target for 'name' in 'module'.
109  AnnoTarget lookup(FModuleLike module, StringRef name) {
110  return getOrCreateCacheFor(module).getTargetForName(name);
111  }
112 
113 private:
114  DenseMap<Operation *, AnnoTargetCache> targetCaches;
115 };

184 /// Implements the same behavior as DictionaryAttr::getAs<A> to return the
185 /// value of a specific type associated with a key in a dictionary. However,
186 /// this is specialized to print a useful error message, specific to custom
187 /// annotation process, on failure.
188 template <typename A>
189 A tryGetAs(DictionaryAttr &dict, const Attribute &root, StringRef key,
190  Location loc, Twine className, Twine path = Twine()) {
191  // Check that the key exists.
192  auto value = dict.get(key);
193  if (!value) {
194  SmallString<128> msg;
195  if (path.isTriviallyEmpty())
196  msg = ("Annotation '" + className + "' did not contain required key '" +
197  key + "'.")
198  .str();
199  else
200  msg = ("Annotation '" + className + "' with path '" + path +
201  "' did not contain required key '" + key + "'.")
202  .str();
203  mlir::emitError(loc, msg).attachNote()
204  << "The full Annotation is reproduced here: " << root << "\n";
205  return nullptr;
206  }
207  // Check that the value has the correct type.
208  auto valueA = value.dyn_cast_or_null<A>();
209  if (!valueA) {
210  SmallString<128> msg;
211  if (path.isTriviallyEmpty())
212  msg = ("Annotation '" + className +
213  "' did not contain the correct type for key '" + key + "'.")
214  .str();
215  else
216  msg = ("Annotation '" + className + "' with path '" + path +
217  "' did not contain the correct type for key '" + key + "'.")
218  .str();
219  mlir::emitError(loc, msg).attachNote()
220  << "The full Annotation is reproduced here: " << root << "\n";
221  return nullptr;
222  }
223  return valueA;
224 }
