95 static bool applyToPort(AnnotationSet annos, Operation *op, size_t portCount,
96  size_t portNo) {
97  assert(portNo < portCount && "port index out of range.");
98  auto *context = op->getContext();
99  auto before = op->getAttrOfType<ArrayAttr>(getPortAnnotationAttrName());
100  SmallVector<Attribute> portAnnotations;
101  if (!before || before.empty())
102  portAnnotations.assign(portCount, ArrayAttr::get(context, {}));
103  else
104  portAnnotations.append(before.begin(), before.end());
105  portAnnotations[portNo] = annos.getArrayAttr();
106  auto after = ArrayAttr::get(context, portAnnotations);
107  op->setAttr(getPortAnnotationAttrName(), after);
108  return before != after;
109 }

141 static DictionaryAttr applyToDictionaryAttrImpl(const AnnotationSet &annoSet,
142  StringRef key,
143  ArrayRef<NamedAttribute> attrs,
144  bool sorted,
145  DictionaryAttr originalDict) {
146  // Find the location in the dictionary where the entry would go.
147  ArrayRef<NamedAttribute>::iterator it;
148  if (sorted) {
149  it = llvm::lower_bound(attrs, key);
150  if (it != attrs.end() && it->getName() != key)
151  it = attrs.end();
152  } else {
153  it = llvm::find_if(
154  attrs, [key](NamedAttribute attr) { return attr.getName() == key; });
155  }
156 
157  // Fast path in case there are no annotations in the dictionary and we are not
158  // supposed to add any.
159  if (it == attrs.end() && annoSet.empty())
160  return originalDict;
161 
162  // Fast path in case there already is an entry in the dictionary, it matches
163  // the set, and, in the case we're supposed to remove empty sets, we're not
164  // leaving an empty entry in the dictionary.
165  if (it != attrs.end() && it->getValue() == annoSet.getArrayAttr() &&
166  !annoSet.empty())
167  return originalDict;
168 
169  // If we arrive here, we are supposed to assemble a new dictionary.
170  SmallVector<NamedAttribute> newAttrs;
171  newAttrs.reserve(attrs.size() + 1);
172  newAttrs.append(attrs.begin(), it);
173  if (!annoSet.empty())
174  newAttrs.push_back(
175  {StringAttr::get(annoSet.getContext(), key), annoSet.getArrayAttr()});
176  if (it != attrs.end())
177  newAttrs.append(it + 1, attrs.end());
178  return sorted ? DictionaryAttr::getWithSorted(annoSet.getContext(), newAttrs)
179  : DictionaryAttr::get(annoSet.getContext(), newAttrs);
180 }

349 /// Remove all annotations from this annotation set for which `predicate`
350 /// returns true.
351 bool AnnotationSet::removeAnnotations(
352  llvm::function_ref<bool(Annotation)> predicate) {
353  // Fast path for empty sets.
354  auto attr = getArrayAttr();
355  if (!attr)
356  return false;
357 
358  // Search for the first match.
359  ArrayRef<Attribute> annos = getArrayAttr().getValue();
360  auto it = annos.begin();
361  while (it != annos.end() && !predicate(Annotation(*it)))
362  ++it;
363 
364  // Fast path for sets where the predicate never matched.
365  if (it == annos.end())
366  return false;
367 
368  // Build a filtered list of annotations.
369  SmallVector<Attribute> filteredAnnos;
370  filteredAnnos.reserve(annos.size());
371  filteredAnnos.append(annos.begin(), it);
372  ++it;
373  while (it != annos.end()) {
374  if (!predicate(Annotation(*it)))
375  filteredAnnos.push_back(*it);
376  ++it;
377  }
378  annotations = ArrayAttr::get(getContext(), filteredAnnos);
379  return true;
380 }

398 /// Remove all port annotations from a module or extmodule for which `predicate`
399 /// returns true.
400 bool AnnotationSet::removePortAnnotations(
401  Operation *module,
402  llvm::function_ref<bool(unsigned, Annotation)> predicate) {
403  auto ports = module->getAttr("portAnnotations").dyn_cast_or_null<ArrayAttr>();
404  if (!ports || ports.empty())
405  return false;
406 
407  // Collect results
408  SmallVector<Attribute> newAnnos;
409 
410  // Filter the annotations on each argument.
411  bool changed = false;
412  for (unsigned argNum = 0, argNumEnd = ports.size(); argNum < argNumEnd;
413  ++argNum) {
414  AnnotationSet annos(AnnotationSet(ports[argNum].cast<ArrayAttr>()));
415 
416  // Go through all annotations on this port and extract the interesting
417  // ones. If any modifications were done, keep a reduced set of attributes
418  // around for the port, otherwise just stick with the existing ones.
419  if (!annos.empty())
420  changed |= annos.removeAnnotations(
421  [&](Annotation anno) { return predicate(argNum, anno); });
422  newAnnos.push_back(annos.getArrayAttr());
423  }
424 
425  // If we have made any changes, apply them to the operation.
426  if (changed)
427  module->setAttr("portAnnotations",
428  ArrayAttr::get(module->getContext(), newAnnos));
429  return changed;
430 }

464 void Annotation::setMember(StringRef name, Attribute value) {
465  // Binary search for the matching field.
466  auto dict = getDict();
467  auto [it, found] = mlir::impl::findAttrSorted(dict.begin(), dict.end(), name);
468  auto index = std::distance(dict.begin(), it);
469  // Create an array for the new members.
470  SmallVector<NamedAttribute> attributes;
471  attributes.reserve(dict.size() + 1);
472  // Copy over the leading annotations.
473  for (auto field : dict.getValue().take_front(index))
474  attributes.push_back(field);
475  // Push the new member.
476  auto nameAttr = StringAttr::get(dict.getContext(), name);
477  attributes.push_back(NamedAttribute(nameAttr, value));
478  // Copy remaining members, skipping the old field value.
479  for (auto field : dict.getValue().drop_front(index + found))
480  attributes.push_back(field);
481  // Commit the dictionary.
482  setDict(DictionaryAttr::getWithSorted(dict.getContext(), attributes));
483 }

485 void Annotation::removeMember(StringAttr name) {
486  auto dict = getDict();
487  SmallVector<NamedAttribute> attributes;
488  attributes.reserve(dict.size() - 1);
489  auto i = dict.begin();
490  auto e = dict.end();
491  while (i != e && i->getValue() != name)
492  attributes.push_back(*(i++));
493  // If the member was not here, just return.
494  if (i == e)
495  return;
496  // Copy the rest of the members over.
497  attributes.append(++i, e);
498  // Commit the dictionary.
499  setDict(DictionaryAttr::getWithSorted(dict.getContext(), attributes));
500 }

506  auto index = std::distance(dict.begin(), it);
