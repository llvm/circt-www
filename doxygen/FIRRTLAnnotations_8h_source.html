230 /// This class provides a read-only projection of an annotation.
231 class Annotation {
232 public:
233  Annotation(DictionaryAttr attrDict) : attrDict(attrDict) {
234  assert(attrDict && "null dictionaries not allowed");
235  }
236 
237  DictionaryAttr getDict() const { return attrDict; }
238 
239  /// Return the 'class' that this annotation is representing.
240  StringAttr getClassAttr() const;
241  StringRef getClass() const;
242 
243  /// Return true if this annotation matches any of the specified class names.
244  template <typename... Args>
245  bool isClass(Args... names) const {
246  return ClassIsa{getClassAttr()}(names...);
247  }
248 
249  /// Return a member of the annotation.
250  template <typename AttrClass = Attribute>
251  AttrClass getMember(StringAttr name) const {
252  // TODO: Once https://reviews.llvm.org/D103822 lands, the `const_cast` can
253  // go away.
254  return const_cast<DictionaryAttr &>(attrDict).getAs<AttrClass>(name);
255  }
256  template <typename AttrClass = Attribute>
257  AttrClass getMember(StringRef name) const {
258  // TODO: Once https://reviews.llvm.org/D103822 lands, the `const_cast` can
259  // go away.
260  return const_cast<DictionaryAttr &>(attrDict).getAs<AttrClass>(name);
261  }
262 
263  bool operator==(const Annotation &other) const {
264  return attrDict == other.attrDict;
265  }
266  bool operator!=(const Annotation &other) const { return !(*this == other); }
267 
268 private:
269  DictionaryAttr attrDict;
270 
271  /// Helper struct to perform variadic class equality check.
272  struct ClassIsa {
273  StringAttr cls;
274 
275  bool operator()() const { return false; }
276  template <typename T, typename... Rest>
277  bool operator()(T name, Rest... rest) const {
278  return compare(name) || (*this)(rest...);
279  }
280 
281  private:
282  bool compare(StringAttr name) const { return cls == name; }
283  bool compare(StringRef name) const { return cls && cls.getValue() == name; }
284  };
