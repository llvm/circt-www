35  StringRef attrName;
36  Attribute attr;
37  if (p.parseKeyword(&attrName))
38  return Attribute();
39  auto parseResult =
40  generatedAttributeParser(getContext(), p, attrName, type, attr);
41  if (parseResult.hasValue())
42  return attr;
43  p.emitError(p.getNameLoc(), "Unexpected FIRRTL attribute '" + attrName + "'");
