37 IntegerAttr direction::packAttribute(MLIRContext *context,
38  ArrayRef<Direction> directions) {
39  // Pack the array of directions into an APInt. Input is zero, output is one.
40  auto size = directions.size();
41  APInt portDirections(size, 0);
42  for (size_t i = 0; i != size; ++i)
43  if (directions[i] == Direction::Out)
44  portDirections.setBit(i);
45  return IntegerAttr::get(IntegerType::get(context, size), portDirections);
46 }

50  "Direction attributes must be signless integers");

52  auto size = value.getBitWidth();
53  SmallVector<Direction> result;

92 Attribute InnerSymPropertiesAttr::parse(AsmParser &parser, Type type) {
93  StringAttr name;
94  NamedAttrList dummyList;
95  int64_t fieldId = 0;
96  StringRef visibility;
97  if (parser.parseLess() || parser.parseSymbolName(name, "name", dummyList) ||
98  parser.parseComma() || parser.parseInteger(fieldId) ||
99  parser.parseComma() ||
100  parser.parseOptionalKeyword(&visibility,
101  {"public", "private", "nested"}) ||
102  parser.parseGreater())
103  return Attribute();
104  StringAttr visibilityAttr = parser.getBuilder().getStringAttr(visibility);

106  return InnerSymPropertiesAttr::get(parser.getContext(), name, fieldId,

124 Attribute InnerSymAttr::parse(AsmParser &parser, Type type) {
125  StringAttr sym;
126  NamedAttrList dummyList;
127  SmallVector<InnerSymPropertiesAttr, 4> names;
128  if (!parser.parseOptionalSymbolName(sym, "dummy", dummyList))
129  names.push_back(InnerSymPropertiesAttr::get(sym));
130  else if (parser.parseCommaSeparatedList(
131  OpAsmParser::Delimiter::Square, [&]() -> ParseResult {
132  InnerSymPropertiesAttr prop;
133  if (parser.parseCustomAttributeWithFallback(
134  prop, mlir::Type{}, "dummy", dummyList))
135  return failure();
136 
137  names.push_back(prop);
138 
139  return success();
140  }))
141  return Attribute();
142 
143  std::sort(names.begin(), names.end(),
144  [&](InnerSymPropertiesAttr a, InnerSymPropertiesAttr b) {
145  return a.getFieldID() < b.getFieldID();
146  });
147 
148  return InnerSymAttr::get(parser.getContext(), names);
149 }
