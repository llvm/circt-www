29 FieldRef circt::firrtl::getFieldRefFromValue(Value value) {
30  // This code walks upwards from the subfield and calculates the field ID at
31  // each level. At each stage, it must take the current id, and re-index it as
32  // a nested bundle under the parent field.. This is accomplished by using the
33  // parent field's ID as a base, and adding the field ID of the child.
34  unsigned id = 0;
35  while (value) {
36  Operation *op = value.getDefiningOp();
37 
38  // If this is a block argument, we are done.
39  if (!op)
40  break;
41 
42  if (auto subfieldOp = dyn_cast<SubfieldOp>(op)) {
43  value = subfieldOp.input();
44  auto bundleType = value.getType().cast<BundleType>();
45  // Rebase the current index on the parent field's index.
46  id += bundleType.getFieldID(subfieldOp.fieldIndex());
47  } else if (auto subindexOp = dyn_cast<SubindexOp>(op)) {
48  auto vecType = subindexOp.input().getType().cast<FVectorType>();
49  // Rebase the current index on the parent field's index.
50  id += vecType.getFieldID(subindexOp.index());
51  } else {
52  break;
53  }
54  }
55  return {value, id};
56 }

58 /// Get the string name of a value which is a direct child of a declaration op.
59 static void getDeclName(Value value, SmallString<64> &string) {
60  if (auto arg = value.dyn_cast<BlockArgument>()) {
61  // Get the module ports and get the name.
62  auto module = cast<FModuleOp>(arg.getOwner()->getParentOp());
63  SmallVector<ModulePortInfo> ports = module.getPorts();
64  string += ports[arg.getArgNumber()].name.getValue();
65  return;
66  }
67 
68  auto *op = value.getDefiningOp();
69  TypeSwitch<Operation *>(op)
70  .Case<InstanceOp, MemOp>([&](auto op) {
71  string += op.name();
72  string += ".";
73  string +=
74  op.getPortName(value.cast<OpResult>().getResultNumber()).getValue();
75  })
76  .Case<WireOp, RegOp, RegResetOp>([&](auto op) { string += op.name(); });
77 }

79 std::string circt::firrtl::getFieldName(const FieldRef &fieldRef) {
80 
81  SmallString<64> name;
82  auto value = fieldRef.getValue();
83  getDeclName(value, name);
84 
85  auto type = value.getType();
86  auto localID = fieldRef.getFieldID();
87  while (localID) {
88  if (auto bundleType = type.dyn_cast<BundleType>()) {
89  auto index = bundleType.getIndexForFieldID(localID);
90  // Add the current field string, and recurse into a subfield.
91  auto &element = bundleType.getElements()[index];
92  name += ".";
93  name += element.name.getValue();
94  // Recurse in to the element type.
95  type = element.type;
96  localID = localID - bundleType.getFieldID(index);
97  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
98  auto index = vecType.getIndexForFieldID(localID);
99  name += "[";
100  name += std::to_string(index);
101  name += "]";
102  // Recurse in to the element type.
103  type = vecType.getElementType();
104  localID = localID - vecType.getFieldID(index);
105  } else {
106  // If we reach here, the field ref is pointing inside some aggregate type
107  // that isn't a bundle or a vector. If the type is a ground type, then the
108  // localID should be 0 at this point, and we should have broken from the
109  // loop.
110  llvm_unreachable("unsupported type");
111  }
112  }
113 
114  return name.str().str();
115 }

136 struct FIRRTLOpAsmDialectInterface : public OpAsmDialectInterface {
137  using OpAsmDialectInterface::OpAsmDialectInterface;

139  /// Get a special name to use when printing the given operation. See
140  /// OpAsmInterface.td#getAsmResultNames for usage details and documentation.
141  void getAsmResultNames(Operation *op,
142  OpAsmSetValueNameFn setNameFn) const override {
143  if (auto instance = dyn_cast<InstanceOp>(op)) {
144  StringRef base;
145  if (auto nameAttr = op->getAttrOfType<StringAttr>("name"))
146  base = nameAttr.getValue();
147  if (base.empty())
148  base = "inst";
149 
150  for (size_t i = 0, e = op->getNumResults(); i != e; ++i) {
151  setNameFn(instance.getResult(i),
152  (base + "_" + instance.getPortNameStr(i)).str());
153  }
154  return;
155  }
156 
157  if (auto memory = dyn_cast<MemOp>(op)) {
158  StringRef base;
159  if (auto nameAttr = op->getAttrOfType<StringAttr>("name"))
160  base = nameAttr.getValue();
161  if (base.empty())
162  base = "mem";
163 
164  for (size_t i = 0, e = op->getNumResults(); i != e; ++i) {
165  setNameFn(memory.getResult(i),
166  (base + "_" + memory.getPortNameStr(i)).str());
167  }
168  return;
169  }
170 
171  // Many firrtl dialect operations have an optional 'name' attribute. If
172  // present, use it.
173  if (op->getNumResults() == 1)
174  if (auto nameAttr = op->getAttrOfType<StringAttr>("name"))
175  setNameFn(op->getResult(0), nameAttr.getValue());
176 
177  // For constants in particular, propagate the value into the result name to
178  // make it easier to read the IR.
179  if (auto constant = dyn_cast<ConstantOp>(op)) {
180  auto intTy = constant.getType().dyn_cast<IntType>();
181 
182  // Otherwise, build a complex name with the value and type.
183  SmallString<32> specialNameBuffer;
184  llvm::raw_svector_ostream specialName(specialNameBuffer);
185  specialName << 'c';
186  if (intTy) {
187  constant.value().print(specialName, /*isSigned:*/ intTy.isSigned());
188 
189  specialName << (intTy.isSigned() ? "_si" : "_ui");
190  auto width = intTy.getWidthOrSentinel();
191  if (width != -1)
192  specialName << width;
193  } else {
194  constant.value().print(specialName, /*isSigned:*/ false);
195  }
196  setNameFn(constant.getResult(), specialName.str());
197  }
198 
199  if (auto specialConstant = dyn_cast<SpecialConstantOp>(op)) {
200  SmallString<32> specialNameBuffer;
201  llvm::raw_svector_ostream specialName(specialNameBuffer);
202  specialName << 'c';
203  specialName << static_cast<unsigned>(specialConstant.value());
204  auto type = specialConstant.getType();
205  if (type.isa<ClockType>()) {
206  specialName << "_clock";
207  } else if (type.isa<ResetType>()) {
208  specialName << "_reset";
209  } else if (type.isa<AsyncResetType>()) {
210  specialName << "_asyncreset";
211  }
212  setNameFn(specialConstant.getResult(), specialName.str());
213  }
214 
215  // Set invalid values to have a distinct name.
216  if (auto invalid = dyn_cast<InvalidValueOp>(op)) {
217  std::string name;
218  if (auto ty = invalid.getType().dyn_cast<IntType>()) {
219  const char *base = ty.isSigned() ? "invalid_si" : "invalid_ui";
220  auto width = ty.getWidthOrSentinel();
221  if (width == -1)
222  name = base;
223  else
224  name = (Twine(base) + Twine(width)).str();
225  } else if (auto ty = invalid.getType().dyn_cast<AnalogType>()) {
226  auto width = ty.getWidthOrSentinel();
227  if (width == -1)
228  name = "invalid_analog";
229  else
230  name = ("invalid_analog" + Twine(width)).str();
231  } else if (invalid.getType().isa<AsyncResetType>())
232  name = "invalid_asyncreset";
233  else if (invalid.getType().isa<ResetType>())
234  name = "invalid_reset";
235  else if (invalid.getType().isa<ClockType>())
236  name = "invalid_clock";
237  else
238  name = "invalid";
239 
240  setNameFn(invalid.getResult(), name);
241  }
242  }

244  /// Get a special name to use when printing the entry block arguments of the
245  /// region contained by an operation in this dialect.
246  void getAsmBlockArgumentNames(Block *block,
247  OpAsmSetValueNameFn setNameFn) const override {
248  // Check to see if the operation containing the arguments has 'firrtl.name'
249  // attributes for them. If so, use that as the name.
250  auto *parentOp = block->getParentOp();
251  auto argAttr = getModulePortNames(parentOp);
252 
253  // Do not crash on invalid IR.
254  if (!argAttr || argAttr.size() != block->getNumArguments())
255  return;
256 
257  for (size_t i = 0, e = block->getNumArguments(); i != e; ++i) {
258  auto str = argAttr[i].cast<StringAttr>().getValue();
259  if (!str.empty())
260  setNameFn(block->getArgument(i), str);
261  }
262  }
263 };
264 } // end anonymous namespace

285 /// Registered hook to materialize a single constant operation from a given
286 /// attribute value with the desired resultant type. This method should use
287 /// the provided builder to create the operation without changing the
288 /// insertion position. The generated operation is expected to be constant
289 /// like, i.e. single result, zero operands, non side-effecting, etc. On
290 /// success, this hook should return the value generated to represent the
291 /// constant value. Otherwise, it should return null on failure.
292 Operation *FIRRTLDialect::materializeConstant(OpBuilder &builder,
293  Attribute value, Type type,
294  Location loc) {
295 
296  // Boolean constants. Boolean attributes are always a special constant type
297  // like ClockType and ResetType. Since BoolAttrs are also IntegerAttrs, its
298  // important that this goes first.
299  if (auto attrValue = value.dyn_cast<BoolAttr>()) {
300  assert(
301  type.isa<ClockType>() || type.isa<AsyncResetType>() ||
302  type.isa<ResetType>() &&
303  "BoolAttrs can only be materialized for special constant types.");
304  return builder.create<SpecialConstantOp>(loc, type, attrValue);
305  }
306 
307  // Integer constants.
308  if (auto attrValue = value.dyn_cast<IntegerAttr>()) {
309  // Integer attributes (ui1) might still be special constant types.
310  if (attrValue.getValue().getBitWidth() == 1 &&
311  (type.isa<ClockType>() || type.isa<AsyncResetType>() ||
312  type.isa<ResetType>()))
313  return builder.create<SpecialConstantOp>(
314  loc, type,
315  builder.getBoolAttr(attrValue.getValue().isAllOnesValue()));
316 
317  auto intType = type.cast<IntType>();
318  assert((!intType.hasWidth() || (unsigned)intType.getWidthOrSentinel() ==
319  attrValue.getValue().getBitWidth()) &&
320  "type/value width mismatch materializing constant");
321  return builder.create<ConstantOp>(loc, type, attrValue);
322  }
323 
324  // InvalidValue constants.
325  if (auto invalidValue = value.dyn_cast<InvalidValueAttr>())
326  return builder.create<InvalidValueOp>(loc, type);
327 
328  return nullptr;
329 }
