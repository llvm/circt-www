28 ArrayRef<InstancePath> InstancePathCache::getAbsolutePaths(Operation *op) {
29  assert(isa<FModuleLike>(op));
30 
31  // If we have reached the circuit root, we're done.
32  if (op == instanceGraph.getTopLevelNode()->getModule()) {
33  static InstancePath empty{};
34  return empty; // array with single empty path
35  }
36 
37  // Fast path: hit the cache.
38  auto cached = absolutePathsCache.find(op);
39  if (cached != absolutePathsCache.end())
40  return cached->second;
41 
42  // For each instance, collect the instance paths to its parent and append the
43  // instance itself to each.
44  SmallVector<InstancePath, 8> extendedPaths;
45  for (auto *inst : instanceGraph[op]->uses()) {
46  auto instPaths = getAbsolutePaths(inst->getParent()->getModule());
47  extendedPaths.reserve(instPaths.size());
48  for (auto path : instPaths) {
49  extendedPaths.push_back(
50  appendInstance(path, cast<InstanceOp>(*inst->getInstance())));
51  }
52  }
53 
54  // Move the list of paths into the bump allocator for later quick retrieval.
55  ArrayRef<InstancePath> pathList;
56  if (!extendedPaths.empty()) {
57  auto *paths = allocator.Allocate<InstancePath>(extendedPaths.size());
58  std::copy(extendedPaths.begin(), extendedPaths.end(), paths);
59  pathList = ArrayRef<InstancePath>(paths, extendedPaths.size());
60  }
61 
62  absolutePathsCache.insert({op, pathList});
63  return pathList;
64 }

66 InstancePath InstancePathCache::appendInstance(InstancePath path,
67  InstanceOp inst) {
68  size_t n = path.size() + 1;
69  auto *newPath = allocator.Allocate<InstanceOp>(n);
70  std::copy(path.begin(), path.end(), newPath);
71  newPath[path.size()] = inst;
72  return InstancePath(newPath, n);
73 }
