28 /// This graph tracks modules and where they are instantiated. This is intended
29 /// to be used as a cached analysis on FIRRTL circuits. This class can be used
30 /// to walk the modules efficiently in a bottom-up or top-down order.
31 ///
32 /// To use this class, retrieve a cached copy from the analysis manager:
33 /// auto &instanceGraph = getAnalysis<InstanceGraph>(getOperation());
34 class InstanceGraph : public hw::InstanceGraphBase {
35 public:
36  /// Create a new module graph of a circuit. This must be called on a FIRRTL
37  /// CircuitOp or MLIR ModuleOp.
38  explicit InstanceGraph(Operation *operation);
39 
40  /// Get the node corresponding to the top-level module of a circuit.
41  InstanceGraphNode *getTopLevelNode() override { return topLevelNode; }
42 
43  /// Get the module corresponding to the top-lebel module of a circuit.
44  FModuleLike getTopLevelModule() {
45  return cast<FModuleLike>(*getTopLevelNode()->getModule());
46  }
47 
48 private:
49  InstanceGraphNode *topLevelNode;
50 };

55 template <typename T>
56 inline static T &formatInstancePath(T &into, const InstancePath &path) {
57  into << "$root";
58  for (auto inst : path)
59  into << "/" << inst.name() << ":" << inst.moduleName();
60  return into;
61 }

63 template <typename T>
64 static T &operator<<(T &os, const InstancePath &path) {
65  return formatInstancePath(os, path);
