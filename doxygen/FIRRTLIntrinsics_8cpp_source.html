117 LogicalResult IntrinsicLowerings::lower(CircuitOp circuit,
118  bool allowUnknownIntrinsics) {
119  unsigned numFailures = 0;
120  for (auto op : llvm::make_early_inc_range(circuit.getOps<FModuleLike>())) {
121  if (auto extMod = dyn_cast<FExtModuleOp>(*op)) {
122  // Special-case some extmodules, identifying them by name.
123  auto it = extmods.find(extMod.getDefnameAttr());
124  if (it != extmods.end()) {
125  if (succeeded(it->second(op))) {
126  op.erase();
127  ++numConverted;
128  } else {
129  ++numFailures;
130  }
131  }
132  continue;
133  }
134 
135  auto intMod = dyn_cast<FIntModuleOp>(*op);
136  if (!intMod)
137  continue;
138 
139  auto intname = intMod.getIntrinsicAttr();
140 
141  // Find the converter and apply it.
142  auto it = intmods.find(intname);
143  if (it == intmods.end()) {
144  if (allowUnknownIntrinsics)
145  continue;
146  return op.emitError() << "intrinsic not recognized";
147  }
148  if (failed(it->second(op))) {
149  ++numFailures;
150  continue;
151  }
152  ++numConverted;
153  op.erase();
154  }
155 
156  return success(numFailures == 0);
157 }
