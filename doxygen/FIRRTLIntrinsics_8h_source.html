23 /// Base class for Intrinsic Converters.
24 ///
25 /// Intrinsic converters contain validation logic, along with a converter
26 /// method to transform instances to extmodules/intmodules into ops.
27 class IntrinsicConverter {
28 protected:
29  StringRef name;
30  FModuleLike mod;
31 
32 public:
33  IntrinsicConverter(StringRef name, FModuleLike mod) : name(name), mod(mod) {}
34 
35  virtual ~IntrinsicConverter();
36 
37  /// Checks whether the intrinsic module is well-formed.
38  ///
39  /// This or's multiple ParseResults together, returning true on failure.
40  virtual bool check() { return false; }
41 
42  /// Transform an instance of the intrinsic.
43  virtual LogicalResult convert(InstanceOp op) = 0;
44 
45 protected:
46  ParseResult hasNPorts(unsigned n);
47 
48  ParseResult namedPort(unsigned n, StringRef portName);
49 
50  ParseResult resetPort(unsigned n);
51 
52  ParseResult hasNParam(unsigned n, unsigned c = 0);
53 
54  ParseResult namedParam(StringRef paramName, bool optional = false);
55 
56  ParseResult namedIntParam(StringRef paramName, bool optional = false);
57 
58  template <typename T>
59  ParseResult typedPort(unsigned n) {
60  auto ports = mod.getPorts();
61  if (n >= ports.size()) {
62  mod.emitError(name) << " missing port " << n;
63  return failure();
64  }
65  if (!isa<T>(ports[n].type)) {
66  mod.emitError(name) << " port " << n << " not of correct type";
67  return failure();
68  }
69  return success();
70  }
71 
72  template <typename T>
73  ParseResult sizedPort(unsigned n, int32_t size) {
74  auto ports = mod.getPorts();
75  if (failed(typedPort<T>(n)))
76  return failure();
77  if (cast<T>(ports[n].type).getWidth() != size) {
78  mod.emitError(name) << " port " << n << " not size " << size;
79  return failure();
80  }
81  return success();
82  }
83 };

85 /// Lowering helper which collects all intrinsic converters.
86 class IntrinsicLowerings {
87 private:
88  using ConverterFn = std::function<LogicalResult(FModuleLike)>;
89 
90  /// Reference to the MLIR context.
91  MLIRContext *context;
92 
93  /// Reference to the instance graph to find module instances.
94  InstanceGraph &graph;
95 
96  /// Mapping from intrinsic names to converters.
97  DenseMap<StringAttr, ConverterFn> intmods;
98  /// Mapping from extmodule names to converters.
99  DenseMap<StringAttr, ConverterFn> extmods;
100 
101 public:
102  IntrinsicLowerings(MLIRContext *context, InstanceGraph &graph)
103  : context(context), graph(graph) {}
104 
105  /// Registers a converter to an intrinsic name.
106  template <typename T>
107  void add(StringRef name) {
108  addConverter<T>(intmods, name);
109  }
110 
111  /// Registers a converter to an extmodule name.
112  template <typename T>
113  void addExtmod(StringRef name) {
114  addConverter<T>(extmods, name);
115  }
116 
117  /// Registers a converter to multiple intrinsic names.
118  template <typename T, typename... Args>
119  void add(StringRef name, Args... args) {
120  add<T>(name);
121  add<T>(args...);
122  }
123 
124  /// Lowers a module to an intrinsic, given an intrinsic name.
125  LogicalResult lower(CircuitOp circuit, bool allowUnknownIntrinsics = false);
126 
127  /// Return the number of intrinsics converted.
128  unsigned getNumConverted() const { return numConverted; }
129 
130 private:
131  template <typename T>
132  void addConverter(DenseMap<StringAttr, ConverterFn> &map, StringRef name) {
133  auto nameAttr = StringAttr::get(context, name);
134  map.try_emplace(nameAttr,
135  [this, nameAttr](FModuleLike mod) -> LogicalResult {
136  T conv(nameAttr.getValue(), mod);
137  return doLowering(mod, conv);
138  });
139  }
140 
141  LogicalResult doLowering(FModuleLike mod, IntrinsicConverter &conv);
142 
143  unsigned numConverted = 0;
144 };
