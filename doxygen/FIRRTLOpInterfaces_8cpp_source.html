23 LogicalResult circt::firrtl::verifyModuleLikeOpInterface(FModuleLike module) {
24  // Verify port types first. This is used as the basis for the number of
25  // ports required everywhere else.
26  auto portTypes = module.getPortTypesAttr();
27  if (!portTypes)
28  return module.emitOpError("requires valid port types");
29  if (llvm::any_of(portTypes.getValue(), [](Attribute attr) {
30  auto typeAttr = attr.dyn_cast<TypeAttr>();
31  return !typeAttr || !typeAttr.getValue().isa<FIRRTLType>();
32  }))
33  return module.emitOpError("ports should all be FIRRTL types");
34 
35  auto numPorts = portTypes.size();
36 
37  // Verify the port dirctions.
38  auto portDirections = module.getPortDirectionsAttr();
39  if (!portDirections)
40  return module.emitOpError("requires valid port direction");
41  // TODO: bitwidth is 1 when there are no ports, since APInt previously did not
42  // support 0 bit widths.
43  auto bitWidth = portDirections.getValue().getBitWidth();
44  if (bitWidth != numPorts)
45  return module.emitOpError("requires ") << numPorts << " port directions";
46 
47  // Verify the port names.
48  auto portNames = module.getPortNamesAttr();
49  if (!portNames)
50  return module.emitOpError("requires valid port names");
51  if (portNames.size() != numPorts)
52  return module.emitOpError("requires ") << numPorts << " port names";
53  if (llvm::any_of(portNames.getValue(),
54  [](Attribute attr) { return !attr.isa<StringAttr>(); }))
55  return module.emitOpError("port names should all be string attributes");
56 
57  // Verify the port annotations.
58  auto portAnnotations = module.getPortAnnotationsAttr();
59  if (!portAnnotations)
60  return module.emitOpError("requires valid port annotations");
61  // TODO: it seems weird to allow empty port annotations.
62  if (!portAnnotations.empty() && portAnnotations.size() != numPorts)
63  return module.emitOpError("requires ") << numPorts << " port annotations";
64  // TODO: Move this into an annotation verifier.
65  for (auto annos : portAnnotations.getValue()) {
66  auto arrayAttr = annos.dyn_cast<ArrayAttr>();
67  if (!arrayAttr)
68  return module.emitOpError(
69  "requires port annotations be array attributes");
70  if (llvm::any_of(arrayAttr.getValue(), [](Attribute attr) {
71  return !attr.isa<DictionaryAttr>();
72  }))
73  return module.emitOpError(
74  "annotations must be dictionaries or subannotations");
75  }
76 
77  // Verify the port symbols.
78  auto portSymbols = module.getPortSymbolsAttr();
79  if (!portSymbols)
80  return module.emitOpError("requires valid port symbols");
81  if (!portSymbols.empty() && portSymbols.size() != numPorts)
82  return module.emitOpError("requires ") << numPorts << " port symbols";
83  if (llvm::any_of(portSymbols.getValue(),
84  [](Attribute attr) { return !attr.isa<StringAttr>(); }))
85  return module.emitOpError("port symbols should all be string attributes");
86 
87  // Verify the body.
88  if (module->getNumRegions() != 1)
89  return module.emitOpError("requires one region");
90 
91  // Verify the block arguments.
92  auto &body = module->getRegion(0);
93  if (!body.empty()) {
94  auto &block = body.front();
95  if (block.getNumArguments() != numPorts)
96  return module.emitOpError("entry block must have ")
97  << numPorts << " arguments to match module signature";
98 
99  if (llvm::any_of(
100  llvm::zip(block.getArguments(), portTypes.getValue()),
101  [](auto pair) {
102  auto blockType = std::get<0>(pair).getType();
103  auto portType =
104  std::get<1>(pair).template cast<TypeAttr>().getValue();
105  return blockType != portType;
106  }))
107  return module.emitOpError(
108  "block argument types should match signature types");
109  }
110 
111  return success();
112 }
