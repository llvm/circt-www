34 /// This holds the name and type that describes the module's ports.
35 struct PortInfo {
36  StringAttr name;
37  FIRRTLType type;
38  Direction direction;
39  StringAttr sym = {};
40  Location loc = UnknownLoc::get(type.getContext());
41  AnnotationSet annotations = AnnotationSet(type.getContext());
42 
43  StringRef getName() const { return name ? name.getValue() : ""; }
44 
45  /// Return true if this is a simple output-only port. If you want the
46  /// direction of the port, use the \p direction parameter.
47  bool isOutput() {
48  auto flags = type.getRecursiveTypeProperties();
49  return flags.isPassive && !flags.containsAnalog &&
50  direction == Direction::Out;
51  }
52 
53  /// Return true if this is a simple input-only port. If you want the
54  /// direction of the port, use the \p direction parameter.
55  bool isInput() {
56  auto flags = type.getRecursiveTypeProperties();
57  return flags.isPassive && !flags.containsAnalog &&
58  direction == Direction::In;
59  }
60 
61  /// Return true if this is an inout port. This will be true if the port
62  /// contains either bi-directional signals or analog types.
63  bool isInOut() { return !isOutput() && !isInput(); }
64 };

79 /// This trait is for operations that define a scope for resolving InnerRef's,
80 /// and provides verification for InnerRef users (via InnerRefUserOpInterface).
81 template <typename ConcreteType>
82 class InnerRefNamespace : public TraitBase<ConcreteType, InnerRefNamespace> {
83 public:
84  static LogicalResult verifyRegionTrait(Operation *op) {
85  static_assert(
86  ConcreteType::template hasTrait<::mlir::OpTrait::SymbolTable>(),
87  "expected operation to be a SymbolTable");
88 
89  if (op->getNumRegions() != 1)
90  return op->emitError("expected operation to have a single region");
91  if (!op->getRegion(0).hasOneBlock())
92  return op->emitError("expected operation to have a single block");
93 
94  // Verify all InnerRef users.
95  return ::circt::firrtl::detail::verifyInnerRefs(op);
96  }
97 };

99 /// A trait for inner symbol table functionality on an operation.
100 template <typename ConcreteType>
101 class InnerSymbolTable : public TraitBase<ConcreteType, InnerSymbolTable> {
102 public:
103  static LogicalResult verifyRegionTrait(Operation *op) {
104  // Insist that ops with InnerSymbolTable's provide a Symbol, this is
105  // essential to how InnerRef's work.
106  static_assert(
107  ConcreteType::template hasTrait<::mlir::SymbolOpInterface::Trait>(),
108  "expected operation to define a Symbol");
109 
110  if (op->getNumRegions() != 1)
111  return op->emitError("expected operation to have a single region");
112  if (!op->getRegion(0).hasOneBlock())
113  return op->emitError("expected operation to have a single block");
114 
115  // InnerSymbolTable's must be directly nested within an InnerRefNamespace.
116  auto *parent = op->getParentOp();
117  if (!parent || !parent->hasTrait<InnerRefNamespace>())
118  return op->emitError(
119  "InnerSymbolTable must have InnerRefNamespace parent");
120 
121  return success();
122  }
123 };
