44 /// Remove elements at the specified indices from the input array, returning the
45 /// elements not mentioned. The indices array is expected to be sorted and
46 /// unique.
47 template <typename T>
48 static SmallVector<T>
49 removeElementsAtIndices(ArrayRef<T> input, ArrayRef<unsigned> indicesToDrop) {
50 #ifndef NDEBUG // Check sortedness.
51  if (!input.empty()) {
52  for (size_t i = 1, e = indicesToDrop.size(); i != e; ++i)
53  assert(indicesToDrop[i - 1] < indicesToDrop[i] &&
54  "indicesToDrop isn't sorted and unique");
55  assert(indicesToDrop.back() < input.size() && "index out of range");
56  }
57 #endif
58 
59  // If the input is empty (which is an optimization we do for certain array
60  // attributes), simply return an empty vector.
61  if (input.empty())
62  return {};
63 
64  // Copy over the live chunks.
65  size_t lastCopied = 0;
66  SmallVector<T> result;
67  result.reserve(input.size() - indicesToDrop.size());
68 
69  for (unsigned indexToDrop : indicesToDrop) {
70  // If we skipped over some valid elements, copy them over.
71  if (indexToDrop > lastCopied) {
72  result.append(input.begin() + lastCopied, input.begin() + indexToDrop);
73  lastCopied = indexToDrop;
74  }
75  // Ignore this value so we don't copy it in the next iteration.
76  ++lastCopied;
77  }
78 
79  // If there are live elements at the end, copy them over.
80  if (lastCopied < input.size())
81  result.append(input.begin() + lastCopied, input.end());
82 
83  return result;
84 }

110 Flow firrtl::foldFlow(Value val, Flow accumulatedFlow) {
111  auto swap = [&accumulatedFlow]() -> Flow {
112  return swapFlow(accumulatedFlow);
113  };
114 
115  if (auto blockArg = val.dyn_cast<BlockArgument>()) {
116  auto op = val.getParentBlock()->getParentOp();
117  auto direction =
118  cast<FModuleLike>(op).getPortDirection(blockArg.getArgNumber());
119  if (direction == Direction::Out)
120  return swap();
121  return accumulatedFlow;
122  }
123 
124  Operation *op = val.getDefiningOp();
125 
126  return TypeSwitch<Operation *, Flow>(op)
127  .Case<SubfieldOp>([&](auto op) {
128  return foldFlow(op.input(),
129  op.isFieldFlipped() ? swap() : accumulatedFlow);
130  })
131  .Case<SubindexOp, SubaccessOp>(
132  [&](auto op) { return foldFlow(op.input(), accumulatedFlow); })
133  // Registers, Wires, and behavioral memory ports are always Duplex.
134  .Case<RegOp, RegResetOp, WireOp, MemoryPortOp>(
135  [](auto) { return Flow::Duplex; })
136  .Case<InstanceOp>([&](auto inst) {
137  auto resultNo = val.cast<OpResult>().getResultNumber();
138  if (inst.getPortDirection(resultNo) == Direction::Out)
139  return accumulatedFlow;
140  return swap();
141  })
142  .Case<MemOp>([&](auto op) { return swap(); })
143  // Anything else acts like a universal source.
144  .Default([&](auto) { return accumulatedFlow; });
145 }

263 LogicalResult CircuitOp::verify() {
264  StringRef main = name();
265 
266  // Check that the circuit has a non-empty name.
267  if (main.empty()) {
268  emitOpError("must have a non-empty name");
269  return failure();
270  }
271 
272  // Check that a module matching the "main" module exists in the circuit.
273  auto mainModule = getMainModule();
274  if (!mainModule) {
275  emitOpError("must contain one module that matches main name '" + main +
276  "'");
277  return failure();
278  }
279 
280  // Check that the main module is public.
281  if (!cast<hw::HWModuleLike>(*mainModule).isPublic()) {
282  emitOpError("main module '" + main + "' must be public");
283  return failure();
284  }

286  // Store a mapping of defname to either the first external module
287  // that defines it or, preferentially, the first external module
288  // that defines it and has no parameters.
289  llvm::DenseMap<Attribute, FExtModuleOp> defnameMap;
290 
291  auto verifyExtModule = [&](FExtModuleOp extModule) {
292  if (!extModule)
293  return success();
294 
295  auto defname = extModule.defnameAttr();
296  if (!defname)
297  return success();
298 
299  // Check that this extmodule's defname does not conflict with
300  // the symbol name of any module.
301  auto *collidingModule = lookupSymbol(defname.getValue());
302  if (isa_and_nonnull<FModuleOp>(collidingModule)) {
303  auto diag =
304  extModule.emitOpError()
305  << "attribute 'defname' with value " << defname
306  << " conflicts with the name of another module in the circuit";
307  diag.attachNote(collidingModule->getLoc())
308  << "previous module declared here";
309  return failure();
310  }
311 
312  // Find an optional extmodule with a defname collision. Update
313  // the defnameMap if this is the first extmodule with that
314  // defname or if the current extmodule takes no parameters and
315  // the collision does. The latter condition improves later
316  // extmodule verification as checking against a parameterless
317  // module is stricter.
318  FExtModuleOp collidingExtModule;
319  if (auto &value = defnameMap[defname]) {
320  collidingExtModule = value;
321  if (!value.parameters().empty() && extModule.parameters().empty())
322  value = extModule;
323  } else {
324  value = extModule;
325  // Go to the next extmodule if no extmodule with the same
326  // defname was found.
327  return success();
328  }
329 
330  // Check that the number of ports is exactly the same.
331  SmallVector<PortInfo> ports = extModule.getPorts();
332  SmallVector<PortInfo> collidingPorts = collidingExtModule.getPorts();
333 
334  if (ports.size() != collidingPorts.size()) {
335  auto diag = extModule.emitOpError()
336  << "with 'defname' attribute " << defname << " has "
337  << ports.size()
338  << " ports which is different from a previously defined "
339  "extmodule with the same 'defname' which has "
340  << collidingPorts.size() << " ports";
341  diag.attachNote(collidingExtModule.getLoc())
342  << "previous extmodule definition occurred here";
343  return failure();
344  }
345 
346  // Check that ports match for name and type. Since parameters
347  // *might* affect widths, ignore widths if either module has
348  // parameters. Note that this allows for misdetections, but
349  // has zero false positives.
350  for (auto p : llvm::zip(ports, collidingPorts)) {
351  StringAttr aName = std::get<0>(p).name, bName = std::get<1>(p).name;
352  FIRRTLType aType = std::get<0>(p).type, bType = std::get<1>(p).type;
353 
354  if (!extModule.parameters().empty() ||
355  !collidingExtModule.parameters().empty()) {
356  aType = aType.getWidthlessType();
357  bType = bType.getWidthlessType();
358  }
359  if (aName != bName) {
360  auto diag = extModule.emitOpError()
361  << "with 'defname' attribute " << defname
362  << " has a port with name " << aName
363  << " which does not match the name of the port "
364  << "in the same position of a previously defined "
365  << "extmodule with the same 'defname', expected port "
366  "to have name "
367  << bName;
368  diag.attachNote(collidingExtModule.getLoc())
369  << "previous extmodule definition occurred here";
370  return failure();
371  }
372  if (aType != bType) {
373  auto diag = extModule.emitOpError()
374  << "with 'defname' attribute " << defname
375  << " has a port with name " << aName
376  << " which has a different type " << aType
377  << " which does not match the type of the port in "
378  "the same position of a previously defined "
379  "extmodule with the same 'defname', expected port "
380  "to have type "
381  << bType;
382  diag.attachNote(collidingExtModule.getLoc())
383  << "previous extmodule definition occurred here";
384  return failure();
385  }
386  }
387  return success();
388  };
389 
390  for (auto &op : *getBody()) {
391  // Verify external modules.
392  if (auto extModule = dyn_cast<FExtModuleOp>(op)) {
393  if (verifyExtModule(extModule).failed())
394  return failure();
395  }
396  }
397 
398  return success();
399 }

446 /// Inserts the given ports. The insertion indices are expected to be in order.
447 /// Insertion occurs in-order, such that ports with the same insertion index
448 /// appear in the module in the same order they appeared in the list.
449 void FModuleOp::insertPorts(ArrayRef<std::pair<unsigned, PortInfo>> ports) {
450  if (ports.empty())
451  return;
452  unsigned oldNumArgs = getNumPorts();
453  unsigned newNumArgs = oldNumArgs + ports.size();
454 
455  auto *body = getBody();
456 
457  // Add direction markers and names for new ports.
458  SmallVector<Direction> existingDirections =
459  direction::unpackAttribute(this->getPortDirectionsAttr());
460  ArrayRef<Attribute> existingNames = this->getPortNames();
461  ArrayRef<Attribute> existingTypes = this->getPortTypes();
462  assert(existingDirections.size() == oldNumArgs);
463  assert(existingNames.size() == oldNumArgs);
464  assert(existingTypes.size() == oldNumArgs);
465 
466  SmallVector<Direction> newDirections;
467  SmallVector<Attribute> newNames;
468  SmallVector<Attribute> newTypes;
469  SmallVector<Attribute> newAnnos;
470  SmallVector<Attribute> newSyms;
471  newDirections.reserve(newNumArgs);
472  newNames.reserve(newNumArgs);
473  newTypes.reserve(newNumArgs);
474  newAnnos.reserve(newNumArgs);
475  newSyms.reserve(newNumArgs);
476 
477  auto emptyArray = ArrayAttr::get(getContext(), {});
478  auto emptyString = StringAttr::get(getContext(), "");
479 
480  unsigned oldIdx = 0;
481  auto migrateOldPorts = [&](unsigned untilOldIdx) {
482  while (oldIdx < oldNumArgs && oldIdx < untilOldIdx) {
483  newDirections.push_back(existingDirections[oldIdx]);
484  newNames.push_back(existingNames[oldIdx]);
485  newTypes.push_back(existingTypes[oldIdx]);
486  newAnnos.push_back(getAnnotationsAttrForPort(oldIdx));
487  newSyms.push_back(getPortSymbolAttr(oldIdx));
488  ++oldIdx;
489  }
490  };
491  for (auto &pair : llvm::enumerate(ports)) {
492  auto idx = pair.value().first;
493  auto &port = pair.value().second;
494  auto newIdx = pair.index();
495  migrateOldPorts(idx);
496  newDirections.push_back(port.direction);
497  newNames.push_back(port.name);
498  newTypes.push_back(TypeAttr::get(port.type));
499  auto annos = port.annotations.getArrayAttr();
500  newAnnos.push_back(annos ? annos : emptyArray);
501  newSyms.push_back(port.sym ? port.sym : emptyString);
502  // Block arguments are inserted one at a time, so for each argument we
503  // insert we have to increase the index by 1.
504  body->insertArgument(idx + newIdx, port.type, port.loc);
505  }
506  migrateOldPorts(oldNumArgs);
507 
508  // The lack of *any* port annotations is represented by an empty
509  // `portAnnotations` array as a shorthand.
510  if (llvm::all_of(newAnnos, [](Attribute attr) {
511  return attr.cast<ArrayAttr>().empty();
512  }))
513  newAnnos.clear();
514 
515  // The lack of *any* port symbol is represented by an empty `portSyms` array
516  // as a shorthand.
517  if (llvm::all_of(newSyms, [](Attribute attr) {
518  return attr.cast<StringAttr>().getValue().empty();
519  }))
520  newSyms.clear();
521 
522  // Apply these changed markers.
523  (*this)->setAttr("portDirections",
524  direction::packAttribute(getContext(), newDirections));
525  (*this)->setAttr("portNames", ArrayAttr::get(getContext(), newNames));
526  (*this)->setAttr("portTypes", ArrayAttr::get(getContext(), newTypes));
527  (*this)->setAttr("portAnnotations", ArrayAttr::get(getContext(), newAnnos));
528  (*this)->setAttr("portSyms", ArrayAttr::get(getContext(), newSyms));
529 }

618  ArrayRef<Attribute> portTypes = this->getPortTypes();

622  assert(portNames.size() == getNumPorts());
623  assert(portAnnos.size() == getNumPorts() || portAnnos.empty());
624  assert(portTypes.size() == getNumPorts());
625  assert(portSyms.size() == getNumPorts() || portSyms.empty());
626 
627  SmallVector<Direction> newPortDirections =
628  removeElementsAtIndices<Direction>(portDirections, portIndices);
629  SmallVector<Attribute> newPortNames =
630  removeElementsAtIndices(portNames, portIndices);
631  SmallVector<Attribute> newPortTypes =
632  removeElementsAtIndices(portTypes, portIndices);
633  SmallVector<Attribute> newPortAnnos =
634  removeElementsAtIndices(portAnnos, portIndices);
635  SmallVector<Attribute> newPortSyms =
636  removeElementsAtIndices(portSyms, portIndices);

638  direction::packAttribute(getContext(), newPortDirections));
639  (*this)->setAttr("portNames", ArrayAttr::get(getContext(), newPortNames));
640  (*this)->setAttr("portAnnotations",
641  ArrayAttr::get(getContext(), newPortAnnos));
642  (*this)->setAttr("portTypes", ArrayAttr::get(getContext(), newPortTypes));
643  (*this)->setAttr("portSyms", ArrayAttr::get(getContext(), newPortSyms));

646  getBody()->eraseArguments(portIndices);

649 static void buildModule(OpBuilder &builder, OperationState &result,
650  StringAttr name, ArrayRef<PortInfo> ports,
651  ArrayAttr annotations) {
652  // Add an attribute for the name.
653  result.addAttribute(::mlir::SymbolTable::getSymbolAttrName(), name);
654 
655  // Record the names of the arguments if present.
656  SmallVector<Direction, 4> portDirections;
657  SmallVector<Attribute, 4> portNames;
658  SmallVector<Attribute, 4> portTypes;
659  SmallVector<Attribute, 4> portAnnotations;
660  SmallVector<Attribute, 4> portSyms;
661  for (size_t i = 0, e = ports.size(); i != e; ++i) {
662  portDirections.push_back(ports[i].direction);
663  portNames.push_back(ports[i].name);
664  portTypes.push_back(TypeAttr::get(ports[i].type));
665  portAnnotations.push_back(ports[i].annotations.getArrayAttr());
666  portSyms.push_back(ports[i].sym ? ports[i].sym : builder.getStringAttr(""));
667  }
668 
669  // The lack of *any* port annotations is represented by an empty
670  // `portAnnotations` array as a shorthand.
671  if (llvm::all_of(portAnnotations, [](Attribute attr) {
672  return attr.cast<ArrayAttr>().empty();
673  }))
674  portAnnotations.clear();
675 
676  // The lack of *any* port symbol is represented by an empty `portSyms` array
677  // as a shorthand.
678  if (llvm::all_of(portSyms, [](Attribute attr) {
679  return attr.cast<StringAttr>().getValue().empty();
680  }))
681  portSyms.clear();
682 
683  // Both attributes are added, even if the module has no ports.
684  result.addAttribute(
685  "portDirections",
686  direction::packAttribute(builder.getContext(), portDirections));
687  result.addAttribute("portNames", builder.getArrayAttr(portNames));
688  result.addAttribute("portTypes", builder.getArrayAttr(portTypes));
689  result.addAttribute("portAnnotations", builder.getArrayAttr(portAnnotations));
690  result.addAttribute("portSyms", builder.getArrayAttr(portSyms));
691 
692  if (!annotations)
693  annotations = builder.getArrayAttr({});
694  result.addAttribute("annotations", annotations);
695 
696  result.addRegion();
697 }

749 /// Print a list of module ports in the following form:
750 /// in x: !firrtl.uint<1> [{class = "DontTouch}], out "_port": !firrtl.uint<2>
751 ///
752 /// When there is no block specified, the port names print as MLIR identifiers,
753 /// wrapping in quotes if not legal to print as-is. When there is no block
754 /// specified, this function always return false, indicating that there was no
755 /// issue printing port names.
756 ///
757 /// If there is a block specified, then port names will be printed as SSA
758 /// values. If there is a reason the printed SSA values can't match the true
759 /// port name, then this function will return true. When this happens, the
760 /// caller should print the port names as a part of the `attr-dict`.
761 static bool printModulePorts(OpAsmPrinter &p, Block *block,
762  ArrayRef<Direction> portDirections,
763  ArrayRef<Attribute> portNames,
764  ArrayRef<Attribute> portTypes,
765  ArrayRef<Attribute> portAnnotations,
766  ArrayRef<Attribute> portSyms) {
767  // When printing port names as SSA values, we can fail to print them
768  // identically.
769  bool printedNamesDontMatch = false;
770 
771  // If we are printing the ports as block arguments the op must have a first
772  // block.
773  SmallString<32> resultNameStr;
774  p << '(';
775  for (unsigned i = 0, e = portTypes.size(); i < e; ++i) {
776  if (i > 0)
777  p << ", ";
778 
779  // Print the port direction.
780  p << portDirections[i] << " ";
781 
782  // Print the port name. If there is a valid block, we print it as a block
783  // argument.
784  if (block) {
785  // Get the printed format for the argument name.
786  resultNameStr.clear();
787  llvm::raw_svector_ostream tmpStream(resultNameStr);
788  p.printOperand(block->getArgument(i), tmpStream);
789  // If the name wasn't printable in a way that agreed with portName, make
790  // sure to print out an explicit portNames attribute.
791  auto portName = portNames[i].cast<StringAttr>().getValue();
792  if (!portName.empty() && tmpStream.str().drop_front() != portName)
793  printedNamesDontMatch = true;
794  p << tmpStream.str();
795  } else {
796  p.printKeywordOrString(portNames[i].cast<StringAttr>().getValue());
797  }
798 
799  // Print the port type.
800  p << ": ";
801  auto portType = portTypes[i].cast<TypeAttr>().getValue();
802  p.printType(portType);
803 
804  // Print the optional port symbol.
805  if (!portSyms.empty()) {
806  auto symValue = portSyms[i].cast<StringAttr>().getValue();
807  if (!symValue.empty()) {
808  p << " sym ";
809  p.printSymbolName(symValue);
810  }
811  }
812 
813  // Print the port specific annotations. The port annotations array will be
814  // empty if there are none.
815  if (!portAnnotations.empty() &&
816  !portAnnotations[i].cast<ArrayAttr>().empty()) {
817  p << " ";
818  p.printAttribute(portAnnotations[i]);
819  }
820  }
821 
822  p << ')';
823  return printedNamesDontMatch;
824 }

826 /// Parse a list of module ports. If port names are SSA identifiers, then this
827 /// will populate `entryArgs`.
828 static ParseResult
829 parseModulePorts(OpAsmParser &parser, bool hasSSAIdentifiers,
830  SmallVectorImpl<OpAsmParser::Argument> &entryArgs,
831  SmallVectorImpl<Direction> &portDirections,
832  SmallVectorImpl<Attribute> &portNames,
833  SmallVectorImpl<Attribute> &portTypes,
834  SmallVectorImpl<Attribute> &portAnnotations,
835  SmallVectorImpl<Attribute> &portSyms) {
836  auto *context = parser.getContext();
837 
838  auto parseArgument = [&]() -> ParseResult {
839  // Parse port direction.
840  if (succeeded(parser.parseOptionalKeyword("out")))
841  portDirections.push_back(Direction::Out);
842  else if (succeeded(parser.parseKeyword("in", "or 'out'")))
843  portDirections.push_back(Direction::In);
844  else
845  return failure();
846 
847  // Parse the port name.
848  if (hasSSAIdentifiers) {
849  OpAsmParser::Argument arg;
850  if (parser.parseArgument(arg))
851  return failure();
852  entryArgs.push_back(arg);
853  // The name of an argument is of the form "%42" or "%id", and since
854  // parsing succeeded, we know it always has one character.
855  assert(arg.ssaName.name.size() > 1 && arg.ssaName.name[0] == '%' &&
856  "Unknown MLIR name");
857  if (isdigit(arg.ssaName.name[1]))
858  portNames.push_back(StringAttr::get(context, ""));
859  else
860  portNames.push_back(
861  StringAttr::get(context, arg.ssaName.name.drop_front()));
862  } else {
863  std::string portName;
864  if (parser.parseKeywordOrString(&portName))
865  return failure();
866  portNames.push_back(StringAttr::get(context, portName));
867  }
868 
869  // Parse the port type.
870  Type portType;
871  if (parser.parseColonType(portType))
872  return failure();
873  portTypes.push_back(TypeAttr::get(portType));
874 
875  if (hasSSAIdentifiers)
876  entryArgs.back().type = portType;
877 
878  // Parse the optional port symbol.
879  StringAttr portSym;
880  if (succeeded(parser.parseOptionalKeyword("sym"))) {
881  NamedAttrList dummyAttrs;
882  if (parser.parseSymbolName(portSym, "dummy", dummyAttrs))
883  return failure();
884  } else {
885  portSym = StringAttr::get(context, "");
886  }
887  portSyms.push_back(portSym);
888 
889  // Parse the port annotations.
890  ArrayAttr annos;
891  auto parseResult = parser.parseOptionalAttribute(annos);
892  if (!parseResult.hasValue())
893  annos = parser.getBuilder().getArrayAttr({});
894  else if (failed(*parseResult))
895  return failure();
896  portAnnotations.push_back(annos);
897 
898  return success();
899  };
900 
901  // Parse all ports.
902  return parser.parseCommaSeparatedList(OpAsmParser::Delimiter::Paren,
903  parseArgument);
904 }

1015 static ParseResult parseFModuleLikeOp(OpAsmParser &parser,
1016  OperationState &result,
1017  bool hasSSAIdentifiers) {
1018  auto *context = result.getContext();
1019  auto &builder = parser.getBuilder();
1020 
1021  // Parse the visibility attribute.
1022  (void)mlir::impl::parseOptionalVisibilityKeyword(parser, result.attributes);
1023 
1024  // Parse the name as a symbol.
1025  StringAttr nameAttr;
1026  if (parser.parseSymbolName(nameAttr, ::mlir::SymbolTable::getSymbolAttrName(),
1027  result.attributes))
1028  return failure();
1029 
1030  // Parse optional parameters.
1031  SmallVector<Attribute, 4> parameters;
1032  if (parseOptionalParameters(parser, parameters))
1033  return failure();
1034  result.addAttribute("parameters", builder.getArrayAttr(parameters));
1035 
1036  // Parse the module ports.
1037  SmallVector<OpAsmParser::Argument> entryArgs;
1038  SmallVector<Direction, 4> portDirections;
1039  SmallVector<Attribute, 4> portNames;
1040  SmallVector<Attribute, 4> portTypes;
1041  SmallVector<Attribute, 4> portAnnotations;
1042  SmallVector<Attribute, 4> portSyms;
1043  if (parseModulePorts(parser, hasSSAIdentifiers, entryArgs, portDirections,
1044  portNames, portTypes, portAnnotations, portSyms))
1045  return failure();
1046 
1047  // If module attributes are present, parse them.
1048  if (parser.parseOptionalAttrDictWithKeyword(result.attributes))
1049  return failure();
1050 
1051  assert(portNames.size() == portTypes.size());
1052 
1053  // Record the argument and result types as an attribute. This is necessary
1054  // for external modules.
1055 
1056  // Add port directions.
1057  if (!result.attributes.get("portDirections"))
1058  result.addAttribute("portDirections",
1059  direction::packAttribute(context, portDirections));
1060 
1061  // Add port names.
1062  if (!result.attributes.get("portNames")) {
1063  result.addAttribute("portNames", builder.getArrayAttr(portNames));
1064  }
1065 
1066  // Add the port types.
1067  if (!result.attributes.get("portTypes"))
1068  result.addAttribute("portTypes", ArrayAttr::get(context, portTypes));
1069 
1070  // Add the port annotations.
1071  if (!result.attributes.get("portAnnotations")) {
1072  // If there are no portAnnotations, don't add the attribute.
1073  if (llvm::any_of(portAnnotations, [&](Attribute anno) {
1074  return !anno.cast<ArrayAttr>().empty();
1075  }))
1076  result.addAttribute("portAnnotations",
1077  ArrayAttr::get(context, portAnnotations));
1078  }
1079 
1080  // Add port symbols.
1081  if (!result.attributes.get("portSyms")) {
1082  result.addAttribute("portSyms", builder.getArrayAttr(portSyms));
1083  }
1084 
1085  // The annotations attribute is always present, but not printed when empty.
1086  if (!result.attributes.get("annotations"))
1087  result.addAttribute("annotations", builder.getArrayAttr({}));
1088 
1089  // The portAnnotations attribute is always present, but not printed when
1090  // empty.
1091  if (!result.attributes.get("portAnnotations"))
1092  result.addAttribute("portAnnotations", builder.getArrayAttr({}));
1093 
1094  // Parse the optional function body.
1095  auto *body = result.addRegion();
1096 
1097  if (hasSSAIdentifiers) {
1098  if (parser.parseRegion(*body, entryArgs))
1099  return failure();
1100  if (body->empty())
1101  body->push_back(new Block());
1102  }
1103  return success();
1104 }

1352  auto module = (*this)->getParentOfType<FModuleOp>();

1355  if (!referencedModule) {

1376  // lets us safely use the port name in error messages below.
1377  size_t numResults = getNumResults();

1396  auto instanceNames = portNames();
1397  auto moduleNames = referencedModule.getPortNamesAttr();
1398  // First compare the sizes:
1399  if (instanceNames.size() != moduleNames.size()) {
1400  return emitNote(emitOpError()

1402  << moduleNames.size() << " but got "
1403  << instanceNames.size());
1404  }
1405  // Next check the values:
1406  for (size_t i = 0; i != numResults; ++i) {
1407  if (instanceNames[i] != moduleNames[i]) {
1408  return emitNote(emitOpError()
1409  << "name for port " << i << " must be "
1410  << moduleNames[i] << ", but got " << instanceNames[i]);
1411  }
1412  }
1413  llvm_unreachable("should have found something wrong");
1414  }
1415 
1416  // Check that the types match.
1417  for (size_t i = 0; i != numResults; i++) {
1418  auto resultType = getResult(i).getType();
1419  auto expectedType = referencedModule.getPortType(i);
1420  if (resultType != expectedType) {
1421  return emitNote(emitOpError()
1422  << "result type for " << getPortName(i) << " must be "
1423  << expectedType << ", but got " << resultType);
1424  }
1425  }
1426 
1427  // Check that the port directions are consistent with the referenced module's.
1428  if (portDirectionsAttr() != referencedModule.getPortDirectionsAttr()) {
1429  // We know there is an error, try to figure out whats wrong.
1430  auto instanceDirectionAttr = portDirectionsAttr();
1431  auto moduleDirectionAttr = referencedModule.getPortDirectionsAttr();
1432  // First compare the sizes:
1433  auto expectedWidth = moduleDirectionAttr.getValue().getBitWidth();
1434  auto actualWidth = instanceDirectionAttr.getValue().getBitWidth();
1435  if (expectedWidth != actualWidth) {
1436  return emitNote(emitOpError()
1437  << "has a wrong number of directions; expected "
1438  << expectedWidth << " but got " << actualWidth);
1439  }
1440  // Next check the values.
1441  auto instanceDirs = direction::unpackAttribute(instanceDirectionAttr);
1442  auto moduleDirs = direction::unpackAttribute(moduleDirectionAttr);
1443  for (size_t i = 0; i != numResults; ++i) {
1444  if (instanceDirs[i] != moduleDirs[i]) {
1445  return emitNote(emitOpError()
1446  << "direction for " << getPortName(i) << " must be \""
1447  << direction::toString(moduleDirs[i])
1448  << "\", but got \""
1449  << direction::toString(instanceDirs[i]) << "\"");
1450  }
1451  }
1452  llvm_unreachable("should have found something wrong");
1453  }
1454 
1455  return success();
1456 }

1644 /// Verify the correctness of a MemOp.
1645 LogicalResult MemOp::verify() {
1646 
1647  // Store the port names as we find them. This lets us check quickly
1648  // for uniqueneess.
1649  llvm::SmallDenseSet<Attribute, 8> portNamesSet;
1650 
1651  // Store the previous data type. This lets us check that the data
1652  // type is consistent across all ports.
1653  FIRRTLType oldDataType;
1654 
1655  for (size_t i = 0, e = getNumResults(); i != e; ++i) {
1656  auto portName = getPortName(i);
1657 
1658  // Get a bundle type representing this port, stripping an outer
1659  // flip if it exists. If this is not a bundle<> or
1660  // flip<bundle<>>, then this is an error.
1661  BundleType portBundleType =
1662  TypeSwitch<FIRRTLType, BundleType>(
1663  getResult(i).getType().cast<FIRRTLType>())
1664  .Case<BundleType>([](BundleType a) { return a; })
1665  .Default([](auto) { return nullptr; });
1666  if (!portBundleType) {
1667  emitOpError() << "has an invalid type on port " << portName
1668  << " (expected '!firrtl.bundle<...>')";
1669  return failure();
1670  }
1671 
1672  // Require that all port names are unique.
1673  if (!portNamesSet.insert(portName).second) {
1674  emitOpError() << "has non-unique port name " << portName;
1675  return failure();
1676  }
1677 
1678  // Determine the kind of the memory. If the kind cannot be
1679  // determined, then it's indicative of the wrong number of fields
1680  // in the type (but we don't know any more just yet).
1681  MemOp::PortKind portKind;
1682  {
1683  auto elt = getPortNamed(portName);
1684  if (!elt) {
1685  emitOpError() << "could not get port with name " << portName;
1686  return failure();
1687  }
1688  auto firrtlType = elt.getType().cast<FIRRTLType>();
1689  auto portType = firrtlType.dyn_cast<BundleType>();
1690  switch (portType.getNumElements()) {
1691  case 4:
1692  portKind = MemOp::PortKind::Read;
1693  break;
1694  case 5:
1695  portKind = MemOp::PortKind::Write;
1696  break;
1697  case 7:
1698  portKind = MemOp::PortKind::ReadWrite;
1699  break;
1700  default:
1701  emitOpError()
1702  << "has an invalid number of fields on port " << portName
1703  << " (expected 4 for read, 5 for write, or 7 for read/write)";
1704  return failure();
1705  }
1706  }
1707 
1708  // Safely search for the "data" field, erroring if it can't be
1709  // found.
1710  FIRRTLType dataType;
1711  {
1712  auto dataTypeOption = portBundleType.getElement("data");
1713  if (!dataTypeOption && portKind == MemOp::PortKind::ReadWrite)
1714  dataTypeOption = portBundleType.getElement("wdata");
1715  if (!dataTypeOption) {
1716  emitOpError() << "has no data field on port " << portName
1717  << " (expected to see \"data\" for a read or write "
1718  "port or \"rdata\" for a read/write port)";
1719  return failure();
1720  }
1721  dataType = dataTypeOption.getValue().type;
1722  // Read data is expected to ba a flip.
1723  if (portKind == MemOp::PortKind::Read) {
1724  // FIXME error on missing bundle flip
1725  }
1726  }
1727 
1728  // Error if the data type isn't passive.
1729  if (!dataType.isPassive()) {
1730  emitOpError() << "has non-passive data type on port " << portName
1731  << " (memory types must be passive)";
1732  return failure();
1733  }
1734 
1735  // Error if the data type contains analog types.
1736  if (dataType.containsAnalog()) {
1737  emitOpError() << "has a data type that contains an analog type on port "
1738  << portName
1739  << " (memory types cannot contain analog types)";
1740  return failure();
1741  }
1742 
1743  // Check that the port type matches the kind that we determined
1744  // for this port. This catches situations of extraneous port
1745  // fields beind included or the fields being named incorrectly.
1746  FIRRTLType expectedType = getTypeForPort(
1747  depth(), dataType, portKind, dataType.isGround() ? getMaskBits() : 0);
1748  // Compute the original port type as portBundleType may have
1749  // stripped outer flip information.
1750  auto originalType = getResult(i).getType();
1751  if (originalType != expectedType) {
1752  StringRef portKindName;
1753  switch (portKind) {
1754  case MemOp::PortKind::Read:
1755  portKindName = "read";
1756  break;
1757  case MemOp::PortKind::Write:
1758  portKindName = "write";
1759  break;
1760  case MemOp::PortKind::ReadWrite:
1761  portKindName = "readwrite";
1762  break;
1763  }
1764  emitOpError() << "has an invalid type for port " << portName
1765  << " of determined kind \"" << portKindName
1766  << "\" (expected " << expectedType << ", but got "
1767  << originalType << ")";
1768  return failure();
1769  }
1770 
1771  // Error if the type of the current port was not the same as the
1772  // last port, but skip checking the first port.
1773  if (oldDataType && oldDataType != dataType) {
1774  emitOpError() << "port " << getPortName(i)
1775  << " has a different type than port " << getPortName(i - 1)
1776  << " (expected " << oldDataType << ", but got " << dataType
1777  << ")";
1778  return failure();
1779  }
1780 
1781  oldDataType = dataType;
1782  }
1783 
1784  auto maskWidth = getMaskBits();
1785 
1786  auto dataWidth = getDataType().getBitWidthOrSentinel();
1787  if (dataWidth > 0 && maskWidth > (size_t)dataWidth)
1788  return emitOpError("the mask width cannot be greater than "
1789  "data width");
1790 
1791  if (portAnnotations().size() != getNumResults())
1792  return emitOpError("the number of result annotations should be "
1793  "equal to the number of results");
1794 
1795  return success();
1796 }

2405 bool firrtl::isConstant(Operation *op) {
2406  // Worklist of ops that need to be examined that should all be constant in
2407  // order for the input operation to be constant.
2408  SmallVector<Operation *, 8> worklist({op});
2409

2411  // and look for counterexamples.
2412  bool constant = true;

2415  // worklist, pull ops off the worklist. If it provides a counterexample, set
2416  // the `constant` to false (and exit on the next loop iteration). Otherwise,
2417  // look through the op or spawn off more ops to look at.
2418  while (constant && !(worklist.empty()))
2419  TypeSwitch<Operation *>(worklist.pop_back_val())
2420  .Case<NodeOp, AsSIntPrimOp, AsUIntPrimOp>([&](auto op) {

2426  for (auto &use : op.getResult().getUses())
2427  worklist.push_back(use.getOwner());
2428  })
2429  .Case<ConstantOp, SpecialConstantOp>([](auto) {})
2430  .Default([&](auto) { constant = false; });

2951 /// Infer the result type for a multiplexer given its two operand types, which
2952 /// may be aggregates.
2953 ///
2954 /// This essentially performs a pairwise comparison of fields and elements, as
2955 /// follows:
2956 /// - Identical operands inferred to their common type
2957 /// - Integer operands inferred to the larger one if both have a known width, a
2958 /// widthless integer otherwise.
2959 /// - Vectors inferred based on the element type.
2960 /// - Bundles inferred in a pairwise fashion based on the field types.
2961 static FIRRTLType inferMuxReturnType(FIRRTLType high, FIRRTLType low,
2962  Optional<Location> loc) {
2963  // If the types are identical we're done.
2964  if (high == low)
2965  return low;
2966 
2967  // The base types need to be equivalent.
2968  if (high.getTypeID() != low.getTypeID()) {
2969  if (loc) {
2970  auto d = mlir::emitError(*loc, "incompatible mux operand types");
2971  d.attachNote() << "true value type: " << high;
2972  d.attachNote() << "false value type: " << low;
2973  }
2974  return {};
2975  }
2976 
2977  // Two different Int types can be compatible. If either has unknown width,
2978  // then return it. If both are known but different width, then return the
2979  // larger one.
2980  if (low.isa<IntType>()) {
2981  int32_t highWidth = high.getBitWidthOrSentinel();
2982  int32_t lowWidth = low.getBitWidthOrSentinel();
2983  if (lowWidth == -1)
2984  return low;
2985  if (highWidth == -1)
2986  return high;
2987  return lowWidth > highWidth ? low : high;
2988  }
2989 
2990  // Infer vector types by comparing the element types.
2991  auto highVector = high.dyn_cast<FVectorType>();
2992  auto lowVector = low.dyn_cast<FVectorType>();
