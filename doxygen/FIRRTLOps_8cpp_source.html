33 /// Remove elements at the specified indices from the input array, returning the
34 /// elements not mentioned. The indices array is expected to be sorted and
35 /// unique.
36 template <typename T>
37 static SmallVector<T>
38 removeElementsAtIndices(ArrayRef<T> input, ArrayRef<unsigned> indicesToDrop) {
39 #ifndef NDEBUG // Check sortedness.
40  if (!input.empty()) {
41  for (size_t i = 1, e = indicesToDrop.size(); i != e; ++i)
42  assert(indicesToDrop[i - 1] < indicesToDrop[i] &&
43  "indicesToDrop isn't sorted and unique");
44  assert(indicesToDrop.back() < input.size() && "index out of range");
45  }
46 #endif
47 
48  // Copy over the live chunks.
49  size_t lastCopied = 0;
50  SmallVector<T> result;
51  result.reserve(input.size() - indicesToDrop.size());
52 
53  for (unsigned indexToDrop : indicesToDrop) {
54  // If we skipped over some valid elements, copy them over.
55  if (indexToDrop > lastCopied) {
56  result.append(input.begin() + lastCopied, input.begin() + indexToDrop);
57  lastCopied = indexToDrop;
58  }
59  // Ignore this value so we don't copy it in the next iteration.
60  ++lastCopied;
61  }
62 
63  // If there are live elements at the end, copy them over.
64  if (lastCopied < input.size())
65  result.append(input.begin() + lastCopied, input.end());
66 
67  return result;
68 }

94 Flow firrtl::foldFlow(Value val, Flow accumulatedFlow) {
95  auto swap = [&accumulatedFlow]() -> Flow {
96  return swapFlow(accumulatedFlow);
97  };
98 
99  if (auto blockArg = val.dyn_cast<BlockArgument>()) {
100  auto op = val.getParentBlock()->getParentOp();
101  auto direction = (Direction)getModulePortDirections(op)
102  .getValue()[blockArg.getArgNumber()];
103  if (direction == Direction::Output)
104  return swap();
105  return accumulatedFlow;
106  }
107 
108  Operation *op = val.getDefiningOp();
109 
110  return TypeSwitch<Operation *, Flow>(op)
111  .Case<SubfieldOp>([&](auto op) {
112  return foldFlow(op.input(),
113  op.isFieldFlipped() ? swap() : accumulatedFlow);
114  })
115  .Case<SubindexOp, SubaccessOp>(
116  [&](auto op) { return foldFlow(op.input(), accumulatedFlow); })
117  // Registers, Wires, and behavioral memory ports are always Duplex.
118  .Case<RegOp, RegResetOp, WireOp, MemoryPortOp>(
119  [](auto) { return Flow::Duplex; })
120  .Case<InstanceOp>([&](auto inst) {
121  for (auto arg : llvm::enumerate(inst.getResults()))
122  if (arg.value() == val) {
123  if (getModulePortDirection(inst.getReferencedModule(),
124  arg.index()) == Direction::Output)
125  return accumulatedFlow;
126  else
127  return swap();
128  }
129  llvm_unreachable("couldn't find result in results");
130  })
131  .Case<MemOp>([&](auto op) { return swap(); })
132  // Anything else acts like a universal source.
133  .Default([&](auto) { return accumulatedFlow; });
134 }

193 static LogicalResult verifyCircuitOp(CircuitOp circuit) {
194  StringRef main = circuit.name();
195 
196  // Check that the circuit has a non-empty name.
197  if (main.empty()) {
198  circuit.emitOpError("must have a non-empty name");
199  return failure();
200  }
201 
202  // Check that a module matching the "main" module exists in the circuit.
203  if (!circuit.getMainModule()) {
204  circuit.emitOpError("must contain one module that matches main name '" +
205  main + "'");
206  return failure();
207  }
208 
209  // Store a mapping of defname to either the first external module
210  // that defines it or, preferentially, the first external module
211  // that defines it and has no parameters.
212  llvm::DenseMap<Attribute, FExtModuleOp> defnameMap;
213 
214  // Verify external modules.
215  for (auto &op : *circuit.getBody()) {
216  auto extModule = dyn_cast<FExtModuleOp>(op);
217  if (!extModule)
218  continue;
219 
220  auto defname = extModule.defnameAttr();
221  if (!defname)
222  continue;
223 
224  // Check that this extmodule's defname does not conflict with
225  // the symbol name of any module.
226  auto collidingModule = circuit.lookupSymbol(defname.getValue());
227  if (isa_and_nonnull<FModuleOp>(collidingModule)) {
228  auto diag =
229  op.emitOpError()
230  << "attribute 'defname' with value " << defname
231  << " conflicts with the name of another module in the circuit";
232  diag.attachNote(collidingModule->getLoc())
233  << "previous module declared here";
234  return failure();
235  }
236 
237  // Find an optional extmodule with a defname collision. Update
238  // the defnameMap if this is the first extmodule with that
239  // defname or if the current extmodule takes no parameters and
240  // the collision does. The latter condition improves later
241  // extmodule verification as checking against a parameterless
242  // module is stricter.
243  FExtModuleOp collidingExtModule;
244  if (auto &value = defnameMap[defname]) {
245  collidingExtModule = value;
246  if (value.parameters() && !extModule.parameters())
247  value = extModule;
248  } else {
249  value = extModule;
250  // Go to the next extmodule if no extmodule with the same
251  // defname was found.
252  continue;
253  }
254 
255  // Check that the number of ports is exactly the same.
256  SmallVector<ModulePortInfo> ports = extModule.getPorts();
257  SmallVector<ModulePortInfo> collidingPorts = collidingExtModule.getPorts();
258 
259  if (ports.size() != collidingPorts.size()) {
260  auto diag = op.emitOpError()
261  << "with 'defname' attribute " << defname << " has "
262  << ports.size()
263  << " ports which is different from a previously defined "
264  "extmodule with the same 'defname' which has "
265  << collidingPorts.size() << " ports";
266  diag.attachNote(collidingExtModule.getLoc())
267  << "previous extmodule definition occurred here";
268  return failure();
269  }
270 
271  // Check that ports match for name and type. Since parameters
272  // *might* affect widths, ignore widths if either module has
273  // parameters. Note that this allows for misdetections, but
274  // has zero false positives.
275  for (auto p : llvm::zip(ports, collidingPorts)) {
276  StringAttr aName = std::get<0>(p).name, bName = std::get<1>(p).name;
277  FIRRTLType aType = std::get<0>(p).type, bType = std::get<1>(p).type;
278 
279  if (extModule.parameters() || collidingExtModule.parameters()) {
280  aType = aType.getWidthlessType();
281  bType = bType.getWidthlessType();
282  }
283  if (aName != bName) {
284  auto diag = op.emitOpError()
285  << "with 'defname' attribute " << defname
286  << " has a port with name " << aName
287  << " which does not match the name of the port "
288  << "in the same position of a previously defined "
289  << "extmodule with the same 'defname', expected port "
290  "to have name "
291  << bName;
292  diag.attachNote(collidingExtModule.getLoc())
293  << "previous extmodule definition occurred here";
294  return failure();
295  }
296  if (aType != bType) {
297  auto diag = op.emitOpError()
298  << "with 'defname' attribute " << defname
299  << " has a port with name " << aName
300  << " which has a different type " << aType
301  << " which does not match the type of the port in "
302  "the same position of a previously defined "
303  "extmodule with the same 'defname', expected port "
304  "to have type "
305  << bType;
306  diag.attachNote(collidingExtModule.getLoc())
307  << "previous extmodule definition occurred here";
308  return failure();
309  }
310  }
311  }
312 
313  return success();
314 }

328 /// This function can extract information about ports from a module and an
329 /// extmodule.
330 SmallVector<ModulePortInfo> firrtl::getModulePortInfo(Operation *op) {
331  SmallVector<ModulePortInfo> results;
332 
333  auto portNamesAttr = getModulePortNames(op);
334  auto portDirections = getModulePortDirections(op).getValue();
335  if (isa<FExtModuleOp>(op)) {
336  // FExtModuleOp's don't have block arguments or locations for their ports.
337  auto argTypes = getModuleType(op).getInputs();
338  auto loc = op->getLoc();
339  for (unsigned i = 0, e = argTypes.size(); i < e; ++i) {
340  auto name = portNamesAttr[i].cast<StringAttr>();
341  auto type = argTypes[i].cast<FIRRTLType>();
342  auto direction = direction::get(portDirections[i]);
343  results.push_back(
344  {name, type, direction, loc, AnnotationSet::forPort(op, i)});
345  }
346  } else {
347  // FModuleOp has the ports as the BlockArgument's of the first block.
348  auto moduleBlock = cast<FModuleOp>(op).getBodyBlock();
349  for (auto portArgAndIndex : llvm::enumerate(moduleBlock->getArguments())) {
350  BlockArgument portArg = portArgAndIndex.value();
351  size_t portIdx = portArgAndIndex.index();
352  auto name = portNamesAttr[portIdx].cast<StringAttr>();
353  auto direction = direction::get(portDirections[portIdx]);
354  results.push_back({name, portArg.getType().cast<FIRRTLType>(), direction,
355  portArg.getLoc(),
356  AnnotationSet::forPort(op, portIdx)});
357  }
358  }
359  return results;
360 }

378 /// Erases the ports listed in `portIndices`. `portIndices` is expected to
379 /// be in order and unique.
380 void FModuleOp::erasePorts(ArrayRef<unsigned> portIndices) {
381  if (portIndices.empty())
382  return;
383 
384  // Drop the direction markers for dead ports.
385  SmallVector<Direction> directions = direction::unpackAttribute(*this);
386  ArrayRef<Attribute> portNames = this->portNames().getValue();
387  assert(directions.size() == portNames.size());
388 
389  SmallVector<Direction> newDirections =
390  removeElementsAtIndices<Direction>(directions, portIndices);
391  SmallVector<Attribute> newPortNames =
392  removeElementsAtIndices(portNames, portIndices);
393  (*this)->setAttr(direction::attrKey,
394  direction::packAttribute(newDirections, getContext()));
395  (*this)->setAttr("portNames", ArrayAttr::get(getContext(), newPortNames));
396 
397  // Erase the common function-like stuff, including the block arguments, and
398  // argument attributes (incl port annotations).
399  eraseArguments(portIndices);
400 }

402 static void buildModule(OpBuilder &builder, OperationState &result,
403  StringAttr name, ArrayRef<ModulePortInfo> ports,
404  ArrayAttr annotations) {
405  using namespace mlir::function_like_impl;
406 
407  // Add an attribute for the name.
408  result.addAttribute(::mlir::SymbolTable::getSymbolAttrName(), name);
409 
410  SmallVector<Type, 4> argTypes;
411  for (auto elt : ports)
412  argTypes.push_back(elt.type);
413 
414  // Record the argument and result types as an attribute.
415  auto type = builder.getFunctionType(argTypes, /*resultTypes*/ {});
416  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
417 
418  // Record the names of the arguments if present.
419  SmallVector<Attribute, 4> argAttrs;
420  SmallVector<Attribute, 4> portNames;
421  SmallVector<Direction, 4> portDirections;
422  for (size_t i = 0, e = ports.size(); i != e; ++i) {
423  portNames.push_back(ports[i].name);
424  portDirections.push_back(ports[i].direction);
425  argAttrs.push_back(ports[i].annotations.getArgumentAttrDict());
426  }
427 
428  // Both attributes are added, even if the module has no ports.
429  result.addAttribute(getArgDictAttrName(), builder.getArrayAttr(argAttrs));
430  result.addAttribute("portNames", builder.getArrayAttr(portNames));
431  result.addAttribute(
432  direction::attrKey,
433  direction::packAttribute(portDirections, builder.getContext()));
434 
435  if (!annotations)
436  annotations = builder.getArrayAttr({});
437  result.addAttribute("annotations", annotations);
438 
439  result.addRegion();
440 }

512 static ParseResult parseFunctionArgumentList2(
513  OpAsmParser &parser, bool allowAttributes, bool allowVariadic,
514  SmallVectorImpl<OpAsmParser::OperandType> &argNames,
515  SmallVectorImpl<Type> &argTypes, SmallVectorImpl<Direction> &argDirections,
516  SmallVectorImpl<NamedAttrList> &argAttrs, bool &isVariadic) {
517  if (parser.parseLParen())
518  return failure();
519 
520  // The argument list either has to consistently have ssa-id's followed by
521  // types, or just be a type list. It isn't ok to sometimes have SSA ID's and
522  // sometimes not.
523  auto parseArgument = [&]() -> ParseResult {
524  llvm::SMLoc loc = parser.getCurrentLocation();
525 
526  // Parse argument name if present.
527  OpAsmParser::OperandType argument;
528  Type argumentType;
529  // TODO: is this safe?
530  SmallVector<StringRef, 2> directions({{"in"}, {"out"}});
531  StringRef direction;
532  if (succeeded(parser.parseOptionalKeyword(&direction, directions)) &&
533  succeeded(parser.parseOptionalRegionArgument(argument)) &&
534  !argument.name.empty()) {
535  // Reject this if the preceding argument was missing a name.
536  if (argNames.empty() && !argTypes.empty())
537  return parser.emitError(loc, "expected type instead of SSA identifier");
538  argNames.push_back(argument);
539  argDirections.push_back(direction::get(direction == "out"));
540 
541  if (parser.parseColonType(argumentType))
542  return failure();
543  } else if (allowVariadic && succeeded(parser.parseOptionalEllipsis())) {
544  isVariadic = true;
545  return success();
546  } else if (!argNames.empty()) {
547  // Reject this if the preceding argument had a name.
548  return parser.emitError(loc, "expected SSA identifier");
549  } else if (parser.parseType(argumentType)) {
550  return failure();
551  }
552 
553  // Add the argument type.
554  argTypes.push_back(argumentType);
555 
556  // Parse any argument attributes.
557  NamedAttrList attrs;
558  if (parser.parseOptionalAttrDict(attrs))
559  return failure();
560  if (!allowAttributes && !attrs.empty())
561  return parser.emitError(loc, "expected arguments without attributes");
562  argAttrs.push_back(attrs);
563  return success();
564  };
565 
566  // Parse the function arguments.
567  isVariadic = false;
568  if (failed(parser.parseOptionalRParen())) {
569  do {
570  unsigned numTypedArguments = argTypes.size();
571  if (parseArgument())
572  return failure();
573 
574  llvm::SMLoc loc = parser.getCurrentLocation();
575  if (argTypes.size() == numTypedArguments &&
576  succeeded(parser.parseOptionalComma()))
577  return parser.emitError(
578  loc, "variadic arguments must be in the end of the argument list");
579  } while (succeeded(parser.parseOptionalComma()));
580  parser.parseRParen();
581  }
582 
583  return success();
584 }

588  SmallVectorImpl<Type> &resultTypes,
589  SmallVectorImpl<NamedAttrList> &resultAttrs) {

611  return failure();
612  }
613  } while (succeeded(parser.parseOptionalComma()));

681 static ParseResult parseFModuleOp(OpAsmParser &parser, OperationState &result,
682  bool isExtModule = false) {
683  using namespace mlir::function_like_impl;
684 
685  // TODO: Should refactor mlir::function_like_impl::parseFunctionLikeOp to
686  // allow these customizations for implicit argument names. Need to not print
687  // the terminator.
688 
689  SmallVector<OpAsmParser::OperandType, 4> entryArgs;
690  SmallVector<NamedAttrList, 4> portNamesAttrs;
691  SmallVector<NamedAttrList, 4> resultAttrs;
692  SmallVector<Type, 4> argTypes;
693  SmallVector<Type, 4> resultTypes;
694  SmallVector<Direction, 4> argDirections;
695  auto &builder = parser.getBuilder();
696 
697  // Parse the name as a symbol.
698  StringAttr nameAttr;
699  if (parser.parseSymbolName(nameAttr, ::mlir::SymbolTable::getSymbolAttrName(),
700  result.attributes))
701  return failure();
702 
703  // Parse the function signature.
704  bool isVariadic = false;
705  if (parseFunctionSignature2(parser, /*allowVariadic*/ false, entryArgs,
706  argTypes, argDirections, portNamesAttrs,
707  isVariadic, resultTypes, resultAttrs))
708  return failure();
709 
710  // Record the argument and result types as an attribute. This is necessary
711  // for external modules.
712  auto type = builder.getFunctionType(argTypes, resultTypes);
713  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
714 
715  // If function attributes are present, parse them.
716  if (parser.parseOptionalAttrDictWithKeyword(result.attributes))
717  return failure();
718 
719  assert(portNamesAttrs.size() == argTypes.size());
720  assert(resultAttrs.size() == resultTypes.size());
721 
722  auto *context = result.getContext();
723 
724  // Add the port directions attribute indiciating which port is.
725  result.addAttribute(direction::attrKey,
726  direction::packAttribute(argDirections, context));
727 
728  SmallVector<Attribute> portNames;
729  if (!result.attributes.get("portNames")) {
730  // Postprocess each of the arguments. If there was no portNames
731  // attribute, and if the argument name was non-numeric, then add the
732  // portNames attribute with the textual name from the IR. The name in the
733  // text file is a load-bearing part of the IR, but we don't want the
734  // verbosity in dumps of including it explicitly in the attribute
735  // dictionary.
736  for (size_t i = 0, e = entryArgs.size(); i != e; ++i) {
737  auto &arg = entryArgs[i];
738 
739  // The name of an argument is of the form "%42" or "%id", and since
740  // parsing succeeded, we know it always has one character.
741  assert(arg.name.size() > 1 && arg.name[0] == '%' && "Unknown MLIR name");
742  if (isdigit(arg.name[1]))
743  portNames.push_back(StringAttr::get(context, ""));
744  else
745  portNames.push_back(StringAttr::get(context, arg.name.drop_front()));
746  }
747  result.addAttribute("portNames", builder.getArrayAttr(portNames));
748  }
749  // Add the attributes to the function arguments.
750  addArgAndResultAttrs(builder, result, portNamesAttrs, resultAttrs);
751 
752  // The annotations attribute is always present, but not printed when empty.
753  if (!result.attributes.get("annotations"))
754  result.addAttribute("annotations", builder.getArrayAttr({}));
755 
756  // Parse the optional function body.
757  auto *body = result.addRegion();
758  if (!isExtModule) {
759  if (parser.parseRegion(*body, entryArgs,
760  entryArgs.empty() ? ArrayRef<Type>() : argTypes))
761  return failure();
762  if (body->empty())
763  body->push_back(new Block());
764  }
765  return success();
766 }

893 
894  // Check that this instance is inside a module.
895  auto module = instance->getParentOfType<FModuleOp>();
896  if (!module) {
897  instance.emitOpError("should be embedded in a 'firrtl.module'");

925  << "original module declared here";
926  return failure();
927  }
928 
929  for (size_t i = 0; i != numResults; i++) {
930  auto resultType = instance.getResult(i).getType();
931  auto expectedType = modulePorts[i].type;
932  if (resultType != expectedType) {
933  auto diag = instance.emitOpError()

936 
937  diag.attachNote(referencedModule->getLoc())
938  << "original module declared here";
939  return failure();
940  }
941  }
942 
943  if (instance.portAnnotations().size() != instance.getNumResults())
944  return instance.emitOpError("the number of result annotations should be "
945  "equal to the number of results");
946 
947  return success();
948 }

995 /// Verify the correctness of a MemOp.
996 static LogicalResult verifyMemOp(MemOp mem) {
997 
998  // Store the port names as we find them. This lets us check quickly
999  // for uniqueneess.
1000  llvm::SmallDenseSet<Attribute, 8> portNamesSet;
1001 
1002  // Store the previous data type. This lets us check that the data
1003  // type is consistent across all ports.
1004  FIRRTLType oldDataType;
1005 
1006  for (size_t i = 0, e = mem.getNumResults(); i != e; ++i) {
1007  auto portName = mem.getPortName(i);
1008 
1009  // Get a bundle type representing this port, stripping an outer
1010  // flip if it exists. If this is not a bundle<> or
1011  // flip<bundle<>>, then this is an error.
1012  BundleType portBundleType =
1013  TypeSwitch<FIRRTLType, BundleType>(
1014  mem.getResult(i).getType().cast<FIRRTLType>())
1015  .Case<BundleType>([](BundleType a) { return a; })
1016  .Default([](auto) { return nullptr; });
1017  if (!portBundleType) {
1018  mem.emitOpError() << "has an invalid type on port " << portName
1019  << " (expected '!firrtl.bundle<...>')";
1020  return failure();
1021  }
1022 
1023  // Require that all port names are unique.
1024  if (!portNamesSet.insert(portName).second) {
1025  mem.emitOpError() << "has non-unique port name " << portName;
1026  return failure();
1027  }
1028 
1029  // Determine the kind of the memory. If the kind cannot be
1030  // determined, then it's indicative of the wrong number of fields
1031  // in the type (but we don't know any more just yet).
1032  MemOp::PortKind portKind;
1033  {
1034  auto elt = mem.getPortNamed(portName);
1035  if (!elt) {
1036  mem.emitOpError() << "could not get port with name " << portName;
1037  return failure();
1038  }
1039  auto firrtlType = elt.getType().cast<FIRRTLType>();
1040  auto portType = firrtlType.dyn_cast<BundleType>();
1041  switch (portType.getNumElements()) {
1042  case 4:
1043  portKind = MemOp::PortKind::Read;
1044  break;
1045  case 5:
1046  portKind = MemOp::PortKind::Write;
1047  break;
1048  case 7:
1049  portKind = MemOp::PortKind::ReadWrite;
1050  break;
1051  default:
1052  mem.emitOpError()
1053  << "has an invalid number of fields on port " << portName
1054  << " (expected 4 for read, 5 for write, or 7 for read/write)";
1055  return failure();
1056  }
1057  }
1058 
1059  // Safely search for the "data" field, erroring if it can't be
1060  // found.
1061  FIRRTLType dataType;
1062  {
1063  auto dataTypeOption = portBundleType.getElement("data");
1064  if (!dataTypeOption && portKind == MemOp::PortKind::ReadWrite)
1065  dataTypeOption = portBundleType.getElement("wdata");
1066  if (!dataTypeOption) {
1067  mem.emitOpError() << "has no data field on port " << portName
1068  << " (expected to see \"data\" for a read or write "
1069  "port or \"rdata\" for a read/write port)";
1070  return failure();
1071  }
1072  dataType = dataTypeOption.getValue().type;
1073  // Read data is expected to ba a flip.
1074  if (portKind == MemOp::PortKind::Read) {
1075  // FIXME error on missing bundle flip
1076  }
1077  }
1078 
1079  // Error if the data type isn't passive.
1080  if (!dataType.isPassive()) {
1081  mem.emitOpError() << "has non-passive data type on port " << portName
1082  << " (memory types must be passive)";
1083  return failure();
1084  }
1085 
1086  // Error if the data type contains analog types.
1087  if (dataType.containsAnalog()) {
1088  mem.emitOpError()
1089  << "has a data type that contains an analog type on port " << portName
1090  << " (memory types cannot contain analog types)";
1091  return failure();
1092  }
1093 
1094  // Check that the port type matches the kind that we determined
1095  // for this port. This catches situations of extraneous port
1096  // fields beind included or the fields being named incorrectly.
1097  FIRRTLType expectedType =
1098  mem.getTypeForPort(mem.depth(), dataType, portKind);
1099  // Compute the original port type as portBundleType may have
1100  // stripped outer flip information.
1101  auto originalType = mem.getResult(i).getType();
1102  if (originalType != expectedType) {
1103  StringRef portKindName;
1104  switch (portKind) {
1105  case MemOp::PortKind::Read:
1106  portKindName = "read";
1107  break;
1108  case MemOp::PortKind::Write:
1109  portKindName = "write";
1110  break;
1111  case MemOp::PortKind::ReadWrite:
1112  portKindName = "readwrite";
1113  break;
1114  }
1115  mem.emitOpError() << "has an invalid type for port " << portName
1116  << " of determined kind \"" << portKindName
1117  << "\" (expected " << expectedType << ", but got "
1118  << originalType << ")";
1119  return failure();
1120  }
1121 
1122  // Error if the type of the current port was not the same as the
1123  // last port, but skip checking the first port.
1124  if (oldDataType && oldDataType != dataType) {
1125  mem.emitOpError() << "port " << mem.getPortName(i)
1126  << " has a different type than port "
1127  << mem.getPortName(i - 1) << " (expected "
1128  << oldDataType << ", but got " << dataType << ")";
1129  return failure();
1130  }
1131 
1132  oldDataType = dataType;
1133  }
1134 
1135  if (mem.portAnnotations().size() != mem.getNumResults())
1136  return mem.emitOpError("the number of result annotations should be "
1137  "equal to the number of results");
1138 
1139  return success();
1140 }

1437 static ParseResult parseConstantOp(OpAsmParser &parser,
1438  OperationState &result) {
1439  // Parse the constant value, without knowing its width.
1440  APInt value;
1441  auto loc = parser.getCurrentLocation();
1442  auto valueResult = parser.parseOptionalInteger(value);
1443  if (!valueResult.hasValue())
1444  return parser.emitError(loc, "expected integer value");
1445 
1446  // Parse the result firrtl integer type.
1447  IntType resultType;
1448  if (failed(*valueResult) || parser.parseColonType(resultType) ||
1449  parser.parseOptionalAttrDict(result.attributes))
1450  return failure();
1451  result.addTypes(resultType);
1452 
1453  // Now that we know the width and sign of the result type, we can munge the
1454  // APInt as appropriate.
1455  if (resultType.hasWidth()) {
1456  auto width = (unsigned)resultType.getWidthOrSentinel();
1457  if (width == 0)
1458  return parser.emitError(loc, "zero bit constants aren't allowed");
1459 
1460  if (width > value.getBitWidth()) {
1461  // sext is always safe here, even for unsigned values, because the
1462  // parseOptionalInteger method will return something with a zero in the
1463  // top bits if it is a positive number.
1464  value = value.sext(width);
1465  } else if (width < value.getBitWidth()) {
1466  // The parser can return an unnecessarily wide result with leading zeros.
1467  // This isn't a problem, but truncating off bits is bad.
1468  if (value.getNumSignBits() < value.getBitWidth() - width)
1469  return parser.emitError(loc, "constant too large for result type ")
1470  << resultType;
1471  value = value.trunc(width);
1472  }
1473  }
1474 
1475  auto intType = parser.getBuilder().getIntegerType(value.getBitWidth(),
1476  resultType.isSigned());
1477  auto valueAttr = parser.getBuilder().getIntegerAttr(intType, value);
1478  result.addAttribute("value", valueAttr);
1479  return success();
1480 }

1638 /// width of them if known. If unknown, return -1 for the widths.

2452  MLIRContext *ctx) {

2459  if (size == 0)

2463  APInt portDirections(size, 0);
2464  for (size_t i = 0, e = directions.size(); i != e; ++i)
2465  if (directions[i] == Direction::Output)
2466  portDirections.setBit(i);
