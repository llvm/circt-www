59 /// This holds the name and type that describes the module's ports.
60 struct ModulePortInfo {
61  StringAttr name;
62  FIRRTLType type;
63  Direction direction;
64  Location loc;
65  AnnotationSet annotations = AnnotationSet(type.getContext());
66 
67  StringRef getName() const { return name ? name.getValue() : ""; }
68 
69  /// Return true if this is a simple output-only port. If you want the
70  /// direction of the port, use the \p direction parameter.
71  bool isOutput() {
72  auto flags = type.getRecursiveTypeProperties();
73  return flags.isPassive && !flags.containsAnalog &&
74  direction == Direction::Output;
75  }
76 
77  /// Return true if this is a simple input-only port. If you want the
78  /// direction of the port, use the \p direction parameter.
79  bool isInput() {
80  auto flags = type.getRecursiveTypeProperties();
81  return flags.isPassive && !flags.containsAnalog &&
82  direction == Direction::Input;
83  }
84 
85  /// Return true if this is an inout port. This will be true if the port
86  /// contains either bi-directional signals or analog types.
87  bool isInOut() { return !isOutput() && !isInput(); }
88 };

123 /// Compute the flow for a Value, \p val, as determined by the FIRRTL
124 /// specification. This recursively walks backwards from \p val to the
125 /// declaration. The resulting flow is a combination of the declaration flow
126 /// (output ports and instance inputs are sinks, registers and wires are duplex,
127 /// anything else is a source) and the number of intermediary flips. An even
128 /// number of flips will result in the same flow as the declaration. An odd
129 /// number of flips will result in reversed flow being returned. The reverse of
130 /// source is sink. The reverse of sink is source. The reverse of duplex is
131 /// duplex. The \p accumulatedFlow parameter sets the initial flow. A user
132 /// should normally \a not have to change this from its default of \p
133 /// Flow::Source.
134 Flow foldFlow(Value val, Flow accumulatedFlow = Flow::Source);
