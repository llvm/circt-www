64 /// Compute the flow for a Value, \p val, as determined by the FIRRTL
65 /// specification. This recursively walks backwards from \p val to the
66 /// declaration. The resulting flow is a combination of the declaration flow
67 /// (output ports and instance inputs are sinks, registers and wires are
68 /// duplex, anything else is a source) and the number of intermediary flips.
69 /// An even number of flips will result in the same flow as the declaration.
70 /// An odd number of flips will result in reversed flow being returned. The
71 /// reverse of source is sink. The reverse of sink is source. The reverse of
72 /// duplex is duplex. The \p accumulatedFlow parameter sets the initial flow.
73 /// A user should normally \a not have to change this from its default of \p
74 /// Flow::Source.
75 Flow foldFlow(Value val, Flow accumulatedFlow = Flow::Source);

215 // Record of the inner sym name, the module name and the corresponding
216 // operation. Also the port index, if the symbol is on a module port.
217 // This is used to record the operations or ports, that have an inner sym.
218 // The operation and portIdx, can be null, when we have an InnerRefAttr, that is
219 // the module name and sym name, but we don't yet have a handle to the operation
220 // which has the Reference. So, InnerRefRecord can be used to construct illegal
221 // InnerRefAttr, which do not exist in the circt. That is the reason, the
222 // comparison operators here, only care for module name and symbol name.
223 struct InnerRefRecord {
224  mlir::StringAttr mod, innerSym;
225  mlir::Operation *op = nullptr;
226  unsigned portIdx = 0;
227  InnerRefRecord(StringAttr mod, StringAttr innerSym, Operation *op)
228  : mod(mod), innerSym(innerSym), op(op) {}
229  InnerRefRecord(StringAttr mod, StringAttr innerSym, Operation *op,
230  unsigned portIdx)
231  : mod(mod), innerSym(innerSym), op(op), portIdx(portIdx) {}
232  InnerRefRecord(hw::InnerRefAttr ref)
233  : mod(ref.getModule()), innerSym(ref.getName()) {}
234  bool operator<(const InnerRefRecord &rhs) const {
235  return (innerSym.getValue() < rhs.innerSym.getValue() ||
236  (innerSym == rhs.innerSym && mod.getValue() < rhs.mod));
237  }
238  bool operator==(const InnerRefRecord &rhs) const {
239  return (innerSym == rhs.innerSym && mod == rhs.mod);
240  }
241  bool operator!=(const InnerRefRecord &rhs) const { return !(*this == rhs); }
242 };

244 // A data structure to record and lookup an InnerSym and the corresponding
245 // operation. Can be used when the list is populated first, then sorted and then
246 // only used for lookup. This does not handle duplicate entries explicitly. The
247 // list must be sorted, before any lookup. This is based on an observation that
248 // Dense arrays can be efficient lookup structures. Especially when we have
249 // insert-phase and lookup-phase based code.
250 // TODO: Generalize this data structure.
251 struct InnerRefList {
252  InnerRefList(MLIRContext *context)
253  : InnerSymAttr(StringAttr::get(context, "inner_sym")) {}
254 
255  void sort() {
256  llvm::sort(list);
257  sorted = true;
258  }
259  int search(const InnerRefRecord &key) const {
260  assert(sorted && "Sort the list before search");
261  if (!sorted || list.empty())
262  return -1;
263  const auto *iter = std::lower_bound(list.begin(), list.end(), key);
264  if (iter == list.end())
265  return -1;
266  return (iter - list.begin());
267  }
268  bool exists(const InnerRefRecord &key) const { return search(key) != -1; }
269  Operation *getOpIfExists(const hw::InnerRefAttr ref) const {
270  auto index = search(InnerRefRecord(ref));
271  if (index != -1 && list[index].op != nullptr)
272  return list[index].op;
273  return nullptr;
274  }
275  const InnerRefRecord *getRecordIfExists(const hw::InnerRefAttr ref) const {
276  auto index = search(InnerRefRecord(ref));
277  if (index != -1)
278  return &list[index];
279  return nullptr;
280  }
281  void pushBack(InnerRefRecord &key) {
282  list.push_back(key);
283  sorted = false;
284  }
285  // Inesrt the op with the module modName, if it has an inner sym.
286  bool insert(Operation *op, StringAttr modName) {
287  if (op == nullptr)
288  return false;
289  auto innerSym = op->getAttrOfType<StringAttr>(InnerSymAttr);
290  if (!innerSym)
291  return false;
292  list.emplace_back(modName, innerSym, op);
293  sorted = false;
294  return true;
295  }
296  // Insert all the ports for the op, if they have the inner sym.
297  bool insert(FModuleLike op) {
298  StringAttr modName = op.moduleNameAttr();
299  bool inserted = false;
300  for (auto sym : llvm::enumerate(op.getPortSymbols()))
301  if (sym.value()) {
302  list.emplace_back(modName, sym.value().cast<StringAttr>(), op,
303  sym.index());
304  inserted = true;
305  }
306  sorted = !inserted;
307  return inserted;
308  }
309 
310 private:
311  StringAttr InnerSymAttr;
312  SmallVector<InnerRefRecord> list;
313  bool sorted = false;
314 };
