38 /// Print a type with a custom printer implementation.
39 ///
40 /// This only prints a subset of all types in the dialect. Use `printNestedType`
41 /// instead, which will call this function in turn, as appropriate.
42 static LogicalResult customTypePrinter(Type type, AsmPrinter &os) {
43  auto printWidthQualifier = [&](Optional<int32_t> width) {
44  if (width)
45  os << '<' << width.getValue() << '>';
46  };
47  bool anyFailed = false;
48  TypeSwitch<Type>(type)
49  .Case<ClockType>([&](auto) { os << "clock"; })
50  .Case<ResetType>([&](auto) { os << "reset"; })
51  .Case<AsyncResetType>([&](auto) { os << "asyncreset"; })
52  .Case<SIntType>([&](auto sIntType) {
53  os << "sint";
54  printWidthQualifier(sIntType.getWidth());
55  })
56  .Case<UIntType>([&](auto uIntType) {
57  os << "uint";
58  printWidthQualifier(uIntType.getWidth());
59  })
60  .Case<AnalogType>([&](auto analogType) {
61  os << "analog";
62  printWidthQualifier(analogType.getWidth());
63  })
64  .Case<BundleType>([&](auto bundleType) {
65  os << "bundle<";
66  llvm::interleaveComma(bundleType, os,
67  [&](BundleType::BundleElement element) {
68  os << element.name.getValue();
69  if (element.isFlip)
70  os << " flip";
71  os << ": ";
72  printNestedType(element.type, os);
73  });
74  os << '>';
75  })
76  .Case<FVectorType>([&](auto vectorType) {
77  os << "vector<";
78  printNestedType(vectorType.getElementType(), os);
79  os << ", " << vectorType.getNumElements() << '>';
80  })
81  .Default([&](auto) { anyFailed = true; });
82  return failure(anyFailed);
83 }

99 /// Parse a type with a custom parser implementation.
100 ///
101 /// This only accepts a subset of all types in the dialect. Use `parseType`
102 /// instead, which will call this function in turn, as appropriate.
103 ///
104 /// Returns `llvm::None` if the type `name` is not covered by the custom
105 /// parsers. Otherwise returns success or failure as appropriate. On success,
106 /// `result` is set to the resulting type.
107 ///
108 /// ```plain
109 /// firrtl-type
110 /// ::= clock
111 /// ::= reset
112 /// ::= asyncreset
113 /// ::= sint ('<' int '>')?
114 /// ::= uint ('<' int '>')?
115 /// ::= analog ('<' int '>')?
116 /// ::= bundle '<' (bundle-elt (',' bundle-elt)*)? '>'
117 /// ::= vector '<' type ',' int '>'
118 ///
119 /// bundle-elt ::= identifier ':' type
120 /// ```
121 static OptionalParseResult customTypeParser(AsmParser &parser, StringRef name,
122  Type &result) {
123  auto *context = parser.getContext();
124  if (name.equals("clock"))
125  return result = ClockType::get(context), success();
126  if (name.equals("reset"))
127  return result = ResetType::get(context), success();
128  if (name.equals("asyncreset"))
129  return result = AsyncResetType::get(context), success();
130 
131  if (name.equals("sint") || name.equals("uint") || name.equals("analog")) {
132  // Parse the width specifier if it exists.
133  int32_t width = -1;
134  if (!parser.parseOptionalLess()) {
135  if (parser.parseInteger(width) || parser.parseGreater())
136  return failure();
137 
138  if (width < 0)
139  return parser.emitError(parser.getNameLoc(), "unknown width"),
140  failure();
141  }
142 
143  if (name.equals("sint"))
144  result = SIntType::get(context, width);
145  else if (name.equals("uint"))
146  result = UIntType::get(context, width);
147  else {
148  assert(name.equals("analog"));
149  result = AnalogType::get(context, width);
150  }
151  return success();
152  }
153 
154  if (name.equals("bundle")) {
155  SmallVector<BundleType::BundleElement, 4> elements;
156 
157  auto parseBundleElement = [&]() -> ParseResult {
158  std::string nameStr;
159  StringRef name;
160  FIRRTLType type;
161 
162  // The 'name' can be an identifier or an integer.
163  uint32_t fieldIntName;
164  auto intName = parser.parseOptionalInteger(fieldIntName);
165  if (intName.hasValue()) {
166  if (failed(intName.getValue()))
167  return failure();
168  nameStr = llvm::utostr(fieldIntName);
169  name = nameStr;
170  } else {
171  // Otherwise must be an identifier.
172  if (parser.parseKeyword(&name))
173  return failure();
174  }
175 
176  bool isFlip = succeeded(parser.parseOptionalKeyword("flip"));
177  if (parser.parseColon() || parseNestedType(type, parser))
178  return failure();
179 
180  elements.push_back({StringAttr::get(context, name), isFlip, type});
181  return success();
182  };
183 
184  if (parser.parseCommaSeparatedList(mlir::AsmParser::Delimiter::LessGreater,
185  parseBundleElement))
186  return failure();
187 
188  return result = BundleType::get(elements, context), success();
189  }
190 
191  if (name.equals("vector")) {
192  FIRRTLType elementType;
193  uint64_t width = 0;
194 
195  if (parser.parseLess() || parseNestedType(elementType, parser) ||
196  parser.parseComma() || parser.parseInteger(width) ||
197  parser.parseGreater())
198  return failure();
199 
200  return result = FVectorType::get(elementType, width), success();
201  }
202 
203  return {};
204 }

206 /// Parse a type defined by this dialect.
207 ///
208 /// This will first try the generated type parsers and then resort to the custom
209 /// parser implementation. Emits an error and returns failure if `name` does not
210 /// refer to a type defined in this dialect.
211 static ParseResult parseType(Type &result, StringRef name, AsmParser &parser) {
212  // Try the custom type parser.
213  OptionalParseResult parseResult = customTypeParser(parser, name, result);
214  if (parseResult.hasValue())
215  return parseResult.getValue();
216 
217  // None of the above recognized the type, so we bail.
218  parser.emitError(parser.getNameLoc(), "unknown FIRRTL dialect type: \"")
219  << name << "\"";
220  return failure();
221 }

223 /// Parse a `FIRRTLType` with a `name` that has already been parsed.
224 ///
225 /// Note that only a subset of types defined in the FIRRTL dialect inherit from
226 /// `FIRRTLType`. Use `parseType` to parse *any* of the defined types.
227 static ParseResult parseFIRRTLType(FIRRTLType &result, StringRef name,
228  AsmParser &parser) {
229  Type type;
230  if (failed(parseType(type, name, parser)))
231  return failure();
232  result = type.dyn_cast<FIRRTLType>();
233  if (result)
234  return success();
235  parser.emitError(parser.getNameLoc(), "unknown FIRRTL type: \"")
236  << name << "\"";
237  return failure();
238 }

240 /// Parse a `FIRRTLType`.
241 ///
242 /// Note that only a subset of types defined in the FIRRTL dialect inherit from
243 /// `FIRRTLType`. Use `parseType` to parse *any* of the defined types.
244 ParseResult circt::firrtl::parseNestedType(FIRRTLType &result,

486 /// Helper to implement the equivalence logic for a pair of bundle elements.
487 /// Note that the FIRRTL spec requires bundle elements to have the same
488 /// orientation, but this only compares their passive types. The FIRRTL dialect
489 /// differs from the spec in how it uses flip types for module output ports and
490 /// canonicalizes flips in bundles, so only passive types can be compared here.
491 static bool areBundleElementsEquivalent(BundleType::BundleElement destElement,
492  BundleType::BundleElement srcElement) {
493  if (destElement.name != srcElement.name)
494  return false;
495  if (destElement.isFlip != srcElement.isFlip)
496  return false;
497 
498  return areTypesEquivalent(destElement.type, srcElement.type);
499 }

501 /// Returns whether the two types are equivalent. This implements the exact
502 /// definition of type equivalence in the FIRRTL spec. If the types being
503 /// compared have any outer flips that encode FIRRTL module directions (input or
504 /// output), these should be stripped before using this method.
505 bool firrtl::areTypesEquivalent(FIRRTLType destType, FIRRTLType srcType) {
506  // Reset types can be driven by UInt<1>, AsyncReset, or Reset types.
507  if (destType.isa<ResetType>())
508  return srcType.isResetType();
509 
510  // Reset types can drive UInt<1>, AsyncReset, or Reset types.
511  if (srcType.isa<ResetType>())
512  return destType.isResetType();
513 
514  // Vector types can be connected if they have the same size and element type.
515  auto destVectorType = destType.dyn_cast<FVectorType>();
516  auto srcVectorType = srcType.dyn_cast<FVectorType>();
517  if (destVectorType && srcVectorType)
518  return destVectorType.getNumElements() == srcVectorType.getNumElements() &&
519  areTypesEquivalent(destVectorType.getElementType(),
520  srcVectorType.getElementType());
521 
522  // Bundle types can be connected if they have the same size, element names,
523  // and element types.
524  auto destBundleType = destType.dyn_cast<BundleType>();
525  auto srcBundleType = srcType.dyn_cast<BundleType>();
526  if (destBundleType && srcBundleType) {
527  auto destElements = destBundleType.getElements();
528  auto srcElements = srcBundleType.getElements();
529  size_t numDestElements = destElements.size();
530  if (numDestElements != srcElements.size())
531  return false;
532 
533  for (size_t i = 0; i < numDestElements; ++i) {
534  auto destElement = destElements[i];
535  auto srcElement = srcElements[i];
536  if (!areBundleElementsEquivalent(destElement, srcElement))
537  return false;
538  }
539  return true;
540  }
541 
542  // Ground types can be connected if their passive, widthless versions
543  // are equal.
544  return destType.getWidthlessType() == srcType.getWidthlessType();
545 }

547 /// Returns whether the two types are weakly equivalent.
548 bool firrtl::areTypesWeaklyEquivalent(FIRRTLType destType, FIRRTLType srcType,
549  bool destFlip, bool srcFlip) {
550 
551  // Reset types can be driven by UInt<1>, AsyncReset, or Reset types.
552  if (destType.isa<ResetType>())
553  return srcType.isResetType();
554 
555  // Reset types can drive UInt<1>, AsyncReset, or Reset types.
556  if (srcType.isa<ResetType>())
557  return destType.isResetType();
558 
559  // Vector types can be connected if their element types are weakly equivalent.
560  // Size doesn't matter.
561  auto destVectorType = destType.dyn_cast<FVectorType>();
562  auto srcVectorType = srcType.dyn_cast<FVectorType>();
563  if (destVectorType && srcVectorType)
564  return areTypesWeaklyEquivalent(destVectorType.getElementType(),
565  srcVectorType.getElementType(), destFlip,
566  srcFlip);
567 
568  // Bundle types are weakly equivalent if all common elements are weakly
569  // equivalent. Non-matching fields are ignored. Flips are "pushed" into
570  // recursive weak type equivalence checks.
571  auto destBundleType = destType.dyn_cast<BundleType>();
572  auto srcBundleType = srcType.dyn_cast<BundleType>();
573  if (destBundleType && srcBundleType)
574  return llvm::all_of(destBundleType, [&](auto destElt) -> bool {
575  auto destField = destElt.name.getValue();
576  auto srcElt = srcBundleType.getElement(destField);
577  // If the src doesn't contain the destination's field, that's okay.
578  if (!srcElt)
579  return true;
580  return areTypesWeaklyEquivalent(destElt.type, srcElt.getValue().type,
581  destFlip ^ destElt.isFlip,
582  srcFlip ^ srcElt.getValue().isFlip);
583  });
584 
585  // Ground types can be connected if their passive, widthless versions
586  // are equal and leaf flippedness matches.
587  return destType.getWidthlessType() == srcType.getWidthlessType() &&
588  destFlip == srcFlip;
589 }

591 /// Returns true if the destination is at least as wide as an equivalent source.
592 bool firrtl::isTypeLarger(FIRRTLType dstType, FIRRTLType srcType) {
593  return TypeSwitch<FIRRTLType, bool>(dstType)
594  .Case<BundleType>([&](auto dstBundle) {
595  auto srcBundle = srcType.cast<BundleType>();
596  for (size_t i = 0, n = dstBundle.getNumElements(); i < n; ++i) {
597  auto srcElem = srcBundle.getElement(i);
598  auto dstElem = dstBundle.getElement(i);
599  if (dstElem.isFlip) {
600  if (!isTypeLarger(srcElem.type, dstElem.type))
601  return false;
602  } else {
603  if (!isTypeLarger(dstElem.type, srcElem.type))
604  return false;
605  }
606  }
607  return true;
608  })
609  .Case<FVectorType>([&](auto vector) {
610  return isTypeLarger(vector.getElementType(),
611  srcType.cast<FVectorType>().getElementType());
612  })

727 struct BundleTypeStorage : mlir::TypeStorage {
728  using KeyTy = ArrayRef<BundleType::BundleElement>;
729 
730  BundleTypeStorage(KeyTy elements)
731  : elements(elements.begin(), elements.end()) {
732  RecursiveTypeProperties props{true, false, false};
733  unsigned fieldID = 0;
734  fieldIDs.reserve(elements.size());
735  for (auto &element : elements) {
736  auto type = element.type;
737  auto eltInfo = type.getRecursiveTypeProperties();
738  props.isPassive &= eltInfo.isPassive & !element.isFlip;
739  props.containsAnalog |= eltInfo.containsAnalog;
740  props.hasUninferredWidth |= eltInfo.hasUninferredWidth;
741  fieldID += 1;
742  fieldIDs.push_back(fieldID);
743  // Increment the field ID for the next field by the number of subfields.
744  fieldID += type.getMaxFieldID();
745  }
746  maxFieldID = fieldID;
747  passiveContainsAnalogTypeInfo.setInt(props.toFlags());
748  }
749 
750  bool operator==(const KeyTy &key) const { return key == KeyTy(elements); }
751 
752  static llvm::hash_code hashKey(const KeyTy &key) {
753  return llvm::hash_combine_range(key.begin(), key.end());
754  }
755 
756  static BundleTypeStorage *construct(TypeStorageAllocator &allocator,
757  KeyTy key) {
758  return new (allocator.allocate<BundleTypeStorage>()) BundleTypeStorage(key);
759  }
760 
761  SmallVector<BundleType::BundleElement, 4> elements;
762  SmallVector<unsigned, 4> fieldIDs;
763  unsigned maxFieldID;
764 
765  /// This holds the bits for the type's recursive properties, and can hold a
766  /// pointer to a passive version of the type.
767  llvm::PointerIntPair<Type, RecursiveTypeProperties::numBits, unsigned>
768  passiveContainsAnalogTypeInfo;

790 /// Return this type with any flip types recursively removed from itself.
791 FIRRTLType BundleType::getPassiveType() {
792  auto *impl = getImpl();
793 
794  // If we've already determined and cached the passive type, use it.
795  if (auto passiveType = impl->passiveContainsAnalogTypeInfo.getPointer())
796  return passiveType.cast<FIRRTLType>();
797 
798  // If this type is already passive, use it and remember for next time.
799  if (impl->passiveContainsAnalogTypeInfo.getInt() & IsPassiveBitMask) {
800  impl->passiveContainsAnalogTypeInfo.setPointer(*this);
801  return *this;
802  }
803 
804  // Otherwise at least one element is non-passive, rebuild a passive version.
805  SmallVector<BundleType::BundleElement, 16> newElements;
806  newElements.reserve(impl->elements.size());
807  for (auto &elt : impl->elements) {
808  newElements.push_back({elt.name, false, elt.type.getPassiveType()});
809  }
810 
811  auto passiveType = BundleType::get(newElements, getContext());
812  impl->passiveContainsAnalogTypeInfo.setPointer(passiveType);
813  return passiveType;
814 }

1020 // Get the bit width for this type, return None if unknown. Unlike
1021 // getBitWidthOrSentinel(), this can recursively compute the bitwidth of
1022 // aggregate types. For bundle and vectors, recursively get the width of each
1023 // field element and return the total bit width of the aggregate type. This
1024 // returns None, if any of the bundle fields is a flip type, or ground type with
1025 // unknown bit width.
1026 llvm::Optional<int64_t> firrtl::getBitWidth(FIRRTLType type) {
1027  std::function<llvm::Optional<int64_t>(FIRRTLType)> getWidth =
1028  [&](FIRRTLType type) -> llvm::Optional<int64_t> {
1029  return TypeSwitch<FIRRTLType, llvm::Optional<int64_t>>(type)
1030  .Case<BundleType>([&](BundleType bundle) {
1031  int64_t width = 0;
1032  for (auto &elt : bundle) {
1033  if (elt.isFlip)
1034  return llvm::Optional<int64_t>(None);
1035  auto w = getBitWidth(elt.type);
1036  if (!w.hasValue())
1037  return llvm::Optional<int64_t>(None);
1038  width += w.getValue();
1039  }
1040  return llvm::Optional<int64_t>(width);
1041  })
1042  .Case<FVectorType>([&](auto vector) {
1043  auto w = getBitWidth(vector.getElementType());
1044  if (!w.hasValue())
1045  return llvm::Optional<int64_t>(None);
1046  return llvm::Optional<int64_t>(w.getValue() *
1047  vector.getNumElements());
1048  })
1049  .Case<IntType>([&](IntType iType) {
1050  auto retval = iType.getWidth();
1051  if (retval)
1052  return llvm::Optional<int64_t>(*retval);
1053  else
1054  return llvm::Optional<int64_t>(None);
1055  })
1056  .Case<ClockType, ResetType, AsyncResetType>([](Type) { return 1; })
1057  .Default([&](auto t) { return llvm::Optional<int64_t>(None); });
1058  };
1059  return getWidth(type);
1060 }
