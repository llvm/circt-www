38 /// A collection of bits indicating the recursive properties of a type.
39 struct RecursiveTypeProperties {
40  /// Whether the type only contains passive elements.
41  bool isPassive : 1;
42  /// Whether the type contains an analog type.
43  bool containsAnalog : 1;
44  /// Whether the type has any uninferred bit widths.
45  bool hasUninferredWidth : 1;
46 
47  /// The number of bits required to represent a type's recursive properties.
48  static constexpr unsigned numBits = 3;
49  /// Unpack `RecursiveTypeProperties` from a bunch of bits.
50  static RecursiveTypeProperties fromFlags(unsigned bits);
51  /// Pack `RecursiveTypeProperties` as a bunch of bits.
52  unsigned toFlags() const;
53 };

55 // This is a common base class for all FIRRTL types.
56 class FIRRTLType : public Type {
57 public:
58  /// Return true if this is a "passive" type - one that contains no "flip"
59  /// types recursively within itself.
60  bool isPassive() { return getRecursiveTypeProperties().isPassive; }
61 
62  /// Returns true if this is a "passive" that which is not analog.
63  bool isRegisterType() { return isPassive() && !containsAnalog(); }
64 
65  /// Return true if this is a 'ground' type, aka a non-aggregate type.
66  bool isGround();
67 
68  /// Return true if this is or contains an Analog type.
69  bool containsAnalog() { return getRecursiveTypeProperties().containsAnalog; }
70 
71  /// Return true if this type contains an uninferred bit width.
72  bool hasUninferredWidth() {
73  return getRecursiveTypeProperties().hasUninferredWidth;
74  }
75 
76  /// Return the recursive properties of the type, containing the `isPassive`,
77  /// `containsAnalog`, and `hasUninferredWidth` bits.
78  RecursiveTypeProperties getRecursiveTypeProperties();
79 
80  /// Return this type with any flip types recursively removed from itself.
81  FIRRTLType getPassiveType();
82 
83  /// Return this type with all ground types replaced with UInt<1>. This is
84  /// used for `mem` operations.
85  FIRRTLType getMaskType();
86 
87  /// Return this type with widths of all ground types removed. This
88  /// enables two types to be compared by structure and name ignoring
89  /// widths.
90  FIRRTLType getWidthlessType();
91 
92  /// If this is an IntType, AnalogType, or sugar type for a single bit (Clock,
93  /// Reset, etc) then return the bitwidth. Return -1 if the is one of these
94  /// types but without a specified bitwidth. Return -2 if this isn't a simple
95  /// type.
96  int32_t getBitWidthOrSentinel();
97 
98  /// Support method to enable LLVM-style type casting.
99  static bool classof(Type type) {
100  return llvm::isa<FIRRTLDialect>(type.getDialect());
101  }
102 
103  /// Return true if this is a valid "reset" type.
104  bool isResetType();
105 
106  /// Get the maximum field ID of this type. For integers and other ground
107  /// types, there are no subfields and the maximum field ID is 0. For bundle
108  /// types and vector types, each field is assigned a field ID in a depth-first
109  /// walk order. This function is used to calculate field IDs when this type is
110  /// nested under another type.
111  unsigned getMaxFieldID();
112 
113  /// Get the sub-type of a type for a field ID, and the subfield's ID. Strip
114  /// off a single layer of this type and return the sub-type and a field ID
115  /// targeting the same field, but rebased on the sub-type.
116  std::pair<FIRRTLType, unsigned> getSubTypeByFieldID(unsigned fieldID);
117 
118  /// Return the final type targeted by this field ID by recursively walking all
119  /// nested aggregate types. This is the identity function for ground types.
120  FIRRTLType getFinalTypeByFieldID(unsigned fieldID);
121 
122  /// Returns the effective field id when treating the index field as the
123  /// root of the type. Essentially maps a fieldID to a fieldID after a
124  /// subfield op. Returns the new id and whether the id is in the given
125  /// child.
126  std::pair<unsigned, bool> rootChildFieldID(unsigned fieldID, unsigned index);
127 
128 protected:
129  using Type::Type;
130 };

136 bool areTypesEquivalent(FIRRTLType destType, FIRRTLType srcType);

138 /// Returns true if two types are weakly equivalent. See the FIRRTL spec,
139 /// Section 4.6, for a full definition of this. Roughly, the oriented types

141 /// types with flips in different positions to be equivalent.

145 /// Returns true if the destination is at least as wide as a source. The source
146 /// and destination types must be equivalent non-analog types. The types are
147 /// recursively connected to ensure that the destination is larger than the

190 template <typename ConcreteType, typename ParentType>
191 class WidthQualifiedType
192  : public FIRRTLType::TypeBase<ConcreteType, ParentType,
193  detail::WidthTypeStorage> {
194 public:
195  using FIRRTLType::TypeBase<ConcreteType, ParentType,
196  detail::WidthTypeStorage>::Base::Base;
197 
198  /// Return the bitwidth of this type or None if unknown.
199  Optional<int32_t> getWidth() {
200  return static_cast<ConcreteType *>(this)->getWidth();
201  }
202 
203  /// Return the width of this type, or -1 if it has none specified.
204  int32_t getWidthOrSentinel() {
205  auto width = getWidth();
206  return width.hasValue() ? width.getValue() : -1;
207  }
208 
209  /// Return true if this type has a known width.
210  bool hasWidth() { return getWidth().hasValue(); }
211 
212  /// Return a new type with the width changed to a different value.
213  ConcreteType changeWidth(int32_t width) {
214  return ConcreteType::get(static_cast<ConcreteType *>(this)->getContext(),
215  width);
216  }
217 };

290 /// BundleType is an aggregate of named elements. This is effectively a struct
291 /// for FIRRTL.
292 class BundleType : public FIRRTLType::TypeBase<BundleType, FIRRTLType,
293  detail::BundleTypeStorage> {
294 public:
295  using Base::Base;
296 
297  // Each element of a bundle, which is a name and type.
298  struct BundleElement {
299  StringAttr name;
300  bool isFlip;
301  FIRRTLType type;
302 
303  BundleElement(StringAttr name, bool isFlip, FIRRTLType type)
304  : name(name), isFlip(isFlip), type(type) {}
305 
306  bool operator==(const BundleElement &rhs) const {
307  return name == rhs.name && isFlip == rhs.isFlip && type == rhs.type;
308  }
309  bool operator!=(const BundleElement &rhs) const { return !operator==(rhs); }
310  };
311 
312  static FIRRTLType get(ArrayRef<BundleElement> elements, MLIRContext *context);
313 
314  ArrayRef<BundleElement> getElements() const;
315 
316  size_t getNumElements() { return getElements().size(); }
317 
318  /// Look up an element's index by name. This returns None on failure.
319  llvm::Optional<unsigned> getElementIndex(StringAttr name);
320  llvm::Optional<unsigned> getElementIndex(StringRef name);
321 
322  /// Look up an element's name by index. This asserts if index is invalid.
323  StringRef getElementName(size_t index);
324 
325  /// Look up an element by name. This returns None on failure.
326  llvm::Optional<BundleElement> getElement(StringAttr name);
327  llvm::Optional<BundleElement> getElement(StringRef name);
328 
329  /// Look up an element by index. This asserts if index is invalid.
330  BundleElement getElement(size_t index);
331 
332  /// Look up an element type by name.
333  FIRRTLType getElementType(StringAttr name);
334  FIRRTLType getElementType(StringRef name);
335 
336  /// Look up an element type by index.
337  FIRRTLType getElementType(size_t index);
338 
339  /// Return the recursive properties of the type.
340  RecursiveTypeProperties getRecursiveTypeProperties();
341 
342  /// Return this type with any flip types recursively removed from itself.
343  FIRRTLType getPassiveType();
344 
345  /// Get an integer ID for the field. Field IDs start at 1, and are assigned
346  /// to each field in a bundle in a recursive pre-order walk of all fields,
347  /// visiting all nested bundle fields. A field ID of 0 is used to reference
348  /// the bundle itself. The ID can be used to uniquely identify any specific
349  /// field in this bundle.
350  unsigned getFieldID(unsigned index);
351 
352  /// Find the element index corresponding to the desired fieldID. If the
353  /// fieldID corresponds to a field in a nested bundle, it will return the
354  /// index of the parent field.
355  unsigned getIndexForFieldID(unsigned fieldID);
356 
357  /// Strip off a single layer of this type and return the sub-type and a field
358  /// ID targeting the same field, but rebased on the sub-type.
359  std::pair<FIRRTLType, unsigned> getSubTypeByFieldID(unsigned fieldID);
360 
361  /// Get the maximum field ID in this bundle. This is helpful for constructing
362  /// field IDs when this BundleType is nested in another aggregate type.
363  unsigned getMaxFieldID();
364 
365  /// Returns the effective field id when treating the index field as the root
366  /// of the type. Essentially maps a fieldID to a fieldID after a subfield op.
367  /// Returns the new id and whether the id is in the given child.
368  std::pair<unsigned, bool> rootChildFieldID(unsigned fieldID, unsigned index);
369 
370  using iterator = ArrayRef<BundleElement>::iterator;
371  iterator begin() const { return getElements().begin(); }
372  iterator end() const { return getElements().end(); }
373 };
