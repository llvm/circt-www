28 /// Emit a connect between two values.
29 void circt::firrtl::emitConnect(ImplicitLocOpBuilder &builder, Value dst,
30  Value src) {
31  auto dstType = dst.getType().cast<FIRRTLType>();
32  auto srcType = src.getType().cast<FIRRTLType>();
33 
34  // If the types are the exact same we can just connect them.
35  if (dstType == srcType) {
36  // Strict connect does not allow uninferred widths.
37  if (dstType.hasUninferredWidth())
38  builder.create<ConnectOp>(dst, src);
39  else
40  builder.create<StrictConnectOp>(dst, src);
41  return;
42  }
43 
44  if (auto dstBundle = dstType.dyn_cast<BundleType>()) {
45  // Connect all the bundle elements pairwise.
46  auto numElements = dstBundle.getNumElements();
47  // Check if we are trying to create an illegal connect - just create the
48  // connect and let the verifier catch it.
49  auto srcBundle = srcType.dyn_cast<BundleType>();
50  if (!srcBundle || numElements != srcBundle.getNumElements()) {
51  builder.create<ConnectOp>(dst, src);
52  return;
53  }
54  for (size_t i = 0; i < numElements; ++i) {
55  auto dstField = builder.create<SubfieldOp>(dst, i);
56  auto srcField = builder.create<SubfieldOp>(src, i);
57  if (dstBundle.getElement(i).isFlip)
58  std::swap(dstField, srcField);
59  emitConnect(builder, dstField, srcField);
60  }
61  return;
62  }
63 
64  if (auto dstVector = dstType.dyn_cast<FVectorType>()) {
65  // Connect all the vector elements pairwise.
66  auto numElements = dstVector.getNumElements();
67  // Check if we are trying to create an illegal connect - just create the
68  // connect and let the verifier catch it.
69  auto srcVector = srcType.dyn_cast<FVectorType>();
70  if (!srcVector || numElements != srcVector.getNumElements()) {
71  builder.create<ConnectOp>(dst, src);
72  return;
73  }
74  for (size_t i = 0; i < numElements; ++i) {
75  auto dstField = builder.create<SubindexOp>(dst, i);
76  auto srcField = builder.create<SubindexOp>(src, i);
77  emitConnect(builder, dstField, srcField);
78  }
79  return;
80  }
81 
82  // Handle ground types with possibly uninferred widths.
83  auto dstWidth = dstType.getBitWidthOrSentinel();
84  auto srcWidth = srcType.getBitWidthOrSentinel();
85  if (dstWidth < 0 || srcWidth < 0) {
86  // If one of these types has an uninferred width, we connect them with a
87  // regular connect operation.
88  builder.create<ConnectOp>(dst, src);
89  return;
90  }
91 
92  // The source must be extended or truncated.
93  if (dstWidth < srcWidth) {
94  // firrtl.tail always returns uint even for sint operands.
95  IntType tmpType = dstType.cast<IntType>();
96  if (tmpType.isSigned())
97  tmpType = UIntType::get(dstType.getContext(), dstWidth);
98  src = builder.create<TailPrimOp>(tmpType, src, srcWidth - dstWidth);
99  // Insert the cast back to signed if needed.
100  if (tmpType != dstType)
101  src = builder.create<AsSIntPrimOp>(dstType, src);
102  } else if (srcWidth < dstWidth) {
103  // Need to extend arg.
104  src = builder.create<PadPrimOp>(src, dstWidth);
105  }
106 
107  // Strict connect requires the types to be completely equal, including
108  // connecting uint<1> to abstract reset types.
109  if (dstType == src.getType())
110  builder.create<StrictConnectOp>(dst, src);
111  else
112  builder.create<ConnectOp>(dst, src);
113 }

134 /// Return the value that drives another FIRRTL value within module scope. Only
135 /// look backwards through one connection. This is intended to be used in
136 /// situations where you only need to look at the most recent connect, e.g., to
137 /// know if a wire has been driven to a constant. Return null if no driver via
138 /// a connect was found.
139 Value circt::firrtl::getDriverFromConnect(Value val) {
140  for (auto *user : val.getUsers()) {
141  if (auto connect = dyn_cast<FConnectLike>(user)) {
142  if (connect.dest() != val)
143  continue;
144  return connect.src();
145  }
146  }
147  return nullptr;
148 }

150 /// Return the value that drives another FIRRTL value within module scope. This
151 /// is parameterized by looking through or not through certain constructs. This
152 /// assumes a single driver and should only be run after `ExpandWhens`.
153 Value circt::firrtl::getModuleScopedDriver(Value val, bool lookThroughWires,
154  bool lookThroughNodes,
155  bool lookThroughCasts) {
156  // Update `val` to the source of the connection driving `thisVal`. This walks
157  // backwards across users to find the first connection and updates `val` to
158  // the source. This assumes that only one connect is driving `thisVal`, i.e.,
159  // this pass runs after `ExpandWhens`.
160  auto updateVal = [&](Value thisVal) {
161  for (auto *user : thisVal.getUsers()) {
162  if (auto connect = dyn_cast<FConnectLike>(user)) {
163  if (connect.dest() != val)
164  continue;
165  val = connect.src();
166  return;
167  }
168  }
169  val = nullptr;
170  return;
171  };
172 
173  while (val) {
174  // The value is a port.
175  if (auto blockArg = val.dyn_cast<BlockArgument>()) {
176  FModuleOp op = cast<FModuleOp>(val.getParentBlock()->getParentOp());
177  auto direction = op.getPortDirection(blockArg.getArgNumber());
178  // Base case: this is one of the module's input ports.
179  if (direction == Direction::In)
180  return blockArg;
181  updateVal(blockArg);
182  continue;
183  }
184 
185  auto *op = val.getDefiningOp();
186 
187  // The value is an instance port.
188  if (auto inst = dyn_cast<InstanceOp>(op)) {
189  auto resultNo = val.cast<OpResult>().getResultNumber();
190  // Base case: this is an instance's output port.
191  if (inst.getPortDirection(resultNo) == Direction::Out)
192  return inst.getResult(resultNo);
193  updateVal(val);
194  continue;
195  }
196 
197  // If told to look through wires, continue from the driver of the wire.
198  if (lookThroughWires && isa<WireOp>(op)) {
199  updateVal(op->getResult(0));
200  continue;
201  }
202 
203  // If told to look through nodes, continue from the node input.
204  if (lookThroughNodes && isa<NodeOp>(op)) {
205  val = cast<NodeOp>(op).input();
206  continue;
207  }
208 
209  if (lookThroughCasts &&
210  isa<AsUIntPrimOp, AsSIntPrimOp, AsClockPrimOp, AsAsyncResetPrimOp>(
211  op)) {
212  val = op->getOperand(0);
213  continue;
214  }
215 
216  // Look through unary ops generated by emitConnect
217  if (isa<PadPrimOp, TailPrimOp>(op)) {
218  val = op->getOperand(0);
219  continue;
220  }
221 
222  // Base case: this is a constant/invalid or primop.
223  //
224  // TODO: If needed, this could be modified to look through unary ops which
225  // have an unambiguous single driver. This should only be added if a need
226  // arises for it.
227  break;
228  };
229  return val;
230 }

236 FieldRef circt::firrtl::getFieldRefFromValue(Value value) {
237  // This code walks upwards from the subfield and calculates the field ID at
238  // each level. At each stage, it must take the current id, and re-index it as
239  // a nested bundle under the parent field.. This is accomplished by using the
240  // parent field's ID as a base, and adding the field ID of the child.
241  unsigned id = 0;
242  while (value) {
243  Operation *op = value.getDefiningOp();
244 
245  // If this is a block argument, we are done.
246  if (!op)
247  break;
248 
249  if (auto subfieldOp = dyn_cast<SubfieldOp>(op)) {
250  value = subfieldOp.input();
251  auto bundleType = value.getType().cast<BundleType>();
252  // Rebase the current index on the parent field's index.
253  id += bundleType.getFieldID(subfieldOp.fieldIndex());
254  } else if (auto subindexOp = dyn_cast<SubindexOp>(op)) {
255  value = subindexOp.input();
256  auto vecType = value.getType().cast<FVectorType>();
257  // Rebase the current index on the parent field's index.
258  id += vecType.getFieldID(subindexOp.index());
259  } else {
260  break;
261  }
262  }
263  return {value, id};
264 }

266 /// Get the string name of a value which is a direct child of a declaration op.
267 static void getDeclName(Value value, SmallString<64> &string) {
268  if (auto arg = value.dyn_cast<BlockArgument>()) {
269  // Get the module ports and get the name.
270  auto module = cast<FModuleOp>(arg.getOwner()->getParentOp());
271  SmallVector<PortInfo> ports = module.getPorts();
272  string += ports[arg.getArgNumber()].name.getValue();
273  return;
274  }
275 
276  auto *op = value.getDefiningOp();
277  TypeSwitch<Operation *>(op)
278  .Case<InstanceOp, MemOp>([&](auto op) {
279  string += op.name();
280  string += ".";
281  string +=
282  op.getPortName(value.cast<OpResult>().getResultNumber()).getValue();
283  })
284  .Case<WireOp, RegOp, RegResetOp>([&](auto op) { string += op.name(); });
285 }

292 std::string circt::firrtl::getFieldName(const FieldRef &fieldRef,
293  bool &rootKnown) {
294  SmallString<64> name;
295  auto value = fieldRef.getValue();
296  getDeclName(value, name);
297  rootKnown = !name.empty();
298 
299  auto type = value.getType();
300  auto localID = fieldRef.getFieldID();
301  while (localID) {
302  if (auto bundleType = type.dyn_cast<BundleType>()) {
303  auto index = bundleType.getIndexForFieldID(localID);
304  // Add the current field string, and recurse into a subfield.
305  auto &element = bundleType.getElements()[index];
306  if (!name.empty())
307  name += ".";
308  name += element.name.getValue();
309  // Recurse in to the element type.
310  type = element.type;
311  localID = localID - bundleType.getFieldID(index);
312  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
313  auto index = vecType.getIndexForFieldID(localID);
314  name += "[";
315  name += std::to_string(index);
316  name += "]";
317  // Recurse in to the element type.
318  type = vecType.getElementType();
319  localID = localID - vecType.getFieldID(index);
320  } else {
321  // If we reach here, the field ref is pointing inside some aggregate type
322  // that isn't a bundle or a vector. If the type is a ground type, then the
323  // localID should be 0 at this point, and we should have broken from the
324  // loop.
325  llvm_unreachable("unsupported type");
326  }
327  }
328 
329  return name.str().str();
330 }

332 /// This gets the value targeted by a field id. If the field id is targeting
333 /// the value itself, it returns it unchanged. If it is targeting a single field
334 /// in a aggregate value, such as a bundle or vector, this will create the
335 /// necessary subaccesses to get the value.
336 Value circt::firrtl::getValueByFieldID(ImplicitLocOpBuilder builder,
337  Value value, unsigned fieldID) {
338  // When the fieldID hits 0, we've found the target value.
339  while (fieldID != 0) {
340  auto type = value.getType();
341  if (auto bundle = type.dyn_cast<BundleType>()) {
342  auto index = bundle.getIndexForFieldID(fieldID);
343  value = builder.create<SubfieldOp>(value, index);
344  fieldID -= bundle.getFieldID(index);
345  } else {
346  auto vector = type.cast<FVectorType>();
347  auto index = vector.getIndexForFieldID(fieldID);
348  value = builder.create<SubindexOp>(value, index);
349  fieldID -= vector.getFieldID(index);
350  }
351  }
352  return value;
353 }

416 /// Parse a string that may encode a FIRRTL location into a LocationAttr.
417 std::pair<bool, Optional<mlir::LocationAttr>>
418 circt::firrtl::maybeStringToLocation(StringRef spelling, bool skipParsing,
419  StringAttr &locatorFilenameCache,
420  FileLineColLoc &fileLineColLocCache,
421  MLIRContext *context) {
422  // The spelling of the token looks something like "@[Decoupled.scala 221:8]".
423  if (!spelling.startswith("@[") || !spelling.endswith("]"))
424  return {false, None};
425 
426  spelling = spelling.drop_front(2).drop_back(1);
427 
428  // Decode the locator in "spelling", returning the filename and filling in
429  // lineNo and colNo on success. On failure, this returns an empty filename.
430  auto decodeLocator = [&](StringRef input, unsigned &resultLineNo,
431  unsigned &resultColNo) -> StringRef {
432  // Split at the last space.
433  auto spaceLoc = input.find_last_of(' ');
434  if (spaceLoc == StringRef::npos)
435  return {};
436 
437  auto filename = input.take_front(spaceLoc);
438  auto lineAndColumn = input.drop_front(spaceLoc + 1);
439 
440  // Decode the line/column. If the colon is missing, then it will be empty
441  // here.
442  StringRef lineStr, colStr;
443  std::tie(lineStr, colStr) = lineAndColumn.split(':');
444 
445  // Decode the line number and the column number if present.
446  if (lineStr.getAsInteger(10, resultLineNo))
447  return {};
448  if (!colStr.empty()) {
449  if (colStr.front() != '{') {
450  if (colStr.getAsInteger(10, resultColNo))
451  return {};
452  } else {
453  // compound locator, just parse the first part for now
454  if (colStr.drop_front().split(',').first.getAsInteger(10, resultColNo))
455  return {};
456  }
457  }
458  return filename;
459  };
460 
461  // Decode the locator spelling, reporting an error if it is malformed.
462  unsigned lineNo = 0, columnNo = 0;
463  StringRef filename = decodeLocator(spelling, lineNo, columnNo);
464  if (filename.empty())
465  return {false, None};
466 
467  // If info locators are ignored, don't actually apply them. We still do all
468  // the verification above though.
469  if (skipParsing)
470  return {true, None};
471 
472  /// Return an FileLineColLoc for the specified location, but use a bit of
473  /// caching to reduce thrasing the MLIRContext.
474  auto getFileLineColLoc = [&](StringRef filename, unsigned lineNo,
475  unsigned columnNo) -> FileLineColLoc {
476  // Check our single-entry cache for this filename.
477  StringAttr filenameId = locatorFilenameCache;
478  if (filenameId.str() != filename) {
479  // We missed! Get the right identifier.
480  locatorFilenameCache = filenameId = StringAttr::get(context, filename);
481 
482  // If we miss in the filename cache, we also miss in the FileLineColLoc
483  // cache.
484  return fileLineColLocCache =
485  FileLineColLoc::get(filenameId, lineNo, columnNo);
486  }
487 
488  // If we hit the filename cache, check the FileLineColLoc cache.
489  auto result = fileLineColLocCache;
490  if (result && result.getLine() == lineNo && result.getColumn() == columnNo)
491  return result;
492 
493  return fileLineColLocCache =
494  FileLineColLoc::get(filenameId, lineNo, columnNo);
495  };
496 
497  // Compound locators will be combined with spaces, like:
498  // @[Foo.scala 123:4 Bar.scala 309:14]
499  // and at this point will be parsed as a-long-string-with-two-spaces at
500  // 309:14. We'd like to parse this into two things and represent it as an
501  // MLIR fused locator, but we want to be conservatively safe for filenames
502  // that have a space in it. As such, we are careful to make sure we can
503  // decode the filename/loc of the result. If so, we accumulate results,
504  // backward, in this vector.
505  SmallVector<Location> extraLocs;
506  auto spaceLoc = filename.find_last_of(' ');
507  while (spaceLoc != StringRef::npos) {
508  // Try decoding the thing before the space. Validates that there is another
509  // space and that the file/line can be decoded in that substring.
510  unsigned nextLineNo = 0, nextColumnNo = 0;
511  auto nextFilename =
512  decodeLocator(filename.take_front(spaceLoc), nextLineNo, nextColumnNo);
513 
514  // On failure we didn't have a joined locator.
515  if (nextFilename.empty())
516  break;
517 
518  // On success, remember what we already parsed (Bar.Scala / 309:14), and
519  // move on to the next chunk.
520  auto loc =
521  getFileLineColLoc(filename.drop_front(spaceLoc + 1), lineNo, columnNo);
522  extraLocs.push_back(loc);
523  filename = nextFilename;
524  lineNo = nextLineNo;
525  columnNo = nextColumnNo;
526  spaceLoc = filename.find_last_of(' ');
527  }
528 
529  mlir::LocationAttr result = getFileLineColLoc(filename, lineNo, columnNo);
530  if (!extraLocs.empty()) {
531  extraLocs.push_back(result);
532  std::reverse(extraLocs.begin(), extraLocs.end());
533  result = FusedLoc::get(context, extraLocs);
534  }
535  return {true, result};
536 }
