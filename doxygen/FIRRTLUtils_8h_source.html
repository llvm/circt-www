39 /// Return true if a value is module-scoped driven by a value of a specific
40 /// type.
41 template <typename A, typename... B>
42 static bool isModuleScopedDrivenBy(Value val, bool lookThroughWires,
43  bool lookThroughNodes,
44  bool lookThroughCasts) {
45  val = getModuleScopedDriver(val, lookThroughWires, lookThroughNodes,
46  lookThroughCasts);
47 
48  if (!val)
49  return false;
50 
51  auto *op = val.getDefiningOp();
52  if (!op)
53  return false;
54 
55  return isa<A, B...>(op);
56 }
