23 /// ExprVisitor is a visitor for FIRRTL expression nodes.
24 template <typename ConcreteType, typename ResultType = void,
25  typename... ExtraArgs>
26 class ExprVisitor {
27 public:
28  ResultType dispatchExprVisitor(Operation *op, ExtraArgs... args) {
29  auto *thisCast = static_cast<ConcreteType *>(this);
30  return TypeSwitch<Operation *, ResultType>(op)
31  // Basic Expressions
32  .template Case<
33  ConstantOp, SpecialConstantOp, InvalidValueOp, SubfieldOp,
34  SubindexOp, SubaccessOp, MultibitMuxOp,
35  // Arithmetic and Logical Binary Primitives.
36  AddPrimOp, SubPrimOp, MulPrimOp, DivPrimOp, RemPrimOp, AndPrimOp,
37  OrPrimOp, XorPrimOp,
38  // Comparisons.
39  LEQPrimOp, LTPrimOp, GEQPrimOp, GTPrimOp, EQPrimOp, NEQPrimOp,
40  // Misc Binary Primitives.
41  CatPrimOp, DShlPrimOp, DShlwPrimOp, DShrPrimOp,
42  // Unary operators.
43  AsSIntPrimOp, AsUIntPrimOp, AsAsyncResetPrimOp, AsClockPrimOp,
44  CvtPrimOp, NegPrimOp, NotPrimOp, AndRPrimOp, OrRPrimOp, XorRPrimOp,
45  // Miscellaneous.
46  BitsPrimOp, HeadPrimOp, MuxPrimOp, PadPrimOp, ShlPrimOp, ShrPrimOp,
47  TailPrimOp, VerbatimExprOp, HWStructCastOp, BitCastOp,
48  mlir::UnrealizedConversionCastOp>([&](auto expr) -> ResultType {
49  return thisCast->visitExpr(expr, args...);
50  })
51  .Default([&](auto expr) -> ResultType {
52  return thisCast->visitInvalidExpr(op, args...);
53  });
54  }
55 
56  /// This callback is invoked on any non-expression operations.
57  ResultType visitInvalidExpr(Operation *op, ExtraArgs... args) {
58  op->emitOpError("unknown FIRRTL expression");
59  abort();
60  }
61 
62  /// This callback is invoked on any expression operations that are not handled
63  /// by the concrete visitor.
64  ResultType visitUnhandledExpr(Operation *op, ExtraArgs... args) {
65  return ResultType();
66  }
67 
68  /// This fallback is invoked on any unary expr that isn't explicitly handled.
69  /// The default implementation delegates to the unhandled expression fallback.
70  ResultType visitUnaryExpr(Operation *op, ExtraArgs... args) {
71  return static_cast<ConcreteType *>(this)->visitUnhandledExpr(op, args...);
72  }
73 
74  /// This fallback is invoked on any binary expr that isn't explicitly handled.
75  /// The default implementation delegates to the unhandled expression fallback.
76  ResultType visitBinaryExpr(Operation *op, ExtraArgs... args) {
77  return static_cast<ConcreteType *>(this)->visitUnhandledExpr(op, args...);
78  }
79 
80 #define HANDLE(OPTYPE, OPKIND) \
81  ResultType visitExpr(OPTYPE op, ExtraArgs... args) { \
82  return static_cast<ConcreteType *>(this)->visit##OPKIND##Expr(op, \
83  args...); \
84  }
85 
86  // Basic expressions.
87  HANDLE(ConstantOp, Unhandled);
88  HANDLE(SpecialConstantOp, Unhandled);
89  HANDLE(SubfieldOp, Unhandled);
90  HANDLE(SubindexOp, Unhandled);
91  HANDLE(SubaccessOp, Unhandled);
92  HANDLE(MultibitMuxOp, Unhandled);
93 
94  // Arithmetic and Logical Binary Primitives.
95  HANDLE(AddPrimOp, Binary);
96  HANDLE(SubPrimOp, Binary);
97  HANDLE(MulPrimOp, Binary);
98  HANDLE(DivPrimOp, Binary);
99  HANDLE(RemPrimOp, Binary);
100  HANDLE(AndPrimOp, Binary);
101  HANDLE(OrPrimOp, Binary);
102  HANDLE(XorPrimOp, Binary);
103 
104  // Comparisons.
105  HANDLE(LEQPrimOp, Binary);
106  HANDLE(LTPrimOp, Binary);
107  HANDLE(GEQPrimOp, Binary);
108  HANDLE(GTPrimOp, Binary);
109  HANDLE(EQPrimOp, Binary);
110  HANDLE(NEQPrimOp, Binary);
111 
112  // Misc Binary Primitives.
113  HANDLE(CatPrimOp, Binary);
114  HANDLE(DShlPrimOp, Binary);
115  HANDLE(DShlwPrimOp, Binary);
116  HANDLE(DShrPrimOp, Binary);
117 
118  // Unary operators.
119  HANDLE(AsSIntPrimOp, Unary);
120  HANDLE(AsUIntPrimOp, Unary);
121  HANDLE(AsAsyncResetPrimOp, Unary);
122  HANDLE(AsClockPrimOp, Unary);
123  HANDLE(CvtPrimOp, Unary);
124  HANDLE(NegPrimOp, Unary);
125  HANDLE(NotPrimOp, Unary);
126  HANDLE(AndRPrimOp, Unary);
127  HANDLE(OrRPrimOp, Unary);
128  HANDLE(XorRPrimOp, Unary);
129 
130  // Miscellaneous.
131  HANDLE(BitsPrimOp, Unhandled);
132  HANDLE(HeadPrimOp, Unhandled);
133  HANDLE(InvalidValueOp, Unhandled);
134  HANDLE(MuxPrimOp, Unhandled);
135  HANDLE(PadPrimOp, Unhandled);
136  HANDLE(ShlPrimOp, Unhandled);
137  HANDLE(ShrPrimOp, Unhandled);
138  HANDLE(TailPrimOp, Unhandled);
139  HANDLE(VerbatimExprOp, Unhandled);
140 
141  // Conversions.
142  HANDLE(HWStructCastOp, Unhandled);
143  HANDLE(mlir::UnrealizedConversionCastOp, Unhandled);
144  HANDLE(BitCastOp, Unhandled);
145 #undef HANDLE
146 };

148 /// ExprVisitor is a visitor for FIRRTL statement nodes.
149 template <typename ConcreteType, typename ResultType = void,
150  typename... ExtraArgs>
151 class StmtVisitor {
152 public:
153  ResultType dispatchStmtVisitor(Operation *op, ExtraArgs... args) {
154  auto *thisCast = static_cast<ConcreteType *>(this);
155  return TypeSwitch<Operation *, ResultType>(op)
156  .template Case<AttachOp, ConnectOp, StrictConnectOp, ForceOp, PrintFOp,
157  SkipOp, StopOp, WhenOp, AssertOp, AssumeOp, CoverOp,
158  ProbeOp>([&](auto opNode) -> ResultType {
159  return thisCast->visitStmt(opNode, args...);
160  })
161  .Default([&](auto expr) -> ResultType {
162  return thisCast->visitInvalidStmt(op, args...);
163  });
164  }
165 
166  /// This callback is invoked on any non-Stmt operations.
167  ResultType visitInvalidStmt(Operation *op, ExtraArgs... args) {
168  op->emitOpError("unknown firrtl stmt");
169  abort();
170  }
171 
172  /// This callback is invoked on any Stmt operations that are not handled
173  /// by the concrete visitor.
174  ResultType visitUnhandledStmt(Operation *op, ExtraArgs... args) {
175  return ResultType();
176  }
177 
178 #define HANDLE(OPTYPE) \
179  ResultType visitStmt(OPTYPE op, ExtraArgs... args) { \
180  return static_cast<ConcreteType *>(this)->visitUnhandledStmt(op, args...); \
181  }
182 
183  HANDLE(AttachOp);
184  HANDLE(ConnectOp);
185  HANDLE(StrictConnectOp);
186  HANDLE(ForceOp);
187  HANDLE(PrintFOp);
188  HANDLE(SkipOp);
189  HANDLE(StopOp);
190  HANDLE(WhenOp);
191  HANDLE(AssertOp);
192  HANDLE(AssumeOp);
193  HANDLE(CoverOp);
194  HANDLE(ProbeOp);
195 
196 #undef HANDLE
197 };

199 /// ExprVisitor is a visitor for FIRRTL declaration nodes.
200 template <typename ConcreteType, typename ResultType = void,
201  typename... ExtraArgs>
202 class DeclVisitor {
203 public:
204  ResultType dispatchDeclVisitor(Operation *op, ExtraArgs... args) {
205  auto *thisCast = static_cast<ConcreteType *>(this);
206  return TypeSwitch<Operation *, ResultType>(op)
207  .template Case<InstanceOp, MemOp, NodeOp, RegOp, RegResetOp, WireOp,
208  VerbatimWireOp>([&](auto opNode) -> ResultType {
209  return thisCast->visitDecl(opNode, args...);
210  })
211  .Default([&](auto expr) -> ResultType {
212  return thisCast->visitInvalidDecl(op, args...);
213  });
214  }
215 
216  /// This callback is invoked on any non-Decl operations.
217  ResultType visitInvalidDecl(Operation *op, ExtraArgs... args) {
218  op->emitOpError("unknown firrtl decl");
219  abort();
220  }
221 
222  /// This callback is invoked on any Decl operations that are not handled
223  /// by the concrete visitor.
224  ResultType visitUnhandledDecl(Operation *op, ExtraArgs... args) {
225  return ResultType();
226  }
227 
228 #define HANDLE(OPTYPE) \
229  ResultType visitDecl(OPTYPE op, ExtraArgs... args) { \
230  return static_cast<ConcreteType *>(this)->visitUnhandledDecl(op, args...); \
231  }
232 
233  HANDLE(InstanceOp);
234  HANDLE(MemOp);
235  HANDLE(NodeOp);
236  HANDLE(RegOp);
237  HANDLE(RegResetOp);
238  HANDLE(WireOp);
239  HANDLE(VerbatimWireOp);
240 #undef HANDLE
241 };

243 /// FIRRTLVisitor allows you to visit all of the expr/stmt/decls with one class
244 /// declaration.
245 ///
246 /// Clients call dispatchVisitor to invoke the dispatch, and may implement
247 /// visitInvalidOp() to get notified about non-FIRRTL dialect nodes and
248 /// visitUnhandledOp() to get notified about FIRRTL dialect ops that are not
249 /// handled specifically.
250 template <typename ConcreteType, typename ResultType = void,
251  typename... ExtraArgs>
252 class FIRRTLVisitor
253  : public ExprVisitor<ConcreteType, ResultType, ExtraArgs...>,
254  public StmtVisitor<ConcreteType, ResultType, ExtraArgs...>,
255  public DeclVisitor<ConcreteType, ResultType, ExtraArgs...> {
256 public:
257  /// This is the main entrypoint for the FIRRTLVisitor.
258  ResultType dispatchVisitor(Operation *op, ExtraArgs... args) {
259  return this->dispatchExprVisitor(op, args...);
260  }
261 
262  // Chain from each visitor onto the next one.
263  ResultType visitInvalidExpr(Operation *op, ExtraArgs... args) {
264  return this->dispatchStmtVisitor(op, args...);
265  }
266  ResultType visitInvalidStmt(Operation *op, ExtraArgs... args) {
267  return this->dispatchDeclVisitor(op, args...);
268  }
269  ResultType visitInvalidDecl(Operation *op, ExtraArgs... args) {
270  return static_cast<ConcreteType *>(this)->visitInvalidOp(op, args...);
271  }
272 
273  // Default to chaining visitUnhandledXXX to visitUnhandledOp.
274  ResultType visitUnhandledExpr(Operation *op, ExtraArgs... args) {
275  return static_cast<ConcreteType *>(this)->visitUnhandledOp(op, args...);
276  }
277  ResultType visitUnhandledStmt(Operation *op, ExtraArgs... args) {
278  return static_cast<ConcreteType *>(this)->visitUnhandledOp(op, args...);
279  }
280  ResultType visitUnhandledDecl(Operation *op, ExtraArgs... args) {
281  return static_cast<ConcreteType *>(this)->visitUnhandledOp(op, args...);
282  }
283 
284  /// visitInvalidOp is an override point for non-FIRRTL dialect operations.
285  ResultType visitInvalidOp(Operation *op, ExtraArgs... args) {
286  return ResultType();
287  }
288 
289  /// visitUnhandledOp is an override point for FIRRTL dialect ops that the
290  /// concrete visitor didn't bother to implement.
291  ResultType visitUnhandledOp(Operation *op, ExtraArgs... args) {
292  return ResultType();
293  }
294 };
