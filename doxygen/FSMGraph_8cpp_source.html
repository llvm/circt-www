46 FSMGraph::FSMGraph(Operation *op) {
47  machine = dyn_cast<MachineOp>(op);
48  assert(machine && "Expected a fsm::MachineOp");
49 
50  // Find all states in the machine.
51  for (auto stateOp : machine.getOps<StateOp>()) {
52  // Add an edge to indicate that this state transitions to some other state.
53  auto *currentStateNode = getOrAddState(stateOp);
54 
55  for (auto transitionOp : stateOp.transitions().getOps<TransitionOp>()) {
56  auto *nextStateNode = getOrAddState(transitionOp.getNextState());
57  currentStateNode->addTransitionEdge(nextStateNode, transitionOp);
58  }
59  }
60 }

91 FSMTransitionEdge *FSMGraph::createTransition(OpBuilder &builder, Location loc,
92  StateOp from, StateOp to) {
93  auto *currentStateNode = getOrAddState(from);
94  auto *nextStateNode = getOrAddState(to);
95  OpBuilder::InsertionGuard g(builder);
96  // Set the insertion point to the end of the transitions. This will always be
97  // right before the implicit fsm.output terminator operation. @todo: the
98  // transition region should obviously not have a terminator, but this needs a
99  // significant change in the ODS/parser of the op.
100  builder.setInsertionPoint(
101  *from.transitions().getBlocks().front().getOps<OutputOp>().begin());
102  auto transition = builder.create<TransitionOp>(loc, to);
103  return currentStateNode->addTransitionEdge(nextStateNode, transition);
