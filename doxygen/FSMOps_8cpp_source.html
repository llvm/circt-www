25 void MachineOp::build(OpBuilder &builder, OperationState &state, StringRef name,
26  StringRef initialStateName, FunctionType type,
27  ArrayRef<NamedAttribute> attrs,
28  ArrayRef<DictionaryAttr> argAttrs) {
29  state.addAttribute(mlir::SymbolTable::getSymbolAttrName(),
30  builder.getStringAttr(name));
31  state.addAttribute(getTypeAttrName(), TypeAttr::get(type));
32  state.addAttribute("initialState",
33  StringAttr::get(state.getContext(), initialStateName));
34  state.attributes.append(attrs.begin(), attrs.end());
35  Region *region = state.addRegion();
36  Block *body = new Block();
37  region->push_back(body);
38  body->addArguments(
39  type.getInputs(),
40  SmallVector<Location, 4>(type.getNumInputs(), builder.getUnknownLoc()));
41 
42  if (argAttrs.empty())
43  return;
44  assert(type.getNumInputs() == argAttrs.size());
45  function_interface_impl::addArgAndResultAttrs(builder, state, argAttrs,
46  /*resultAttrs=*/llvm::None);
47 }

93 static LogicalResult compareTypes(TypeRange rangeA, TypeRange rangeB) {
94  if (rangeA.size() != rangeB.size())
95  return failure();
96 
97  int64_t index = 0;
98  for (auto zip : llvm::zip(rangeA, rangeB)) {
99  if (std::get<0>(zip) != std::get<1>(zip))
100  return failure();
101  ++index;
102  }
103 
104  return success();
105 }

107 LogicalResult MachineOp::verify() {
108  // If this function is external there is nothing to do.
109  if (isExternal())
110  return success();
111 
112  // Verify that the argument list of the function and the arg list of the entry
113  // block line up. The trait already verified that the number of arguments is
114  // the same between the signature and the block.
115  if (failed(compareTypes(getArgumentTypes(), front().getArgumentTypes())))
116  return emitOpError(
117  "entry block argument types must match the machine input types");
118 
119  // Verify that the machine only has one block terminated with OutputOp.
120  if (!llvm::hasSingleElement(*this))
121  return emitOpError("must only have a single block");
122 
123  // Verify that the initial state exists
124  if (!getInitialStateOp())
125  return emitOpError("initial state '" + initialState() +
126  "' was not defined in the machine");
127 
128  return success();
129 }

158 template <typename OpType>
159 static LogicalResult verifyCallerTypes(OpType op) {
160  auto machine = op.getMachine();
161  if (!machine)
162  return op.emitError("cannot find machine definition");
163 
164  // Check operand types first.
165  if (failed(
166  compareTypes(machine.getArgumentTypes(), op.inputs().getTypes()))) {
167  auto diag =
168  op.emitOpError("operand types must match the machine input types");
169  diag.attachNote(machine->getLoc()) << "original machine declared here";
170  return failure();
171  }
172 
173  // Check result types.
174  if (failed(compareTypes(machine.getResultTypes(), op.outputs().getTypes()))) {
175  auto diag =
176  op.emitOpError("result types must match the machine output types");
177  diag.attachNote(machine->getLoc()) << "original machine declared here";
178  return failure();
179  }
180 
181  return success();
182 }

229 LogicalResult StateOp::canonicalize(StateOp op, PatternRewriter &rewriter) {
230  bool hasAlwaysTakenTransition = false;
231  SmallVector<TransitionOp, 4> transitionsToErase;
232  // Remove all transitions after an "always-taken" transition.
233  for (auto transition : op.transitions().getOps<TransitionOp>()) {
234  if (!hasAlwaysTakenTransition)
235  hasAlwaysTakenTransition = transition.isAlwaysTaken();
236  else
237  transitionsToErase.push_back(transition);
238  }
239 
240  for (auto transition : transitionsToErase)
241  rewriter.eraseOp(transition);
242 
243  return failure(transitionsToErase.empty());
244 }

304 bool TransitionOp::isAlwaysTaken() {
305  if (!hasGuard())
306  return true;
307 
308  auto guardReturn = getGuardReturn();
309  if (guardReturn.getNumOperands() == 0)
310  return true;
311 
312  if (auto constantOp =
313  guardReturn.getOperand(0).getDefiningOp<mlir::arith::ConstantOp>())
314  return constantOp.getValue().cast<BoolAttr>().getValue();
315 
316  return false;
317 }

319 LogicalResult TransitionOp::canonicalize(TransitionOp op,
320  PatternRewriter &rewriter) {
321  if (op.hasGuard()) {
322  auto guardReturn = op.getGuardReturn();
323  if (guardReturn.getNumOperands() == 1)
324  if (auto constantOp = guardReturn.getOperand(0)
325  .getDefiningOp<mlir::arith::ConstantOp>()) {
326  // Simplify when the guard region returns a constant value.
327  if (constantOp.getValue().cast<BoolAttr>().getValue()) {
328  // Replace the original return op with a new one without any operands
329  // if the constant is TRUE.
330  rewriter.setInsertionPoint(guardReturn);
331  rewriter.create<fsm::ReturnOp>(guardReturn.getLoc());
332  rewriter.eraseOp(guardReturn);
333  } else {
334  // Erase the whole transition op if the constant is FALSE, because the
335  // transition will never be taken.
336  rewriter.eraseOp(op);
337  }
338  return success();
339  }
340  }
341 
342  return failure();
343 }

345 LogicalResult TransitionOp::verify() {
346  if (!getNextState())
347  return emitOpError("cannot find the definition of the next state `")
348  << nextState() << "`";
349 
350  // Verify the action region, if present.
351  if (hasGuard()) {
352  if (guard().front().empty() ||
353  !isa_and_nonnull<fsm::ReturnOp>(&guard().front().back()))
354  return emitOpError("guard region must terminate with a ReturnOp");
355  }
356 
357  // Verify the transition is located in the correct region.
358  if ((*this)->getParentRegion() != &getCurrentState().transitions())
359  return emitOpError("must only be located in the transitions region");
360 
361  return success();
362 }
