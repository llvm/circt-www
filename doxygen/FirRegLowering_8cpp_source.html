23 // Reimplemented from SliceAnalysis to use a worklist rather than recursion and
24 // non-insert ordered set.
25 static void
26 getForwardSliceSimple(Operation *root,
27  llvm::DenseSet<Operation *> &forwardSlice,
28  llvm::function_ref<bool(Operation *)> filter = nullptr) {
29  SmallVector<Operation *> worklist({root});
30 
31  while (!worklist.empty()) {
32  Operation *op = worklist.pop_back_val();
33 
34  if (!op)
35  continue;
36 
37  if (filter && !filter(op))
38  continue;
39 
40  for (Region &region : op->getRegions())
41  for (Block &block : region)
42  for (Operation &blockOp : block)
43  if (forwardSlice.insert(&blockOp).second)
44  worklist.push_back(&blockOp);
45  for (Value result : op->getResults())
46  for (Operation *userOp : result.getUsers())
47  if (forwardSlice.insert(userOp).second)
48  worklist.push_back(userOp);
49 
50  forwardSlice.insert(op);
51  }
52 }

54 void FirRegLowering::addToIfBlock(OpBuilder &builder, Value cond,
55  const std::function<void()> &trueSide,
56  const std::function<void()> &falseSide) {
57  auto op = ifCache.lookup({builder.getBlock(), cond});
58  // Always build both sides of the if, in case we want to use an empty else
59  // later. This way we don't have to build a new if and replace it.
60  if (!op) {
61  auto newIfOp =
62  builder.create<sv::IfOp>(cond.getLoc(), cond, trueSide, falseSide);
63  ifCache.insert({{builder.getBlock(), cond}, newIfOp});
64  } else {
65  OpBuilder::InsertionGuard guard(builder);
66  builder.setInsertionPointToEnd(op.getThenBlock());
67  trueSide();
68  builder.setInsertionPointToEnd(op.getElseBlock());
69  falseSide();
70  }

73 void FirRegLowering::lower() {
74  // Find all registers to lower in the module.
75  auto regs = module.getOps<seq::FirRegOp>();
76  if (regs.empty())
77  return;
78 
79  // Lower the regs to SV regs. Group them by initializer and reset kind.
80  SmallVector<RegLowerInfo> randomInit, presetInit;
81  llvm::MapVector<Value, SmallVector<RegLowerInfo>> asyncResets;
82  for (auto reg : llvm::make_early_inc_range(regs)) {
83  auto svReg = lower(reg);
84  if (svReg.preset)
85  presetInit.push_back(svReg);
86  else if (!disableRegRandomization)
87  randomInit.push_back(svReg);
88 
89  if (svReg.asyncResetSignal)
90  asyncResets[svReg.asyncResetSignal].emplace_back(svReg);
91  }
92 
93  // Compute total width of random space. Place non-chisel registers at the end
94  // of the space. The Random space is unique to the initial block, due to
95  // verilog thread rules, so we can drop trailing random calls if they are
96  // unused.
97  uint64_t maxBit = 0;
98  for (auto reg : randomInit)
99  if (reg.randStart >= 0)
100  maxBit = std::max(maxBit, (uint64_t)reg.randStart + reg.width);
101 
102  for (auto &reg : randomInit) {
103  if (reg.randStart == -1) {
104  reg.randStart = maxBit;
105  maxBit += reg.width;
106  }
107  }
108 
109  // Create an initial block at the end of the module where random
110  // initialisation will be inserted. Create two builders into the two
111  // `ifdef` ops where the registers will be placed.
112  //
113  // `ifndef SYNTHESIS
114  // `ifdef RANDOMIZE_REG_INIT
115  // ... regBuilder ...
116  // `endif
117  // initial
118  // `INIT_RANDOM_PROLOG_
119  // ... initBuilder ..
120  // `endif
121  if (randomInit.empty() && presetInit.empty() && asyncResets.empty())
122  return;
123 
124  needsRandom = true;
125 
126  auto loc = module.getLoc();
127  MLIRContext *context = module.getContext();
128  auto randInitRef = sv::MacroIdentAttr::get(context, "RANDOMIZE_REG_INIT");
129 
130  auto builder =
131  ImplicitLocOpBuilder::atBlockTerminator(loc, module.getBodyBlock());
132 
133  builder.create<sv::IfDefOp>("ENABLE_INITIAL_REG_", [&] {
134  builder.create<sv::OrderedOutputOp>([&] {
135  builder.create<sv::IfDefOp>("FIRRTL_BEFORE_INITIAL", [&] {
136  builder.create<sv::VerbatimOp>("`FIRRTL_BEFORE_INITIAL");
137  });
138 
139  builder.create<sv::InitialOp>([&] {
140  if (!randomInit.empty()) {
141  builder.create<sv::IfDefProceduralOp>("INIT_RANDOM_PROLOG_", [&] {
142  builder.create<sv::VerbatimOp>("`INIT_RANDOM_PROLOG_");
143  });
144  builder.create<sv::IfDefProceduralOp>(randInitRef, [&] {
145  // Create randomization vector
146  SmallVector<Value> randValues;
147  auto numRandomCalls = (maxBit + 31) / 32;
148  auto logic = builder.create<sv::LogicOp>(
149  loc,
150  hw::UnpackedArrayType::get(builder.getIntegerType(32),
151  numRandomCalls),
152  "_RANDOM");
153  // Indvar's width must be equal to `ceil(log2(numRandomCalls +
154  // 1))` to avoid overflow.
155  auto inducionVariableWidth = llvm::Log2_64_Ceil(numRandomCalls + 1);
156  auto arrayIndexWith = llvm::Log2_64_Ceil(numRandomCalls);
157  auto lb =
158  getOrCreateConstant(loc, APInt::getZero(inducionVariableWidth));
159  auto ub = getOrCreateConstant(
160  loc, APInt(inducionVariableWidth, numRandomCalls));
161  auto step =
162  getOrCreateConstant(loc, APInt(inducionVariableWidth, 1));
163  auto forLoop = builder.create<sv::ForOp>(
164  loc, lb, ub, step, "i", [&](BlockArgument iter) {
165  auto rhs = builder.create<sv::MacroRefExprSEOp>(
166  loc, builder.getIntegerType(32), "RANDOM");
167  Value iterValue = iter;
168  if (!iter.getType().isInteger(arrayIndexWith))
169  iterValue = builder.create<comb::ExtractOp>(
170  loc, iterValue, 0, arrayIndexWith);
171  auto lhs = builder.create<sv::ArrayIndexInOutOp>(loc, logic,
172  iterValue);
173  builder.create<sv::BPAssignOp>(loc, lhs, rhs);
174  });
175  builder.setInsertionPointAfter(forLoop);
176  for (uint64_t x = 0; x < numRandomCalls; ++x) {
177  auto lhs = builder.create<sv::ArrayIndexInOutOp>(
178  loc, logic,
179  getOrCreateConstant(loc, APInt(arrayIndexWith, x)));
180  randValues.push_back(lhs.getResult());
181  }
182 
183  // Create initialisers for all registers.
184  for (auto &svReg : randomInit)
185  initialize(builder, svReg, randValues);
186  });
187  }
188 
189  if (!presetInit.empty()) {
190  for (auto &svReg : presetInit) {
191  auto loc = svReg.reg.getLoc();
192  auto elemTy = svReg.reg.getType().getElementType();
193  auto cst = getOrCreateConstant(loc, svReg.preset.getValue());
194 
195  Value rhs;
196  if (cst.getType() == elemTy)
197  rhs = cst;
198  else
199  rhs = builder.create<hw::BitcastOp>(loc, elemTy, cst);
200 
201  builder.create<sv::BPAssignOp>(loc, svReg.reg, rhs);
202  }
203  }
204 
205  if (!asyncResets.empty()) {
206  // If the register is async reset, we need to insert extra
207  // initialization in post-randomization so that we can set the
208  // reset value to register if the reset signal is enabled.
209  for (auto &reset : asyncResets) {
210  // if (reset) begin
211  // ..
212  // end
213  builder.create<sv::IfOp>(reset.first, [&] {
214  for (auto &reg : reset.second)
215  builder.create<sv::BPAssignOp>(reg.reg.getLoc(), reg.reg,
216  reg.asyncResetValue);
217  });
218  }
219  }
220  });
221 
222  builder.create<sv::IfDefOp>("FIRRTL_AFTER_INITIAL", [&] {
223  builder.create<sv::VerbatimOp>("`FIRRTL_AFTER_INITIAL");
224  });
225  });
226  });
227 
228  module->removeAttr("firrtl.random_init_width");
229 }

231 // Return true if two arguments are equivalent, or if both of them are the same
232 // array indexing.
233 // NOLINTNEXTLINE(misc-no-recursion)
234 static bool areEquivalentValues(Value term, Value next) {
235  if (term == next)
236  return true;
237  // Check whether these values are equivalent array accesses with constant
238  // index. We have to check the equivalence recursively because they might not
239  // be CSEd.
240  if (auto t1 = term.getDefiningOp<hw::ArrayGetOp>())
241  if (auto t2 = next.getDefiningOp<hw::ArrayGetOp>())
242  if (auto c1 = t1.getIndex().getDefiningOp<hw::ConstantOp>())
243  if (auto c2 = t2.getIndex().getDefiningOp<hw::ConstantOp>())
244  return c1.getType() == c2.getType() &&
245  c1.getValue() == c2.getValue() &&
246  areEquivalentValues(t1.getInput(), t2.getInput());
247  // Otherwise, regard as different.
248  // TODO: Handle struct if necessary.
249  return false;
250 }

252 static llvm::SetVector<Value> extractConditions(Value value) {
253  auto andOp = value.getDefiningOp<comb::AndOp>();
254  // If the value is not AndOp with a bin flag, use it as a condition.
255  if (!andOp || !andOp.getTwoState()) {
256  llvm::SetVector<Value> ret;
257  ret.insert(value);
258  return ret;
259  }
260 
261  return llvm::SetVector<Value>(andOp.getOperands().begin(),
262  andOp.getOperands().end());
263 }

272 // Return a tuple <cond, idx, val> if the array register update can be
273 // represented with a dynamic index assignment:
274 // if (cond)
275 // reg[idx] <= val;
276 //
277 std::optional<std::tuple<Value, Value, Value>>
278 FirRegLowering::tryRestoringSubaccess(OpBuilder &builder, Value reg, Value term,
279  hw::ArrayCreateOp nextRegValue) {
280  Value trueVal;
281  SmallVector<Value> muxConditions;
282  // Compat fix for GCC12's libstdc++, cannot use
283  // llvm::enumerate(llvm::reverse(OperandRange)). See #4900.
284  SmallVector<Value> reverseOpValues(llvm::reverse(nextRegValue.getOperands()));
285  if (!llvm::all_of(llvm::enumerate(reverseOpValues), [&](auto idxAndValue) {
286  // Check that `nextRegValue[i]` is `cond_i ? val : reg[i]`.
287  auto [i, value] = idxAndValue;
288  auto mux = value.template getDefiningOp<comb::MuxOp>();
289  // Ensure that mux has binary flag.
290  if (!mux || !mux.getTwoState())
291  return false;
292  // The next value must be same.
293  if (trueVal && trueVal != mux.getTrueValue())
294  return false;
295  if (!trueVal)
296  trueVal = mux.getTrueValue();
297  muxConditions.push_back(mux.getCond());
298  // Check that ith element is an element of the register we are
299  // currently lowering.
300  auto arrayGet =
301  mux.getFalseValue().template getDefiningOp<hw::ArrayGetOp>();
302  if (!arrayGet)
303  return false;
304  return areEquivalentValues(arrayGet.getInput(), term) &&
305  getConstantValue(arrayGet.getIndex()) == i;
306  }))
307  return {};
308 
309  // Extract common expressions among mux conditions.
310  llvm::SetVector<Value> commonConditions =
311  extractConditions(muxConditions.front());
312  for (auto condition : ArrayRef(muxConditions).drop_front()) {
313  auto cond = extractConditions(condition);
314  commonConditions.remove_if([&](auto v) { return !cond.contains(v); });
315  }
316  Value indexValue;
317  for (auto [idx, condition] : llvm::enumerate(muxConditions)) {
318  llvm::SetVector<Value> extractedConditions = extractConditions(condition);
319  // Remove common conditions and check the remaining condition is only an
320  // index comparision.
321  extractedConditions.remove_if(
322  [&](auto v) { return commonConditions.contains(v); });
323  if (extractedConditions.size() != 1)
324  return {};
325 
326  auto indexCompare =
327  (*extractedConditions.begin()).getDefiningOp<comb::ICmpOp>();
328  if (!indexCompare || !indexCompare.getTwoState() ||
329  indexCompare.getPredicate() != comb::ICmpPredicate::eq)
330  return {};
331  // `IndexValue` must be same.
332  if (indexValue && indexValue != indexCompare.getLhs())
333  return {};
334  if (!indexValue)
335  indexValue = indexCompare.getLhs();
336  if (getConstantValue(indexCompare.getRhs()) != idx)
337  return {};
338  }
339 
340  OpBuilder::InsertionGuard guard(builder);
341  builder.setInsertionPointAfterValue(reg);
342  Value commonConditionValue;
343  if (commonConditions.empty())
344  commonConditionValue = getOrCreateConstant(reg.getLoc(), APInt(1, 1));
345  else
346  commonConditionValue = builder.createOrFold<comb::AndOp>(
347  reg.getLoc(), builder.getI1Type(), commonConditions.takeVector(), true);
348  return std::make_tuple(commonConditionValue, indexValue, trueVal);
349 }

351 void FirRegLowering::createTree(OpBuilder &builder, Value reg, Value term,
352  Value next) {
353  // Get the fanout from this register before we build the tree. While we are
354  // creating the tree of if/else statements from muxes, we only want to turn
355  // muxes that are on the register's fanout into if/else statements. This is
356  // required to get the correct enable inference. But other muxes in the tree
357  // should be left as ternary operators. This is desirable because we don't
358  // want to create if/else structure for logic unrelated to the register's
359  // enable.
360  auto firReg = term.getDefiningOp<seq::FirRegOp>();
361  DenseSet<Operation *> regMuxFanout;
362  getForwardSliceSimple(firReg, regMuxFanout, [&](Operation *op) {
363  return op == firReg || !isa<sv::RegOp, seq::FirRegOp, hw::InstanceOp>(op);
364  });
365 
366  SmallVector<std::tuple<Block *, Value, Value, Value>> worklist;
367  auto addToWorklist = [&](Value reg, Value term, Value next) {
368  worklist.push_back({builder.getBlock(), reg, term, next});
369  };
370 
371  auto getArrayIndex = [&](Value reg, Value idx) {
372  // Create an array index op just after `reg`.
373  OpBuilder::InsertionGuard guard(builder);
374  builder.setInsertionPointAfterValue(reg);
375  return builder.create<sv::ArrayIndexInOutOp>(reg.getLoc(), reg, idx);
376  };
377 
378  SmallVector<Value, 8> opsToDelete;
379  addToWorklist(reg, term, next);
380  while (!worklist.empty()) {
381  OpBuilder::InsertionGuard guard(builder);
382  Block *block;
383  Value reg, term, next;
384  std::tie(block, reg, term, next) = worklist.pop_back_val();
385  builder.setInsertionPointToEnd(block);
386  if (areEquivalentValues(term, next))
387  continue;
388 
389  // If this is a two-state mux within the fanout from the register, we use
390  // if/else structure for proper enable inference.
391  auto mux = next.getDefiningOp<comb::MuxOp>();
392  if (mux && mux.getTwoState() && regMuxFanout.contains(mux)) {
393  addToIfBlock(
394  builder, mux.getCond(),
395  [&]() { addToWorklist(reg, term, mux.getTrueValue()); },
396  [&]() { addToWorklist(reg, term, mux.getFalseValue()); });
397  continue;
398  }
399  // If the next value is an array creation, split the value into
400  // invidial elements and construct trees recursively.
401  if (auto array = next.getDefiningOp<hw::ArrayCreateOp>()) {
402  // First, try restoring subaccess assignments.
403  if (auto matchResultOpt =
404  tryRestoringSubaccess(builder, reg, term, array)) {
405  Value cond, index, trueValue;
406  std::tie(cond, index, trueValue) = *matchResultOpt;
407  addToIfBlock(
408  builder, cond,
409  [&]() {
410  Value nextReg = getArrayIndex(reg, index);
411  // Create a value to use for equivalence checking in the
412  // recursive calls. Add the value to `opsToDelete` so that it can
413  // be deleted afterwards.
414  auto termElement =
415  builder.create<hw::ArrayGetOp>(term.getLoc(), term, index);
416  opsToDelete.push_back(termElement);
417  addToWorklist(nextReg, termElement, trueValue);
418  },
419  []() {});
420  ++numSubaccessRestored;
421  continue;
422  }
423  // Compat fix for GCC12's libstdc++, cannot use
424  // llvm::enumerate(llvm::reverse(OperandRange)). See #4900.
425  // SmallVector<Value> reverseOpValues(llvm::reverse(array.getOperands()));
426  for (auto [idx, value] : llvm::enumerate(array.getOperands())) {
427  idx = array.getOperands().size() - idx - 1;
428  // Create an index constant.
429  auto idxVal = getOrCreateConstant(
430  array.getLoc(),
431  APInt(std::max(1u, llvm::Log2_64_Ceil(array.getOperands().size())),
432  idx));
433 
434  auto &index = arrayIndexCache[{reg, idx}];
435  if (!index)
436  index = getArrayIndex(reg, idxVal);
437 
438  // Create a value to use for equivalence checking in the
439  // recursive calls. Add the value to `opsToDelete` so that it can
440  // be deleted afterwards.
441  auto termElement =
442  builder.create<hw::ArrayGetOp>(term.getLoc(), term, idxVal);
443  opsToDelete.push_back(termElement);
444  addToWorklist(index, termElement, value);
445  }
446  continue;
447  }
448 
449  builder.create<sv::PAssignOp>(term.getLoc(), reg, next);
450  }
451 
452  while (!opsToDelete.empty()) {
453  auto value = opsToDelete.pop_back_val();
454  assert(value.use_empty());
455  value.getDefiningOp()->erase();
456  }
457 }

459 FirRegLowering::RegLowerInfo FirRegLowering::lower(FirRegOp reg) {
460  Location loc = reg.getLoc();
461  Type regTy = typeConverter.convertType(reg.getType());
462 
463  ImplicitLocOpBuilder builder(reg.getLoc(), reg);
464  RegLowerInfo svReg{nullptr, reg.getPresetAttr(), nullptr, nullptr, -1, 0};
465  svReg.reg = builder.create<sv::RegOp>(loc, regTy, reg.getNameAttr());
466  svReg.width = hw::getBitWidth(regTy);
467 
468  if (auto attr = reg->getAttrOfType<IntegerAttr>("firrtl.random_init_start"))
469  svReg.randStart = attr.getUInt();
470 
471  // Don't move these over
472  reg->removeAttr("firrtl.random_init_start");
473 
474  // Move Attributes
475  svReg.reg->setDialectAttrs(reg->getDialectAttrs());
476 
477  if (auto innerSymAttr = reg.getInnerSymAttr())
478  svReg.reg.setInnerSymAttr(innerSymAttr);
479 
480  auto regVal = builder.create<sv::ReadInOutOp>(loc, svReg.reg);
481 
482  if (reg.hasReset()) {
483  addToAlwaysBlock(
484  module.getBodyBlock(), sv::EventControl::AtPosEdge, reg.getClk(),
485  [&](OpBuilder &b) {
486  // If this is an AsyncReset, ensure that we emit a self connect to
487  // avoid erroneously creating a latch construct.
488  if (reg.getIsAsync() && areEquivalentValues(reg, reg.getNext()))
489  b.create<sv::PAssignOp>(reg.getLoc(), svReg.reg, reg);
490  else
491  createTree(b, svReg.reg, reg, reg.getNext());
492  },
493  reg.getIsAsync() ? ResetType::AsyncReset : ResetType::SyncReset,
494  sv::EventControl::AtPosEdge, reg.getReset(),
495  [&](OpBuilder &builder) {
496  builder.create<sv::PAssignOp>(loc, svReg.reg, reg.getResetValue());
497  });
498  if (reg.getIsAsync()) {
499  svReg.asyncResetSignal = reg.getReset();
500  svReg.asyncResetValue = reg.getResetValue();
501  }
502  } else {
503  addToAlwaysBlock(
504  module.getBodyBlock(), sv::EventControl::AtPosEdge, reg.getClk(),
505  [&](OpBuilder &b) { createTree(b, svReg.reg, reg, reg.getNext()); });
506  }
507 
508  reg.replaceAllUsesWith(regVal.getResult());
509  reg.erase();
510 
511  return svReg;
512 }

514 // Initialize registers by assigning each element recursively instead of
515 // initializing entire registers. This is necessary as a workaround for
516 // verilator which allocates many local variables for concat op.
517 // NOLINTBEGIN(misc-no-recursion)
518 void FirRegLowering::initializeRegisterElements(Location loc,
519  OpBuilder &builder, Value reg,
520  Value randomSource,
521  unsigned &pos) {
522  auto type = reg.getType().cast<sv::InOutType>().getElementType();
523  if (auto intTy = hw::type_dyn_cast<IntegerType>(type)) {
524  // Use randomSource[pos-1:pos-width] as a random value.
525  pos -= intTy.getWidth();
526  auto elem = builder.createOrFold<comb::ExtractOp>(loc, randomSource, pos,
527  intTy.getWidth());
528  builder.create<sv::BPAssignOp>(loc, reg, elem);
529  } else if (auto array = hw::type_dyn_cast<hw::ArrayType>(type)) {
530  for (unsigned i = 0, e = array.getNumElements(); i < e; ++i) {
531  auto index = getOrCreateConstant(loc, APInt(llvm::Log2_64_Ceil(e), i));
532  initializeRegisterElements(
533  loc, builder, builder.create<sv::ArrayIndexInOutOp>(loc, reg, index),
534  randomSource, pos);
535  }
536  } else if (auto structType = hw::type_dyn_cast<hw::StructType>(type)) {
537  for (auto e : structType.getElements())
538  initializeRegisterElements(
539  loc, builder,
540  builder.create<sv::StructFieldInOutOp>(loc, reg, e.name),
541  randomSource, pos);
542  } else {
543  assert(false && "unsupported type");
544  }
545 }

548 void FirRegLowering::initialize(OpBuilder &builder, RegLowerInfo reg,
549  ArrayRef<Value> rands) {
550  auto loc = reg.reg.getLoc();
551  SmallVector<Value> nibbles;
552  if (reg.width == 0)
553  return;
554 
555  uint64_t width = reg.width;
556  uint64_t offset = reg.randStart;
557  while (width) {
558  auto index = offset / 32;
559  auto start = offset % 32;
560  auto nwidth = std::min(32 - start, width);
561  auto elemVal = builder.create<sv::ReadInOutOp>(loc, rands[index]);
562  auto elem =
563  builder.createOrFold<comb::ExtractOp>(loc, elemVal, start, nwidth);
564  nibbles.push_back(elem);
565  offset += nwidth;
566  width -= nwidth;
567  }
568  auto concat = builder.createOrFold<comb::ConcatOp>(loc, nibbles);
569  unsigned pos = reg.width;
570  // Initialize register elements.
571  initializeRegisterElements(loc, builder, reg.reg, concat, pos);
572 }

574 void FirRegLowering::addToAlwaysBlock(
575  Block *block, sv::EventControl clockEdge, Value clock,
576  const std::function<void(OpBuilder &)> &body, ::ResetType resetStyle,
577  sv::EventControl resetEdge, Value reset,
578  const std::function<void(OpBuilder &)> &resetBody) {
579  auto loc = clock.getLoc();
580  auto builder = ImplicitLocOpBuilder::atBlockTerminator(loc, block);
581  AlwaysKeyType key{builder.getBlock(), clockEdge, clock,
582  resetStyle, resetEdge, reset};
583 
584  sv::AlwaysOp alwaysOp;
585  sv::IfOp insideIfOp;
586  if (!emitSeparateAlwaysBlocks) {
587  std::tie(alwaysOp, insideIfOp) = alwaysBlocks[key];
588  }
589 
590  if (!alwaysOp) {
591  if (reset) {

593  // Here, we want to create the following structure with sv.always and
