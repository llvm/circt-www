26 LogicalResult firtool::populatePreprocessTransforms(mlir::PassManager &pm,
27  const FirtoolOptions &opt) {
28  // Legalize away "open" aggregates to hw-only versions.
29  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerOpenAggsPass());
30 
31  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createResolvePathsPass());
32 
33  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerFIRRTLAnnotationsPass(
34  opt.shouldDisableUnknownAnnotations(),
35  opt.shouldDisableClasslessAnnotations(),
36  opt.shouldLowerNoRefTypePortAnnotations()));
37 
38  if (opt.shouldEnableDebugInfo())
39  pm.nest<firrtl::CircuitOp>().addNestedPass<firrtl::FModuleOp>(
40  firrtl::createMaterializeDebugInfoPass());
41 
42  pm.nest<firrtl::CircuitOp>().addPass(
43  firrtl::createLowerIntrinsicsPass(opt.shouldFixupEICGWrapper()));
44 
45  return success();
46 }

48 LogicalResult firtool::populateCHIRRTLToLowFIRRTL(mlir::PassManager &pm,
49  const FirtoolOptions &opt,
50  StringRef inputFilename) {
51  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerSignaturesPass());
52 
53  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInjectDUTHierarchyPass());
54 
55  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
56  firrtl::createPassiveWiresPass());
57 
58  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
59  firrtl::createDropNamesPass(opt.getPreserveMode()));
60 
61  if (!opt.shouldDisableOptimization())
62  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
63  mlir::createCSEPass());
64 
65  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
66  firrtl::createLowerCHIRRTLPass());
67 
68  // Run LowerMatches before InferWidths, as the latter does not support the
69  // match statement, but it does support what they lower to.
70  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
71  firrtl::createLowerMatchesPass());
72 
73  // Width inference creates canonicalization opportunities.
74  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInferWidthsPass());
75 
76  pm.nest<firrtl::CircuitOp>().addPass(
77  firrtl::createMemToRegOfVecPass(opt.shouldReplicateSequentialMemories(),
78  opt.shouldIgnoreReadEnableMemories()));
79 
80  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInferResetsPass());
81 
82  if (opt.shouldExportChiselInterface()) {
83  StringRef outdir = opt.getChiselInterfaceOutputDirectory();
84  if (opt.isDefaultOutputFilename() && outdir.empty()) {
85  pm.nest<firrtl::CircuitOp>().addPass(createExportChiselInterfacePass());
86  } else {
87  if (outdir.empty())
88  outdir = opt.getOutputFilename();
89  pm.nest<firrtl::CircuitOp>().addPass(
90  createExportSplitChiselInterfacePass(outdir));
91  }
92  }
93 
94  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createDropConstPass());
95 
96  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createHoistPassthroughPass(
97  /*hoistHWDrivers=*/!opt.shouldDisableOptimization() &&
98  !opt.shouldDisableHoistingHWPassthrough()));
99  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createProbeDCEPass());
100 
101  if (opt.shouldDedup())
102  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createDedupPass());
103 
104  if (opt.shouldConvertVecOfBundle()) {
105  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createLowerFIRRTLTypesPass(
106  firrtl::PreserveAggregate::All, firrtl::PreserveAggregate::All));
107  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createVBToBVPass());
108  }
109 
110  if (!opt.shouldLowerMemories())
111  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
112  firrtl::createFlattenMemoryPass());
113 
114  // The input mlir file could be firrtl dialect so we might need to clean
115  // things up.
116  // pm.addNestedPass<firrtl::CircuitOp>(firrtl::createLowerSignaturesPass());
117  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createLowerFIRRTLTypesPass(
118  opt.getPreserveAggregate(), firrtl::PreserveAggregate::None));
119 
120  pm.nest<firrtl::CircuitOp>().nestAny().addPass(
121  firrtl::createExpandWhensPass());
122 
123  auto &modulePM = pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>();
124  modulePM.addPass(firrtl::createSFCCompatPass());
125  modulePM.addPass(firrtl::createLayerMergePass());
126  modulePM.addPass(firrtl::createLayerSinkPass());
127 
128  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerLayersPass());
129 
130  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInlinerPass());
131 
132  // Preset the random initialization parameters for each module. The current
133  // implementation assumes it can run at a time where every register is
134  // currently in the final module it will be emitted in, all registers have
135  // been created, and no registers have yet been removed.
136  if (opt.isRandomEnabled(FirtoolOptions::RandomKind::Reg))
137  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
138  firrtl::createRandomizeRegisterInitPass());
139 
140  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createCheckCombLoopsPass());
141 
142  // If we parsed a FIRRTL file and have optimizations enabled, clean it up.
143  if (!opt.shouldDisableOptimization())
144  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
145  createSimpleCanonicalizerPass());
146 
147  // Run the infer-rw pass, which merges read and write ports of a memory with
148  // mutually exclusive enables.
149  if (!opt.shouldDisableOptimization())
150  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
151  firrtl::createInferReadWritePass());
152 
153  if (opt.shouldReplicateSequentialMemories())
154  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerMemoryPass());
155 
156  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createPrefixModulesPass());
157 
158  if (!opt.shouldDisableOptimization()) {
159  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createIMConstPropPass());
160 
161  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createHoistPassthroughPass(
162  /*hoistHWDrivers=*/!opt.shouldDisableOptimization() &&
163  !opt.shouldDisableHoistingHWPassthrough()));
164  // Cleanup after hoisting passthroughs, for separation-of-concerns.
165  pm.addPass(firrtl::createIMDeadCodeElimPass());
166  }
167 
168  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createAddSeqMemPortsPass());
169 
170  pm.addPass(firrtl::createCreateSiFiveMetadataPass(
171  opt.shouldReplicateSequentialMemories(),
172  opt.getReplaceSequentialMemoriesFile()));
173 
174  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createExtractInstancesPass());
175 
176  // Run passes to resolve Grand Central features. This should run before
177  // BlackBoxReader because Grand Central needs to inform BlackBoxReader where
178  // certain black boxes should be placed. Note: all Grand Central Taps related
179  // collateral is resolved entirely by LowerAnnotations.
180  pm.addNestedPass<firrtl::CircuitOp>(
181  firrtl::createGrandCentralPass(opt.getCompanionMode()));
182 
183  // Read black box source files into the IR.
184  StringRef blackBoxRoot = opt.getBlackBoxRootPath().empty()
185  ? llvm::sys::path::parent_path(inputFilename)
186  : opt.getBlackBoxRootPath();
187  pm.nest<firrtl::CircuitOp>().addPass(
188  firrtl::createBlackBoxReaderPass(blackBoxRoot));
189 
190  // Run SymbolDCE as late as possible, but before InnerSymbolDCE. This is for
191  // hierpathop's and just for general cleanup.
192  pm.addNestedPass<firrtl::CircuitOp>(mlir::createSymbolDCEPass());
193 
194  // Run InnerSymbolDCE as late as possible, but before IMDCE.
195  pm.addPass(firrtl::createInnerSymbolDCEPass());
196 
197  // The above passes, IMConstProp in particular, introduce additional
198  // canonicalization opportunities that we should pick up here before we
199  // proceed to output-specific pipelines.
200  if (!opt.shouldDisableOptimization()) {
201  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
202  createSimpleCanonicalizerPass());
203  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
204  circt::firrtl::createRegisterOptimizerPass());
205  // Re-run IMConstProp to propagate constants produced by register
206  // optimizations.
207  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createIMConstPropPass());
208  pm.addPass(firrtl::createIMDeadCodeElimPass());
209  }
210 
211  if (opt.shouldEmitOMIR())
212  pm.nest<firrtl::CircuitOp>().addPass(
213  firrtl::createEmitOMIRPass(opt.getOmirOutputFile()));
214 
215  // Always run this, required for legalization.
216  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
217  firrtl::createMergeConnectionsPass(
218  !opt.shouldDisableAggressiveMergeConnections()));
219 
220  if (!opt.shouldDisableOptimization())
221  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
222  firrtl::createVectorizationPass());
223 
224  return success();
225 }

227 LogicalResult firtool::populateLowFIRRTLToHW(mlir::PassManager &pm,
228  const FirtoolOptions &opt) {
229  // Remove TraceAnnotations and write their updated paths to an output
230  // annotation file.
231  pm.nest<firrtl::CircuitOp>().addPass(
232  firrtl::createResolveTracesPass(opt.getOutputAnnotationFilename()));
233 
234  // Lower the ref.resolve and ref.send ops and remove the RefType ports.
235  // LowerToHW cannot handle RefType so, this pass must be run to remove all
236  // RefType ports and ops.
237  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerXMRPass());
238 
239  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerClassesPass());
240 
241  // Check for static asserts.
242  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
243  circt::firrtl::createLintingPass());
244 
245  pm.addPass(createLowerFIRRTLToHWPass(opt.shouldEnableAnnotationWarning(),
246  opt.shouldEmitChiselAssertsAsSVA()));
247 
248  if (!opt.shouldDisableOptimization()) {
249  auto &modulePM = pm.nest<hw::HWModuleOp>();
250  modulePM.addPass(mlir::createCSEPass());
251  modulePM.addPass(createSimpleCanonicalizerPass());
252  }
253 
254  // Check inner symbols and inner refs.
255  pm.addPass(hw::createVerifyInnerRefNamespacePass());
256 
257  return success();
258 }

260 LogicalResult firtool::populateHWToSV(mlir::PassManager &pm,
261  const FirtoolOptions &opt) {
262  if (opt.shouldExtractTestCode())
263  pm.addPass(sv::createSVExtractTestCodePass(
264  opt.shouldEtcDisableInstanceExtraction(),
265  opt.shouldEtcDisableRegisterExtraction(),
266  opt.shouldEtcDisableModuleInlining()));
267 
268  pm.addPass(seq::createExternalizeClockGatePass(opt.getClockGateOptions()));
269  pm.addPass(circt::createLowerSimToSVPass());
270  pm.addPass(circt::createLowerSeqToSVPass(
271  {/*disableRegRandomization=*/!opt.isRandomEnabled(
272  FirtoolOptions::RandomKind::Reg),
273  /*disableMemRandomization=*/
274  !opt.isRandomEnabled(FirtoolOptions::RandomKind::Mem),
275  /*emitSeparateAlwaysBlocks=*/
276  opt.shouldEmitSeparateAlwaysBlocks()}));
277  pm.addNestedPass<hw::HWModuleOp>(createLowerVerifToSVPass());
278  pm.addPass(seq::createHWMemSimImplPass(
279  {/*disableMemRandomization=*/!opt.isRandomEnabled(
280  FirtoolOptions::RandomKind::Mem),
281  /*disableRegRandomization=*/
282  !opt.isRandomEnabled(FirtoolOptions::RandomKind::Reg),
283  /*replSeqMem=*/opt.shouldReplicateSequentialMemories(),
284  /*readEnableMode=*/opt.shouldIgnoreReadEnableMemories()
285  ? seq::ReadEnableMode::Ignore
286  : seq::ReadEnableMode::Undefined,
287  /*addMuxPragmas=*/opt.shouldAddMuxPragmas(),
288  /*addVivadoRAMAddressConflictSynthesisBugWorkaround=*/
289  opt.shouldAddVivadoRAMAddressConflictSynthesisBugWorkaround()}));
290 
291  // If enabled, run the optimizer.
292  if (!opt.shouldDisableOptimization()) {
293  auto &modulePM = pm.nest<hw::HWModuleOp>();
294  modulePM.addPass(mlir::createCSEPass());
295  modulePM.addPass(createSimpleCanonicalizerPass());
296  modulePM.addPass(mlir::createCSEPass());
297  modulePM.addPass(sv::createHWCleanupPass(
298  /*mergeAlwaysBlocks=*/!opt.shouldEmitSeparateAlwaysBlocks()));
299  }
300 
301  // Check inner symbols and inner refs.
302  pm.addPass(hw::createVerifyInnerRefNamespacePass());
303 
304  return success();
305 }

308 LogicalResult
309 populatePrepareForExportVerilog(mlir::PassManager &pm,
310  const firtool::FirtoolOptions &opt) {
311 
312  // Legalize unsupported operations within the modules.
313  pm.nest<hw::HWModuleOp>().addPass(sv::createHWLegalizeModulesPass());
314 
315  // Tidy up the IR to improve verilog emission quality.
316  if (!opt.shouldDisableOptimization())
317  pm.nest<hw::HWModuleOp>().addPass(sv::createPrettifyVerilogPass());
318 
319  if (opt.shouldStripFirDebugInfo())
320  pm.addPass(circt::createStripDebugInfoWithPredPass([](mlir::Location loc) {
321  if (auto fileLoc = loc.dyn_cast<FileLineColLoc>())
322  return fileLoc.getFilename().getValue().ends_with(".fir");
323  return false;
324  }));
325 
326  if (opt.shouldStripDebugInfo())
327  pm.addPass(circt::createStripDebugInfoWithPredPass(
328  [](mlir::Location loc) { return true; }));
329 
330  // Emit module and testbench hierarchy JSON files.
331  if (opt.shouldExportModuleHierarchy())
332  pm.addPass(sv::createHWExportModuleHierarchyPass());
333 
334  // Check inner symbols and inner refs.
335  pm.addPass(hw::createVerifyInnerRefNamespacePass());
336 
337  return success();
338 }
