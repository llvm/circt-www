36 //===- Terminology and useful information ---------------------------------===//
37 //
38 // For SignalDriver annotations, "local" refers to the subcircuit that
39 // drives/probes values in the main "remote" circuit.
40 //
41 // Driven values ("sinks"), are set via 'force XMR = input_port;' statements.
42 // Probes ("sources"), are read via 'assign output_port = XMR;' statements.
43 //
44 // The end goal is a mappings module that is generated and instantiated in the
45 // subcircuit which interacts with targeted values in the main circuit.
46 //
47 //===- Flow ---------------------------------------------------------------===//
48 //
49 // Signal driver annotations are needed when processing both circuits:
50 //
51 // First the original annotations are used with the main circuit, in order to
52 // emit extra wires around forced ports and to emit updated annotations
53 // pointing to the values across naming or hierarchy changes performed.
54 //
55 // Second these updated annotations are used when processing the subcircuit
56 // which is when the mappings module is created and instantiated.
57 //
58 //===----------------------------------------------------------------------===//

64 namespace {
65 enum class MappingDirection {
66  /// The local signal forces the remote signal.
67  DriveRemote, // Sink
68  /// The local signal is driven to the value of the remote signal through a
69  /// connect.
70  ProbeRemote, // Source
71 };
72 
73 /// The information necessary to connect a local signal in a module to a remote
74 /// value (in a different module and/or circuit).
75 struct SignalMapping {
76  /// Whether we force or read from the target.
77  MappingDirection dir;
78  /// The reference target of the thing we are forcing or probing.
79  StringAttr remoteTarget;
80  /// The type of the signal being mapped.
81  FIRRTLType type;
82  /// The block argument or result that interacts with the remote target, either
83  /// by forcing it or by reading from it through a connect.
84  Value localValue;
85  /// The name of the local value, for reuse in the generated signal mappings
86  /// module.
87  StringAttr localName;
88  /// Which "side" we're on
89  bool isLocal;
90  /// NLA, if present
91  FlatSymbolRefAttr nlaSym;
92 };

124 LogicalResult circt::firrtl::applyGCTSignalMappings(const AnnoPathValue &target,
125  DictionaryAttr anno,
126  ApplyState &state) {
127  auto id = state.newID();
128  auto *context = state.circuit.getContext();
129  auto loc = state.circuit.getLoc();
130 
131  // Rework the circuit-level annotation to no longer include the
132  // information we are scattering away anyway.
133  NamedAttrList fields;
134  auto annotationsAttr = tryGetAs<ArrayAttr>(anno, anno, "annotations", loc,
135  signalDriverAnnoClass);
136  auto circuitAttr =
137  tryGetAs<StringAttr>(anno, anno, "circuit", loc, signalDriverAnnoClass);
138  auto circuitPackageAttr = tryGetAs<StringAttr>(anno, anno, "circuitPackage",
139  loc, signalDriverAnnoClass);
140  if (!annotationsAttr || !circuitAttr || !circuitPackageAttr)
141  return failure();
142  fields.append("class", StringAttr::get(context, signalDriverAnnoClass));
143  fields.append("id", id);
144  fields.append("annotations", annotationsAttr);
145  fields.append("circuit", circuitAttr);
146  fields.append("circuitPackage", circuitPackageAttr);
147 
148  // Handle the source and sink targets.
149  auto sourcesAttr = tryGetAs<ArrayAttr>(anno, anno, "sourceTargets", loc,
150  signalDriverAnnoClass);
151  auto sinksAttr = tryGetAs<ArrayAttr>(anno, anno, "sinkTargets", loc,
152  signalDriverAnnoClass);
153  if (!sourcesAttr || !sinksAttr)
154  return failure();
155 
156  // Add field indicating if we're the subcircuit or not. Do this by looking at
157  // the first source or sink and seeing what its circuit target is.
158  DictionaryAttr firstSourceOrSink;
159  if (!sourcesAttr.empty())
160  firstSourceOrSink = sourcesAttr.begin()->dyn_cast<DictionaryAttr>();
161  else if (!sinksAttr.empty())
162  firstSourceOrSink = sinksAttr.begin()->dyn_cast<DictionaryAttr>();
163  else
164  return success();
165  if (!firstSourceOrSink)
166  return failure();
167  auto prefix = tryGetAs<StringAttr>(firstSourceOrSink, anno, "_1", loc,
168  signalDriverAnnoClass)
169  .getValue();
170  bool isSubCircuit = !(prefix.consume_front("~") &&
171  prefix.consume_front(state.circuit.name()) &&
172  prefix.consume_front("|"));
173  fields.append("isSubCircuit", BoolAttr::get(context, isSubCircuit));
174 
175  auto dontTouch = [&](StringRef targetStr) {
176  NamedAttrList anno;
177  anno.append("class", StringAttr::get(context, dontTouchAnnoClass));
178  anno.append("target", StringAttr::get(context, targetStr));
179  state.addToWorklistFn(DictionaryAttr::get(context, anno));
180  };
181 
182  StringSet<> modules;
183  auto buildSourceOrSinkAnno = [&](Attribute attr,
184  bool isSource) -> DictionaryAttr {
185  auto dict = attr.dyn_cast<DictionaryAttr>();
186  if (!dict)
187  return {};
188  NamedAttrList fields;
189  fields.append("class",
190  StringAttr::get(context, signalDriverTargetAnnoClass));
191  StringAttr targetAttr = tryGetAs<StringAttr>(
192  dict, dict, isSubCircuit ? "_2" : "_1", loc, signalDriverAnnoClass);
193  StringAttr peer = tryGetAs<StringAttr>(
194  dict, dict, isSubCircuit ? "_1" : "_2", loc, signalDriverAnnoClass);
195  if (!targetAttr || !peer)
196  return {};
197 
198  auto target = tokenizePath(targetAttr);
199  if (!target)
200  return {};
201  target->component.clear();
202  target->name = "";
203  modules.insert(target->str());
204 
205  auto targetId = state.newID();
206  fields.append("dir",
207  StringAttr::get(context, isSource ? "source" : "sink"));
208  fields.append("id", id);
209  fields.append("peer", peer);
210  fields.append("side",
211  StringAttr::get(context, isSubCircuit ? "local" : "remote"));
212  fields.append("target", targetAttr);
213  fields.append("targetId", targetId);
214 
215  // TODO: Always applying DontTouchAnnotation to a target is too restrictive.
216  // This should only be necessary if this is a sink (which will produce a
217  // force) and we are in the circuit where the force will be applied. Relax
218  // this restriction in the future.
219  dontTouch(targetAttr.getValue());
220 
221  return DictionaryAttr::get(context, fields);
222  };
223 
224  // Add any sinks or sources to the worklist for later processing.
225  for (auto attr : sourcesAttr) {
226  auto sourceAnno = buildSourceOrSinkAnno(attr, true);
227  if (!sourceAnno)
228  return failure();
229  state.addToWorklistFn(sourceAnno);
230  }
231  for (auto attr : sinksAttr) {
232  auto sinkAnno = buildSourceOrSinkAnno(attr, false);
233  if (!sinkAnno)
234  return failure();
235  state.addToWorklistFn(sinkAnno);
236  }
237 
238  AnnotationSet annotations(state.circuit);
239  annotations.addAnnotations({DictionaryAttr::get(context, fields)});
240  annotations.applyToOperation(state.circuit);
241 
242  for (auto &module : modules) {
243  NamedAttrList fields;
244  fields.append("class",
245  StringAttr::get(context, signalDriverModuleAnnoClass));
246  fields.append("id", id);
247  fields.append("target", StringAttr::get(context, module.getKey()));
248  state.addToWorklistFn(DictionaryAttr::get(context, fields));
249  }
250 
251  return success();
252 }

254 /// Analyze the `module` of this `ModuleSignalMappings` and generate the
255 /// corresponding auxiliary `FModuleOp` with the necessary cross-module
256 /// references and `ForceOp`s to probe and drive remote signals. This is
257 /// dictated by the presence of `SignalDriverAnnotation.module` on the module
258 /// and individual operations inside it. Anything targeted by a source/sink will
259 /// show up with a `SignalDriverAnnotation.target`. Global circuit-level
260 /// information lives in a circuit-level `SignalDriverAnnotation`.
261 ///
262 /// If the module is the "remote" (main) circuit, gather those mappings for use
263 /// handling forced ports and creating updated mappings.
264 void ModuleSignalMappings::run() {
265  // Check whether this module has any `SignalDriverAnnotation.module`s. These
266  // indicate whether the module contains any operations with such annotations
267  // and requires processing.
268  if (!AnnotationSet::removeAnnotations(module, signalDriverModuleAnnoClass)) {
269  LLVM_DEBUG(llvm::dbgs() << "Skipping `" << module.getName()
270  << "` (has no annotations)\n");
271  return;
272  }
273  LLVM_DEBUG(llvm::dbgs() << "Running on module `" << module.getName()
274  << "`\n");
275 
276  // Gather the `SignalDriverAnnotation.target`s on the ports of this module.
277  LLVM_DEBUG(llvm::dbgs() << "- Gather port annotations\n");
278  AnnotationSet::removePortAnnotations(
279  module, [&](unsigned i, Annotation anno) {
280  if (!anno.isClass(signalDriverTargetAnnoClass))
281  return false;
282  addTarget(module.getArgument(i), anno);
283  return true;
284  });
285 
286  // Gather the `SignalDriverAnnotation.target`s of the operations within this
287  // module.
288  LLVM_DEBUG(llvm::dbgs() << "- Gather operation annotations\n");
289  module.walk([&](Operation *op) {
290  AnnotationSet::removeAnnotations(op, [&](Annotation anno) {
291  if (!anno.isClass(signalDriverTargetAnnoClass))
292  return false;
293  for (auto result : op->getResults())
294  addTarget(result, anno);
295  return true;
296  });
297  });
298 
299  auto localMappings =
300  llvm::make_filter_range(mappings, [](auto &m) { return m.isLocal; });
301 
302  // Remove connections to sources in the subcircuit. This is done to cleanup
303  // invalidations that occur from the Chisel API of Grand Central.
304  // Only the remote value "source" should be connected to local source values.
305  for (auto mapping : localMappings)
306  if (mapping.dir == MappingDirection::ProbeRemote) {
307  for (auto &use : llvm::make_early_inc_range(mapping.localValue.getUses()))
308  if (auto connect = dyn_cast<FConnectLike>(use.getOwner()))
309  if (connect.dest() == mapping.localValue) {
310  connect.erase();
311  allAnalysesPreserved = false;
312  };
313  }
314 
315  // If there aren't any local-side (subcircuit-side) mappings, we're done.
316  if (localMappings.empty()) {
317  LLVM_DEBUG(llvm::dbgs() << "Skipping `" << module.getName()
318  << "` (has no non-zero sources or sinks)\n");
319  return;
320  }
321 
322  // Changes are coming, mark analyses not preserved
323  allAnalysesPreserved = false;
324 
325  if (!llvm::all_of(mappings, [](auto &m) { return m.isLocal; })) {
326  emitWarning(module->getLoc(),
327  "both remote and local SignalDriverAnnotation mappings found");
328  }
329 
330  // Pick a name for the module that implements the signal mappings.
331  CircuitNamespace circuitNamespace(module->getParentOfType<CircuitOp>());
332  mappingsModuleName =
333  circuitNamespace.newName(Twine(module.getName()) + "_signal_mappings");
334 
335  // Generate the mappings module.
336  auto mappingsModule = emitMappingsModule();
337 
338  // Instantiate the mappings module.
339  instantiateMappingsModule(mappingsModule);
340 }

342 /// Mark a `value` inside the `module` as being the target of the
343 /// `SignalDriverAnnotation` `anno`. This generates the necessary
344 /// `SignalMapping` information and adds an entry to the `mappings` array, to be
345 /// later consumed when the mappings module is constructed.
346 void ModuleSignalMappings::addTarget(Value value, Annotation anno) {
347  // Ignore the target if it is zero width.
348  if (!value.getType().cast<FIRRTLType>().getBitWidthOrSentinel())
349  return;
350 
351  SignalMapping mapping;
352  mapping.dir = anno.getMember<StringAttr>("dir").getValue() == "source"
353  ? MappingDirection::ProbeRemote
354  : MappingDirection::DriveRemote;
355  mapping.remoteTarget = anno.getMember<StringAttr>("peer");
356  mapping.localValue = value;
357  mapping.type = value.getType().cast<FIRRTLType>();
358  mapping.isLocal = anno.getMember<StringAttr>("side").getValue() == "local";
359  mapping.nlaSym = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal");
360 
361  // Guess a name for the local value. This is only for readability's sake,
362  // giving the pass a hint for picking the names of the generated module ports.
363  if (auto blockArg = value.dyn_cast<BlockArgument>()) {
364  mapping.localName = module.getPortNameAttr(blockArg.getArgNumber());
365  } else if (auto *op = value.getDefiningOp()) {
366  mapping.localName = op->getAttrOfType<StringAttr>("name");
367  }
368 
369  LLVM_DEBUG(llvm::dbgs() << " - " << mapping << "\n");
370  mappings.push_back(std::move(mapping));
371 }

373 /// Create a separate mappings module that contains cross-module references and
374 /// `ForceOp`s for each entry in the `mappings` array.
375 FModuleOp ModuleSignalMappings::emitMappingsModule() {
376  LLVM_DEBUG(llvm::dbgs() << "- Generating `" << mappingsModuleName << "`\n");
377 
378  // Determine what ports this module will have, given the signal mappings we
379  // are supposed to emit.
380  SmallVector<PortInfo> ports;
381  auto localMappings =
382  llvm::make_filter_range(mappings, [](auto &m) { return m.isLocal; });
383  for (auto &mapping : localMappings) {
384  ports.push_back(PortInfo{mapping.localName,
385  mapping.type,
386  mapping.dir == MappingDirection::DriveRemote
387  ? Direction::In
388  : Direction::Out,
389  {},
390  module.getLoc()});
391  LLVM_DEBUG(llvm::dbgs() << " - Adding port " << mapping.localName << "\n");
392  }
393 
394  // Create the actual module.
395  ImplicitLocOpBuilder builder(module.getLoc(), module);
396  auto mappingsModule = builder.create<FModuleOp>(
397  StringAttr::get(module.getContext(), mappingsModuleName), ports);
398 
399  // Generate the connect and force statements inside the module.
400  builder.setInsertionPointToStart(mappingsModule.getBody());
401  unsigned portIdx = 0;
402  for (auto &mapping : localMappings) {
403  // TODO: Actually generate a proper XMR here. For now just do some textual
404  // replacements. Generating a real IR node (like a proper XMR op) would be
405  // much better, but the modules that `EmitSignalMappings` interacts with
406  // generally live in a separate circuit. Multiple circuits are not fully
407  // supported at the moment.
408  auto tokenized = tokenizePath(mapping.remoteTarget);
409  if (!tokenized) {
410  emitWarning(module.getLoc(),
411  llvm::formatv("unable to parse remote target '{0}' found in "
412  "SignalDriverAnnotation, skipping",
413  mapping.remoteTarget));
414  continue;
415  }
416 
417  SmallString<32> remoteXmrName;
418  if (tokenized->instances.empty()) {
419  // If no instance path, just use module directly
420  remoteXmrName = tokenized->module;
421  } else {
422  // First module encountered is our starting point
423  auto &[firstMod, _] = tokenized->instances.front();
424  remoteXmrName = firstMod;
425  for (auto &[mod, inst] : tokenized->instances) {
426  remoteXmrName.push_back('.');
427  remoteXmrName += inst;
428  }
429  }
430  assert(!remoteXmrName.empty());
431 
432  remoteXmrName.push_back('.');
433  remoteXmrName += tokenized->name;
434 
435  // TODO: Actually index into vectors/bundles?
436  // For now, convert to expected name of expanded aggregate
437  for (auto &comp : tokenized->component) {
438  remoteXmrName.push_back('_');
439  remoteXmrName += comp.name;
440  }
441 
442  // llvm::errs() << "XMR: " << remoteXmrName << "\n\n";
443  if (mapping.dir == MappingDirection::DriveRemote) {
444  auto xmr = builder.create<VerbatimWireOp>(mapping.type, remoteXmrName);
445  builder.create<ForceOp>(xmr, mappingsModule.getArgument(portIdx++));
446  } else {
447  auto xmr = builder.create<VerbatimWireOp>(mapping.type, remoteXmrName);
448  builder.create<ConnectOp>(mappingsModule.getArgument(portIdx++), xmr);
449  }
450  }
451  return mappingsModule;
452 }

454 /// Instantiate the generated mappings module inside the `module` we are working
455 /// on, and generated the necessary connections to local signals.
456 void ModuleSignalMappings::instantiateMappingsModule(FModuleOp mappingsModule) {
457  LLVM_DEBUG(llvm::dbgs() << "- Instantiating `" << mappingsModuleName
458  << "`\n");
459  // Create the actual module.
460  auto builder =
461  ImplicitLocOpBuilder::atBlockEnd(module.getLoc(), module.getBody());
462  auto inst = builder.create<InstanceOp>(mappingsModule, "signal_mappings");
463 
464  // Generate the connections to and from the instance.
465  unsigned portIdx = 0;
466  auto localMappings =
467  llvm::make_filter_range(mappings, [](auto &m) { return m.isLocal; });
468  for (auto &mapping : localMappings) {
469  Value dst = inst.getResult(portIdx++);
470  Value src = mapping.localValue;
471  if (mapping.dir == MappingDirection::ProbeRemote)
472  std::swap(src, dst);
473  builder.create<ConnectOp>(dst, src);
474  }
475 }

521 void GrandCentralSignalMappingsPass::runOnOperation() {
522  CircuitOp circuit = getOperation();
523 
524  StringAttr circuitPackage;
525  BoolAttr isSubCircuitAttr;
526  bool circuitHasAnno =
527  AnnotationSet::removeAnnotations(circuit, [&](Annotation anno) {
528  if (!anno.isClass(signalDriverAnnoClass))
529  return false;
530 
531  isSubCircuitAttr = anno.getMember<BoolAttr>("isSubCircuit");
532  circuitPackage = anno.getMember<StringAttr>("circuitPackage");
533  return true;
534  });
535 
536  if (!circuitHasAnno) {
537  // Nothing to do here
538  markAllAnalysesPreserved();
539  return;
540  }
541 
542  if (!isSubCircuitAttr) {
543  emitError(
544  circuit->getLoc(),
545  "has invalid SignalDriverAnnotation (missing 'isSubCircuit' field')");
546  return signalPassFailure();
547  }
548  bool isSubCircuit = isSubCircuitAttr.getValue();
549 
550  if (isSubCircuit && !circuitPackage) {
551  emitError(circuit->getLoc())
552  << "has invalid SignalDriverAnnotation (subcircuit JSON emission is "
553  "enabled, but no circuitPackage was provided)";
554  return signalPassFailure();
555  }
556 
557  SmallVector<FModuleOp> modules;
558  ExtModules extmodules;
559 
560  for (auto op : circuit.body().getOps<FModuleLike>()) {
561  if (auto *extModule = dyn_cast<FExtModuleOp>(&op)) {
562  AnnotationSet annotations(*extModule);
563  if (annotations.hasAnnotation(blackBoxInlineAnnoClass)) {
564  extmodules.vsrc.push_back(*extModule);
565  continue;
566  }
567  extmodules.json.push_back(*extModule);
568  } else if (auto *module = dyn_cast<FModuleOp>(&op)) {
569  modules.push_back(*module);
570  }
571  }
572 
573  // Pre-allocate result vector
574  SmallVector<ModuleMappingResult> results;
575  results.resize(modules.size());
576 
577  auto updateStats = [this](auto &mappings) {
578  size_t sources = 0, sinks = 0;
579  for (auto &mapping : mappings) {
580  if (mapping.dir == MappingDirection::ProbeRemote)
581  ++sources;
582  else
583  ++sinks;
584  }
585  if (sources)
586  numSources += sources;
587  if (sinks)
588  numSinks += sinks;
589  if (sources || sinks)
590  ++numModules;
591  };
592 
593  mlir::parallelFor(
594  circuit.getContext(), 0, modules.size(),
595  [&modules, &results, &updateStats](size_t index) {
596  ModuleSignalMappings mapper(modules[index]);
597  mapper.run();
598 
599  // Update statistics
600  updateStats(mapper.mappings);
601 
602  results[index] = {
603  mapper.allAnalysesPreserved, modules[index],
604  SmallVector<SignalMapping>{llvm::make_filter_range(
605  mapper.mappings, [](auto &m) { return !m.isLocal; })}};
606  });
607  bool allAnalysesPreserved =
608  llvm::all_of(results, [](auto &r) { return r.allAnalysesPreserved; });
609 
610  // If this is a subcircuit, then emit JSON information necessary to drive
611  // SiFive tools.
612  // Otherwise handle any forced ports and emit updated mappings.
613  if (isSubCircuit) {
614  emitSubCircuitJSON(circuit, circuitPackage, outputFilename, extmodules);
615  } else {
616  auto result = emitUpdatedMappings(circuit, circuitPackage, results);
617  if (failed(result))
618  return signalPassFailure();
619  allAnalysesPreserved &= *result;
620  }
621 
622  if (allAnalysesPreserved)
623  markAllAnalysesPreserved();
624 }

626 void GrandCentralSignalMappingsPass::emitSubCircuitJSON(
627  CircuitOp circuit, StringAttr circuitPackage, StringRef outputFilename,
628  const ExtModules &extmodules) {
629  std::string jsonString;
630  llvm::raw_string_ostream jsonStream(jsonString);
631  json::OStream j(jsonStream, 2);
632  j.object([&] {
633  j.attributeObject("vendor", [&]() {
634  j.attributeObject("vcs", [&]() {
635  j.attributeArray("vsrcs", [&]() {
636  for (FModuleOp module : circuit.body().getOps<FModuleOp>()) {

673 FailureOr<bool> GrandCentralSignalMappingsPass::emitUpdatedMappings(
674  CircuitOp circuit, StringAttr circuitPackage,
675  MutableArrayRef<ModuleMappingResult> results) {
676  bool allAnalysesPreserved = true;
677 
678  // 1) Sanity checking
679  // 2) Fixup ports that are driven from subcircuit
680  // 3) Generate new remote references
681  // 4) Emit updated mappings as SignalDriverAnnotations
682  // for consumption when processing the subcircuit
683  // (as replacements for the original annotations)
684 
685  // Helper to instantiate module namespaces as-needed
686  DenseMap<Operation *, ModuleNamespace> moduleNamespaces;
687  auto getModuleNamespace =
688  [&moduleNamespaces](Operation *module) -> ModuleNamespace & {
689  return moduleNamespaces.try_emplace(module, module).first->second;
690  };
691 
692  // (1) Scan mappings for unexpected or unsupported values
693  for (auto const &[_, mod, mappings] : results) {
694  for (auto &mapping : mappings) {
695  // No local-side mappings on main circuit side
696  if (mapping.isLocal) {
697  emitError(mapping.localValue.getLoc())
698  << "local-side SignalDriverAnnotation found in main circuit, "
699  "mixing not supported";
700  return failure();
701  }
702  }
703  }
704 
705  // (2) Find ports that are sinks, insert buffer wires to break the net.
706  // Input ports are buffered at instantiation points, outputs in the module.
707  auto *instanceGraph = &getAnalysis<InstanceGraph>();
708  llvm::SmallVector<unsigned, 32> forcedInputPorts;
709  llvm::SmallVector<unsigned, 32> forcedOutputPorts;
710  for (auto &[_, mod, mappings] : results) {
711  // Walk mappings looking for module ports that are being driven,
712  // and gather their indices for fixing up.
713  forcedInputPorts.clear();
714  forcedOutputPorts.clear();
715  for (auto &mapping : mappings) {
716  if (mapping.dir == MappingDirection::DriveRemote /* Sink */) {
717  if (auto blockArg = mapping.localValue.dyn_cast<BlockArgument>()) {
718  auto portIdx = blockArg.getArgNumber();
719  // Port may be driven multiple times along different instance paths
720  if (mod.getPortDirection(portIdx) == Direction::In) {
721  forcedInputPorts.push_back(portIdx);
722  } else {
723  forcedOutputPorts.push_back(portIdx);
724  }
725  }
726  }
727  }
728  llvm::sort(forcedInputPorts);
729  forcedInputPorts.erase(
730  std::unique(forcedInputPorts.begin(), forcedInputPorts.end()),
731  forcedInputPorts.end());
732  llvm::sort(forcedOutputPorts);
733  forcedOutputPorts.erase(
734  std::unique(forcedOutputPorts.begin(), forcedOutputPorts.end()),
735  forcedOutputPorts.end());
736 
737  // Replace uses of each driven port with a wire that's connected to a wire
738  // that is connected to the port. This is done to cause an 'assign' to be
739  // created, disconnecting the forced port's net from its uses.
740  auto breakNet = [modName = mod.moduleName()](
741  OpBuilder &builder, Value port,
742  ModuleNamespace &moduleNamespace, StringRef portName) {
743  // Create chain like:
744  // port_result <= foo_dataIn_x_buffer
745  // foo_dataIn_x_buffer <= foo_dataIn_x
746  auto replacementWireName = builder.getStringAttr(
747  moduleNamespace.newName(modName + "_" + portName));
748  auto bufferWireName = builder.getStringAttr(
749  moduleNamespace.newName(replacementWireName.getValue() + "_buffer"));
750  auto bufferWire =
751  builder.create<WireOp>(builder.getUnknownLoc(), port.getType(),
752  bufferWireName, NameKindEnum::DroppableName,
753  builder.getArrayAttr({}), bufferWireName);
754  auto replacementWire = builder.create<WireOp>(
755  builder.getUnknownLoc(), port.getType(), replacementWireName,
756  NameKindEnum::DroppableName, builder.getArrayAttr({}),
757  replacementWireName);
758  port.replaceAllUsesWith(replacementWire);
759  builder.create<StrictConnectOp>(builder.getUnknownLoc(), port,
760  bufferWire);
761  builder.create<StrictConnectOp>(builder.getUnknownLoc(), bufferWire,
762  replacementWire);
763  };
764 
765  // Update the forced ports statistics.
766  numForcedInputPorts += forcedInputPorts.size();
767  numForcedOutputPorts += forcedOutputPorts.size();
768 
769  if (!forcedOutputPorts.empty()) {
770  ModuleNamespace &moduleNamespace = getModuleNamespace(mod);
771  auto builder = OpBuilder::atBlockBegin(mod.getBody());
772 
773  // Update statistic
774  numBufferWirePairsAdded += forcedOutputPorts.size();
775 
776  for (auto portIdx : forcedOutputPorts) {
777  auto port = mod.getArgument(portIdx);
778  breakNet(builder, port, moduleNamespace, mod.getPortName(portIdx));
779  }
780  }
781 
782  if (forcedInputPorts.empty())
783  continue;
784 
785  for (auto *use : instanceGraph->lookup(mod)->uses()) {
786  allAnalysesPreserved = false;
787 
788  auto inst = use->getInstance();
789  OpBuilder builder(inst.getContext());
790  builder.setInsertionPointAfter(inst);
791  auto parentModule = inst->getParentOfType<FModuleOp>();
792  ModuleNamespace &moduleNamespace = getModuleNamespace(parentModule);
793 
794  // Update statistic
795  numBufferWirePairsAdded += forcedInputPorts.size();
796 
797  for (auto portIdx : forcedInputPorts) {
798  auto port = inst->getResult(portIdx);
799  breakNet(builder, port, moduleNamespace, mod.getPortName(portIdx));
800  }
801  }
802  }
803 
804  // (3) Generate updated remote references
805 
806  // For instance path references, determine where to start.
807  // This is either the main module or the DUT if any.
808  auto dut = circuit.getMainModule();
809  for (auto op : circuit.body().getOps<FModuleOp>())
810  if (AnnotationSet(op).hasAnnotation(dutAnnoClass)) {
811  dut = op;
812  break;
813  }
814 
815  // Helpers to emit XMR's for the mappings.
816  SmallVector<Attribute> symbols;
817  SmallDenseMap<Attribute, size_t> symMap;
818 
819  auto getSymIdx = [&](Attribute symbol) {
820  auto it = symMap.find(symbol);
821  if (it != symMap.end())
822  return it->second;
823 
824  auto id = symbols.size();
825  symbols.push_back(symbol);
826  symMap.insert({symbol, id});
827  return id;
828  };
829  auto mkRef = [&](FModuleOp module,
830  const SignalMapping &mapping) -> std::string {
831  // Generate placeholder into specified storage, return StringRef for it
832  auto mkSymPlaceholder = [&](Attribute symbol, auto &out) -> StringRef {
833  ("{{" + Twine(getSymIdx(symbol)) + "}}").toVector(out);
834  return out;
835  };
836 
837  // Construct target using placeholders:
838  SmallString<32> circuitStr, moduleStr, nameStr;
839  // Storage for <mod,inst> strings along instance path
840  SmallVector<std::pair<SmallString<8>, SmallString<8>>, 16> stringStorage;
841  TokenAnnoTarget target;
842  target.circuit = mkSymPlaceholder(FlatSymbolRefAttr::get(dut), circuitStr);
843  target.module = mkSymPlaceholder(FlatSymbolRefAttr::get(module), moduleStr);
844  target.component = {};
845 
846  if (auto port = mapping.localValue.dyn_cast<BlockArgument>())
847  target.name = mkSymPlaceholder(
848  ::getInnerRefTo(module, port.getArgNumber(), mapping.localName,
849  getModuleNamespace),
850  nameStr);
851  else
852  target.name =
853  mkSymPlaceholder(::getInnerRefTo(mapping.localValue.getDefiningOp(),
854  "", getModuleNamespace),
855  nameStr);
856 
857  // If there's an NLA, add instance path information.
858  if (mapping.nlaSym) {
859  auto nla =
860  cast<HierPathOp>(circuit.lookupSymbol(mapping.nlaSym.getAttr()));
861  assert(!nla.namepath().empty());
862 
863  // Start from root of NLA, or from top/DUT if through it
864  bool seenRoot = false;
865  bool usesTop = nla.hasModule(dut.moduleNameAttr());
866  ArrayRef<Attribute> path = nla.namepath().getValue();
867  stringStorage.resize(path.drop_back().size());
868  for (const auto &attr : llvm::enumerate(path.drop_back())) {
869  auto ref = attr.value().cast<hw::InnerRefAttr>();
870  if (usesTop && !seenRoot) {
871  if (ref.getModule() == dut.moduleNameAttr())
872  seenRoot = true;
873  }
874 
875  if (!usesTop || seenRoot) {
876  auto &store = stringStorage[attr.index()];
877  auto modStr = mkSymPlaceholder(ref.getModuleRef(), store.first);
878  auto instStr = mkSymPlaceholder(
879  hw::InnerRefAttr::get(ref.getModule(), ref.getName()),
880  store.second);
881  target.instances.emplace_back(modStr, instStr);
882  }
883  };
884  }
885  return target.str();
886  };
887 
888  // Generate and sort new mappings for (better) output stability
889  SmallVector<std::pair<std::string, StringRef>, 32> sinks, sources;
890  auto addUpdatedMappings = [&](MappingDirection filterDir, auto &vec) {
891  for (auto const &[_, mod, mappings] : results)
892  for (auto &mapping : mappings)
893  if (mapping.dir == filterDir)
894  vec.emplace_back(mkRef(mod, mapping),
895  mapping.remoteTarget.getValue());
896  };
897  addUpdatedMappings(MappingDirection::DriveRemote, sinks);
898  addUpdatedMappings(MappingDirection::ProbeRemote, sources);
899  // Sort on local targets only, remote may contain substitution placeholders
900  llvm::sort(sinks, llvm::less_second());
901  llvm::sort(sources, llvm::less_second());
902 
903  // (4) Emit updated mappings for consumption by subcircuit
904  std::string jsonString;
905  llvm::raw_string_ostream jsonStream(jsonString);
906  json::OStream j(jsonStream, 2);
907 
908  j.array([&] {
909  j.object([&] {
910  j.attribute("class", signalDriverAnnoClass);
911  j.attributeArray("sinkTargets", [&]() {
912  for (auto &[remote, local] : sinks) {
913  j.objectBegin();
914  j.attribute("_1", remote);
915  j.attribute("_2", local);
916  j.objectEnd();
917  };
918  });
919  j.attributeArray("sourceTargets", [&]() {
920  for (auto &[remote, local] : sources) {
921  j.objectBegin();
922  j.attribute("_1", remote);
923  j.attribute("_2", local);
924  j.objectEnd();
925  };
926  });
927  // Emit these but don't attempt to plumb through their original values
928  j.attribute("circuit", "circuit empty :\n module empty :\n\n skip\n");
929  j.attributeArray("annotations", [&]() {});
930  if (circuitPackage)
931  j.attribute("circuitPackage", circuitPackage.getValue());
932  });
933  });
934  auto b = OpBuilder::atBlockEnd(circuit.getBody());
935  auto jsonOp = b.create<sv::VerbatimOp>(b.getUnknownLoc(), jsonString,
936  ValueRange{}, b.getArrayAttr(symbols));
937  // TODO: plumb option to specify path/name?
938  constexpr char jsonOutFile[] = "sigdrive.json";
939  jsonOp->setAttr("output_file", hw::OutputFileAttr::getFromFilename(
940  b.getContext(), jsonOutFile, true));
941 
942  return allAnalysesPreserved;
943 }
