139 ArrayRef<InstancePath> InstancePaths::getAbsolutePaths(Operation *op) {
140  assert((isa<FModuleOp, FExtModuleOp>(op))); // extra parens makes parser smile
141 
142  // If we have reached the circuit root, we're done.
143  if (op == instanceGraph.getTopLevelNode()->getModule()) {
144  static InstancePath empty{};
145  return empty; // array with single empty path
146  }
147 
148  // Fast path: hit the cache.
149  auto cached = absolutePathsCache.find(op);
150  if (cached != absolutePathsCache.end())
151  return cached->second;
152 
153  // For each instance, collect the instance paths to its parent and append the
154  // instance itself to each.
155  SmallVector<InstancePath, 8> extendedPaths;
156  for (auto inst : instanceGraph[op]->uses()) {
157  auto instPaths = getAbsolutePaths(inst->getParent()->getModule());
158  extendedPaths.reserve(instPaths.size());
159  for (auto path : instPaths) {
160  extendedPaths.push_back(appendInstance(path, inst->getInstance()));
161  }
162  }
163 
164  // Move the list of paths into the bump allocator for later quick retrieval.
165  ArrayRef<InstancePath> pathList;
166  if (!extendedPaths.empty()) {
167  auto paths = allocator.Allocate<InstancePath>(extendedPaths.size());
168  std::copy(extendedPaths.begin(), extendedPaths.end(), paths);
169  pathList = ArrayRef<InstancePath>(paths, extendedPaths.size());
170  }
171 
172  absolutePathsCache.insert({op, pathList});
173  return pathList;
174 }

176 InstancePath InstancePaths::appendInstance(InstancePath path, InstanceOp inst) {
177  size_t n = path.size() + 1;
178  auto newPath = allocator.Allocate<InstanceOp>(n);
179  std::copy(path.begin(), path.end(), newPath);
180  newPath[path.size()] = inst;
181  return InstancePath(newPath, n);
182 }

184 /// Return a version of `path` that skips all front instances it has in common
185 /// with `other`.
186 static InstancePath stripCommonPrefix(InstancePath path, InstancePath other) {
187  while (!path.empty() && !other.empty() && path.front() == other.front()) {
188  path = path.drop_front();
189  other = other.drop_front();
190  }
191  return path;
192 }

264 void GrandCentralTapsPass::runOnOperation() {
265  auto circuitOp = getOperation();
266  LLVM_DEBUG(llvm::dbgs() << "Running the GCT Data Taps pass\n");
267 
268  // Here's a rough idea of what the Scala code is doing:
269  // - Gather the `source` of all `keys` of all `DataTapsAnnotation`s throughout
270  // the design.
271  // - Convert the sources, which are specified on modules, to the absolute
272  // paths in all instances. E.g. module M tap x will produce a.x and b.x if
273  // there are two instances a and b of module M in the design.
274  // - All data tap keys are specified on black box ports.
275  // - The code then processes every DataTapsAnnotation separately as follows
276  // (with the targeted blackbox and keys):
277  // - Check for collisions between SV keywords and key port names (skip this).
278  // - Find all instances of the blackbox, but then just pick the first one (we
279  // should probably do this for each?)
280  // - Process each key independently as follows:
281  // - Look up the absolute path of the source in the map. Ensure it exists and
282  // is unambiguous. Make it relative to the blackbox instance path.
283  // - Look up the port on the black box.
284  // - Create a hierarchical SV name and store this as an assignment to the
285  // blackbox port.
286  // - DeletedDataTapKey: skip and don't create a wiring
287  // - LiteralDataTapKey: just the literal
288  // - ReferenceDataTapKey: relative path with "." + source name
289  // - DataTapModuleSignalKey: relative path with "." + internal path
290  // - Generate a body for the blackbox module with the signal mapping
291 
292  memPortIdx.clear();
293 
294  // Gather a list of extmodules that have data or mem tap annotations to be
295  // expanded.
296  SmallVector<AnnotatedExtModule, 4> modules;
297  for (auto &op : *circuitOp.getBody()) {
298  auto extModule = dyn_cast<FExtModuleOp>(&op);
299  if (!extModule)
300  continue;
301 
302  // Go through the module ports and collect the annotated ones.
303  AnnotatedExtModule result{extModule, {}, {}, {}};
304  result.filteredPortAnnos.reserve(extModule.getNumArguments());
305  for (unsigned argNum = 0, e = extModule.getNumArguments(); argNum < e;
306  ++argNum) {
307  // Go through all annotations on this port and add the data tap key and
308  // mem tap ones to the list.
309  auto annos = AnnotationSet::forPort(extModule, argNum);
310  annos.removeAnnotations([&](Annotation anno) {
311  if (anno.isClass(memTapClass, deletedKeyClass, literalKeyClass,
312  internalKeyClass) ||
313  isReferenceDataTapPortName(anno)) {
314  result.portAnnos.push_back({argNum, anno});
315  return true;
316  }
317  return false;
318  });
319  result.filteredPortAnnos.push_back(annos);
320  }
321 
322  // If there are data tap annotations on the module, which is likely the
323  // case, create a filtered array of annotations with them removed.
324  AnnotationSet annos(extModule.getOperation());
325  annos.removeAnnotations(
326  [&](Annotation anno) { return anno.isClass(dataTapsClass); });
327  result.filteredModuleAnnos = annos.getArrayAttr();
328 
329  if (!result.portAnnos.empty())
330  modules.push_back(std::move(result));
331  }
332 
333  LLVM_DEBUG({
334  for (auto m : modules) {
335  llvm::dbgs() << "Extmodule " << m.extModule.getName() << " has "
336  << m.portAnnos.size() << " port annotations\n";
337  }
338  });
339 
340  // Fast path if there's nothing to do.
341  if (modules.empty()) {
342  markAllAnalysesPreserved();
343  return;
344  }
345 
346  // Build a generator for absolute module and instance paths in the design.
347  InstancePaths instancePaths(getAnalysis<InstanceGraph>());
348 
349  // Gather the annotated ports and operations throughout the design that we are
350  // supposed to tap in one way or another.
351  tappedPorts.clear();
352  tappedOps.clear();
353  circuitOp.walk([&](Operation *op) { gatherAnnotations(op); });
354 
355  LLVM_DEBUG({
356  llvm::dbgs() << "Tapped ports:\n";
357  for (auto it : tappedPorts)
358  llvm::dbgs() << "- " << it.first << ": "
359  << it.second.first->getAttr("sym_name") << " port #"
360  << it.second.second << "\n";
361  llvm::dbgs() << "Tapped ops:\n";
362  for (auto it : tappedOps)
363  llvm::dbgs() << "- " << it.first << ": " << *it.second << "\n";
364  });
365 
366  // Process each black box independently.
367  for (auto blackBox : modules) {
368  LLVM_DEBUG(llvm::dbgs() << "Generating impls for "
369  << blackBox.extModule.getName() << "\n");
370 
371  // As a first step, gather a list of all absolute paths to instances of
372  // this black box.
373  auto paths = instancePaths.getAbsolutePaths(blackBox.extModule);
374  LLVM_DEBUG({
375  for (auto path : paths)
376  llvm::dbgs() << "- " << path << "\n";
377  });
378 
379  // Go through the port annotations of the tap module and generate a
380  // hierarchical path for each.
381  portWiring.clear();
382  portWiring.reserve(blackBox.portAnnos.size());
383  for (auto portAnno : blackBox.portAnnos) {
384  processAnnotation(portAnno, blackBox, instancePaths);
385  }
386 
387  LLVM_DEBUG({
388  llvm::dbgs() << "- Wire up as follows:\n";
389  for (auto wiring : portWiring) {
390  llvm::dbgs() << "- Port " << wiring.portNum << ":\n";
391  for (auto path : wiring.prefices) {
392  llvm::dbgs() << " - " << path << "." << wiring.suffix << "\n";
393  }
394  }
395  });
396 
397  // Now we have an awkward mapping problem. We have multiple data tap
398  // module instances, which reference things in modules that in turn have
399  // multiple instances. This is a side-effect of how Grand Central
400  // annotates things on modules rather than instances. (However in practice
401  // these will have a one-to-one correspondence due to CHIRRTL having fully
402  // uniquified instances.) To solve this issue, create a dedicated
403  // implementation for every data tap instance, and among the possible
404  // targets for the data taps choose the one with the shortest relative
405  // path to the data tap instance.
406  ImplicitLocOpBuilder builder(blackBox.extModule->getLoc(),
407  blackBox.extModule);
408  unsigned implIdx = 0;
409  for (auto path : paths) {
410  builder.setInsertionPointAfter(blackBox.extModule);
411 
412  // Get the list of ports from the original extmodule, and update the
413  // annotations such that they no longer contain any data/mem taps.
414  auto ports = blackBox.extModule.getPorts();
415  for (auto port : llvm::zip(ports, blackBox.filteredPortAnnos)) {
416  std::get<0>(port).annotations = std::get<1>(port);
417  }
418 
419  // Create a new firrtl.module that implements the data tap.
420  auto name =
421  StringAttr::get(&getContext(), Twine(blackBox.extModule.getName()) +
422  "_impl_" + Twine(implIdx++));
423  LLVM_DEBUG(llvm::dbgs()
424  << "Implementing " << name << " ("
425  << blackBox.extModule.getName() << " for " << path << ")\n");
426  auto impl =
427  builder.create<FModuleOp>(name, ports, blackBox.filteredModuleAnnos);
428  builder.setInsertionPointToEnd(impl.getBodyBlock());
429 
430  // Connect the output ports to the appropriate tapped object.
431  for (auto port : portWiring) {
432  LLVM_DEBUG(llvm::dbgs() << "- Wiring up port " << port.portNum << "\n");
433  // Determine the shortest hierarchical prefix from this black box
434  // instance to the tapped object.
435  Optional<InstancePath> shortestPrefix;
436  for (auto prefix : port.prefices) {
437  auto relative = stripCommonPrefix(prefix, path);
438  if (!shortestPrefix.hasValue() ||
439  relative.size() < shortestPrefix->size())
440  shortestPrefix = relative;
441  }
442  if (!shortestPrefix.hasValue()) {
443  LLVM_DEBUG(llvm::dbgs() << " - Has no prefix, skipping\n");
444  continue;
445  }
446  LLVM_DEBUG(llvm::dbgs()
447  << " - Shortest prefix " << *shortestPrefix << "\n");
448 
449  // Concatenate the prefix into a proper full hierarchical name.
450  SmallString<128> hname;
451  for (auto inst : shortestPrefix.getValue()) {
452  if (!hname.empty())
453  hname += '.';
454  hname += inst.name();
455  }
456  if (!hname.empty())
457  hname += '.';
458  hname += port.suffix;
459  LLVM_DEBUG(llvm::dbgs() << " - Connecting as " << hname << "\n");
460 
461  // Add a verbatim op that assigns this module port.
462  auto arg = impl.getArgument(port.portNum);
463  auto hnameExpr = builder.create<VerbatimExprOp>(
464  arg.getType().cast<FIRRTLType>(), hname);
465  builder.create<ConnectOp>(arg, hnameExpr);
466  }
467 
468  // Switch the instance from the original extmodule to this
469  // implementation. CAVEAT: If the same black box data tap module is
470  // instantiated in a parent module that itself is instantiated in
471  // different locations, this will pretty arbitrarily pick one of those
472  // locations.
473  path.back()->setAttr("moduleName",
474  builder.getSymbolRefAttr(name.getValue()));
475  }
476 
477  // Drop the original black box module.
478  blackBox.extModule.erase();
479  }
480 }

482 /// Gather the annotations on ports and operations into the `tappedPorts` and
483 /// `tappedOps` maps.
484 void GrandCentralTapsPass::gatherAnnotations(Operation *op) {
485  if (isa<FModuleOp, FExtModuleOp>(op)) {
486  // Handle port annotations on module/extmodule ops.
487  auto gather = [&](unsigned argNum, Annotation anno) {
488  if (isReferenceDataTapSource(anno)) {
489  gatherTap(anno, Port{op, argNum});
490  return true;
491  }
492  return false;
493  };
494  AnnotationSet::removePortAnnotations(op, gather);
495 
496  // Handle internal data taps on extmodule ops.
497  if (isa<FExtModuleOp>(op)) {
498  auto gather = [&](Annotation anno) {
499  if (anno.isClass(internalKeyClass)) {
500  gatherTap(anno, op);
501  return true;
502  }
503  return false;
504  };
505  AnnotationSet::removeAnnotations(op, gather);
506  }
507 
508  return;
509  }
510 
511  // Go through all annotations on this op and extract the interesting
512  // ones. Note that the way tap annotations are scattered to their
513  // targets, we should never see multiple values or memories annotated
514  // with the exact same annotation (hence the asserts).
515  AnnotationSet::removeAnnotations(op, [&](Annotation anno) {
516  if (anno.isClass(memTapClass) || isReferenceDataTapSource(anno)) {
517  gatherTap(anno, op);
518  return true;
519  }
520  return false;
521  });
522 }

524 void GrandCentralTapsPass::processAnnotation(AnnotatedPort &portAnno,
525  AnnotatedExtModule &blackBox,
526  InstancePaths &instancePaths) {
527  LLVM_DEBUG(llvm::dbgs() << "- Processing port " << portAnno.portNum
528  << " anno " << portAnno.anno.getDict() << "\n");
529  auto key = getKey(portAnno.anno);
530  auto portName = getModulePortName(blackBox.extModule, portAnno.portNum);
531  PortWiring wiring = {portAnno.portNum, {}, {}};
532 
533  // Handle data taps on signals and ports.
534  if (portAnno.anno.isClass(referenceKeyClass)) {
535  // Handle ports.
536  if (auto port = tappedPorts.lookup(key)) {
537  wiring.prefices = instancePaths.getAbsolutePaths(port.first);
538  wiring.suffix = getModulePortName(port.first, port.second).getValue();
539  portWiring.push_back(std::move(wiring));
540  return;
541  }
542 
543  // Handle operations.
544  if (auto op = tappedOps.lookup(key)) {
545  // We require the target to be a wire or node, such that it gets a name
546  // during Verilog emission.
547  if (!isa<WireOp, NodeOp, RegOp, RegResetOp>(op)) {
548  auto diag = blackBox.extModule.emitError("ReferenceDataTapKey on port ")
549  << portName << " must be a wire, node, or reg";
550  diag.attachNote(op->getLoc()) << "referenced operation is here:";
551  signalPassFailure();
552  return;
553  }
554 
555  // We currently require the target to be named.
556  // TODO: If we were to use proper cross-module reference ops in the IR
557  // then this could be anonymous, with ExportVerilog resolving the name
558  // at the last moment.
559  auto name = op->getAttrOfType<StringAttr>("name");
560  if (!name) {
561  auto diag =
562  op->emitError("declaration targeted by data tap must have a name");
563  diag.attachNote(blackBox.extModule->getLoc())
564  << "used by ReferenceDataTapKey on port " << portName << " here:";
565  signalPassFailure();
566  return;
567  }
568 
569  wiring.prefices =
570  instancePaths.getAbsolutePaths(op->getParentOfType<FModuleOp>());
571  wiring.suffix = name.getValue();
572  portWiring.push_back(std::move(wiring));
573  return;
574  }
575 
576  // The annotation scattering must have placed this annotation on some
577  // target operation or block argument, which we should have picked up in
578  // the tapped args or ops maps.
579  blackBox.extModule.emitOpError(
580  "ReferenceDataTapKey annotation was not scattered to "
581  "an operation: ")
582  << portAnno.anno.getDict();
583  signalPassFailure();
584  return;
585  }
586 
587  // Handle data taps on black boxes.
588  if (portAnno.anno.isClass(internalKeyClass)) {
589  auto op = tappedOps.lookup(key);
590  if (!op) {
591  blackBox.extModule.emitOpError(
592  "DataTapModuleSignalKey annotation was not scattered to "
593  "an operation: ")
594  << portAnno.anno.getDict();
595  signalPassFailure();
596  return;
597  }
598 
599  // Extract the internal path we're supposed to append.
600  auto internalPath = portAnno.anno.getMember<StringAttr>("internalPath");
601  if (!internalPath) {
602  blackBox.extModule.emitError("DataTapModuleSignalKey annotation on port ")
603  << portName << " missing \"internalPath\" attribute";
604  signalPassFailure();
605  return;
606  }
607 
608  wiring.prefices = instancePaths.getAbsolutePaths(op);
609  wiring.suffix = internalPath.getValue();
610  portWiring.push_back(std::move(wiring));
611  return;
612  }
613 
614  // Handle data taps with literals.
615  if (portAnno.anno.isClass(literalKeyClass)) {
616  blackBox.extModule.emitError(
617  "LiteralDataTapKey annotations not yet supported (on port ")
618  << portName << ")";
619  signalPassFailure();
620  return;
621  }
622 
623  // Handle memory taps.
624  if (portAnno.anno.isClass(memTapClass)) {
625  auto op = tappedOps.lookup(key);
626  if (!op) {
627  blackBox.extModule.emitOpError(
628  "MemTapAnnotation annotation was not scattered to "
629  "an operation: ")
630  << portAnno.anno.getDict();
631  signalPassFailure();
632  return;
633  }
634 
635  // Extract the name of the memory.
636  // TODO: This would be better handled through a proper cross-module
637  // reference preserved in the IR, such that ExportVerilog can insert a
638  // proper name here at the last moment.
639  auto name = op->getAttrOfType<StringAttr>("name");
640  if (!name) {

661  portWiring.push_back(std::move(wiring));
