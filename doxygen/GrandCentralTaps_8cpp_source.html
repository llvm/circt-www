153 /// Parse a FIRRTL `UInt`/`SInt` literal.
154 static FailureOr<Literal> parseIntegerLiteral(MLIRContext *context,
155  StringRef literal, Location loc) {
156  auto initial = literal; // used for error reporting
157  auto consumed = [&]() {
158  return initial.take_front(initial.size() - literal.size());
159  };
160  auto bail = [&](const Twine &message) {
161  mlir::emitError(loc, "expected ")
162  << message << " in literal after `" << consumed() << "`";
163  return failure();
164  };
165  auto consume = [&](StringRef text) {
166  if (!literal.consume_front(text)) {
167  (void)bail(Twine("`") + text + "`");
168  return false;
169  }
170  return true;
171  };
172 
173  // Parse the leading keyword.
174  bool isSigned;
175  if (literal.consume_front("UInt")) {
176  isSigned = false;
177  } else if (literal.consume_front("SInt")) {
178  isSigned = true;
179  } else {
180  mlir::emitError(loc, "expected leading `UInt` or `SInt` in literal");
181  return failure();
182  }
183 
184  // Parse the optional width.
185  Optional<APInt> width = {};
186  if (literal.consume_front("<")) {
187  auto widthLiteral = literal.take_while(llvm::isDigit);
188  APInt parsedWidth;
189  if (widthLiteral.getAsInteger(10, parsedWidth))
190  return bail("integer width");
191  literal = literal.drop_front(widthLiteral.size());
192  if (!literal.consume_front(">"))
193  return bail("closing `>`");
194  width = parsedWidth;
195  }
196 
197  // Parse the opening parenthesis.
198  if (!consume("("))
199  return failure();
200 
201  // Parse the opening quotes and base specifier.
202  unsigned base = 10;
203  bool hasQuotes = false;
204  if (literal.consume_front("\"")) {
205  hasQuotes = true;
206  if (literal.consume_front("h"))
207  base = 16;
208  else if (literal.consume_front("o"))
209  base = 8;
210  else if (literal.consume_front("b"))
211  base = 2;
212  else
213  return bail("base specifier (`h`, `o`, or `b`)");
214  }
215 
216  // Parse the optional sign.
217  bool isNegative = false;
218  if (literal.consume_front("-"))
219  isNegative = true;
220  else if (literal.consume_front("+"))
221  isNegative = false;
222 
223  // Parse the actual value.
224  APInt parsedValue;
225  auto valueLiteral = literal.take_while(llvm::isHexDigit);
226  if (valueLiteral.getAsInteger(base, parsedValue))
227  return bail("integer value");
228  literal = literal.drop_front(valueLiteral.size());
229 
230  // Parse the closing quotes.
231  if (hasQuotes && !literal.consume_front("\""))
232  return bail("closing quotes");
233 
234  // Parse the closing parenthesis.
235  if (!consume(")"))
236  return failure();
237 
238  // Ensure that there's no junk afterwards.
239  if (!literal.empty()) {
240  mlir::emitError(loc, "extraneous `")
241  << literal << "` in literal after `" << consumed() << "`";
242  return failure();
243  }
244 
245  // Ensure we have a 0 bit at the top to properly hold a negative value.
246  if (parsedValue.isNegative())
247  parsedValue = parsedValue.zext(parsedValue.getBitWidth() + 1);
248  if (isNegative)
249  parsedValue = -parsedValue;
250 
251  // Ensure the width is sound.
252  int32_t saneWidth = -1;
253  if (width) {
254  saneWidth = (int32_t)width->getLimitedValue(INT32_MAX);
255  if (saneWidth != *width) {
256  mlir::emitError(loc, "width of literal `")
257  << initial << "` is too big to handle";
258  return failure();
259  }
260  parsedValue = isSigned ? parsedValue.sextOrTrunc(saneWidth)
261  : parsedValue.zextOrTrunc(saneWidth);
262  }
263 
264  // Assemble the literal.
265  Literal lit;
266  lit.type = IntType::get(context, isSigned, saneWidth);
267  lit.value = IntegerAttr::get(
268  IntegerType::get(context, parsedValue.getBitWidth(),
269  isSigned ? IntegerType::Signed : IntegerType::Unsigned),
270  parsedValue);
271  return lit;
272 }

278 // Describes tap points into the design. This has the following structure:
279 // blackBox: ModuleTarget
280 // keys: Seq[DataTapKey]
281 // DataTapKey has multiple implementations:
282 // - ReferenceDataTapKey: (tapping a point which exists in the FIRRTL)
283 // portName: ReferenceTarget
284 // source: ReferenceTarget
285 // - DataTapModuleSignalKey: (tapping a point, by name, in a blackbox)
286 // portName: ReferenceTarget
287 // module: IsModule
288 // internalPath: String
289 // - DeletedDataTapKey: (not implemented here)
290 // portName: ReferenceTarget
291 // - LiteralDataTapKey: (not implemented here)
292 // portName: ReferenceTarget
293 // literal: Literal
294 // A Literal is a FIRRTL IR literal serialized to a string. For now, just
295 // store the string.
296 // TODO: Parse the literal string into a UInt or SInt literal.
297 LogicalResult circt::firrtl::applyGCTDataTaps(const AnnoPathValue &target,
298  DictionaryAttr anno,
299  ApplyState &state) {
300 
301  auto *context = state.circuit.getContext();
302  auto loc = state.circuit.getLoc();
303 
304  auto id = state.newID();
305  NamedAttrList attrs;
306  attrs.append("class", StringAttr::get(context, dataTapsBlackboxClass));
307  auto blackBoxAttr =
308  tryGetAs<StringAttr>(anno, anno, "blackBox", loc, dataTapsClass);
309  if (!blackBoxAttr)
310  return failure();
311  auto canonicalTarget = canonicalizeTarget(blackBoxAttr.getValue());
312  if (!tokenizePath(canonicalTarget))
313  return failure();
314  attrs.append("target", StringAttr::get(context, canonicalTarget));
315  state.addToWorklistFn(DictionaryAttr::getWithSorted(context, attrs));
316 
317  // Process all the taps.
318  auto keyAttr = tryGetAs<ArrayAttr>(anno, anno, "keys", loc, dataTapsClass);
319  if (!keyAttr)
320  return failure();
321  for (size_t i = 0, e = keyAttr.size(); i != e; ++i) {
322  auto b = keyAttr[i];
323  auto path = ("keys[" + Twine(i) + "]").str();
324  auto bDict = b.cast<DictionaryAttr>();
325  auto classAttr =
326  tryGetAs<StringAttr>(bDict, anno, "class", loc, dataTapsClass, path);
327  if (!classAttr)
328  return failure();
329 
330  // The "portName" field is common across all sub-types of DataTapKey.
331  NamedAttrList port;
332  auto portNameAttr =
333  tryGetAs<StringAttr>(bDict, anno, "portName", loc, dataTapsClass, path);
334  if (!portNameAttr)
335  return failure();
336  auto portTarget = canonicalizeTarget(portNameAttr.getValue());
337  if (!tokenizePath(portTarget))
338  return failure();
339 
340  if (classAttr.getValue() == referenceKeyClass) {
341  NamedAttrList source;
342  auto portID = state.newID();
343  source.append("class", StringAttr::get(context, referenceKeySourceClass));
344  source.append("id", id);
345  source.append("portID", portID);
346  auto sourceAttr =
347  tryGetAs<StringAttr>(bDict, anno, "source", loc, dataTapsClass, path);
348  if (!sourceAttr)
349  return failure();
350  auto sourceTarget = canonicalizeTarget(sourceAttr.getValue());
351  if (!tokenizePath(sourceTarget))
352  return failure();
353 
354  source.append("target", StringAttr::get(context, sourceTarget));
355 
356  state.addToWorklistFn(DictionaryAttr::get(context, source));
357 
358  // Annotate the data tap module port.
359  port.append("class", StringAttr::get(context, referenceKeyPortClass));
360  port.append("id", id);
361  port.append("portID", portID);
362  port.append("target", StringAttr::get(context, portTarget));
363  state.addToWorklistFn(DictionaryAttr::getWithSorted(context, port));
364  continue;
365  }
366 
367  if (classAttr.getValue() == internalKeyClass) {
368  NamedAttrList module;
369  auto portID = state.newID();
370  module.append("class", StringAttr::get(context, internalKeySourceClass));
371  module.append("id", id);
372  auto internalPathAttr = tryGetAs<StringAttr>(bDict, anno, "internalPath",
373  loc, dataTapsClass, path);
374  auto moduleAttr =
375  tryGetAs<StringAttr>(bDict, anno, "module", loc, dataTapsClass, path);
376  if (!internalPathAttr || !moduleAttr)
377  return failure();
378  module.append("internalPath", internalPathAttr);
379  module.append("portID", portID);
380  auto moduleTarget = canonicalizeTarget(moduleAttr.getValue());
381  if (!tokenizePath(moduleTarget))
382  return failure();
383 
384  module.append("target", StringAttr::get(context, moduleTarget));
385  state.addToWorklistFn(DictionaryAttr::getWithSorted(context, module));
386 
387  // Annotate the data tap module port.
388  port.append("class", StringAttr::get(context, internalKeyPortClass));
389  port.append("id", id);
390  port.append("portID", portID);
391  port.append("target", StringAttr::get(context, portTarget));
392  state.addToWorklistFn(DictionaryAttr::get(context, port));
393  continue;
394  }
395 
396  if (classAttr.getValue() == deletedKeyClass) {
397  // Annotate the data tap module port.
398  port.append("class", classAttr);
399  port.append("id", id);
400  port.append("target", StringAttr::get(context, portTarget));
401  state.addToWorklistFn(DictionaryAttr::get(context, port));
402  continue;
403  }
404 
405  if (classAttr.getValue() == literalKeyClass) {
406  auto literalAttr = tryGetAs<StringAttr>(bDict, anno, "literal", loc,
407  dataTapsClass, path);
408  if (!literalAttr)
409  return failure();
410 
411  // Annotate the data tap module port.
412  port.append("class", classAttr);
413  port.append("id", id);
414  port.append("literal", literalAttr);
415  port.append("target", StringAttr::get(context, portTarget));
416  state.addToWorklistFn(DictionaryAttr::get(context, port));
417  continue;
418  }
419 
420  mlir::emitError(
421  loc, "Annotation '" + Twine(dataTapsClass) + "' with path '" + path +
422  ".class" +
423  +"' contained an unknown/unimplemented DataTapKey class '" +
424  classAttr.getValue() + "'.")
425  .attachNote()
426  << "The full Annotation is reprodcued here: " << anno << "\n";
427  return failure();
428  }
429 
430  return success();
431 }

433 LogicalResult circt::firrtl::applyGCTMemTaps(const AnnoPathValue &target,
434  DictionaryAttr anno,
435  ApplyState &state) {
436 
437  auto *context = state.circuit.getContext();
438  auto loc = state.circuit.getLoc();
439 
440  auto id = state.newID();
441  NamedAttrList attrs;
442  auto sourceAttr =
443  tryGetAs<StringAttr>(anno, anno, "source", loc, memTapClass);
444  if (!sourceAttr)
445  return failure();
446  auto sourceTarget = canonicalizeTarget(sourceAttr.getValue());
447  if (!tokenizePath(sourceTarget))
448  return failure();
449  attrs.append("class", StringAttr::get(context, memTapSourceClass));
450  attrs.append("id", id);
451  attrs.append("target", StringAttr::get(context, sourceTarget));
452  state.addToWorklistFn(DictionaryAttr::get(context, attrs));
453 
454  auto tapsAttr = tryGetAs<ArrayAttr>(anno, anno, "taps", loc, memTapClass);
455  if (!tapsAttr)
456  return failure();
457  StringSet<> memTapBlackboxes;
458  for (size_t i = 0, e = tapsAttr.size(); i != e; ++i) {
459  auto tap = tapsAttr[i].dyn_cast_or_null<StringAttr>();
460  if (!tap) {
461  mlir::emitError(
462  loc, "Annotation '" + Twine(memTapClass) + "' with path '.taps[" +
463  Twine(i) +
464  "]' contained an unexpected type (expected a string).")
465  .attachNote()
466  << "The full Annotation is reprodcued here: " << anno << "\n";
467  return failure();
468  }
469  NamedAttrList port;
470  port.append("class", StringAttr::get(context, memTapPortClass));
471  port.append("id", id);
472  port.append("portID", IntegerAttr::get(IntegerType::get(context, 64), i));
473  auto canonTarget = canonicalizeTarget(tap.getValue());
474  if (!tokenizePath(canonTarget))
475  return failure();
476  port.append("target", StringAttr::get(context, canonTarget));
477  state.addToWorklistFn(DictionaryAttr::get(context, port));
478 
479  auto blackboxTarget = tokenizePath(canonTarget).getValue();
480  blackboxTarget.name = {};
481  blackboxTarget.component.clear();
482  auto blackboxTargetStr = blackboxTarget.str();
483  if (!memTapBlackboxes.insert(blackboxTargetStr).second)
484  continue;
485 
486  NamedAttrList blackbox;
487  blackbox.append("class", StringAttr::get(context, memTapBlackboxClass));
488  blackbox.append("target", StringAttr::get(context, blackboxTargetStr));
489  state.addToWorklistFn(DictionaryAttr::getWithSorted(context, blackbox));
490  }
491 
492  return success();
493 }

505  // Helpers to simplify collecting taps on the various things.
506  void gatherTap(Annotation anno, Port port) {
507  auto key = getKey(anno);
508  annos.insert({key, anno});
509  assert(!tappedPorts.count(key) && "ambiguous tap annotation");
510  auto portWidth = cast<FModuleLike>(port.first)
511  .getPortType(port.second)
512  .getBitWidthOrSentinel();
513  // If the port width is non-zero, process it normally. Otherwise, record it
514  // as being zero-width.
515  if (portWidth)
516  tappedPorts.insert({key, port});
517  else
518  zeroWidthTaps.insert(key);
519  if (auto sym = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal"))
520  deadNLAs.insert(sym.getAttr());
521  }
522  void gatherTap(Annotation anno, Operation *op) {
523  auto key = getKey(anno);
524  annos.insert({key, anno});
525  assert(!tappedOps.count(key) && "ambiguous tap annotation");
526  // If the port width is targeting a module (e.g., a blackbox) or if it has a
527  // non-zero width, process it normally. Otherwise, record it as being
528  // zero-width.
529  if (isa<FModuleLike>(op) ||
530  op->getResult(0).getType().cast<FIRRTLType>().getBitWidthOrSentinel())
531  tappedOps.insert({key, op});
532  else
533  zeroWidthTaps.insert(key);
534  if (auto sym = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal"))
535  deadNLAs.insert(sym.getAttr());
536  }

589 void GrandCentralTapsPass::runOnOperation() {
590  auto circuitOp = getOperation();
591  LLVM_DEBUG(llvm::dbgs() << "Running the GCT Data Taps pass\n");
592  SymbolTable symtbl(circuitOp);
593  circuitSymbols = &symtbl;
594  InnerSymbolTableCollection innerSymTblCol;
595  innerSymTblCol.populateTables(circuitOp);
596 
597  // Here's a rough idea of what the Scala code is doing:
598  // - Gather the `source` of all `keys` of all `DataTapsAnnotation`s throughout
599  // the design.
600  // - Convert the sources, which are specified on modules, to the absolute
601  // paths in all instances. E.g. module M tap x will produce a.x and b.x if
602  // there are two instances a and b of module M in the design.
603  // - All data tap keys are specified on black box ports.
604  // - The code then processes every DataTapsAnnotation separately as follows
605  // (with the targeted blackbox and keys):
606  // - Check for collisions between SV keywords and key port names (skip this).
607  // - Find all instances of the blackbox, but then just pick the first one (we
608  // should probably do this for each?)
609  // - Process each key independently as follows:
610  // - Look up the absolute path of the source in the map. Ensure it exists and
611  // is unambiguous. Make it relative to the blackbox instance path.
612  // - Look up the port on the black box.
613  // - Create a hierarchical SV name and store this as an assignment to the
614  // blackbox port.
615  // - DeletedDataTapKey: skip and don't create a wiring
616  // - LiteralDataTapKey: just the literal
617  // - ReferenceDataTapKey: relative path with "." + source name
618  // - DataTapModuleSignalKey: relative path with "." + internal path
619  // - Generate a body for the blackbox module with the signal mapping
620 
621  AnnotationSet circuitAnnotations(circuitOp);
622  if (auto extractAnno =
623  circuitAnnotations.getAnnotation(extractGrandCentralClass)) {
624  auto directory = extractAnno.getMember<StringAttr>("directory");
625  if (!directory) {
626  circuitOp->emitError()
627  << "contained an invalid 'ExtractGrandCentralAnnotation' that does "

817  // instance (path being the absolute path to that instance) to the
818  // tapped thing (prefix being the path to the tapped port, wire, or
819  // memory) by calling stripCommonPrefix(prefix, path). If the tapped
820  // thing includes an NLA, then the NLA path is appended to the rest of
821  // the path before the common prefix stripping is done.

824  // instance to the tapped object.

826  for (auto prefix : port.prefices) {

832  auto refPart = getRefPart(segment);
833  if (!refPart)
834  continue;

836  symtbl.lookup(getModPart(segment)));
837  prefixWithNLA.push_back(
838  cast<InstanceOp>(innerSymTbl.lookup(refPart)));
839  }
840  }
841

843  if (!shortestPrefix.hasValue() ||
844  relative.size() < shortestPrefix->size())
845  shortestPrefix.emplace(relative.begin(), relative.end());
846  }
847  if (!shortestPrefix.hasValue()) {
848  LLVM_DEBUG(llvm::dbgs() << " - Has no prefix, skipping\n");
849  continue;
850  }
851  LLVM_DEBUG(llvm::dbgs()
852  << " - Shortest prefix " << *shortestPrefix << "\n");
853 
854  // Determine the module at which the hierarchical name should start.
855  FModuleLike rootModule;
856  if (shortestPrefix->empty()) {
857  if (port.target.hasPort())
858  rootModule = cast<FModuleLike>(port.target.getOp());
859  else
860  rootModule = port.target.getOp()->getParentOfType<FModuleLike>();
861  } else
862  rootModule = shortestPrefix->front()->getParentOfType<FModuleLike>();
863 
864  SmallVector<Attribute> symbols;
865  SmallString<128> hname;
866  auto addSymbol = [&](Attribute symbol) {
867  auto id = symbols.size();
868  symbols.push_back(symbol);
869  if (!hname.empty())

871  ("{{" + Twine(id) + "}}").toVector(hname);
872  };
873

875  addSymbol(
876  FlatSymbolRefAttr::get(SymbolTable::getSymbolName(rootModule)));
877  for (auto inst : shortestPrefix.getValue())
878  addSymbol(getInnerRefTo(inst));
879 
880  if (port.target.getOp()) {
881  Attribute leaf;
882  if (port.target.hasPort())
883  leaf = getInnerRefTo(port.target.getOp(), port.target.getPort());
884  else
885  leaf = getInnerRefTo(port.target.getOp());
886  addSymbol(leaf);
887  }
888 
889  if (!port.suffix.empty()) {
890  hname += '.';
891  hname += port.suffix;
892  }
893  LLVM_DEBUG({
894  llvm::dbgs() << " - Connecting as " << hname;
895  if (!symbols.empty()) {
896  llvm::dbgs() << " (";
897  llvm::interleave(
898  symbols, llvm::dbgs(),
899  [&](Attribute sym) { llvm::dbgs() << sym; }, ".");
900  llvm::dbgs() << ")";
901  }
902  llvm::dbgs() << "\n";
903  });
904 
905  // Add a verbatim op that assigns this module port.
906  auto arg = impl.getArgument(port.portNum);
907  auto hnameExpr = builder.create<VerbatimExprOp>(
908  arg.getType().cast<FIRRTLType>(), hname, ValueRange{}, symbols);
909  builder.create<ConnectOp>(arg, hnameExpr);
910  }
911 
912  // Switch the instance from the original extmodule to this
913  // implementation. CAVEAT: If the same black box data tap module is
914  // instantiated in a parent module that itself is instantiated in
915  // different locations, this will pretty arbitrarily pick one of those
916  // locations.
917  path.back()->setAttr("moduleName", SymbolRefAttr::get(name));
918  }
919 
920  // Drop the original black box module.
921  blackBox.extModule.erase();
922  }
923 
924  // Garbage collect NLAs which were removed.
925  for (auto &op :
926  llvm::make_early_inc_range(circuitOp.getBody()->getOperations())) {
927  // Remove NLA anchors whose leaf annotations were removed.
928  if (auto nla = dyn_cast<HierPathOp>(op)) {
929  if (deadNLAs.contains(nla.getNameAttr()))
930  nla.erase();
931  continue;
932  }
933  }

978 void GrandCentralTapsPass::processAnnotation(AnnotatedPort &portAnno,
979  AnnotatedExtModule &blackBox,
980  InstancePathCache &instancePaths) {
981  LLVM_DEBUG(llvm::dbgs() << "- Processing port " << portAnno.portNum
982  << " anno " << portAnno.anno.getDict() << "\n");
983  auto key = getKey(portAnno.anno);
984  auto portName = blackBox.extModule.getPortNameAttr(portAnno.portNum);
985  PortWiring wiring;
986  wiring.portNum = portAnno.portNum;
987 
988  // Lookup the sibling annotation no the target. This may not exist, e.g. in
989  // the case of a `LiteralDataTapKey`, in which use the annotation on the
990  // data tap module port again.
991  auto targetAnnoIt = annos.find(key);
992  if (portAnno.anno.isClass(memTapPortClass) && targetAnnoIt == annos.end())
993  targetAnnoIt = annos.find({key.first, {}});
994  auto targetAnno =
995  targetAnnoIt != annos.end() ? targetAnnoIt->second : portAnno.anno;
996  LLVM_DEBUG(llvm::dbgs() << " Target anno " << targetAnno.getDict() << "\n");
997 
998  // NOTE:
999  // - portAnno holds the "*.port" flavor of the annotation
1000  // - targetAnno holds the "*.source" flavor of the annotation
1001 
1002  // If the annotation is non-local, look up the corresponding NLA operation to
1003  // find the exact instance path. We basically make the `wiring.prefices` array
1004  // of instance paths list all the possible paths to the beginning of the NLA
1005  // path. During wiring of the ports, we generate hierarchical names of the
1006  // form `<prefix>.<nla-path>.<suffix>`. If we don't have an NLA, we leave it
1007  // to the key-class-specific code below to come up with the possible prefices.
1008  HierPathOp nla;
1009  if (auto nlaSym = targetAnno.getMember<FlatSymbolRefAttr>("circt.nonlocal")) {
1010  nla = dyn_cast<HierPathOp>(circuitSymbols->lookup(nlaSym.getAttr()));
1011  assert(nla);
1012  // Find all paths to the root of the NLA.
1013  Operation *root = circuitSymbols->lookup(nla.root());
1014  wiring.nla = nla;
1015  wiring.prefices = instancePaths.getAbsolutePaths(root);
1016  }
1017 
1018  // Handle data taps on signals and ports.
1019  if (targetAnno.isClass(referenceKeySourceClass)) {
1020  // Handle ports.
1021  if (auto port = tappedPorts.lookup(key)) {
1022  if (!nla)
1023  wiring.prefices = instancePaths.getAbsolutePaths(port.first);
1024  wiring.target = PortWiring::Target(port.first, port.second);
1025  portWiring.push_back(std::move(wiring));
1026  return;
1027  }
1028 
1029  // Handle operations.
1030  if (auto op = tappedOps.lookup(key)) {
1031  // We require the target to be a wire or node, such that it gets a name
1032  // during Verilog emission.
1033  if (!isa<WireOp, NodeOp, RegOp, RegResetOp>(op)) {
1034  auto diag = blackBox.extModule.emitError("ReferenceDataTapKey on port ")
1035  << portName << " must be a wire, node, or reg";
1036  diag.attachNote(op->getLoc()) << "referenced operation is here:";
1037  signalPassFailure();
1038  return;
1039  }
1040 
1041  if (!nla)
1042  wiring.prefices =
1043  instancePaths.getAbsolutePaths(op->getParentOfType<FModuleOp>());
1044  wiring.target = PortWiring::Target(op);
1045 
1046  // If the tapped operation is trivially driven by a constant, set
1047  // information about the literal so that this can later be used instead of
1048  // an XMR.
1049  if (auto driver = getDriverFromConnect(op->getResult(0)))
1050  if (auto constant =
1051  dyn_cast_or_null<ConstantOp>(driver.getDefiningOp()))
1052  wiring.literal = {constant.valueAttr(), constant.getType()};
1053 
1054  portWiring.push_back(std::move(wiring));
1055  return;
1056  }
1057 
1058  // If the port is zero-width, then mark it as
1059  if (zeroWidthTaps.contains(key)) {
1060  wiring.zeroWidth = true;
1061  portWiring.push_back(std::move(wiring));
1062  return;
1063  }
1064 
1065  // The annotation scattering must have placed this annotation on some
1066  // target operation or block argument, which we should have picked up in
1067  // the tapped args or ops maps.
1068  blackBox.extModule.emitOpError(
1069  "ReferenceDataTapKey annotation was not scattered to "
1070  "an operation: ")
1071  << targetAnno.getDict();
1072  signalPassFailure();
1073  return;
1074  }
1075 
1076  // Handle data taps on black boxes.
1077  if (targetAnno.isClass(internalKeySourceClass)) {
1078  auto op = tappedOps.lookup(key);
1079  if (!op) {
1080  blackBox.extModule.emitOpError(
1081  "DataTapModuleSignalKey annotation was not scattered to "
1082  "an operation: ")
1083  << targetAnno.getDict();
1084  signalPassFailure();
1085  return;
1086  }
1087 
1088  // Extract the internal path we're supposed to append.
1089  auto internalPath = targetAnno.getMember<StringAttr>("internalPath");
1090  if (!internalPath) {
1091  blackBox.extModule.emitError("DataTapModuleSignalKey annotation on port ")
1092  << portName << " missing \"internalPath\" attribute";
1093  signalPassFailure();
1094  return;
1095  }
1096 
1097  if (!nla)
1098  wiring.prefices = instancePaths.getAbsolutePaths(op);
1099  wiring.suffix = internalPath.getValue();
1100  portWiring.push_back(std::move(wiring));
1101  return;
1102  }
1103 
1104  // Handle data taps with literals.
1105  if (targetAnno.isClass(literalKeyClass)) {
1106  auto literal = portAnno.anno.getMember<StringAttr>("literal");
1107  if (!literal) {
1108  blackBox.extModule.emitError("LiteralDataTapKey annotation on port ")
1109  << portName << " missing \"literal\" attribute";
1110  signalPassFailure();
1111  return;
1112  }
1113 
1114  // Parse the literal.
1115  auto parsed =
1116  parseIntegerLiteral(blackBox.extModule.getContext(), literal.getValue(),
1117  blackBox.extModule.getLoc());
1118  if (failed(parsed)) {
1119  blackBox.extModule.emitError("LiteralDataTapKey annotation on port ")
1120  << portName << " has invalid literal \"" << literal.getValue()
1121  << "\"";
1122  signalPassFailure();
1123  return;
1124  }
1125 
1126  wiring.literal = *parsed;
1127  portWiring.push_back(std::move(wiring));
1128  return;
1129  }
1130 
1131  // Handle memory taps.
1132  if (targetAnno.isClass(memTapSourceClass)) {
1133 
1134  // Handle operations.
1135  if (auto *op = tappedOps.lookup(key)) {
1136  // We require the target to be a wire or node, such that it gets a name
1137  // during Verilog emission.
1138  if (!isa<WireOp, NodeOp, RegOp, RegResetOp>(op)) {
1139  auto diag = blackBox.extModule.emitError("MemTapAnnotation on port ")
1140  << portName << " must be a wire, node, or reg";
1141  diag.attachNote(op->getLoc()) << "referenced operation is here:";
1142  signalPassFailure();
1143  return;
1144  }
1145 
1146  if (!nla)
1147  wiring.prefices =
1148  instancePaths.getAbsolutePaths(op->getParentOfType<FModuleOp>());
1149  wiring.target = PortWiring::Target(op);
1150  portWiring.push_back(std::move(wiring));
1151  return;
1152  }
1153  // This handles the case when the memtap is on a MemOp, which shouldn't have
1154  // the PortID attribute. Lookup the op without the portID key.
1155  if (auto *op = tappedOps.lookup({key.first, {}})) {
1156 
1157  // Extract the memory location we're supposed to access.
1158  auto word = portAnno.anno.getMember<IntegerAttr>("portID");
1159  if (!word) {
1160  blackBox.extModule.emitError("MemTapAnnotation annotation on port ")
1161  << portName << " missing \"word\" attribute";
1162  signalPassFailure();
1163  return;
1164  }
1165  // Formulate a hierarchical reference into the memory.
1166  // CAVEAT: This just assumes that the memory will map to something that
1167  // can be indexed in the final Verilog. If the memory gets turned into
1168  // an instance of some sort, we lack the information necessary to go in
1169  // and access individual elements of it. This will break horribly since
1170  // we generate memory impls out-of-line already, and memories coming
1171  // from an external generator are even worse. This needs a special node
1172  // in the IR that can properly inject the memory array on emission.
1173  if (!nla)
1174  wiring.prefices =
1175  instancePaths.getAbsolutePaths(op->getParentOfType<FModuleOp>());
1176  wiring.target = PortWiring::Target(op);
1177  ("Memory[" + Twine(word.getValue().getLimitedValue()) + "]")
1178  .toVector(wiring.suffix);
1179  portWiring.push_back(std::move(wiring));
1180  return;
1181  }
1182  blackBox.extModule.emitOpError(
1183  "MemTapAnnotation annotation was not scattered to "
1184  "an operation: ")
1185  << targetAnno.getDict();
1186  signalPassFailure();
1187  return;
1188  }
1189 
1190  // We never arrive here since the above code that populates the portAnnos
1191  // list only adds annotations that we handle in one of the if statements
1192  // above.
1193  llvm_unreachable("portAnnos is never populated with unsupported annos");
1194 }
