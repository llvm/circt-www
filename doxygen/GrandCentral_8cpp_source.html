30 /// Mutable store of information about an Element in an interface. This is
31 /// derived from information stored in the "elements" field of an
32 /// "AugmentedBundleType". This is updated as more information is known about
33 /// an Element.
34 struct ElementInfo {
35  /// Encodes the "tpe" of an element. This is called "Kind" to avoid
36  /// overloading the meaning of "Type" (which also conflicts with mlir::Type).
37  enum Kind {
38  Error = -1,
39  Ground,
40  Vector,
41  Bundle,
42  String,
43  Boolean,
44  Integer,
45  Double
46  };
47  /// The "tpe" field indicating if this element of the interface is a ground
48  /// type, a vector type, or a bundle type. Bundle types are nested
49  /// interfaces.
50  Kind tpe;
51  /// A string description that will show up as a comment in the output Verilog.
52  StringRef description;
53  /// The width of this interface. This is only non-negative for ground or
54  /// vector types.
55  int32_t width = -1;
56  /// The depth of the interface. This is one for ground types and greater
57  /// than one for vector types.
58  uint32_t depth = 0;
59  /// Indicate if this element was found in the circuit.
60  bool found = false;
61  /// Trakcs location information about what was used to build this element.
62  SmallVector<Location> locations = {};
63  /// True if this is a ground or vector type and it was not (statefully) found.
64  /// This indicates that an interface element, which is composed of ground and
65  /// vector types, found no matching, annotated components in the circuit.
66  bool isMissing() { return !found && (tpe == Ground || tpe == Vector); }
67 };

69 /// Stores a decoded Grand Central AugmentedField
70 struct AugmentedField {
71  /// The name of the field.
72  StringRef name;
73  /// An optional descripton that the user provided for the field. This should

75  StringRef description;
76  /// The "type" of the field.
77  ElementInfo::Kind tpe;

83  StringRef defName;

88 /// Convert an arbitrary attributes into an optional AugmentedField. Returns
89 /// None if the attribute is an invalid AugmentedField.
90 static Optional<AugmentedField> decodeField(Attribute maybeField) {
91  auto field = maybeField.dyn_cast_or_null<DictionaryAttr>();
92  if (!field)
93  return {};
94  auto tpeString = field.getAs<StringAttr>("tpe");
95  auto name = field.getAs<StringAttr>("name");
96  if (!name || !tpeString)
97  return {};
98  auto tpe = llvm::StringSwitch<ElementInfo::Kind>(tpeString.getValue())
99  .Case("sifive.enterprise.grandcentral.AugmentedBundleType",
100  ElementInfo::Bundle)
101  .Case("sifive.enterprise.grandcentral.AugmentedVectorType",
102  ElementInfo::Vector)
103  .Case("sifive.enterprise.grandcentral.AugmentedGroundType",
104  ElementInfo::Ground)
105  .Case("sifive.enterprise.grandcentral.AugmentedStringType",
106  ElementInfo::String)
107  .Case("sifive.enterprise.grandcentral.AugmentedBooleanType",
108  ElementInfo::Boolean)
109  .Case("sifive.enterprise.grandcentral.AugmentedIntegerType",
110  ElementInfo::Integer)
111  .Case("sifive.enterprise.grandcentral.AugmentedDoubleType",
112  ElementInfo::Double)
113  .Default(ElementInfo::Error);
114  if (tpe == ElementInfo::Error)
115  return {};
116 
117  StringRef description = {};
118  if (auto maybeDescription = field.getAs<StringAttr>("description"))
119  description = maybeDescription.getValue();
120  return Optional<AugmentedField>({name.getValue(), description, tpe});
121 };

123 /// Convert an Annotation into an optional AugmentedBundleType. Returns None if
124 /// the annotation is not an AugmentedBundleType.
125 static Optional<AugmentedBundleType> decodeBundleType(Annotation anno) {
126  auto defName = anno.getMember<StringAttr>("defName");
127  auto elements = anno.getMember<ArrayAttr>("elements");
128  if (!defName || !elements)
129  return {};
130  AugmentedBundleType bundle(
131  {defName.getValue(), SmallVector<AugmentedField>()});
132  for (auto element : elements) {
133  auto field = decodeField(element);
134  if (!field)
135  return {};
136  bundle.elements.push_back(field.getValue());
137  }
138  return Optional<AugmentedBundleType>(bundle);
139 };

141 /// Remove Grand Central Annotations associated with SystemVerilog interfaces
142 /// that should emitted. This pass works in three major phases:
143 ///
144 /// 1. The circuit's annotations are examnined to figure out _what_ interfaces
145 /// there are. This includes information about the name of the interface
146 /// ("defName") and each of the elements (sv::InterfaceSignalOp) that make up
147 /// the interface. However, no information about the _type_ of the elements
148 /// is known.
149 ///
150 /// 2. With this, information, walk through the circuit to find scattered
151 /// information about the types of the interface elements. Annotations are
152 /// scattered during FIRRTL parsing to attach all the annotations associated
153 /// with elements on the right components.
154 ///
155 /// 3. Add interface ops and populate the elements.
156 ///
157 /// Grand Central supports three "normal" element types and four "weird" element
158 /// types. The normal ones are ground types (SystemVerilog logic), vector types
159 /// (SystemVerilog unpacked arrays), and nested interface types (another
160 /// SystemVerilog interface). The Chisel API provides "weird" elements that
161 /// include: Boolean, Integer, String, and Double. The SFC implementation
162 /// currently drops these, but this pass emits them as commented out strings.
163 struct GrandCentralPass : public GrandCentralBase<GrandCentralPass> {
164  void runOnOperation() override;
165 
166  // A map storing mutable information about an element in an interface. This
167  // is keyed using a (defName, name) tuple where defname is the name of the
168  // interface and name is the name of the element.
169  typedef DenseMap<std::pair<StringRef, StringRef>, ElementInfo> InterfaceMap;
170 
171 private:
172  // Store a mapping of interface name to InterfaceOp.
173  llvm::StringMap<sv::InterfaceOp> interfaces;
174 
175  // Discovered interfaces that need to be constructed.
176  InterfaceMap interfaceMap;
177 
178  // Track the order that interfaces should be emitted in.
179  SmallVector<std::pair<StringRef, StringRef>> interfaceKeys;
180 };

237 /// Process all other operations. This will throw an error if the operation
238 /// contains any annotations that indicates that this should be included in an
239 /// interface. Otherwise, this is a valid nop.
240 void GrandCentralVisitor::visitUnhandledDecl(Operation *op) {
241  AnnotationSet annotations(op);
242  auto anno = annotations.getAnnotation(
243  "sifive.enterprise.grandcentral.AugmentedGroundType");
244  if (!anno)
245  return;
246  auto diag =
247  op->emitOpError()
248  << "is marked as a an interface element, but this op or its ports are "
249  "not supposed to be interface elements (Are your annotations "
250  "malformed? Is this a missing feature that should be supported?)";
251  diag.attachNote()
252  << "this annotation marked the op as an interface element: '" << anno
253  << "'";
254  failed = true;
255 }

257 /// Process annotations associated with an operation and having some type.
258 /// Return the annotations with the processed annotations removed. If all
259 /// annotations are removed, this returns an empty ArrayAttr.
260 void GrandCentralVisitor::handleRefLike(mlir::Operation *op,
261  AnnotationSet &annotations,
262  FIRRTLType type) {
263  if (annotations.empty())
264  return;
265 
266  bool foundAnnotations = false;
267  for (auto anno : annotations) {
268  if (!anno.isClass("sifive.enterprise.grandcentral.AugmentedGroundType"))
269  continue;
270 
271  auto defName = anno.getMember<StringAttr>("defName");
272  auto name = anno.getMember<StringAttr>("name");
273  if (!defName || !name) {
274  op->emitOpError(
275  "is marked as part of an interface, but is missing 'defName' or "
276  "'name' fields (did you forget to add these?)")
277  .attachNote()
278  << "the full annotation is: " << anno.getDict();
279  failed = true;
280  continue;
281  }
282 
283  // TODO: This is ignoring situations where the leaves of and interface are
284  // not ground types. This enforces the requirement that this runs after
285  // LowerTypes. However, this could eventually be relaxed.
286  if (!type.isGround()) {
287  auto diag = op->emitOpError()
288  << "cannot be added to interface '" << defName.getValue()
289  << "', component '" << name.getValue()
290  << "' because it is not a ground type. (Got type '" << type
291  << "'.) This will be dropped from the interface. (Did you "
292  "forget to run LowerTypes?)";
293  diag.attachNote()
294  << "The annotation indicating that this should be added was: '"
295  << anno.getDict();
296  failed = true;
297  continue;
298  }
299 
300  auto &component = interfaceMap[{defName.getValue(), name.getValue()}];
301  component.found = true;
302 
303  switch (component.tpe) {
304  case ElementInfo::Vector:
305  component.width = type.getBitWidthOrSentinel();
306  component.depth++;
307  component.locations.push_back(op->getLoc());
308  break;
309  case ElementInfo::Ground:
310  component.width = type.getBitWidthOrSentinel();
311  component.locations.push_back(op->getLoc());
312  break;
313  case ElementInfo::Bundle:
314  case ElementInfo::String:
315  case ElementInfo::Boolean:
316  case ElementInfo::Integer:
317  case ElementInfo::Double:
318  break;
319  case ElementInfo::Error:
320  llvm_unreachable("Shouldn't be here");
321  break;
322  }
323  annotations.removeAnnotation(anno);
324  foundAnnotations = true;
325  }
326 }

336 /// Remove Grand Central Annotations from ports of modules or external modules.
337 /// Return argument attributes with annotations removed.
338 void GrandCentralVisitor::handlePorts(Operation *op) {
339 
340  SmallVector<Attribute> newArgAttrs;
341  auto ports = getModulePortInfo(op);
342  for (size_t i = 0, e = ports.size(); i != e; ++i) {
343  auto port = ports[i];
344  handleRefLike(op, port.annotations, port.type);
345 
346  newArgAttrs.push_back(port.annotations.applyToPortDictionaryAttr(
347  DictionaryAttr::get(op->getContext(), {})));
348  }
349 
350  mlir::function_like_impl::setAllArgAttrDicts(op, newArgAttrs);
351 }

353 void GrandCentralVisitor::visitDecl(InstanceOp op) {
354 
355  // If this instance's underlying module has a "companion" annotation, then
356  // move this onto the actual instance op.
357  AnnotationSet annotations(op.getReferencedModule());
358  if (auto anno = annotations.getAnnotation(
359  "sifive.enterprise.grandcentral.GrandCentralView$"
360  "SerializedViewAnnotation")) {
361  auto tpe = anno.getAs<StringAttr>("type");
362  if (!tpe) {
363  op.getReferencedModule()->emitOpError(
364  "contains a GrandcCentralView$SerializedViewAnnotation that does not "
365  "contain a \"type\" field");
366  failed = true;
367  return;
368  }
369  if (tpe.getValue() == "companion")
370  op->setAttr("lowerToBind", BoolAttr::get(op.getContext(), true));
371  }
372 
373  visitUnhandledDecl(op);
374 }

376 void GrandCentralPass::runOnOperation() {
377  CircuitOp circuitOp = getOperation();
378 
379  AnnotationSet annotations(circuitOp);
380  if (annotations.empty())
381  return;
382 
383  auto builder = OpBuilder::atBlockEnd(circuitOp->getBlock());
384 
385  // Utility that acts like emitOpError, but does _not_ include a note. The
386  // note in emitOpError includes the entire op which means the **ENTIRE**
387  // FIRRTL circuit. This doesn't communicate anything useful to the user other
388  // than flooding their terminal.
389  auto emitCircuitError = [&circuitOp](StringRef message = {}) {
390  return emitError(circuitOp.getLoc(), message);
391  };
392 
393  // Examine the Circuit's Annotations doing work to remove Grand Central
394  // Annotations. Ignore any unprocesssed annotations and rewrite the Circuit's
395  // Annotations with these when done.
396  bool removalError = false;
397  annotations.removeAnnotations([&](auto anno) {
398  if (!anno.isClass("sifive.enterprise.grandcentral.AugmentedBundleType"))
399  return false;
400 
401  AugmentedBundleType bundle;
402  if (auto maybeBundle = decodeBundleType(anno))
403  bundle = maybeBundle.getValue();
404  else {
405  emitCircuitError(
406  "'firrtl.circuit' op contained an 'AugmentedBundleType' "
407  "Annotation which did not conform to the expected format")
408  .attachNote()
409  << "the problematic 'AugmentedBundleType' is: '" << anno.getDict()
410  << "'";
411  removalError = true;
412  return false;
413  }
414 
415  for (auto elt : bundle.elements) {
416  std::pair<StringRef, StringRef> key = {bundle.defName, elt.name};
417  interfaceMap[key] = {elt.tpe, elt.description};
418  interfaceKeys.push_back(key);
419  }
420 
421  // If the interface already exists, don't create it.
422  if (interfaces.count(bundle.defName))
423  return true;
424 
425  // Create the interface. This will be populated later.
426  interfaces[bundle.defName] =
427  builder.create<sv::InterfaceOp>(circuitOp->getLoc(), bundle.defName);
428 
429  return true;
430  });
431 
432  if (removalError)
433  return signalPassFailure();
434 
435  // Remove the processed annotations.
436  circuitOp->setAttr("annotations", annotations.getArrayAttr());
437 
438  // Walk through the circuit to collect additional information. If this fails,
439  // signal pass failure. Walk in reverse order so that annotations can be
440  // removed from modules after all referring instances have consumed their
441  // annotations.
442  for (auto &op : llvm::reverse(circuitOp.getBody()->getOperations())) {
443  if (isa<FModuleOp, FExtModuleOp>(op)) {
444  GrandCentralVisitor visitor(interfaceMap);
445  visitor.visitModule(&op);
446  if (visitor.hasFailed())
447  return signalPassFailure();
448  AnnotationSet annotations(&op);
449  annotations.removeAnnotations([](auto anno) {
450  return anno.isClass("sifive.enterprise.grandcentral.GrandCentralView$"
451  "SerializedViewAnnotation");
452  });
453  annotations.applyToOperation(&op);
454  }
455  }
456 
457  // Populate interfaces.
458  for (auto &a : interfaceKeys) {
459  auto defName = a.first;
460  auto name = a.second;
461 
462  auto &info = interfaceMap[{defName, name}];
463  if (info.isMissing()) {
464  emitCircuitError()
465  << "'firrtl.circuit' op contained a Grand Central Interface '"
466  << defName << "' that had an element '" << name
467  << "' which did not have a scattered companion annotation (is there "
468  "an invalid target in your annotation file?)";
469  continue;
470  }
471 
472  builder.setInsertionPointToEnd(interfaces[defName].getBodyBlock());
473 
474  auto loc = builder.getFusedLoc(info.locations);
475  auto description = info.description;
476  if (!description.empty())
477  builder.create<sv::VerbatimOp>(loc, "\n// " + description);
478 
479  switch (info.tpe) {
480  case ElementInfo::Bundle:
481  // TODO: Change this to actually use an interface type. This currently
482  // does not work because: (1) interfaces don't have a defined way to get
483  // their bit width and (2) interfaces have a symbol table that is used to
484  // verify internal ops, but this requires looking arbitrarily far upwards
485  // to find other symbols.
486  builder.create<sv::VerbatimOp>(loc, name + " " + name + "();");
487  break;
488  case ElementInfo::Vector: {
489  auto type = hw::UnpackedArrayType::get(builder.getIntegerType(info.width),
490  info.depth);
491  builder.create<sv::InterfaceSignalOp>(loc, name, type);
492  break;
493  }
494  case ElementInfo::Ground: {
495  auto type = builder.getIntegerType(info.width);
496  builder.create<sv::InterfaceSignalOp>(loc, name, type);
497  break;
498  }
499  case ElementInfo::String:
500  builder.create<sv::VerbatimOp>(loc, "// " + name +
501  " = <unsupported string type>;");
502  break;
503  case ElementInfo::Boolean:
504  builder.create<sv::VerbatimOp>(loc, "// " + name +
505  " = <unsupported boolean type>;");
506  break;
507  case ElementInfo::Integer:
508  builder.create<sv::VerbatimOp>(loc, "// " + name +
509  " = <unsupported integer type>;");
510  break;
511  case ElementInfo::Double:
512  builder.create<sv::VerbatimOp>(loc, "// " + name +
513  " = <unsupported double type>;");
514  break;
515  case ElementInfo::Error:
516  llvm_unreachable("Shouldn't be here");
517  break;
518  }
519  }
520 
521  interfaces.clear();
522  interfaceMap.clear();
523  interfaceKeys.clear();
524 }
