122 /// Convert newlines and comments to remove the comments. This produces better
123 /// looking YAML output. E.g., this will convert the following:
124 ///
125 /// // foo
126 /// // bar
127 ///
128 /// Into the following:
129 ///
130 /// foo
131 /// bar
132 std::string static stripComment(StringRef str) {
133  std::string descriptionString;
134  llvm::raw_string_ostream stream(descriptionString);
135  SmallVector<StringRef> splits;
136  str.split(splits, "\n");
137  llvm::interleave(
138  splits,
139  [&](auto substr) {
140  substr.consume_front("//");
141  stream << substr.drop_while([](auto c) { return c == ' '; });
142  },
143  [&]() { stream << "\n"; });
144  return descriptionString;
145 }

147 /// Conversion from a `DescribedSignal` to YAML. This is
148 /// implemented using YAML normalization to first convert this to an internal
149 /// `Field` structure which has a one-to-one mapping to the YAML represntation.
150 template <>
151 struct MappingContextTraits<DescribedSignal, Context> {
152  /// A one-to-one representation with a YAML representation of a signal/field.
153  struct Field {
154  /// The name of the field.
155  StringRef name;
156 
157  /// An optional, textual description of what the field is.
158  Optional<std::string> description;
159 
160  /// The dimensions of the field.
161  SmallVector<unsigned, 2> dimensions;
162 
163  /// The width of the underlying type.
164  unsigned width;
165 
166  /// Construct a `Field` from a `DescribedSignal` (an `sv::InterfaceSignalOp`
167  /// with an optional description).
168  Field(IO &io, DescribedSignal &op)
169  : name(op.signal.sym_nameAttr().getValue()) {
170 
171  // Convert the description from a `StringAttr` (which may be null) to an
172  // `Optional<StringRef>`. This aligns exactly with the YAML
173  // representation.
174  if (op.description)
175  description = stripComment(op.description.getValue());
176 
177  // Unwrap the type of the field into an array of dimensions and a width.
178  // By example, this is going from the following hardware type:
179  //
180  // !hw.uarray<1xuarray<2xuarray<3xi8>>>
181  //
182  // To the following representation:
183  //
184  // dimensions: [ 3, 2, 1 ]
185  // width: 8
186  //
187  // Note that the above is equivalenet to the following Verilog
188  // specification.
189  //
190  // wire [7:0] foo [2:0][1:0][0:0]
191  //
192  // Do this by repeatedly unwrapping unpacked array types until you get to
193  // the underlying type. The dimensions need to be reversed as this
194  // unwrapping happens in reverse order of the final representation.
195  auto tpe = op.signal.type();
196  while (auto vector = tpe.dyn_cast<hw::UnpackedArrayType>()) {
197  dimensions.push_back(vector.getSize());
198  tpe = vector.getElementType();
199  }
200  dimensions = SmallVector<unsigned>(llvm::reverse(dimensions));
201 
202  // The final non-array type must be an integer. Leave this as an assert
203  // with a blind cast because we generated this type in this pass (and we
204  // therefore cannot fail this cast).
205  assert(tpe.isa<IntegerType>());
206  width = tpe.cast<IntegerType>().getWidth();
207  }
208 
209  /// A no-argument constructor is necessary to work with LLVM's YAML library.
210  Field(IO &io){UNIMPLEMENTED_DEFAULT("Field")}
211 
212  /// This cannot be denomralized back to an interface op.
213  DescribedSignal denormalize(IO &) {
214  UNIMPLEMENTED_DENORM("DescribedSignal")
215  }
216  };

278 /// Conversion from an `sv::InterfaceOp` to YAML. This is implemented using
279 /// YAML normalization to first convert the interface to an internal `Interface`
280 /// which reformats the Grand Central-generated interface into the YAML format.
281 template <>
282 struct MappingContextTraits<sv::InterfaceOp, Context> {
283  /// A YAML-serializable representation of an interface. This consists of
284  /// fields (vector or ground types) and nested interfaces.
285  struct Interface {
286  /// The name of the interface.
287  StringRef name;
288 
289  /// All ground or vectors that make up the interface.
290  std::vector<DescribedSignal> fields;
291 
292  /// Instantiations of _other_ interfaces.
293  std::vector<DescribedInstance> instances;
294 
295  /// Construct an `Interface` from an `sv::InterfaceOp`. This is tuned to
296  /// "parse" the structure of an interface that the Grand Central pass
297  /// generates. The structure of `Field`s and `Instance`s is documented
298  /// below.
299  ///
300  /// A field will look like the following. The verbatim description is
301  /// optional:
302  ///
303  /// sv.verbatim "// <description>" {
304  /// firrtl.grandcentral.yaml.type = "description",
305  /// symbols = []}
306  /// sv.interface.signal @<name> : <type>
307  ///
308  /// An interface instanctiation will look like the following. The verbatim
309  /// description is optional.
310  ///
311  /// sv.verbatim "// <description>" {
312  /// firrtl.grandcentral.type = "description",
313  /// symbols = []}
314  /// sv.verbatim "<name> <symbol>();" {
315  /// firrtl.grandcentral.yaml.name = "<name>",
316  /// firrtl.grandcentral.yaml.dimensions = [<first dimension>, ...],
317  /// firrtl.grandcentral.yaml.symbol = @<symbol>,
318  /// firrtl.grandcentral.yaml.type = "instance",
319  /// symbols = []}
320  ///
321  Interface(IO &io, sv::InterfaceOp &op) : name(op.getName()) {
322  // A mutable store of the description. This occurs in the op _before_ the
323  // field or instance, so we need someplace to put it until we use it.
324  StringAttr description = {};
325 
326  for (auto &op : op.getBodyBlock()->getOperations()) {
327  TypeSwitch<Operation *>(&op)
328  // A verbatim op is either a description or an interface
329  // instantiation.
330  .Case<sv::VerbatimOp>([&](sv::VerbatimOp op) {
331  auto tpe = op->getAttrOfType<StringAttr>(
332  "firrtl.grandcentral.yaml.type");
333 
334  // This is a descripton. Update the mutable description and
335  // continue;
336  if (tpe.getValue() == "description") {
337  description = op.stringAttr();
338  return;
339  }
340 
341  // This is an unsupported construct. Just drop it.
342  if (tpe.getValue() == "unsupported") {
343  description = {};
344  return;
345  }
346 
347  // This is an instance of another interface. Add the symbol to
348  // the vector of instances.
349  auto name = op->getAttrOfType<StringAttr>(
350  "firrtl.grandcentral.yaml.name");
351  auto dimensions = op->getAttrOfType<ArrayAttr>(
352  "firrtl.grandcentral.yaml.dimensions");
353  auto symbol = op->getAttrOfType<FlatSymbolRefAttr>(
354  "firrtl.grandcentral.yaml.symbol");
355  instances.push_back(
356  DescribedInstance({name, description, dimensions, symbol}));
357  description = {};
358  })
359  // An interface signal op is a field.
360  .Case<sv::InterfaceSignalOp>([&](sv::InterfaceSignalOp op) {
361  fields.push_back(DescribedSignal({description, op}));
362  description = {};
363  });
364  }
365  }
366 
367  /// A no-argument constructor is necessary to work with LLVM's YAML library.
368  Interface(IO &io){UNIMPLEMENTED_DEFAULT("Interface")}
369 
370  /// This cannot be denomralized back to an interface op.
371  sv::InterfaceOp denormalize(IO &) {
372  UNIMPLEMENTED_DENORM("sv::InterfaceOp")
373  }
374  };

379  io.mapRequired("fields", keys->fields, ctx);
380  io.mapRequired("instances", keys->instances, ctx);
381  }
382 };

393 /// A helper to build verbatim strings with symbol placeholders. Provides a
394 /// mechanism to snapshot the current string and symbols and restore back to
395 /// this state after modifications. These snapshots are particularly useful when
396 /// the string is assembled through hierarchical traversal of some sort, which
397 /// populates the string with a prefix common to all children of a hierarchy
398 /// (like the interface field traversal in the `GrandCentralPass`).
399 ///
400 /// The intended use is as follows:
401 ///
402 /// void baz(VerbatimBuilder &v) {
403 /// foo(v.snapshot().append("bar"));
404 /// }
405 ///
406 /// The function `baz` takes a snapshot of the current verbatim text `v`, adds
407 /// "bar" to it and calls `foo` with that appended verbatim text. After the call
408 /// to `foo` returns, any changes made by `foo` as well as the "bar" are dropped
409 /// from the verbatim text `v`, as the temporary snapshot goes out of scope.
410 struct VerbatimBuilder {
411  struct Base {
412  SmallString<128> string;
413  SmallVector<Attribute> symbols;
414  VerbatimBuilder builder() { return VerbatimBuilder(*this); }
415  operator VerbatimBuilder() { return builder(); }
416  };
417 
418  /// Constructing a builder will snapshot the `Base` which holds the actual
419  /// string and symbols.
420  VerbatimBuilder(Base &base)
421  : base(base), stringBaseSize(base.string.size()),
422  symbolsBaseSize(base.symbols.size()) {}
423 
424  /// Destroying a builder will reset the `Base` to the original string and
425  /// symbols.
426  ~VerbatimBuilder() {
427  base.string.resize(stringBaseSize);
428  base.symbols.resize(symbolsBaseSize);
429  }
430 
431  // Disallow copying.
432  VerbatimBuilder(const VerbatimBuilder &) = delete;
433  VerbatimBuilder &operator=(const VerbatimBuilder &) = delete;
434 
435  /// Take a snapshot of the current string and symbols. This returns a new
436  /// `VerbatimBuilder` that will reset to the current state of the string once
437  /// destroyed.
438  VerbatimBuilder snapshot() { return VerbatimBuilder(base); }
439 
440  /// Get the current string.
441  StringRef getString() const { return base.string; }
442  /// Get the current symbols;
443  ArrayRef<Attribute> getSymbols() const { return base.symbols; }
444 
445  /// Append to the string.
446  VerbatimBuilder &append(char c) {
447  base.string.push_back(c);
448  return *this;
449  }
450 
451  /// Append to the string.
452  VerbatimBuilder &append(const Twine &twine) {
453  twine.toVector(base.string);
454  return *this;
455  }
456 
457  /// Append a placeholder and symbol to the string.
458  VerbatimBuilder &append(Attribute symbol) {
459  unsigned id = base.symbols.size();
460  base.symbols.push_back(symbol);
461  append("{{" + Twine(id) + "}}");
462  return *this;
463  }
464 
465  VerbatimBuilder &operator+=(char c) { return append(c); }
466  VerbatimBuilder &operator+=(const Twine &twine) { return append(twine); }
467  VerbatimBuilder &operator+=(Attribute symbol) { return append(symbol); }
468 
469 private:
470  Base &base;
471  size_t stringBaseSize;
472  size_t symbolsBaseSize;
473 };

475 /// A wrapper around a string that is used to encode a type which cannot be
476 /// represented by an mlir::Type for some reason. This is currently used to
477 /// represent either an interface, a n-dimensional vector of interfaces, or a
478 /// tombstone for an actually unsupported type (e.g., an AugmentedBooleanType).
479 struct VerbatimType {
480  /// The textual representation of the type.
481  std::string str;
482 
483  /// True if this is a type which must be "instatiated" and requires a trailing
484  /// "()".
485  bool instantiation;
486 
487  /// A vector storing the width of each dimension of the type.
488  SmallVector<int32_t, 4> dimensions = {};
489

491  std::string toStr(StringRef name) {
492  SmallString<64> stringType(str);
493  stringType.append(" ");
494  stringType.append(name);
495  for (auto d : llvm::reverse(dimensions)) {
496  stringType.append("[");

498  stringType.append("]");
499  }

501  stringType.append("()");

504  }

730 /// Recursively walk a sifive.enterprise.grandcentral.AugmentedType to extract
731 /// any annotations it may contain. This is going to generate two types of
732 /// annotations:
733 /// 1) Annotations necessary to build interfaces and store them at "~"
734 /// 2) Scattered annotations for how components bind to interfaces
735 static Optional<DictionaryAttr>
736 parseAugmentedType(ApplyState &state, DictionaryAttr augmentedType,
737  DictionaryAttr root, StringRef companion, StringAttr name,
738  StringAttr defName, Optional<IntegerAttr> id,
739  Optional<StringAttr>(description), Twine clazz,
740  Twine path = {}) {
741 
742  auto *context = state.circuit.getContext();
743  auto loc = state.circuit.getLoc();
744 
745  /// Optionally unpack a ReferenceTarget encoded as a DictionaryAttr. Return
746  /// either a pair containing the Target string (up to the reference) and an
747  /// array of components or none if the input is malformed. The input
748  /// DicionaryAttr encoding is a JSON object of a serialized ReferenceTarget
749  /// Scala class. By example, this is converting:
750  /// ~Foo|Foo>a.b[0]
751  /// To:
752  /// {"~Foo|Foo>a", {".b", "[0]"}}
753  /// The format of a ReferenceTarget object like:
754  /// circuit: String
755  /// module: String
756  /// path: Seq[(Instance, OfModule)]
757  /// ref: String
758  /// component: Seq[TargetToken]
759  auto refToTarget =
760  [&](DictionaryAttr refTarget) -> llvm::Optional<std::string> {
761  auto circuitAttr =
762  tryGetAs<StringAttr>(refTarget, refTarget, "circuit", loc, clazz, path);
763  auto moduleAttr =
764  tryGetAs<StringAttr>(refTarget, refTarget, "module", loc, clazz, path);
765  auto pathAttr =
766  tryGetAs<ArrayAttr>(refTarget, refTarget, "path", loc, clazz, path);
767  auto componentAttr = tryGetAs<ArrayAttr>(refTarget, refTarget, "component",
768  loc, clazz, path);
769  if (!circuitAttr || !moduleAttr || !pathAttr || !componentAttr)
770  return {};
771 
772  // Parse non-local annotations.
773  SmallString<32> strpath;
774  for (auto p : pathAttr) {
775  auto dict = p.dyn_cast_or_null<DictionaryAttr>();
776  if (!dict) {
777  mlir::emitError(loc, "annotation '" + clazz +
778  " has invalid type (expected DictionaryAttr)");
779  return {};
780  }
781  auto instHolder =
782  tryGetAs<DictionaryAttr>(dict, dict, "_1", loc, clazz, path);
783  auto modHolder =
784  tryGetAs<DictionaryAttr>(dict, dict, "_2", loc, clazz, path);
785  if (!instHolder || !modHolder) {
786  mlir::emitError(loc, "annotation '" + clazz +
787  " has invalid type (expected DictionaryAttr)");
788  return {};
789  }
790  auto inst = tryGetAs<StringAttr>(instHolder, instHolder, "value", loc,
791  clazz, path);
792  auto mod =
793  tryGetAs<StringAttr>(modHolder, modHolder, "value", loc, clazz, path);
794  if (!inst || !mod) {
795  mlir::emitError(loc, "annotation '" + clazz +
796  " has invalid type (expected DictionaryAttr)");
797  return {};
798  }
799  strpath += "/" + inst.getValue().str() + ":" + mod.getValue().str();
800  }
801 
802  SmallVector<Attribute> componentAttrs;
803  SmallString<32> componentStr;
804  for (size_t i = 0, e = componentAttr.size(); i != e; ++i) {
805  auto cPath = (path + ".component[" + Twine(i) + "]").str();
806  auto component = componentAttr[i];
807  auto dict = component.dyn_cast_or_null<DictionaryAttr>();
808  if (!dict) {
809  mlir::emitError(loc, "annotation '" + clazz + "' with path '" + cPath +
810  " has invalid type (expected DictionaryAttr)");
811  return {};
812  }
813  auto classAttr =
814  tryGetAs<StringAttr>(dict, refTarget, "class", loc, clazz, cPath);
815  if (!classAttr)
816  return {};
817 
818  auto value = dict.get("value");
819 
820  // A subfield like "bar" in "~Foo|Foo>foo.bar".
821  if (auto field = value.dyn_cast<StringAttr>()) {
822  assert(classAttr.getValue() == "firrtl.annotations.TargetToken$Field" &&
823  "A StringAttr target token must be found with a subfield target "
824  "token.");
825  componentStr.append((Twine(".") + field.getValue()).str());
826  continue;
827  }
828 
829  // A subindex like "42" in "~Foo|Foo>foo[42]".
830  if (auto index = value.dyn_cast<IntegerAttr>()) {
831  assert(classAttr.getValue() == "firrtl.annotations.TargetToken$Index" &&
832  "An IntegerAttr target token must be found with a subindex "
833  "target token.");
834  componentStr.append(
835  (Twine("[") + Twine(index.getValue().getZExtValue()) + "]").str());
836  continue;
837  }
838 
839  mlir::emitError(loc,
840  "Annotation '" + clazz + "' with path '" + cPath +
841  ".value has unexpected type (should be StringAttr "
842  "for subfield or IntegerAttr for subindex).")
843  .attachNote()
844  << "The value received was: " << value << "\n";
845  return {};
846  }
847 
848  auto refAttr =
849  tryGetAs<StringAttr>(refTarget, refTarget, "ref", loc, clazz, path);
850 
851  return llvm::Optional<std::string>(
852  {(Twine("~" + circuitAttr.getValue() + "|" + moduleAttr.getValue() +
853  strpath + ">" + refAttr.getValue()) +
854  componentStr)
855  .str()});
856  };
857 
858  auto classAttr =
859  tryGetAs<StringAttr>(augmentedType, root, "class", loc, clazz, path);
860  if (!classAttr)
861  return None;
862  StringRef classBase = classAttr.getValue();
863  if (!classBase.consume_front("sifive.enterprise.grandcentral.Augmented")) {
864  mlir::emitError(loc,
865  "the 'class' was expected to start with "
866  "'sifive.enterprise.grandCentral.Augmented*', but was '" +
867  classAttr.getValue() + "' (Did you misspell it?)")
868  .attachNote()
869  << "see annotation: " << augmentedType;
870  return None;
871  }
872 
873  // An AugmentedBundleType looks like:
874  // "defName": String
875  // "elements": Seq[AugmentedField]
876  if (classBase == "BundleType") {
877  defName =
878  tryGetAs<StringAttr>(augmentedType, root, "defName", loc, clazz, path);
879  if (!defName)
880  return None;
881 
882  // Each element is an AugmentedField with members:
883  // "name": String
884  // "description": Option[String]
885  // "tpe": AugmenetedType
886  SmallVector<Attribute> elements;
887  auto elementsAttr =
888  tryGetAs<ArrayAttr>(augmentedType, root, "elements", loc, clazz, path);
889  if (!elementsAttr)
890  return None;
891  for (size_t i = 0, e = elementsAttr.size(); i != e; ++i) {
892  auto field = elementsAttr[i].dyn_cast_or_null<DictionaryAttr>();
893  if (!field) {
894  mlir::emitError(
895  loc,
896  "Annotation '" + Twine(clazz) + "' with path '.elements[" +
897  Twine(i) +
898  "]' contained an unexpected type (expected a DictionaryAttr).")
899  .attachNote()
900  << "The received element was: " << elementsAttr[i] << "\n";
901  return None;
902  }
903  auto ePath = (path + ".elements[" + Twine(i) + "]").str();
904  auto name = tryGetAs<StringAttr>(field, root, "name", loc, clazz, ePath);
905  auto tpe =
906  tryGetAs<DictionaryAttr>(field, root, "tpe", loc, clazz, ePath);
907  Optional<StringAttr> description = None;
908  if (auto maybeDescription = field.get("description"))
909  description = maybeDescription.cast<StringAttr>();
910  auto eltAttr =
911  parseAugmentedType(state, tpe, root, companion, name, defName, None,
912  description, clazz, path);
913  if (!name || !tpe || !eltAttr)
914  return None;
915 
916  // Collect information necessary to build a module with this view later.
917  // This includes the optional description and name.
918  NamedAttrList attrs;
919  if (auto maybeDescription = field.get("description"))
920  attrs.append("description", maybeDescription.cast<StringAttr>());
921  attrs.append("name", name);
922  attrs.append("tpe", tpe.getAs<StringAttr>("class"));
923  elements.push_back(eltAttr.getValue());
924  }
925  // Add an annotation that stores information necessary to construct the
926  // module for the view. This needs the name of the module (defName) and the
927  // names of the components inside it.
928  NamedAttrList attrs;
929  attrs.append("class", classAttr);
930  attrs.append("defName", defName);
931  if (description)
932  attrs.append("description", description.getValue());
933  attrs.append("elements", ArrayAttr::get(context, elements));
934  if (id)
935  attrs.append("id", id.getValue());
936  attrs.append("name", name);
937  return DictionaryAttr::getWithSorted(context, attrs);
938  }
939 
940  // An AugmentedGroundType looks like:
941  // "ref": ReferenceTarget
942  // "tpe": GroundType
943  // The ReferenceTarget is not serialized to a string. The GroundType will
944  // either be an actual FIRRTL ground type or a GrandCentral uninferred type.
945  // This can be ignored for us.
946  if (classBase == "GroundType") {
947  auto maybeTarget = refToTarget(augmentedType.getAs<DictionaryAttr>("ref"));
948  if (!maybeTarget) {
949  mlir::emitError(loc, "Failed to parse ReferenceTarget").attachNote()
950  << "See the full Annotation here: " << root;
951  return None;
952  }
953 
954  auto id = state.newID();
955 
956  // TODO: We don't support non-local annotations, so force this annotation
957  // into a local annotation. This does not properly check that the
958  // non-local and local targets are totally equivalent.
959  auto target = maybeTarget.getValue();
960 
961  NamedAttrList elementIface, elementScattered;
962 
963  // Populate the annotation for the interface element.
964  elementIface.append("class", classAttr);
965  if (description)
966  elementIface.append("description", description.getValue());
967  elementIface.append("id", id);
968  elementIface.append("name", name);
969  // Populate an annotation that will be scattered onto the element.
970  elementScattered.append("class", classAttr);
971  elementScattered.append("id", id);
972  // If there are sub-targets, then add these.
973  auto targetAttr = StringAttr::get(context, target);
974  elementScattered.append("target", targetAttr);
975 
976  state.addToWorklistFn(
977  DictionaryAttr::getWithSorted(context, elementScattered));
978 
979  return DictionaryAttr::getWithSorted(context, elementIface);
980  }
981 
982  // An AugmentedVectorType looks like:
983  // "elements": Seq[AugmentedType]
984  if (classBase == "VectorType") {
985  auto elementsAttr =
986  tryGetAs<ArrayAttr>(augmentedType, root, "elements", loc, clazz, path);
987  if (!elementsAttr)
988  return None;
989  SmallVector<Attribute> elements;
990  for (auto elt : elementsAttr) {
991  auto eltAttr = parseAugmentedType(
992  state, elt.cast<DictionaryAttr>(), root, companion, name,
993  StringAttr::get(context, ""), id, None, clazz, path);
994  if (!eltAttr)
995  return None;
996  elements.push_back(eltAttr.getValue());
997  }
998  NamedAttrList attrs;
999  attrs.append("class", classAttr);
1000  if (description)
1001  attrs.append("description", description.getValue());
1002  attrs.append("elements", ArrayAttr::get(context, elements));
1003  attrs.append("name", name);
1004  return DictionaryAttr::getWithSorted(context, attrs);
1005  }
1006 
1007  // Any of the following are known and expected, but are legacy AugmentedTypes
1008  // do not have a target:
1009  // - AugmentedStringType
1010  // - AugmentedBooleanType
1011  // - AugmentedIntegerType
1012  // - AugmentedDoubleType
1013  bool isIgnorable =
1014  llvm::StringSwitch<bool>(classBase)
1015  .Cases("StringType", "BooleanType", "IntegerType", "DoubleType", true)
1016  .Default(false);
1017  if (isIgnorable)
1018  return augmentedType;
1019 
1020  // Anything else is unexpected or a user error if they manually wrote
1021  // annotations. Print an error and error out.
1022  mlir::emitError(loc, "found unknown AugmentedType '" + classAttr.getValue() +
1023  "' (Did you misspell it?)")
1024  .attachNote()
1025  << "see annotation: " << augmentedType;
1026  return None;
1027 }

1102  emitCircuitError() << "has an invalid AugmentedBundleType that does not "

1118  "contain 'id' and 'name' fields: "
1119  << dict;

1121  emitCircuitError() << "has an AugmentedGroundType with 'id == "

1123  << "' that does not have a scattered leaf to connect "
1124  "to in the circuit "
1125  "(was the leaf deleted or constant prop'd away?)";

1127  if (auto name = dict.getAs<StringAttr>("name"))

1156 bool GrandCentralPass::traverseField(Attribute field, IntegerAttr id,
1157  VerbatimBuilder &path) {
1158  return TypeSwitch<Attribute, bool>(field)
1159  .Case<AugmentedGroundTypeAttr>([&](AugmentedGroundTypeAttr ground) {
1160  auto [fieldRef, sym] = leafMap.lookup(ground.getID());
1161  HierPathOp nla;
1162  if (sym)
1163  nla = nlaTable->getNLA(sym.getAttr());
1164  Value leafValue = fieldRef.getValue();
1165  unsigned fieldID = fieldRef.getFieldID();
1166  assert(leafValue && "leafValue not found");
1167 
1168  auto builder =
1169  OpBuilder::atBlockEnd(companionIDMap.lookup(id).mapping.getBody());
1170 
1171  FIRRTLType tpe = leafValue.getType().cast<FIRRTLType>();
1172 
1173  // If the type is zero-width then do not emit an XMR.
1174  if (!tpe.getBitWidthOrSentinel())
1175  return true;
1176 
1177  // Generate the path from the LCA to the module that contains the leaf.
1178  path += " = ";
1179 
1180  // There are two posisibilites for what this is tapping:
1181  // 1. This is a constant that will be synced into the mappings file.
1182  // 2. This is something else and we need an XMR.
1183  // Handle case (1) here and exit. Handle case (2) following.
1184  auto driver = getDriverFromConnect(leafValue);
1185  if (driver) {
1186  if (auto constant =
1187  dyn_cast_or_null<ConstantOp>(driver.getDefiningOp())) {
1188  path.append(Twine(constant.value().getBitWidth()));
1189  path += "'h";
1190  SmallString<32> valueStr;
1191  constant.value().toStringUnsigned(valueStr, 16);
1192  path.append(valueStr);
1193  builder.create<sv::VerbatimOp>(
1194  constant.getLoc(),
1195  StringAttr::get(&getContext(),
1196  "assign " + path.getString() + ";"),
1197  ValueRange{}, ArrayAttr::get(&getContext(), path.getSymbols()));
1198  return true;
1199  }
1200  }
1201 
1202  // Populate a hierarchical path to the leaf. For an NLA this is just
1203  // the namepath of the associated hierarchical path. For a local
1204  // annotation, this is computed from the instance path.
1205  SmallVector<Attribute> fullLeafPath;
1206  if (nla) {
1207  fullLeafPath.append(nla.namepath().begin(), nla.namepath().end());
1208  } else {
1209  FModuleLike enclosing = getEnclosingModule(leafValue, sym);
1210  auto enclosingPaths = instancePaths->getAbsolutePaths(enclosing);
1211  assert(enclosingPaths.size() == 1 &&
1212  "Unable to handle multiply instantiated companions");
1213  if (enclosingPaths.size() != 1)
1214  return false;
1215  StringAttr root =
1216  instancePaths->instanceGraph.getTopLevelModule().moduleNameAttr();
1217  for (auto segment : enclosingPaths[0]) {
1218  fullLeafPath.push_back(getInnerRefTo(segment));
1219  root = segment.moduleNameAttr().getAttr();
1220  }
1221  fullLeafPath.push_back(FlatSymbolRefAttr::get(root));
1222  }
1223 
1224  // Compute the lowest common ancestor (LCA) of the leaf path and the
1225  // parent module. This enables the generated XMR to be as short as
1226  // possible while not losing specificity.
1227  ArrayRef<Attribute> minimalLeafPath(fullLeafPath);
1228  StringAttr parentNameAttr =
1229  parentIDMap.lookup(id).second.moduleNameAttr();
1230  minimalLeafPath = minimalLeafPath.drop_until(
1231  [&](Attribute attr) { return getModPart(attr) == parentNameAttr; });
1232 
1233  path += FlatSymbolRefAttr::get(getModPart(minimalLeafPath.front()));
1234  if (minimalLeafPath.size() > 0) {
1235  for (auto segment : minimalLeafPath.drop_back()) {
1236  path += ".";
1237  path += segment;
1238  }
1239  }
1240 
1241  // Add the leaf value to the path.
1242  auto uloc = builder.getUnknownLoc();
1243  path += '.';
1244  if (auto blockArg = leafValue.dyn_cast<BlockArgument>()) {
1245  auto module = cast<FModuleOp>(blockArg.getOwner()->getParentOp());
1246  path += getInnerRefTo(module, blockArg.getArgNumber());
1247  } else {
1248  path += getInnerRefTo(leafValue.getDefiningOp());
1249  }
1250 
1251  if (fieldID > tpe.getMaxFieldID()) {
1252  leafValue.getDefiningOp()->emitError()
1253  << "subannotation with fieldID=" << fieldID
1254  << " is too large for type " << tpe;
1255  return false;
1256  }
1257 
1258  // Construct a path given by fieldID.
1259  while (fieldID) {
1260  TypeSwitch<FIRRTLType>(tpe)
1261  .template Case<FVectorType>([&](FVectorType vector) {
1262  unsigned index = vector.getIndexForFieldID(fieldID);
1263  tpe = vector.getElementType();
1264  fieldID -= vector.getFieldID(index);
1265  path.append("[" + Twine(index) + "]");
1266  })
1267  .template Case<BundleType>([&](BundleType bundle) {
1268  unsigned index = bundle.getIndexForFieldID(fieldID);
1269  tpe = bundle.getElementType(index);
1270  fieldID -= bundle.getFieldID(index);
1271  // FIXME: Invalid verilog names (e.g. "begin", "reg", .. ) will
1272  // be renamed at ExportVerilog so the path constructed here
1273  // might become invalid. We can use an inner name ref to encode
1274  // a reference to a subfield.
1275  path.append("." + Twine(bundle.getElement(index).name));
1276  })
1277  .Default([&](auto op) {
1278  llvm_unreachable(
1279  "fieldID > maxFieldID case must be already handled");
1280  });
1281  }
1282 
1283  // Assemble the verbatim op.
1284  builder.create<sv::VerbatimOp>(
1285  uloc,
1286  StringAttr::get(&getContext(), "assign " + path.getString() + ";"),
1287  ValueRange{}, ArrayAttr::get(&getContext(), path.getSymbols()));
1288  ++numXMRs;
1289  return true;
1290  })
1291  .Case<AugmentedVectorTypeAttr>([&](auto vector) {
1292  bool notFailed = true;
1293  auto elements = vector.getElements();
1294  for (size_t i = 0, e = elements.size(); i != e; ++i) {
1295  auto field = fromAttr(elements[i]);
1296  if (!field)
1297  return false;
1298  notFailed &=
1299  traverseField(field.getValue(), id,
1300  path.snapshot().append("[" + Twine(i) + "]"));
1301  }
1302  return notFailed;
1303  })
1304  .Case<AugmentedBundleTypeAttr>([&](AugmentedBundleTypeAttr bundle) {
1305  bool anyFailed = true;
1306  for (auto element : bundle.getElements()) {
1307  auto field = fromAttr(element);
1308  if (!field)
1309  return false;
1310  auto name = element.cast<DictionaryAttr>().getAs<StringAttr>("name");
1311  if (!name)
1312  name = element.cast<DictionaryAttr>().getAs<StringAttr>("defName");
1313  anyFailed &=
1314  traverseField(field.getValue(), id,
1315  path.snapshot().append("." + name.getValue()));
1316  }
1317 
1318  return anyFailed;
1319  })
1320  .Case<AugmentedStringTypeAttr>([&](auto a) { return false; })
1321  .Case<AugmentedBooleanTypeAttr>([&](auto a) { return false; })
1322  .Case<AugmentedIntegerTypeAttr>([&](auto a) { return false; })
1323  .Case<AugmentedDoubleTypeAttr>([&](auto a) { return false; })
1324  .Case<AugmentedLiteralTypeAttr>([&](auto a) { return false; })
1325  .Case<AugmentedDeletedTypeAttr>([&](auto a) { return false; })
1326  .Default([](auto a) { return true; });
1327 }

1329 Optional<TypeSum> GrandCentralPass::computeField(Attribute field,
1330  IntegerAttr id,
1331  StringAttr prefix,
1332  VerbatimBuilder &path) {
1333 
1334  auto unsupported = [&](StringRef name, StringRef kind) {
1335  return VerbatimType({("// <unsupported " + kind + " type>").str(), false});
1336  };
1337 
1338  return TypeSwitch<Attribute, Optional<TypeSum>>(field)
1339  .Case<AugmentedGroundTypeAttr>(
1340  [&](AugmentedGroundTypeAttr ground) -> Optional<TypeSum> {
1341  // Traverse to generate mappings.
1342  if (!traverseField(field, id, path))
1343  return None;
1344  FieldRef fieldRef = leafMap.lookup(ground.getID()).field;
1345  auto value = fieldRef.getValue();
1346  auto fieldID = fieldRef.getFieldID();
1347  auto tpe = value.getType().cast<FIRRTLType>().getFinalTypeByFieldID(
1348  fieldID);
1349  if (!tpe.isGround()) {
1350  value.getDefiningOp()->emitOpError()
1351  << "cannot be added to interface with id '"
1352  << id.getValue().getZExtValue()
1353  << "' because it is not a ground type";
1354  return None;
1355  }
1356  return TypeSum(IntegerType::get(getOperation().getContext(),
1357  tpe.getBitWidthOrSentinel()));
1358  })
1359  .Case<AugmentedVectorTypeAttr>(
1360  [&](AugmentedVectorTypeAttr vector) -> Optional<TypeSum> {
1361  auto elements = vector.getElements();
1362  auto firstElement = fromAttr(elements[0]);
1363  auto elementType =
1364  computeField(firstElement.getValue(), id, prefix,
1365  path.snapshot().append("[" + Twine(0) + "]"));
1366  if (!elementType)
1367  return None;
1368 
1369  for (size_t i = 1, e = elements.size(); i != e; ++i) {
1370  auto subField = fromAttr(elements[i]);
1371  if (!subField)
1372  return None;
1373  (void)traverseField(subField.getValue(), id,
1374  path.snapshot().append("[" + Twine(i) + "]"));
1375  }
1376 
1377  if (auto *tpe = std::get_if<Type>(&elementType.getValue()))
1378  return TypeSum(
1379  hw::UnpackedArrayType::get(*tpe, elements.getValue().size()));
1380  auto str = std::get<VerbatimType>(elementType.getValue());
1381  str.dimensions.push_back(elements.getValue().size());
1382  return TypeSum(str);
1383  })
1384  .Case<AugmentedBundleTypeAttr>(
1385  [&](AugmentedBundleTypeAttr bundle) -> TypeSum {
1386  auto iface = traverseBundle(bundle, id, prefix, path);
1387  assert(iface && iface.getValue());
1388  (void)iface;
1389  return VerbatimType({getInterfaceName(prefix, bundle), true});
1390  })
1391  .Case<AugmentedStringTypeAttr>([&](auto field) -> TypeSum {
1392  return unsupported(field.getName().getValue(), "string");
1393  })
1394  .Case<AugmentedBooleanTypeAttr>([&](auto field) -> TypeSum {
1395  return unsupported(field.getName().getValue(), "boolean");
1396  })
1397  .Case<AugmentedIntegerTypeAttr>([&](auto field) -> TypeSum {
1398  return unsupported(field.getName().getValue(), "integer");
1399  })
1400  .Case<AugmentedDoubleTypeAttr>([&](auto field) -> TypeSum {
1401  return unsupported(field.getName().getValue(), "double");
1402  })
1403  .Case<AugmentedLiteralTypeAttr>([&](auto field) -> TypeSum {
1404  return unsupported(field.getName().getValue(), "literal");
1405  })
1406  .Case<AugmentedDeletedTypeAttr>([&](auto field) -> TypeSum {
1407  return unsupported(field.getName().getValue(), "deleted");
1408  });
1409 }

1411 /// Traverse an Annotation that is an AugmentedBundleType. During
1412 /// traversal, construct any discovered SystemVerilog interfaces. If this
1413 /// is the root interface, instantiate that interface in the parent. Recurse
1414 /// into fields of the AugmentedBundleType to construct nested interfaces
1415 /// and generate stringy-typed SystemVerilog hierarchical references to
1416 /// drive the interface. Returns false on any failure and true on success.
1417 Optional<sv::InterfaceOp>
1418 GrandCentralPass::traverseBundle(AugmentedBundleTypeAttr bundle, IntegerAttr id,
1419  StringAttr prefix, VerbatimBuilder &path) {
1420  auto builder = OpBuilder::atBlockEnd(getOperation().getBody());
1421  sv::InterfaceOp iface;
1422  builder.setInsertionPointToEnd(getOperation().getBody());
1423  auto loc = getOperation().getLoc();
1424  auto iFaceName = getNamespace().newName(getInterfaceName(prefix, bundle));
1425  iface = builder.create<sv::InterfaceOp>(loc, iFaceName);
1426  ++numInterfaces;
1427  if (dut &&
1428  !instancePaths->instanceGraph.isAncestor(companionIDMap[id].companion,
1429  cast<hw::HWModuleLike>(*dut)) &&
1430  testbenchDir)
1431  iface->setAttr("output_file",
1432  hw::OutputFileAttr::getFromDirectoryAndFilename(
1433  &getContext(), testbenchDir.getValue(),
1434  iFaceName + ".sv",
1435  /*excludFromFileList=*/true));
1436  else if (maybeExtractInfo)
1437  iface->setAttr("output_file",
1438  hw::OutputFileAttr::getFromDirectoryAndFilename(
1439  &getContext(), getOutputDirectory().getValue(),
1440  iFaceName + ".sv",
1441  /*excludFromFileList=*/true));
1442 
1443  builder.setInsertionPointToEnd(cast<sv::InterfaceOp>(iface).getBody());
1444 
1445  for (auto element : bundle.getElements()) {
1446  auto field = fromAttr(element);
1447  if (!field)
1448  return None;
1449 
1450  auto name = element.cast<DictionaryAttr>().getAs<StringAttr>("name");
1451  // auto signalSym = hw::InnerRefAttr::get(iface.sym_nameAttr(), name);
1452  // TODO: The `append(name.getValue())` in the following should actually be
1453  // `append(signalSym)`, but this requires that `computeField` and the
1454  // functions it calls always return a type for which we can construct an
1455  // `InterfaceSignalOp`. Since nested interface instances are currently
1456  // busted (due to the interface being a symbol table), this doesn't work at
1457  // the moment. Passing a `name` works most of the time, but can be brittle
1458  // if the interface field requires renaming in the output (e.g. due to
1459  // naming conflicts).
1460  auto elementType =
1461  computeField(field.getValue(), id, prefix,
1462  path.snapshot().append(".").append(name.getValue()));
1463  if (!elementType)
1464  return None;
1465 
1466  auto uloc = builder.getUnknownLoc();
1467  auto description =
1468  element.cast<DictionaryAttr>().getAs<StringAttr>("description");
1469  if (description) {
1470  auto descriptionOp = builder.create<sv::VerbatimOp>(
1471  uloc, ("// " + cleanupDescription(description.getValue())));
1472 
1473  // If we need to generate a YAML representation of this interface, then
1474  // add an attribute indicating that this `sv::VerbatimOp` is actually a
1475  // description.
1476  if (maybeHierarchyFileYAML)
1477  descriptionOp->setAttr("firrtl.grandcentral.yaml.type",
1478  builder.getStringAttr("description"));
1479  }
1480 
1481  if (auto *str = std::get_if<VerbatimType>(&elementType.getValue())) {
1482  auto instanceOp =
1483  builder.create<sv::VerbatimOp>(uloc, str->toStr(name.getValue()));
1484 
1485  // If we need to generate a YAML representation of the interface, then add
1486  // attirbutes that describe what this `sv::VerbatimOp` is.
1487  if (maybeHierarchyFileYAML) {
1488  if (str->instantiation)
1489  instanceOp->setAttr("firrtl.grandcentral.yaml.type",
1490  builder.getStringAttr("instance"));
1491  else
1492  instanceOp->setAttr("firrtl.grandcentral.yaml.type",
1493  builder.getStringAttr("unsupported"));
1494  instanceOp->setAttr("firrtl.grandcentral.yaml.name", name);
1495  instanceOp->setAttr("firrtl.grandcentral.yaml.dimensions",
1496  builder.getI32ArrayAttr(str->dimensions));
1497  instanceOp->setAttr(
1498  "firrtl.grandcentral.yaml.symbol",
1499  FlatSymbolRefAttr::get(builder.getContext(), str->str));
1500  }
1501  continue;
1502  }
1503 
1504  auto tpe = std::get<Type>(elementType.getValue());
1505  builder.create<sv::InterfaceSignalOp>(uloc, name.getValue(), tpe);
1506  }
1507 
1508  interfaceMap[FlatSymbolRefAttr::get(builder.getContext(), iFaceName)] = iface;
1509  return iface;
1510 }
