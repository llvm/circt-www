43 Attribute HWDialect::parseAttribute(DialectAsmParser &p, Type type) const {
44  StringRef attrName;
45  Attribute attr;
46  if (p.parseKeyword(&attrName))
47  return Attribute();
48  auto parseResult = generatedAttributeParser(p, attrName, type, attr);
49  if (parseResult.hasValue())
50  return attr;
51 
52  // Parse "#hw.param.expr.add" as ParamExprAttr.
53  if (attrName.startswith(ParamExprAttr::getMnemonic())) {
54  auto string = attrName.drop_front(ParamExprAttr::getMnemonic().size());
55  if (string.front() == '.')
56  return parseParamExprWithOpcode(string.drop_front(), p, type);
57  }
58 
59  p.emitError(p.getNameLoc(), "Unexpected hw attribute '" + attrName + "'");
60  return {};
61 }

73 static std::string canonicalizeFilename(const Twine &directory,
74  const Twine &filename) {
75  SmallString<128> fullPath;
76 
77  // If the filename is an absolute path, we don't need the directory.
78  if (llvm::sys::path::is_absolute(filename))
79  filename.toVector(fullPath);
80  else
81  llvm::sys::path::append(fullPath, directory, filename);
82 
83  // If this is a directory target, we need to ensure it ends with a `/`
84  if (filename.isTriviallyEmpty() && !fullPath.endswith("/"))
85  fullPath += "/";
86 
87  return std::string(fullPath);
88 }

119 /// Option ::= 'excludeFromFileList' | 'includeReplicatedOp'
120 /// OutputFileAttr ::= 'output_file<' directory ',' name (',' Option)* '>'
121 Attribute OutputFileAttr::parse(AsmParser &p, Type type) {
122  StringAttr filename;
123  if (p.parseLess() || p.parseAttribute<StringAttr>(filename))
124  return Attribute();
125 
126  // Parse the additional keyword attributes. Its easier to let people specify
127  // these more than once than to detect the problem and do something about it.
128  bool excludeFromFileList = false;
129  bool includeReplicatedOps = false;
130  while (true) {
131  if (p.parseOptionalComma())
132  break;
133  if (!p.parseOptionalKeyword("excludeFromFileList"))
134  excludeFromFileList = true;
135  else if (!p.parseKeyword("includeReplicatedOps",
136  "or 'excludeFromFileList'"))
137  includeReplicatedOps = true;
138  else
139  return Attribute();
140  }
141 
142  if (p.parseGreater())
143  return Attribute();
144 
145  auto *context = p.getContext();
146 
147  return OutputFileAttr::get(context, filename,
148  BoolAttr::get(context, excludeFromFileList),
149  BoolAttr::get(context, includeReplicatedOps));
150 }

281 /// Given a binary function, if the two operands are known constant integers,
282 /// use the specified fold function to compute the result.
283 static Attribute foldBinaryOp(
284  ArrayRef<Attribute> operands,
285  llvm::function_ref<APInt(const APInt &, const APInt &)> calculate) {
286  assert(operands.size() == 2 && "binary operator always has two operands");
287  if (auto lhs = operands[0].dyn_cast<IntegerAttr>())
288  if (auto rhs = operands[1].dyn_cast<IntegerAttr>())
289  return IntegerAttr::get(lhs.getType(),
290  calculate(lhs.getValue(), rhs.getValue()));
291  return {};
292 }

314 /// This implements a < comparison for two operands to an associative operation
315 /// imposing an ordering upon them.
316 ///
317 /// The ordering provided puts more complex things to the start of the list,
318 /// from left to right:
319 /// expressions :: verbatims :: decl.refs :: constant
320 ///
321 static bool paramExprOperandSortPredicate(Attribute lhs, Attribute rhs) {
322  // Simplify the code below - we never have to care about exactly equal values.
323  if (lhs == rhs)
324  return false;
325 
326  // All expressions are "less than" a constant, since they appear on the right.
327  if (rhs.isa<IntegerAttr>()) {
328  // We don't bother to order constants w.r.t. each other since they will be
329  // folded - they can all compare equal.
330  return !lhs.isa<IntegerAttr>();
331  }
332  if (lhs.isa<IntegerAttr>())
333  return false;
334 
335  // Next up are named parameters.
336  if (auto rhsParam = rhs.dyn_cast<ParamDeclRefAttr>()) {
337  // Parameters are sorted lexically w.r.t. each other.
338  if (auto lhsParam = lhs.dyn_cast<ParamDeclRefAttr>())
339  return lhsParam.getName().getValue() < rhsParam.getName().getValue();
340  // They otherwise appear on the right of other things.
341  return true;
342  }
343  if (lhs.isa<ParamDeclRefAttr>())
344  return false;
345 
346  // Next up are verbatim parameters.
347  if (auto rhsParam = rhs.dyn_cast<ParamVerbatimAttr>()) {
348  // Verbatims are sorted lexically w.r.t. each other.
349  if (auto lhsParam = lhs.dyn_cast<ParamVerbatimAttr>())
350  return lhsParam.getValue().getValue() < rhsParam.getValue().getValue();
351  // They otherwise appear on the right of other things.
352  return true;
353  }
354  if (lhs.isa<ParamVerbatimAttr>())
355  return false;
356 
357  // The only thing left are nested expressions.
358  auto lhsExpr = lhs.cast<ParamExprAttr>(), rhsExpr = rhs.cast<ParamExprAttr>();
359  // Sort by the string form of the opcode, e.g. add, .. mul,... then xor.
360  if (lhsExpr.getOpcode() != rhsExpr.getOpcode())
361  return stringifyPEO(lhsExpr.getOpcode()) <
362  stringifyPEO(rhsExpr.getOpcode());
363 
364  // If they are the same opcode, then sort by arity: more complex to the left.
365  ArrayRef<Attribute> lhsOperands = lhsExpr.getOperands(),
366  rhsOperands = rhsExpr.getOperands();
367  if (lhsOperands.size() != rhsOperands.size())
368  return lhsOperands.size() > rhsOperands.size();
369 
370  // We know the two subexpressions are different (they'd otherwise be pointer
371  // equivalent) so just go compare all of the elements.
372  for (size_t i = 0, e = lhsOperands.size(); i != e; ++i) {
373  if (paramExprOperandSortPredicate(lhsOperands[i], rhsOperands[i]))
374  return true;
375  if (paramExprOperandSortPredicate(rhsOperands[i], lhsOperands[i]))
376  return false;
377  }
378 
379  llvm_unreachable("expressions should never be equivalent");
380  return false;
381 }

383 /// Given a fully associative variadic integer operation, constant fold any
384 /// constant operands and move them to the right. If the whole expression is
385 /// constant, then return that, otherwise update the operands list.
386 static Attribute simplifyAssocOp(
387  PEO opcode, SmallVector<Attribute, 4> &operands,
388  llvm::function_ref<APInt(const APInt &, const APInt &)> calculateFn,
389  llvm::function_ref<bool(const APInt &)> identityConstantFn,
390  llvm::function_ref<bool(const APInt &)> destructiveConstantFn = {}) {
391  auto type = operands[0].getType();
392  assert(isHWIntegerType(type));
393  if (operands.size() == 1)
394  return operands[0];
395 
396  // Flatten any of the same operation into the operand list:
397  // `(add x, (add y, z))` => `(add x, y, z)`.
398  for (size_t i = 0, e = operands.size(); i != e; ++i) {
399  if (auto subexpr = dyn_castPE(opcode, operands[i])) {
400  std::swap(operands[i], operands.back());
401  operands.pop_back();
402  --e;
403  --i;
404  operands.append(subexpr.getOperands().begin(),
405  subexpr.getOperands().end());
406  }
407  }
408 
409  // Impose an ordering on the operands, pushing subexpressions to the left and
410  // constants to the right, with verbatims and parameters in the middle - but
411  // predictably ordered w.r.t. each other.
412  llvm::stable_sort(operands, paramExprOperandSortPredicate);
413 
414  // Merge any constants, they will appear at the back of the operand list now.
415  if (operands.back().isa<IntegerAttr>()) {
416  while (operands.size() >= 2 &&
417  operands[operands.size() - 2].isa<IntegerAttr>()) {
418  APInt c1 = operands.pop_back_val().cast<IntegerAttr>().getValue();
419  APInt c2 = operands.pop_back_val().cast<IntegerAttr>().getValue();
420  auto resultConstant = IntegerAttr::get(type, calculateFn(c1, c2));
421  operands.push_back(resultConstant);
422  }
423 
424  auto resultCst = operands.back().cast<IntegerAttr>();
425 
426  // If the resulting constant is the destructive constant (e.g. `x*0`), then
427  // return it.
428  if (destructiveConstantFn && destructiveConstantFn(resultCst.getValue()))
429  return resultCst;
430 
431  // Remove the constant back to our operand list if it is the identity
432  // constant for this operator (e.g. `x*1`) and there are other operands.
433  if (identityConstantFn(resultCst.getValue()) && operands.size() != 1)
434  operands.pop_back();
435  }
436 
437  return operands.size() == 1 ? operands[0] : Attribute();
438 }

440 /// Analyze an operand to an add. If it is a multiplication by a constant (e.g.
441 /// `(a*b*42)` then split it into the non-constant and the constant portions
442 /// (e.g. `a*b` and `42`). Otherwise return the operand as the first value and
443 /// null as the second (standin for "multiplication by 1").
444 static std::pair<Attribute, Attribute> decomposeAddend(Attribute operand) {
445  if (auto mul = dyn_castPE(PEO::Mul, operand))
446  if (auto cst = mul.getOperands().back().dyn_cast<IntegerAttr>()) {
447  auto nonCst = ParamExprAttr::get(PEO::Mul, mul.getOperands().drop_back());
448  return {nonCst, cst};
449  }
450  return {operand, Attribute()};
451 }

454  return IntegerAttr::get(type, APInt(type.getIntOrFloatBitWidth(), 1));

457 static Attribute simplifyAdd(SmallVector<Attribute, 4> &operands) {
458  if (auto result = simplifyAssocOp(
459  PEO::Add, operands, [](auto a, auto b) { return a + b; },
460  /*identityCst*/ [](auto cst) { return cst.isZero(); }))
461  return result;
462 
463  // Canonicalize the add by splitting all addends into their variable and
464  // constant factors.
465  SmallVector<std::pair<Attribute, Attribute>> decomposedOperands;
466  llvm::SmallDenseSet<Attribute> nonConstantParts;
467  for (auto &op : operands) {
468  decomposedOperands.push_back(decomposeAddend(op));
469 
470  // Keep track of non-constant parts we've already seen. If we see multiple
471  // uses of the same value, then we can fold them together with a multiply.
472  // This handles things like `(a+b+a)` => `(a*2 + b)` and `(a*2 + b + a)` =>
473  // `(a*3 + b)`.
474  if (!nonConstantParts.insert(decomposedOperands.back().first).second) {
475  // The thing we multiply will be the common expression.
476  Attribute mulOperand = decomposedOperands.back().first;
477 
478  // Find the index of the first occurrence.
479  size_t i = 0;
480  while (decomposedOperands[i].first != mulOperand)
481  ++i;
482  // Remove both occurrences from the operand list.
483  operands.erase(operands.begin() + (&op - &operands[0]));
484  operands.erase(operands.begin() + i);
485 
486  auto type = mulOperand.getType();
487  auto c1 = decomposedOperands[i].second,
488  c2 = decomposedOperands.back().second;
489  // Fill in missing constant multiplicands with 1.
490  if (!c1)
491  c1 = getOneOfType(type);
492  if (!c2)
493  c2 = getOneOfType(type);
494  // Re-add the "a"*(c1+c2) expression to the operand list and
495  // re-canonicalize.
496  auto constant = ParamExprAttr::get(PEO::Add, c1, c2);
497  auto mulCst = ParamExprAttr::get(PEO::Mul, mulOperand, constant);
498  operands.push_back(mulCst);
499  return ParamExprAttr::get(PEO::Add, operands);
500  }
501  }
502 
503  return {};
504 }

506 static Attribute simplifyMul(SmallVector<Attribute, 4> &operands) {
507  if (auto result = simplifyAssocOp(
508  PEO::Mul, operands, [](auto a, auto b) { return a * b; },
509  /*identityCst*/ [](auto cst) { return cst.isOne(); },
510  /*destructiveCst*/ [](auto cst) { return cst.isZero(); }))
511  return result;
512 
513  // We always build a sum-of-products representation, so if we see an addition
514  // as a subexpr, we need to pull it out: (a+b)*c*d ==> (a*c*d + b*c*d).
515  for (size_t i = 0, e = operands.size(); i != e; ++i) {
516  if (auto subexpr = dyn_castPE(PEO::Add, operands[i])) {
517  // Pull the `c*d` operands out - it is whatever operands remain after
518  // removing the `(a+b)` term.
519  SmallVector<Attribute> mulOperands(operands.begin(), operands.end());
520  mulOperands.erase(mulOperands.begin() + i);
521 
522  // Build each add operand.
523  SmallVector<Attribute> addOperands;
524  for (auto addOperand : subexpr.getOperands()) {
525  mulOperands.push_back(addOperand);
526  addOperands.push_back(ParamExprAttr::get(PEO::Mul, mulOperands));
527  mulOperands.pop_back();
528  }
529  // Canonicalize and form the add expression.
530  return ParamExprAttr::get(PEO::Add, addOperands);
531  }
532  }
533 
534  return {};
535 }

556 static Attribute simplifyShl(SmallVector<Attribute, 4> &operands) {
557  assert(isHWIntegerType(operands[0].getType()));
558 
559  if (auto rhs = operands[1].dyn_cast<IntegerAttr>()) {
560  // Constant fold simple integers.
561  if (auto lhs = operands[0].dyn_cast<IntegerAttr>())
562  return IntegerAttr::get(lhs.getType(),
563  lhs.getValue().shl(rhs.getValue()));
564 
565  // Canonicalize `x << cst` => `x * (1<<cst)` to compose correctly with
566  // add/mul canonicalization.
567  auto rhsCst = APInt::getOneBitSet(rhs.getValue().getBitWidth(),
568  rhs.getValue().getZExtValue());
569  return ParamExprAttr::get(PEO::Mul, operands[0],
570  IntegerAttr::get(rhs.getType(), rhsCst));
571  }
572  return {};
573 }

643 static Attribute simplifyStrConcat(SmallVector<Attribute, 4> &operands) {
644  // Combine all adjacent strings.
645  SmallVector<Attribute> newOperands;
646  SmallVector<StringAttr> stringsToCombine;
647  auto combineAndPush = [&]() {
648  if (stringsToCombine.empty())
649  return;
650  // Concatenate buffered strings, push to ops.
651  SmallString<32> newString;
652  for (auto part : stringsToCombine)
653  newString.append(part.getValue());
654  newOperands.push_back(
655  StringAttr::get(stringsToCombine[0].getContext(), newString));
656  stringsToCombine.clear();
657  };
658 
659  for (Attribute op : operands) {
660  if (auto strOp = op.dyn_cast<StringAttr>()) {
661  // Queue up adjacent strings.
662  stringsToCombine.push_back(strOp);
663  } else {
664  combineAndPush();
665  newOperands.push_back(op);
666  }
667  }
668  combineAndPush();
669 
670  assert(!newOperands.empty());
671  if (newOperands.size() == 1)
672  return newOperands[0];
673  if (newOperands.size() < operands.size())
674  return ParamExprAttr::get(PEO::StrConcat, newOperands);
675  return {};
676 }

678 /// Build a parameter expression. This automatically canonicalizes and
679 /// folds, so it may not necessarily return a ParamExprAttr.
680 Attribute ParamExprAttr::get(PEO opcode, ArrayRef<Attribute> operandsIn) {
681  assert(!operandsIn.empty() && "Cannot have expr with no operands");
682  // All operands must have the same type, which is the type of the result.
683  auto type = operandsIn.front().getType();
684  assert(llvm::all_of(operandsIn.drop_front(),
685  [&](auto op) { return op.getType() == type; }));
686 
687  SmallVector<Attribute, 4> operands(operandsIn.begin(), operandsIn.end());
688 
689  // Verify and canonicalize parameter expressions.
690  Attribute result;
691  switch (opcode) {
692  case PEO::Add:
693  result = simplifyAdd(operands);
694  break;
695  case PEO::Mul:
696  result = simplifyMul(operands);
697  break;
698  case PEO::And:
699  result = simplifyAnd(operands);
700  break;
701  case PEO::Or:
702  result = simplifyOr(operands);
703  break;
704  case PEO::Xor:
705  result = simplifyXor(operands);
706  break;
707  case PEO::Shl:
708  result = simplifyShl(operands);
709  break;
710  case PEO::ShrU:
711  result = simplifyShrU(operands);
712  break;
713  case PEO::ShrS:
714  result = simplifyShrS(operands);
715  break;
716  case PEO::DivU:
717  result = simplifyDivU(operands);
718  break;
719  case PEO::DivS:
720  result = simplifyDivS(operands);
721  break;
722  case PEO::ModU:
723  result = simplifyModU(operands);
724  break;
725  case PEO::ModS:
726  result = simplifyModS(operands);
727  break;
728  case PEO::CLog2:
729  result = simplifyCLog2(operands);
730  break;
731  case PEO::StrConcat:
732  result = simplifyStrConcat(operands);
733  break;
734  }
735 
736  // If we folded to an operand, return it.
737  if (result)
738  return result;
739 
740  return Base::get(operands[0].getContext(), opcode, operands, type);
741 }

778 // Replaces any ParamDeclRefAttr within a parametric expression with its
779 // corresponding value from the map of provided parameters.
780 static FailureOr<Attribute>
781 replaceDeclRefInExpr(Location loc,
782  const std::map<std::string, Attribute> &parameters,
783  Attribute paramAttr) {
784  if (paramAttr.dyn_cast<IntegerAttr>()) {
785  // Nothing to do, constant value.
786  return paramAttr;
787  }
788  if (auto paramRefAttr = paramAttr.dyn_cast<hw::ParamDeclRefAttr>()) {
789  // Get the value from the provided parameters.
790  auto it = parameters.find(paramRefAttr.getName().str());
791  if (it == parameters.end())
792  return emitError(loc)
793  << "Could not find parameter " << paramRefAttr.getName().str()
794  << " in the provided parameters for the expression!";
795  return it->second;
796  }
797  if (auto paramExprAttr = paramAttr.dyn_cast<hw::ParamExprAttr>()) {
798  // Recurse into all operands of the expression.
799  llvm::SmallVector<Attribute, 4> replacedOperands;
800  for (auto operand : paramExprAttr.getOperands()) {
801  auto res = replaceDeclRefInExpr(loc, parameters, operand);
802  if (failed(res))
803  return {failure()};
804  replacedOperands.push_back(res.getValue());
805  }
806  return {
807  hw::ParamExprAttr::get(paramExprAttr.getOpcode(), replacedOperands)};
808  }
809  llvm_unreachable("Unhandled parametric attribute");
810  return {};
811 }

813 FailureOr<Attribute> hw::evaluateParametricAttr(Location loc,
814  ArrayAttr parameters,
815  Attribute paramAttr) {
816  // Create a map of the provided parameters for faster lookup.
817  std::map<std::string, Attribute> parameterMap;
818  for (auto param : parameters) {
819  auto paramDecl = param.cast<ParamDeclAttr>();
820  parameterMap[paramDecl.getName().str()] = paramDecl.getValue();
821  }
822 
823  // First, replace any ParamDeclRefAttr in the expression with its
824  // corresponding value in 'parameters'.
825  auto paramAttrRes = replaceDeclRefInExpr(loc, parameterMap, paramAttr);
826  if (failed(paramAttrRes))
827  return {failure()};
828  paramAttr = paramAttrRes.getValue();
829 
830  // Then, evaluate the parametric attribute.
831  if (paramAttr.isa<IntegerAttr>() || paramAttr.isa<hw::ParamDeclRefAttr>())
832  return paramAttr;
833  if (auto paramExprAttr = paramAttr.dyn_cast<hw::ParamExprAttr>()) {
834  // Since any ParamDeclRefAttr was replaced within the expression,
835  // we re-evaluate the expression through the existing ParamExprAttr
836  // canonicalizer.
837  return ParamExprAttr::get(paramExprAttr.getOpcode(),
838  paramExprAttr.getOperands());
839  }
840 
841  llvm_unreachable("Unhandled parametric attribute");
842  return Attribute();
843 }

845 FailureOr<Type> hw::evaluateParametricType(Location loc, ArrayAttr parameters,
846  Type type) {
847  return llvm::TypeSwitch<Type, Type>(type)
848  .Case<hw::IntType>([&](hw::IntType t) -> FailureOr<Type> {
849  auto evaluatedWidth =
850  evaluateParametricAttr(loc, parameters, t.getWidth());
851  if (failed(evaluatedWidth))
852  return {failure()};
853 
854  // If the width was evaluated to a constant, return an `IntegerType`
855  if (auto intAttr = evaluatedWidth->dyn_cast<IntegerAttr>())
856  return {IntegerType::get(type.getContext(),
857  intAttr.getValue().getSExtValue())};
858 
859  // Otherwise parameter references are still involved
860  return hw::IntType::get(evaluatedWidth.getValue());
861  })
862  .Case<hw::ArrayType>([&](hw::ArrayType arrayType) -> FailureOr<Type> {
863  auto size =
864  evaluateParametricAttr(loc, parameters, arrayType.getSizeAttr());
865  if (failed(size))
866  return failure();
867  auto elementType =
868  evaluateParametricType(loc, parameters, arrayType.getElementType());
869  if (failed(elementType))
870  return failure();
871 
872  // If the size was evaluated to a constant, use a 64-bit integer
873  // attribute version of it
874  if (auto intAttr = size->dyn_cast<IntegerAttr>())
875  return hw::ArrayType::get(
876  arrayType.getContext(), elementType.getValue(),
877  IntegerAttr::get(IntegerType::get(type.getContext(), 64),
878  intAttr.getValue().getSExtValue()));
879 
880  // Otherwise parameter references are still involved
881  return hw::ArrayType::get(arrayType.getContext(),
882  elementType.getValue(), size.getValue());
883  })
884  .Default([&](auto) { return type; });
885 }
