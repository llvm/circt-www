82 // Registered hook to materialize a single constant operation from a given
83 /// attribute value with the desired resultant type. This method should use
84 /// the provided builder to create the operation without changing the
85 /// insertion position. The generated operation is expected to be constant
86 /// like, i.e. single result, zero operands, non side-effecting, etc. On
87 /// success, this hook should return the value generated to represent the
88 /// constant value. Otherwise, it should return null on failure.
89 Operation *HWDialect::materializeConstant(OpBuilder &builder, Attribute value,
90  Type type, Location loc) {
91  // Integer constants can materialize into hw.constant
92  if (auto intType = type.dyn_cast<IntegerType>())
93  if (auto attrValue = value.dyn_cast<IntegerAttr>())
94  return builder.create<ConstantOp>(loc, type, attrValue);
95 
96  // Parameter expressions materialize into hw.param.value.
97  auto parentOp = builder.getBlock()->getParentOp();
98  auto curModule = dyn_cast<HWModuleOp>(parentOp);
99  if (!curModule)
100  curModule = parentOp->getParentOfType<HWModuleOp>();
101  if (curModule && isValidParameterExpression(value, curModule))
102  return builder.create<ParamValueOp>(loc, type, value);
103 
104  return nullptr;
105 }
