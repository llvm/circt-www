42 /// Recursively print the module hierarchy as serialized as JSON.
43 static void printHierarchy(hw::InstanceOp &inst, SymbolTable &symbolTable,
44  llvm::json::OStream &j) {
45  auto moduleOp = symbolTable.lookup(inst.moduleNameAttr().getValue());
46 
47  j.object([&] {
48  j.attribute("instance_name", inst.instanceName());
49  j.attribute("module_name", hw::getVerilogModuleName(moduleOp));
50  j.attributeArray("instances", [&] {
51  // Only recurse on module ops, not extern or generated ops, whose internal
52  // are opaque.
53  if (auto module = dyn_cast<hw::HWModuleOp>(moduleOp)) {
54  for (auto op : module.getOps<hw::InstanceOp>()) {
55  printHierarchy(op, symbolTable, j);
56  }
57  }
58  });
59  });
60 }

62 /// Return the JSON-serialized module hierarchy for the given module as the top
63 /// of the hierarchy.
64 static void extractHierarchyFromTop(hw::HWModuleOp op, SymbolTable &symbolTable,
65  llvm::raw_ostream &os) {
66  llvm::json::OStream j(os, 2);
67 
68  // As a special case for top-level module, set instance name to module name,
69  // since the top-level module is not instantiated.
70  j.object([&] {
71  j.attribute("instance_name", op.getName());
72  j.attribute("module_name", hw::getVerilogModuleName(op));
73  j.attributeArray("instances", [&] {
74  for (auto op : op.getOps<hw::InstanceOp>())
75  printHierarchy(op, symbolTable, j);
76  });
77  });
78 }

80 /// Find the modules corresponding to the firrtl mainModule and DesignUnderTest,
81 /// and if they exist, emit a verbatim op with the module hierarchy for each.
82 void HWExportModuleHierarchyPass::runOnOperation() {
83  mlir::ModuleOp mlirModule = getOperation();
84  Optional<SymbolTable> symbolTable = None;
85  bool directoryCreated = false;
86 
87  for (auto op : mlirModule.getOps<hw::HWModuleOp>()) {
88  auto attr = op->getAttrOfType<ArrayAttr>("firrtl.moduleHierarchyFile");
89  if (!attr)
90  continue;
91  for (auto file : attr.getAsRange<hw::OutputFileAttr>()) {
92  if (!symbolTable)
93  symbolTable = SymbolTable(mlirModule);
94 
95  if (!directoryCreated) {
96  auto error = llvm::sys::fs::create_directories(directoryName);
97  if (error) {
98  op->emitError("Error creating directory in HWExportModuleHierarchy: ")
99  << error.message();
100  signalPassFailure();
101  return;
102  }
103  directoryCreated = true;
104  }
105  SmallString<128> outputPath(directoryName);
106  appendPossiblyAbsolutePath(outputPath, file.getFilename().getValue());
107  std::string errorMessage;
108  std::unique_ptr<llvm::ToolOutputFile> outputFile(
109  mlir::openOutputFile(outputPath, &errorMessage));
110  if (!outputFile) {
111  op->emitError("Error creating file in HWExportModuleHierarchy: ")
112  << errorMessage;
113  signalPassFailure();
114  return;
115  }
116 
117  extractHierarchyFromTop(op, symbolTable.getValue(), outputFile->os());
118 
119  outputFile->keep();
120  }
121  }
122 
123  markAllAnalysesPreserved();
124 }
