55 /// Try to lower a hw.array_get in module that doesn't support packed arrays.
56 /// This returns a replacement operation if lowering was successful, null
57 /// otherwise.
58 Operation *HWLegalizeModulesPass::tryLoweringArrayGet(hw::ArrayGetOp getOp) {
59  // If the operand is an array_create, then we can lower this into a casez.
60  auto createOp = getOp.input().getDefiningOp<hw::ArrayCreateOp>();
61  if (!createOp)
62  return nullptr;
63 
64  // array_get(idx, array_create(a,b,c,d)) ==> casez(idx).
65  Value index = getOp.index();
66 
67  // Create the wire for the result of the casez in the hw.module.
68  OpBuilder builder(&thisHWModule.getBodyBlock()->front());
69 
70  auto theWire = builder.create<sv::RegOp>(getOp.getLoc(), getOp.getType(),
71  builder.getStringAttr("casez_tmp"));
72  builder.setInsertionPoint(getOp);
73 
74  // A casez is a procedural operation, so if we're in a non-procedural region
75  // we need to inject an always_comb block.
76  if (!getOp->getParentOp()->hasTrait<sv::ProceduralRegion>()) {
77  auto alwaysComb = builder.create<sv::AlwaysCombOp>(createOp.getLoc());
78  builder.setInsertionPointToEnd(alwaysComb.getBodyBlock());
79  }
80 
81  // If we are missing elements in the array (it is non-power of two), then
82  // add a default 'X' value.
83  SmallVector<Value> caseValues(llvm::reverse(createOp.getOperands()));
84  if (1ULL << index.getType().getIntOrFloatBitWidth() !=
85  createOp.getNumOperands()) {
86  caseValues.push_back(
87  builder.create<sv::ConstantXOp>(getOp.getLoc(), getOp.getType()));
88  }
89 
90  APInt caseValue(index.getType().getIntOrFloatBitWidth(), 0);
91  auto *context = builder.getContext();
92 
93  // Create the casez itself.
94  builder.create<sv::CaseOp>(
95  createOp.getLoc(), CaseStmtType::CaseZStmt, index, caseValues.size(),
96  [&](size_t caseIdx) -> std::unique_ptr<sv::CasePattern> {
97  // Use a default pattern for the last value, even if we are complete.
98  // This avoids tools thinking they need to insert a latch due to
99  // potentially incomplete case coverage.
100  bool isDefault = caseIdx == caseValues.size() - 1;
101  Value theValue = caseValues[caseIdx];
102  std::unique_ptr<sv::CasePattern> thePattern;
103 
104  if (isDefault)
105  thePattern = std::make_unique<sv::CaseDefaultPattern>(context);
106  else
107  thePattern = std::make_unique<sv::CaseBitPattern>(caseValue, context);
108  ++caseValue;
109  builder.create<sv::BPAssignOp>(createOp.getLoc(), theWire, theValue);
110  return thePattern;
111  });
112 
113  // Ok, emit the read from the wire to get the value out.
114  builder.setInsertionPoint(getOp);
115  auto readWire = builder.create<sv::ReadInOutOp>(getOp.getLoc(), theWire);
116  getOp.getResult().replaceAllUsesWith(readWire);
117  getOp->erase();
118  return readWire;
119 }

121 void HWLegalizeModulesPass::processPostOrder(Block &body) {
122  if (body.empty())
123  return;
124 
125  // Walk the block bottom-up, processing the region tree inside out.
126  Block::iterator it = std::prev(body.end());
127  while (it != body.end()) {
128  auto &op = *it;
129 
130  // Advance the iterator, using the end iterator as a sentinel that we're at
131  // the top of the block.
132  if (it == body.begin())
133  it = body.end();
134  else
135  --it;
136 
137  if (op.getNumRegions()) {
138  for (auto &region : op.getRegions())
139  for (auto &regionBlock : region.getBlocks())
140  processPostOrder(regionBlock);
141  }
142 
143  if (options.disallowPackedArrays) {
144  // Try idioms for lowering array_get operations.
145  if (auto getOp = dyn_cast<hw::ArrayGetOp>(op))
146  if (auto *replacement = tryLoweringArrayGet(getOp)) {
147  it = Block::iterator(replacement);
148  anythingChanged = true;
149  continue;
150  }
151 
152  // If this is a dead array_create, then we can just delete it. This is
153  // probably left over from get/create lowering.
154  if (isa<hw::ArrayCreateOp>(op) && op.use_empty()) {
155  op.erase();
156  continue;
157  }
158 
159  // Otherwise, if we aren't allowing multi-dimensional arrays, reject the
160  // IR as invalid.
161  // TODO: We should eventually implement a "lower types" like feature in
162  // this pass.
163  for (auto value : op.getResults()) {
164  if (value.getType().isa<hw::ArrayType>()) {
165  op.emitError("unsupported packed array expression");
166  }
167  }
168  }
169  }
170 }

172 void HWLegalizeModulesPass::runOnOperation() {
173  thisHWModule = getOperation();
174 
175  // Parse the lowering options if necessary.
176  auto optionsAttr = LoweringOptions::getAttributeFrom(
177  cast<ModuleOp>(thisHWModule->getParentOp()));
178  if (optionsAttr != lastParsedOptions) {
179  if (optionsAttr)
180  options = LoweringOptions(optionsAttr.getValue(), [&](Twine error) {
181  thisHWModule.emitError(error);
182  });
183  else
184  options = LoweringOptions();
185  lastParsedOptions = optionsAttr;
186  }
187 
188  // Keeps track if anything changed during this pass, used to determine if
189  // the analyses were preserved.
190  anythingChanged = false;
191 
192  // Walk the operations in post-order, transforming any that are interesting.
193  processPostOrder(*thisHWModule.getBodyBlock());
194 
195  // If we did not change anything in the IR mark all analysis as preserved.
196  if (!anythingChanged)
197  markAllAnalysesPreserved();
198 }
