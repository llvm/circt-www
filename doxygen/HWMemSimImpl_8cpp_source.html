63 static Value addPipelineStages(ImplicitLocOpBuilder &b, size_t stages,
64  Value clock, Value data) {
65  if (!stages)
66  return data;
67 
68  while (stages--) {
69  auto reg = b.create<sv::RegOp>(data.getType());
70 
71  // pipeline stage
72  b.create<sv::AlwaysFFOp>(sv::EventControl::AtPosEdge, clock,
73  [&]() { b.create<sv::PAssignOp>(reg, data); });
74  data = b.create<sv::ReadInOutOp>(reg);
75  }
76 
77  return data;
78 }

80 void HWMemSimImplPass::generateMemory(hw::HWModuleOp op, FirMemory mem) {
81  ImplicitLocOpBuilder b(UnknownLoc::get(&getContext()), op.getBody());
82 
83  // Create a register for the memory.
84  auto dataType = b.getIntegerType(mem.dataWidth);
85  Value reg =
86  b.create<sv::RegOp>(hw::UnpackedArrayType::get(dataType, mem.depth),
87  b.getStringAttr("Memory"));
88 
89  SmallVector<Value, 4> outputs;
90 
91  size_t inArg = 0;
92  for (size_t i = 0; i < mem.numReadPorts; ++i) {
93  Value clock = op.body().getArgument(inArg++);
94  Value en = op.body().getArgument(inArg++);
95  Value addr = op.body().getArgument(inArg++);
96  // Add pipeline stages
97  en = addPipelineStages(b, mem.readLatency, clock, en);
98  addr = addPipelineStages(b, mem.readLatency, clock, addr);
99 
100  // Read Logic
101  Value ren =
102  b.create<sv::ReadInOutOp>(b.create<sv::ArrayIndexInOutOp>(reg, addr));
103  Value x = b.create<sv::ConstantXOp>(dataType);
104 
105  Value rdata = b.create<comb::MuxOp>(en, ren, x);
106  outputs.push_back(rdata);
107  }
108 
109  for (size_t i = 0; i < mem.numReadWritePorts; ++i) {
110  auto numStages = std::max(mem.readLatency, mem.writeLatency) - 1;
111  Value clock = op.body().getArgument(inArg++);
112  Value en = op.body().getArgument(inArg++);
113  Value addr = op.body().getArgument(inArg++);
114  Value wmode = op.body().getArgument(inArg++);
115  Value wmask = op.body().getArgument(inArg++);
116  Value wdata = op.body().getArgument(inArg++);
117 
118  // Add pipeline stages
119  en = addPipelineStages(b, numStages, clock, en);
120  addr = addPipelineStages(b, numStages, clock, addr);
121  wmode = addPipelineStages(b, numStages, clock, wmode);
122  wmask = addPipelineStages(b, numStages, clock, wmask);
123  wdata = addPipelineStages(b, numStages, clock, wdata);
124 
125  // wire to store read result
126  auto rWire = b.create<sv::WireOp>(wdata.getType());
127  Value rdata = b.create<sv::ReadInOutOp>(rWire);
128 
129  // RW logic
130  b.create<sv::AlwaysFFOp>(sv::EventControl::AtPosEdge, clock, [&]() {
131  auto slot = b.create<sv::ArrayIndexInOutOp>(reg, addr);
132  auto rcond = b.createOrFold<comb::AndOp>(
133  en, b.createOrFold<comb::ICmpOp>(
134  comb::ICmpPredicate::eq, wmode,
135  b.createOrFold<hw::ConstantOp>(wmode.getType(), 0)));
136  auto wcond = b.createOrFold<comb::AndOp>(
137  en, b.createOrFold<comb::AndOp>(wmask, wmode));
138 
139  b.create<sv::PAssignOp>(rWire, b.create<sv::ConstantXOp>(dataType));
140  b.create<sv::IfOp>(
141  wcond, [&]() { b.create<sv::PAssignOp>(slot, wdata); },
142  [&]() {
143  b.create<sv::IfOp>(rcond, [&]() {
144  b.create<sv::PAssignOp>(rWire, b.create<sv::ReadInOutOp>(slot));
145  });
146  });
147  });
148  outputs.push_back(rdata);
149  }
150 
151  for (size_t i = 0; i < mem.numWritePorts; ++i) {
152  auto numStages = mem.writeLatency - 1;
153  Value clock = op.body().getArgument(inArg++);
154  Value en = op.body().getArgument(inArg++);
155  Value addr = op.body().getArgument(inArg++);
156  Value wmask = op.body().getArgument(inArg++);
157  Value wdata = op.body().getArgument(inArg++);
158  // Add pipeline stages
159  en = addPipelineStages(b, numStages, clock, en);
160  addr = addPipelineStages(b, numStages, clock, addr);
161  wmask = addPipelineStages(b, numStages, clock, wmask);
162  wdata = addPipelineStages(b, numStages, clock, wdata);
163 
164  // Write logic
165  b.create<sv::AlwaysFFOp>(sv::EventControl::AtPosEdge, clock, [&]() {
166  auto wcond = b.createOrFold<comb::AndOp>(en, wmask);
167  b.create<sv::IfOp>(wcond, [&]() {
168  auto slot = b.create<sv::ArrayIndexInOutOp>(reg, addr);
169  b.create<sv::PAssignOp>(slot, wdata);
170  });
171  });
172  }
173 
174  auto outputOp = op.getBodyBlock()->getTerminator();
175  outputOp->setOperands(outputs);
176 }

178 void HWMemSimImplPass::runOnOperation() {
179  auto topModule = getOperation().getBody();
180 
181  SmallVector<hw::HWModuleGeneratedOp> toErase;
182  bool anythingChanged = false;
183 
184  for (auto op : llvm::make_early_inc_range(
185  topModule->getOps<hw::HWModuleGeneratedOp>())) {
186  auto oldModule = cast<hw::HWModuleGeneratedOp>(op);
187  auto gen = oldModule.generatorKind();
188  auto genOp = cast<hw::HWGeneratorSchemaOp>(
189  SymbolTable::lookupSymbolIn(getOperation(), gen));
190 
191  if (genOp.descriptor() == "FIRRTL_Memory") {
192  auto mem = analyzeMemOp(oldModule);
193 
194  OpBuilder builder(oldModule);
195  auto nameAttr = builder.getStringAttr(oldModule.getName());
196  auto newModule = builder.create<hw::HWModuleOp>(
197  oldModule.getLoc(), nameAttr, oldModule.getPorts());
198  generateMemory(newModule, mem);
199  oldModule.erase();
200  anythingChanged = true;
201  }
202  }
203 
204  if (!anythingChanged)
205  markAllAnalysesPreserved();
206 }
