75 /// Check parameter specified by `value` to see if it is valid within the scope
76 /// of the specified module `module`. If not, emit an error at the location of
77 /// `usingOp` and return failure, otherwise return success. If `usingOp` is
78 /// null, then no diagnostic is generated.
79 ///
80 /// If `disallowParamRefs` is true, then parameter references are not allowed.
81 LogicalResult hw::checkParameterInContext(Attribute value, Operation *module,
82  Operation *usingOp,
83  bool disallowParamRefs) {
84  // Literals are always ok. Their types are already known to match
85  // expectations.
86  if (value.isa<IntegerAttr>() || value.isa<FloatAttr>() ||
87  value.isa<StringAttr>() || value.isa<ParamVerbatimAttr>())
88  return success();
89 
90  // Check both subexpressions of an expression.
91  if (auto expr = value.dyn_cast<ParamExprAttr>()) {
92  for (auto op : expr.getOperands())
93  if (failed(
94  checkParameterInContext(op, module, usingOp, disallowParamRefs)))
95  return failure();
96  return success();
97  }
98 
99  // Parameter references need more analysis to make sure they are valid within
100  // this module.
101  if (auto parameterRef = value.dyn_cast<ParamDeclRefAttr>()) {
102  auto nameAttr = parameterRef.getName();
103 
104  // Don't allow references to parameters from the default values of a
105  // parameter list.
106  if (disallowParamRefs) {
107  if (usingOp)
108  usingOp->emitOpError("parameter ")
109  << nameAttr << " cannot be used as a default value for a parameter";
110  return failure();
111  }
112 
113  // Find the corresponding attribute in the module.
114  for (auto param : module->getAttrOfType<ArrayAttr>("parameters")) {
115  auto paramAttr = param.cast<ParamDeclAttr>();
116  if (paramAttr.getName() != nameAttr)
117  continue;
118 
119  // If the types match then the reference is ok.
120  if (paramAttr.getType().getValue() == parameterRef.getType())
121  return success();
122 
123  if (usingOp) {
124  auto diag = usingOp->emitOpError("parameter ")
125  << nameAttr << " used with type " << parameterRef.getType()
126  << "; should have type " << paramAttr.getType().getValue();
127  diag.attachNote(module->getLoc()) << "module declared here";
128  }
129  return failure();
130  }
131 
132  if (usingOp) {
133  auto diag = usingOp->emitOpError("use of unknown parameter ") << nameAttr;
134  diag.attachNote(module->getLoc()) << "module declared here";
135  }
136  return failure();
137  }
138 
139  if (usingOp)
140  usingOp->emitOpError("invalid parameter value ") << value;
141  return failure();
142 }

363 static void buildModule(OpBuilder &builder, OperationState &result,
364  StringAttr name, const ModulePortInfo &ports,
365  ArrayAttr parameters,
366  ArrayRef<NamedAttribute> attributes,
367  StringAttr comment) {
368  using namespace mlir::function_interface_impl;
369 
370  // Add an attribute for the name.
371  result.addAttribute(SymbolTable::getSymbolAttrName(), name);
372 
373  SmallVector<Attribute> argNames, resultNames;
374  SmallVector<Type, 4> argTypes, resultTypes;
375  SmallVector<Attribute> argAttrs, resultAttrs;
376  auto exportPortIdent = StringAttr::get(builder.getContext(), "hw.exportPort");
377 
378  for (auto elt : ports.inputs) {
379  if (elt.direction == PortDirection::INOUT && !elt.type.isa<hw::InOutType>())
380  elt.type = hw::InOutType::get(elt.type);
381  argTypes.push_back(elt.type);
382  argNames.push_back(elt.name);
383  Attribute attr;
384  if (elt.sym && !elt.sym.getValue().empty())
385  attr = builder.getDictionaryAttr(
386  {{exportPortIdent, FlatSymbolRefAttr::get(elt.sym)}});
387  else
388  attr = builder.getDictionaryAttr({});
389  argAttrs.push_back(attr);
390  }
391 
392  for (auto elt : ports.outputs) {
393  resultTypes.push_back(elt.type);
394  resultNames.push_back(elt.name);
395  Attribute attr;
396  if (elt.sym && !elt.sym.getValue().empty())
397  attr = builder.getDictionaryAttr(
398  {{exportPortIdent, FlatSymbolRefAttr::get(elt.sym)}});
399  else
400  attr = builder.getDictionaryAttr({});
401  resultAttrs.push_back(attr);
402  }
403 
404  // Allow clients to pass in null for the parameters list.
405  if (!parameters)
406  parameters = builder.getArrayAttr({});
407 
408  // Record the argument and result types as an attribute.
409  auto type = builder.getFunctionType(argTypes, resultTypes);
410  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
411  result.addAttribute("argNames", builder.getArrayAttr(argNames));
412  result.addAttribute("resultNames", builder.getArrayAttr(resultNames));
413  result.addAttribute(mlir::function_interface_impl::getArgDictAttrName(),
414  builder.getArrayAttr(argAttrs));
415  result.addAttribute(mlir::function_interface_impl::getResultDictAttrName(),
416  builder.getArrayAttr(resultAttrs));
417  result.addAttribute("parameters", parameters);
418  if (!comment)
419  comment = builder.getStringAttr("");
420  result.addAttribute("comment", comment);
421  result.addAttributes(attributes);
422  result.addRegion();
423 }

425 /// Internal implementation of argument/result insertion and removal on modules.
426 static void modifyModuleArgs(
427  MLIRContext *context, ArrayRef<std::pair<unsigned, PortInfo>> insertArgs,
428  ArrayRef<unsigned> removeArgs, ArrayRef<Attribute> oldArgNames,
429  ArrayRef<Type> oldArgTypes, ArrayRef<Attribute> oldArgAttrs,
430  SmallVector<Attribute> &newArgNames, SmallVector<Type> &newArgTypes,
431  SmallVector<Attribute> &newArgAttrs) {
432 
433 #ifndef NDEBUG
434  // Check that the `insertArgs` and `removeArgs` indices are in ascending
435  // order.
436  assert(llvm::is_sorted(insertArgs,
437  [](auto &a, auto &b) { return a.first < b.first; }) &&
438  "insertArgs must be in ascending order");
439  assert(llvm::is_sorted(removeArgs, [](auto &a, auto &b) { return a < b; }) &&
440  "removeArgs must be in ascending order");
441 #endif
442 
443  auto oldArgCount = oldArgTypes.size();
444  auto newArgCount = oldArgCount + insertArgs.size() - removeArgs.size();
445  assert((int)newArgCount >= 0);
446 
447  newArgNames.reserve(newArgCount);
448  newArgTypes.reserve(newArgCount);
449  newArgAttrs.reserve(newArgCount);
450 
451  auto exportPortAttrName = StringAttr::get(context, "hw.exportPort");
452  auto emptyDictAttr = DictionaryAttr::get(context, {});
453 
454  for (unsigned argIdx = 0; argIdx <= oldArgCount; ++argIdx) {
455  // Insert new ports at this position.
456  while (!insertArgs.empty() && insertArgs[0].first == argIdx) {
457  auto port = insertArgs[0].second;
458  if (port.direction == PortDirection::INOUT && !port.type.isa<InOutType>())
459  port.type = InOutType::get(port.type);
460  Attribute attr =
461  (port.sym && !port.sym.getValue().empty())
462  ? DictionaryAttr::get(
463  context,
464  {{exportPortAttrName, FlatSymbolRefAttr::get(port.sym)}})
465  : emptyDictAttr;
466  newArgNames.push_back(port.name);
467  newArgTypes.push_back(port.type);
468  newArgAttrs.push_back(attr);
469  insertArgs = insertArgs.drop_front();
470  }
471  if (argIdx == oldArgCount)
472  break;
473 
474  // Migrate the old port at this position.
475  bool removed = false;
476  while (!removeArgs.empty() && removeArgs[0] == argIdx) {
477  removeArgs = removeArgs.drop_front();
478  removed = true;
479  }
480  if (!removed) {
481  newArgNames.push_back(oldArgNames[argIdx]);
482  newArgTypes.push_back(oldArgTypes[argIdx]);
483  newArgAttrs.push_back(oldArgAttrs.empty() ? emptyDictAttr
484  : oldArgAttrs[argIdx]);
485  }
486  }
487 
488  assert(newArgNames.size() == newArgCount);
489  assert(newArgTypes.size() == newArgCount);
490  assert(newArgAttrs.size() == newArgCount);
491 }

493 /// Insert and remove ports of a module. The insertion and removal indices must
494 /// be in ascending order. The indices refer to the port positions before any
495 /// insertion or removal occurs. Ports inserted at the same index will appear in
496 /// the module in the same order as they were listed in the `insert*` array.
497 ///
498 /// The operation must be any of the module-like operations.
499 void hw::modifyModulePorts(
500  Operation *op, ArrayRef<std::pair<unsigned, PortInfo>> insertInputs,
501  ArrayRef<std::pair<unsigned, PortInfo>> insertOutputs,
502  ArrayRef<unsigned> removeInputs, ArrayRef<unsigned> removeOutputs) {
503  auto moduleOp = cast<mlir::FunctionOpInterface>(op);
504 
505  auto arrayOrEmpty = [](ArrayAttr attr) {
506  return attr ? attr.getValue() : ArrayRef<Attribute>{};
507  };
508 
509  // Dig up the old argument and result data.
510  ArrayRef<Attribute> oldArgNames =
511  moduleOp->getAttrOfType<ArrayAttr>("argNames").getValue();
512  ArrayRef<Type> oldArgTypes = moduleOp.getArgumentTypes();
513  ArrayRef<Attribute> oldArgAttrs =
514  arrayOrEmpty(moduleOp->getAttrOfType<ArrayAttr>(
515  mlir::function_interface_impl::getArgDictAttrName()));
516 
517  ArrayRef<Attribute> oldResultNames =
518  moduleOp->getAttrOfType<ArrayAttr>("resultNames").getValue();
519  ArrayRef<Type> oldResultTypes = moduleOp.getResultTypes();
520  ArrayRef<Attribute> oldResultAttrs =
521  arrayOrEmpty(moduleOp->getAttrOfType<ArrayAttr>(
522  mlir::function_interface_impl::getResultDictAttrName()));
523 
524  // Modify the ports.
525  SmallVector<Attribute> newArgNames, newResultNames;
526  SmallVector<Type> newArgTypes, newResultTypes;
527  SmallVector<Attribute> newArgAttrs, newResultAttrs;
528 
529  modifyModuleArgs(moduleOp.getContext(), insertInputs, removeInputs,
530  oldArgNames, oldArgTypes, oldArgAttrs, newArgNames,
531  newArgTypes, newArgAttrs);
532 
533  modifyModuleArgs(moduleOp.getContext(), insertOutputs, removeOutputs,
534  oldResultNames, oldResultTypes, oldResultAttrs,
535  newResultNames, newResultTypes, newResultAttrs);
536 
537  // Update the module operation types and attributes.
538  moduleOp.setType(
539  FunctionType::get(moduleOp.getContext(), newArgTypes, newResultTypes));
540  moduleOp->setAttr("argNames",
541  ArrayAttr::get(moduleOp.getContext(), newArgNames));
542  moduleOp->setAttr("resultNames",
543  ArrayAttr::get(moduleOp.getContext(), newResultNames));
544  moduleOp->setAttr(mlir::function_interface_impl::getArgDictAttrName(),
545  ArrayAttr::get(moduleOp.getContext(), newArgAttrs));
546  moduleOp->setAttr(mlir::function_interface_impl::getResultDictAttrName(),
547  ArrayAttr::get(moduleOp.getContext(), newResultAttrs));
548 }

764  SmallVector<Attribute> &parameters) {
765

789 static ParseResult parseHWModuleOp(OpAsmParser &parser, OperationState &result,
790  ExternModKind modKind = PlainMod) {
791 
792  using namespace mlir::function_interface_impl;
793 
794  auto loc = parser.getCurrentLocation();
795 
796  SmallVector<OpAsmParser::Argument, 4> entryArgs;
797  SmallVector<DictionaryAttr> resultAttrs;
798  SmallVector<Type, 4> resultTypes;
799  SmallVector<Attribute> parameters;
800  auto &builder = parser.getBuilder();
801 
802  // Parse the visibility attribute.
803  (void)mlir::impl::parseOptionalVisibilityKeyword(parser, result.attributes);
804 
805  // Parse the name as a symbol.
806  StringAttr nameAttr;
807  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
808  result.attributes))
809  return failure();
810 
811  FlatSymbolRefAttr kindAttr;
812  if (modKind == GenMod) {
813  if (parser.parseComma() ||
814  parser.parseAttribute(kindAttr, "generatorKind", result.attributes)) {
815  return failure();
816  }
817  }
818 
819  // Parse the function signature.
820  bool isVariadic = false;
821  SmallVector<Attribute> resultNames;
822  if (parseOptionalParameters(parser, parameters) ||
823  module_like_impl::parseModuleFunctionSignature(
824  parser, entryArgs, isVariadic, resultTypes, resultAttrs,
825  resultNames) ||
826  // If function attributes are present, parse them.
827  parser.parseOptionalAttrDictWithKeyword(result.attributes))
828  return failure();
829 
830  // Record the argument and result types as an attribute. This is necessary
831  // for external modules.
832  SmallVector<Type> argTypes;
833  for (auto &arg : entryArgs)
834  argTypes.push_back(arg.type);
835 
836  auto type = builder.getFunctionType(argTypes, resultTypes);
837  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
838 
839  auto *context = result.getContext();
840 
841  if (hasAttribute("resultNames", result.attributes) ||
842  hasAttribute("parameters", result.attributes)) {
843  parser.emitError(
844  loc, "explicit `resultNames` / `parameters` attributes not allowed");
845  return failure();
846  }
847 
848  // Use the argument and result names if not already specified.
849  SmallVector<Attribute> argNames;
850  if (!entryArgs.empty()) {
851  for (auto &arg : entryArgs)
852  argNames.push_back(
853  module_like_impl::getPortNameAttr(context, arg.ssaName.name));
854  } else if (!argTypes.empty()) {
855  // The parser returns empty names in a special way.
856  argNames.assign(argTypes.size(), StringAttr::get(context, ""));
857  }
858 
859  // An explicit `argNames` attribute overrides the MLIR names. This is how
860  // we represent port names that aren't valid MLIR identifiers. Result and
861  // parameter names are printed quoted when they aren't valid identifiers, so
862  // they don't need this affordance.
863  if (!hasAttribute("argNames", result.attributes))
864  result.addAttribute("argNames", ArrayAttr::get(context, argNames));
865  result.addAttribute("resultNames", ArrayAttr::get(context, resultNames));
866  result.addAttribute("parameters", ArrayAttr::get(context, parameters));
867  if (!hasAttribute("comment", result.attributes))
868  result.addAttribute("comment", StringAttr::get(context, ""));
869 
870  assert(resultAttrs.size() == resultTypes.size());
871 
872  // Add the attributes to the function arguments.
873  addArgAndResultAttrs(builder, result, entryArgs, resultAttrs);
874 
875  // Parse the optional function body.
876  auto *body = result.addRegion();
877  if (modKind == PlainMod) {
878  if (parser.parseRegion(*body, entryArgs))
879  return failure();
880 
881  HWModuleOp::ensureTerminator(*body, parser.getBuilder(), result.location);
882  }
883  return success();
884 }

985 static LogicalResult verifyModuleCommon(Operation *module) {
986  assert(isAnyModule(module) &&
987  "verifier hook should only be called on modules");
988 
989  auto moduleType = getModuleType(module);
990  auto argNames = module->getAttrOfType<ArrayAttr>("argNames");
991  auto resultNames = module->getAttrOfType<ArrayAttr>("resultNames");
992  if (argNames.size() != moduleType.getNumInputs())
993  return module->emitOpError("incorrect number of argument names");
994  if (resultNames.size() != moduleType.getNumResults())
995  return module->emitOpError("incorrect number of result names");
996 
997  SmallPtrSet<Attribute, 4> paramNames;
998 
999  // Check parameter default values are sensible.
1000  for (auto param : module->getAttrOfType<ArrayAttr>("parameters")) {
1001  auto paramAttr = param.cast<ParamDeclAttr>();
1002 
1003  // Check that we don't have any redundant parameter names. These are
1004  // resolved by string name: reuse of the same name would cause ambiguities.
1005  if (!paramNames.insert(paramAttr.getName()).second)
1006  return module->emitOpError("parameter ")
1007  << paramAttr << " has the same name as a previous parameter";
1008 
1009  // Default values are allowed to be missing, check them if present.
1010  auto value = paramAttr.getValue();
1011  if (!value)
1012  continue;
1013 
1014  if (value.getType() != paramAttr.getType().getValue())
1015  return module->emitOpError("parameter ")
1016  << paramAttr << " should have type "
1017  << paramAttr.getType().getValue() << "; has type "
1018  << value.getType();
1019 
1020  // Verify that this is a valid parameter value, disallowing parameter
1021  // references. We could allow parameters to refer to each other in the
1022  // future with lexical ordering if there is a need.
1023  if (failed(checkParameterInContext(value, module, module,
1024  /*disallowParamRefs=*/true)))
1025  return failure();
1026  }
1027  return success();
1028 }

1155  return emitError("symbol reference '")

1159  // module.
1160  // Emit an error message on the instance, with a note indicating which module
1161  // is being referenced.

1163  [&](std::function<void(InFlightDiagnostic & diag)> fn) -> LogicalResult {
1164  auto diag = emitOpError();

1166  diag.attachNote(module->getLoc()) << "module declared here";
1167  return failure();
1168  };

1171  ArrayAttr argNames = argNamesAttr();
1172  ArrayAttr modArgNames = module->getAttrOfType<ArrayAttr>("argNames");
1173

1175  auto numOperands = getOperation()->getNumOperands();
1176  auto expectedOperandTypes = getModuleType(module).getInputs();
1177

1181  << expectedOperandTypes.size() << " but got " << numOperands;

1192  evaluateParametricType(getLoc(), parameters(), expectedOperandTypes[i]);
1193  if (failed(expectedType))

1195  diag << "failed to resolve parametric input of instantiated module";
1196  });

1198  if (operandType != expectedType.getValue()) {
1199  return emitError([&](auto &diag) {
1200  diag << "operand type #" << i << " must be " << expectedType.getValue()
1201  << ", but got " << operandType;
1202  });
1203  }

1209  });
1210  }

1216  ArrayAttr modResultNames = module->getAttrOfType<ArrayAttr>("resultNames");
1217

1221  << expectedResultTypes.size() << " but got " << numResults;
1222  });

1235  });

1244  return emitError([&](auto &diag) {
1245  diag << "input label #" << i << " must be " << modResultNames[i]
1246  << ", but got " << resultNames[i];
1247  });
1248  }
1249 
1250  // Check parameters match up.
1251  ArrayAttr parameters = this->parameters();
1252  ArrayAttr modParameters = module->getAttrOfType<ArrayAttr>("parameters");
1253  auto numParameters = parameters.size();

1255  return emitError([&](auto &diag) {
1256  diag << "expected " << modParameters.size() << " parameters but had "
1257  << numParameters;
1258  });
1259

1261  auto param = parameters[i].cast<ParamDeclAttr>();
1262  auto modParam = modParameters[i].cast<ParamDeclAttr>();
1263 
1264  auto paramName = param.getName();

1266  return emitError([&](auto &diag) {
1267  diag << "parameter #" << i << " should have name " << modParam.getName()
1268  << " but has name " << paramName;
1269  });

1278  // a module's default value if you want the default.

1312  StringAttr sym_nameAttr;

1314  SmallVector<OpAsmParser::UnresolvedOperand, 4> inputsOperands;
1315  SmallVector<Type> inputsTypes;

1348  allResultTypes.push_back({});
1349  return parser.parseColonType(allResultTypes.back());

1357  parser.getCurrentLocation(&inputsOperandsLoc) ||

1360  parser.resolveOperands(inputsOperands, inputsTypes, inputsOperandsLoc,
1361  result.operands) ||
1362  parser.parseArrow() ||
1363  parser.parseCommaSeparatedList(OpAsmParser::Delimiter::Paren,
1364  parseResultPort) ||
1365  parser.parseOptionalAttrDict(result.attributes)) {
1366  return failure();
1367  }

1370  result.addAttribute("resultNames",
1371  parser.getBuilder().getArrayAttr(resultNames));
1372  result.addAttribute("parameters",
1373  parser.getBuilder().getArrayAttr(parameters));
1374  result.addTypes(allResultTypes);
1375  return success();
1376 }

1512 LogicalResult
1513 GlobalRefOp::verifySymbolUses(mlir::SymbolTableCollection &symTables) {
1514  Operation *parent = (*this)->getParentOp();
1515  SymbolTable &symTable = symTables.getSymbolTable(parent);
1516  StringAttr symNameAttr = (*this).sym_nameAttr();
1517  auto hasGlobalRef = [&](Attribute attr) -> bool {
1518  if (!attr)
1519  return false;
1520  for (auto ref : attr.cast<ArrayAttr>().getAsRange<GlobalRefAttr>())
1521  if (ref.getGlblSym().getAttr() == symNameAttr)
1522  return true;
1523  return false;
1524  };
1525  // For all inner refs in the namepath, ensure they have a corresponding
1526  // GlobalRefAttr to this GlobalRefOp.
1527  for (auto innerRef : namepath().getAsRange<hw::InnerRefAttr>()) {
1528  StringAttr modName = innerRef.getModule();
1529  StringAttr innerSym = innerRef.getName();
1530  Operation *mod = symTable.lookup(modName);
1531  if (!mod) {
1532  (*this)->emitOpError("module:'" + modName.str() + "' not found");
1533  return failure();
1534  }
1535  bool glblSymNotFound = true;
1536  bool innerSymOpNotFound = true;
1537  mod->walk([&](Operation *op) -> WalkResult {
1538  StringAttr attr = op->getAttrOfType<StringAttr>("inner_sym");
1539  // If this is one of the ops in the instance path for the GlobalRefOp.
1540  if (attr && attr == innerSym) {
1541  innerSymOpNotFound = false;
1542  // Each op can have an array of GlobalRefAttr, check if this op is one
1543  // of them.
1544  if (hasGlobalRef(op->getAttr(GlobalRefAttr::DialectAttrName))) {
1545  glblSymNotFound = false;
1546  return WalkResult::interrupt();
1547  }
1548  // If cannot find the ref, then its an error.
1549  return failure();
1550  }
1551  return WalkResult::advance();
1552  });
1553  if (glblSymNotFound) {
1554  // TODO: Doesn't yet work for symbls on FIRRTL module ports. Need to
1555  // implement an interface.
1556  if (isa<HWModuleOp, HWModuleExternOp>(mod)) {
1557  if (auto argAttrs = mod->getAttr(
1558  mlir::function_interface_impl::getArgDictAttrName()))
1559  for (auto attr :
1560  argAttrs.cast<ArrayAttr>().getAsRange<DictionaryAttr>())
1561  if (auto symRef = attr.get("hw.exportPort"))
1562  if (symRef.cast<FlatSymbolRefAttr>().getValue() == innerSym)
1563  if (hasGlobalRef(attr.get(GlobalRefAttr::DialectAttrName)))
1564  return success();
1565 
1566  if (auto resAttrs = mod->getAttr(
1567  mlir::function_interface_impl::getResultDictAttrName()))
1568  for (auto attr :
1569  resAttrs.cast<ArrayAttr>().getAsRange<DictionaryAttr>())
1570  if (auto symRef = attr.get("hw.exportPort"))
1571  if (symRef.cast<FlatSymbolRefAttr>().getValue() == innerSym)
1572  if (hasGlobalRef(attr.get(GlobalRefAttr::DialectAttrName)))
1573  return success();
1574  }
1575  }
1576  if (innerSymOpNotFound)
1577  return (*this)->emitOpError("operation:'" + innerSym.str() +
1578  "' in module:'" + modName.str() +
1579  "' could not be found");
1580  if (glblSymNotFound)
1581  return (*this)->emitOpError(
1582  "operation:'" + innerSym.str() + "' in module:'" + modName.str() +
1583  "' does not contain a reference to '" + symNameAttr.str() + "'");
1584  }
1585  return success();
1586 }
