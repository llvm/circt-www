43 /// This holds the name, type, direction of a module's ports
44 struct PortInfo {
45  StringAttr name;
46  PortDirection direction;
47  Type type;
48 
49  /// This is the argument index or the result index depending on the direction.
50  /// "0" for an output means the first output, "0" for a in/inout means the
51  /// first argument.
52  size_t argNum = ~0U;
53 
54  /// The optional symbol for this port.
55  StringAttr sym = {};
56 
57  StringRef getName() const { return name.getValue(); }
58  bool isInput() const { return direction == PortDirection::INPUT; }
59  bool isOutput() const { return direction == PortDirection::OUTPUT; }
60  bool isInOut() const { return direction == PortDirection::INOUT; }
61 
62  /// Return a unique numeric identifier for this port.
63  ssize_t getId() const { return isOutput() ? argNum : (-1 - argNum); };
64 };

66 /// This holds a decoded list of input/inout and output ports for a module or
67 /// instance.
68 struct ModulePortInfo {
69  explicit ModulePortInfo(ArrayRef<PortInfo> inputs, ArrayRef<PortInfo> outputs)
70  : inputs(inputs.begin(), inputs.end()),
71  outputs(outputs.begin(), outputs.end()) {}
72 
73  explicit ModulePortInfo(ArrayRef<PortInfo> mergedPorts) {
74  inputs.reserve(mergedPorts.size());
75  outputs.reserve(mergedPorts.size());
76  for (auto port : mergedPorts) {
77  if (port.isOutput())
78  outputs.push_back(port);
79  else
80  inputs.push_back(port);
81  }
82  }
83 
84  /// This contains a list of the input and inout ports.
85  SmallVector<PortInfo> inputs;
86  /// This is a list of the output ports.
87  SmallVector<PortInfo> outputs;
88 };
