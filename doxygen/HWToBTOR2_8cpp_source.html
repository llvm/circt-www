64  // instructions to reference the given sorts (key: width, value: LID)

66  // Keeps track of {constant, width} -> LID mappings

76  // Keeps track of operation aliases. This is used for wire inlining, as
77  // btor2 does not have the concept of a wire. This means that wires in

106  // Checks if an operation was declared
107  // If so, its lid will be returned
108  // Otherwise a new lid will be assigned to the op
109  size_t getOpLID(Operation *op) {
110  // Look for the original operation declaration
111  // Make sure that wires are considered when looking for an lid
112  Operation *defOp = getOpAlias(op);
113  auto &f = opLIDMap[defOp];
114 
115  // If the op isn't associated to an lid, assign it a new one
116  if (!f)
117  f = lid++;
118  return f;
119  }

121  // Associates the current lid to an operation
122  // The LID is then incremented to maintain uniqueness
123  size_t setOpLID(Operation *op) {
124  size_t oplid = lid++;
125  opLIDMap[op] = oplid;
126  return oplid;
127  }

129  // Checks if an operation was declared
130  // If so, its lid will be returned
131  // Otherwise -1 will be returned
132  size_t getOpLID(Value value) {
133  // Check for an operation alias
134  Operation *defOp = getOpAlias(value.getDefiningOp());
135 
136  if (auto it = opLIDMap.find(defOp); it != opLIDMap.end())
137  return it->second;
138 
139  // Check for special case where op is actually a port
140  // To do so, we start by checking if our operation is a block argument
141  if (BlockArgument barg = dyn_cast<BlockArgument>(value)) {
142  // Extract the block argument index and use that to get the line number
143  size_t argIdx = barg.getArgNumber();
144 
145  // Check that the extracted argument is in range before using it
146  if (auto it = inputLIDs.find(argIdx); it != inputLIDs.end())
147  return it->second;
148  }
149 
150  // Return -1 if no LID was found
151  return noLID;
152  }

470  // Verifies that the sort required for the given operation's btor2 emission
471  // has been generated
472  int64_t requireSort(mlir::Type type) {
473  // Start by figuring out what sort needs to be generated
474  int64_t width = hw::getBitWidth(type);
475 
476  // Sanity check: getBitWidth can technically return -1 it is a type with no
477  // width (like a clock). This shouldn't be allowed as width is required to
478  // generate a sort
479  assert(width != noWidth);
480 
481  // Generate the sort regardles of resulting width (nothing will be added if
482  // the sort already exists)
483  genSort("bitvec", width);
484  return width;
485  }

487  // Generates the transitions required to finalize the register to state
488  // transition system conversion
489  void finalizeRegVisit(Operation *op) {
490  int64_t width;
491  Value next, resetVal;
492 
493  // Extract the operands depending on the register type
494  if (auto reg = dyn_cast<seq::CompRegOp>(op)) {
495  width = hw::getBitWidth(reg.getType());
496  next = reg.getInput();
497  resetVal = reg.getResetValue();
498  } else if (auto reg = dyn_cast<seq::FirRegOp>(op)) {
499  width = hw::getBitWidth(reg.getType());
500  next = reg.getNext();
501  resetVal = reg.getResetValue();
502  } else {
503  op->emitError("Invalid register operation !");
504  return;
505  }
506 
507  genSort("bitvec", width);
508 
509  // Next should already be associated to an LID at this point
510  // As we are going to override it, we need to keep track of the original
511  // instruction
512  size_t nextLID = noLID;
513 
514  // We need to check if the next value is a port to avoid nullptrs
515  // To do so, we start by checking if our operation is a block argument
516  if (BlockArgument barg = dyn_cast<BlockArgument>(next)) {
517  // Extract the block argument index and use that to get the line number
518  size_t argIdx = barg.getArgNumber();
519 
520  // Check that the extracted argument is in range before using it
521  nextLID = inputLIDs[argIdx];
522 
523  } else {
524  nextLID = getOpLID(next);
525  }
526 
527  // Check if the register has a reset
528  if (resetLID != noLID) {
529  size_t resetValLID = noLID;
530 
531  // Check for a reset value, if none exists assume it's zero
532  if (resetVal)
533  resetValLID = getOpLID(resetVal.getDefiningOp());
534  else
535  resetValLID = genZero(width);
536 
537  // Assign a new LID to next
538  setOpLID(next.getDefiningOp());
539 
540  // Sanity check: at this point the next operation should have had it's
541  // btor2 counterpart emitted if not then something terrible must have
542  // happened.
543  assert(nextLID != noLID);
544 
545  // Generate the ite for the register update reset condition
546  // i.e. reg <= reset ? 0 : next
547  genIte(next.getDefiningOp(), resetLID, resetValLID, nextLID, width);
548  } else {
549  // Assign a new LID to next and perform a sanity check
550  setOpLID(next.getDefiningOp());
551  assert(nextLID != noLID);
552  }
553 
554  // Finally generate the next statement
555  genNext(next, op, width);
556  }

561  // Visitor for the inputs of the module.
562  // This will generate additional sorts and input declaration explicitly for
563  // btor2 Note that outputs are ignored in btor2 as they do not contribute to
564  // the final assertions
565  void visit(hw::PortInfo &port) {
566  // Separate the inputs from outputs and generate the first btor2 lines for
567  // input declaration We only consider ports with an explicit bit-width (so
568  // ignore clocks)
569  if (port.isInput() && !port.type.isa<seq::ClockType>()) {
570  // Generate the associated btor declaration for the inputs
571  StringRef iName = port.getName();
572 
573  // Guarantees that a sort will exist for the generation of this port's
574  // translation into btor2
575  int64_t w = requireSort(port.type);
576 
577  // Save lid for later
578  size_t inlid = lid;
579 
580  // Record the defining operation's line ID (the module itself in the case
581  // of ports)
582  inputLIDs[port.argNum] = lid;
583 
584  // We assume that the explicit name is always %reset for reset ports
585  if (iName == "reset")
586  resetLID = lid;
587 
588  // Increment the lid to keep it unique
589  lid++;
590 
591  genInput(inlid, w, iName);
592  }
593  }

804  StringRef regName = reg.getName().value();

872 void ConvertHWToBTOR2Pass::runOnOperation() {
873  // Btor2 does not have the concept of modules or module
874  // hierarchies, so we assume that no nested modules exist at this point.
875  // This greatly simplifies translation.
876  getOperation().walk([&](hw::HWModuleOp module) {
877  // Start by extracting the inputs and generating appropriate instructions
878  for (auto &port : module.getPortList()) {
879  visit(port);
880  }
881 
882  // Previsit all registers in the module in order to avoid dependency cylcles
883  module.walk([&](Operation *op) {
884  TypeSwitch<Operation *, void>(op)
885  .Case<seq::FirRegOp, seq::CompRegOp>([&](auto reg) {
886  visit(reg);
887  handledOps.insert(op);
888  })
889  .Default([&](auto expr) {});
890  });
891 
892  // Visit all of the operations in our module
893  module.walk([&](Operation *op) {
894  // Check: instances are not (yet) supported
895  if (isa<hw::InstanceOp>(op)) {
896  op->emitOpError("not supported in BTOR2 conversion");
897  return;
898  }
899 
900  // Don't process ops that have already been emitted
901  if (handledOps.contains(op))
902  return;
903 
904  // Fill in our worklist
905  worklist.insert({op, op->operand_begin()});
906 
907  // Process the elements in our worklist
908  while (!worklist.empty()) {
909  auto &[op, operandIt] = worklist.back();
910  if (operandIt == op->operand_end()) {
911  // All of the operands have been emitted, it is safe to emit our op
912  dispatchTypeOpVisitor(op);
913 
914  // Record that our op has been emitted
915  handledOps.insert(op);
916  worklist.pop_back();
917  continue;
918  }
919 
920  // Send the operands of our op to the worklist in case they are still
921  // un-emitted
922  Value operand = *(operandIt++);
923  auto *defOp = operand.getDefiningOp();
924 
925  // Make sure that we don't emit the same operand twice
926  if (!defOp || handledOps.contains(defOp))
927  continue;
928 
929  // This is triggered if our operand is already in the worklist and
930  // wasn't handled
931  if (!worklist.insert({defOp, defOp->operand_begin()}).second) {
932  op->emitError("dependency cycle");
933  return;
934  }
935  }
936  });
937 
938  // Iterate through the registers and generate the `next` instructions
939  for (size_t i = 0; i < regOps.size(); ++i) {
940  finalizeRegVisit(regOps[i]);
941  }
942  });
943  // Clear data structures to allow for pass reuse
944  sortToLIDMap.clear();
945  constToLIDMap.clear();
946  opLIDMap.clear();
947  opAliasMap.clear();
948  inputLIDs.clear();
949  regOps.clear();
950  handledOps.clear();
951  worklist.clear();
952 }
