52 void circt::populateHWToSMTTypeConverter(TypeConverter &converter) {
53  // The semantics of the builtin integer at the CIRCT core level is currently
54  // not very well defined. It is used for two-valued, four-valued, and possible
55  // other multi-valued logic. Here, we interpret it as two-valued for now.
56  // From a formal perspective, CIRCT would ideally define its own types for
57  // two-valued, four-valued, nine-valued (etc.) logic each. In MLIR upstream
58  // the integer type also carries poison information (which we don't have in
59  // CIRCT?).
60  converter.addConversion([](IntegerType type) -> std::optional<Type> {
61  if (type.getWidth() <= 0)
62  return std::nullopt;
63  return smt::BitVectorType::get(type.getContext(), type.getWidth());
64  });
65 
66  // Default target materialization to convert from illegal types to legal
67  // types, e.g., at the boundary of an inlined child block.
68  converter.addTargetMaterialization([&](OpBuilder &builder, Type resultType,
69  ValueRange inputs,
70  Location loc) -> std::optional<Value> {
71  return builder
72  .create<mlir::UnrealizedConversionCastOp>(loc, resultType, inputs)
73  ->getResult(0);
74  });
75 
76  // Convert a 'smt.bool'-typed value to a 'smt.bv<N>'-typed value
77  converter.addTargetMaterialization(
78  [&](OpBuilder &builder, smt::BitVectorType resultType, ValueRange inputs,
79  Location loc) -> std::optional<Value> {
80  if (inputs.size() != 1)
81  return std::nullopt;
82 
83  if (!isa<smt::BoolType>(inputs[0].getType()))
84  return std::nullopt;
85 
86  unsigned width = resultType.getWidth();
87  Value constZero = builder.create<smt::BVConstantOp>(loc, 0, width);
88  Value constOne = builder.create<smt::BVConstantOp>(loc, 1, width);
89  return builder.create<smt::IteOp>(loc, inputs[0], constOne, constZero);
90  });
91 
92  // Convert a 'smt.bv<1>'-typed value to a 'smt.bool'-typed value
93  converter.addTargetMaterialization(
94  [&](OpBuilder &builder, smt::BoolType resultType, ValueRange inputs,
95  Location loc) -> std::optional<Value> {
96  if (inputs.size() != 1)
97  return std::nullopt;
98 
99  auto bvType = dyn_cast<smt::BitVectorType>(inputs[0].getType());
100  if (!bvType || bvType.getWidth() != 1)
101  return std::nullopt;
102 
103  Value constOne = builder.create<smt::BVConstantOp>(loc, 1, 1);
104  return builder.create<smt::EqOp>(loc, inputs[0], constOne);
105  });
106 
107  // Default source materialization to convert from illegal types to legal
108  // types, e.g., at the boundary of an inlined child block.
109  converter.addSourceMaterialization([&](OpBuilder &builder, Type resultType,
110  ValueRange inputs,
111  Location loc) -> std::optional<Value> {
112  return builder
113  .create<mlir::UnrealizedConversionCastOp>(loc, resultType, inputs)
114  ->getResult(0);
115  });
116 }

126  target.addLegalDialect<smt::SMTDialect>();
127 
128  RewritePatternSet patterns(&getContext());
129  TypeConverter converter;
130  populateHWToSMTTypeConverter(converter);
131  populateHWToSMTConversionPatterns(converter, patterns);
132 
133  if (failed(mlir::applyPartialConversion(getOperation(), target,
134  std::move(patterns))))
135  return signalPassFailure();
