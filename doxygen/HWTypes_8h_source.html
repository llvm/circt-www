72 template <typename... BaseTy>
73 bool type_isa(Type type) {
74  // First check if the type is the requested type.
75  if (type.isa<BaseTy...>())
76  return true;
77 
78  // Then check if it is a type alias wrapping the requested type.
79  if (auto alias = type.dyn_cast<TypeAliasType>())
80  return alias.getInnerType().isa<BaseTy...>();
81 
82  return false;
83 }
84 
85 // type_isa for a nullable argument.
86 template <typename... BaseTy>
87 bool type_isa_and_nonnull(Type type) { // NOLINT(readability-identifier-naming)
88  if (!type)
89  return false;
90  return type_isa<BaseTy...>(type);
91 }

93 template <typename BaseTy>
94 BaseTy type_cast(Type type) {
95  assert(type_isa<BaseTy>(type) && "type must convert to requested type");
96 
97  // If the type is the requested type, return it.
98  if (type.isa<BaseTy>())
99  return type.cast<BaseTy>();
100 
101  // Otherwise, it must be a type alias wrapping the requested type.
102  return type.cast<TypeAliasType>().getInnerType().cast<BaseTy>();
103 }
104 
105 template <typename BaseTy>
106 BaseTy type_dyn_cast(Type type) {
107  if (!type_isa<BaseTy>(type))
108  return BaseTy();
109 
110  return type_cast<BaseTy>(type);
111 }
