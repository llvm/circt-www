22 /// This helps visit TypeOp nodes.
23 template <typename ConcreteType, typename ResultType = void,
24  typename... ExtraArgs>
25 class TypeOpVisitor {
26 public:
27  ResultType dispatchTypeOpVisitor(Operation *op, ExtraArgs... args) {
28  auto *thisCast = static_cast<ConcreteType *>(this);
29  return TypeSwitch<Operation *, ResultType>(op)
30  .template Case<ConstantOp,
31  // Array operations
32  ArraySliceOp, ArrayCreateOp, ArrayConcatOp, ArrayGetOp,
33  // Struct operations
34  StructCreateOp, StructExtractOp, StructInjectOp,
35  // Cast operation
36  BitcastOp, ParamValueOp,
37  // Enum operations
38  EnumConstantOp>([&](auto expr) -> ResultType {
39  return thisCast->visitTypeOp(expr, args...);
40  })
41  .Default([&](auto expr) -> ResultType {
42  return thisCast->visitInvalidTypeOp(op, args...);
43  });
44  }
45 
46  /// This callback is invoked on any non-expression operations.
47  ResultType visitInvalidTypeOp(Operation *op, ExtraArgs... args) {
48  op->emitOpError("unknown HW combinational node");
49  abort();
50  }
51 
52  /// This callback is invoked on any combinational operations that are not
53  /// handled by the concrete visitor.
54  ResultType visitUnhandledTypeOp(Operation *op, ExtraArgs... args) {
55  return ResultType();
56  }
57 
58 #define HANDLE(OPTYPE, OPKIND) \
59  ResultType visitTypeOp(OPTYPE op, ExtraArgs... args) { \
60  return static_cast<ConcreteType *>(this)->visit##OPKIND##TypeOp(op, \
61  args...); \
62  }
63 
64  HANDLE(ConstantOp, Unhandled);
65  HANDLE(BitcastOp, Unhandled);
66  HANDLE(ParamValueOp, Unhandled);
67  HANDLE(StructCreateOp, Unhandled);
68  HANDLE(StructExtractOp, Unhandled);
69  HANDLE(StructInjectOp, Unhandled);
70  HANDLE(ArraySliceOp, Unhandled);
71  HANDLE(ArrayGetOp, Unhandled);
72  HANDLE(ArrayCreateOp, Unhandled);
73  HANDLE(ArrayConcatOp, Unhandled);
74  HANDLE(EnumConstantOp, Unhandled);
75 #undef HANDLE
76 };

78 /// This helps visit TypeOp nodes.
79 template <typename ConcreteType, typename ResultType = void,
80  typename... ExtraArgs>
81 class StmtVisitor {
82 public:
83  ResultType dispatchStmtVisitor(Operation *op, ExtraArgs... args) {
84  auto *thisCast = static_cast<ConcreteType *>(this);
85  return TypeSwitch<Operation *, ResultType>(op)
86  .template Case<ProbeOp, OutputOp, InstanceOp, TypeScopeOp, TypedeclOp>(
87  [&](auto expr) -> ResultType {
88  return thisCast->visitStmt(expr, args...);
89  })
90  .Default([&](auto expr) -> ResultType {
91  return thisCast->visitInvalidStmt(op, args...);
92  });
93  }
94 
95  /// This callback is invoked on any non-expression operations.
96  ResultType visitInvalidStmt(Operation *op, ExtraArgs... args) {
97  op->emitOpError("unknown hw statement");
98  abort();
99  }
100 
101  /// This callback is invoked on any combinational operations that are not
102  /// handled by the concrete visitor.
103  ResultType visitUnhandledTypeOp(Operation *op, ExtraArgs... args) {
104  return ResultType();
105  }
106 
107  /// This fallback is invoked on any binary node that isn't explicitly handled.
108  /// The default implementation delegates to the 'unhandled' fallback.
109  ResultType visitBinaryTypeOp(Operation *op, ExtraArgs... args) {
110  return static_cast<ConcreteType *>(this)->visitUnhandledTypeOp(op, args...);
111  }
112 
113  ResultType visitUnaryTypeOp(Operation *op, ExtraArgs... args) {
114  return static_cast<ConcreteType *>(this)->visitUnhandledTypeOp(op, args...);
115  }
116 
117 #define HANDLE(OPTYPE, OPKIND) \
118  ResultType visitStmt(OPTYPE op, ExtraArgs... args) { \
119  return static_cast<ConcreteType *>(this)->visit##OPKIND##Stmt(op, \
120  args...); \
121  }
122 
123  // Basic nodes.
124  HANDLE(ProbeOp, Unhandled);
125  HANDLE(OutputOp, Unhandled);
126  HANDLE(InstanceOp, Unhandled);
127  HANDLE(TypeScopeOp, Unhandled);
128  HANDLE(TypedeclOp, Unhandled);
129 #undef HANDLE
130 };
