77 MlirType hwStructTypeGet(MlirContext ctx, intptr_t numElements,
78  HWStructFieldInfo const *elements) {
79  SmallVector<StructType::FieldInfo> fieldInfos;
80  fieldInfos.reserve(numElements);
81  for (intptr_t i = 0; i < numElements; ++i) {
82  fieldInfos.push_back(StructType::FieldInfo{
83  unwrap(elements[i].name).cast<StringAttr>(), unwrap(elements[i].type)});
84  }
85  return wrap(StructType::get(unwrap(ctx), fieldInfos));
86 }
