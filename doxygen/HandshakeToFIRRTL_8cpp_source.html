55 /// Get the corresponding FIRRTL type given the built-in data type. Current
56 /// supported data types are integer (signed, unsigned, and signless), index,
57 /// and none.
58 static FIRRTLType getFIRRTLType(Type type) {
59  MLIRContext *context = type.getContext();
60  return TypeSwitch<Type, FIRRTLType>(type)
61  .Case<IntegerType>([&](IntegerType integerType) -> FIRRTLType {
62  unsigned width = integerType.getWidth();
63 
64  switch (integerType.getSignedness()) {
65  case IntegerType::Signed:
66  return SIntType::get(context, width);
67  case IntegerType::Unsigned:
68  return UIntType::get(context, width);
69  // ISSUE: How to handle signless integers? Should we use the
70  // AsSIntPrimOp or AsUIntPrimOp to convert?
71  case IntegerType::Signless:
72  return UIntType::get(context, width);
73  }
74  llvm_unreachable("invalid IntegerType");
75  })
76  .Case<IndexType>([&](IndexType indexType) -> FIRRTLType {
77  // Currently we consider index type as 64-bits unsigned integer.
78  unsigned width = indexType.kInternalStorageBitWidth;
79  return UIntType::get(context, width);
80  })
81  .Case<TupleType>([&](TupleType tupleType) -> FIRRTLType {
82  using BundleElement = BundleType::BundleElement;
83  llvm::SmallVector<BundleElement> elements;
84  for (auto it : llvm::enumerate(tupleType.getTypes()))
85  elements.emplace_back(BundleElement(
86  StringAttr::get(context, llvm::formatv("field{0}", it.index())),
87  false, getFIRRTLType(it.value())));
88  return BundleType::get(elements, context);
89  })
90  .Default([&](Type) { return FIRRTLType(); });
91 }

105 /// Return a FIRRTL bundle type (with data, valid, and ready subfields) given a
106 /// standard data type.
107 static FIRRTLType getBundleType(Type type) {
108  // If the input is already converted to a bundle type elsewhere, itself will
109  // be returned after cast.
110  if (auto firrtlType = type.dyn_cast<FIRRTLType>())
111  return firrtlType;
112 
113  MLIRContext *context = type.getContext();
114  using BundleElement = BundleType::BundleElement;
115  llvm::SmallVector<BundleElement, 3> elements;
116 
117  // Add valid and ready subfield to the bundle.
118  auto validId = StringAttr::get(context, "valid");
119  auto readyId = StringAttr::get(context, "ready");
120  auto signalType = UIntType::get(context, 1);
121  elements.push_back(BundleElement(validId, false, signalType));
122  elements.push_back(BundleElement(readyId, true, signalType));
123 
124  // Add data subfield to the bundle if dataType is not a null.
125  auto dataType = getFIRRTLType(type);
126  if (dataType) {
127  auto dataId = StringAttr::get(context, "data");
128  elements.push_back(BundleElement(dataId, false, dataType));
129  }
130 
131  auto bundleType = BundleType::get(elements, context);
132  return bundleType;
133 }

180 /// Returns a vector of PortInfo's which defines the FIRRTL interface of the
181 /// to-be-converted op.
182 llvm::SmallVector<PortInfo>
183 getPortInfoForOp(ConversionPatternRewriter &rewriter, Operation *op) {
184  llvm::SmallVector<PortInfo> ports;
185  auto loc = op->getLoc();
186  bool hasClock = op->hasTrait<mlir::OpTrait::HasClock>();
187  PortNameGenerator portNames(op);
188 
189  // Add all inputs of oldOp.
190  for (auto portType : llvm::enumerate(op->getOperandTypes())) {
191  auto bundlePortType = getBundleType(portType.value());
192 
193  if (!bundlePortType)
194  op->emitError("Unsupported data type. Supported data types: integer "
195  "(signed, unsigned, signless), index, none.");
196 
197  ports.push_back({portNames.inputName(portType.index()), bundlePortType,
198  Direction::In, StringAttr{}, loc});
199  }
200 
201  // Add all outputs of oldOp.
202  for (auto portType : llvm::enumerate(op->getResultTypes())) {
203  auto bundlePortType = getBundleType(portType.value());
204 
205  if (!bundlePortType)
206  op->emitError("Unsupported data type. Supported data types: integer "
207  "(signed, unsigned, signless), index, none.");
208 
209  ports.push_back({portNames.outputName(portType.index()), bundlePortType,
210  Direction::Out, StringAttr{}, loc});
211  }
212 
213  // Add clock and reset signals.
214  if (hasClock) {
215  ports.push_back({rewriter.getStringAttr("clock"),
216  rewriter.getType<ClockType>(), Direction::In, StringAttr{},
217  loc});
218  ports.push_back({rewriter.getStringAttr("reset"),
219  rewriter.getType<UIntType>(1), Direction::In, StringAttr{},
220  loc});
221  }
222 
223  return ports;
224 }

350 /// Get type name. Currently we only support integer or index types.
351 /// The emitted type aligns with the getFIRRTLType() method. Thus all integers
352 /// other than signed integers will be emitted as unsigned.
353 static std::string getTypeName(Location loc, Type type) {
354  std::string typeName;
355  // Builtin types
356  if (type.isIntOrIndex()) {
357  if (auto indexType = type.dyn_cast<IndexType>())
358  typeName += "_ui" + std::to_string(indexType.kInternalStorageBitWidth);
359  else if (type.isSignedInteger())
360  typeName += "_si" + std::to_string(type.getIntOrFloatBitWidth());
361  else
362  typeName += "_ui" + std::to_string(type.getIntOrFloatBitWidth());
363  } else if (auto tupleType = type.dyn_cast<TupleType>()) {
364  typeName += "_tuple";
365  for (auto elementType : tupleType.getTypes())
366  typeName += getTypeName(loc, elementType);
367  }
368  // FIRRTL types
369  else if (type.isa<SIntType, UIntType>()) {
370  if (auto sintType = type.dyn_cast<SIntType>(); sintType)
371  typeName += "_si" + std::to_string(sintType.getWidthOrSentinel());
372  else {
373  auto uintType = type.cast<UIntType>();
374  typeName += "_ui" + std::to_string(uintType.getWidthOrSentinel());
375  }
376  } else if (auto bundleType = type.dyn_cast<BundleType>()) {
377  typeName += "_tuple";
378  for (auto element : bundleType.getElements())
379  typeName += getTypeName(loc, element.type);
380  } else
381  emitError(loc) << "unsupported data type '" << type << "'";
382 
383  return typeName;
384 }

397 /// Construct a name for creating FIRRTL sub-module.
398 static std::string getSubModuleName(Operation *oldOp) {
399  if (auto instanceOp = dyn_cast<handshake::InstanceOp>(oldOp); instanceOp)
400  return instanceOp.getModule().str();
401 
402  std::string subModuleName = getBareSubModuleName(oldOp);
403 
404  // Add value of the constant operation.
405  if (auto constOp = dyn_cast<handshake::ConstantOp>(oldOp)) {
406  if (auto intAttr = constOp.getValue().dyn_cast<IntegerAttr>()) {
407  auto intType = intAttr.getType();
408 
409  if (intType.isSignedInteger())
410  subModuleName += "_c" + std::to_string(intAttr.getSInt());
411  else if (intType.isUnsignedInteger())
412  subModuleName += "_c" + std::to_string(intAttr.getUInt());
413  else
414  subModuleName += "_c" + std::to_string((uint64_t)intAttr.getInt());
415  } else
416  oldOp->emitError("unsupported constant type");
417  }
418 
419  // Add discriminating in- and output types.
420  auto [inTypes, outTypes] = getHandshakeDiscriminatingTypes(oldOp);
421  if (!inTypes.empty())
422  subModuleName += "_in";
423  for (auto inType : inTypes)
424  subModuleName += getTypeName(oldOp->getLoc(), inType);
425 
426  if (!outTypes.empty())
427  subModuleName += "_out";
428  for (auto outType : outTypes)
429  subModuleName += getTypeName(oldOp->getLoc(), outType);
430 
431  // Add memory ID.
432  if (auto memOp = dyn_cast<handshake::MemoryOp>(oldOp))
433  subModuleName += "_id" + std::to_string(memOp.id());
434 
435  // Add compare kind.
436  if (auto comOp = dyn_cast<mlir::arith::CmpIOp>(oldOp))
437  subModuleName += "_" + stringifyEnum(comOp.getPredicate()).str();
438 
439  // Add buffer information.
440  if (auto bufferOp = dyn_cast<handshake::BufferOp>(oldOp)) {
441  subModuleName += "_" + std::to_string(bufferOp.getNumSlots()) + "slots";
442  if (bufferOp.isSequential())
443  subModuleName += "_seq";
444  else
445  subModuleName += "_fifo";
446 
447  if (bufferOp.initValues().hasValue()) {
448  subModuleName += "_init";
449  ArrayAttr values = bufferOp.initValues().getValue();
450  for (const Attribute e : values) {
451  assert(e.isa<IntegerAttr>());
452  subModuleName +=
453  "_" + std::to_string(e.dyn_cast<IntegerAttr>().getInt());
454  }
455  }
456  }
457 
458  // Add control information.
459  if (auto ctrlInterface = dyn_cast<handshake::ControlInterface>(oldOp);
460  ctrlInterface && ctrlInterface.isControl()) {
461  // Add some additional discriminating info for non-typed operations.
462  subModuleName += "_" + std::to_string(oldOp->getNumOperands()) + "ins_" +
463  std::to_string(oldOp->getNumResults()) + "outs";
464  subModuleName += "_ctrl";
465  } else {
466  assert((!inTypes.empty() || !outTypes.empty()) &&
467  "Non-control operators must provide discriminating type info");
468  }
469 
470  return subModuleName;
471 }

478 /// Construct a tree of 1-bit muxes to multiplex arbitrary numbers of signals
479 /// using a binary-encoded select value.
480 static Value createMuxTree(ArrayRef<Value> inputs, Value select,
481  Location insertLoc, OpBuilder &builder) {
482  Type muxDataType = inputs[0].getType();
483  auto createBits = [&](Value select, size_t idx) {
484  return builder.create<BitsPrimOp>(insertLoc, select, idx, idx);
485  };
486 
487  std::function<Value(ArrayRef<Value>)> buildTreeRec =
488  [&](ArrayRef<Value> operands) {
489  assert(!operands.empty());
490  Value retVal;
491  // Base case
492  if (operands.size() == 1)
493  retVal = operands.front();
494  else {
495  // Mux case. In each layer we take a look at the significant bit wrt.
496  // the # of operands provided, and split the operands at the log2
497  // boundary. By doing so, every subsequent layer can the ignore the
498  // MSBs considered by its preceding layer.
499  unsigned selectBit = llvm::Log2_64_Ceil(operands.size()) - 1;
500  // Split the operands at the selected index
501  unsigned splitIdx = 1 << selectBit;
502 
503  auto [front, back] = splitArrayRef(operands, splitIdx);
504  auto lowerTree = buildTreeRec(front);
505  auto upperTree = buildTreeRec(back);
506  auto layerSelect = createBits(select, selectBit);
507  retVal = builder
508  .create<MuxPrimOp>(insertLoc, muxDataType, layerSelect,
509  upperTree, lowerTree)
510  .result();
511  }
512  return retVal;
513  };
514 
515  return buildTreeRec(inputs);
516 }

518 /// Construct a tree of 1-bit muxes to multiplex arbitrary numbers of signals
519 /// using a one-hot select value. Assumes select has a UIntType.
520 static Value createOneHotMuxTree(ArrayRef<Value> inputs, Value select,
521  Location insertLoc,
522  ConversionPatternRewriter &rewriter) {
523  // Confirm the select input can be a one-hot encoding for the inputs.
524  int32_t numInputs = inputs.size();
525  assert(numInputs == select.getType().cast<UIntType>().getWidthOrSentinel() &&
526  "one-hot select can't mux inputs");
527 
528  // Start the mux tree with zero value.
529  auto inputType = inputs[0].getType().cast<FIRRTLType>();
530  auto inputWidth = inputType.getBitWidthOrSentinel();
531  auto muxValue =
532  createConstantOp(inputType, APInt(inputWidth, 0), insertLoc, rewriter);
533 
534  // Iteratively chain together muxes from the high bit to the low bit.
535  for (size_t i = numInputs; i > 0; --i) {
536  size_t inputIndex = i - 1;
537 
538  Value input = inputs[inputIndex];
539 
540  Value selectBit =
541  rewriter.create<BitsPrimOp>(insertLoc, select, inputIndex, inputIndex);
542 
543  muxValue = rewriter.create<MuxPrimOp>(insertLoc, input.getType(), selectBit,
544  input, muxValue);
545  }
546 
547  return muxValue;
548 }

569 /// Construct an arbiter based on a simple priority-encoding scheme. In addition
570 /// to returning the arbiter result, the index for each input is added to a
571 /// mapping for other lowerings to make use of.
572 static Value createPriorityArbiter(ArrayRef<Value> inputs, Value defaultValue,
573  DenseMap<size_t, Value> &indexMapping,
574  Location insertLoc,
575  ConversionPatternRewriter &rewriter) {
576  auto numInputs = inputs.size();
577  auto indexType = UIntType::get(rewriter.getContext(), numInputs);
578  auto priorityArb = defaultValue;
579 
580  for (size_t i = numInputs; i > 0; --i) {
581  size_t inputIndex = i - 1;
582  size_t oneHotIndex = 1 << inputIndex;
583 
584  auto constIndex = createConstantOp(indexType, APInt(numInputs, oneHotIndex),
585  insertLoc, rewriter);

631 /// Currently we are not considering clock and registers, thus the generated
632 /// circuit is pure combinational logic. If graph cycle exists, at least one
633 /// buffer is required to be inserted for breaking the cycle, which will be
634 /// supported in the next patch.
635 static FModuleOp createTopModuleOp(handshake::FuncOp funcOp, unsigned numClocks,
636  ConversionPatternRewriter &rewriter,
637  bool setFlattenAttr) {
638  llvm::SmallVector<PortInfo, 8> ports;
639 
640  // Add all inputs of funcOp.
641  for (auto &arg : llvm::enumerate(funcOp.getArguments())) {
642  auto portName = funcOp.getArgName(arg.index());
643  FIRRTLType bundlePortType;
644  if (arg.value().getType().isa<MemRefType>())
645  bundlePortType =
646  getMemrefBundleType(rewriter, arg.value(), /*flip=*/true);
647  else
648  bundlePortType = getBundleType(arg.value().getType());
649 
650  if (!bundlePortType)
651  funcOp.emitError("Unsupported data type. Supported data types: integer "
652  "(signed, unsigned, signless), index, none.");
653 
654  ports.push_back({portName, bundlePortType, Direction::In, StringAttr{},
655  arg.value().getLoc()});
656  }
657 
658  auto funcLoc = funcOp.getLoc();
659 
660  // Add all outputs of funcOp.
661  for (auto portType : llvm::enumerate(funcOp.getResultTypes())) {
662  auto portName = funcOp.getResName(portType.index());
663  auto bundlePortType = getBundleType(portType.value());
664 
665  if (!bundlePortType)
666  funcOp.emitError("Unsupported data type. Supported data types: integer "
667  "(signed, unsigned, signless), index, none.");
668 
669  ports.push_back(
670  {portName, bundlePortType, Direction::Out, StringAttr{}, funcLoc});
671  }
672 
673  // Add clock and reset signals.
674  if (numClocks == 1) {
675  ports.push_back({rewriter.getStringAttr("clock"),
676  rewriter.getType<ClockType>(), Direction::In, StringAttr{},
677  funcLoc});
678  ports.push_back({rewriter.getStringAttr("reset"),
679  rewriter.getType<UIntType>(1), Direction::In, StringAttr{},
680  funcLoc});
681  } else if (numClocks > 1) {
682  for (unsigned i = 0; i < numClocks; ++i) {
683  auto clockName = "clock" + std::to_string(i);
684  auto resetName = "reset" + std::to_string(i);
685  ports.push_back({rewriter.getStringAttr(clockName),
686  rewriter.getType<ClockType>(), Direction::In,
687  StringAttr{}, funcLoc});
688  ports.push_back({rewriter.getStringAttr(resetName),
689  rewriter.getType<UIntType>(1), Direction::In,
690  StringAttr{}, funcLoc});
691  }
692  }
693 
694  // Create a FIRRTL module and inline the funcOp into the FIRRTL module.
695  auto topModuleOp = rewriter.create<FModuleOp>(
696  funcOp.getLoc(), rewriter.getStringAttr(funcOp.getName()), ports);
697 
698  if (setFlattenAttr) {
699  topModuleOp->setAttr(
700  "annotations",
701  rewriter.getArrayAttr(rewriter.getDictionaryAttr(
702  llvm::SmallVector<NamedAttribute>{rewriter.getNamedAttr(
703  "class", rewriter.getStringAttr(
704  "firrtl.transforms.FlattenAnnotation"))})));
705  }
706 
707  rewriter.inlineRegionBefore(funcOp.body(), topModuleOp.body(),
708  topModuleOp.body().end());
709 
710  // In the following section, we manually merge the two regions and manually
711  // replace arguments. This is an alternative to using rewriter.mergeBlocks; we
712  // do this to ensure that argument SSA values are replaced instantly, instead
713  // of late, as would be the case for mergeBlocks.
714 
715  // Merge the second block (inlined from funcOp) of the top-module into the
716  // entry block.
717  auto &blockIterator = topModuleOp.body().getBlocks();
718  Block *entryBlock = &blockIterator.front();
719  Block *secondBlock = &*std::next(blockIterator.begin());
720 
721  // Replace uses of each argument of the second block with the corresponding
722  // argument of the entry block.
723  for (auto &oldArg : enumerate(secondBlock->getArguments()))
724  oldArg.value().replaceAllUsesWith(entryBlock->getArgument(oldArg.index()));
725 
726  // Move all operations of the second block to the entry block.
727  entryBlock->getOperations().splice(entryBlock->end(),
728  secondBlock->getOperations());
729  rewriter.eraseBlock(secondBlock);
730  return topModuleOp;
731 }

961 template <typename OpType, bool fstOpSigned, bool sndOpSigned>
962 void StdExprBuilder::buildBinaryLogic() {
963  ValueVector arg0Subfield = portList[0];
964  ValueVector arg1Subfield = portList[1];
965  ValueVector resultSubfields = portList[2];
966 
967  Value arg0Valid = arg0Subfield[0];
968  Value arg0Ready = arg0Subfield[1];
969  Value arg0Data = arg0Subfield[2];
970  Value arg1Valid = arg1Subfield[0];
971  Value arg1Ready = arg1Subfield[1];
972  Value arg1Data = arg1Subfield[2];
973  Value resultValid = resultSubfields[0];
974  Value resultReady = resultSubfields[1];
975  Value resultData = resultSubfields[2];
976 
977  if (fstOpSigned)
978  arg0Data = rewriter.create<AsSIntPrimOp>(insertLoc, arg0Data);
979 
980  if (sndOpSigned)
981  arg1Data = rewriter.create<AsSIntPrimOp>(insertLoc, arg1Data);
982 
983  // Carry out the binary operation.
984  Value resultDataOp = rewriter.create<OpType>(insertLoc, arg0Data, arg1Data);
985  auto resultTy = resultDataOp.getType().cast<FIRRTLType>();
986 
987  // Truncate the result type down if needed.
988  auto resultWidth = resultData.getType()
989  .cast<FIRRTLType>()
990  .getPassiveType()
991  .getBitWidthOrSentinel();
992 
993  if (resultWidth < resultTy.getBitWidthOrSentinel()) {
994  resultDataOp = rewriter.create<BitsPrimOp>(insertLoc, resultDataOp,
995  resultWidth - 1, 0);
996  } else if (fstOpSigned) {
997  // BitsPrimOp already casts to correct type, thus only do this when no
998  // BitsPrimOp was created
999  resultDataOp = rewriter.create<AsUIntPrimOp>(insertLoc, resultDataOp);
1000  }
1001 
1002  rewriter.create<ConnectOp>(insertLoc, resultData, resultDataOp);
1003 
1004  // Generate valid signal.
1005  auto resultValidOp = rewriter.create<AndPrimOp>(
1006  insertLoc, arg0Valid.getType(), arg0Valid, arg1Valid);
1007  rewriter.create<ConnectOp>(insertLoc, resultValid, resultValidOp);
1008 
1009  // Generate ready signals.
1010  auto argReadyOp = rewriter.create<AndPrimOp>(insertLoc, resultReady.getType(),
1011  resultReady, resultValidOp);
1012  rewriter.create<ConnectOp>(insertLoc, arg0Ready, argReadyOp);
1013  rewriter.create<ConnectOp>(insertLoc, arg1Ready, argReadyOp);
1014 }

1729  auto bitType = UIntType::get(rewriter.getContext(), 1);
1730  auto falseConst = createConstantOp(bitType, APInt(1, 0), insertLoc, rewriter);

1733  SmallVector<Value, 4> doneWires;
1734  for (unsigned i = 0; i < resultNum; ++i) {
1735  auto doneWire =
1736  rewriter.create<WireOp>(insertLoc, bitType, "done" + std::to_string(i));
1737  doneWires.push_back(doneWire);

1739

1741  // result ports are handshaked (done), the argument port is ready to accept
1742  // the next token.
1743  Value allDoneWire = rewriter.create<WireOp>(insertLoc, bitType, "allDone");
1744  buildReductionTree<AndPrimOp>(doneWires, allDoneWire);
1745

1747  rewriter.create<ConnectOp>(insertLoc, argReady, allDoneWire);

1750  auto notAllDoneWire =
1751  rewriter.create<WireOp>(insertLoc, bitType, "notAllDone");

1753  insertLoc, notAllDoneWire,
1754  rewriter.create<NotPrimOp>(insertLoc, bitType, allDoneWire));

1757  unsigned idx = 0;
1758  for (auto doneWire : doneWires) {
1759  if (outputs[idx] == nullptr)
1760  return false;
1761

1763  auto resultSubfields = *outputs[idx];
1764  Value resultValid = resultSubfields[0];
1765  Value resultReady = resultSubfields[1];
1766

1768  // port and connect it with the argument data.
1769  if (!isControl) {
1770  Value argData = argSubfields[2];

1772  rewriter.create<ConnectOp>(insertLoc, resultData, argData);
1773  }

1777  rewriter.create<RegResetOp>(insertLoc, bitType, clock, reset,
1778  falseConst, "emtd" + std::to_string(idx));
1779

1781  // notallDone, the emtdReg will be set to the value of doneWire. Otherwise,
1782  // all emtdRegs will be cleared to zero.
1783  auto emtd = rewriter.create<AndPrimOp>(insertLoc, bitType, doneWire,
1784  notAllDoneWire);
1785  rewriter.create<ConnectOp>(insertLoc, emtdReg, emtd);
1786 
1787  // Create a notEmtdWire for later use.
1788  auto notEmtdWire = rewriter.create<WireOp>(insertLoc, bitType,
1789  "notEmtd" + std::to_string(idx));
1790  rewriter.create<ConnectOp>(
1791  insertLoc, notEmtdWire,
1792  rewriter.create<NotPrimOp>(insertLoc, bitType, emtdReg));
1793

1795  // AndPrimOp is each result can only be emitted once.
1796  auto valid =
1797  rewriter.create<AndPrimOp>(insertLoc, bitType, notEmtdWire, argValid);
1798  rewriter.create<ConnectOp>(insertLoc, resultValid, valid);

1801  // the current clock cycle.

1803  insertLoc, bitType, "validReady" + std::to_string(idx));
1804  rewriter.create<ConnectOp>(
1805  insertLoc, validReadyWire,
1806  rewriter.create<AndPrimOp>(insertLoc, bitType, resultReady, valid));

1810  // handshake in a previous clock cycle.
1811  rewriter.create<ConnectOp>(
1812  insertLoc, doneWire,
1813  rewriter.create<OrPrimOp>(insertLoc, bitType, validReadyWire, emtdReg));
1814 
1815  // All done, move to the next result port.
1816  ++idx;
1817  }
1818  return true;

2972 /// Process of lowering:
2973 ///
2974 /// 0) Create and go into a new FIRRTL circuit;
2975 /// 1) Create and go into a new FIRRTL top-module;
2976 /// 2) Inline Handshake FuncOp region into the FIRRTL top-module;
2977 /// 3) Traverse and convert each Standard or Handshake operation:
2978 /// i) Check if an identical sub-module exists. If so, skip to vi);
2979 /// ii) Create and go into a new FIRRTL sub-module;
2980 /// iii) Extract data (if applied), valid, and ready subfield from each port
2981 /// of the sub-module;
2982 /// iv) Build combinational logic;
2983 /// v) Exit the sub-module and go back to the top-module;
2984 /// vi) Create an new instance for the sub-module;
2985 /// vii) Connect the instance with its predecessors and successors;
2986 /// 4) Erase the Handshake FuncOp.
2987 ///
2988 /// createTopModuleOp(): 1) and 2)
2989 /// checkSubModuleOp(): 3.i)
2990 /// createSubModuleOp(): 3.ii)
2991 /// extractSubfields(): 3.iii)
2992 /// build*Logic(): 3.iv)
