72 /// Returns a set of types which may uniquely identify the provided op. Return
73 /// value is <inputTypes, outputTypes>.
74 using DiscriminatingTypes = std::pair<SmallVector<Type>, SmallVector<Type>>;
75 static DiscriminatingTypes getHandshakeDiscriminatingTypes(Operation *op) {
76  return TypeSwitch<Operation *, DiscriminatingTypes>(op)
77  .Case<MemoryOp>([&](auto memOp) {
78  return DiscriminatingTypes{{}, {memOp.memRefType().getElementType()}};
79  })
80  .Default([&](auto) {
81  // By default, all in- and output types which is not a control type
82  // (NoneType) are discriminating types.
83  std::vector<Type> inTypes, outTypes;
84  llvm::transform(op->getOperands(), std::back_inserter(inTypes),
85  getOperandDataType);
86  llvm::transform(op->getResults(), std::back_inserter(outTypes),
87  getOperandDataType);
88  return DiscriminatingTypes{filterNoneTypes(inTypes),
89  filterNoneTypes(outTypes)};
90  });
91 }

93 // Wraps a type into an ESI ChannelPort type. The inner type is converted to
94 // ensure comprehensability by the RTL dialects.
95 static Type esiWrapper(Type t) {
96  // Translate none- and index types to something HW understands.
97  if (t.isa<NoneType>())
98  t = IntegerType::get(t.getContext(), 0);
99  else if (t.isa<IndexType>())
100  t = IntegerType::get(t.getContext(), 64);
101 
102  return esi::ChannelPort::get(t.getContext(), t);

105 /// Get type name. Currently we only support integer or index types.
106 /// The emitted type aligns with the getFIRRTLType() method. Thus all integers
107 /// other than signed integers will be emitted as unsigned.
108 static std::string getTypeName(Location loc, Type type) {
109  std::string typeName;
110  // Builtin types
111  if (type.isIntOrIndex()) {
112  if (auto indexType = type.dyn_cast<IndexType>())
113  typeName += "_ui" + std::to_string(indexType.kInternalStorageBitWidth);
114  else if (type.isSignedInteger())
115  typeName += "_si" + std::to_string(type.getIntOrFloatBitWidth());
116  else
117  typeName += "_ui" + std::to_string(type.getIntOrFloatBitWidth());
118  } else
119  emitError(loc) << "unsupported data type '" << type << "'";
120 
121  return typeName;
122 }

126 /// names by the Handshake NamedIOInterface; and other operations, such as
127 /// arith ops, are assigned default names.
128 class HandshakePortNameGenerator {
129 public:
130  explicit HandshakePortNameGenerator(Operation *op)
131  : builder(op->getContext()) {
132  auto namedOpInterface = dyn_cast<handshake::NamedIOInterface>(op);
133  if (namedOpInterface)
134  inferFromNamedOpInterface(namedOpInterface);
135  else if (auto funcOp = dyn_cast<handshake::FuncOp>(op))
136  inferFromFuncOp(funcOp);
137  else
138  inferDefault(op);
139  }

141  StringAttr inputName(unsigned idx) { return inputs[idx]; }
142  StringAttr outputName(unsigned idx) { return outputs[idx]; }

145  using IdxToStrF = const std::function<std::string(unsigned)> &;

147  llvm::transform(

149  [&](auto it) { return builder.getStringAttr(inF(it.index())); });
150  llvm::transform(
151  llvm::enumerate(op->getResultTypes()), std::back_inserter(outputs),
152  [&](auto it) { return builder.getStringAttr(outF(it.index())); });

175  [&](auto it) { return builder.getStringAttr(outF(it.index())); });

183 /// Construct a name for creating HW sub-module.
184 static std::string getSubModuleName(Operation *oldOp) {
185  if (auto instanceOp = dyn_cast<handshake::InstanceOp>(oldOp); instanceOp)
186  return instanceOp.getModule().str();
187 
188  std::string subModuleName = getBareSubModuleName(oldOp);
189 
190  // Add value of the constant operation.
191  if (auto constOp = dyn_cast<handshake::ConstantOp>(oldOp)) {
192  if (auto intAttr = constOp.getValue().dyn_cast<IntegerAttr>()) {
193  auto intType = intAttr.getType();
194 
195  if (intType.isSignedInteger())
196  subModuleName += "_c" + std::to_string(intAttr.getSInt());
197  else if (intType.isUnsignedInteger())
198  subModuleName += "_c" + std::to_string(intAttr.getUInt());
199  else
200  subModuleName += "_c" + std::to_string((uint64_t)intAttr.getInt());
201  } else
202  oldOp->emitError("unsupported constant type");
203  }
204 
205  // Add discriminating in- and output types.
206  auto [inTypes, outTypes] = getHandshakeDiscriminatingTypes(oldOp);
207  if (!inTypes.empty())
208  subModuleName += "_in";
209  for (auto inType : inTypes)
210  subModuleName += getTypeName(oldOp->getLoc(), inType);
211 
212  if (!outTypes.empty())
213  subModuleName += "_out";
214  for (auto outType : outTypes)
215  subModuleName += getTypeName(oldOp->getLoc(), outType);
216 
217  // Add memory ID.
218  if (auto memOp = dyn_cast<handshake::MemoryOp>(oldOp))
219  subModuleName += "_id" + std::to_string(memOp.id());
220 
221  // Add compare kind.
222  if (auto comOp = dyn_cast<mlir::arith::CmpIOp>(oldOp))
223  subModuleName += "_" + stringifyEnum(comOp.getPredicate()).str();
224 
225  // Add buffer information.
226  if (auto bufferOp = dyn_cast<handshake::BufferOp>(oldOp)) {
227  subModuleName += "_" + std::to_string(bufferOp.getNumSlots()) + "slots";
228  if (bufferOp.isSequential())
229  subModuleName += "_seq";
230  else
231  subModuleName += "_fifo";
232  }
233 
234  // Add control information.
235  if (auto ctrlInterface = dyn_cast<handshake::ControlInterface>(oldOp);
236  ctrlInterface && ctrlInterface.isControl()) {
237  // Add some additional discriminating info for non-typed operations.
238  subModuleName += "_" + std::to_string(oldOp->getNumOperands()) + "ins_" +
239  std::to_string(oldOp->getNumResults()) + "outs";
240  subModuleName += "_ctrl";
241  } else {
242  assert(
243  (!inTypes.empty() || !outTypes.empty()) &&
244  "Insufficient discriminating type info generated for the operation!");
245  }
246 
247  return subModuleName;
248 }

277 static llvm::SmallVector<PortInfo>
278 getPortInfoForOp(ConversionPatternRewriter &rewriter, Operation *op,
279  TypeRange inputs, TypeRange outputs) {
280  llvm::SmallVector<PortInfo> ports;
281  HandshakePortNameGenerator portNames(op);
282 
283  // Add all inputs of funcOp.
284  unsigned inIdx = 0;
285  for (auto &arg : llvm::enumerate(inputs)) {
286  ports.push_back({portNames.inputName(arg.index()), PortDirection::INPUT,
287  esiWrapper(arg.value()), arg.index(), StringAttr{}});
288  inIdx++;
289  }
290 
291  // Add all outputs of funcOp.
292  for (auto &res : llvm::enumerate(outputs)) {
293  ports.push_back({portNames.outputName(res.index()), PortDirection::OUTPUT,
294  esiWrapper(res.value()), res.index(), StringAttr{}});
295  }
296 
297  // Add clock and reset signals.
298  if (op->hasTrait<mlir::OpTrait::HasClock>()) {
299  ports.push_back({rewriter.getStringAttr("clock"), PortDirection::INPUT,
300  rewriter.getI1Type(), inIdx++, StringAttr{}});
301  ports.push_back({rewriter.getStringAttr("reset"), PortDirection::INPUT,
302  rewriter.getI1Type(), inIdx, StringAttr{}});
303  }
304 
305  return ports;
306 }

334 static hw::HWModuleOp createTopModuleOp(handshake::FuncOp funcOp,
335  ConversionPatternRewriter &rewriter) {
336  llvm::SmallVector<PortInfo, 8> ports = getPortInfoForOp(
337  rewriter, funcOp, funcOp.getArgumentTypes(), funcOp.getResultTypes());
338 
339  // Create a HW module.
340  auto hwModuleOp = rewriter.create<hw::HWModuleOp>(
341  funcOp.getLoc(), rewriter.getStringAttr(funcOp.getName()), ports);
342 
343  // Remove the default created hw_output operation.
344  auto outputOps = hwModuleOp.getOps<hw::OutputOp>();
345  assert(std::distance(outputOps.begin(), outputOps.end()) == 1 &&
346  "Expected exactly 1 default created hw_output operation");
347  rewriter.eraseOp(*outputOps.begin());
348 
349  return hwModuleOp;
350 }

376 static Operation *createOpInHWModule(ConversionPatternRewriter &rewriter,
377  HandshakeLoweringState &ls,
378  Operation *op) {
379  OpBuilder::InsertionGuard g(rewriter);
380  rewriter.setInsertionPointToEnd(ls.hwModuleOp.getBodyBlock());
381 
382  // Create a mapper between the operands of 'op' and the replacement operands
383  // in the target hwModule. For any missing operands, we create a new backedge.
384  llvm::SmallVector<Value> hwOperands =
385  ls.mapper.get(op->getOperands(), esiWrapper);
386 
387  // Add clock and reset if needed.
388  if (op->hasTrait<mlir::OpTrait::HasClock>())
389  hwOperands.append({ls.clock, ls.reset});
390 
391  // Check if a sub-module for the operation already exists.
392  Operation *subModuleSymOp = checkSubModuleOp(ls.parentModule, op);
393  if (!subModuleSymOp) {
394  subModuleSymOp = createSubModuleOp(ls.parentModule, op, rewriter);
395  // TODO: fill the subModuleSymOp with the meat of the handshake
396  // operations.
397  }
398 
399  // Instantiate the new created sub-module.
400  rewriter.setInsertionPointToEnd(ls.hwModuleOp.getBodyBlock());
401  auto submoduleInstanceOp = rewriter.create<hw::InstanceOp>(
402  op->getLoc(), subModuleSymOp, rewriter.getStringAttr(ls.nameUniquer(op)),
403  hwOperands);
404 
405  // Resolve any previously created backedges that referred to the results of
406  // 'op'.
407  ls.mapper.set(op->getResults(), submoduleInstanceOp.getResults());
408 
409  return submoduleInstanceOp;
410 }

419 struct HandshakeFuncOpLowering : public OpConversionPattern<handshake::FuncOp> {
420  using OpConversionPattern<handshake::FuncOp>::OpConversionPattern;
421  HandshakeFuncOpLowering(MLIRContext *context)
422  : OpConversionPattern<handshake::FuncOp>(context) {}
423 
424  LogicalResult
425  matchAndRewrite(handshake::FuncOp funcOp, OpAdaptor adaptor,
426  ConversionPatternRewriter &rewriter) const override {
427  if (failed(verifyHandshakeFuncOp(funcOp)))
428  return failure();
429 
430  ModuleOp parentModule = funcOp->getParentOfType<ModuleOp>();
431  rewriter.setInsertionPointToStart(funcOp->getBlock());
432  HWModuleOp topModuleOp = createTopModuleOp(funcOp, rewriter);
433  Value clockPort =
434  topModuleOp.getArgument(topModuleOp.getNumArguments() - 2);
435  Value resetPort =
436  topModuleOp.getArgument(topModuleOp.getNumArguments() - 1);
437 
438  // Create a uniquer function for creating instance names.
439  NameUniquer instanceUniquer = [&](Operation *op) {
440  std::string instName = getCallName(op);
441  if (auto idAttr = op->getAttrOfType<IntegerAttr>("handshake_id");
442  idAttr) {
443  // We use a special naming convention for operations which have a
444  // 'handshake_id' attribute.
445  instName += "_id" + std::to_string(idAttr.getValue().getZExtValue());
446  } else {
447  // Fallback to just prefixing with an integer.
448  instName += std::to_string(instanceNameCntr[instName]++);
449  }
450  return instName;
451  };
452 
453  // Initialize the Handshake lowering state.
454  BackedgeBuilder bb(rewriter, funcOp.getLoc());
455  ValueMapper valuemapper(&bb);
456  auto ls = HandshakeLoweringState{valuemapper, parentModule, topModuleOp,
457  instanceUniquer, clockPort, resetPort};
458 
459  // Extend value mapper with input arguments. Drop the 2 last inputs from
460  // the HW module (clock and reset).
461  valuemapper.set(funcOp.getArguments(),
462  topModuleOp.getArguments().drop_back(2));
463 
464  // Traverse and convert each operation in funcOp.
465  for (Operation &op : funcOp.front()) {
466  if (isa<hw::OutputOp>(op)) {
467  // Skip the default created HWModule terminator, for now.
468  continue;
469  }
470  if (auto returnOp = dyn_cast<handshake::ReturnOp>(op)) {
471  convertReturnOp(returnOp, ls, rewriter);
472  } else {
473  // Regular operation.
474  createOpInHWModule(rewriter, ls, &op);
475  }
476  }
477  rewriter.eraseOp(funcOp);
478  return success();
479  }

482  /// Maintain a map from module names to the # of times the module has been
483  /// instantiated inside this module. This is used to generate unique names for
484  /// each instance.
485  mutable std::map<std::string, unsigned> instanceNameCntr;
486 };

488 class HandshakeToHWPass : public HandshakeToHWBase<HandshakeToHWPass> {
489 public:
490  void runOnOperation() override {
491  auto op = getOperation();
492 
493  // Lowering to HW requires that every value is used exactly once. Check
494  // whether this precondition is met, and if not, exit.
495  if (llvm::any_of(op.getOps<handshake::FuncOp>(), [](auto f) {
496  return failed(verifyAllValuesHasOneUse(f));
497  })) {
498  signalPassFailure();
499  return;
500  }
501 
502  // Resolve the instance graph to get a top-level module.
503  std::string topLevel;
504  handshake::InstanceGraph uses;
505  SmallVector<std::string> sortedFuncs;
506  if (resolveInstanceGraph(op, uses, topLevel, sortedFuncs).failed()) {
507  signalPassFailure();
508  return;
509  }
510 
511  ConversionTarget target(getContext());
512  target.addLegalDialect<HWDialect>();
513  target.addIllegalDialect<handshake::HandshakeDialect>();
514 
515  // Convert the handshake.func operations in post-order wrt. the instance
516  // graph. This ensures that any referenced submodules (through
517  // handshake.instance) has already been lowered, and their HW module
518  // equivalents are available.
519  for (auto &funcName : llvm::reverse(sortedFuncs)) {
520  RewritePatternSet patterns(op.getContext());
521  patterns.insert<HandshakeFuncOpLowering>(op.getContext());
522  auto *funcOp = op.lookupSymbol(funcName);
523  assert(funcOp && "Symbol not found in module!");
524  if (failed(applyPartialConversion(funcOp, target, std::move(patterns)))) {
525  signalPassFailure();
526  funcOp->emitOpError() << "error during conversion";
527  return;
528  }
529  }
530  }
