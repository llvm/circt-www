44 /// This class represents a single lattice value. A lattive value corresponds to
45 /// the various different states that a value in the SCCP dataflow analysis can
46 /// take. See 'Kind' below for more details on the different states a value can
47 /// take.
48 class LatticeValue {
49  enum Kind {
50  /// A value with a yet-to-be-determined value. This state may be changed to
51  /// anything, it hasn't been processed by IMConstProp.
52  Unknown,
53 
54  /// An FIRRTL 'invalidvalue' value, carrying the result of an
55  /// InvalidValueOp. Wires and other stateful values start out in this
56  /// state.
57  ///
58  /// This is named "InvalidValue" instead of "Invalid" to avoid confusion
59  /// about whether the lattice value is corrupted. "InvalidValue" is a
60  /// valid lattice state, and a can move up to Constant or Overdefined.
61  InvalidValue,
62 
63  /// A value that is known to be a constant. This state may be changed to
64  /// overdefined.
65  Constant,
66 
67  /// A value that cannot statically be determined to be a constant. This
68  /// state cannot be changed.
69  Overdefined
70  };
71 
72 public:
73  /// Initialize a lattice value with "Unknown".
74  /*implicit*/ LatticeValue() : valueAndTag(nullptr, Kind::Unknown) {}
75  /// Initialize a lattice value with a constant.
76  /*implicit*/ LatticeValue(IntegerAttr attr)
77  : valueAndTag(attr, Kind::Constant) {}
78 
79  /// Initialize a lattice value with an InvalidValue constant.
80  /*implicit*/ LatticeValue(InvalidValueAttr attr)
81  : valueAndTag(attr, Kind::InvalidValue) {}
82 
83  static LatticeValue getOverdefined() {
84  LatticeValue result;
85  result.markOverdefined();
86  return result;
87  }
88 
89  bool isUnknown() const { return valueAndTag.getInt() == Kind::Unknown; }
90  bool isInvalidValue() const {
91  return valueAndTag.getInt() == Kind::InvalidValue;
92  }
93  bool isConstant() const { return valueAndTag.getInt() == Kind::Constant; }
94  bool isOverdefined() const {
95  return valueAndTag.getInt() == Kind::Overdefined;
96  }
97 
98  /// Mark the lattice value as overdefined.
99  void markOverdefined() {
100  valueAndTag.setPointerAndInt(nullptr, Kind::Overdefined);
101  }
102 
103  void markInvalidValue(InvalidValueAttr value) {
104  valueAndTag.setPointerAndInt(value, Kind::InvalidValue);
105  }
106 
107  /// Mark the lattice value as constant.
108  void markConstant(IntegerAttr value) {
109  valueAndTag.setPointerAndInt(value, Kind::Constant);
110  }
111 
112  /// If this lattice is constant or invalid value, return the attribute.
113  /// Returns nullptr otherwise.
114  Attribute getValue() const { return valueAndTag.getPointer(); }
115 
116  /// If this is in the constant state, return the IntegerAttr.
117  IntegerAttr getConstant() const {
118  assert(isConstant());
119  return getValue().dyn_cast_or_null<IntegerAttr>();
120  }
121 
122  /// Merge in the value of the 'rhs' lattice into this one. Returns true if the
123  /// lattice value changed.
124  bool mergeIn(LatticeValue rhs) {
125  // If we are already overdefined, or rhs is unknown, there is nothing to do.
126  if (isOverdefined() || rhs.isUnknown())
127  return false;
128 
129  // If we are unknown, just take the value of rhs.
130  if (isUnknown()) {
131  valueAndTag = rhs.valueAndTag;
132  return true;
133  }
134 
135  // If the right side is InvalidValue then it won't contribute anything to
136  // our state since we're either already InvalidValue or a constant here.
137  if (rhs.isInvalidValue())
138  return false;
139 
140  // If we are an InvalidValue, then upgrade to Constant or Overdefined.
141  if (isInvalidValue()) {
142  valueAndTag = rhs.valueAndTag;
143  return true;
144  }
145 
146  // Otherwise, if this value doesn't match rhs go straight to overdefined.
147  // This happens when we merge "3" and "4" from two different instance sites
148  // for example.
149  if (valueAndTag != rhs.valueAndTag) {
150  markOverdefined();
151  return true;
152  }
153  return false;
154  }
155 
156  bool operator==(const LatticeValue &other) const {
157  return valueAndTag == other.valueAndTag;
158  }
159  bool operator!=(const LatticeValue &other) const {
160  return valueAndTag != other.valueAndTag;
161  }
162 
163 private:
164  /// The attribute value if this is a constant and the tag for the element
165  /// kind. The attribute is always an IntegerAttr.
166  llvm::PointerIntPair<Attribute, 2, Kind> valueAndTag;
167 };
168 } // end anonymous namespace

177  return os << "<Invalid>";

188  /// Returns true if the given block is executable.
189  bool isBlockExecutable(Block *block) const {
190  return executableBlocks.count(block);

193  bool isOverdefined(Value value) const {
194  auto it = latticeValues.find(value);
195  return it != latticeValues.end() && it->second.isOverdefined();
196  }
197 
198  /// Mark the given value as overdefined. This means that we cannot refine a
199  /// specific constant for this value.
200  void markOverdefined(Value value) {
201  auto &entry = latticeValues[value];
202  if (!entry.isOverdefined()) {
203  entry.markOverdefined();
204  changedLatticeValueWorklist.push_back(value);
205  }
206  }
207 
208  /// Merge information from the 'from' lattice value into value. If it
209  /// changes, then users of the value are added to the worklist for
210  /// revisitation.
211  void mergeLatticeValue(Value value, LatticeValue &valueEntry,
212  LatticeValue source) {
213  if (!source.isOverdefined() &&
214  (!isa_and_nonnull<InstanceOp>(value.getDefiningOp()) &&
215  hasDontTouch(value)))
216  source = LatticeValue::getOverdefined();
217  if (valueEntry.mergeIn(source))
218  changedLatticeValueWorklist.push_back(value);
219  }
220  void mergeLatticeValue(Value value, LatticeValue source) {
221  // Don't even do a map lookup if from has no info in it.
222  if (source.isUnknown())
223  return;
224  mergeLatticeValue(value, latticeValues[value], source);
225  }
226  void mergeLatticeValue(Value result, Value from) {
227  // If 'from' hasn't been computed yet, then it is unknown, don't do
228  // anything.
229  auto it = latticeValues.find(from);
230  if (it == latticeValues.end())
231  return;
232  mergeLatticeValue(result, it->second);
233  }
234 
235  /// setLatticeValue - This is used when a new LatticeValue is computed for
236  /// the result of the specified value that replaces any previous knowledge,
237  /// e.g. because a fold() function on an op returned a new thing. This should
238  /// not be used on operations that have multiple contributors to it, e.g.
239  /// wires or ports.
240  void setLatticeValue(Value value, LatticeValue source) {
241  // Don't even do a map lookup if from has no info in it.
242  if (source.isUnknown())
243  return;
244 
245  if (!source.isOverdefined() &&
246  (!isa_and_nonnull<InstanceOp>(value.getDefiningOp()) &&
247  hasDontTouch(value)))
248  source = LatticeValue::getOverdefined();
249  // If we've changed this value then revisit all the users.
250  auto &valueEntry = latticeValues[value];
251  if (valueEntry != source) {
252  changedLatticeValueWorklist.push_back(value);
253  valueEntry = source;
254  }
255  }
256 
257  /// Return the lattice value for the specified SSA value, extended to the
258  /// width of the specified destType. If allowTruncation is true, then this
259  /// allows truncating the lattice value to the specified type.
260  LatticeValue getExtendedLatticeValue(Value value, FIRRTLType destType,
261  bool allowTruncation = false);
262 
263  /// Mark the given block as executable.
264  void markBlockExecutable(Block *block);
265  void markWireRegOp(Operation *wireOrReg);
266  void markMemOp(MemOp mem);
267 
268  void markInvalidValueOp(InvalidValueOp invalid);
269  void markConstantOp(ConstantOp constant);
270  void markSpecialConstantOp(SpecialConstantOp specialConstant);
271  void markInstanceOp(InstanceOp instance);
272 
273  void visitConnect(ConnectOp connect);
274  void visitStrictConnect(StrictConnectOp connect);
275  void visitRegResetOp(RegResetOp regReset);
276  void visitOperation(Operation *op);

278 private:
279  /// This is the current instance graph for the Circuit.
280  InstanceGraph *instanceGraph = nullptr;
281 
282  /// This keeps track of the current state of each tracked value.
283  DenseMap<Value, LatticeValue> latticeValues;
284 
285  /// The set of blocks that are known to execute, or are intrinsically live.
286  SmallPtrSet<Block *, 16> executableBlocks;
287 
288  /// A worklist of values whose LatticeValue recently changed, indicating the
289  /// users need to be reprocessed.
290  SmallVector<Value, 64> changedLatticeValueWorklist;
291 
292  /// This keeps track of users the instance results that correspond to output
293  /// ports.
294  DenseMap<BlockArgument, llvm::TinyPtrVector<Value>>
295  resultPortToInstanceResultMapping;
296 };

299 // TODO: handle annotations: [[OptimizableExtModuleAnnotation]]
300 void IMConstPropPass::runOnOperation() {
301  auto circuit = getOperation();
302 
303  instanceGraph = &getAnalysis<InstanceGraph>();
304 
305  // Mark the input ports of public modules as being overdefined.
306  for (auto module : circuit.getBody()->getOps<FModuleOp>()) {
307  if (module.isPublic()) {
308  markBlockExecutable(module.getBody());
309  for (auto port : module.getBody()->getArguments())
310  markOverdefined(port);
311  }
312  }
313 
314  // If a value changed lattice state then reprocess any of its users.
315  while (!changedLatticeValueWorklist.empty()) {
316  Value changedVal = changedLatticeValueWorklist.pop_back_val();
317  for (Operation *user : changedVal.getUsers()) {
318  if (isBlockExecutable(user->getBlock()))
319  visitOperation(user);
320  }
321  }
322 
323  // Rewrite any constants in the modules.
324  mlir::parallelForEach(circuit.getContext(),
325  circuit.getBody()->getOps<FModuleOp>(),
326  [&](auto op) { rewriteModuleBody(op); });
327 
328  // Clean up our state for next time.
329  instanceGraph = nullptr;
330  latticeValues.clear();
331  executableBlocks.clear();
332  resultPortToInstanceResultMapping.clear();
333 }

335 /// Return the lattice value for the specified SSA value, extended to the width
336 /// of the specified destType. If allowTruncation is true, then this allows
337 /// truncating the lattice value to the specified type.
338 LatticeValue IMConstPropPass::getExtendedLatticeValue(Value value,
339  FIRRTLType destType,
340  bool allowTruncation) {
341  // If 'value' hasn't been computed yet, then it is unknown.
342  auto it = latticeValues.find(value);
343  if (it == latticeValues.end())
344  return LatticeValue();
345 
346  auto result = it->second;
347  // Unknown/overdefined stay whatever they are.
348  if (result.isUnknown() || result.isOverdefined())
349  return result;
350  // InvalidValue gets wider.
351  if (result.isInvalidValue())
352  return InvalidValueAttr::get(destType);
353 
354  auto constant = result.getConstant();
355 
356  // If this is a BoolAttr then we are dealing with a special constant.
357  if (auto boolAttr = constant.dyn_cast<BoolAttr>()) {
358  // No extOrTrunc necessary for clock or reset types.
359  return LatticeValue(boolAttr);
360  }
361 
362  // If destType is wider than the source constant type, extend it.
363  auto resultConstant = result.getConstant().getAPSInt();
364  auto destWidth = destType.getBitWidthOrSentinel();
365  if (destWidth == -1) // We don't support unknown width FIRRTL.
366  return LatticeValue::getOverdefined();
367  if (resultConstant.getBitWidth() == (unsigned)destWidth)
368  return result; // Already the right width, we're done.
369 
370  // Otherwise, extend the constant using the signedness of the source.
371  resultConstant = extOrTruncZeroWidth(resultConstant, destWidth);
372  return LatticeValue(IntegerAttr::get(destType.getContext(), resultConstant));
373 }

375 /// Mark a block executable if it isn't already. This does an initial scan of
376 /// the block, processing nullary operations like wires, instances, and
377 /// constants that only get processed once.
378 void IMConstPropPass::markBlockExecutable(Block *block) {
379  if (!executableBlocks.insert(block).second)
380  return; // Already executable.
381 
382  for (auto &op : *block) {
383 
384  // Handle each of the special operations in the firrtl dialect.
385  if (isWireOrReg(&op))
386  markWireRegOp(&op);
387  else if (auto constant = dyn_cast<ConstantOp>(op))
388  markConstantOp(constant);
389  else if (auto specialConstant = dyn_cast<SpecialConstantOp>(op))
390  markSpecialConstantOp(specialConstant);
391  else if (auto invalid = dyn_cast<InvalidValueOp>(op))
392  markInvalidValueOp(invalid);
393  else if (auto instance = dyn_cast<InstanceOp>(op))
394  markInstanceOp(instance);
395  else if (auto mem = dyn_cast<MemOp>(op))
396  markMemOp(mem);
397  }
398 }

400 void IMConstPropPass::markWireRegOp(Operation *wireOrReg) {
401  // If the wire/reg has a non-ground type, then it is too complex for us to
402  // handle, mark it as overdefined.
403  // TODO: Eventually add a field-sensitive model.
404  auto resultValue = wireOrReg->getResult(0);
405  if (!resultValue.getType().cast<FIRRTLType>().getPassiveType().isGround())
406  return markOverdefined(resultValue);
407 
408  // Otherwise, this starts out as InvalidValue and is upgraded by connects.
409  mergeLatticeValue(resultValue, InvalidValueAttr::get(resultValue.getType()));
410 }

429 /// Instances have no operands, so they are visited exactly once when their
430 /// enclosing block is marked live. This sets up the def-use edges for ports.
431 void IMConstPropPass::markInstanceOp(InstanceOp instance) {
432  // Get the module being reference or a null pointer if this is an extmodule.
433  Operation *op = instanceGraph->getReferencedModule(instance);
434 
435  // If this is an extmodule, just remember that any results and inouts are
436  // overdefined.
437  if (!isa<FModuleOp>(op)) {
438  auto module = dyn_cast<FModuleLike>(op);
439  for (size_t resultNo = 0, e = instance.getNumResults(); resultNo != e;
440  ++resultNo) {
441  auto portVal = instance.getResult(resultNo);
442  // If this is an input to the extmodule, we can ignore it.
443  if (module.getPortDirection(resultNo) == Direction::In)
444  continue;
445 
446  // Otherwise this is a result from it or an inout, mark it as overdefined.
447  markOverdefined(portVal);
448  }
449  return;
450  }
451 
452  // Otherwise this is a defined module.
453  auto fModule = cast<FModuleOp>(op);
454  markBlockExecutable(fModule.getBody());
455 
456  // Ok, it is a normal internal module reference. Populate
457  // resultPortToInstanceResultMapping, and forward any already-computed values.
458  for (size_t resultNo = 0, e = instance.getNumResults(); resultNo != e;
459  ++resultNo) {
460  auto instancePortVal = instance.getResult(resultNo);
461  // If this is an input to the instance, it will
462  // get handled when any connects to it are processed.
463  if (fModule.getPortDirection(resultNo) == Direction::In)
464  continue;
465  // We only support simple values so far.
466  if (!instancePortVal.getType().cast<FIRRTLType>().isGround()) {
467  // TODO: Add field sensitivity.
468  markOverdefined(instancePortVal);
469  continue;
470  }
471 
472  // Otherwise we have a result from the instance. We need to forward results
473  // from the body to this instance result's SSA value, so remember it.
474  BlockArgument modulePortVal = fModule.getArgument(resultNo);
475 
476  // Mark don't touch results as overdefined
477  if (hasDontTouch(modulePortVal))
478  markOverdefined(modulePortVal);
479 
480  resultPortToInstanceResultMapping[modulePortVal].push_back(instancePortVal);
481 
482  // If there is already a value known for modulePortVal make sure to forward
483  // it here.
484  mergeLatticeValue(instancePortVal, modulePortVal);
485  }
486 }

488 // We merge the value from the RHS into the value of the LHS.
489 void IMConstPropPass::visitConnect(ConnectOp connect) {
490  auto destType = connect.dest().getType().cast<FIRRTLType>().getPassiveType();
491 
492  // Handle implicit extensions.
493  auto srcValue = getExtendedLatticeValue(connect.src(), destType);
494  if (srcValue.isUnknown())
495  return;
496 
497  // Driving result ports propagates the value to each instance using the
498  // module.
499  if (auto blockArg = connect.dest().dyn_cast<BlockArgument>()) {
500  for (auto userOfResultPort : resultPortToInstanceResultMapping[blockArg])
501  mergeLatticeValue(userOfResultPort, srcValue);
502  // Output ports are wire-like and may have users.
503  mergeLatticeValue(connect.dest(), srcValue);
504  return;
505  }
506 
507  auto dest = connect.dest().cast<mlir::OpResult>();
508 
509  // For wires and registers, we drive the value of the wire itself, which
510  // automatically propagates to users.
511  if (isWireOrReg(dest.getOwner()))
512  return mergeLatticeValue(connect.dest(), srcValue);
513 
514  // Driving an instance argument port drives the corresponding argument of the
515  // referenced module.
516  if (auto instance = dest.getDefiningOp<InstanceOp>()) {
517  // Update the dest, when its an instance op.
518  mergeLatticeValue(connect.dest(), srcValue);
519  auto module =
520  dyn_cast<FModuleOp>(*instanceGraph->getReferencedModule(instance));
521  if (!module)
522  return;
523 
524  BlockArgument modulePortVal = module.getArgument(dest.getResultNumber());
525  return mergeLatticeValue(modulePortVal, srcValue);
526  }
527 
528  // Driving a memory result is ignored because these are always treated as
529  // overdefined.
530  if (auto subfield = dest.getDefiningOp<SubfieldOp>()) {
531  if (subfield.getOperand().getDefiningOp<MemOp>())
532  return;
533  }
534 
535  connect.emitError("connect unhandled by IMConstProp")
536  .attachNote(connect.dest().getLoc())
537  << "connect destination is here";
538 }

589  markOverdefined(connect.src());
590  return markOverdefined(connect.dest());

595  << "strictconnect destination is here";
596 }

598 void IMConstPropPass::visitRegResetOp(RegResetOp regReset) {
599  // If the reg has a non-ground type, then it is too complex for us to handle,
600  // mark it as overdefined.
601  // TODO: Eventually add a field-sensitive model.
602  if (!regReset.getType().getPassiveType().isGround())
603  return markOverdefined(regReset);
604 
605  // The reset value may be known - if so, merge it in if the enable is greater
606  // than invalid.
607  auto srcValue = getExtendedLatticeValue(regReset.resetValue(),
608  regReset.getType().cast<FIRRTLType>(),
609  /*allowTruncation=*/true);
610  auto enable = getExtendedLatticeValue(
611  regReset.resetSignal(),
612  regReset.resetSignal().getType().cast<FIRRTLType>(),
613  /*allowTruncation=*/true);
614  if (enable.isOverdefined() ||
615  (enable.isConstant() && !enable.getConstant().getValue().isZero()))
616  mergeLatticeValue(regReset, srcValue);
617 }

619 /// This method is invoked when an operand of the specified op changes its
620 /// lattice value state and when the block containing the operation is first
621 /// noticed as being alive.
622 ///
623 /// This should update the lattice value state for any result values.
624 ///
625 void IMConstPropPass::visitOperation(Operation *op) {
626  // If this is a operation with special handling, handle it specially.
627  if (auto connectOp = dyn_cast<ConnectOp>(op))
628  return visitConnect(connectOp);
629  if (auto strictConnectOp = dyn_cast<StrictConnectOp>(op))
630  return visitStrictConnect(strictConnectOp);
631  if (auto regResetOp = dyn_cast<RegResetOp>(op))
632  return visitRegResetOp(regResetOp);
633 
634  // The clock operand of regop changing doesn't change its result value.
635  if (isa<RegOp>(op))
636  return;
637  // TODO: Handle 'when' operations.
638 
639  // Nodes might not fold since they might have a name, but should prop
640  if (isa<NodeOp>(op)) {
641  mergeLatticeValue(op->getResult(0), op->getOperand(0));
642  return;
643  }
644 
645  // If all of the results of this operation are already overdefined (or if
646  // there are no results) then bail out early: we've converged.
647  auto isOverdefinedFn = [&](Value value) { return isOverdefined(value); };
648  if (llvm::all_of(op->getResults(), isOverdefinedFn))
649  return;
650 
651  // Collect all of the constant operands feeding into this operation. If any
652  // are not ready to be resolved, bail out and wait for them to resolve.
653  SmallVector<Attribute, 8> operandConstants;
654  operandConstants.reserve(op->getNumOperands());
655  for (Value operand : op->getOperands()) {
656  auto &operandLattice = latticeValues[operand];
657 
658  // If the operand is an unknown value, then we generally don't want to
659  // process it - we want to wait until the value is resolved to by the SCCP
660  // algorithm.
661  if (operandLattice.isUnknown())
662  return;
663 
664  // Otherwise, it must be constant, invalid, or overdefined. Translate them
665  // into attributes that the fold hook can look at.
666  if (operandLattice.isConstant() || operandLattice.isInvalidValue())
667  operandConstants.push_back(operandLattice.getValue());
668  else
669  operandConstants.push_back({});
670  }
671 
672  // Simulate the result of folding this operation to a constant. If folding
673  // fails or was an in-place fold, mark the results as overdefined.
674  SmallVector<OpFoldResult, 8> foldResults;
675  foldResults.reserve(op->getNumResults());
676  if (failed(op->fold(operandConstants, foldResults))) {
677  for (auto value : op->getResults())
678  markOverdefined(value);
679  return;
680  }
681 
682  // Fold functions in general are allowed to do in-place updates, but FIRRTL
683  // does not do this and supporting it costs more.
684  assert(!foldResults.empty() &&
685  "FIRRTL fold functions shouldn't do in-place updates!");
686 
687  // Merge the fold results into the lattice for this operation.
688  assert(foldResults.size() == op->getNumResults() && "invalid result size");
689  for (unsigned i = 0, e = foldResults.size(); i != e; ++i) {
690  // Merge in the result of the fold, either a constant or a value.
691  LatticeValue resultLattice;
692  OpFoldResult foldResult = foldResults[i];
693  if (Attribute foldAttr = foldResult.dyn_cast<Attribute>()) {
694  if (auto intAttr = foldAttr.dyn_cast<IntegerAttr>())
695  resultLattice = LatticeValue(intAttr);
696  else if (auto invalidValueAttr = foldAttr.dyn_cast<InvalidValueAttr>())
697  resultLattice = invalidValueAttr;
698  else // Treat non integer constants as overdefined.
699  resultLattice = LatticeValue::getOverdefined();
700  } else { // Folding to an operand results in its value.
701  resultLattice = latticeValues[foldResult.get<Value>()];
702  }
703 
704  // We do not "merge" the lattice value in, we set it. This is because the
705  // fold functions can produce different values over time, e.g. in the
706  // presence of InvalidValue operands that get resolved to other constants.
707  setLatticeValue(op->getResult(i), resultLattice);
708  }
709 }

711 void IMConstPropPass::rewriteModuleBody(FModuleOp module) {
712  auto *body = module.getBody();
713  // If a module is unreachable, just ignore it.
714  if (!executableBlocks.count(body))
715  return;
716 
717  auto builder = OpBuilder::atBlockBegin(body);
718 
719  // Unique constants per <Const,Type> pair, inserted at entry
720  DenseMap<std::pair<Attribute, Type>, Operation *> constPool;
721  auto getConst = [&](Attribute constantValue, Type type, Location loc) {
722  auto constIt = constPool.find({constantValue, type});
723  if (constIt != constPool.end()) {
724  auto *cst = constIt->second;
725  // Add location to the constant
726  cst->setLoc(builder.getFusedLoc(cst->getLoc(), loc));
727  return cst->getResult(0);
728  }
729  auto savedIP = builder.saveInsertionPoint();
730  builder.setInsertionPointToStart(body);
731  auto *cst = module->getDialect()->materializeConstant(
732  builder, constantValue, type, loc);
733  builder.restoreInsertionPoint(savedIP);
734  assert(cst && "all FIRRTL constants can be materialized");
735  constPool.insert({{constantValue, type}, cst});
736  return cst->getResult(0);
737  };
738 
739  // If the lattice value for the specified value is a constant or
740  // InvalidValue, update it and return true. Otherwise return false.
741  auto replaceValueIfPossible = [&](Value value) -> bool {
742  auto it = latticeValues.find(value);
743  if (it == latticeValues.end() || it->second.isOverdefined() ||
744  it->second.isUnknown())
745  return false;
746 
747  auto cstValue =
748  getConst(it->second.getValue(), value.getType(), value.getLoc());
749 
750  // Replace all uses of this value with the constant, unless this is the
751  // destination of a connect. We leave those alone to avoid upsetting flow.
752  value.replaceUsesWithIf(cstValue, [](OpOperand &operand) {
753  if (isa<ConnectOp>(operand.getOwner()) && operand.getOperandNumber() == 0)
754  return false;
755  if (isa<StrictConnectOp>(operand.getOwner()) &&
756  operand.getOperandNumber() == 0)
757  return false;
758  return true;
759  });
760  return true;
761  };
762 
763  // Constant propagate any ports that are always constant.
764  for (auto &port : body->getArguments())
765  replaceValueIfPossible(port);
766 
767  // TODO: Walk 'when's preorder with `walk`.
768 
769  // Walk the IR bottom-up when folding. We often fold entire chains of
770  // operations into constants, which make the intermediate nodes dead. Going
771  // bottom up eliminates the users of the intermediate ops, allowing us to
772  // aggressively delete them.
773  for (auto &op : llvm::make_early_inc_range(llvm::reverse(*body))) {
774  // Connects to values that we found to be constant can be dropped.
775  if (auto connect = dyn_cast<ConnectOp>(op)) {
776  if (auto *destOp = connect.dest().getDefiningOp()) {
777  if (isDeletableWireOrReg(destOp) && !isOverdefined(connect.dest())) {
778  connect.erase();
779  ++numErasedOp;
780  }
781  }
782  continue;
783  }
784  if (auto connect = dyn_cast<StrictConnectOp>(op)) {
785  if (auto *destOp = connect.dest().getDefiningOp()) {
786  if (isDeletableWireOrReg(destOp) && !isOverdefined(connect.dest())) {
787  connect.erase();
788  ++numErasedOp;
789  }
790  }
791  continue;
792  }
793 
794  // We only fold single-result ops and instances in practice, because they
795  // are the expressions.
796  if (op.getNumResults() != 1 && !isa<InstanceOp>(op))
797  continue;
798 
799  // If this operation is already dead, then go ahead and remove it.
800  if (op.use_empty() &&
801  (wouldOpBeTriviallyDead(&op) || isDeletableWireOrReg(&op))) {
802  op.erase();
803  continue;
804  }
805 
806  // Don't "refold" constants. TODO: Unique in the module entry block.
807  if (isa<ConstantOp, SpecialConstantOp, InvalidValueOp>(op))
808  continue;
809 
810  // If the op had any constants folded, replace them.
811  builder.setInsertionPoint(&op);
812  bool foldedAny = false;
813  for (auto result : op.getResults())
814  foldedAny |= replaceValueIfPossible(result);
815 
816  if (foldedAny)
817  ++numFoldedOp;
818 
819  // If the operation folded to a constant then we can probably nuke it.
820  if (foldedAny && op.use_empty() &&
821  (wouldOpBeTriviallyDead(&op) || isDeletableWireOrReg(&op))) {
822  op.erase();
823  ++numErasedOp;
824  continue;
825  }
826  }
827 }
