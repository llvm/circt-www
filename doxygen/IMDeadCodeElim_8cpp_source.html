27 /// Return true if this is a wire or register we're allowed to delete.
28 static bool isDeletableWireOrRegOrNode(Operation *op) {
29  if (auto name = dyn_cast<FNamableOp>(op))

31  return false;

36 struct IMDeadCodeElimPass : public IMDeadCodeElimBase<IMDeadCodeElimPass> {
37  void runOnOperation() override;
38 
39  void rewriteModuleSignature(FModuleOp module);
40  void rewriteModuleBody(FModuleOp module);
41 
42  void markAlive(Value value) {
43  // If the value is already in `liveSet`, skip it.
44  if (liveSet.insert(value).second)
45  worklist.push_back(value);
46  }
47 
48  /// Return true if the value is known alive.
49  bool isKnownAlive(Value value) const {
50  assert(value && "null should not be used");
51  return liveSet.count(value);
52  }
53 
54  /// Return true if the value is assumed dead.
55  bool isAssumedDead(Value value) const { return !isKnownAlive(value); }
56 
57  /// Return true if the block is alive.
58  bool isBlockExecutable(Block *block) const {
59  return executableBlocks.count(block);
60  }
61 
62  void visitUser(Operation *op);
63  void visitValue(Value value);
64  void visitConnect(FConnectLike connect);
65  void visitSubelement(Operation *op);
66  void markBlockExecutable(Block *block);
67  void markWireOrRegOrNode(Operation *op);
68  void markMemOp(MemOp op);
69  void markInstanceOp(InstanceOp instanceOp);
70  void markUnknownSideEffectOp(Operation *op);
71 
72 private:
73  /// The set of blocks that are known to execute, or are intrinsically alive.
74  DenseSet<Block *> executableBlocks;
75 
76  /// This keeps track of users the instance results that correspond to output
77  /// ports.
78  DenseMap<BlockArgument, llvm::TinyPtrVector<Value>>
79  resultPortToInstanceResultMapping;
80  InstanceGraph *instanceGraph;
81 
82  /// A worklist of values whose liveness recently changed, indicating the
83  /// users need to be reprocessed.
84  SmallVector<Value, 64> worklist;
85  llvm::DenseSet<Value> liveSet;
86 };

101  // For operations with side effects, pessimistically mark results and
102  // operands as alive.

117 void IMDeadCodeElimPass::markInstanceOp(InstanceOp instance) {
118  // Get the module being referenced.
119  Operation *op = instanceGraph->getReferencedModule(instance);
120 
121  // If this is an extmodule, just remember that any inputs and inouts are
122  // alive.
123  if (!isa<FModuleOp>(op)) {
124  auto module = dyn_cast<FModuleLike>(op);
125  for (auto resultNo : llvm::seq(0u, instance.getNumResults())) {
126  auto portVal = instance.getResult(resultNo);
127  // If this is an output to the extmodule, we can ignore it.
128  if (module.getPortDirection(resultNo) == Direction::Out)
129  continue;
130 
131  // Otherwise this is an inuput from it or an inout, mark it as alive.
132  markAlive(portVal);
133  }
134  return;
135  }
136 
137  // Otherwise this is a defined module.
138  auto fModule = cast<FModuleOp>(op);
139  markBlockExecutable(fModule.getBody());
140 
141  // Ok, it is a normal internal module reference so populate
142  // resultPortToInstanceResultMapping.
143  for (auto resultNo : llvm::seq(0u, instance.getNumResults())) {
144  auto instancePortVal = instance.getResult(resultNo);
145 
146  // Otherwise we have a result from the instance. We need to forward results
147  // from the body to this instance result's SSA value, so remember it.
148  BlockArgument modulePortVal = fModule.getArgument(resultNo);
149 
150  resultPortToInstanceResultMapping[modulePortVal].push_back(instancePortVal);
151  }
152 }

154 void IMDeadCodeElimPass::markBlockExecutable(Block *block) {
155  if (!executableBlocks.insert(block).second)
156  return; // Already executable.
157 
158  // Mark ports with don't touch as alive.
159  for (auto blockArg : block->getArguments())
160  if (hasDontTouch(blockArg))
161  markAlive(blockArg);
162 
163  for (auto &op : *block) {
164  if (isWireOrRegOrNode(&op))
165  markWireOrRegOrNode(&op);
166  else if (auto instance = dyn_cast<InstanceOp>(op))
167  markInstanceOp(instance);
168  else if (isa<FConnectLike>(op))
169  // Skip connect op.
170  continue;
171  else if (auto mem = dyn_cast<MemOp>(op))
172  markMemOp(mem);
173  else if (!mlir::MemoryEffectOpInterface::hasNoEffect(&op))
174  markUnknownSideEffectOp(&op);
175 
176  // TODO: Handle attach etc.
177  }
178 }

180 void IMDeadCodeElimPass::runOnOperation() {
181  LLVM_DEBUG(llvm::dbgs() << "===----- Remove unused ports -----==="
182  << "\n");
183  auto circuit = getOperation();
184  instanceGraph = &getAnalysis<InstanceGraph>();
185  for (auto module : circuit.getBody()->getOps<FModuleOp>()) {
186  // Mark the ports of public modules as alive.
187  if (module.isPublic()) {
188  markBlockExecutable(module.getBody());
189  for (auto port : module.getBody()->getArguments())
190  markAlive(port);
191  }
192  }
193 
194  // If a value changed liveness then propagate liveness through its users and
195  // definition.
196  while (!worklist.empty())
197  visitValue(worklist.pop_back_val());
198 
199  // Rewrite module signatures.
200  for (auto module : circuit.getBody()->getOps<FModuleOp>())
201  rewriteModuleSignature(module);
202 
203  // Rewrite module bodies parallelly.
204  mlir::parallelForEach(circuit.getContext(),
205  circuit.getBody()->getOps<FModuleOp>(),
206  [&](auto op) { rewriteModuleBody(op); });
207 }

209 void IMDeadCodeElimPass::visitValue(Value value) {
210  assert(isKnownAlive(value) && "only alive values reach here");
211 
212  // Propagate liveness through users.
213  for (Operation *user : value.getUsers())
214  visitUser(user);
215 
216  // Requiring an input port propagates the liveness to each instance.
217  if (auto blockArg = value.dyn_cast<BlockArgument>()) {
218  auto module = cast<FModuleOp>(blockArg.getParentBlock()->getParentOp());
219  auto portDirection = module.getPortDirection(blockArg.getArgNumber());
220  // If the port is input, it's necessary to mark corresponding input ports of
221  // instances as alive. We don't have to propagate the liveness of output
222  // ports.
223  if (portDirection == Direction::In)
224  for (auto userOfResultPort : resultPortToInstanceResultMapping[blockArg])
225  markAlive(userOfResultPort);
226  return;
227  }
228 
229  // Marking an instance port as alive propagates to the corresponding port of
230  // the module.
231  if (auto instance = value.getDefiningOp<InstanceOp>()) {
232  auto instanceResult = value.cast<mlir::OpResult>();
233  // Update the src, when it's an instance op.
234  auto module =
235  dyn_cast<FModuleOp>(*instanceGraph->getReferencedModule(instance));
236  if (!module)
237  return;
238 
239  BlockArgument modulePortVal =
240  module.getArgument(instanceResult.getResultNumber());
241  return markAlive(modulePortVal);
242  }
243 
244  // If op is defined by an operation, mark its operands as alive.
245  if (auto op = value.getDefiningOp())
246  for (auto operand : op->getOperands())
247  markAlive(operand);
248 }

261 void IMDeadCodeElimPass::rewriteModuleBody(FModuleOp module) {
262  auto *body = module.getBody();
263  // If the module is unreachable, just ignore it.
264  // TODO: Erase this module from circuit op.
265  if (!isBlockExecutable(body))
266  return;
267 
268  // Walk the IR bottom-up when deleting operations.
269  for (auto &op : llvm::make_early_inc_range(llvm::reverse(*body))) {
270  // Connects to values that we found to be dead can be dropped.
271  if (auto connect = dyn_cast<FConnectLike>(op)) {
272  if (isAssumedDead(connect.dest())) {
273  LLVM_DEBUG(llvm::dbgs() << "DEAD: " << connect << "\n";);
274  connect.erase();
275  }
276  continue;
277  }
278 
279  // Delete dead wires, regs and nodes.
280  if (isWireOrRegOrNode(&op) && isAssumedDead(op.getResult(0))) {
281  LLVM_DEBUG(llvm::dbgs() << "DEAD: " << op << "\n";);
282  assert(op.use_empty() && "users should be already removed");
283  op.erase();
284  continue;
285  }
286 
287  // Remove non-sideeffect op using `isOpTriviallyDead`.
288  if (mlir::isOpTriviallyDead(&op))
289  op.erase();
290  }
291 }

293 void IMDeadCodeElimPass::rewriteModuleSignature(FModuleOp module) {
294  // If the module is unreachable, just ignore it.
295  // TODO: Erase this module from circuit op.
296  if (!isBlockExecutable(module.getBody()))
297  return;
298 
299  // Ports of public modules cannot be modified.
300  if (module.isPublic())
301  return;
302 
303  InstanceGraphNode *instanceGraphNode =
304  instanceGraph->lookup(module.moduleNameAttr());
305  LLVM_DEBUG(llvm::dbgs() << "Prune ports of module: " << module.getName()
306  << "\n");
307  SmallVector<unsigned> deadPortIndexes;
308  unsigned numOldPorts = module.getNumPorts();
309 
310  ImplicitLocOpBuilder builder(module.getLoc(), module.getContext());
311  builder.setInsertionPointToStart(module.getBody());
312 
313  for (auto index : llvm::seq(0u, numOldPorts)) {
314  auto argument = module.getArgument(index);
315  assert((!hasDontTouch(argument) || isKnownAlive(argument)) &&
316  "If the port has don't touch, it should be known alive");
317 
318  // If the port has dontTouch, skip.
319  if (hasDontTouch(argument))
320  continue;
321 
322  // If the port is known alive, then we can't delete it except for write-only
323  // output ports.
324  if (isKnownAlive(argument)) {
325  bool deadOutputPortAtAnyInstantiation =
326  module.getPortDirection(index) == Direction::Out &&
327  llvm::all_of(resultPortToInstanceResultMapping[argument],
328  [&](Value result) { return isAssumedDead(result); });
329 
330  if (!deadOutputPortAtAnyInstantiation)
331  continue;
332 
333  // Ok, this port is used only within its defined module. So we can replace
334  // the port with a wire.
335  WireOp wire = builder.create<WireOp>(argument.getType());
336 
337  // Since `liveSet` contains the port, we have to erase it from the set.
338  liveSet.erase(argument);
339  liveSet.insert(wire);
340  argument.replaceAllUsesWith(wire);
341  deadPortIndexes.push_back(index);
342  continue;
343  }
344 
345  // Replace the port with a dummy wire. This wire should be erased within
346  // `rewriteModuleBody`.
347  WireOp wire = builder.create<WireOp>(argument.getType());
348  argument.replaceAllUsesWith(wire);
349  assert(isAssumedDead(wire) && "dummy wire must be dead");
350  deadPortIndexes.push_back(index);
351  }
352 
353  // If there is nothing to remove, abort.
354  if (deadPortIndexes.empty())
355  return;
356 
357  // Erase arguments of the old module from liveSet to prevent from creating
358  // dangling pointers.
359  for (auto arg : module.getArguments())
360  liveSet.erase(arg);
361 
362  // Delete ports from the module.
363  module.erasePorts(deadPortIndexes);
364 
365  // Add arguments of the new module to liveSet.
366  for (auto arg : module.getArguments())
367  liveSet.insert(arg);
368 
369  // Rewrite all uses.
370  for (auto *use : instanceGraphNode->uses()) {
371  auto instance = cast<InstanceOp>(*use->getInstance());
372  ImplicitLocOpBuilder builder(instance.getLoc(), instance);
373  // Since we will rewrite instance op, it is necessary to remove old instance
374  // results from liveSet.
375  for (auto oldResult : instance.getResults())
376  liveSet.erase(oldResult);
377 
378  // Replace old instance results with dummy wires.
379  for (auto index : deadPortIndexes) {
380  auto result = instance.getResult(index);
381  assert(isAssumedDead(result) &&
382  "instance results of dead ports must be dead");
383  WireOp wire = builder.create<WireOp>(result.getType());
384  result.replaceAllUsesWith(wire);
385  }
386 
387  // Create a new instance op without dead ports.
388  auto newInstance = instance.erasePorts(builder, deadPortIndexes);
389 
390  // Mark new results as alive.
391  for (auto newResult : newInstance.getResults())
392  liveSet.insert(newResult);
393 
394  instanceGraph->replaceInstance(instance, newInstance);
395  // Remove old one.
396  instance.erase();
397  }
398 
399  numRemovedPorts += deadPortIndexes.size();
400 }
