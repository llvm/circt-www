25 struct OutlineContainerPattern : public OpConversionPattern<InnerContainerOp> {
26  OutlineContainerPattern(MLIRContext *context, Namespace &ns)
27  : OpConversionPattern<InnerContainerOp>(context), ns(ns) {}
28 
29  using OpAdaptor = typename OpConversionPattern<InnerContainerOp>::OpAdaptor;
30 
31  LogicalResult
32  matchAndRewrite(InnerContainerOp op, OpAdaptor adaptor,
33  ConversionPatternRewriter &rewriter) const override {
34  // Outline the container into the module scope, by prefixing it with the
35  // parent class name.
36  auto parentClass =
37  dyn_cast_or_null<ClassOp>(op.getOperation()->getParentOp());
38  assert(parentClass && "This pattern should never be called on a container "
39  "that is not nested within a class.");
40 
41  rewriter.setInsertionPoint(parentClass);
42  StringAttr newContainerName = rewriter.getStringAttr(
43  ns.newName(parentClass.getName() + "_" + op.getModuleName()));
44  auto newContainer =
45  rewriter.create<OuterContainerOp>(op.getLoc(), newContainerName);
46 
47  rewriter.mergeBlocks(op.getBodyBlock(), newContainer.getBodyBlock(), {});
48 
49  // Rename the ibis.this operation to refer to the proper op.
50  auto thisOp =
51  cast<ThisOp>(cast<ScopeOpInterface>(*newContainer.getOperation())
52  .getThis()
53  .getDefiningOp());
54  rewriter.setInsertionPoint(thisOp);
55  rewriter.replaceOpWithNewOp<ThisOp>(thisOp, newContainer.getNameAttr());
56 
57  // Create a container instance op in the parent class.
58  rewriter.setInsertionPoint(op);
59  rewriter.create<ContainerInstanceOp>(
60  parentClass.getLoc(), hw::InnerSymAttr::get(newContainerName),
61  newContainer.getNameAttr());
62  rewriter.eraseOp(op);
63  return success();
64  }
65 
66  Namespace &ns;
67 };

69 struct ClassToContainerPattern : public OpConversionPattern<ClassOp> {
70  using OpConversionPattern<ClassOp>::OpConversionPattern;
71 
72  LogicalResult
73  matchAndRewrite(ClassOp op, OpAdaptor adaptor,
74  ConversionPatternRewriter &rewriter) const override {
75  // Replace the class by a container of the same name.
76  auto newContainer =
77  rewriter.create<OuterContainerOp>(op.getLoc(), op.getNameAttr());
78  rewriter.mergeBlocks(op.getBodyBlock(), newContainer.getBodyBlock(), {});
79  rewriter.eraseOp(op);

81  }

111 LogicalResult ContainerizePass::outlineContainers() {
112  auto *context = &getContext();
113  ConversionTarget target(*context);
114  target.addLegalDialect<IbisDialect>();
115  target.addDynamicallyLegalOp<InnerContainerOp>(
116  [&](auto *op) { return !isa<ibis::ClassOp>(op->getParentOp()); });
117  RewritePatternSet patterns(context);
118 
119  // Setup a namespace to ensure that the new container names are unique.
120  SymbolCache symCache;
121  Namespace ns;
122  symCache.addDefinitions(getOperation());
123  ns.add(symCache);
124  patterns.insert<OutlineContainerPattern>(context, ns);
125  return applyPartialConversion(getOperation(), target, std::move(patterns));
126 }

128 LogicalResult ContainerizePass::containerizeClasses() {
129  auto *context = &getContext();
130  ConversionTarget target(*context);
131  target.addLegalDialect<IbisDialect>();
132  target.addIllegalOp<ClassOp, InstanceOp>();
133  RewritePatternSet patterns(context);
134  patterns.insert<ClassToContainerPattern, InstanceToContainerInstancePattern>(
135  context);
136  return applyPartialConversion(getOperation(), target, std::move(patterns));
137 }
