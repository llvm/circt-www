30 // Analysis result for generating the port interface of a container + a bit of
31 // port op caching.
32 struct ContainerPortInfo {
33  std::unique_ptr<hw::ModulePortInfo> hwPorts;
34 
35  // A mapping between the port name and the port op within the container.
36  llvm::DenseMap<StringAttr, InputPortOp> opInputs;
37 
38  // A mapping between the port name and the port op within the container.
39  llvm::DenseMap<StringAttr, OutputPortOp> opOutputs;
40 
41  ContainerPortInfo() = default;
42  ContainerPortInfo(ContainerOpInterface container) {
43  SmallVector<hw::PortInfo, 4> inputs, outputs;
44 
45  // Copies all attributes from a port, except for the port symbol and type.
46  auto copyPortAttrs = [](auto port) {
47  llvm::DenseSet<StringAttr> elidedAttrs;
48  elidedAttrs.insert(port.getInnerSymAttrName());
49  elidedAttrs.insert(port.getTypeAttrName());
50  llvm::SmallVector<NamedAttribute> attrs;
51  for (NamedAttribute namedAttr : port->getAttrs()) {
52  if (elidedAttrs.contains(namedAttr.getName()))
53  continue;
54  attrs.push_back(namedAttr);
55  }
56  return DictionaryAttr::get(port.getContext(), attrs);
57  };
58 
59  // Gather in and output port ops.
60  for (auto input : container.getBodyBlock()->getOps<InputPortOp>()) {
61  opInputs[input.getInnerSym().getSymName()] = input;
62 
63  hw::PortInfo portInfo;
64  portInfo.name = input.getInnerSym().getSymName();
65  portInfo.type = cast<PortOpInterface>(input.getOperation()).getPortType();
66  portInfo.dir = hw::ModulePort::Direction::Input;
67  portInfo.attrs = copyPortAttrs(input);
68  inputs.push_back(portInfo);
69  }
70 
71  for (auto output : container.getBodyBlock()->getOps<OutputPortOp>()) {
72  opOutputs[output.getInnerSym().getSymName()] = output;
73 
74  hw::PortInfo portInfo;
75  portInfo.name = output.getInnerSym().getSymName();
76  portInfo.type =
77  cast<PortOpInterface>(output.getOperation()).getPortType();
78  portInfo.dir = hw::ModulePort::Direction::Output;
79  portInfo.attrs = copyPortAttrs(output);
80  outputs.push_back(portInfo);
81  }
82  hwPorts = std::make_unique<hw::ModulePortInfo>(inputs, outputs);
83  }
84 };

88 template <typename ContainerOp>
89 struct ContainerOpConversionPattern : public OpConversionPattern<ContainerOp> {
90  ContainerOpConversionPattern(MLIRContext *ctx,
91  ContainerPortInfoMap &portOrder)
92  : OpConversionPattern<ContainerOp>(ctx), portOrder(portOrder) {}
93 
94  LogicalResult
95  matchAndRewrite(ContainerOp op, typename ContainerOp::Adaptor adaptor,
96  ConversionPatternRewriter &rewriter) const override {
97  rewriter.setInsertionPoint(op);
98 
99  const ContainerPortInfo &cpi = portOrder.at(op.getSymNameAttr());
100  auto hwMod = rewriter.create<hw::HWModuleOp>(
101  op.getLoc(), op.getSymNameAttr(), *cpi.hwPorts);
102 
103  hw::OutputOp outputOp =
104  cast<hw::OutputOp>(hwMod.getBodyBlock()->getTerminator());
105 
106  // Replace all of the reads of the inputs to use the input block arguments.
107  for (auto [idx, input] : llvm::enumerate(cpi.hwPorts->getInputs())) {
108  Value barg = hwMod.getBodyBlock()->getArgument(idx);
109  InputPortOp inputPort = cpi.opInputs.at(input.name);
110  // Replace all reads of the input port with the input block argument.
111  for (auto *user : inputPort.getOperation()->getUsers()) {
112  auto reader = dyn_cast<PortReadOp>(user);
113  if (!reader)
114  return rewriter.notifyMatchFailure(
115  user, "expected only ibis.port.read ops of the input port");
116 
117  rewriter.replaceOp(reader, barg);
118  }
119 
120  rewriter.eraseOp(inputPort);
121  }
122 
123  // Adjust the hw.output op to use ibis.port.write values
124  llvm::SmallVector<Value> outputValues;
125  for (auto [idx, output] : llvm::enumerate(cpi.hwPorts->getOutputs())) {
126  auto outputPort = cpi.opOutputs.at(output.name);
127  // Locate the write to the output op.
128  auto users = outputPort->getUsers();
129  size_t nUsers = std::distance(users.begin(), users.end());
130  if (nUsers != 1)
131  return outputPort->emitOpError()
132  << "expected exactly one ibis.port.write op of the output "
133  "port: "
134  << output.name.str() << " found: " << nUsers;
135  auto writer = cast<PortWriteOp>(*users.begin());
136  outputValues.push_back(writer.getValue());
137  rewriter.eraseOp(outputPort);
138  rewriter.eraseOp(writer);
139  }
140 
141  rewriter.mergeBlocks(&op.getBodyRegion().front(), hwMod.getBodyBlock());
142 
143  // Rewrite the hw.output op.
144  rewriter.eraseOp(outputOp);
145  rewriter.setInsertionPointToEnd(hwMod.getBodyBlock());
146  outputOp = rewriter.create<hw::OutputOp>(op.getLoc(), outputValues);
147  rewriter.eraseOp(op);
148  return success();
149  }
150 
151  ContainerPortInfoMap &portOrder;
152 };

167 struct ContainerInstanceOpConversionPattern
168  : public OpConversionPattern<ContainerInstanceOp> {
169 
170  ContainerInstanceOpConversionPattern(MLIRContext *ctx,
171  ContainerPortInfoMap &portOrder)
172  : OpConversionPattern<ContainerInstanceOp>(ctx), portOrder(portOrder) {}
173 
174  LogicalResult
175  matchAndRewrite(ContainerInstanceOp op, OpAdaptor adaptor,
176  ConversionPatternRewriter &rewriter) const override {
177  rewriter.setInsertionPoint(op);
178  llvm::SmallVector<Value> operands;
179 
180  // Gather the get_port ops that target the instance
181  llvm::DenseMap<StringAttr, PortReadOp> outputReadsToReplace;
182  llvm::DenseMap<StringAttr, PortWriteOp> inputWritesToUse;
183  llvm::SmallVector<Operation *> getPortsToErase;
184  for (auto *user : op->getUsers()) {
185  auto getPort = dyn_cast<GetPortOp>(user);
186  if (!getPort)
187  return rewriter.notifyMatchFailure(
188  user, "expected only ibis.get_port op usage of the instance");
189 
190  for (auto *user : getPort->getUsers()) {
191  auto res =
192  llvm::TypeSwitch<Operation *, LogicalResult>(user)
193  .Case<PortReadOp>([&](auto read) {
194  auto [it, inserted] = outputReadsToReplace.insert(
195  {getPort.getPortSymbolAttr().getAttr(), read});
196  if (!inserted)
197  return rewriter.notifyMatchFailure(
198  read, "expected only one ibis.port.read op of the "
199  "output port");
200  return success();
201  })
202  .Case<PortWriteOp>([&](auto write) {
203  auto [it, inserted] = inputWritesToUse.insert(
204  {getPort.getPortSymbolAttr().getAttr(), write});
205  if (!inserted)
206  return rewriter.notifyMatchFailure(
207  write,
208  "expected only one ibis.port.write op of the input "
209  "port");
210  return success();
211  })
212  .Default([&](auto op) {
213  return rewriter.notifyMatchFailure(
214  op, "expected only ibis.port.read or ibis.port.write ops "
215  "of the "
216  "instance");
217  });
218  if (failed(res))
219  return failure();
220  }
221  getPortsToErase.push_back(getPort);
222  }
223 
224  // Grab the operands in the order of the hw.module ports.
225  const ContainerPortInfo &cpi = portOrder.at(rewriter.getStringAttr(
226  getScopeRefModuleName(op.getResult().getType())));
227  size_t nInputPorts = std::distance(cpi.hwPorts->getInputs().begin(),
228  cpi.hwPorts->getInputs().end());
229  if (nInputPorts != inputWritesToUse.size()) {
230  std::string errMsg;
231  llvm::raw_string_ostream ers(errMsg);
232  ers << "Error when lowering instance ";
233  op.print(ers, mlir::OpPrintingFlags().printGenericOpForm());
234 
235  ers << "\nexpected exactly one ibis.port.write op of each input port. "
236  "Mising port assignments were:\n";
237  for (auto input : cpi.hwPorts->getInputs()) {
238  if (inputWritesToUse.find(input.name) == inputWritesToUse.end())
239  ers << "\t" << input.name << "\n";
240  }
241  return rewriter.notifyMatchFailure(op, errMsg);
242  }
243  for (auto input : cpi.hwPorts->getInputs()) {
244  auto writeOp = inputWritesToUse.at(input.name);
245  operands.push_back(writeOp.getValue());
246  rewriter.eraseOp(writeOp);
247  }
248 
249  // Determine the result types.
250  llvm::SmallVector<Type> retTypes;
251  for (auto output : cpi.hwPorts->getOutputs())
252  retTypes.push_back(output.type);
253 
254  // Gather arg and res names
255  // TODO: @mortbopet - this should be part of ModulePortInfo
256  llvm::SmallVector<Attribute> argNames, resNames;
257  llvm::transform(cpi.hwPorts->getInputs(), std::back_inserter(argNames),
258  [](auto port) { return port.name; });
259  llvm::transform(cpi.hwPorts->getOutputs(), std::back_inserter(resNames),
260  [](auto port) { return port.name; });
261 
262  // Create the hw.instance op.
263  StringRef moduleName = getScopeRefModuleName(op.getType());
264  auto hwInst = rewriter.create<hw::InstanceOp>(
265  op.getLoc(), retTypes, op.getInnerSym().getSymName(), moduleName,
266  operands, rewriter.getArrayAttr(argNames),
267  rewriter.getArrayAttr(resNames),
268  /*parameters*/ rewriter.getArrayAttr({}), /*innerSym*/ nullptr);
269 
270  // Replace the reads of the output ports with the hw.instance results.
271  for (auto [output, value] :
272  llvm::zip(cpi.hwPorts->getOutputs(), hwInst.getResults())) {
273  auto outputReadIt = outputReadsToReplace.find(output.name);
274  if (outputReadIt == outputReadsToReplace.end())
275  continue;
276  // TODO: RewriterBase::replaceAllUsesWith is not currently supported by
277  // DialectConversion. Using it may lead to assertions about mutating
278  // replaced/erased ops. For now, do this RAUW directly, until
279  // ConversionPatternRewriter properly supports RAUW.
280  // See https://github.com/llvm/circt/issues/6795.
281  outputReadIt->second.getResult().replaceAllUsesWith(value);
282  rewriter.eraseOp(outputReadIt->second);
283  }
284 
285  // Erase the get_port ops.
286  for (auto *getPort : getPortsToErase)
287  rewriter.eraseOp(getPort);
288 
289  // And finally erase the instance op.
290  rewriter.eraseOp(op);
291  return success();
292  }

302 void ContainersToHWPass::runOnOperation() {
303  auto *ctx = &getContext();
304 
305  // Generate module signatures.
306  ContainerPortInfoMap portOrder;
307  for (auto container : getOperation().getOps<ContainerOpInterface>())
308  portOrder.try_emplace(container.getSymNameAttr(),
309  ContainerPortInfo(container));
310 
311  ConversionTarget target(*ctx);
312  target.addIllegalOp<OuterContainerOp, InnerContainerOp, ContainerInstanceOp,
313  ThisOp>();
314  target.markUnknownOpDynamicallyLegal([](Operation *) { return true; });
315 
316  // Parts of the conversion patterns will update operations in place, which in
317  // turn requires the updated operations to be legalizeable. These in-place ops
318  // also include ibis ops that eventually will get replaced once all of the
319  // patterns apply.
320  target.addLegalDialect<IbisDialect>();
321 
322  RewritePatternSet patterns(ctx);
323  patterns.add<ContainerOpConversionPattern<OuterContainerOp>,
324  ContainerOpConversionPattern<InnerContainerOp>,
325  ContainerInstanceOpConversionPattern>(ctx, portOrder);
326  patterns.add<ThisOpConversionPattern>(ctx);
327 
328  if (failed(
329  applyPartialConversion(getOperation(), target, std::move(patterns))))
330  signalPassFailure();
331 }
