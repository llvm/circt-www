31 struct InferReadWritePass : public InferReadWriteBase<InferReadWritePass> {
32 
33  /// This pass performs two memory transformations:
34  /// 1. If the multi-bit enable port is connected to a constant 1,
35  /// then, replace with a single bit mask. Create a new memory with a
36  /// 1 bit mask, and replace the old memory with it. The single bit mask
37  /// memory is always lowered to an unmasked memory.
38  /// 2. If the read and write enable ports are trivially mutually exclusive,
39  /// then create a new memory with a single read/write port, and replace
40  /// the old memory with it.
41  void runOnOperation() override {
42  LLVM_DEBUG(llvm::dbgs() << "\n Running Infer Read Write on module:"
43  << getOperation().getName());
44  ModuleNamespace modNamespace(getOperation());
45  SmallVector<Operation *> opsToErase;
46  for (MemOp memOp : llvm::make_early_inc_range(
47  getOperation().getBody()->getOps<MemOp>())) {
48  inferUnmasked(memOp, opsToErase);
49  size_t nReads, nWrites, nRWs;
50  memOp.getNumPorts(nReads, nWrites, nRWs);
51  // Run the analysis only for Seq memories (latency=1) and a single read
52  // and write ports.
53  if (!(nReads == 1 && nWrites == 1 && nRWs == 0) ||
54  !(memOp.readLatency() == 1 && memOp.writeLatency() == 1))
55  continue;
56  Value rClock, wClock;
57  // The memory has exactly two ports.
58  SmallVector<Value> portTerms[2];
59  for (const auto &portIt : llvm::enumerate(memOp.results())) {
60  // Get the port value.
61  Value portVal = portIt.value();
62  // Get the port kind.
63  bool readPort =
64  memOp.getPortKind(portIt.index()) == MemOp::PortKind::Read;
65  // Iterate over all users of the port.
66  for (Operation *u : portVal.getUsers())
67  if (auto sf = dyn_cast<SubfieldOp>(u)) {
68  // Get the field name.
69  auto fName = sf.input().getType().cast<BundleType>().getElementName(
70  sf.fieldIndex());
71  // If this is the enable field, record the product terms(the And
72  // expression tree).
73  if (fName.equals("en"))
74  getProductTerms(sf, portTerms[portIt.index()]);
75  else if (fName.equals("clk")) {
76  if (readPort)
77  rClock = getConnectSrc(sf);
78  else
79  wClock = getConnectSrc(sf);
80  }
81  }
82  // End of loop for getting MemOp port users.
83  }
84  if (!sameDriver(rClock, wClock))
85  continue;
86 
87  rClock = wClock;
88  LLVM_DEBUG(
89  llvm::dbgs() << "\n read clock:" << rClock
90  << " --- write clock:" << wClock;
91  llvm::dbgs() << "\n Read terms==>"; for (auto t
92  : portTerms[0]) llvm::dbgs()
93  << "\n term::" << t;
94 
95  llvm::dbgs() << "\n Write terms==>"; for (auto t
96  : portTerms[1]) llvm::dbgs()
97  << "\n term::" << t;
98 
99  );
100  // If the read and write clocks are the same, check if any of the product
101  // terms are a complement of each other.
102  if (!checkComplement(portTerms))
103  continue;
104 
105  SmallVector<Attribute, 4> resultNames;
106  SmallVector<Type, 4> resultTypes;
107  SmallVector<Attribute, 4> portAnnotations;
108  // Create the merged rw port for the new memory.
109  resultNames.push_back(
110  StringAttr::get(memOp.getContext(), modNamespace.newName("rw")));
111  // Set the type of the rw port.
112  resultTypes.push_back(MemOp::getTypeForPort(
113  memOp.depth(), memOp.getDataType(), MemOp::PortKind::ReadWrite,
114  memOp.getMaskBits()));
115  SmallVector<Attribute> portAtts;
116  // Append the annotations from the two ports.
117  if (!memOp.portAnnotations()[0].cast<ArrayAttr>().empty())
118  portAtts.push_back(memOp.portAnnotations()[0]);
119  if (!memOp.portAnnotations()[1].cast<ArrayAttr>().empty())
120  portAtts.push_back(memOp.portAnnotations()[1]);
121  ImplicitLocOpBuilder builder(memOp.getLoc(), memOp);
122  portAnnotations.push_back(builder.getArrayAttr(portAtts));
123  // Create the new rw memory.
124  auto rwMem = builder.create<MemOp>(
125  resultTypes, memOp.readLatency(), memOp.writeLatency(), memOp.depth(),
126  RUWAttr::Undefined, builder.getArrayAttr(resultNames),
127  memOp.nameAttr(), memOp.nameKind(), memOp.annotations(),
128  builder.getArrayAttr(portAnnotations), memOp.inner_symAttr(),
129  memOp.groupIDAttr());
130  auto rwPort = rwMem->getResult(0);
131  // Create the subfield access to all fields of the port.
132  // The addr should be connected to read/write address depending on the
133  // read/write mode.
134  auto addr = builder.create<SubfieldOp>(rwPort, "addr");
135  // Enable is high whenever the memory is written or read.
136  auto enb = builder.create<SubfieldOp>(rwPort, "en");
137  // Read/Write clock.
138  auto clk = builder.create<SubfieldOp>(rwPort, "clk");
139  auto readData = builder.create<SubfieldOp>(rwPort, "rdata");
140  // wmode is high when the port is in write mode. That is this can be
141  // connected to the write enable.
142  auto wmode = builder.create<SubfieldOp>(rwPort, "wmode");
143  auto writeData = builder.create<SubfieldOp>(rwPort, "wdata");
144  auto mask = builder.create<SubfieldOp>(rwPort, "wmask");
145  // Temp wires to replace the original memory connects.
146  auto rAddr = builder.create<WireOp>(addr.getType(), "readAddr");
147  auto wAddr = builder.create<WireOp>(addr.getType(), "writeAddr");
148  auto wEnWire = builder.create<WireOp>(enb.getType(), "writeEnable");
149  auto rEnWire = builder.create<WireOp>(enb.getType(), "readEnable");
150  auto writeClock =
151  builder.create<WireOp>(ClockType::get(enb.getContext()));
152  // addr = Mux(WriteEnable, WriteAddress, ReadAddress).
153  builder.create<StrictConnectOp>(
154  addr, builder.create<MuxPrimOp>(wEnWire, wAddr, rAddr));
155  // Enable = Or(WriteEnable, ReadEnable).
156  builder.create<StrictConnectOp>(
157  enb, builder.create<OrPrimOp>(rEnWire, wEnWire));
158  // WriteMode = WriteEnable.
159  builder.create<StrictConnectOp>(wmode, wEnWire);
160  // Now iterate over the original memory read and write ports.
161  for (const auto &portIt : llvm::enumerate(memOp.results())) {
162  // Get the port value.
163  Value portVal = portIt.value();
164  // Get the port kind.
165  bool readPort =
166  memOp.getPortKind(portIt.index()) == MemOp::PortKind::Read;
167  // Iterate over all users of the port, which are the subfield ops, and
168  // replace them.
169  for (Operation *u : portVal.getUsers())
170  if (auto sf = dyn_cast<SubfieldOp>(u)) {
171  StringRef fName =
172  sf.input().getType().cast<BundleType>().getElementName(
173  sf.fieldIndex());
174  Value repl;
175  if (readPort)
176  repl = llvm::StringSwitch<Value>(fName)
177  .Case("en", rEnWire)
178  .Case("clk", clk)
179  .Case("addr", rAddr)
180  .Case("data", readData);
181  else
182  repl = llvm::StringSwitch<Value>(fName)
183  .Case("en", wEnWire)
184  .Case("clk", writeClock)
185  .Case("addr", wAddr)
186  .Case("data", writeData)
187  .Case("mask", mask);
188  sf.replaceAllUsesWith(repl);
189  // Once all the uses of the subfield op replaced, delete it.
190  opsToErase.push_back(sf);
191  }
192  }
193  // All uses for all results of mem removed, now erase the memOp.
194  opsToErase.push_back(memOp);
195  }
196  for (auto *o : opsToErase)
197  o->erase();
198  }

200 private:
201  // Get the source value which is connected to the dst.
202  Value getConnectSrc(Value dst) {
203  for (auto *c : dst.getUsers())
204  if (auto connect = dyn_cast<FConnectLike>(c))
205  if (connect.dest() == dst)
206  return connect.src();
207 
208  return nullptr;
209  }
210 
211  /// If the ports are not directly connected to the same clock, then check
212  /// if indirectly connected to the same clock.
213  bool sameDriver(Value rClock, Value wClock) {
214  if (rClock == wClock)
215  return true;
216  DenseSet<Value> rClocks, wClocks;
217  while (rClock) {
218  // Record all the values which are indirectly connected to the clock
219  // port.
220  rClocks.insert(rClock);
221  rClock = getConnectSrc(rClock);
222  }
223 
224  bool sameClock = false;
225  // Now check all the indirect connections to the write memory clock
226  // port.
227  while (wClock) {
228  if (rClocks.find(wClock) != rClocks.end()) {
229  sameClock = true;
230  break;
231  }
232  wClock = getConnectSrc(wClock);
233  }
234  return sameClock;
235  }

237  void getProductTerms(Value enValue, SmallVector<Value> &terms) {
238  if (!enValue)
239  return;
240  SmallVector<Value> worklist;
241  worklist.push_back(enValue);
242  while (!worklist.empty()) {
243  auto term = worklist.back();
244  worklist.pop_back();
245  terms.push_back(term);
246  if (term.isa<BlockArgument>())
247  continue;
248  TypeSwitch<Operation *>(term.getDefiningOp())
249  .Case<NodeOp>([&](auto n) { worklist.push_back(n.input()); })
250  .Case<AndPrimOp>([&](AndPrimOp andOp) {
251  worklist.push_back(andOp.getOperand(0));
252  worklist.push_back(andOp.getOperand(1));
253  })
254  .Case<MuxPrimOp>([&](auto muxOp) {
255  // Check for the pattern when low is 0, which is equivalent to (sel
256  // & high)
257  // term = mux (sel, high, 0) => term = sel & high
258  if (ConstantOp cLow =
259  dyn_cast_or_null<ConstantOp>(muxOp.low().getDefiningOp()))
260  if (cLow.value().isZero()) {
261  worklist.push_back(muxOp.sel());
262  worklist.push_back(muxOp.high());
263  }
264  })
265  .Default([&](auto) {
266  if (auto src = getConnectSrc(term))
267  worklist.push_back(src);
268  });
269  }
270  }
271 
272  /// Check if any of the terms in the prodTerms[0] is a complement of any of
273  /// the terms in prodTerms[1]. prodTerms[0], prodTerms[1] is a vector of
274  /// Value, each of which correspond to the two product terms of read/write
275  /// enable.
276  bool checkComplement(SmallVector<Value> prodTerms[2]) {
277  bool isComplement = false;
278  // Foreach Value in first term, check if it is the complement of any of the
279  // Value in second term.
280  for (auto t1 : prodTerms[0])
281  for (auto t2 : prodTerms[1]) {
282  // Return true if t1 is a Not of t2.
283  if (!t1.isa<BlockArgument>() && isa<NotPrimOp>(t1.getDefiningOp()))
284  if (cast<NotPrimOp>(t1.getDefiningOp()).input() == t2)
285  return true;
286  // Else Return true if t2 is a Not of t1.
287  if (!t2.isa<BlockArgument>() && isa<NotPrimOp>(t2.getDefiningOp()))
288  if (cast<NotPrimOp>(t2.getDefiningOp()).input() == t1)
289  return true;
290  }
291 
292  return isComplement;
293  }

295  void inferUnmasked(MemOp &memOp, SmallVector<Operation *> &opsToErase) {
296  bool isMasked = true;
297 
298  // Iterate over all results, and check if the mask field of the result is
299  // connected to a multi-bit constant 1.
300  for (const auto &portIt : llvm::enumerate(memOp.results())) {
301  // Read ports donot have the mask field.
302  if (memOp.getPortKind(portIt.index()) == MemOp::PortKind::Read)
303  continue;
304  Value portVal = portIt.value();
305  // Iterate over all users of the write/rw port.
306  for (Operation *u : portVal.getUsers())
307  if (auto sf = dyn_cast<SubfieldOp>(u)) {
308  // Get the field name.
309  auto fName = sf.input().getType().cast<BundleType>().getElementName(
310  sf.fieldIndex());
311  // Check if this is the mask field.
312  if (fName.contains("mask")) {
313  // Already 1 bit, nothing to do.
314  if (sf.result()
315  .getType()
316  .cast<FIRRTLType>()
317  .getBitWidthOrSentinel() == 1)
318  continue;
319  // Check what is the mask field directly connected to.
320  // If, a constant 1, then we can replace with unMasked memory.
321  if (auto maskVal = getConnectSrc(sf))
322  if (auto constVal = dyn_cast<ConstantOp>(maskVal.getDefiningOp()))
323  if (constVal.value().isAllOnes())
324  isMasked = false;
325  }
326  }
327  }
328 
329  if (!isMasked) {
330  // Replace with a new memory of 1 bit mask.
331  ImplicitLocOpBuilder builder(memOp.getLoc(), memOp);
332  // Copy the result type, except the mask bits.
333  SmallVector<Type, 4> resultTypes;
334  for (size_t i = 0, e = memOp.getNumResults(); i != e; ++i)
335  resultTypes.push_back(
336  MemOp::getTypeForPort(memOp.depth(), memOp.getDataType(),
337  memOp.getPortKind(i), /*maskBits=*/1));
338 
339  // Copy everything from old memory, except the result type.
340  auto newMem = builder.create<MemOp>(
341  resultTypes, memOp.readLatency(), memOp.writeLatency(), memOp.depth(),
342  memOp.ruw(), memOp.portNames().getValue(), memOp.nameAttr(),
343  memOp.nameKind(), memOp.annotations().getValue(),
344  memOp.portAnnotations().getValue(), memOp.inner_symAttr());
345  // Now replace the result of old memory with the new one.
346  for (const auto &portIt : llvm::enumerate(memOp.results())) {
347  // Old result.
348  Value oldPort = portIt.value();
349  // New result.
350  auto newPortVal = newMem->getResult(portIt.index());
351  // If read port, then blindly replace.
352  if (memOp.getPortKind(portIt.index()) == MemOp::PortKind::Read) {
353  oldPort.replaceAllUsesWith(newPortVal);
354  continue;
355  }
356  // Otherwise, all fields can be blindly replaced, except mask field.
357  for (Operation *u : oldPort.getUsers()) {
358  auto oldRes = dyn_cast<SubfieldOp>(u);
359  auto sf = builder.create<SubfieldOp>(newPortVal, oldRes.fieldIndex());
360  auto fName = sf.input().getType().cast<BundleType>().getElementName(
361  sf.fieldIndex());
362  // Replace all mask fields with a one bit constant 1.
363  // Replace all other fields with the new port.
364  if (fName.contains("mask")) {
365  WireOp dummy = builder.create<WireOp>(oldRes.getType());
366  oldRes->replaceAllUsesWith(dummy);
367  builder.create<StrictConnectOp>(
368  sf, builder.create<ConstantOp>(
369  UIntType::get(builder.getContext(), 1), APInt(1, 1)));
370  } else
371  oldRes->replaceAllUsesWith(sf);
372 
373  opsToErase.push_back(oldRes);
374  }
375  }
376  opsToErase.push_back(memOp);
377  memOp = newMem;
378  }
379  }
