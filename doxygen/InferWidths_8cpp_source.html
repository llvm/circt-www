127 /// An identity expression.
128 ///
129 /// This expression evaluates to its inner expression. It is used in a very
130 /// specific case of constraints on variables, in order to be able to track
131 /// where the constraint was imposed. Constraints on variables are represented
132 /// as `var >= <expr>`. When the first constraint `a` is imposed, it is stored
133 /// as the constraint expression (`var >= a`). When the second constraint `b` is
134 /// imposed, a *new* max expression is allocated (`var >= max(a, b)`).
135 /// Expressions are annotated with a location when they are created, which in
136 /// this case are connect ops. Since imposing the first constraint does not
137 /// create any new expression, the location information of that connect would be
138 /// lost. With an identity expression, imposing the first constraint becomes
139 /// `var >= identity(a)`, which is a *new* expression and properly tracks the
140 /// location info.
141 struct IdExpr : public ExprBase<IdExpr, Expr::Kind::Id> {
142  IdExpr(Expr *arg) : arg(arg) { assert(arg); }
143  void print(llvm::raw_ostream &os) const { os << "*" << *arg; }
144  iterator begin() const { return &arg; }
145  iterator end() const { return &arg + (arg ? 1 : 0); }
146  bool operator==(const IdExpr &other) const {
147  return kind == other.kind && arg == other.arg;
148  }
149  llvm::hash_code hash_value() const {
150  return llvm::hash_combine(Expr::hash_value(), arg);
151  }
152 
153  /// The inner expression.
154  Expr *const arg;
155 };

310 /// A simple bump allocator that ensures only ever one copy per object exists.
311 /// The allocated objects must not have a destructor.
312 template <typename T, typename std::enable_if_t<
313  std::is_trivially_destructible<T>::value, int> = 0>
314 class InternedAllocator {
315  using Slot = InternedSlot<T>;
316  llvm::DenseSet<Slot> interned;
317  llvm::BumpPtrAllocator &allocator;
318 
319 public:
320  InternedAllocator(llvm::BumpPtrAllocator &allocator) : allocator(allocator) {}
321 
322  /// Allocate a new object if it does not yet exist, or return a pointer to the
323  /// existing one. `R` is the type of the object to be allocated. `R` must be
324  /// derived from or be the type `T`.
325  template <typename R = T, typename... Args>
326  std::pair<R *, bool> alloc(Args &&...args) {
327  auto stack_value = R(std::forward<Args>(args)...);
328  auto stack_slot = Slot(&stack_value);
329  auto it = interned.find(stack_slot);
330  if (it != interned.end())
331  return std::make_pair(static_cast<R *>(it->ptr), false);
332  auto heap_value = new (allocator) R(std::move(stack_value));
333  interned.insert(Slot(heap_value));
334  return std::make_pair(heap_value, true);
335  }
336 };

362 /// A canonicalized linear inequality that maps a constraint on var `x` to the
363 /// linear inequality `x >= max(a*x+b, c) + (failed ? ∞ : 0)`.
364 ///
365 /// The inequality separately tracks recursive (a, b) and non-recursive (c)
366 /// constraints on `x`. This allows it to properly identify the combination of
367 /// the two constraints constraints `x >= x-1` and `x >= 4` to be satisfiable as
368 /// `x >= max(x-1, 4)`. If it only tracked inequality as `x >= a*x+b`, the
369 /// combination of these two constraints would be `x >= x+4` (due to max(-1,4) =
370 /// 4), which would be unsatisfiable.
371 ///
372 /// The `failed` flag acts as an additional `∞` term that renders the inequality
373 /// unsatisfiable. It is used as a tombstone value in case an operation renders
374 /// the equality unsatisfiable (e.g. `x >= 2**x` would be represented as the
375 /// inequality `x >= ∞`).
376 ///
377 /// Inequalities represented in this form can easily be checked for
378 /// unsatisfiability in the presence of recursion by inspecting the coefficients
379 /// a and b. The `sat` function performs this action.
380 struct LinIneq {
381  // x >= max(a*x+b, c) + (failed ? ∞ : 0)
382  int32_t rec_scale = 0; // a
383  int32_t rec_bias = 0; // b
384  int32_t nonrec_bias = 0; // c
385  bool failed = false;
386 
387  /// Create a new unsatisfiable inequality `x >= ∞`.
388  static LinIneq unsat() { return LinIneq(true); }
389 
390  /// Create a new inequality `x >= (failed ? ∞ : 0)`.
391  explicit LinIneq(bool failed = false) : failed(failed) {}
392 
393  /// Create a new inequality `x >= bias`.
394  explicit LinIneq(int32_t bias) : nonrec_bias(bias) {}
395 
396  /// Create a new inequality `x >= scale*x+bias`.
397  explicit LinIneq(int32_t scale, int32_t bias) {
398  if (scale != 0) {
399  rec_scale = scale;
400  rec_bias = bias;
401  } else {
402  nonrec_bias = bias;
403  }
404  }
405 
406  /// Create a new inequality `x >= max(rec_scale*x+rec_bias, nonrec_bias) +
407  /// (failed ? ∞ : 0)`.
408  explicit LinIneq(int32_t rec_scale, int32_t rec_bias, int32_t nonrec_bias,
409  bool failed = false)
410  : failed(failed) {
411  if (rec_scale != 0) {
412  this->rec_scale = rec_scale;
413  this->rec_bias = rec_bias;
414  this->nonrec_bias = nonrec_bias;
415  } else {
416  this->nonrec_bias = std::max(rec_bias, nonrec_bias);
417  }
418  }
419 
420  /// Combine two inequalities by taking the maxima of corresponding
421  /// coefficients.
422  ///
423  /// This essentially combines `x >= max(a1*x+b1, c1)` and `x >= max(a2*x+b2,
424  /// c2)` into a new `x >= max(max(a1,a2)*x+max(b1,b2), max(c1,c2))`. This is
425  /// a pessimistic upper bound, since e.g. `x >= 2x-10` and `x >= x-5` may both
426  /// hold, but the resulting `x >= 2x-5` may pessimistically not hold.
427  static LinIneq max(const LinIneq &lhs, const LinIneq &rhs) {
428  return LinIneq(std::max(lhs.rec_scale, rhs.rec_scale),
429  std::max(lhs.rec_bias, rhs.rec_bias),
430  std::max(lhs.nonrec_bias, rhs.nonrec_bias),
431  lhs.failed || rhs.failed);
432  }
433 
434  /// Combine two inequalities by summing up the two right hand sides.
435  ///
436  /// This is a tricky one, since the addition of the two max terms will lead to
437  /// a maximum over four possible terms (similar to a binomial expansion). In
438  /// order to shoehorn this back into a two-term maximum, we have to pick the
439  /// recursive term that will grow the fastest.
440  ///
441  /// As an example for this problem, consider the following addition:
442  ///
443  /// x >= max(a1*x+b1, c1) + max(a2*x+b2, c2)
444  ///
445  /// We would like to expand and rearrange this again into a maximum:
446  ///
447  /// x >= max(a1*x+b1 + max(a2*x+b2, c2), c1 + max(a2*x+b2, c2))
448  /// x >= max(max(a1*x+b1 + a2*x+b2, a1*x+b1 + c2),
449  /// max(c1 + a2*x+b2, c1 + c2))
450  /// x >= max((a1+a2)*x+(b1+b2), a1*x+(b1+c2), a2*x+(b2+c1), c1+c2)
451  ///
452  /// Since we are combining two two-term maxima, there are four possible ways
453  /// how the terms can combine, leading to the above four-term maximum. An easy
454  /// upper bound of the form we want would be the following:
455  ///
456  /// x >= max(max(a1+a2, a1, a2)*x + max(b1+b2, b1+c2, b2+c1), c1+c2)
457  ///
458  /// However, this is a very pessimistic upper-bound that will declare very
459  /// common patterns in the IR as unbreakable cycles, despite them being very
460  /// much breakable. For example:
461  ///
462  /// x >= max(x, 42) + max(0, -3) <-- breakable recursion
463  /// x >= max(max(1+0, 1, 0)*x + max(42+0, -3, 42), 42-2)
464  /// x >= max(x + 42, 39) <-- unbreakable recursion!
465  ///
466  /// A better approach is to take the expanded four-term maximum, retain the
467  /// non-recursive term (c1+c2), and estimate which one of the recursive terms
468  /// (first three) will become dominant as we choose greater values for x.
469  /// Since x never is inferred to be negative, the recursive term in the
470  /// maximum with the highest scaling factor for x will end up dominating as
471  /// x tends to ∞:
472  ///
473  /// x >= max({
474  /// (a1+a2)*x+(b1+b2) if a1+a2 >= max(a1+a2, a1, a2) and a1>0 and a2>0,
475  /// a1*x+(b1+c2) if a1 >= max(a1+a2, a1, a2) and a1>0,
476  /// a2*x+(b2+c1) if a2 >= max(a1+a2, a1, a2) and a2>0,
477  /// 0 otherwise
478  /// }, c1+c2)
479  ///
480  /// In case multiple cases apply, the highest bias of the recursive term is
481  /// picked. With this, the above problematic example triggers the second case
482  /// and becomes:
483  ///
484  /// x >= max(1*x+(0-3), 42-3) = max(x-3, 39)
485  ///
486  /// Of which the first case is chosen, as it has the lower bias value.
487  static LinIneq add(const LinIneq &lhs, const LinIneq &rhs) {
488  // Determine the maximum scaling factor among the three possible recursive
489  // terms.
490  auto enable1 = lhs.rec_scale > 0 && rhs.rec_scale > 0;
491  auto enable2 = lhs.rec_scale > 0;
492  auto enable3 = rhs.rec_scale > 0;
493  auto scale1 = lhs.rec_scale + rhs.rec_scale; // (a1+a2)
494  auto scale2 = lhs.rec_scale; // a1
495  auto scale3 = rhs.rec_scale; // a2
496  auto bias1 = lhs.rec_bias + rhs.rec_bias; // (b1+b2)
497  auto bias2 = lhs.rec_bias + rhs.nonrec_bias; // (b1+c2)
498  auto bias3 = rhs.rec_bias + lhs.nonrec_bias; // (b2+c1)
499  auto maxScale = std::max(scale1, std::max(scale2, scale3));
500 
501  // Among those terms that have a maximum scaling factor, determine the
502  // largest bias value.
503  Optional<int32_t> maxBias = llvm::None;
504  if (enable1 && scale1 == maxScale)
505  maxBias = bias1;
506  if (enable2 && scale2 == maxScale && (!maxBias || bias2 > *maxBias))
507  maxBias = bias2;
508  if (enable3 && scale3 == maxScale && (!maxBias || bias3 > *maxBias))
509  maxBias = bias3;
510 
511  // Pick from the recursive terms the one with maximum scaling factor and
512  // minimum bias value.
513  auto nonrec_bias = lhs.nonrec_bias + rhs.nonrec_bias; // c1+c2
514  auto failed = lhs.failed || rhs.failed;
515  if (enable1 && scale1 == maxScale && bias1 == *maxBias)
516  return LinIneq(scale1, bias1, nonrec_bias, failed);
517  if (enable2 && scale2 == maxScale && bias2 == *maxBias)
518  return LinIneq(scale2, bias2, nonrec_bias, failed);
519  if (enable3 && scale3 == maxScale && bias3 == *maxBias)
520  return LinIneq(scale3, bias3, nonrec_bias, failed);
521  return LinIneq(0, 0, nonrec_bias, failed);
522  }
523 
524  /// Check if the inequality is satisfiable.
525  ///
526  /// The inequality becomes unsatisfiable if the RHS is ∞, or a>1, or a==1 and
527  /// b <= 0. Otherwise there exists as solution for `x` that satisfies the
528  /// inequality.
529  bool sat() const {
530  if (failed)
531  return false;
532  if (rec_scale > 1)
533  return false;
534  if (rec_scale == 1 && rec_bias > 0)
535  return false;
536  return true;
537  }

539  /// Dump the inequality in human-readable form.
540  void print(llvm::raw_ostream &os) const {
541  bool any = false;
542  bool both = (rec_scale != 0 || rec_bias != 0) && nonrec_bias != 0;
543  os << "x >= ";
544  if (both)
545  os << "max(";
546  if (rec_scale != 0) {
547  any = true;
548  if (rec_scale != 1)
549  os << rec_scale << "*";
550  os << "x";
551  }
552  if (rec_bias != 0) {
553  if (any) {
554  if (rec_bias < 0)
555  os << " - " << -rec_bias;
556  else
557  os << " + " << rec_bias;
558  } else {
559  any = true;
560  os << rec_bias;
561  }
562  }
563  if (both)
564  os << ", ";
565  if (nonrec_bias != 0) {
566  any = true;
567  os << nonrec_bias;
568  }
569  if (both)
570  os << ")";
571  if (failed) {
572  if (any)
573  os << " + ";

575  }
576  if (!any)
577  os << "0";
578  }

689 /// Compute the canonicalized linear inequality expression starting at `expr`,
690 /// for the `var` as the left hand side `x` of the inequality. `seenVars` is
691 /// used as a recursion breaker. Occurrences of `var` itself within the
692 /// expression are mapped to the `a` coefficient in the inequality. Any other
693 /// variables are substituted and, in the presence of a recursion in a variable
694 /// other than `var`, treated as zero. `info` is a mapping from constraint
695 /// expressions to values and operations that produced the expression, and is
696 /// used during error reporting. If `reportInto` is present, the function will
697 /// additionally attach unsatisfiable inequalities as notes to the diagnostic as
698 /// it encounters them.
699 LinIneq ConstraintSolver::checkCycles(VarExpr *var, Expr *expr,
700  SmallPtrSetImpl<Expr *> &seenVars,
701  InFlightDiagnostic *reportInto,
702  unsigned indent) {
703  auto ineq =
704  TypeSwitch<Expr *, LinIneq>(expr)
705  .Case<KnownExpr>([&](auto *expr) { return LinIneq(*expr->solution); })
706  .Case<VarExpr>([&](auto *expr) {
707  if (expr == var)
708  return LinIneq(1, 0); // x >= 1*x + 0
709  if (!seenVars.insert(expr).second)
710  // Count recursions in other variables as 0. This is sane
711  // since the cycle is either breakable, in which case the
712  // recursion does not modify the resulting value of the
713  // variable, or it is not breakable and will be caught by
714  // this very function once it is called on that variable.
715  return LinIneq(0);
716  if (!expr->constraint)
717  // Count unconstrained variables as `x >= 0`.
718  return LinIneq(0);
719  auto l = checkCycles(var, expr->constraint, seenVars, reportInto,
720  indent + 1);
721  seenVars.erase(expr);
722  return l;
723  })
724  .Case<IdExpr>([&](auto *expr) {
725  return checkCycles(var, expr->arg, seenVars, reportInto,
726  indent + 1);
727  })
728  .Case<PowExpr>([&](auto *expr) {
729  // If we can evaluate `2**arg` to a sensible constant, do
730  // so. This is the case if a == 0 and c < 31 such that 2**c is
731  // representable.
732  auto arg =
733  checkCycles(var, expr->arg, seenVars, reportInto, indent + 1);
734  if (arg.rec_scale != 0 || arg.nonrec_bias < 0 ||
735  arg.nonrec_bias >= 31)
736  return LinIneq::unsat();
737  return LinIneq(1 << arg.nonrec_bias); // x >= 2**arg
738  })
739  .Case<AddExpr>([&](auto *expr) {
740  return LinIneq::add(
741  checkCycles(var, expr->lhs(), seenVars, reportInto, indent + 1),
742  checkCycles(var, expr->rhs(), seenVars, reportInto,
743  indent + 1));
744  })
745  .Case<MaxExpr, MinExpr>([&](auto *expr) {
746  // Combine the inequalities of the LHS and RHS into a single overly
747  // pessimistic inequality. We treat `MinExpr` the same as `MaxExpr`,
748  // since `max(a,b)` is an upper bound to `min(a,b)`.
749  return LinIneq::max(
750  checkCycles(var, expr->lhs(), seenVars, reportInto, indent + 1),
751  checkCycles(var, expr->rhs(), seenVars, reportInto,
752  indent + 1));
753  })
754  .Default([](auto) { return LinIneq::unsat(); });
755 
756  // If we were passed an in-flight diagnostic and the current inequality is
757  // unsatisfiable, attach notes to the diagnostic indicating the values or
758  // operations that contributed to this part of the constraint expression.
759  if (reportInto && !ineq.sat()) {
760  auto report = [&](Location loc) {
761  auto &note = reportInto->attachNote(loc);
762  note << "constrained width W >= ";
763  if (ineq.rec_scale == -1)
764  note << "-";
765  if (ineq.rec_scale != 1)
766  note << ineq.rec_scale;
767  note << "W";
768  if (ineq.rec_bias < 0)
769  note << "-" << -ineq.rec_bias;
770  if (ineq.rec_bias > 0)
771  note << "+" << ineq.rec_bias;
772  note << " here:";
773  };
774  auto it = locs.find(expr);
775  if (it != locs.end())
776  for (auto loc : it->second)
777  report(loc);
778  }
779  if (!reportInto)
780  LLVM_DEBUG(llvm::dbgs().indent(indent * 2)
781  << "- Visited " << *expr << ": " << ineq << "\n");
782 
783  return ineq;
784 }

795 static ExprSolution
796 computeBinary(ExprSolution lhs, ExprSolution rhs,
797  llvm::function_ref<int32_t(int32_t, int32_t)> operation) {
798  auto result = ExprSolution{llvm::None, lhs.second || rhs.second};

808 /// Compute the value of a constraint expression`expr`. `seenVars` is used as a
809 /// recursion breaker. Recursive variables are treated as zero. Returns the
810 /// computed value and a boolean indicating whether a recursion was detected.
811 /// This may be used to memoize the result of expressions in case they were not
812 /// involved in a cycle (which may alter their value from the perspective of a
813 /// variable).
814 static ExprSolution solveExpr(Expr *expr, SmallPtrSetImpl<Expr *> &seenVars,
815  unsigned indent = 1) {
816  // See if we have a memoized result we can return.
817  if (expr->solution) {
818  LLVM_DEBUG({
819  if (!isa<KnownExpr>(expr))
820  llvm::dbgs().indent(indent * 2)
821  << "- Cached " << *expr << " = " << *expr->solution << "\n";
822  });
823  return {*expr->solution, false};
824  }
825 
826  // Otherwise compute the value of the expression.
827  LLVM_DEBUG({
828  if (!isa<KnownExpr>(expr))
829  llvm::dbgs().indent(indent * 2) << "- Solving " << *expr << "\n";
830  });
831  auto solution =
832  TypeSwitch<Expr *, ExprSolution>(expr)
833  .Case<KnownExpr>([&](auto *expr) {
834  return ExprSolution{*expr->solution, false};
835  })
836  .Case<VarExpr>([&](auto *expr) {
837  // Unconstrained variables produce no solution.
838  if (!expr->constraint)
839  return ExprSolution{llvm::None, false};
840  // Return no solution for recursions in the variables. This is sane
841  // and will cause the expression to be ignored when computing the
842  // parent, e.g. `a >= max(a, 1)` will become just `a >= 1`.
843  if (!seenVars.insert(expr).second)
844  return ExprSolution{llvm::None, true};
845  auto solution = solveExpr(expr->constraint, seenVars, indent + 1);
846  seenVars.erase(expr);
847  // Constrain variables >= 0.
848  if (solution.first && *solution.first < 0)
849  solution.first = 0;
850  return solution;
851  })
852  .Case<IdExpr>([&](auto *expr) {
853  return solveExpr(expr->arg, seenVars, indent + 1);
854  })
855  .Case<PowExpr>([&](auto *expr) {
856  auto arg = solveExpr(expr->arg, seenVars, indent + 1);
857  return computeUnary(arg, [](int32_t arg) { return 1 << arg; });
858  })
859  .Case<AddExpr>([&](auto *expr) {
860  auto lhs = solveExpr(expr->lhs(), seenVars, indent + 1);
861  auto rhs = solveExpr(expr->rhs(), seenVars, indent + 1);
862  return computeBinary(
863  lhs, rhs, [](int32_t lhs, int32_t rhs) { return lhs + rhs; });
864  })
865  .Case<MaxExpr>([&](auto *expr) {
866  auto lhs = solveExpr(expr->lhs(), seenVars, indent + 1);
867  auto rhs = solveExpr(expr->rhs(), seenVars, indent + 1);
868  return computeBinary(lhs, rhs, [](int32_t lhs, int32_t rhs) {
869  return std::max(lhs, rhs);
870  });
871  })
872  .Case<MinExpr>([&](auto *expr) {
873  auto lhs = solveExpr(expr->lhs(), seenVars, indent + 1);
874  auto rhs = solveExpr(expr->rhs(), seenVars, indent + 1);
875  return computeBinary(lhs, rhs, [](int32_t lhs, int32_t rhs) {
876  return std::min(lhs, rhs);
877  });
878  })
879  .Default([](auto) {
880  return ExprSolution{llvm::None, false};
881  });
882 
883  // Memoize the result.
884  if (solution.first && !solution.second)
885  expr->solution = *solution.first;
886 
887  // Produce some useful debug prints.
888  LLVM_DEBUG({
889  if (!isa<KnownExpr>(expr)) {
890  if (solution.first)
891  llvm::dbgs().indent(indent * 2)
892  << "= Solved " << *expr << " = " << *solution.first;
893  else
894  llvm::dbgs().indent(indent * 2) << "= Skipped " << *expr;
895  llvm::dbgs() << " (" << (solution.second ? "cycle broken" : "unique")
896  << ")\n";
897  }
898  });
899 
900  return solution;
901 }

903 /// Solve the constraint problem. This is a very simple implementation that
904 /// does not fully solve the problem if there are weird dependency cycles
905 /// present.
906 LogicalResult ConstraintSolver::solve() {
907  LLVM_DEBUG({
908  llvm::dbgs() << "\n===----- Constraints -----===\n\n";
909  dumpConstraints(llvm::dbgs());
910  });
911 
912  // Ensure that there are no adverse cycles around.
913  LLVM_DEBUG(
914  llvm::dbgs() << "\n===----- Checking for unbreakable loops -----===\n\n");
915  SmallPtrSet<Expr *, 16> seenVars;
916  bool anyFailed = false;
917 
918  for (auto *expr : exprs) {
919  // Only work on variables.
920  auto *var = dyn_cast<VarExpr>(expr);
921  if (!var || !var->constraint)
922  continue;
923  LLVM_DEBUG(llvm::dbgs()
924  << "- Checking " << *var << " >= " << *var->constraint << "\n");
925 
926  // Canonicalize the variable's constraint expression into a form that allows
927  // us to easily determine if any recursion leads to an unsatisfiable
928  // constraint. The `seenVars` set acts as a recursion breaker.
929  seenVars.insert(var);
930  auto ineq = checkCycles(var, var->constraint, seenVars);
931  seenVars.clear();
932 
933  // If the constraint is satisfiable, we're done.
934  // TODO: It's possible that this result is already sufficient to arrive at a
935  // solution for the constraint, and the second pass further down is not
936  // necessary. This would require more proper handling of `MinExpr` in the
937  // cycle checking code.
938  if (ineq.sat()) {
939  LLVM_DEBUG(llvm::dbgs()
940  << " = Breakable since " << ineq << " satisfiable\n");
941  continue;
942  }
943 
944  // If we arrive here, the constraint is not satisfiable at all. To provide
945  // some guidance to the user, we call the cycle checking code again, but
946  // this time with an in-flight diagnostic to attach notes indicating
947  // unsatisfiable paths in the cycle.
948  LLVM_DEBUG(llvm::dbgs()
949  << " = UNBREAKABLE since " << ineq << " unsatisfiable\n");
950  anyFailed = true;
951  for (auto fieldRef : info.find(var)->second) {
952  // Depending on whether this value stems from an operation or not, create
953  // an appropriate diagnostic identifying the value.
954  auto op = fieldRef.getDefiningOp();
955  auto diag = op ? op->emitOpError()
956  : mlir::emitError(fieldRef.getValue().getLoc())
957  << "value ";
958  diag << "is constrained to be wider than itself";
959 
960  // Re-run the cycle checking, but this time reporting into the diagnostic.
961  seenVars.insert(var);
962  checkCycles(var, var->constraint, seenVars, &diag);
963  seenVars.clear();
964  }
965  }
966 
967  // If there were cycles, return now to avoid complaining to the user about
968  // dependent widths not being inferred.
969  if (anyFailed)
970  return failure();
971 
972  // Iterate over the constraint variables and solve each.
973  LLVM_DEBUG(llvm::dbgs() << "\n===----- Solving constraints -----===\n\n");
974  for (auto *expr : exprs) {
975  // Only work on variables.
976  auto *var = dyn_cast<VarExpr>(expr);
977  if (!var)
978  continue;
979 
980  // Complain about unconstrained variables.
981  if (!var->constraint) {
982  LLVM_DEBUG(llvm::dbgs() << "- Unconstrained " << *var << "\n");
983  if (emitUninferredWidthError(var))
984  anyFailed = true;
985  continue;
986  }
987 
988  // Compute the value for the variable.
989  LLVM_DEBUG(llvm::dbgs()
990  << "- Solving " << *var << " >= " << *var->constraint << "\n");
991  seenVars.insert(var);
992  auto solution = solveExpr(var->constraint, seenVars);
993  seenVars.clear();
994 
995  // Constrain variables >= 0.
996  if (solution.first && *solution.first < 0)
997  solution.first = 0;
998 
999  // In case the width could not be inferred, complain to the user. This might
1000  // be the case if the width depends on an unconstrained variable.
1001  if (!solution.first) {
1002  LLVM_DEBUG(llvm::dbgs() << " - UNSOLVED " << *var << "\n");
1003  if (emitUninferredWidthError(var))
1004  anyFailed = true;
1005  } else {
1006  LLVM_DEBUG(llvm::dbgs()
1007  << " - Solved " << *var << " = " << solution.first << " ("
1008  << (solution.second ? "cycle broken" : "unique") << ")\n");
1009  }
1010  var->solution = solution.first;
1011  }
1012 
1013  return failure(anyFailed);
1014 }

1152 LogicalResult InferenceMapping::map(CircuitOp op) {
1153  LLVM_DEBUG(llvm::dbgs()
1154  << "\n===----- Mapping ops to constraint exprs -----===\n\n");
1155 
1156  // Ensure we have constraint variables established for all module ports.
1157  op.walk<WalkOrder::PostOrder>([&](FModuleOp module) {
1158  for (auto arg : module.getArguments()) {
1159  solver.setCurrentContextInfo(FieldRef(arg, 0));
1160  declareVars(arg, module.getLoc());
1161  }
1162  return WalkResult::skip(); // no need to look inside the module
1163  });
1164 
1165  // Go through the module bodies and populate the constraint problem.
1166  auto result = op.walk<WalkOrder::PostOrder>([&](FModuleOp module) {
1167  // Check if the module contains *any* uninferred widths. This allows us to
1168  // do an early skip if the module is already fully inferred.
1169  bool anyUninferred = false;
1170  for (auto arg : module.getArguments()) {
1171  anyUninferred |= hasUninferredWidth(arg.getType());
1172  if (anyUninferred)
1173  break;
1174  }
1175  module.walk([&](Operation *op) {
1176  for (auto type : op->getResultTypes())
1177  anyUninferred |= hasUninferredWidth(type);
1178  if (anyUninferred)
1179  return WalkResult::interrupt();
1180  return WalkResult::advance();
1181  });
1182  if (!anyUninferred) {
1183  LLVM_DEBUG(llvm::dbgs() << "Skipping fully-inferred module '"
1184  << module.getName() << "'\n");
1185  skippedModules.insert(module);
1186  return WalkResult::skip();
1187  }
1188  allModulesSkipped = false;
1189 
1190  // Go through operations in the module, creating type variables for results,
1191  // and generating constraints.
1192  auto result = module.getBody()->walk(
1193  [&](Operation *op) { return WalkResult(mapOperation(op)); });
1194  if (result.wasInterrupted())
1195  return WalkResult::interrupt();
1196  return WalkResult::skip(); // walk above already visited module body
1197  });
1198  return failure(result.wasInterrupted());
1199 }

1201 LogicalResult InferenceMapping::mapOperation(Operation *op) {
1202  // In case the operation result has a type without uninferred widths, don't
1203  // even bother to populate the constraint problem and treat that as a known
1204  // size directly. This is done in `declareVars`, which will generate
1205  // `KnownExpr` nodes for all known widths -- which are the only ones in this
1206  // case.
1207  bool allWidthsKnown = true;
1208  for (auto result : op->getResults()) {
1209  if (auto mux = dyn_cast<MuxPrimOp>(op))
1210  if (hasUninferredWidth(mux.sel().getType()))
1211  allWidthsKnown = false;
1212  if (!hasUninferredWidth(result.getType()))
1213  declareVars(result, op->getLoc());
1214  else
1215  allWidthsKnown = false;
1216  }
1217  if (allWidthsKnown && !isa<ConnectOp, StrictConnectOp, AttachOp>(op))
1218  return success();
1219 
1220  // Actually generate the necessary constraint expressions.
1221  bool mappingFailed = false;
1222  solver.setCurrentContextInfo(
1223  op->getNumResults() > 0 ? FieldRef(op->getResults()[0], 0) : FieldRef());
1224  solver.setCurrentLocation(op->getLoc());
1225  TypeSwitch<Operation *>(op)
1226  .Case<ConstantOp>([&](auto op) {
1227  // If the constant has a known width, use that. Otherwise pick the
1228  // smallest number of bits necessary to represent the constant.
1229  Expr *e;
1230  if (auto width = op.getType().getWidth())
1231  e = solver.known(*width);
1232  else {
1233  auto v = op.value();
1234  auto w = v.getBitWidth() - (v.isNegative() ? v.countLeadingOnes()
1235  : v.countLeadingZeros());
1236  if (v.isSigned())
1237  w += 1;
1238  e = solver.known(std::max(w, 1u));
1239  }
1240  setExpr(op.getResult(), e);
1241  })
1242  .Case<SpecialConstantOp>([&](auto op) {
1243  // Nothing required.
1244  })
1245  .Case<InvalidValueOp>([&](auto op) {
1246  // We must duplicate the invalid value for each use, since each use can
1247  // be inferred to a different width.
1248  if (!hasUninferredWidth(op.getType()) || op->use_empty())
1249  return;
1250 
1251  auto type = op.getType();
1252  ImplicitLocOpBuilder builder(op->getLoc(), op);
1253  for (auto &use :
1254  llvm::make_early_inc_range(llvm::drop_begin(op->getUses()))) {
1255  // - `make_early_inc_range` since `getUses()` is invalidated upon
1256  // `use.set(...)`.
1257  // - `drop_begin` such that the first use can keep the original op.
1258  use.set(builder.create<InvalidValueOp>(type));
1259  }
1260  })
1261  .Case<WireOp, RegOp>(

1263  .Case<RegResetOp>([&](auto op) {
1264  // The original Scala code also constrains the reset signal to be at
1265  // least 1 bit wide. We don't do this here since the MLIR FIRRTL
1266  // dialect enforces the reset signal to be an async reset or a
1267  // `uint<1>`.
1268  declareVars(op.getResult(), op.getLoc());

1270  // signal.
1271  constrainTypes(op.getResult(), op.resetValue());
1272  })

1332  auto rhs = getExpr(op.rhs());

1475  switch (kind) {
1476  case MemOp::PortKind::Read:
1477  case MemOp::PortKind::Write:
1478  return ArrayRef<unsigned>(indices, 1); // {3}

1480  return ArrayRef<unsigned>(indices); // {3, 5}

1486  // actually want is for all data ports to share the same variable. To do
1487  // this, we find the first data port declared, and use that port's vars
1488  // for all the other ports.
1489  unsigned firstFieldIndex = dataFieldIndices(op.getPortKind(0))[0];
1490  FieldRef firstData(op.getResult(0), op.getPortType(0)
1491  .getPassiveType()
1492  .template cast<BundleType>()
1493  .getFieldID(firstFieldIndex));

1501  op.getPortType(i).getPassiveType().template cast<BundleType>();
1502  for (auto fieldIndex : dataFieldIndices(op.getPortKind(i)))
1503  unifyTypes(FieldRef(result, portType.getFieldID(fieldIndex)),
1504  firstData, dataType);
1505  }
1506  })

1516 /// Declare free variables for the type of a value, and associate the resulting
1517 /// set of variables with that value.
1518 void InferenceMapping::declareVars(Value value, Location loc) {
1519  auto ftype = value.getType().cast<FIRRTLType>();
1520 
1521  // Declare a variable for every unknown width in the type. If this is a Bundle
1522  // type or a FVector type, we will have to potentially create many variables.
1523  unsigned fieldID = 0;
1524  std::function<void(FIRRTLType)> declare = [&](FIRRTLType type) {
1525  auto width = type.getBitWidthOrSentinel();
1526  if (width >= 0) {
1527  // Known width integer create a known expression.
1528  setExpr(FieldRef(value, fieldID), solver.known(width));
1529  fieldID++;
1530  } else if (width == -1) {
1531  // Unknown width integers create a variable.
1532  FieldRef field(value, fieldID);
1533  solver.setCurrentContextInfo(field);
1534  setExpr(field, solver.var());
1535  fieldID++;
1536  } else if (auto bundleType = type.dyn_cast<BundleType>()) {
1537  // Bundle types recursively declare all bundle elements.
1538  fieldID++;
1539  for (auto &element : bundleType) {
1540  declare(element.type);
1541  }
1542  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
1543  fieldID++;
1544  auto save = fieldID;
1545  declare(vecType.getElementType());
1546  // Skip past the rest of the elements
1547  fieldID = save + vecType.getMaxFieldID();
1548  } else {
1549  llvm_unreachable("Unknown type inside a bundle!");
1550  }
1551  };
1552  declare(ftype);
1553 }

1555 /// Assign the constraint expressions of the fields in the `result` argument as
1556 /// the max of expressions in the `rhs` and `lhs` arguments. Both fields must be
1557 /// the same type.
1558 void InferenceMapping::maximumOfTypes(Value result, Value rhs, Value lhs) {
1559  // Recurse to every leaf element and set larger >= smaller.
1560  auto fieldID = 0;
1561  std::function<void(FIRRTLType)> maximize = [&](FIRRTLType type) {
1562  if (auto bundleType = type.dyn_cast<BundleType>()) {
1563  fieldID++;
1564  for (auto &element : bundleType.getElements())
1565  maximize(element.type);
1566  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
1567  fieldID++;
1568  auto save = fieldID;
1569  // Skip 0 length vectors.
1570  if (vecType.getNumElements() > 0)
1571  maximize(vecType.getElementType());
1572  fieldID = save + vecType.getMaxFieldID();
1573  } else if (type.isGround()) {
1574  auto *e = solver.max(getExpr(FieldRef(rhs, fieldID)),
1575  getExpr(FieldRef(lhs, fieldID)));
1576  setExpr(FieldRef(result, fieldID), e);
1577  fieldID++;
1578  } else {
1579  llvm_unreachable("Unknown type inside a bundle!");
1580  }
1581  };
1582  auto type = result.getType().cast<FIRRTLType>();
1583  maximize(type);
1584 }

1586 /// Establishes constraints to ensure the sizes in the `larger` type are greater
1587 /// than or equal to the sizes in the `smaller` type. Types have to be
1588 /// compatible in the sense that they may only differ in the presence or absence
1589 /// of bit widths.
1590 ///
1591 /// This function is used to apply regular connects.
1592 void InferenceMapping::constrainTypes(Value larger, Value smaller) {
1593  // Recurse to every leaf element and set larger >= smaller.
1594  auto type = larger.getType().cast<FIRRTLType>();
1595  auto fieldID = 0;
1596  std::function<void(FIRRTLType, Value, Value)> constrain =
1597  [&](FIRRTLType type, Value larger, Value smaller) {
1598  if (auto bundleType = type.dyn_cast<BundleType>()) {
1599  fieldID++;
1600  for (auto &element : bundleType.getElements()) {
1601  if (element.isFlip)
1602  constrain(element.type, smaller, larger);
1603  else
1604  constrain(element.type, larger, smaller);
1605  }
1606  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
1607  fieldID++;
1608  auto save = fieldID;
1609  // Skip 0 length vectors.
1610  if (vecType.getNumElements() > 0) {
1611  constrain(vecType.getElementType(), larger, smaller);
1612  }
1613  fieldID = save + vecType.getMaxFieldID();
1614  } else if (type.isGround()) {
1615  // Leaf element, look up their expressions, and create the constraint.
1616  constrainTypes(getExpr(FieldRef(larger, fieldID)),
1617  getExpr(FieldRef(smaller, fieldID)));
1618  fieldID++;
1619  } else {
1620  llvm_unreachable("Unknown type inside a bundle!");
1621  }
1622  };
1623 
1624  constrain(type, larger, smaller);
1625 }

1628 /// than or equal to the sizes in the `smaller` type.
1629 void InferenceMapping::constrainTypes(Expr *larger, Expr *smaller) {
1630  assert(larger && "Larger expression should be specified");
1631  assert(smaller && "Smaller expression should be specified");
1632  // Mimic the Scala implementation here by simply doing nothing if the larger
1633  // expr is not a free variable. Apparently there are many cases where
1634  // useless constraints can be added, e.g. on multiple well-known values. As
1635  // long as we don't want to do type checking itself here, but only width
1636  // inference, we should be fine ignoring expr we cannot constraint anyway.
1637  if (auto largerVar = dyn_cast<VarExpr>(larger)) {
1638  LLVM_ATTRIBUTE_UNUSED auto c = solver.addGeqConstraint(largerVar, smaller);
1639  LLVM_DEBUG(llvm::dbgs()
1640  << "Constrained " << *largerVar << " >= " << *c << "\n");
1641  }

1644 /// Assign the constraint expressions of the fields in the `src` argument as the
1645 /// expressions for the `dst` argument. Both fields must be of the given `type`.
1646 void InferenceMapping::unifyTypes(FieldRef lhs, FieldRef rhs, FIRRTLType type) {
1647  // Fast path for `unifyTypes(x, x, _)`.
1648  if (lhs == rhs)
1649  return;
1650 
1651  // Co-iterate the two field refs, recurring into every leaf element and set
1652  // them equal.
1653  auto fieldID = 0;
1654  std::function<void(FIRRTLType)> unify = [&](FIRRTLType type) {
1655  if (type.isGround()) {
1656  // Leaf element, unify the fields!
1657  FieldRef lhsFieldRef(lhs.getValue(), lhs.getFieldID() + fieldID);
1658  FieldRef rhsFieldRef(rhs.getValue(), rhs.getFieldID() + fieldID);
1659  LLVM_DEBUG(llvm::dbgs() << "Unify " << getFieldName(lhsFieldRef) << " = "
1660  << getFieldName(rhsFieldRef) << "\n");
1661  // Abandon variables becoming unconstrainable by the unification.
1662  if (auto *var = dyn_cast_or_null<VarExpr>(getExprOrNull(lhsFieldRef)))
1663  solver.addGeqConstraint(var, solver.known(0));
1664  setExpr(lhsFieldRef, getExpr(rhsFieldRef));
1665  fieldID++;
1666  } else if (auto bundleType = type.dyn_cast<BundleType>()) {
1667  fieldID++;
1668  for (auto &element : bundleType) {
1669  unify(element.type);
1670  }
1671  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
1672  fieldID++;
1673  auto save = fieldID;
1674  // Skip 0 length vectors.
1675  if (vecType.getNumElements() > 0) {
1676  unify(vecType.getElementType());
1677  }
1678  fieldID = save + vecType.getMaxFieldID();
1679  } else {
1680  llvm_unreachable("Unknown type inside a bundle!");
1681  }
1682  };
1683  unify(type);
1684 }

1762 /// Update the result types of an operation.
1763 bool InferenceTypeUpdate::updateOperation(Operation *op) {
1764  bool anyChanged = false;
1765 
1766  // Invalid value operations get their value from a connect which uses them. We
1767  // set the width when we see the connect op later on, so that the destination
1768  // operand width will have definitely been mapped in.
1769  if (isa<InvalidValueOp>(op) &&
1770  hasUninferredWidth(op->getResultTypes().front())) {
1771  if (op->use_empty() ||
1772  !isa<ConnectOp, StrictConnectOp>(*op->getUsers().begin())) {
1773  auto diag = mlir::emitError(
1774  op->getLoc(), "uninferred width: invalid value is unconstrained");
1775  anyFailed = true;
1776  }
1777  return false;
1778  }
1779 
1780  for (Value v : op->getResults()) {
1781  anyChanged |= updateValue(v);
1782  if (anyFailed)
1783  return false;
1784  }
1785 
1786  // If this is a connect operation, width inference might have inferred a RHS
1787  // that is wider than the LHS, in which case an additional BitsPrimOp is
1788  // necessary to truncate the value.
1789  if (auto con = dyn_cast<ConnectOp>(op)) {
1790  auto lhs = con.dest();
1791  auto rhs = con.src();
1792  auto lhsType = lhs.getType().cast<FIRRTLType>();
1793  auto rhsType = rhs.getType().cast<FIRRTLType>();
1794 
1795  // If the source is an InvalidValue of unknown width, infer the type to be
1796  // the same as the destination.
1797  if (dyn_cast_or_null<InvalidValueOp>(rhs.getDefiningOp()) &&
1798  hasUninferredWidth(rhs.getType()))
1799  rhs.setType(lhsType);
1800 
1801  auto lhsWidth = lhsType.getBitWidthOrSentinel();
1802  auto rhsWidth = rhsType.getBitWidthOrSentinel();
1803  if (lhsWidth >= 0 && rhsWidth >= 0 && lhsWidth < rhsWidth) {
1804  OpBuilder builder(op);
1805  auto trunc = builder.createOrFold<TailPrimOp>(con.getLoc(), con.src(),
1806  rhsWidth - lhsWidth);
1807  if (rhsType.isa<SIntType>())
1808  trunc =
1809  builder.createOrFold<AsSIntPrimOp>(con.getLoc(), lhsType, trunc);
1810 
1811  LLVM_DEBUG(llvm::dbgs()
1812  << "Truncating RHS to " << lhsType << " in " << con << "\n");
1813  con->replaceUsesOfWith(con.src(), trunc);
1814  }
1815  }
1816 
1817  // If this is a module, update its ports.
1818  if (auto module = dyn_cast<FModuleOp>(op)) {
1819  // Update the block argument types.
1820  bool argsChanged = false;
1821  SmallVector<Attribute> argTypes;
1822  argTypes.reserve(module.getNumPorts());
1823  for (auto arg : module.getArguments()) {
1824  argsChanged |= updateValue(arg);
1825  argTypes.push_back(TypeAttr::get(arg.getType()));
1826  if (anyFailed)
1827  return false;
1828  }
1829 
1830  // Update the module function type if needed.
1831  if (argsChanged) {
1832  module->setAttr(FModuleLike::getPortTypesAttrName(),
1833  ArrayAttr::get(module.getContext(), argTypes));
1834  anyChanged = true;
1835  }
1836  }
1837  return anyChanged;
1838 }

1853 /// Update the type of a value.
1854 bool InferenceTypeUpdate::updateValue(Value value) {
1855  // Check if the value has a type which we can update.
1856  auto type = value.getType().dyn_cast<FIRRTLType>();
1857  if (!type)
1858  return false;
1859 
1860  // Fast path for types that have fully inferred widths.
1861  if (!hasUninferredWidth(type))
1862  return false;
1863 
1864  // If this is an operation that does not generate any free variables that
1865  // are determined during width inference, simply update the value type based
1866  // on the operation arguments.
1867  if (auto op = dyn_cast_or_null<InferTypeOpInterface>(value.getDefiningOp())) {
1868  SmallVector<Type, 2> types;
1869  auto res =
1870  op.inferReturnTypes(op->getContext(), op->getLoc(), op->getOperands(),
1871  op->getAttrDictionary(), op->getRegions(), types);
1872  if (failed(res)) {
1873  anyFailed = true;
1874  return false;
1875  }
1876  assert(types.size() == op->getNumResults());
1877  for (auto it : llvm::zip(op->getResults(), types)) {
1878  LLVM_DEBUG(llvm::dbgs() << "Inferring " << std::get<0>(it) << " as "
1879  << std::get<1>(it) << "\n");
1880  std::get<0>(it).setType(std::get<1>(it));
1881  }
1882  return true;
1883  }
1884 
1885  // Recreate the type, substituting the solved widths.
1886  auto context = type.getContext();
1887  unsigned fieldID = 0;
1888  std::function<FIRRTLType(FIRRTLType)> update = [&](FIRRTLType type) {
1889  auto width = type.getBitWidthOrSentinel();
1890  if (width >= 0) {
1891  // Known width integers return themselves.
1892  fieldID++;
1893  return type;
1894  } else if (width == -1) {
1895  // Unknown width integers return the solved type.
1896  auto newType = updateType(FieldRef(value, fieldID), type);
1897  fieldID++;
1898  return newType;
1899  } else if (auto bundleType = type.dyn_cast<BundleType>()) {
1900  // Bundle types recursively update all bundle elements.
1901  fieldID++;
1902  llvm::SmallVector<BundleType::BundleElement, 3> elements;
1903  for (auto &element : bundleType) {
1904  elements.emplace_back(element.name, element.isFlip,
1905  update(element.type));
1906  }
1907  return BundleType::get(elements, context);
1908  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
1909  fieldID++;
1910  auto save = fieldID;
1911  // TODO: this should recurse into the element type of 0 length vectors and
1912  // set any unknown width to 0.
1913  if (vecType.getNumElements() > 0) {
1914  auto newType = FVectorType::get(update(vecType.getElementType()),
1915  vecType.getNumElements());
1916  fieldID = save + vecType.getMaxFieldID();
1917  return newType;
1918  }
1919  // If this is a 0 length vector return the original type.
1920  return type;
1921  }
1922  llvm_unreachable("Unknown type inside a bundle!");
1923  };
1924 
1925  // Update the type.
1926  auto newType = update(type);
1927  LLVM_DEBUG(llvm::dbgs() << "Update " << value << " to " << newType << "\n");
1928  value.setType(newType);
1929 
1930  // If this is a ConstantOp, adjust the width of the underlying APInt.
1931  // Unsized constants have APInts which are *at least* wide enough to hold
1932  // the value, but may be larger. This can trip up the verifier.
1933  if (auto op = value.getDefiningOp<ConstantOp>()) {
1934  auto k = op.value();
1935  auto bitwidth = op.getType().cast<FIRRTLType>().getBitWidthOrSentinel();
1936  if (k.getBitWidth() > unsigned(bitwidth))
1937  k = k.trunc(bitwidth);
1938  op->setAttr("value", IntegerAttr::get(op.getContext(), k));
1939  }
1940 
1941  return newType != type;
1942 }

1944 /// Update a type.
1945 FIRRTLType InferenceTypeUpdate::updateType(FieldRef fieldRef, FIRRTLType type) {
1946  assert(type.isGround() && "Can only pass in ground types.");
1947  auto value = fieldRef.getValue();
1948  // Get the inferred width.
1949  Expr *expr = mapping.getExprOrNull(fieldRef);
1950  if (!expr || !expr->solution.hasValue()) {
1951  // It should not be possible to arrive at an uninferred width at this point.
1952  // In case the constraints are not resolvable, checks before the calls to
1953  // `updateType` must have already caught the issues and aborted the pass
1954  // early. Might turn this into an assert later.
1955  anyFailed = true;
1956  mlir::emitError(value.getLoc(), "width should have been inferred");
1957  return type;
1958  }
1959  int32_t solution = expr->solution.getValue();
1960  assert(solution >= 0); // The solver infers variables to be 0 or greater.
1961  return resizeType(type, solution);
1962 }
