27 InnerSymbolTable::InnerSymbolTable(Operation *op) {
28  assert(op->hasTrait<OpTrait::InnerSymbolTable>() &&
29  "expected operation to have InnerSymbolTable trait");
30  // Save the operation this table is for.
31  this->innerSymTblOp = op;
32 
33  // Walk the operation and add InnerSymbol's to the table.
34  op->walk([&](InnerSymbolOpInterface symOp) {
35  auto attr = symOp.getInnerNameAttr();
36  if (!attr)
37  return;
38  auto it = symbolTable.insert({attr, symOp});
39  (void)it;
40  assert(it.second && "repeated symbol found");
41  });
42 }

79 void InnerSymbolTableCollection::populateTables(Operation *innerRefNSOp) {
80  // Gather top-level operations.
81  SmallVector<Operation *> childOps(
82  llvm::make_pointer_range(innerRefNSOp->getRegion(0).front()));
83 
84  // Filter these to those that have the InnerSymbolTable trait.
85  SmallVector<Operation *> innerSymTableOps(
86  llvm::make_filter_range(childOps, [&](Operation *op) {
87  return op->hasTrait<OpTrait::InnerSymbolTable>();
88  }));
89 
90  // Ensure entries exist for each operation.
91  llvm::for_each(innerSymTableOps,
92  [&](auto *op) { symbolTables.try_emplace(op, nullptr); });
93 
94  // Construct the tables in parallel (if context allows it).
95  mlir::parallelForEach(
96  innerRefNSOp->getContext(), innerSymTableOps, [&](auto *op) {
97  auto it = symbolTables.find(op);
98  assert(it != symbolTables.end());
99  if (!it->second)
100  it->second = ::std::make_unique<InnerSymbolTable>(op);
101  });
102 }

120 LogicalResult verifyInnerRefs(Operation *op) {
121  // Construct the symbol tables.
122  InnerSymbolTableCollection innerSymTables;
123  SymbolTable symbolTable(op);
124  InnerRefNamespace ns{symbolTable, innerSymTables};
125  innerSymTables.populateTables(op);
126 
127  // Conduct parallel walks of the top-level children of this
128  // InnerRefNamespace, verifying all InnerRefUserOp's discovered within.
129  auto verifySymbolUserFn = [&](Operation *op) -> WalkResult {
130  if (auto user = dyn_cast<InnerRefUserOpInterface>(op))
131  return WalkResult(user.verifyInnerRefs(ns));
132  return WalkResult::advance();
133  };
134  return mlir::failableParallelForEach(
135  op->getContext(), op->getRegion(0).front(), [&](auto &op) {
136  return success(!op.walk(verifySymbolUserFn).wasInterrupted());
137  });
138 }
