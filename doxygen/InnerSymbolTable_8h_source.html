25 /// A table of inner symbols and their resolutions.
26 class InnerSymbolTable {
27 public:
28  /// Build an inner symbol table for the given operation. The operation must
29  /// have the InnerSymbolTable trait.
30  explicit InnerSymbolTable(Operation *op);
31 
32  /// Non-copyable
33  InnerSymbolTable(const InnerSymbolTable &) = delete;
34  InnerSymbolTable &operator=(InnerSymbolTable &) = delete;
35 
36  /// Look up a symbol with the specified name, returning null if no such
37  /// name exists. Names never include the @ on them.
38  Operation *lookup(StringRef name) const;
39  template <typename T>
40  T lookup(StringRef name) const {
41  return dyn_cast_or_null<T>(lookup(name));
42  }
43 
44  /// Look up a symbol with the specified name, returning null if no such
45  /// name exists. Names never include the @ on them.
46  Operation *lookup(StringAttr name) const;
47  template <typename T>
48  T lookup(StringAttr name) const {
49  return dyn_cast_or_null<T>(lookup(name));
50  }
51 
52  /// Return an InnerRef to the given operation which must be within this table.
53  hw::InnerRefAttr getInnerRef(Operation *op);
54 
55  /// Return an InnerRef for the given inner symbol, which must be valid.
56  hw::InnerRefAttr getInnerRef(StringRef name) {
57  return getInnerRef(lookup(name));
58  }
59 
60  /// Return an InnerRef for the given inner symbol, which must be valid.
61  hw::InnerRefAttr getInnerRef(StringAttr name) {
62  return getInnerRef(lookup(name));
63  }
64 
65  /// Get InnerSymbol for an operation.
66  static StringAttr getInnerSymbol(Operation *op);
67 
68  /// Return the name of the attribute used for inner symbol names.
69  static StringRef getInnerSymbolAttrName() { return "inner_sym"; }
70 
71 private:
72  /// This is the operation this table is constructed for, which must have the
73  /// InnerSymbolTable trait.
74  Operation *innerSymTblOp;
75 
76  /// This maps names to operations with that inner symbol.
77  DenseMap<StringAttr, Operation *> symbolTable;
78 };
