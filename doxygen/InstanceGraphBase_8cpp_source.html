15 void InstanceRecord::erase() {
16  // Update the prev node to point to the next node.
17  if (prevUse)
18  prevUse->nextUse = nextUse;

20  target->firstUse = nextUse;
21  // Update the next node to point to the prev node.

53 InstanceGraphBase::InstanceGraphBase(Operation *parent) : parent(parent) {
54  parent->walk([&](HWModuleLike module) {
55  auto name = module.moduleNameAttr();
56  auto *currentNode = getOrAddNode(name);
57  currentNode->module = module;
58 
59  // Find all instance operations in the module body.
60  module.walk([&](HWInstanceLike instanceOp) {
61  // Add an edge to indicate that this module instantiates the target.
62  auto *targetNode = getOrAddNode(instanceOp.referencedModuleNameAttr());
63  currentNode->addInstance(instanceOp, targetNode);
64  });
65  });
66 }

103 void InstanceGraphBase::replaceInstance(HWInstanceLike inst,
104  HWInstanceLike newInst) {
105  assert(inst.referencedModuleName() == newInst.referencedModuleName() &&
106  "Both instances must be targeting the same module");
107 
108  // Find the instance record of this instance.
109  auto *node = lookup(inst.referencedModuleNameAttr());
110  auto it = llvm::find_if(node->uses(), [&](InstanceRecord *record) {
111  return record->getInstance() == inst;
112  });
113  assert(it != node->usesEnd() && "Instance of module not recorded in graph");
114 
115  // We can just replace the instance op in the InstanceRecord without updating
116  // any instance lists.
117  (*it)->instance = newInst;
118 }

120 bool InstanceGraphBase::isAncestor(HWModuleLike child, HWModuleLike parent) {
121  DenseSet<InstanceGraphNode *> seen;
122  SmallVector<InstanceGraphNode *> worklist;
123  auto *cn = lookup(child);
124  worklist.push_back(cn);
125  seen.insert(cn);
126  while (!worklist.empty()) {
127  auto *node = worklist.back();
128  worklist.pop_back();
129  if (node->getModule() == parent)
130  return true;
131  for (auto *use : node->uses()) {
132  auto *mod = use->getParent();
133  if (!seen.count(mod)) {
134  seen.insert(mod);
135  worklist.push_back(mod);
136  }
137  }
138  }
139  return false;
140 }
