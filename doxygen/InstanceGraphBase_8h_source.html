27 /// This just maps a iterator of references to an iterator of addresses.
28 template <typename It>
29 struct AddressIterator
30  : public llvm::mapped_iterator<It, typename It::pointer (*)(
31  typename It::reference)> {
32  // This using statement is to get around a bug in MSVC. Without it, it
33  // tries to look up "It" as a member type of the parent class.
34  using Iterator = It;
35  /* implicit */ AddressIterator(Iterator iterator)
36  : llvm::mapped_iterator<It, typename Iterator::pointer (*)(
37  typename Iterator::reference)>(
38  iterator, &std::addressof<typename Iterator::value_type>) {}
39 };

44 /// This is an edge in the InstanceGraph. This tracks a specific instantiation
45 /// of a module.
46 class InstanceRecord
47  : public llvm::ilist_node_with_parent<InstanceRecord, InstanceGraphNode> {
48 public:
49  /// Get the instance-like op that this is tracking.
50  HWInstanceLike getInstance() const { return instance; }
51 
52  /// Get the module where the instantiation lives.
53  InstanceGraphNode *getParent() const { return parent; }
54 
55  /// Get the module which the instance-like is instantiating.
56  InstanceGraphNode *getTarget() const { return target; }
57 
58  /// Erase this instance record, removing it from the parent module and the
59  /// target's use-list.
60  void erase();
61 
62 private:
63  friend class InstanceGraphBase;
64  friend class InstanceGraphNode;
65 
66  InstanceRecord(InstanceGraphNode *parent, HWInstanceLike instance,
67  InstanceGraphNode *target)
68  : parent(parent), instance(instance), target(target) {}
69  InstanceRecord(const InstanceRecord &) = delete;
70 
71  /// This is the module where the instantiation lives.
72  InstanceGraphNode *parent;
73 
74  /// The InstanceLike that this is tracking.
75  HWInstanceLike instance;
76 
77  /// This is the module which the instance-like is instantiating.
78  InstanceGraphNode *target;
79  /// Intrusive linked list for other uses.
80  InstanceRecord *nextUse = nullptr;
81  InstanceRecord *prevUse = nullptr;
82 };

84 /// This is a Node in the InstanceGraph. Each node represents a Module in a
85 /// Circuit. Both external modules and regular modules can be represented by
86 /// this class. It is possible to efficiently iterate all modules instantiated
87 /// by this module, as well as all instantiations of this module.
88 class InstanceGraphNode : public llvm::ilist_node<InstanceGraphNode> {
89  using InstanceList = llvm::iplist<InstanceRecord>;
90 
91 public:
92  InstanceGraphNode() : module(nullptr) {}
93 
94  /// Get the module that this node is tracking.
95  HWModuleLike getModule() const { return module; }
96 
97  /// Iterate the instance records in this module.
98  using iterator = detail::AddressIterator<InstanceList::iterator>;
99  iterator begin() { return instances.begin(); }
100  iterator end() { return instances.end(); }
101 
102  /// Return true if there are no more instances of this module.
103  bool noUses() { return !firstUse; }
104 
105  /// Return true if this module has exactly one use.
106  bool hasOneUse() { return llvm::hasSingleElement(uses()); }
107 
108  /// Get the number of direct instantiations of this module.
109  size_t getNumUses() { return std::distance(usesBegin(), usesEnd()); }
110 
111  /// Iterator for module uses.
112  struct UseIterator
113  : public llvm::iterator_facade_base<
114  UseIterator, std::forward_iterator_tag, InstanceRecord *> {
115  UseIterator() : current(nullptr) {}
116  UseIterator(InstanceGraphNode *node) : current(node->firstUse) {}
117  InstanceRecord *operator*() const { return current; }
118  using llvm::iterator_facade_base<UseIterator, std::forward_iterator_tag,
119  InstanceRecord *>::operator++;
120  UseIterator &operator++() {
121  assert(current && "incrementing past end");
122  current = current->nextUse;
123  return *this;
124  }
125  bool operator==(const UseIterator &other) const {
126  return current == other.current;
127  }
128 
129  private:
130  InstanceRecord *current;
131  };
132 
133  /// Iterate the instance records which instantiate this module.
134  UseIterator usesBegin() { return {this}; }
135  UseIterator usesEnd() { return {}; }
136  llvm::iterator_range<UseIterator> uses() {
137  return llvm::make_range(usesBegin(), usesEnd());
138  }
139 
140  /// Record a new instance op in the body of this module. Returns a newly
141  /// allocated InstanceRecord which will be owned by this node.
142  InstanceRecord *addInstance(HWInstanceLike instance,
143  InstanceGraphNode *target);
144 
145 private:
146  friend class InstanceRecord;
147 
148  InstanceGraphNode(const InstanceGraphNode &) = delete;
149 
150  /// Record that a module instantiates this module.
151  void recordUse(InstanceRecord *record);
152 
153  /// The module.
154  HWModuleLike module;
155 
156  /// List of instance operations in this module. This member owns the
157  /// InstanceRecords, which may be pointed to by other InstanceGraphNode's use
158  /// lists.
159  InstanceList instances;
160 
161  /// List of instances which instantiate this module.
162  InstanceRecord *firstUse = nullptr;
163 
164  // Provide access to the constructor.
165  friend class InstanceGraphBase;
166 };

168 /// This graph tracks modules and where they are instantiated. This is intended
169 /// to be used as a cached analysis on circuits. This class can be used
170 /// to walk the modules efficiently in a bottom-up or top-down order.
171 ///
172 /// To use this class, retrieve a cached copy from the analysis manager:
173 /// auto &instanceGraph = getAnalysis<InstanceGraph>(getOperation());
174 class InstanceGraphBase {
175  /// This is the list of InstanceGraphNodes in the graph.
176  using NodeList = llvm::iplist<InstanceGraphNode>;
177 
178 public:
179  virtual ~InstanceGraphBase();
180 
181  /// Look up an InstanceGraphNode for a module.
182  InstanceGraphNode *lookup(HWModuleLike op);
183 
184  /// Lookup an module by name.
185  InstanceGraphNode *lookup(StringAttr name);
186 
187  /// Lookup an InstanceGraphNode for a module.
188  InstanceGraphNode *operator[](HWModuleLike op) { return lookup(op); }
189 
190  /// Look up the referenced module from an InstanceOp. This will use a
191  /// hashtable lookup to find the module, where
192  /// InstanceOp.getReferencedModule() will be a linear search through the IR.
193  HWModuleLike getReferencedModule(HWInstanceLike op);
194 
195  /// Check if child is instantiated by a parent.
196  bool isAncestor(HWModuleLike child, HWModuleLike parent);
197 
198  /// Get the node corresponding to the top-level module of a circuit.
199  virtual InstanceGraphNode *getTopLevelNode() = 0;
200 
201  /// Return the parent under which all nodes are nested.
202  Operation *getParent() { return parent; }
203 
204  /// Returns pointer to member of operation list.
205  static NodeList InstanceGraphBase::*
206  getSublistAccessgetSublistAccess(Operation *) {
207  return &InstanceGraphBase::nodes;
208  }
209 
210  /// Iterate through all modules.
211  using iterator = detail::AddressIterator<NodeList::iterator>;
212  iterator begin() { return nodes.begin(); }
213  iterator end() { return nodes.end(); }
214 
215  //===-------------------------------------------------------------------------
216  // Methods to keep an InstanceGraph up to date.
217  //
218  // These methods are not thread safe. Make sure that modifications are
219  // properly synchronized or performed in a serial context. When the
220  // InstanceGraph is used as an analysis, this is only safe when the pass is
221  // on a CircuitOp or a ModuleOp.
222 
223  /// Add a newly created module to the instance graph.
224  virtual InstanceGraphNode *addModule(HWModuleLike module);
225 
226  /// Remove this module from the instance graph. This will also remove all

230 
231  /// Replaces an instance of a module with another instance. The target module
232  /// of both InstanceOps must be the same.
233  virtual void replaceInstance(HWInstanceLike inst, HWInstanceLike newInst);
234

237  /// operation of HWModuleLike ops.
238  InstanceGraphBase(Operation *parent);
239  InstanceGraphBase(const InstanceGraphBase &) = delete;
240 
241  /// Get the node corresponding to the module. If the node has does not exist
242  /// yet, it will be created.
243  InstanceGraphNode *getOrAddNode(StringAttr name);
244 
245  /// The node under which all modules are nested.
246  Operation *parent;
247 
248  /// The storage for graph nodes, with deterministic iteration.
249  NodeList nodes;
250 
251  /// This maps each operation to its graph node.
