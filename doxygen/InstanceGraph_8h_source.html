26 /// This is an edge in the InstanceGraph. This tracks a specific instantiation
27 /// of a module.
28 class InstanceRecord {
29 public:
30  InstanceRecord(InstanceOp instance, InstanceGraphNode *parent,
31  InstanceGraphNode *target)
32  : instance(instance), parent(parent), target(target) {}
33 
34  /// Get the InstanceOp that this is tracking.
35  InstanceOp getInstance() const { return instance; }
36 
37  /// Get the module where the InstanceOp lives.
38  InstanceGraphNode *getParent() const { return parent; }
39 
40  /// Get the module which the InstanceOp is instantiating.
41  InstanceGraphNode *getTarget() const { return target; }
42 
43 private:
44  /// The InstanceOp that this is tracking.
45  InstanceOp instance;
46 
47  /// This is the module where the InstanceOp lives.
48  InstanceGraphNode *parent;
49 
50  /// This is the module which the InstanceOp is instantiating.
51  InstanceGraphNode *target;
52 };

54 /// This is a Node in the InstanceGraph. Each node represents a Module in a
55 /// Circuit. Both external modules and regular modules can be represented by
56 /// this class. It is possible to efficiently iterate all modules instantiated
57 /// by this module, as well as all instantiations of this module.
58 class InstanceGraphNode {
59  using EdgeVec = std::deque<InstanceRecord>;
60  using UseVec = std::vector<InstanceRecord *>;
61 
62  static InstanceRecord *unwrap(EdgeVec::value_type &value) {
63  return &value;
64  }
65  class InstanceIterator final
66  : public llvm::mapped_iterator<EdgeVec::iterator, decltype(&unwrap)> {
67  public:
68  /// Initializes the result type iterator to the specified result iterator.
69  InstanceIterator(EdgeVec::iterator it)
70  : llvm::mapped_iterator<EdgeVec::iterator, decltype(&unwrap)>(
71  it, &unwrap) {}
72  };
73 
74 public:
75  InstanceGraphNode() : module(nullptr) {}
76 
77  /// Get the module that this node is tracking.
78  Operation *getModule() const { return module; }
79 
80  /// Iterate the instance records in this module.
81  using iterator = InstanceIterator;
82  iterator begin() { return iterator(moduleInstances.begin()); }
83  iterator end() { return iterator(moduleInstances.end()); }
84  llvm::iterator_range<iterator> instances() {
85  return llvm::make_range(begin(), end());
86  }
87 
88  /// Iterate the instance records which instantiate this module.
89  using use_iterator = UseVec::iterator;
90  use_iterator uses_begin() { return moduleUses.begin(); }
91  use_iterator uses_end() { return moduleUses.end(); }
92  llvm::iterator_range<use_iterator> uses() {
93  return llvm::make_range(uses_begin(), uses_end());
94  }
95 
96 private:
97  /// Record a new instance op in the body of this module. Returns a newly
98  /// allocated InstanceRecord which will be owned by this node.
99  InstanceRecord *recordInstance(InstanceOp instance,
100  InstanceGraphNode *target);
101 
102  /// Record that a module instantiates this module.
103  void recordUse(InstanceRecord *record);
104 
105  /// The module.
106  Operation *module;
107 
108  /// List of instance operations in this module. This member owns the
109  /// InstanceRecords, which may be pointed to by other InstanceGraohNode's use
110  /// lists.
111  EdgeVec moduleInstances;
112 
113  /// List of instances which instantiate this module.
114  UseVec moduleUses;
115 
116  // Provide access to the constructor.
117  friend class InstanceGraph;
118 };

120 /// This graph tracks modules and where they are instantiated. This is intended
121 /// to be used as a cached analysis on FIRRTL circuits. This class can be used
122 /// to walk the modules efficiently in a bottom-up or top-down order.
123 ///
124 /// To use this class, retrieve a cached copy from the analysis manager:
125 /// auto &instanceGraph = getAnalysis<InstanceGraph>(getOperation());
126 class InstanceGraph {
127 
128  /// Storage for InstanceGraphNodes.
129  using NodeVec = std::deque<InstanceGraphNode>;
130 
131  /// Iterator that unwraps a unique_ptr to return a regular pointer.
132  static InstanceGraphNode *unwrap(NodeVec::value_type &value) {
133  return &value;
134  }
135  struct NodeIterator final
136  : public llvm::mapped_iterator<NodeVec::iterator,
137  decltype(&unwrap)> {
138  /// Initializes the result type iterator to the specified result iterator.
139  NodeIterator(NodeVec::iterator it)
140  : llvm::mapped_iterator<NodeVec::iterator, decltype(&unwrap)>(
141  it, &unwrap) {}
142  };
143 
144 public:
145  /// Create a new module graph of a circuit. This must be called on a FIRRTL
146  /// CircuitOp.
147  explicit InstanceGraph(Operation *operation);
148 
149  /// Get the node corresponding to the top-level module of a circuit.
150  InstanceGraphNode *getTopLevelNode();
151 
152  /// Look up an InstanceGraphNode for a module. Operation must be an FModuleOp
153  /// or an FExtModuleOp.
154  InstanceGraphNode *lookup(Operation *op);
155 
156  /// Lookup an InstanceGraphNode for a module. Operation must be an FModuleOp
157  /// or an FExtModuleOp.
158  InstanceGraphNode *operator[](Operation *op) { return lookup(op); }
159 
160  /// Look up the referenced module from an InstanceOp. This will use a
161  /// hashtable lookup to find the module, where
162  /// InstanceOp.getReferencedModule() will be a linear search through the IR.
163  Operation *getReferencedModule(InstanceOp op);
164 
165  /// Iterate through all modules.
166  using iterator = NodeIterator;
167  iterator begin() { return nodes.begin(); }
168  iterator end() { return nodes.end(); }
169 
170 private:
171  /// Get the node corresponding to the module. If the node has does not exist
172  /// yet, it will be created.
173  InstanceGraphNode *getOrAddNode(StringRef name);
174 
175  /// Lookup an module by name.
176  InstanceGraphNode *lookup(StringRef name);
177 
178  /// The storage for graph nodes, with deterministic iteration.
179  NodeVec nodes;
180 
181  /// This maps each operation to its graph node.
182  llvm::StringMap<unsigned> nodeMap;
183 };
