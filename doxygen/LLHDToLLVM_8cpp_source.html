117 /// Extract the details from the given signal struct. The details are returned
118 /// in the original struct order.
119 static std::vector<Value> getSignalDetail(ConversionPatternRewriter &rewriter,
120  LLVM::LLVMDialect *dialect,
121  Location loc, Value signal,
122  bool extractIndices = false) {
123 
124  auto i8PtrTy =
125  LLVM::LLVMPointerType::get(IntegerType::get(dialect->getContext(), 8));
126  auto i32Ty = IntegerType::get(dialect->getContext(), 32);
127  auto i64Ty = IntegerType::get(dialect->getContext(), 64);
128 
129  std::vector<Value> result;
130 
131  // Extract the value and offset elements.
132  auto zeroC = rewriter.create<LLVM::ConstantOp>(loc, i32Ty,
133  rewriter.getI32IntegerAttr(0));
134  auto oneC = rewriter.create<LLVM::ConstantOp>(loc, i32Ty,
135  rewriter.getI32IntegerAttr(1));
136 
137  auto sigPtrPtr =
138  rewriter.create<LLVM::GEPOp>(loc, LLVM::LLVMPointerType::get(i8PtrTy),
139  signal, ArrayRef<Value>({zeroC, zeroC}));
140  result.push_back(rewriter.create<LLVM::LoadOp>(loc, i8PtrTy, sigPtrPtr));
141 
142  auto offsetPtr =
143  rewriter.create<LLVM::GEPOp>(loc, LLVM::LLVMPointerType::get(i64Ty),
144  signal, ArrayRef<Value>({zeroC, oneC}));
145  result.push_back(rewriter.create<LLVM::LoadOp>(loc, i64Ty, offsetPtr));
146 
147  // Extract the instance and global indices.
148  if (extractIndices) {
149  auto twoC = rewriter.create<LLVM::ConstantOp>(
150  loc, i32Ty, rewriter.getI32IntegerAttr(2));
151  auto threeC = rewriter.create<LLVM::ConstantOp>(
152  loc, i32Ty, rewriter.getI32IntegerAttr(3));
153 
154  auto instIndexPtr =
155  rewriter.create<LLVM::GEPOp>(loc, LLVM::LLVMPointerType::get(i64Ty),
156  signal, ArrayRef<Value>({zeroC, twoC}));
157  result.push_back(rewriter.create<LLVM::LoadOp>(loc, i64Ty, instIndexPtr));
158 
159  auto globalIndexPtr =
160  rewriter.create<LLVM::GEPOp>(loc, LLVM::LLVMPointerType::get(i64Ty),
161  signal, ArrayRef<Value>({zeroC, threeC}));
162  result.push_back(rewriter.create<LLVM::LoadOp>(loc, i64Ty, globalIndexPtr));
163  }
164 
165  return result;
166 }

169 static Value createSubSig(LLVM::LLVMDialect *dialect,
170  ConversionPatternRewriter &rewriter, Location loc,

191  loc, LLVM::LLVMPointerType::get(sigTy), oneC, 4);

223 /// Gather the types of values that are used outside of the block they're
224 /// defined in. An LLVMType structure containing those types, in order of
225 /// appearance, is returned.
226 static Type getProcPersistenceTy(LLVM::LLVMDialect *dialect,
227  TypeConverter *converter, ProcOp &proc) {
228  SmallVector<Type, 3> types = SmallVector<Type, 3>();
229  proc.walk([&](Operation *op) -> void {
230  if (op->isUsedOutsideOfBlock(op->getBlock()) || isWaitDestArg(op)) {
231  auto ty = op->getResult(0).getType();
232  auto convertedTy = converter->convertType(ty);
233  if (ty.isa<PtrType, SigType>()) {
234  // Persist the unwrapped value.
235  types.push_back(unwrapLLVMPtr(convertedTy));
236  } else {
237  // Persist the value as is.
238  types.push_back(convertedTy);
239  }
240  }
241  });
242 
243  // Also persist block arguments escaping their defining block.
244  for (auto &block : proc.getBlocks()) {
245  // Skip entry block (contains the function signature in its args).
246  if (block.isEntryBlock())
247  continue;
248 
249  for (auto arg : block.getArguments()) {
250  if (arg.isUsedOutsideOfBlock(&block)) {
251  types.push_back(converter->convertType(arg.getType()));
252  }
253  }
254  }
255 
256  return LLVM::LLVMStructType::getLiteral(dialect->getContext(), types);
257 }

259 /// Insert a comparison block that either jumps to the trueDest block, if the
260 /// resume index mathces the current index, or to falseDest otherwise. If no
261 /// falseDest is provided, the next block is taken insead.
262 static void insertComparisonBlock(ConversionPatternRewriter &rewriter,
263  LLVM::LLVMDialect *dialect, Location loc,
264  Region *body, Value resumeIdx, int currIdx,
265  Block *trueDest, ValueRange trueDestArgs,
266  Block *falseDest = nullptr) {
267  auto i32Ty = IntegerType::get(dialect->getContext(), 32);
268  auto secondBlock = ++body->begin();
269  auto newBlock = rewriter.createBlock(body, secondBlock);
270  auto cmpIdx = rewriter.create<LLVM::ConstantOp>(
271  loc, i32Ty, rewriter.getI32IntegerAttr(currIdx));
272  auto cmpRes = rewriter.create<LLVM::ICmpOp>(loc, LLVM::ICmpPredicate::eq,
273  resumeIdx, cmpIdx);
274 
275  // Default to jumping to the next block for the false case, if no explicit
276  // block is provided.
277  if (!falseDest)
278  falseDest = &*secondBlock;
279 
280  rewriter.create<LLVM::CondBrOp>(loc, cmpRes, trueDest, trueDestArgs,
281  falseDest, ValueRange());
282 
283  // Redirect the entry block terminator to the new comparison block.
284  auto entryTer = body->front().getTerminator();
285  entryTer->setSuccessor(newBlock, 0);
286 }

305 /// Persist a `Value` by storing it into the process persistence table, and
306 /// substituting the uses that escape the block the operation is defined in with
307 /// a load from the persistence table.
308 static void persistValue(LLVM::LLVMDialect *dialect, Location loc,
309  TypeConverter *converter,
310  ConversionPatternRewriter &rewriter, Type stateTy,
311  int &i, Value state, Value persist) {
312  auto elemTy = stateTy.cast<LLVM::LLVMStructType>()
313  .getBody()[3]
314  .cast<LLVM::LLVMStructType>()
315  .getBody()[i];
316 
317  if (auto arg = persist.dyn_cast<BlockArgument>()) {
318  rewriter.setInsertionPointToStart(arg.getParentBlock());
319  } else {
320  rewriter.setInsertionPointAfter(persist.getDefiningOp());
321  }
322 
323  Value convPersist = converter->materializeTargetConversion(
324  rewriter, loc, converter->convertType(persist.getType()), {persist});
325 
326  auto gep0 = gepPersistenceState(dialect, loc, rewriter, elemTy, i, state);
327 
328  Value toStore;
329  if (auto ptr = persist.getType().dyn_cast<PtrType>()) {
330  // Unwrap the pointer and store it's value.
331  auto elemTy = converter->convertType(ptr.getUnderlyingType());
332  toStore = rewriter.create<LLVM::LoadOp>(loc, elemTy, convPersist);
333  } else if (persist.getType().isa<SigType>()) {
334  // Unwrap and store the signal struct.
335  toStore = rewriter.create<LLVM::LoadOp>(loc, getLLVMSigType(dialect),
336  convPersist);
337  } else {
338  // Store the value directly.
339  toStore = convPersist;
340  }
341 
342  rewriter.create<LLVM::StoreOp>(loc, toStore, gep0);
343 
344  // Load the value from the persistence table and substitute the original
345  // use with it, whenever it is in a different block.
346  for (auto &use : llvm::make_early_inc_range(persist.getUses())) {
347  auto user = use.getOwner();
348  if (persist.getType().isa<PtrType>() && user != toStore.getDefiningOp() &&
349  user != convPersist.getDefiningOp() &&
350  persist.getParentBlock() == user->getBlock()) {
351  // Redirect uses of the pointer in the same block to the pointer in the
352  // persistence state. This ensures that stores and loads all operate on
353  // the same value.
354  use.set(gep0);
355  } else if (persist.getParentBlock() != user->getBlock() ||
356  (isa<WaitOp>(user) &&
357  isWaitDestArg(cast<WaitOp>(user), persist))) {
358  // The destination args of a wait op have to be loaded in the entry block
359  // of the function, before jumping to the resume destination, so they can
360  // be passed as block arguments by the comparison block.
361  if (isa<WaitOp>(user) && isWaitDestArg(cast<WaitOp>(user), persist))
362  rewriter.setInsertionPoint(
363  user->getParentRegion()->front().getTerminator());
364  else
365  rewriter.setInsertionPointToStart(user->getBlock());
366 
367  auto gep1 = gepPersistenceState(dialect, loc, rewriter, elemTy, i, state);
368  // Use the pointer in the state struct directly for pointer and signal
369  // types.
370  if (persist.getType().isa<PtrType, SigType>()) {
371  use.set(gep1);
372  } else {
373  auto load1 = rewriter.create<LLVM::LoadOp>(loc, elemTy, gep1);
374  // Load the value otherwise.
375  use.set(load1);
376  }
377  }
378  }
379  i++;
380 }

382 /// Insert the blocks and operations needed to persist values across suspension,
383 /// as well as ones needed to resume execution at the right spot.
384 static void insertPersistence(TypeConverter *converter,
385  ConversionPatternRewriter &rewriter,
386  LLVM::LLVMDialect *dialect, Location loc,
387  ProcOp &proc, Type &stateTy,
388  LLVM::LLVMFuncOp &converted,
389  Operation *splitEntryBefore) {
390  auto i32Ty = IntegerType::get(dialect->getContext(), 32);
391 
392  auto &firstBB = converted.getBody().front();
393 
394  // Split entry block such that all the operations contained in it in the
395  // original process appear after the comparison blocks.
396  auto splitFirst =
397  rewriter.splitBlock(&firstBB, splitEntryBefore->getIterator());
398 
399  // Insert dummy branch terminator at the new end of the function's entry
400  // block.
401  rewriter.setInsertionPointToEnd(&firstBB);
402  rewriter.create<LLVM::BrOp>(loc, ValueRange(), splitFirst);
403 
404  // Load the resume index from the process state argument.
405  rewriter.setInsertionPoint(firstBB.getTerminator());
406  auto zeroC = rewriter.create<LLVM::ConstantOp>(loc, i32Ty,
407  rewriter.getI32IntegerAttr(0));
408  auto oneC = rewriter.create<LLVM::ConstantOp>(loc, i32Ty,
409  rewriter.getI32IntegerAttr(1));
410  auto gep = rewriter.create<LLVM::GEPOp>(
411  loc, LLVM::LLVMPointerType::get(i32Ty), converted.getArgument(1),
412  ArrayRef<Value>({zeroC, oneC}));
413 
414  auto larg = rewriter.create<LLVM::LoadOp>(loc, i32Ty, gep);
415 
416  auto body = &converted.getBody();
417 
418  // Insert an abort block as the last block.
419  auto abortBlock = rewriter.createBlock(body, body->end());
420  rewriter.create<LLVM::ReturnOp>(loc, ValueRange());
421 
422  // Redirect the entry block to a first comparison block. If on a first
423  // execution, jump to the new (splitted) entry block, else the process is in
424  // an illegal state and jump to the abort block.
425  insertComparisonBlock(rewriter, dialect, loc, body, larg, 0, splitFirst,
426  ValueRange(), abortBlock);
427 
428  // Keep track of the index in the presistence table of the operation we
429  // are currently processing.
430  int i = 0;
431  // Keep track of the current resume index for comparison blocks.
432  int waitInd = 0;
433 
434  // Insert operations required to persist values across process suspension.
435  converted.walk([&](Operation *op) -> void {
436  if ((op->isUsedOutsideOfBlock(op->getBlock()) || isWaitDestArg(op)) &&
437  op->getResult(0) != larg.getResult()) {
438  persistValue(dialect, loc, converter, rewriter, stateTy, i,
439  converted.getArgument(1), op->getResult(0));
440  }
441 
442  // Insert a comparison block for wait operations.
443  if (auto wait = dyn_cast<WaitOp>(op)) {
444  insertComparisonBlock(rewriter, dialect, loc, body, larg, ++waitInd,
445  wait.dest(), wait.destOps());
446 
447  // Insert the resume index update at the wait operation location.
448  rewriter.setInsertionPoint(op);
449  auto procState = op->getParentOfType<LLVM::LLVMFuncOp>().getArgument(1);
450  auto resumeIdxC = rewriter.create<LLVM::ConstantOp>(
451  loc, i32Ty, rewriter.getI32IntegerAttr(waitInd));
452  auto resumeIdxPtr = rewriter.create<LLVM::GEPOp>(
453  loc, LLVM::LLVMPointerType::get(i32Ty), procState,
454  ArrayRef<Value>({zeroC, oneC}));
455  rewriter.create<LLVM::StoreOp>(op->getLoc(), resumeIdxC, resumeIdxPtr);
456  }
457  });
458 
459  // Also persist argument blocks escaping their defining block.
460  for (auto &block : converted.getBlocks()) {
461  // Skip entry block as it contains the function signature.
462  if (block.isEntryBlock())
463  continue;
464 
465  for (auto arg : block.getArguments()) {
466  if (arg.isUsedOutsideOfBlock(&block)) {
467  persistValue(dialect, loc, converter, rewriter, stateTy, i,
468  converted.getArgument(1), arg);
469  }
470  }
471  }
472 }

478  Operation *entity) {

521 /// Recursively clone the init origin of a sig operation into the init function,
522 /// up to the initial constant value(s). This is required to clone the
523 /// initialization of array and struct signals, where the init operant cannot
524 /// originate from a constant operation. Integer constants are currently assumed
525 /// to come from a constant operation.
526 static Operation *recursiveCloneInit(OpBuilder &initBuilder, Operation *op) {
527  if (auto arrayCreateOp = dyn_cast<hw::ArrayCreateOp>(op)) {
528  auto def =
529  cast<hw::ArrayCreateOp>(initBuilder.insert(arrayCreateOp.clone()));
530  initBuilder.setInsertionPoint(def.getOperation());
531  for (size_t i = 0, e = def.inputs().size(); i < e; ++i) {
532  auto clone =
533  recursiveCloneInit(initBuilder, def.inputs()[i].getDefiningOp());
534  def.setOperand(i, clone->getResult(0));
535  }
536  initBuilder.setInsertionPointAfter(def.getOperation());
537  return def;
538  }
539 
540  if (auto structCreateOp = dyn_cast<hw::StructCreateOp>(op)) {
541  auto def =
542  cast<hw::StructCreateOp>(initBuilder.insert(structCreateOp.clone()));
543  initBuilder.setInsertionPoint(def.getOperation());
544  for (size_t i = 0, e = def.input().size(); i < e; ++i) {
545  auto clone =
546  recursiveCloneInit(initBuilder, def.input()[i].getDefiningOp());
547  def.setOperand(i, clone->getResult(0));
548  }
549  initBuilder.setInsertionPointAfter(def.getOperation());
550  return def;
551  }
552 
553  return initBuilder.insert(op->clone());
554 }

662 /// Convert an `llhd.entity` entity to LLVM dialect. The result is an
663 /// `llvm.func` which takes a pointer to the global simulation state, a pointer
664 /// to the entity's local state, and a pointer to the instance's signal table as
665 /// arguments.
666 struct EntityOpConversion : public ConvertToLLVMPattern {
667  explicit EntityOpConversion(MLIRContext *ctx,
668  LLVMTypeConverter &typeConverter,
669  size_t &sigCounter, size_t &regCounter)
670  : ConvertToLLVMPattern(llhd::EntityOp::getOperationName(), ctx,
671  typeConverter),
672  sigCounter(sigCounter), regCounter(regCounter) {}
673 
674  LogicalResult
675  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
676  ConversionPatternRewriter &rewriter) const override {
677  // Get adapted operands.
678  EntityOpAdaptor transformed(operands);
679  // Get entity operation.
680  auto entityOp = cast<EntityOp>(op);
681 
682  // Collect used llvm types.
683  auto voidTy = getVoidType();
684  auto i8PtrTy = getVoidPtrType();
685  auto i32Ty = IntegerType::get(rewriter.getContext(), 32);
686  auto sigTy = getLLVMSigType(&getDialect());
687  auto entityStatePtrTy =
688  LLVM::LLVMPointerType::get(getRegStateTy(&getDialect(), op));
689 
690  regCounter = 0;
691 
692  // Use an intermediate signature conversion to add the arguments for the
693  // state and signal table pointer arguments.
694  LLVMTypeConverter::SignatureConversion intermediate(
695  entityOp.getNumArguments());
696  // Add state and signal table arguments.
697  intermediate.addInputs(std::array<Type, 3>(
698  {i8PtrTy, entityStatePtrTy, LLVM::LLVMPointerType::get(sigTy)}));
699  for (size_t i = 0, e = entityOp.getNumArguments(); i < e; ++i)
700  intermediate.addInputs(i, voidTy);
701  rewriter.applySignatureConversion(&entityOp.getBody(), intermediate,
702  typeConverter);
703 
704  OpBuilder bodyBuilder =
705  OpBuilder::atBlockBegin(&entityOp.getBlocks().front());
706  LLVMTypeConverter::SignatureConversion final(
707  intermediate.getConvertedTypes().size());
708  final.addInputs(0, i8PtrTy);
709  final.addInputs(1, entityStatePtrTy);
710  final.addInputs(2, LLVM::LLVMPointerType::get(sigTy));
711 
712  // The first n elements of the signal table represent the entity arguments,
713  // while the remaining elements represent the entity's owned signals.
714  sigCounter = entityOp.getNumArguments();
715  for (size_t i = 0; i < sigCounter; ++i) {
716  // Create gep operations from the signal table for each original argument.
717  auto index = bodyBuilder.create<LLVM::ConstantOp>(
718  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(i));
719  auto gep = bodyBuilder.create<LLVM::GEPOp>(
720  op->getLoc(), LLVM::LLVMPointerType::get(sigTy),
721  entityOp.getArgument(2), ArrayRef<Value>(index));
722  // Remap i-th original argument to the gep'd signal pointer.
723  final.remapInput(i + 3, gep.getResult());
724  }
725 
726  rewriter.applySignatureConversion(&entityOp.getBody(), final,
727  typeConverter);
728 
729  // Get the converted entity signature.
730  auto funcTy = LLVM::LLVMFunctionType::get(
731  voidTy, {i8PtrTy, entityStatePtrTy, LLVM::LLVMPointerType::get(sigTy)});
732 
733  // Create the a new llvm function to house the lowered entity.
734  auto llvmFunc = rewriter.create<LLVM::LLVMFuncOp>(
735  op->getLoc(), entityOp.getName(), funcTy);
736 
737  // Add a return to the entity for later inclusion into the LLVM function.
738  rewriter.setInsertionPointToEnd(&entityOp.getBlocks().front());
739  rewriter.create<LLVM::ReturnOp>(op->getLoc(), ValueRange{});
740 
741  // Inline the entity region in the new llvm function.
742  rewriter.inlineRegionBefore(entityOp.getBody(), llvmFunc.getBody(),
743  llvmFunc.end());
744 
745  // Erase the original operation.
746  rewriter.eraseOp(op);
747 
748  return success();
749  }
750 
751 private:
752  size_t &sigCounter;
753  size_t &regCounter;
754 };

758 /// Convert an `llhd.proc` operation to LLVM dialect. This inserts the required
759 /// logic to resume execution after an `llhd.wait` operation, as well as state
760 /// keeping for values that need to persist across suspension.
761 struct ProcOpConversion : public ConvertToLLVMPattern {
762  explicit ProcOpConversion(MLIRContext *ctx, LLVMTypeConverter &typeConverter)
763  : ConvertToLLVMPattern(ProcOp::getOperationName(), ctx, typeConverter) {}
764 
765  LogicalResult
766  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
767  ConversionPatternRewriter &rewriter) const override {
768  auto procOp = cast<ProcOp>(op);
769 
770  // Get adapted operands.
771  ProcOpAdaptor transformed(operands);
772 
773  // Collect used llvm types.
774  auto voidTy = getVoidType();
775  auto i8PtrTy = getVoidPtrType();
776  auto i1Ty = IntegerType::get(rewriter.getContext(), 1);
777  auto i32Ty = IntegerType::get(rewriter.getContext(), 32);
778  auto senseTableTy = LLVM::LLVMPointerType::get(
779  LLVM::LLVMArrayType::get(i1Ty, procOp.getNumArguments()));
780  auto stateTy = LLVM::LLVMStructType::getLiteral(
781  rewriter.getContext(),
782  {/* current instance */ i32Ty, /* resume index */ i32Ty,
783  /* sense flags */ senseTableTy, /* persistent types */
784  getProcPersistenceTy(&getDialect(), typeConverter, procOp)});
785  auto sigTy = getLLVMSigType(&getDialect());
786 
787  // Keep track of the original first operation of the process, to know where
788  // to split the first block to insert comparison blocks.
789  auto &firstOp = op->getRegion(0).front().front();
790 
791  // Have an intermediate signature conversion to add the arguments for the
792  // state, process-specific state and signal table.
793  LLVMTypeConverter::SignatureConversion intermediate(
794  procOp.getNumArguments());
795  // Add state, process state table and signal table arguments.
796  std::array<Type, 3> procArgTys({i8PtrTy,
797  LLVM::LLVMPointerType::get(stateTy),
798  LLVM::LLVMPointerType::get(sigTy)});
799  intermediate.addInputs(procArgTys);
800  for (size_t i = 0, e = procOp.getNumArguments(); i < e; ++i)
801  intermediate.addInputs(i, voidTy);
802  rewriter.applySignatureConversion(&procOp.getBody(), intermediate,
803  typeConverter);
804 
805  // Get the final signature conversion.
806  OpBuilder bodyBuilder =
807  OpBuilder::atBlockBegin(&procOp.getBlocks().front());
808  LLVMTypeConverter::SignatureConversion final(
809  intermediate.getConvertedTypes().size());
810  final.addInputs(0, i8PtrTy);
811  final.addInputs(1, LLVM::LLVMPointerType::get(stateTy));
812  final.addInputs(2, LLVM::LLVMPointerType::get(sigTy));
813 
814  for (size_t i = 0, e = procOp.getNumArguments(); i < e; ++i) {
815  // Create gep operations from the signal table for each original argument.
816  auto index = bodyBuilder.create<LLVM::ConstantOp>(
817  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(i));
818  auto gep = bodyBuilder.create<LLVM::GEPOp>(
819  op->getLoc(), LLVM::LLVMPointerType::get(sigTy),
820  procOp.getArgument(2), ArrayRef<Value>({index}));
821 
822  // Remap the i-th original argument to the gep'd value.
823  final.remapInput(i + 3, gep.getResult());
824  }
825 
826  // Get the converted process signature.
827  auto funcTy = LLVM::LLVMFunctionType::get(
828  voidTy, {i8PtrTy, LLVM::LLVMPointerType::get(stateTy),
829  LLVM::LLVMPointerType::get(sigTy)});
830  // Create a new llvm function to house the lowered process.
831  auto llvmFunc = rewriter.create<LLVM::LLVMFuncOp>(op->getLoc(),
832  procOp.getName(), funcTy);
833 
834  // Inline the process region in the new llvm function.
835  rewriter.inlineRegionBefore(procOp.getBody(), llvmFunc.getBody(),
836  llvmFunc.end());
837 
838  insertPersistence(typeConverter, rewriter, &getDialect(), op->getLoc(),
839  procOp, stateTy, llvmFunc, &firstOp);
840 
841  // Convert the block argument types after inserting the persistence, as this
842  // would otherwise interfere with the persistence generation.
843  if (failed(rewriter.convertRegionTypes(&llvmFunc.getBody(), *typeConverter,
844  &final))) {
845  return failure();
846  }
847 
848  rewriter.eraseOp(op);
849 
850  return success();
851  }
852 };

912 /// resume point, sets the observed senses (if present) and schedules the timed
913 /// wake up (if present).
914 struct WaitOpConversion : public ConvertToLLVMPattern {
915  explicit WaitOpConversion(MLIRContext *ctx, LLVMTypeConverter &typeConverter)
916  : ConvertToLLVMPattern(WaitOp::getOperationName(), ctx, typeConverter) {}

918  LogicalResult
919  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
920  ConversionPatternRewriter &rewriter) const override {
921  auto waitOp = cast<WaitOp>(op);
922  WaitOpAdaptor transformed(operands, op->getAttrDictionary());
923  auto llvmFunc = op->getParentOfType<LLVM::LLVMFuncOp>();
924 
925  auto voidTy = getVoidType();
926  auto i8PtrTy = getVoidPtrType();
927  auto i1Ty = IntegerType::get(rewriter.getContext(), 1);
928  auto i32Ty = IntegerType::get(rewriter.getContext(), 32);
929  auto i64Ty = IntegerType::get(rewriter.getContext(), 64);
930 
931  // Get the llhdSuspend runtime function.
932  auto llhdSuspendTy = LLVM::LLVMFunctionType::get(
933  voidTy, {i8PtrTy, i8PtrTy, i64Ty, i64Ty, i64Ty});
934  auto module = op->getParentOfType<ModuleOp>();
935  auto llhdSuspendFunc = getOrInsertFunction(module, rewriter, op->getLoc(),
936  "llhdSuspend", llhdSuspendTy);
937 
938  auto statePtr = llvmFunc.getArgument(0);
939  auto procState = llvmFunc.getArgument(1);
940  auto procStateTy = procState.getType();
941  auto senseTableTy = procStateTy.cast<LLVM::LLVMPointerType>()
942  .getElementType()
943  .cast<LLVM::LLVMStructType>()
944  .getBody()[2]
945  .cast<LLVM::LLVMPointerType>()
946  .getElementType();
947 
948  // Get senses ptr.
949  auto zeroC = rewriter.create<LLVM::ConstantOp>(
950  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(0));
951  auto twoC = rewriter.create<LLVM::ConstantOp>(
952  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(2));
953  auto sensePtrGep = rewriter.create<LLVM::GEPOp>(
954  op->getLoc(),
955  LLVM::LLVMPointerType::get(LLVM::LLVMPointerType::get(senseTableTy)),
956  procState, ArrayRef<Value>({zeroC, twoC}));
957  auto sensePtr = rewriter.create<LLVM::LoadOp>(
958  op->getLoc(), LLVM::LLVMPointerType::get(senseTableTy), sensePtrGep);
959 
960  // Reset sense table, if not all signals are observed.
961  if (waitOp.obs().size() <
962  senseTableTy.cast<LLVM::LLVMArrayType>().getNumElements()) {
963  auto zeroB = rewriter.create<LLVM::ConstantOp>(
964  op->getLoc(), i1Ty, rewriter.getBoolAttr(false));
965  for (size_t i = 0,
966  e = senseTableTy.cast<LLVM::LLVMArrayType>().getNumElements();
967  i < e; ++i) {
968  auto indC = rewriter.create<LLVM::ConstantOp>(
969  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(i));
970  auto senseElemPtr = rewriter.create<LLVM::GEPOp>(
971  op->getLoc(), LLVM::LLVMPointerType::get(i1Ty), sensePtr,
972  ArrayRef<Value>({zeroC, indC}));
973  rewriter.create<LLVM::StoreOp>(op->getLoc(), zeroB, senseElemPtr);
974  }
975  }
976 
977  // Set sense flags for observed signals.
978  for (auto observed : transformed.obs()) {
979  auto instIndexPtr = rewriter.create<LLVM::GEPOp>(
980  op->getLoc(), LLVM::LLVMPointerType::get(i64Ty), observed,
981  ArrayRef<Value>({zeroC, twoC}));
982  auto instIndex =
983  rewriter.create<LLVM::LoadOp>(op->getLoc(), i64Ty, instIndexPtr);
984  auto oneB = rewriter.create<LLVM::ConstantOp>(op->getLoc(), i1Ty,
985  rewriter.getBoolAttr(true));
986  auto senseElementPtr = rewriter.create<LLVM::GEPOp>(
987  op->getLoc(), LLVM::LLVMPointerType::get(i1Ty), sensePtr,
988  ArrayRef<Value>({zeroC, instIndex}));
989  rewriter.create<LLVM::StoreOp>(op->getLoc(), oneB, senseElementPtr);
990  }
991 
992  // Update and store the new resume index in the process state.
993  auto procStateBC =
994  rewriter.create<LLVM::BitcastOp>(op->getLoc(), i8PtrTy, procState);
995 
996  // Spawn scheduled event, if present.
997  if (waitOp.time()) {
998  auto realTime = rewriter.create<LLVM::ExtractValueOp>(
999  op->getLoc(), i64Ty, transformed.time(), rewriter.getI32ArrayAttr(0));
1000  auto delta = rewriter.create<LLVM::ExtractValueOp>(
1001  op->getLoc(), i64Ty, transformed.time(), rewriter.getI32ArrayAttr(1));
1002  auto eps = rewriter.create<LLVM::ExtractValueOp>(
1003  op->getLoc(), i64Ty, transformed.time(), rewriter.getI32ArrayAttr(2));
1004 
1005  std::array<Value, 5> args({statePtr, procStateBC, realTime, delta, eps});
1006  rewriter.create<LLVM::CallOp>(op->getLoc(), llvm::None,
1007  SymbolRefAttr::get(llhdSuspendFunc), args);
1008  }
1009 
1010  rewriter.replaceOpWithNewOp<LLVM::ReturnOp>(op, ValueRange());
1011  return success();
1012  }

1074  "addSigStructElement", addSigStructElemFuncTy);
1075 
1076  // Get or insert allocProc library call definition.
1077  auto allocProcFuncTy =
1078  LLVM::LLVMFunctionType::get(voidTy, {i8PtrTy, i8PtrTy, i8PtrTy});
1079  auto allocProcFunc = getOrInsertFunction(module, rewriter, op->getLoc(),
1080  "allocProc", allocProcFuncTy);
1081 
1082  // Get or insert allocEntity library call definition.
1083  auto allocEntityFuncTy =
1084  LLVM::LLVMFunctionType::get(voidTy, {i8PtrTy, i8PtrTy, i8PtrTy});
1085  auto allocEntityFunc = getOrInsertFunction(
1086  module, rewriter, op->getLoc(), "allocEntity", allocEntityFuncTy);
1087 
1088  Value initStatePtr = initFunc.getArgument(0);
1089 
1090  // Get a builder for the init function.
1091  OpBuilder initBuilder =
1092  OpBuilder::atBlockTerminator(&initFunc.getBody().getBlocks().front());
1093 
1094  // Use the instance name to retrieve the instance from the state.
1095  auto ownerName = entity.getName().str() + "." + instOp.name().str();
1096 
1097  // Get or create owner name string
1098  Value owner;
1099  auto parentSym =
1100  module.lookupSymbol<LLVM::GlobalOp>("instance." + ownerName);
1101  if (!parentSym) {
1102  owner = LLVM::createGlobalString(
1103  op->getLoc(), initBuilder, "instance." + ownerName, ownerName + '\0',
1104  LLVM::Linkage::Internal);
1105  parentSym = module.lookupSymbol<LLVM::GlobalOp>("instance." + ownerName);
1106  } else {
1107  owner =
1108  getGlobalString(op->getLoc(), initBuilder, typeConverter, parentSym);
1109  }
1110 
1111  // Handle entity instantiation.
1112  if (auto child = module.lookupSymbol<EntityOp>(instOp.callee())) {
1113  auto regStateTy = getRegStateTy(&getDialect(), child.getOperation());
1114  auto regStatePtrTy = LLVM::LLVMPointerType::get(regStateTy);
1115 
1116  // Get reg state size.
1117  auto oneC = initBuilder.create<LLVM::ConstantOp>(
1118  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(1));
1119  auto regNull =
1120  initBuilder.create<LLVM::NullOp>(op->getLoc(), regStatePtrTy);
1121  auto regGep = initBuilder.create<LLVM::GEPOp>(
1122  op->getLoc(), regStatePtrTy, regNull, ArrayRef<Value>({oneC}));
1123  auto regSize =
1124  initBuilder.create<LLVM::PtrToIntOp>(op->getLoc(), i64Ty, regGep);
1125 
1126  // Malloc reg state.
1127  auto regMall = initBuilder
1128  .create<LLVM::CallOp>(op->getLoc(), i8PtrTy,
1129  SymbolRefAttr::get(mallFunc),
1130  ArrayRef<Value>({regSize}))
1131  .getResult(0);
1132  auto regMallBC = initBuilder.create<LLVM::BitcastOp>(
1133  op->getLoc(), regStatePtrTy, regMall);
1134  auto zeroB = initBuilder.create<LLVM::ConstantOp>(
1135  op->getLoc(), i1Ty, rewriter.getBoolAttr(false));
1136 
1137  // Zero-initialize reg state entries.
1138  for (size_t i = 0,
1139  e = regStateTy.cast<LLVM::LLVMStructType>().getBody().size();
1140  i < e; ++i) {
1141  size_t f = regStateTy.cast<LLVM::LLVMStructType>()
1142  .getBody()[i]
1143  .cast<LLVM::LLVMArrayType>()
1144  .getNumElements();
1145  for (size_t j = 0; j < f; ++j) {
1146  auto regIndexC = initBuilder.create<LLVM::ConstantOp>(
1147  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(i));
1148  auto triggerIndexC = initBuilder.create<LLVM::ConstantOp>(
1149  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(j));
1150  auto regGep = initBuilder.create<LLVM::GEPOp>(
1151  op->getLoc(), LLVM::LLVMPointerType::get(i1Ty), regMallBC,
1152  ArrayRef<Value>({zeroB, regIndexC, triggerIndexC}));
1153  initBuilder.create<LLVM::StoreOp>(op->getLoc(), zeroB, regGep);
1154  }
1155  }
1156 
1157  // Add reg state pointer to global state.
1158  initBuilder.create<LLVM::CallOp>(
1159  op->getLoc(), llvm::None, SymbolRefAttr::get(allocEntityFunc),
1160  ArrayRef<Value>({initStatePtr, owner, regMall}));
1161 
1162  // Index of the signal in the entity's signal table.
1163  int initCounter = 0;
1164  // Walk over the entity and generate mallocs for each one of its signals.
1165  child.walk([&](SigOp op) -> void {
1166  // if (auto sigOp = dyn_cast<SigOp>(op)) {
1167  auto underlyingTy = typeConverter->convertType(op.init().getType());
1168  // Get index constant of the signal in the entity's signal table.
1169  auto indexConst = initBuilder.create<LLVM::ConstantOp>(
1170  op.getLoc(), i32Ty, rewriter.getI32IntegerAttr(initCounter));
1171  initCounter++;
1172 
1173  // Clone and insert the operation that defines the signal's init
1174  // operand (assmued to be a constant/array op)
1175  auto defOp = op.init().getDefiningOp();
1176  auto initDef = recursiveCloneInit(initBuilder, defOp)->getResult(0);
1177  Value initDefCast = typeConverter->materializeTargetConversion(
1178  initBuilder, initDef.getLoc(),
1179  typeConverter->convertType(initDef.getType()), initDef);
1180 
1181  // Compute the required space to malloc.
1182  auto oneC = initBuilder.create<LLVM::ConstantOp>(
1183  op.getLoc(), i32Ty, rewriter.getI32IntegerAttr(1));
1184  auto twoC = initBuilder.create<LLVM::ConstantOp>(
1185  op.getLoc(), i64Ty, rewriter.getI32IntegerAttr(2));
1186  auto nullPtr = initBuilder.create<LLVM::NullOp>(
1187  op.getLoc(), LLVM::LLVMPointerType::get(underlyingTy));
1188  auto sizeGep = initBuilder.create<LLVM::GEPOp>(
1189  op.getLoc(), LLVM::LLVMPointerType::get(underlyingTy), nullPtr,
1190  ArrayRef<Value>(oneC));
1191  auto size =
1192  initBuilder.create<LLVM::PtrToIntOp>(op.getLoc(), i64Ty, sizeGep);
1193  // Malloc double the required space to make sure signal
1194  // shifts do not segfault.
1195  auto mallocSize =
1196  initBuilder.create<LLVM::MulOp>(op.getLoc(), i64Ty, size, twoC);
1197  std::array<Value, 1> margs({mallocSize});
1198  auto mall =
1199  initBuilder
1200  .create<LLVM::CallOp>(op.getLoc(), i8PtrTy,
1201  SymbolRefAttr::get(mallFunc), margs)
1202  .getResult(0);
1203 
1204  // Store the initial value.
1205  auto bitcast = initBuilder.create<LLVM::BitcastOp>(
1206  op.getLoc(), LLVM::LLVMPointerType::get(underlyingTy), mall);
1207 
1208  initBuilder.create<LLVM::StoreOp>(op.getLoc(), initDefCast, bitcast);
1209 
1210  // Get the amount of bytes required to represent an integer underlying
1211  // type. Use the whole size of the type if not an integer.
1212  Value passSize;
1213  if (auto intTy = underlyingTy.dyn_cast<IntegerType>()) {
1214  auto byteWidth = llvm::divideCeil(intTy.getWidth(), 8);
1215  passSize = initBuilder.create<LLVM::ConstantOp>(
1216  op.getLoc(), i64Ty, rewriter.getI64IntegerAttr(byteWidth));
1217  } else {
1218  passSize = size;
1219  }
1220 
1221  std::array<Value, 5> args(
1222  {initStatePtr, indexConst, owner, mall, passSize});
1223  auto sigIndex =
1224  initBuilder
1225  .create<LLVM::CallOp>(op.getLoc(), i32Ty,
1226  SymbolRefAttr::get(sigFunc), args)
1227  .getResult(0);
1228 
1229  // Add structured underlying type information.
1230  if (auto arrayTy = underlyingTy.dyn_cast<LLVM::LLVMArrayType>()) {
1231  auto zeroC = initBuilder.create<LLVM::ConstantOp>(
1232  op.getLoc(), i32Ty, rewriter.getI32IntegerAttr(0));
1233 
1234  auto numElements = initBuilder.create<LLVM::ConstantOp>(
1235  op.getLoc(), i32Ty,
1236  rewriter.getI32IntegerAttr(arrayTy.getNumElements()));
1237 
1238  // Get element size.
1239  auto null = initBuilder.create<LLVM::NullOp>(
1240  op.getLoc(), LLVM::LLVMPointerType::get(arrayTy));
1241  auto gepFirst = initBuilder.create<LLVM::GEPOp>(
1242  op.getLoc(), LLVM::LLVMPointerType::get(arrayTy.getElementType()),
1243  null, ArrayRef<Value>({zeroC, oneC}));
1244  auto toInt = initBuilder.create<LLVM::PtrToIntOp>(op.getLoc(), i32Ty,
1245  gepFirst);
1246 
1247  // Add information to the state.
1248  initBuilder.create<LLVM::CallOp>(
1249  op.getLoc(), llvm::None, SymbolRefAttr::get(addSigElemFunc),
1250  ArrayRef<Value>({initStatePtr, sigIndex, toInt, numElements}));
1251  } else if (auto structTy =
1252  underlyingTy.dyn_cast<LLVM::LLVMStructType>()) {
1253  auto zeroC = initBuilder.create<LLVM::ConstantOp>(
1254  op.getLoc(), i32Ty, rewriter.getI32IntegerAttr(0));
1255 
1256  auto null = initBuilder.create<LLVM::NullOp>(
1257  op.getLoc(), LLVM::LLVMPointerType::get(structTy));
1258  for (size_t i = 0, e = structTy.getBody().size(); i < e; ++i) {
1259  auto oneC = initBuilder.create<LLVM::ConstantOp>(
1260  op.getLoc(), i32Ty, rewriter.getI32IntegerAttr(1));
1261  auto indexC = initBuilder.create<LLVM::ConstantOp>(
1262  op.getLoc(), i32Ty, rewriter.getI32IntegerAttr(i));
1263 
1264  // Get pointer offset.
1265  auto gepElem = initBuilder.create<LLVM::GEPOp>(
1266  op.getLoc(), LLVM::LLVMPointerType::get(structTy.getBody()[i]),
1267  null, ArrayRef<Value>({zeroC, indexC}));
1268  auto elemToInt = initBuilder.create<LLVM::PtrToIntOp>(
1269  op.getLoc(), i32Ty, gepElem);
1270 
1271  // Get element size.
1272  auto elemNull = initBuilder.create<LLVM::NullOp>(
1273  op.getLoc(), LLVM::LLVMPointerType::get(structTy.getBody()[i]));
1274  auto gepElemSize = initBuilder.create<LLVM::GEPOp>(
1275  op.getLoc(), LLVM::LLVMPointerType::get(structTy.getBody()[i]),
1276  elemNull, ArrayRef<Value>({oneC}));
1277  auto elemSizeToInt = initBuilder.create<LLVM::PtrToIntOp>(
1278  op.getLoc(), i32Ty, gepElemSize);
1279 
1280  // Add information to the state.
1281  initBuilder.create<LLVM::CallOp>(
1282  op.getLoc(), llvm::None, SymbolRefAttr::get(addSigStructFunc),
1283  ArrayRef<Value>(
1284  {initStatePtr, sigIndex, elemToInt, elemSizeToInt}));
1285  }
1286  }
1287  });
1288  } else if (auto proc = module.lookupSymbol<ProcOp>(instOp.callee())) {
1289  // Handle process instantiation.
1290  auto sensesPtrTy = LLVM::LLVMPointerType::get(
1291  LLVM::LLVMArrayType::get(i1Ty, proc.getNumArguments()));
1292  auto procStatePtrTy =
1293  LLVM::LLVMPointerType::get(LLVM::LLVMStructType::getLiteral(
1294  rewriter.getContext(),
1295  {i32Ty, i32Ty, sensesPtrTy,
1296  getProcPersistenceTy(&getDialect(), typeConverter, proc)}));
1297 
1298  auto zeroC = initBuilder.create<LLVM::ConstantOp>(
1299  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(0));
1300  auto oneC = initBuilder.create<LLVM::ConstantOp>(
1301  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(1));
1302  auto twoC = initBuilder.create<LLVM::ConstantOp>(
1303  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(2));
1304 
1305  // Malloc space for the process state.
1306  auto procStateNullPtr =
1307  initBuilder.create<LLVM::NullOp>(op->getLoc(), procStatePtrTy);
1308  auto procStateGep = initBuilder.create<LLVM::GEPOp>(
1309  op->getLoc(), procStatePtrTy, procStateNullPtr,
1310  ArrayRef<Value>({oneC}));
1311  auto procStateSize = initBuilder.create<LLVM::PtrToIntOp>(
1312  op->getLoc(), i64Ty, procStateGep);
1313  std::array<Value, 1> procStateMArgs({procStateSize});
1314  auto procStateMall = initBuilder
1315  .create<LLVM::CallOp>(
1316  op->getLoc(), i8PtrTy,
1317  SymbolRefAttr::get(mallFunc), procStateMArgs)
1318  .getResult(0);
1319 
1320  auto procStateBC = initBuilder.create<LLVM::BitcastOp>(
1321  op->getLoc(), procStatePtrTy, procStateMall);
1322 
1323  // Store the initial resume index.
1324  auto resumeGep = initBuilder.create<LLVM::GEPOp>(
1325  op->getLoc(), LLVM::LLVMPointerType::get(i32Ty), procStateBC,
1326  ArrayRef<Value>({zeroC, oneC}));
1327  initBuilder.create<LLVM::StoreOp>(op->getLoc(), zeroC, resumeGep);
1328 
1329  // Malloc space for the senses table.
1330  auto sensesNullPtr =
1331  initBuilder.create<LLVM::NullOp>(op->getLoc(), sensesPtrTy);
1332  auto sensesGep = initBuilder.create<LLVM::GEPOp>(
1333  op->getLoc(), sensesPtrTy, sensesNullPtr, ArrayRef<Value>({oneC}));
1334  auto sensesSize =
1335  initBuilder.create<LLVM::PtrToIntOp>(op->getLoc(), i64Ty, sensesGep);
1336  std::array<Value, 1> senseMArgs({sensesSize});
1337  auto sensesMall =
1338  initBuilder
1339  .create<LLVM::CallOp>(op->getLoc(), i8PtrTy,
1340  SymbolRefAttr::get(mallFunc), senseMArgs)
1341  .getResult(0);
1342 
1343  auto sensesBC = initBuilder.create<LLVM::BitcastOp>(
1344  op->getLoc(), sensesPtrTy, sensesMall);
1345 
1346  // Set all initial senses to 1.
1347  for (size_t i = 0, e = sensesPtrTy.cast<LLVM::LLVMPointerType>()
1348  .getElementType()
1349  .cast<LLVM::LLVMArrayType>()
1350  .getNumElements();
1351  i < e; ++i) {
1352  auto oneB = initBuilder.create<LLVM::ConstantOp>(
1353  op->getLoc(), i1Ty, rewriter.getBoolAttr(true));
1354  auto gepInd = initBuilder.create<LLVM::ConstantOp>(
1355  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(i));
1356  auto senseGep = initBuilder.create<LLVM::GEPOp>(
1357  op->getLoc(), LLVM::LLVMPointerType::get(i1Ty), sensesBC,
1358  ArrayRef<Value>({zeroC, gepInd}));
1359  initBuilder.create<LLVM::StoreOp>(op->getLoc(), oneB, senseGep);
1360  }
1361 
1362  // Store the senses pointer in the process state.
1363  auto procStateSensesPtr = initBuilder.create<LLVM::GEPOp>(
1364  op->getLoc(), LLVM::LLVMPointerType::get(sensesPtrTy), procStateBC,
1365  ArrayRef<Value>({zeroC, twoC}));
1366  initBuilder.create<LLVM::StoreOp>(op->getLoc(), sensesBC,
1367  procStateSensesPtr);
1368 
1369  std::array<Value, 3> allocProcArgs({initStatePtr, owner, procStateMall});
1370  initBuilder.create<LLVM::CallOp>(op->getLoc(), llvm::None,
1371  SymbolRefAttr::get(allocProcFunc),
1372  allocProcArgs);
1373  }
1374 
1375  rewriter.eraseOp(op);
1376  return success();
1377  }
1378 };

1428 /// Convert an `llhd.prb` operation to LLVM dialect. The result is a library
1429 /// call to the
1430 /// `@probe_signal` function. The signal details are then extracted and used to
1431 /// load the final probe value.
1432 struct PrbOpConversion : public ConvertToLLVMPattern {
1433  explicit PrbOpConversion(MLIRContext *ctx, LLVMTypeConverter &typeConverter)
1434  : ConvertToLLVMPattern(llhd::PrbOp::getOperationName(), ctx,
1435  typeConverter) {}
1436 
1437  LogicalResult
1438  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
1439  ConversionPatternRewriter &rewriter) const override {
1440  // Get the adapted operands.
1441  PrbOpAdaptor transformed(operands);
1442  // Get the prb operation.
1443  auto prbOp = cast<PrbOp>(op);
1444 
1445  // Collect the used llvm types.
1446  auto resTy = prbOp.getType();
1447  auto finalTy = typeConverter->convertType(resTy);
1448 
1449  // Get the signal details from the signal struct.
1450  auto sigDetail = getSignalDetail(rewriter, &getDialect(), op->getLoc(),
1451  transformed.signal());
1452 
1453  if (resTy.isa<IntegerType>()) {
1454  // Get the amount of bytes to load. An extra byte is always loaded to
1455  // cover the case where a subsignal spans halfway in the last byte.
1456  int resWidth = resTy.getIntOrFloatBitWidth();
1457  int loadWidth = (llvm::divideCeil(resWidth, 8) + 1) * 8;
1458  auto loadTy = IntegerType::get(rewriter.getContext(), loadWidth);
1459 
1460  auto bitcast = rewriter.create<LLVM::BitcastOp>(
1461  op->getLoc(), LLVM::LLVMPointerType::get(loadTy), sigDetail[0]);
1462  auto loadSig =
1463  rewriter.create<LLVM::LoadOp>(op->getLoc(), loadTy, bitcast);
1464 
1465  // Shift the loaded value by the offset and truncate to the final width.
1466  auto trOff = adjustBitWidth(op->getLoc(), rewriter, loadTy, sigDetail[1]);
1467  auto shifted =
1468  rewriter.create<LLVM::LShrOp>(op->getLoc(), loadTy, loadSig, trOff);
1469  rewriter.replaceOpWithNewOp<LLVM::TruncOp>(op, finalTy, shifted);
1470 
1471  return success();
1472  }
1473 
1474  if (resTy.isa<hw::ArrayType, hw::StructType>()) {
1475  auto bitcast = rewriter.create<LLVM::BitcastOp>(
1476  op->getLoc(), LLVM::LLVMPointerType::get(finalTy), sigDetail[0]);
1477  rewriter.replaceOpWithNewOp<LLVM::LoadOp>(op, finalTy, bitcast);
1478 
1479  return success();
1480  }
1481 
1482  return failure();
1483  }
1484 };

1488 /// Convert an `llhd.drv` operation to LLVM dialect. The result is a library
1489 /// call to the
1490 /// `@driveSignal` function, which declaration is inserted at the beginning of
1491 /// the module if missing. The required arguments are either generated or
1492 /// fetched.
1493 struct DrvOpConversion : public ConvertToLLVMPattern {
1494  explicit DrvOpConversion(MLIRContext *ctx, LLVMTypeConverter &typeConverter)
1495  : ConvertToLLVMPattern(llhd::DrvOp::getOperationName(), ctx,
1496  typeConverter) {}
1497 
1498  LogicalResult
1499  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
1500  ConversionPatternRewriter &rewriter) const override {
1501  // Get the adapted operands.
1502  DrvOpAdaptor transformed(operands);
1503  // Get the drive operation.
1504  auto drvOp = cast<DrvOp>(op);
1505  // Get the parent module.
1506  auto module = op->getParentOfType<ModuleOp>();
1507 
1508  // Collect used llvm types.
1509  auto voidTy = getVoidType();
1510  auto i8PtrTy = getVoidPtrType();
1511  auto i1Ty = IntegerType::get(rewriter.getContext(), 1);
1512  auto i32Ty = IntegerType::get(rewriter.getContext(), 32);
1513  auto i64Ty = IntegerType::get(rewriter.getContext(), 64);
1514  auto sigTy = getLLVMSigType(&getDialect());
1515 
1516  // Get or insert the drive library call.
1517  auto drvFuncTy = LLVM::LLVMFunctionType::get(
1518  voidTy, {i8PtrTy, LLVM::LLVMPointerType::get(sigTy), i8PtrTy, i64Ty,
1519  i64Ty, i64Ty, i64Ty});
1520  auto drvFunc = getOrInsertFunction(module, rewriter, op->getLoc(),
1521  "driveSignal", drvFuncTy);
1522 
1523  // Get the state pointer from the function arguments.
1524  Value statePtr = op->getParentOfType<LLVM::LLVMFuncOp>().getArgument(0);
1525 
1526  // Get signal width.
1527  Value sigWidth;
1528  auto underlyingTy = drvOp.value().getType();
1529  if (isArrayOrStruct(underlyingTy)) {
1530  auto llvmPtrTy =
1531  LLVM::LLVMPointerType::get(typeConverter->convertType(underlyingTy));
1532  auto oneC = rewriter.create<LLVM::ConstantOp>(
1533  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(1));
1534  auto eightC = rewriter.create<LLVM::ConstantOp>(
1535  op->getLoc(), i64Ty, rewriter.getI64IntegerAttr(8));
1536  auto nullPtr = rewriter.create<LLVM::NullOp>(op->getLoc(), llvmPtrTy);
1537  auto gepOne = rewriter.create<LLVM::GEPOp>(
1538  op->getLoc(), llvmPtrTy, nullPtr, ArrayRef<Value>(oneC));
1539  auto toInt =
1540  rewriter.create<LLVM::PtrToIntOp>(op->getLoc(), i64Ty, gepOne);
1541  sigWidth = rewriter.create<LLVM::MulOp>(op->getLoc(), toInt, eightC);
1542  } else {
1543  sigWidth = rewriter.create<LLVM::ConstantOp>(
1544  op->getLoc(), i64Ty,
1545  rewriter.getI64IntegerAttr(underlyingTy.getIntOrFloatBitWidth()));
1546  }
1547 
1548  // Insert enable comparison. Skip if the enable operand is 0.
1549  if (auto gate = drvOp.enable()) {
1550  auto block = op->getBlock();
1551  auto continueBlock =
1552  rewriter.splitBlock(rewriter.getInsertionBlock(), op->getIterator());
1553  auto drvBlock = rewriter.createBlock(continueBlock);
1554  rewriter.setInsertionPointToEnd(drvBlock);
1555  rewriter.create<LLVM::BrOp>(op->getLoc(), ValueRange(), continueBlock);
1556 
1557  rewriter.setInsertionPointToEnd(block);
1558  auto oneC = rewriter.create<LLVM::ConstantOp>(
1559  op->getLoc(), i1Ty, rewriter.getI16IntegerAttr(1));
1560  auto cmp = rewriter.create<LLVM::ICmpOp>(
1561  op->getLoc(), LLVM::ICmpPredicate::eq, transformed.enable(), oneC);
1562  rewriter.create<LLVM::CondBrOp>(op->getLoc(), cmp, drvBlock,
1563  continueBlock);
1564 
1565  rewriter.setInsertionPointToStart(drvBlock);
1566  }
1567 
1568  Type valTy = typeConverter->convertType(transformed.value().getType());
1569  Value castVal = typeConverter->materializeTargetConversion(
1570  rewriter, transformed.value().getLoc(), valTy, transformed.value());
1571 
1572  auto oneConst = rewriter.create<LLVM::ConstantOp>(
1573  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(1));
1574  auto alloca = rewriter.create<LLVM::AllocaOp>(
1575  op->getLoc(), LLVM::LLVMPointerType::get(valTy), oneConst, 4);
1576  rewriter.create<LLVM::StoreOp>(op->getLoc(), castVal, alloca);
1577  auto bc = rewriter.create<LLVM::BitcastOp>(op->getLoc(), i8PtrTy, alloca);
1578 
1579  // Get the time values.
1580  auto realTime = rewriter.create<LLVM::ExtractValueOp>(
1581  op->getLoc(), i64Ty, transformed.time(), rewriter.getI32ArrayAttr(0));
1582  auto delta = rewriter.create<LLVM::ExtractValueOp>(
1583  op->getLoc(), i64Ty, transformed.time(), rewriter.getI32ArrayAttr(1));
1584  auto eps = rewriter.create<LLVM::ExtractValueOp>(
1585  op->getLoc(), i64Ty, transformed.time(), rewriter.getI32ArrayAttr(2));
1586 
1587  // Define the driveSignal library call arguments.
1588  std::array<Value, 7> args(
1589  {statePtr, transformed.signal(), bc, sigWidth, realTime, delta, eps});
1590  // Create the library call.
1591  rewriter.create<LLVM::CallOp>(op->getLoc(), llvm::None,
1592  SymbolRefAttr::get(drvFunc), args);
1593 
1594  rewriter.eraseOp(op);
1595  return success();
1596  }

1601 /// Convert an `llhd.reg` operation to LLVM dialect. This generates a series of
1602 /// comparisons (blocks) that end up driving the signal with the arguments of
1603 /// the first matching trigger from the trigger list.
1604 struct RegOpConversion : public ConvertToLLVMPattern {
1605  explicit RegOpConversion(MLIRContext *ctx, LLVMTypeConverter &typeConverter,
1606  size_t &regCounter)
1607  : ConvertToLLVMPattern(RegOp::getOperationName(), ctx, typeConverter),
1608  regCounter(regCounter) {}
1609 
1610  LogicalResult
1611  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
1612  ConversionPatternRewriter &rewriter) const override {
1613  auto regOp = cast<RegOp>(op);
1614  RegOpAdaptor transformed(operands, op->getAttrDictionary());
1615 
1616  auto i1Ty = IntegerType::get(rewriter.getContext(), 1);
1617  auto i32Ty = IntegerType::get(rewriter.getContext(), 32);
1618 
1619  auto func = op->getParentOfType<LLVM::LLVMFuncOp>();
1620 
1621  // Retrieve and update previous trigger values for rising/falling edge
1622  // detection.
1623  size_t triggerIndex = 0;
1624  SmallVector<Value, 4> prevTriggers;
1625  for (int i = 0, e = regOp.values().size(); i < e; ++i) {
1626  auto mode = regOp.getRegModeAt(i);
1627  if (mode == RegMode::both || mode == RegMode::fall ||
1628  mode == RegMode::rise) {
1629  auto zeroC = rewriter.create<LLVM::ConstantOp>(
1630  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(0));
1631  auto regIndexC = rewriter.create<LLVM::ConstantOp>(
1632  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(regCounter));
1633  auto triggerIndexC = rewriter.create<LLVM::ConstantOp>(
1634  op->getLoc(), i32Ty, rewriter.getI32IntegerAttr(triggerIndex++));
1635  auto gep = rewriter.create<LLVM::GEPOp>(
1636  op->getLoc(), LLVM::LLVMPointerType::get(i1Ty), func.getArgument(1),
1637  ArrayRef<Value>({zeroC, regIndexC, triggerIndexC}));
1638  prevTriggers.push_back(
1639  rewriter.create<LLVM::LoadOp>(op->getLoc(), gep));
1640  rewriter.create<LLVM::StoreOp>(op->getLoc(), transformed.triggers()[i],
1641  gep);
1642  }
1643  }
1644 
1645  // Create blocks for drive and continue.
1646  auto block = op->getBlock();
1647  auto continueBlock = block->splitBlock(op);
1648 
1649  auto drvBlock = rewriter.createBlock(continueBlock);
1650  auto valArg = drvBlock->addArgument(transformed.values()[0].getType(),
1651  transformed.values()[0].getLoc());
1652  auto delayArg = drvBlock->addArgument(transformed.delays()[0].getType(),
1653  transformed.delays()[0].getLoc());
1654  auto gateArg = drvBlock->addArgument(i1Ty, rewriter.getUnknownLoc());
1655 
1656  // Create a drive with the block arguments.
1657  rewriter.setInsertionPointToStart(drvBlock);
1658  rewriter.create<DrvOp>(op->getLoc(), regOp.signal(), valArg, delayArg,
1659  gateArg);
1660  rewriter.create<LLVM::BrOp>(op->getLoc(), ValueRange(), continueBlock);
1661 
1662  int j = prevTriggers.size() - 1;
1663  // Create a comparison block for each of the reg tuples.
1664  for (int i = regOp.values().size() - 1, e = i; i >= 0; --i) {
1665  auto cmpBlock = rewriter.createBlock(block->getNextNode());
1666  rewriter.setInsertionPointToStart(cmpBlock);
1667 
1668  Value gate;
1669  if (regOp.hasGate(i)) {
1670  gate = regOp.getGateAt(i);
1671  } else {
1672  gate = rewriter.create<LLVM::ConstantOp>(op->getLoc(), i1Ty,
1673  rewriter.getBoolAttr(true));
1674  }
1675 
1676  auto drvArgs = std::array<Value, 3>(
1677  {transformed.values()[i], transformed.delays()[i], gate});
1678 
1679  RegMode mode = regOp.getRegModeAt(i);
1680 
1681  // Create comparison constants for all modes other than both.
1682  Value rhs;
1683  if (mode == RegMode::low || mode == RegMode::fall) {
1684  rhs = rewriter.create<LLVM::ConstantOp>(op->getLoc(), i1Ty,
1685  rewriter.getBoolAttr(false));
1686  } else if (mode == RegMode::high || mode == RegMode::rise) {
1687  rhs = rewriter.create<LLVM::ConstantOp>(op->getLoc(), i1Ty,
1688  rewriter.getBoolAttr(true));
1689  }
1690 
1691  // Create comparison for non-both modes.
1692  Value comp;
1693  if (rhs)
1694  comp =
1695  rewriter.create<LLVM::ICmpOp>(op->getLoc(), LLVM::ICmpPredicate::eq,
1696  transformed.triggers()[i], rhs);
1697 
1698  // Create comparison for modes needing more than one state of the trigger.
1699  Value brCond;
1700  if (mode == RegMode::rise || mode == RegMode::fall ||
1701  mode == RegMode::both) {
1702 
1703  auto cmpPrev = rewriter.create<LLVM::ICmpOp>(
1704  op->getLoc(), LLVM::ICmpPredicate::ne, transformed.triggers()[i],
1705  prevTriggers[j--]);
1706  if (mode == RegMode::both)
1707  brCond = cmpPrev;
1708  else
1709  brCond =
1710  rewriter.create<LLVM::AndOp>(op->getLoc(), i1Ty, comp, cmpPrev);
1711  } else {
1712  brCond = comp;
1713  }
1714 
1715  Block *nextBlock;
1716  nextBlock = cmpBlock->getNextNode();
1717  // Don't go to next block for last comparison's false branch (skip the
1718  // drive block).
1719  if (i == e)
1720  nextBlock = continueBlock;
1721 
1722  rewriter.create<LLVM::CondBrOp>(op->getLoc(), brCond, drvBlock, drvArgs,
1723  nextBlock, ValueRange());
1724  }
1725 
1726  rewriter.setInsertionPointToEnd(block);
1727  rewriter.create<LLVM::BrOp>(op->getLoc(), ArrayRef<Value>(),
1728  block->getNextNode());
1729 
1730  rewriter.eraseOp(op);
1731 
1732  ++regCounter;
1733 
1734  return success();
1735  }
1736 
1737 private:
1738  size_t &regCounter;
1739 }; // namespace

1747 /// Convert an `llhd.shr` operation to LLVM dialect. All the operands are
1748 /// extended to the width obtained by combining the hidden and base values. This
1749 /// combined value is then shifted (exposing the hidden value) and truncated to
1750 /// the base length
1751 struct ShrOpConversion : public ConvertToLLVMPattern {
1752  explicit ShrOpConversion(MLIRContext *ctx, LLVMTypeConverter &typeConverter)
1753  : ConvertToLLVMPattern(llhd::ShrOp::getOperationName(), ctx,
1754  typeConverter) {}
1755 
1756  LogicalResult
1757  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
1758  ConversionPatternRewriter &rewriter) const override {
1759 
1760  ShrOpAdaptor transformed(operands);
1761  auto shrOp = cast<ShrOp>(op);
1762 
1763  if (auto resTy = shrOp.result().getType().dyn_cast<IntegerType>()) {
1764  // Get width of the base and hidden values combined.
1765  auto baseWidth = shrOp.getType().getIntOrFloatBitWidth();
1766  auto hdnWidth = shrOp.hidden().getType().getIntOrFloatBitWidth();
1767  auto full = baseWidth + hdnWidth;
1768 
1769  auto tmpTy = IntegerType::get(rewriter.getContext(), full);
1770 
1771  // Extend all operands the combined width.
1772  auto baseZext =
1773  adjustBitWidth(op->getLoc(), rewriter, tmpTy, transformed.base());
1774  auto hdnZext =
1775  adjustBitWidth(op->getLoc(), rewriter, tmpTy, transformed.hidden());
1776  auto amntZext =
1777  adjustBitWidth(op->getLoc(), rewriter, tmpTy, transformed.amount());
1778 
1779  // Shift the hidden operand such that it can be prepended to the full
1780  // value.
1781  auto hdnShAmnt = rewriter.create<LLVM::ConstantOp>(
1782  op->getLoc(), tmpTy,
1783  rewriter.getIntegerAttr(rewriter.getIntegerType(full), baseWidth));
1784  auto hdnSh =
1785  rewriter.create<LLVM::ShlOp>(op->getLoc(), tmpTy, hdnZext, hdnShAmnt);
1786 
1787  // Combine the base and hidden values.
1788  auto combined =
1789  rewriter.create<LLVM::OrOp>(op->getLoc(), tmpTy, hdnSh, baseZext);
1790 
1791  // Perform the right shift.
1792  auto shifted = rewriter.create<LLVM::LShrOp>(op->getLoc(), tmpTy,
1793  combined, amntZext);
1794 
1795  // Truncate to final width.
1796  rewriter.replaceOpWithNewOp<LLVM::TruncOp>(
1797  op, transformed.base().getType(), shifted);
1798 
1799  return success();
1800  }
1801  if (auto arrTy = shrOp.result().getType().dyn_cast<hw::ArrayType>()) {
1802 
1803  auto combined = rewriter.create<hw::ArrayConcatOp>(
1804  op->getLoc(), ValueRange({shrOp.hidden(), shrOp.base()}));
1805  rewriter.replaceOpWithNewOp<hw::ArraySliceOp>(op, arrTy, combined,
1806  transformed.amount());
1807 
1808  return success();
1809  }
1810 
1811  return failure();
1812  }
1813 };

1817 /// Convert an `llhd.shl` operation to LLVM dialect. All the operands are
1818 /// extended to the width obtained by combining the hidden and base values. This
1819 /// combined value is then shifted right by `hidden_width - amount` (exposing
1820 /// the hidden value) and truncated to the base length

1831  auto shlOp = cast<ShlOp>(op);
1832  assert(!shlOp.getType().isa<llhd::SigType>() && "sig not yet supported");
1833 
1834  // Get the width of the base and hidden operands combined.
1835  auto baseWidth = shlOp.getType().getIntOrFloatBitWidth();
1836  auto hdnWidth = shlOp.hidden().getType().getIntOrFloatBitWidth();

1844  auto hdnZext =

1847  adjustBitWidth(op->getLoc(), rewriter, tmpTy, transformed.amount());

1867  rewriter.create<LLVM::LShrOp>(op->getLoc(), tmpTy, combined, shrAmnt);
1868

1870  rewriter.replaceOpWithNewOp<LLVM::TruncOp>(op, transformed.base().getType(),
1871  shifted);
1872 
1873  return success();
1874  }
