54 StringAttr FieldNameResolver::getRenamedFieldName(StringAttr fieldName) {
55  auto it = renamedFieldNames.find(fieldName);
56  if (it != renamedFieldNames.end())
57  return it->second;
58 
59  // If a field name is not verilog name or used already, we have to rename it.
60  bool hasToBeRenamed = !sv::isNameValid(fieldName.getValue()) ||
61  usedFieldNames.count(fieldName.getValue());
62 
63  if (!hasToBeRenamed) {
64  setRenamedFieldName(fieldName, fieldName);
65  return fieldName;
66  }
67 
68  StringRef newFieldName = sv::legalizeName(
69  fieldName.getValue(), usedFieldNames, nextGeneratedNameID);
70 
71  auto newFieldNameAttr = StringAttr::get(fieldName.getContext(), newFieldName);
72 
73  setRenamedFieldName(fieldName, newFieldNameAttr);
74  return newFieldNameAttr;
75 }

116 /// Construct a GlobalNameResolver and do the initial scan to populate and
117 /// unique the module/interfaces and port/parameter names.
118 GlobalNameResolver::GlobalNameResolver(mlir::ModuleOp topLevel) {
119  // Register the names of external modules which we cannot rename. This has to
120  // occur in a first pass separate from the modules and interfaces which we are
121  // actually allowed to rename, in order to ensure that we don't accidentally
122  // rename a module that later collides with an extern module.
123  for (auto &op : *topLevel.getBody()) {
124  // Note that external modules *often* have name collisions, because they
125  // correspond to the same verilog module with different parameters.
126  if (isa<HWModuleExternOp>(op) || isa<HWModuleGeneratedOp>(op)) {
127  auto name = getVerilogModuleNameAttr(&op).getValue();
128  if (!sv::isNameValid(name))
129  op.emitError("name \"")
130  << name << "\" is not allowed in Verilog output";
131  globalNameResolver.insertUsedName(name);
132  }
133  }
134 
135  // Legalize module and interface names.
136  for (auto &op : *topLevel.getBody()) {
137  if (auto module = dyn_cast<HWModuleOp>(op)) {
138  legalizeModuleNames(module);
139  continue;
140  }
141 
142  // Legalize the name of the interface itself, as well as any signals and
143  // modports within it.
144  if (auto interface = dyn_cast<InterfaceOp>(op)) {
145  legalizeInterfaceNames(interface);
146  continue;
147  }
148  }
149 }

151 /// Check to see if the port names of the specified module conflict with
152 /// keywords or themselves. If so, add the replacement names to
153 /// globalNameTable.
154 void GlobalNameResolver::legalizeModuleNames(HWModuleOp module) {
155  MLIRContext *ctxt = module.getContext();
156  // If the module's symbol itself conflicts, then set a "verilogName" attribute
157  // on the module to reflect the name we need to use.
158  StringRef oldName = module.getName();
159  auto newName = globalNameResolver.getLegalName(oldName);
160  if (newName != oldName)
161  module->setAttr("verilogName", StringAttr::get(ctxt, newName));
162 
163  NameCollisionResolver nameResolver;
164  auto verilogNameAttr = StringAttr::get(ctxt, "hw.verilogName");
165  // Legalize the port names.
166  size_t portIdx = 0;
167  SmallVector<Attribute, 4> argNames, resultNames;
168  for (const PortInfo &port : getAllModulePortInfos(module)) {
169  auto newName = nameResolver.getLegalName(port.name);
170  if (newName != port.name.getValue()) {
171  if (port.isOutput())
172  module.setResultAttr(port.argNum, verilogNameAttr,
173  StringAttr::get(ctxt, newName));
174  else
175  module.setArgAttr(port.argNum, verilogNameAttr,
176  StringAttr::get(ctxt, newName));
177  }
178  ++portIdx;
179  }
180 
181  // Legalize the parameter names.
182  for (auto param : module.parameters()) {
183  auto paramAttr = param.cast<ParamDeclAttr>();
184  auto newName = nameResolver.getLegalName(paramAttr.getName());
185  if (newName != paramAttr.getName().getValue())
186  globalNameTable.addRenamedParam(module, paramAttr.getName(), newName);
187  }
188 
189  // Legalize the value names.
190  module.walk([&](Operation *op) {
191  if (!isa<HWModuleOp>(op))
192  if (auto nameAttr = getDeclarationName(op)) {
193  auto newName = nameResolver.getLegalName(nameAttr);
194  if (newName != nameAttr.getValue()) {
195  op->setAttr(verilogNameAttr, StringAttr::get(ctxt, newName));
196  }
197  }
198  });
199 }

201 void GlobalNameResolver::legalizeInterfaceNames(InterfaceOp interface) {
202  MLIRContext *ctxt = interface.getContext();
203  auto verilogNameAttr = StringAttr::get(ctxt, "hw.verilogName");
204  auto newName = globalNameResolver.getLegalName(interface.getName());
205  if (newName != interface.getName())
206  interface->setAttr(verilogNameAttr, StringAttr::get(ctxt, newName));
207 
208  NameCollisionResolver localNames;
209  // Rename signals and modports.
210  for (auto &op : *interface.getBodyBlock()) {
211  if (isa<InterfaceSignalOp, InterfaceModportOp>(op)) {
212  auto name = SymbolTable::getSymbolName(&op).getValue();
213  auto newName = localNames.getLegalName(name);
214  if (newName != name)
215  op.setAttr(verilogNameAttr, StringAttr::get(ctxt, newName));
216  }
217  }
218 }
