26 /// This class provides logic-exporting functions for the implemented
27 /// operations, along with a dispatcher to visit the correct handler.
28 struct Visitor : public hw::StmtVisitor<Visitor, LogicalResult>,
29  public hw::TypeOpVisitor<Visitor, LogicalResult>,
30  public comb::CombinationalVisitor<Visitor, LogicalResult> {
31  using hw::StmtVisitor<Visitor, LogicalResult>::visitStmt;
32  using hw::TypeOpVisitor<Visitor, LogicalResult>::visitTypeOp;
33  using comb::CombinationalVisitor<Visitor, LogicalResult>::visitComb;
34  friend class hw::StmtVisitor<Visitor, LogicalResult>;
35  friend class hw::TypeOpVisitor<Visitor, LogicalResult>;
36  friend class comb::CombinationalVisitor<Visitor, LogicalResult>;
37 
38  Visitor(Solver::Circuit *circuit) : circuit(circuit) {}
39  Solver::Circuit *circuit;

41  /// Handles `builtin.module` logic exporting.
42  LogicalResult visit(ModuleOp op, llvm::StringRef targetModule) {
43  for (auto hwModule : op.getOps<hw::HWModuleOp>()) {
44  if (targetModule.empty() || hwModule.getName() == targetModule) {
45  LLVM_DEBUG(llvm::dbgs()
46  << "Using module `" << hwModule.getName() << "`\n");
47  return visit(hwModule);
48  }
49  }
50  op.emitError("module not found");
51  return failure();
52  }
53 
54  /// Handles `hw.module` logic exporting.
55  LogicalResult visit(hw::HWModuleOp op) {
56  for (auto argument : op.getBodyBlock()->getArguments())
57  circuit->addInput(argument);
58  for (auto &op : op.getOps())
59  if (failed(dispatch(&op)))
60  return failure();
61  return success();
62  }
63 
64  LogicalResult visitUnhandledOp(Operation *op) {
65  op->emitOpError("not supported");
66  return failure();
67  }
68 
69  /// Dispatches an operation to the appropriate visit function.
70  LogicalResult dispatch(Operation *op) { return dispatchStmtVisitor(op); }

76  LogicalResult visitStmt(hw::InstanceOp op) {
77  Operation *moduleOp =
78  SymbolTable::lookupNearestSymbolFrom(op, op.getModuleNameAttr());
79 
80  if (auto hwModule = llvm::dyn_cast<hw::HWModuleOp>(moduleOp)) {
81  circuit->addInstance(op.getInstanceName(), hwModule, op->getOperands(),
82  op->getResults());
83  return success();
84  }
85  op.emitError("instantiated module `" + op.getModuleName() +
86  "` is not an HW module");
87  return failure();
88  }
89 
90  LogicalResult visitStmt(hw::InstanceChoiceOp op) {
91  return op.emitError("instance choices are not supported");
92  }
93 
94  LogicalResult visitStmt(hw::OutputOp op) {
95  for (auto operand : op.getOperands())
96  circuit->addOutput(operand);
97  return success();
98  }

101  return dispatchTypeOpVisitor(op);
102  }
103  LogicalResult visitUnhandledStmt(Operation *op) {
104  return visitUnhandledOp(op);
105  }

127  // Visit a comb operation with a variadic number of operands.
128  template <typename OpTy, typename FnTy>
129  LogicalResult visitVariadicCombOp(OpTy op, FnTy fn) {
130  if (!op.getTwoState())
131  return op.emitOpError("without 'bin' unsupported");
132  (circuit->*fn)(op.getResult(), op.getOperands());
133  return success();
134  }
135 
136  // Visit a comb operation with two operands.
137  template <typename OpTy, typename FnTy>
138  LogicalResult visitBinaryCombOp(OpTy op, FnTy fn) {
139  if (!op.getTwoState())
140  return op.emitOpError("without 'bin' unsupported");
141  (circuit->*fn)(op.getResult(), op.getLhs(), op.getRhs());
142  return success();
143  }
144 
145  // Visit a comb operation with one operand.
146  template <typename OpTy, typename FnTy>
147  LogicalResult visitUnaryCombOp(OpTy op, FnTy fn) {
148  if (!op.getTwoState())
149  return op.emitOpError("without 'bin' unsupported");
150  (circuit->*fn)(op.getResult(), op.getInput());
151  return success();
152  }
153

155  return visitVariadicCombOp(op, &Solver::Circuit::performAdd);
156  }
157  LogicalResult visitComb(comb::AndOp op) {
158  return visitVariadicCombOp(op, &Solver::Circuit::performAnd);
159  }
160  LogicalResult visitComb(comb::ConcatOp op) {
161  circuit->performConcat(op.getResult(), op.getOperands());
162  return success();
163  }
164  LogicalResult visitComb(comb::DivSOp op) {
165  return visitBinaryCombOp(op, &Solver::Circuit::performDivS);
166  }

168  return visitBinaryCombOp(op, &Solver::Circuit::performDivU);
169  }
170  LogicalResult visitComb(comb::ExtractOp op) {
171  circuit->performExtract(op.getResult(), op.getInput(), op.getLowBit());
172  return success();
173  }
174  LogicalResult visitComb(comb::ICmpOp op) {
175  if (!op.getTwoState())

177  return circuit->performICmp(op.getResult(), op.getPredicate(), op.getLhs(),
178  op.getRhs());
179  }
180  LogicalResult visitComb(comb::ModSOp op) {
181  return visitBinaryCombOp(op, &Solver::Circuit::performModS);
182  }
183  LogicalResult visitComb(comb::ModUOp op) {
184  return visitBinaryCombOp(op, &Solver::Circuit::performModU);
185  }
186  LogicalResult visitComb(comb::MulOp op) {
187  return visitVariadicCombOp(op, &Solver::Circuit::performMul);
188  }
189  LogicalResult visitComb(comb::MuxOp op) {
190  if (!op.getTwoState())
191  return op.emitOpError("without 'bin' unsupported");
192  circuit->performMux(op.getResult(), op.getCond(), op.getTrueValue(),
193  op.getFalseValue());
194  return success();
195  }
196  LogicalResult visitComb(comb::OrOp op) {
197  return visitVariadicCombOp(op, &Solver::Circuit::performOr);
198  }
199  LogicalResult visitComb(comb::ParityOp op) {
200  return visitUnaryCombOp(op, &Solver::Circuit::performParity);
201  }
202  LogicalResult visitComb(comb::ReplicateOp op) {
203  circuit->performReplicate(op.getResult(), op.getInput());
204  return success();
205  }
206  LogicalResult visitComb(comb::ShlOp op) {
207  return visitBinaryCombOp(op, &Solver::Circuit::performShl);
208  }
209  LogicalResult visitComb(comb::ShrSOp op) {
210  return visitBinaryCombOp(op, &Solver::Circuit::performShrS);
211  }
212  LogicalResult visitComb(comb::ShrUOp op) {
213  return visitBinaryCombOp(op, &Solver::Circuit::performShrU);
214  }
215  LogicalResult visitComb(comb::SubOp op) {
216  return visitVariadicCombOp(op, &Solver::Circuit::performSub);
217  }
218  LogicalResult visitComb(comb::XorOp op) {
219  return visitVariadicCombOp(op, &Solver::Circuit::performXor);
