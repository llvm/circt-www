69 /// Apply a new annotation to a resolved target. This handles ports,
70 /// aggregates, modules, wires, etc.
71 static void addAnnotation(AnnoTarget ref, unsigned fieldIdx,
72  ArrayRef<NamedAttribute> anno) {
73  auto *context = ref.getOp()->getContext();
74  DictionaryAttr annotation;
75  if (fieldIdx) {
76  SmallVector<NamedAttribute> annoField(anno.begin(), anno.end());
77  annoField.emplace_back(
78  StringAttr::get(context, "circt.fieldID"),
79  IntegerAttr::get(IntegerType::get(context, 32, IntegerType::Signless),
80  fieldIdx));
81  annotation = DictionaryAttr::get(context, annoField);
82  } else {
83  annotation = DictionaryAttr::get(context, anno);
84  }
85 
86  if (ref.isa<OpAnnoTarget>()) {
87  auto newAnno = appendArrayAttr(getAnnotationsFrom(ref.getOp()), annotation);
88  ref.getOp()->setAttr(getAnnotationAttrName(), newAnno);
89  return;
90  }
91 
92  auto portRef = ref.cast<PortAnnoTarget>();
93  auto portAnnoRaw = ref.getOp()->getAttr(getPortAnnotationAttrName());
94  ArrayAttr portAnno = portAnnoRaw.dyn_cast_or_null<ArrayAttr>();
95  if (!portAnno || portAnno.size() != getNumPorts(ref.getOp())) {
96  SmallVector<Attribute> emptyPortAttr(
97  getNumPorts(ref.getOp()),
98  ArrayAttr::get(ref.getOp()->getContext(), {}));
99  portAnno = ArrayAttr::get(ref.getOp()->getContext(), emptyPortAttr);
100  }
101  portAnno = replaceArrayAttrElement(
102  portAnno, portRef.getPortNo(),
103  appendArrayAttr(dyn_cast<ArrayAttr>(portAnno[portRef.getPortNo()]),
104  annotation));
105  ref.getOp()->setAttr("portAnnotations", portAnno);
106 }

108 /// Make an anchor for a non-local annotation. Use the expanded path to build
109 /// the module and name list in the anchor.
110 static FlatSymbolRefAttr buildNLA(const AnnoPathValue &target,
111  ApplyState &state) {
112  OpBuilder b(state.circuit.getBodyRegion());
113  SmallVector<Attribute> insts;
114  for (auto inst : target.instances) {
115  insts.push_back(OpAnnoTarget(inst).getNLAReference(
116  state.getNamespace(inst->getParentOfType<FModuleLike>())));
117  }
118 
119  insts.push_back(
120  FlatSymbolRefAttr::get(target.ref.getModule().getModuleNameAttr()));
121 
122  auto instAttr = ArrayAttr::get(state.circuit.getContext(), insts);
123  return state.hierPathCache.getRefFor(instAttr);
124 }

147 /// Implementation of standard resolution. First parses the target path, then
148 /// resolves it.
149 static std::optional<AnnoPathValue> stdResolveImpl(StringRef rawPath,
150  ApplyState &state) {
151  auto pathStr = canonicalizeTarget(rawPath);
152  StringRef path{pathStr};
153 
154  auto tokens = tokenizePath(path);
155  if (!tokens) {
156  mlir::emitError(state.circuit.getLoc())
157  << "Cannot tokenize annotation path " << rawPath;
158  return {};
159  }
160 
161  return resolveEntities(*tokens, state.circuit, state.symTbl,
162  state.targetCaches);
163 }

165 /// (SFC) FIRRTL SingleTargetAnnotation resolver. Uses the 'target' field of
166 /// the annotation with standard parsing to resolve the path. This requires
167 /// 'target' to exist and be normalized (per docs/FIRRTLAnnotations.md).
168 std::optional<AnnoPathValue> circt::firrtl::stdResolve(DictionaryAttr anno,
169  ApplyState &state) {
170  auto target = anno.getNamed("target");
171  if (!target) {
172  mlir::emitError(state.circuit.getLoc())
173  << "No target field in annotation " << anno;
174  return {};
175  }
176  if (!isa<StringAttr>(target->getValue())) {
177  mlir::emitError(state.circuit.getLoc())
178  << "Target field in annotation doesn't contain string " << anno;
179  return {};
180  }
181  return stdResolveImpl(cast<StringAttr>(target->getValue()).getValue(), state);
182 }

198 /// An applier which puts the annotation on the target and drops the 'target'
199 /// field from the annotation. Optionally handles non-local annotations.
200 LogicalResult circt::firrtl::applyWithoutTargetImpl(const AnnoPathValue &target,
201 
202  DictionaryAttr anno,
203  ApplyState &state,
204  bool allowNonLocal) {
205  if (!allowNonLocal && !target.isLocal()) {
206  Annotation annotation(anno);
207  auto diag = mlir::emitError(target.ref.getOp()->getLoc())
208  << "is targeted by a non-local annotation \""
209  << annotation.getClass() << "\" with target "
210  << annotation.getMember("target")
211  << ", but this annotation cannot be non-local";
212  diag.attachNote() << "see current annotation: " << anno << "\n";
213  return failure();
214  }
215  SmallVector<NamedAttribute> newAnnoAttrs;
216  for (auto &na : anno) {
217  if (na.getName().getValue() != "target") {
218  newAnnoAttrs.push_back(na);
219  } else if (!target.isLocal()) {
220  auto sym = scatterNonLocalPath(target, state);
221  newAnnoAttrs.push_back(
222  {StringAttr::get(anno.getContext(), "circt.nonlocal"), sym});
223  }
224  }
225  addAnnotation(target.ref, target.fieldIdx, newAnnoAttrs);
226  return success();
227 }

239 static LogicalResult applyDUTAnno(const AnnoPathValue &target,
240  DictionaryAttr anno, ApplyState &state) {
241  auto *op = target.ref.getOp();
242  auto loc = op->getLoc();
243 
244  if (!target.isLocal())
245  return mlir::emitError(loc) << "must be local";
246 
247  if (!target.ref.isa<OpAnnoTarget>() || !isa<FModuleOp>(op))
248  return mlir::emitError(loc) << "can only target to a module";
249 
250  auto moduleOp = cast<FModuleOp>(op);
251 
252  // DUT has public visibility.
253  moduleOp.setPublic();
254  SmallVector<NamedAttribute> newAnnoAttrs;
255  for (auto &na : anno)
256  if (na.getName().getValue() != "target")
257  newAnnoAttrs.push_back(na);
258  addAnnotation(target.ref, target.fieldIdx, newAnnoAttrs);
259  return success();
260 }

269 static LogicalResult applyConventionAnno(const AnnoPathValue &target,
270  DictionaryAttr anno,
271  ApplyState &state) {
272  auto *op = target.ref.getOp();
273  auto loc = op->getLoc();
274  auto error = [&]() {
275  auto diag = mlir::emitError(loc);
276  diag << "circuit.ConventionAnnotation ";
277  return diag;
278  };
279 
280  auto opTarget = target.ref.dyn_cast<OpAnnoTarget>();
281  if (!opTarget)
282  return error() << "must target a module object";
283 
284  if (!target.isLocal())
285  return error() << "must be local";
286 
287  auto conventionStrAttr =
288  tryGetAs<StringAttr>(anno, anno, "convention", loc, conventionAnnoClass);
289  if (!conventionStrAttr)
290  return failure();
291 
292  auto conventionStr = conventionStrAttr.getValue();
293  auto conventionOpt = parseConvention(conventionStr);
294  if (!conventionOpt)
295  return error() << "unknown convention " << conventionStr;
296 
297  auto convention = *conventionOpt;
298 
299  if (auto moduleOp = dyn_cast<FModuleOp>(op)) {
300  moduleOp.setConvention(convention);
301  return success();
302  }
303 
304  if (auto extModuleOp = dyn_cast<FExtModuleOp>(op)) {
305  extModuleOp.setConvention(convention);
306  return success();
307  }
308 
309  return error() << "can only target to a module or extmodule";
310 }

341 /// Update a memory op with attributes about memory file loading.
342 template <bool isInline>
343 static LogicalResult applyLoadMemoryAnno(const AnnoPathValue &target,
344  DictionaryAttr anno,
345  ApplyState &state) {
346  if (!target.isLocal()) {
347  mlir::emitError(state.circuit.getLoc())
348  << "has a " << anno.get("class")
349  << " annotation which is non-local, but this annotation is not allowed "
350  "to be non-local";
351  return failure();
352  }
353 
354  auto *op = target.ref.getOp();
355 
356  if (!target.isOpOfType<MemOp, CombMemOp, SeqMemOp>()) {
357  mlir::emitError(op->getLoc())
358  << "can only apply a load memory annotation to a memory";
359  return failure();
360  }
361 
362  // The two annotations have different case usage in "filename".
363  StringAttr filename = tryGetAs<StringAttr>(
364  anno, anno, isInline ? "filename" : "fileName", op->getLoc(),
365  anno.getAs<StringAttr>("class").getValue());
366  if (!filename)
367  return failure();
368 
369  auto hexOrBinary =
370  tryGetAs<StringAttr>(anno, anno, "hexOrBinary", op->getLoc(),
371  anno.getAs<StringAttr>("class").getValue());
372  if (!hexOrBinary)
373  return failure();
374 
375  auto hexOrBinaryValue = hexOrBinary.getValue();
376  if (hexOrBinaryValue != "h" && hexOrBinaryValue != "b") {
377  auto diag = mlir::emitError(op->getLoc())
378  << "has memory initialization annotation with invalid format, "
379  "'hexOrBinary' field must be either 'h' or 'b'";
380  diag.attachNote() << "the full annotation is: " << anno;
381  return failure();
382  }
383 
384  op->setAttr("init", MemoryInitAttr::get(op->getContext(), filename,
385  hexOrBinaryValue == "b", isInline));
386 
387  return success();
388 }

557 LogicalResult LowerAnnotationsPass::applyAnnotation(DictionaryAttr anno,
558  ApplyState &state) {
559  LLVM_DEBUG(llvm::dbgs() << " - anno: " << anno << "\n";);
560 
561  // Lookup the class
562  StringRef annoClassVal;
563  if (auto annoClass = anno.getNamed("class"))
564  annoClassVal = cast<StringAttr>(annoClass->getValue()).getValue();
565  else if (ignoreAnnotationClassless)
566  annoClassVal = "circt.missing";
567  else
568  return mlir::emitError(state.circuit.getLoc())
569  << "Annotation without a class: " << anno;
570 
571  // See if we handle the class
572  auto *record = getAnnotationHandler(annoClassVal, false);
573  if (!record) {
574  ++numUnhandled;
575  if (!ignoreAnnotationUnknown)
576  return mlir::emitError(state.circuit.getLoc())
577  << "Unhandled annotation: " << anno;
578 
579  // Try again, requesting the fallback handler.
580  record = getAnnotationHandler(annoClassVal, ignoreAnnotationUnknown);
581  assert(record);
582  }
583 
584  // Try to apply the annotation

615 /// Modify the circuit to solve and apply all Wiring Problems in the circuit. A
616 /// Wiring Problem is a mapping from a source to a sink that can be connected
617 /// via a base Type or RefType as requested. This uses a two-step approach.
618 /// First, all Wiring Problems are analyzed to compute pending modifications to
619 /// modules. Second, modules are visited from leaves to roots to apply module
620 /// modifications. Module modifications include addings ports and connecting
621 /// things up.
622 LogicalResult LowerAnnotationsPass::solveWiringProblems(ApplyState &state) {
623  // Utility function to extract the defining module from a value which may be
624  // either a BlockArgument or an Operation result.
625  auto getModule = [](Value value) {
626  if (BlockArgument blockArg = dyn_cast<BlockArgument>(value))
627  return cast<FModuleLike>(blockArg.getParentBlock()->getParentOp());
628  return value.getDefiningOp()->getParentOfType<FModuleLike>();
629  };
630 
631  // Utility function to determine where to insert connection operations.
632  auto findInsertionBlock = [&getModule](Value src, Value dest) -> Block * {
633  // Check for easy case: both are in the same block.
634  if (src.getParentBlock() == dest.getParentBlock())
635  return src.getParentBlock();
636 
637  // If connecting across blocks, figure out where to connect.
638  (void)getModule;
639  assert(getModule(src) == getModule(dest));
640  // Helper to determine if 'a' is available at 'b's block.
641  auto safelyDoms = [&](Value a, Value b) {
642  if (isa<BlockArgument>(a))
643  return true;
644  if (isa<BlockArgument>(b))
645  return false;
646  // Handle cases where 'b' is in child op after 'a'.
647  auto *ancestor =
648  a.getParentBlock()->findAncestorOpInBlock(*b.getDefiningOp());
649  return ancestor && a.getDefiningOp()->isBeforeInBlock(ancestor);
650  };
651  if (safelyDoms(src, dest))
652  return dest.getParentBlock();
653  if (safelyDoms(dest, src))
654  return src.getParentBlock();
655  return {};
656  };
657 
658  auto getNoopCast = [](Value v) -> mlir::UnrealizedConversionCastOp {
659  auto op =
660  dyn_cast_or_null<mlir::UnrealizedConversionCastOp>(v.getDefiningOp());
661  if (op && op.getNumResults() == 1 && op.getNumOperands() == 1 &&
662  op.getResultTypes()[0] == op.getOperandTypes()[0])
663  return op;
664  return {};
665  };
666 
667  // Utility function to connect a destination to a source. Always use a
668  // ConnectOp as the widths may be uninferred.
669  SmallVector<Operation *> opsToErase;
670  auto connect = [&](Value src, Value dest,
671  ImplicitLocOpBuilder &builder) -> LogicalResult {
672  // Strip away noop unrealized_conversion_cast's, used as placeholders.
673  // In the future, these should be created/managed as part of creating WP's.
674  if (auto op = getNoopCast(dest)) {
675  dest = op.getOperand(0);
676  opsToErase.push_back(op);
677  std::swap(src, dest);
678  } else if (auto op = getNoopCast(src)) {
679  src = op.getOperand(0);
680  opsToErase.push_back(op);
681  }
682 
683  if (foldFlow(dest) == Flow::Source)
684  std::swap(src, dest);
685 
686  // Figure out where to insert operations.
687  auto *insertBlock = findInsertionBlock(src, dest);
688  if (!insertBlock)
689  return emitError(src.getLoc())
690  .append("This value is involved with a Wiring Problem where the "
691  "destination is in the same module but neither dominates the "
692  "other, which is not supported.")
693  .attachNote(dest.getLoc())
694  .append("The destination is here.");
695 
696  // Insert at end, past invalidation in same block.
697  builder.setInsertionPointToEnd(insertBlock);
698 
699  // Create RefSend/RefResolve if necessary.
700  if (type_isa<RefType>(dest.getType()) != type_isa<RefType>(src.getType())) {
701  if (type_isa<RefType>(dest.getType()))
702  src = builder.create<RefSendOp>(src);
703  else
704  src = builder.create<RefResolveOp>(src);
705  }
706 
707  // If the sink is a wire with no users, then convert this to a node.
708  auto destOp = dyn_cast_or_null<WireOp>(dest.getDefiningOp());
709  if (destOp && dest.getUses().empty()) {
710  builder.create<NodeOp>(src, destOp.getName())
711  .setAnnotationsAttr(destOp.getAnnotations());
712  opsToErase.push_back(destOp);
713  return success();
714  }
715 
716  // Otherwise, just connect to the source.
717  emitConnect(builder, dest, src);
718 
719  return success();
720  };
721 
722  auto &instanceGraph = state.instancePathCache.instanceGraph;
723  auto *context = state.circuit.getContext();
724 
725  // Examine all discovered Wiring Problems to determine modifications that need
726  // to be made per-module.
727  LLVM_DEBUG({ llvm::dbgs() << "Analyzing wiring problems:\n"; });
728  DenseMap<FModuleLike, ModuleModifications> moduleModifications;
729  DenseSet<Value> visitedSinks;
730  for (auto e : llvm::enumerate(state.wiringProblems)) {
731  auto index = e.index();
732  auto problem = e.value();
733  // This is a unique index that is assigned to this specific wiring problem
734  // and is used as a key during wiring to know which Values (ports, sources,
735  // or sinks) should be connected.
736  auto source = problem.source;
737  auto sink = problem.sink;
738 
739  // Check that no WiringProblems are trying to use the same sink. This
740  // should never happen.
741  if (!visitedSinks.insert(sink).second) {
742  auto diag = mlir::emitError(source.getLoc())
743  << "This sink is involved with a Wiring Problem which is "
744  "targeted by a source used by another Wiring Problem. "
745  "(This is both illegal and should be impossible.)";
746  diag.attachNote(source.getLoc()) << "The source is here";
747  return failure();
748  }
749  FModuleLike sourceModule = getModule(source);
750  FModuleLike sinkModule = getModule(sink);
751  if (isa<FExtModuleOp>(sourceModule) || isa<FExtModuleOp>(sinkModule)) {
752  auto diag = mlir::emitError(source.getLoc())
753  << "This source is involved with a Wiring Problem which "
754  "includes an External Module port and External Module "
755  "ports anre not supported.";
756  diag.attachNote(sink.getLoc()) << "The sink is here.";
757  return failure();
758  }
759 
760  LLVM_DEBUG({
761  llvm::dbgs() << " - index: " << index << "\n"
762  << " source:\n"
763  << " module: " << sourceModule.getModuleName() << "\n"
764  << " value: " << source << "\n"
765  << " sink:\n"
766  << " module: " << sinkModule.getModuleName() << "\n"
767  << " value: " << sink << "\n"
768  << " newNameHint: " << problem.newNameHint << "\n";
769  });
770 
771  // If the source and sink are in the same block, just wire them up.
772  if (sink.getParentBlock() == source.getParentBlock()) {
773  auto builder = ImplicitLocOpBuilder::atBlockEnd(UnknownLoc::get(context),
774  sink.getParentBlock());
775  if (failed(connect(source, sink, builder)))
776  return failure();
777  continue;
778  }
779  // If both are in the same module but not same block, U-turn.
780  // We may not be able to handle this, but that is checked below while
781  // connecting.
782  if (sourceModule == sinkModule) {
783  LLVM_DEBUG(llvm::dbgs()
784  << " LCA: " << sourceModule.getModuleName() << "\n");
785  moduleModifications[sourceModule].connectionMap[index] = source;
786  moduleModifications[sourceModule].uturns.push_back({index, sink});
787  continue;
788  }
789 
790  // Otherwise, get instance paths for source/sink, and compute LCA.
791  auto sourcePaths = state.instancePathCache.getAbsolutePaths(sourceModule);
792  auto sinkPaths = state.instancePathCache.getAbsolutePaths(sinkModule);
793 
794  if (sourcePaths.size() != 1 || sinkPaths.size() != 1) {
795  auto diag =
796  mlir::emitError(source.getLoc())
797  << "This source is involved with a Wiring Problem where the source "
798  "or the sink are multiply instantiated and this is not supported.";
799  diag.attachNote(sink.getLoc()) << "The sink is here.";
800  return failure();
801  }
802 
803  FModuleOp lca =
804  cast<FModuleOp>(instanceGraph.getTopLevelNode()->getModule());
805  auto sources = sourcePaths[0];
806  auto sinks = sinkPaths[0];
807  while (!sources.empty() && !sinks.empty()) {
808  if (sources.top() != sinks.top())
809  break;
810  auto newLCA = cast<InstanceOp>(*sources.top());
811  lca = cast<FModuleOp>(newLCA.getReferencedModule(instanceGraph));
812  sources = sources.dropFront();
813  sinks = sinks.dropFront();
814  }
815 
816  LLVM_DEBUG({
817  llvm::dbgs() << " LCA: " << lca.getModuleName() << "\n"
818  << " sourcePath: " << sourcePaths[0] << "\n"
819  << " sinkPaths: " << sinkPaths[0] << "\n";
820  });
821 
822  // Pre-populate the connectionMap of the module with the source and sink.
823  moduleModifications[sourceModule].connectionMap[index] = source;
824  moduleModifications[sinkModule].connectionMap[index] = sink;
825 
826  // Record port types that should be added to each module along the LCA path.
827  Type sourceType, sinkType;
828  auto useRefTypes =
829  !noRefTypePorts &&
830  problem.refTypeUsage == WiringProblem::RefTypeUsage::Prefer;
831  if (useRefTypes) {
832  // Use RefType ports if possible
833  RefType refType = TypeSwitch<Type, RefType>(source.getType())
834  .Case<FIRRTLBaseType>([](FIRRTLBaseType base) {
835  return RefType::get(base.getPassiveType());
836  })
837  .Case<RefType>([](RefType ref) { return ref; });
838  sourceType = refType;
839  sinkType = refType.getType();
840  } else {
841  // Use specified port types.
842  sourceType = source.getType();
843  sinkType = sink.getType();
844 
845  // Types must be connectable, which means FIRRTLType's.
846  auto sourceFType = type_dyn_cast<FIRRTLType>(sourceType);
847  auto sinkFType = type_dyn_cast<FIRRTLType>(sinkType);
848  if (!sourceFType)
849  return emitError(source.getLoc())
850  << "Wiring Problem source type \"" << sourceType
851  << "\" must be a FIRRTL type";
852  if (!sinkFType)
853  return emitError(sink.getLoc())
854  << "Wiring Problem sink type \"" << sinkType
855  << "\" must be a FIRRTL type";
856 
857  // Otherwise they must be identical or FIRRTL type-equivalent
858  // (connectable).
859  if (sourceFType != sinkFType &&
860  !areTypesEquivalent(sinkFType, sourceFType)) {
861  auto diag = mlir::emitError(source.getLoc())
862  << "Wiring Problem source type " << sourceType
863  << " does not match sink type " << sinkType;
864  diag.attachNote(sink.getLoc()) << "The sink is here.";
865  return failure();
866  }
867  }
868  // If wiring using references, check that the sink value we connect to is
869  // passive.
870  if (auto sinkFType = type_dyn_cast<FIRRTLType>(sink.getType());
871  sinkFType && type_isa<RefType>(sourceType) &&
872  !getBaseType(sinkFType).isPassive())
873  return emitError(sink.getLoc())
874  << "Wiring Problem sink type \"" << sink.getType()
875  << "\" must be passive (no flips) when using references";
876 
877  // Record module modifications related to adding ports to modules.
878  auto addPorts = [&](igraph::InstancePath insts, Value val, Type tpe,
879  Direction dir) {
880  StringRef name, instName;
881  for (auto instNode : llvm::reverse(insts)) {
882  auto inst = cast<InstanceOp>(*instNode);
883  auto mod = inst.getReferencedModule<FModuleOp>(instanceGraph);
884  if (name.empty()) {
885  if (problem.newNameHint.empty())
886  name = state.getNamespace(mod).newName(
887  getFieldName(
888  getFieldRefFromValue(val, /*lookThroughCasts=*/true),
889  /*nameSafe=*/true)
890  .first +
891  "__bore");
892  else
893  name = state.getNamespace(mod).newName(problem.newNameHint);
894  } else {
895  assert(!instName.empty());
896  name = state.getNamespace(mod).newName(instName + "_" + name);
897  }
898  moduleModifications[mod].portsToAdd.push_back(
899  {index, {StringAttr::get(context, name), tpe, dir}});
900  instName = inst.getInstanceName();
901  }
902  };
903 
904  // Record the addition of ports.
905  addPorts(sources, source, sourceType, Direction::Out);
906  addPorts(sinks, sink, sinkType, Direction::In);
907  }
908 
909  // Iterate over modules from leaves to roots, applying ModuleModifications to
910  // each module.
911  LLVM_DEBUG({ llvm::dbgs() << "Updating modules:\n"; });
912  for (auto *op : llvm::post_order(instanceGraph.getTopLevelNode())) {
913  auto fmodule = dyn_cast<FModuleOp>(*op->getModule());
914  // Skip external modules and modules that have no modifications.
915  if (!fmodule || !moduleModifications.count(fmodule))
916  continue;
917 
918  auto modifications = moduleModifications[fmodule];
919  LLVM_DEBUG({
920  llvm::dbgs() << " - module: " << fmodule.getModuleName() << "\n";
921  llvm::dbgs() << " ports:\n";
922  for (auto [index, port] : modifications.portsToAdd) {
923  llvm::dbgs() << " - name: " << port.getName() << "\n"
924  << " id: " << index << "\n"
925  << " type: " << port.type << "\n"
926  << " direction: "
927  << (port.direction == Direction::In ? "in" : "out")
928  << "\n";
929  }
930  });
931 
932  // Add ports to the module after all other existing ports.
933  SmallVector<std::pair<unsigned, PortInfo>> newPorts;
934  SmallVector<unsigned> problemIndices;
935  for (auto [problemIdx, portInfo] : modifications.portsToAdd) {
936  // Create the port.
937  newPorts.push_back({fmodule.getNumPorts(), portInfo});
938  problemIndices.push_back(problemIdx);
939  }
940  auto originalNumPorts = fmodule.getNumPorts();
941  auto portIdx = fmodule.getNumPorts();
942  fmodule.insertPorts(newPorts);
943 
944  auto builder = ImplicitLocOpBuilder::atBlockBegin(UnknownLoc::get(context),
945  fmodule.getBodyBlock());
946 
947  // Connect each port to the value stored in the connectionMap for this
948  // wiring problem index.
949  for (auto [problemIdx, portPair] : llvm::zip(problemIndices, newPorts)) {
950  Value src = moduleModifications[fmodule].connectionMap[problemIdx];
951  assert(src && "there did not exist a driver for the port");
952  Value dest = fmodule.getArgument(portIdx++);
953  if (failed(connect(src, dest, builder)))
954  return failure();
955  }
956 
957  // If a U-turn exists, this is an LCA and we need a U-turn connection. These
958  // are the last connections made for this module.
959  for (auto [problemIdx, dest] : moduleModifications[fmodule].uturns) {
960  Value src = moduleModifications[fmodule].connectionMap[problemIdx];
961  assert(src && "there did not exist a connection for the u-turn");
962  if (failed(connect(src, dest, builder)))
963  return failure();
964  }
965 
966  // Update the connectionMap of all modules for which we created a port.
967  for (auto *inst : instanceGraph.lookup(fmodule)->uses()) {
968  InstanceOp useInst = cast<InstanceOp>(inst->getInstance());
969  auto enclosingModule = useInst->getParentOfType<FModuleOp>();
970  auto clonedInst = useInst.cloneAndInsertPorts(newPorts);
971  state.instancePathCache.replaceInstance(useInst, clonedInst);
972  // When RAUW-ing, ignore the new ports that we added when replacing (they
973  // cannot have uses).
974  useInst->replaceAllUsesWith(
975  clonedInst.getResults().drop_back(newPorts.size()));
976  useInst->erase();
977  // Record information in the moduleModifications strucutre for the module
978  // _where this is instantiated_. This is done so that when that module is
979  // visited later, there will be information available for it to find ports
980  // it needs to wire up. If there is already an existing connection, then
981  // this is a U-turn.
982  for (auto [newPortIdx, problemIdx] : llvm::enumerate(problemIndices)) {
983  auto &modifications = moduleModifications[enclosingModule];
984  auto newPort = clonedInst.getResult(newPortIdx + originalNumPorts);
985  if (modifications.connectionMap.count(problemIdx)) {
986  modifications.uturns.push_back({problemIdx, newPort});
987  continue;
988  }
989  modifications.connectionMap[problemIdx] = newPort;
990  }
991  }
992  }
993 
994  // Delete unused WireOps created by producers of WiringProblems.
995  for (auto *op : opsToErase)
996  op->erase();
997 
998  return success();
999 }

1001 // This is the main entrypoint for the lowering pass.
1002 void LowerAnnotationsPass::runOnOperation() {
1003  CircuitOp circuit = getOperation();
1004  SymbolTable modules(circuit);
1005 
1006  LLVM_DEBUG(debugPassHeader(this) << "\n");
1007 
1008  // Grab the annotations from a non-standard attribute called "rawAnnotations".
1009  // This is a temporary location for all annotations that are earmarked for
1010  // processing by this pass as we migrate annotations from being handled by
1011  // FIRAnnotations/FIRParser into this pass. While we do this, this pass is
1012  // not supposed to touch _other_ annotations to enable this pass to be run
1013  // after FIRAnnotations/FIRParser.
1014  auto annotations = circuit->getAttrOfType<ArrayAttr>(rawAnnotations);
1015  if (!annotations)
1016  return;
1017  circuit->removeAttr(rawAnnotations);
1018 
1019  // Populate the worklist in reverse order. This has the effect of causing
1020  // annotations to be processed in the order in which they appear in the
1021  // original JSON.
1022  for (auto anno : llvm::reverse(annotations.getValue()))
1023  worklistAttrs.push_back(cast<DictionaryAttr>(anno));
1024 
1025  size_t numFailures = 0;
1026  size_t numAdded = 0;
1027  auto addToWorklist = [&](DictionaryAttr anno) {
1028  ++numAdded;
1029  worklistAttrs.push_back(anno);
1030  };
1031  InstancePathCache instancePathCache(getAnalysis<InstanceGraph>());
1032  ApplyState state{circuit, modules, addToWorklist, instancePathCache,
1033  noRefTypePorts};
1034  LLVM_DEBUG(llvm::dbgs() << "Processing annotations:\n");
1035  while (!worklistAttrs.empty()) {
1036  auto attr = worklistAttrs.pop_back_val();
1037  if (applyAnnotation(attr, state).failed())
1038  ++numFailures;
1039  }
1040 
1041  if (failed(legacyToWiringProblems(state)))
1042  ++numFailures;
1043 
1044  if (failed(solveWiringProblems(state)))
1045  ++numFailures;
1046 
1047  // Update statistics
1048  numRawAnnotations += annotations.size();
1049  numAddedAnnos += numAdded;
1050  numAnnos += numAdded + annotations.size();
1051  numReusedHierPathOps += state.numReusedHierPaths;
1052 
1053  if (numFailures)
1054  signalPassFailure();
1055 }
