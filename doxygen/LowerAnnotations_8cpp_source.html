45 /// Construct the annotation array with a new thing appended.

47  if (!array)

62 /// Apply a new annotation to a resolved target. This handles ports,
63 /// aggregates, modules, wires, etc.
64 static void addAnnotation(AnnoTarget ref, unsigned fieldIdx,
65  ArrayRef<NamedAttribute> anno) {
66  auto *context = ref.getOp()->getContext();
67  DictionaryAttr annotation;
68  if (fieldIdx) {
69  SmallVector<NamedAttribute> annoField(anno.begin(), anno.end());
70  annoField.emplace_back(
71  StringAttr::get(context, "circt.fieldID"),
72  IntegerAttr::get(IntegerType::get(context, 32, IntegerType::Signless),
73  fieldIdx));
74  annotation = DictionaryAttr::get(context, annoField);
75  } else {
76  annotation = DictionaryAttr::get(context, anno);
77  }
78 
79  if (ref.isa<OpAnnoTarget>()) {
80  auto newAnno = appendArrayAttr(getAnnotationsFrom(ref.getOp()), annotation);
81  ref.getOp()->setAttr(getAnnotationAttrName(), newAnno);
82  return;
83  }
84 
85  auto portRef = ref.cast<PortAnnoTarget>();
86  auto portAnnoRaw = ref.getOp()->getAttr(getPortAnnotationAttrName());
87  ArrayAttr portAnno = portAnnoRaw.dyn_cast_or_null<ArrayAttr>();
88  if (!portAnno || portAnno.size() != getNumPorts(ref.getOp())) {
89  SmallVector<Attribute> emptyPortAttr(
90  getNumPorts(ref.getOp()),
91  ArrayAttr::get(ref.getOp()->getContext(), {}));
92  portAnno = ArrayAttr::get(ref.getOp()->getContext(), emptyPortAttr);
93  }
94  portAnno = replaceArrayAttrElement(
95  portAnno, portRef.getPortNo(),
96  appendArrayAttr(portAnno[portRef.getPortNo()].dyn_cast<ArrayAttr>(),
97  annotation));
98  ref.getOp()->setAttr("portAnnotations", portAnno);
99 }

101 /// Make an anchor for a non-local annotation. Use the expanded path to build
102 /// the module and name list in the anchor.
103 static FlatSymbolRefAttr buildNLA(const AnnoPathValue &target,
104  ApplyState &state) {
105  OpBuilder b(state.circuit.getBodyRegion());
106  SmallVector<Attribute> insts;
107  for (auto inst : target.instances) {
108  insts.push_back(OpAnnoTarget(inst).getNLAReference(
109  state.getNamespace(inst->getParentOfType<FModuleLike>())));
110  }
111 
112  insts.push_back(
113  FlatSymbolRefAttr::get(target.ref.getModule().moduleNameAttr()));
114  auto instAttr = ArrayAttr::get(state.circuit.getContext(), insts);
115  auto nla = b.create<HierPathOp>(state.circuit.getLoc(), "nla", instAttr);
116  state.symTbl.insert(nla);
117  return FlatSymbolRefAttr::get(nla);
118 }

141 /// Implementation of standard resolution. First parses the target path, then
142 /// resolves it.
143 static Optional<AnnoPathValue> stdResolveImpl(StringRef rawPath,
144  ApplyState &state) {
145  auto pathStr = canonicalizeTarget(rawPath);
146  StringRef path{pathStr};
147 
148  auto tokens = tokenizePath(path);
149  if (!tokens) {
150  mlir::emitError(state.circuit.getLoc())
151  << "Cannot tokenize annotation path " << rawPath;
152  return {};
153  }
154 
155  return resolveEntities(*tokens, state.circuit, state.symTbl,
156  state.targetCaches);
157 }

159 /// (SFC) FIRRTL SingleTargetAnnotation resolver. Uses the 'target' field of
160 /// the annotation with standard parsing to resolve the path. This requires
161 /// 'target' to exist and be normalized (per docs/FIRRTLAnnotations.md).
162 static Optional<AnnoPathValue> stdResolve(DictionaryAttr anno,
163  ApplyState &state) {
164  auto target = anno.getNamed("target");
165  if (!target) {
166  mlir::emitError(state.circuit.getLoc())
167  << "No target field in annotation " << anno;
168  return {};
169  }
170  if (!target->getValue().isa<StringAttr>()) {
171  mlir::emitError(state.circuit.getLoc())
172  << "Target field in annotation doesn't contain string " << anno;
173  return {};
174  }
175  return stdResolveImpl(target->getValue().cast<StringAttr>().getValue(),
176  state);
177 }

179 /// Resolves with target, if it exists. If not, resolves to the circuit.
180 static Optional<AnnoPathValue> tryResolve(DictionaryAttr anno,
181  ApplyState &state) {
182  auto target = anno.getNamed("target");
183  if (target)
184  return stdResolveImpl(target->getValue().cast<StringAttr>().getValue(),

193 /// An applier which puts the annotation on the target and drops the 'target'
194 /// field from the annotaiton. Optionally handles non-local annotations.
195 static LogicalResult applyWithoutTargetImpl(const AnnoPathValue &target,
196  DictionaryAttr anno,
197  ApplyState &state,
198  bool allowNonLocal) {
199  if (!allowNonLocal && !target.isLocal()) {
200  Annotation annotation(anno);
201  auto diag = mlir::emitError(target.ref.getOp()->getLoc())
202  << "is targeted by a non-local annotation \""
203  << annotation.getClass() << "\" with target "
204  << annotation.getMember("target")
205  << ", but this annotation cannot be non-local";
206  diag.attachNote() << "see current annotation: " << anno << "\n";
207  return failure();
208  }
209  SmallVector<NamedAttribute> newAnnoAttrs;
210  for (auto &na : anno) {
211  if (na.getName().getValue() != "target") {
212  newAnnoAttrs.push_back(na);
213  } else if (!target.isLocal()) {
214  auto sym = scatterNonLocalPath(target, state);
215  newAnnoAttrs.push_back(
216  {StringAttr::get(anno.getContext(), "circt.nonlocal"), sym});
217  }
218  }
219  addAnnotation(target.ref, target.fieldIdx, newAnnoAttrs);
220  return success();
221 }

223 /// An applier which puts the annotation on the target and drops the 'target'
224 /// field from the annotaiton. Optionally handles non-local annotations.
225 /// Ensures the target resolves to an expected type of operation.
226 template <bool allowNonLocal, typename T, typename... Tr>
227 static LogicalResult applyWithoutTarget(const AnnoPathValue &target,
228  DictionaryAttr anno,
229  ApplyState &state) {
230  if (!target.isOpOfType<T, Tr...>())
231  return failure();
232  return applyWithoutTargetImpl(target, anno, state, allowNonLocal);
233 }

244 /// Apply a DontTouchAnnotation to the circuit. For almost all operations, this
245 /// just adds a symbol. For CHIRRTL memory ports, this preserves the
246 /// annotation.
247 static LogicalResult applyDontTouch(const AnnoPathValue &target,
248  DictionaryAttr anno, ApplyState &state) {
249 
250  // A DontTouchAnnotation is only allowed to be placed on a ReferenceTarget.
251  // If this winds up on a module. then it indicates that the original
252  // annotation was incorrect.
253  if (target.isOpOfType<FModuleOp, FExtModuleOp>()) {
254  mlir::emitError(target.ref.getOp()->getLoc())
255  << "'firrtl.module' op is targeted by a DontTouchAnotation with target "
256  << Annotation(anno).getMember("target")
257  << ", but this annotation must be a reference target";
258  return failure();
259  }
260 
261  // If the annotation is on a MemoryPortOp or if the annotation is on part of
262  // an aggregate, then keep the DontTouchAnnotation around.
263  if (isa<chirrtl::MemoryPortOp>(target.ref.getOp()) || target.fieldIdx)
264  return applyWithoutTarget<true>(target, anno, state);
265 
266  target.ref.getInnerSym(state.getNamespace(target.ref.getModule()));
267  return success();
268 }

355 LogicalResult LowerAnnotationsPass::applyAnnotation(DictionaryAttr anno,
356  ApplyState &state) {
357  LLVM_DEBUG(llvm::dbgs() << " - anno: " << anno << "\n";);
358 
359  // Lookup the class
360  StringRef annoClassVal;
361  if (auto annoClass = anno.getNamed("class"))
362  annoClassVal = annoClass->getValue().cast<StringAttr>().getValue();
363  else if (ignoreClasslessAnno)
364  annoClassVal = "circt.missing";
365  else
366  return state.circuit.emitError("Annotation without a class: ") << anno;
367 
368  // See if we handle the class
369  auto *record = getAnnotationHandler(annoClassVal, false);
370  if (!record) {
371  ++numUnhandled;
372  if (!ignoreUnhandledAnno)
373  return state.circuit->emitWarning("Unhandled annotation: ") << anno;
374 
375  // Try again, requesting the fallback handler.
376  record = getAnnotationHandler(annoClassVal, ignoreUnhandledAnno);
377  assert(record);
378  }
379 
380  // Try to apply the annotation
381  auto target = record->resolver(anno, state);
382  if (!target)
383  return mlir::emitError(state.circuit.getLoc())
384  << "Unable to resolve target of annotation: " << anno;
385  if (record->applier(*target, anno, state).failed())
386  return mlir::emitError(state.circuit.getLoc())
387  << "Unable to apply annotation: " << anno;
388  return success();
389 }

391 // This is the main entrypoint for the lowering pass.
392 void LowerAnnotationsPass::runOnOperation() {
393  CircuitOp circuit = getOperation();
394  SymbolTable modules(circuit);
395 
396  LLVM_DEBUG(llvm::dbgs() << "===- Running LowerAnnotations Pass "
397  "------------------------------------------===\n");
398 
399  // Grab the annotations from a non-standard attribute called "rawAnnotations".
400  // This is a temporary location for all annotations that are earmarked for
401  // processing by this pass as we migrate annotations from being handled by
402  // FIRAnnotations/FIRParser into this pass. While we do this, this pass is
403  // not supposed to touch _other_ annotations to enable this pass to be run
404  // after FIRAnnotations/FIRParser.
405  auto annotations = circuit->getAttrOfType<ArrayAttr>(rawAnnotations);
406  if (!annotations)
407  return;
408  circuit->removeAttr(rawAnnotations);
409 
410  // Populate the worklist in reverse order. This has the effect of causing
411  // annotations to be processed in the order in which they appear in the
412  // original JSON.
413  for (auto anno : llvm::reverse(annotations.getValue()))
414  worklistAttrs.push_back(anno.cast<DictionaryAttr>());
415 
416  size_t numFailures = 0;
417  size_t numAdded = 0;
418  auto addToWorklist = [&](DictionaryAttr anno) {
419  ++numAdded;
420  worklistAttrs.push_back(anno);
421  };
422  ApplyState state{circuit, modules, addToWorklist};
423  LLVM_DEBUG(llvm::dbgs() << "Processing annotations:\n");
424  while (!worklistAttrs.empty()) {
425  auto attr = worklistAttrs.pop_back_val();
426  if (applyAnnotation(attr, state).failed())
427  ++numFailures;
428  }
429 
430  // Update statistics
431  numRawAnnotations += annotations.size();
432  numAddedAnnos += numAdded;
433  numAnnos += numAdded + annotations.size();
434 
435  if (numFailures)
436  signalPassFailure();
437 }
