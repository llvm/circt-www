121 /// Performs the callback for each leaf element of a value. This will create
122 /// any subindex and subfield operations needed to access the leaf values of the
123 /// aggregate value.
124 static void forEachLeaf(ImplicitLocOpBuilder &builder, Value value,
125  llvm::function_ref<void(Value)> func) {
126  auto type = value.getType();
127  if (auto bundleType = type.dyn_cast<BundleType>()) {
128  for (size_t i = 0, e = bundleType.getNumElements(); i < e; ++i)
129  forEachLeaf(builder, builder.create<SubfieldOp>(value, i), func);
130  } else if (auto vectorType = type.dyn_cast<FVectorType>()) {
131  for (size_t i = 0, e = vectorType.getNumElements(); i != e; ++i)
132  forEachLeaf(builder, builder.create<SubindexOp>(value, i), func);
133  } else {
134  func(value);
135  }
136 }

138 /// Drive a value to all leafs of the input aggregate value. This only makes
139 /// sense when all leaf values have the same type, since the same value will be
140 /// connected to each leaf. This does not work for aggregates with flip types.
141 static void connectLeafsTo(ImplicitLocOpBuilder &builder, Value bundle,
142  Value value) {
143  forEachLeaf(builder, bundle,
144  [&](Value leaf) { emitConnect(builder, leaf, value); });

159 /// Converts a CHIRRTL memory port direction to a MemoryOp port type. The
160 /// biggest difference is that there is no match for the Infer port type.
161 static MemOp::PortKind memDirAttrToPortKind(MemDirAttr direction) {
162  switch (direction) {
163  case MemDirAttr::Read:
164  return MemOp::PortKind::Read;
165  case MemDirAttr::Write:
166  return MemOp::PortKind::Write;
167  case MemDirAttr::ReadWrite:
168  return MemOp::PortKind::ReadWrite;
169  default:
170  llvm_unreachable(
171  "Unhandled MemDirAttr, was the port direction not inferred?");
172  }
173 }

175 /// This function infers the memory direction of each CHIRRTL memory port. Each
176 /// memory port has an initial memory direction which is explicitly declared in
177 /// the MemoryPortOp, which is used as a starting point. For example, if the
178 /// port is declared to be Write, but it is only ever read from, the port will
179 /// become a ReadWrite port.
180 ///
181 /// When the memory port is eventually replaced with a memory, we will go from
182 /// having a single data value to having separate rdata and wdata values. In
183 /// this function we record how the result of each data subfield operation is
184 /// used, so that later on we can make sure the SubfieldOp is cloned to index
185 /// into the correct rdata and wdata fields of the memory.
186 MemDirAttr LowerCHIRRTLPass::inferMemoryPortKind(MemoryPortOp memPort) {
187  // This function does a depth-first walk of the use-lists of the memport
188  // operation to look through subindex operations and find the places where it
189  // is ultimately used. At each node we record how the children ops are using
190  // the result of the current operation. When we are done visiting the current
191  // operation we store how it is used into a global hashtable for later use.
192  // This records how both the MemoryPort and Subfield operations are used.
193  struct StackElement {
194  StackElement(Value value, Value::use_iterator iterator, MemDirAttr mode)
195  : value(value), iterator(iterator), mode(mode) {}
196  Value value;
197  Value::use_iterator iterator;
198  MemDirAttr mode;
199  };
200 
201  SmallVector<StackElement> stack;
202  stack.emplace_back(memPort.data(), memPort.data().use_begin(),
203  memPort.direction());
204  MemDirAttr mode = MemDirAttr::Infer;
205 
206  while (!stack.empty()) {
207  auto *iter = &stack.back().iterator;
208  auto end = stack.back().value.use_end();
209  stack.back().mode |= mode;
210 
211  while (*iter != end) {
212  auto &element = stack.back();
213  auto &use = *(*iter);
214  auto *user = use.getOwner();
215  ++(*iter);
216  if (isa<SubindexOp, SubfieldOp>(user)) {
217  // We recurse into Subindex ops to find the leaf-uses.
218  auto output = user->getResult(0);
219  stack.emplace_back(output, output.use_begin(), MemDirAttr::Infer);
220  mode = MemDirAttr::Infer;
221  iter = &stack.back().iterator;
222  end = output.use_end();
223  continue;
224  }
225  if (auto subaccessOp = dyn_cast<SubaccessOp>(user)) {
226  // Subaccess has two arguments, the vector and the index. If we are
227  // using the memory port as an index, we can ignore it. If we are using
228  // the memory as the vector, we need to recurse.
229  auto input = subaccessOp.input();
230  if (use.get() == input) {
231  auto output = subaccessOp.getResult();
232  stack.emplace_back(output, output.use_begin(), MemDirAttr::Infer);
233  mode = MemDirAttr::Infer;
234  iter = &stack.back().iterator;
235  end = output.use_end();
236  continue;
237  }
238  // Otherwise we are reading from a memory for the index.
239  element.mode |= MemDirAttr::Read;
240  } else if (auto connectOp = dyn_cast<ConnectOp>(user)) {
241  if (use.get() == connectOp.dest()) {
242  element.mode |= MemDirAttr::Write;
243  } else {
244  element.mode |= MemDirAttr::Read;
245  }
246  } else if (auto connectOp = dyn_cast<StrictConnectOp>(user)) {
247  if (use.get() == connectOp.dest()) {
248  element.mode |= MemDirAttr::Write;
249  } else {
250  element.mode |= MemDirAttr::Read;
251  }
252  } else {
253  // Every other use of a memory is a read operation.
254  element.mode |= MemDirAttr::Read;
255  }
256  }
257  mode = stack.back().mode;
258 
259  // Store the direction of the current operation in the global map. This will
260  // be used later to determine if this subaccess operation needs to be cloned
261  // into rdata, wdata, and wmask.
262  subfieldDirs[stack.back().value.getDefiningOp()] = mode;
263  stack.pop_back();
264  }
265 
266  return mode;
267 }

269 void LowerCHIRRTLPass::replaceMem(Operation *cmem, StringRef name,
270  bool isSequential, RUWAttr ruw,
271  ArrayAttr annotations) {
272  assert(isa<CombMemOp>(cmem) || isa<SeqMemOp>(cmem));
273 
274  // We have several early breaks in this function, so we record the CHIRRTL
275  // memory for deletion here.
276  opsToDelete.push_back(cmem);
277 
278  auto cmemType = cmem->getResult(0).getType().cast<CMemoryType>();
279  auto depth = cmemType.getNumElements();
280  auto type = cmemType.getElementType();
281 
282  // Collect the information from each of the CMemoryPorts.
283  struct PortInfo {
284  StringAttr name;
285  Type type;
286  Attribute annotations;
287  MemOp::PortKind portKind;
288  MemoryPortOp cmemPort;
289  };
290  SmallVector<PortInfo, 4> ports;
291  for (auto *user : cmem->getUsers()) {
292  auto cmemoryPort = cast<MemoryPortOp>(user);
293 
294  // Infer the type of memory port we need to create.
295  auto portDirection = inferMemoryPortKind(cmemoryPort);
296 
297  // If the memory port is never used, it will have the Infer type and should
298  // just be deleted. TODO: this is mirroring SFC, but should we be checking
299  // for annotations on the memory port before removing it?
300  if (portDirection == MemDirAttr::Infer)
301  continue;
302  auto portKind = memDirAttrToPortKind(portDirection);
303 
304  // Add the new port.
305  ports.push_back({cmemoryPort.nameAttr(),
306  MemOp::getTypeForPort(depth, type, portKind),
307  cmemoryPort.annotationsAttr(), portKind, cmemoryPort});
308  }
309 
310  // If there are no valid memory ports, don't create a memory.
311  if (ports.empty())
312  return;
313 
314  // Canonicalize the ports into alphabetical order.
315  llvm::array_pod_sort(ports.begin(), ports.end(),
316  [](const PortInfo *lhs, const PortInfo *rhs) -> int {
317  return lhs->name.getValue().compare(
318  rhs->name.getValue());
319  });
320 
321  SmallVector<Attribute, 4> resultNames;
322  SmallVector<Type, 4> resultTypes;
323  SmallVector<Attribute, 4> portAnnotations;
324  for (auto port : ports) {
325  resultNames.push_back(port.name);
326  resultTypes.push_back(port.type);
327  portAnnotations.push_back(port.annotations);
328  }
329 
330  // Write latency is always 1, while the read latency depends on the memory
331  // type.
332  auto readLatency = isSequential ? 1 : 0;
333  auto writeLatency = 1;
334 
335  // Create the memory.
336  ImplicitLocOpBuilder memBuilder(cmem->getLoc(), cmem);
337  auto memory = memBuilder.create<MemOp>(
338  resultTypes, readLatency, writeLatency, depth, ruw,
339  memBuilder.getArrayAttr(resultNames), name,
340  cmem->getAttrOfType<firrtl::NameKindEnumAttr>("nameKind").getValue(),
341  annotations, memBuilder.getArrayAttr(portAnnotations), InnerSymAttr(),
342  IntegerAttr());
343  if (auto innerSym = cmem->getAttr("inner_sym"))
344  memory->setAttr("inner_sym", innerSym);
345 
346  // Process each memory port, initializing the memory port and inferring when
347  // to set the enable signal high.
348  for (unsigned i = 0, e = memory.getNumResults(); i < e; ++i) {
349  auto cmemoryPort = ports[i].cmemPort;
350  auto cmemoryPortAccess = cmemoryPort.getAccess();
351  auto memoryPort = memory.getResult(i);
352  auto portKind = ports[i].portKind;
353 
354  // Most fields on the newly created memory will be assigned an initial value
355  // immediately following the memory decl, and then will be assigned a second
356  // value at the location of the CHIRRTL memory port.
357 
358  // Initialization at the MemoryOp.
359  ImplicitLocOpBuilder portBuilder(cmemoryPortAccess.getLoc(),
360  cmemoryPortAccess);
361  auto address = memBuilder.create<SubfieldOp>(memoryPort, "addr");
362  emitInvalid(memBuilder, address);
363  auto enable = memBuilder.create<SubfieldOp>(memoryPort, "en");
364  emitConnect(memBuilder, enable, getConst(0));
365  auto clock = memBuilder.create<SubfieldOp>(memoryPort, "clk");
366  emitInvalid(memBuilder, clock);
367 
368  // Initialization at the MemoryPortOp.
369  emitConnect(portBuilder, address, cmemoryPortAccess.index());
370  // Sequential+Read ports have a more complicated "enable inference".
371  auto useEnableInference = isSequential && portKind == MemOp::PortKind::Read;
372  auto *addressOp = cmemoryPortAccess.index().getDefiningOp();
373  // If the address value is not something with a "name", then we do not use
374  // enable inference.
375  useEnableInference &=
376  !addressOp || isa<WireOp, NodeOp, RegOp, RegResetOp>(addressOp);
377 
378  // Most memory ports just tie their enable line to one.
379  if (!useEnableInference)
380  emitConnect(portBuilder, enable, getConst(1));
381 
382  emitConnect(portBuilder, clock, cmemoryPortAccess.clock());
383 
384  if (portKind == MemOp::PortKind::Read) {
385  // Store the read information for updating subfield ops.
386  auto data = memBuilder.create<SubfieldOp>(memoryPort, "data");
387  rdataValues[cmemoryPort.data()] = data;
388  } else if (portKind == MemOp::PortKind::Write) {
389  // Initialization at the MemoryOp.
390  auto data = memBuilder.create<SubfieldOp>(memoryPort, "data");
391  emitInvalid(memBuilder, data);
392  auto mask = memBuilder.create<SubfieldOp>(memoryPort, "mask");
393  emitInvalid(memBuilder, mask);
394 
395  // Initialization at the MemoryPortOp.
396  connectLeafsTo(portBuilder, mask, getConst(0));
397 
398  // Store the write information for updating subfield ops.
399  wdataValues[cmemoryPort.data()] = {data, mask, nullptr};
400  } else if (portKind == MemOp::PortKind::ReadWrite) {
401  // Initialization at the MemoryOp.
402  auto rdata = memBuilder.create<SubfieldOp>(memoryPort, "rdata");
403  auto wmode = memBuilder.create<SubfieldOp>(memoryPort, "wmode");
404  emitConnect(memBuilder, wmode, getConst(0));
405  auto wdata = memBuilder.create<SubfieldOp>(memoryPort, "wdata");
406  emitInvalid(memBuilder, wdata);
407  auto wmask = memBuilder.create<SubfieldOp>(memoryPort, "wmask");
408  emitInvalid(memBuilder, wmask);
409 
410  // Initialization at the MemoryPortOp.
411  connectLeafsTo(portBuilder, wmask, getConst(0));
412 
413  // Store the read and write information for updating subfield ops.
414  wdataValues[cmemoryPort.data()] = {wdata, wmask, wmode};
415  rdataValues[cmemoryPort.data()] = rdata;
416  }
417 
418  // Sequential read only memory ports have "enable inference", which
419  // detects when to set the enable high. All other memory ports set the
420  // enable high when the memport is declared. This is higly questionable
421  // logic that is easily defeated. This behaviour depends on the kind of
422  // operation used as the memport index.
423  if (useEnableInference) {
424  auto *indexOp = cmemoryPortAccess.index().getDefiningOp();
425  bool success = false;
426  if (!indexOp) {
427  // TODO: SFC does not infer any enable when using a module port as the
428  // address. This seems like something that should be fixed sooner
429  // rather than later.
430  } else if (isa<WireOp, RegResetOp, RegOp>(indexOp)) {
431  // If the address is a reference, then we set the enable whenever the
432  // address is driven.
433 
434  // Find the uses of the address that write a value to it, ignoring the
435  // ones driving an invalid value.
436  auto drivers =
437  make_filter_range(indexOp->getUsers(), [&](Operation *op) {
438  if (auto connectOp = dyn_cast<ConnectOp>(op)) {
439  if (cmemoryPortAccess.index() == connectOp.dest())
440  return !dyn_cast_or_null<InvalidValueOp>(
441  connectOp.src().getDefiningOp());
442  } else if (auto connectOp = dyn_cast<StrictConnectOp>(op)) {
443  if (cmemoryPortAccess.index() == connectOp.dest())
444  return !dyn_cast_or_null<InvalidValueOp>(
445  connectOp.src().getDefiningOp());
446  }
447  return false;
448  });
449 
450  // At each location where we drive a value to the index, set the enable.
451  for (auto *driver : drivers) {
452  OpBuilder(driver).create<StrictConnectOp>(driver->getLoc(), enable,
453  getConst(1));
454  success = true;
455  }
456  } else if (isa<NodeOp>(indexOp)) {
457  // If using a Node for the address, then the we place the enable at the
458  // Node op's
459  OpBuilder(indexOp).create<StrictConnectOp>(indexOp->getLoc(), enable,
460  getConst(1));
461  success = true;
462  }
463 
464  // If we don't infer any enable points, it is almost always a user error.
465  if (!success)
466  cmemoryPort.emitWarning("memory port is never enabled");
467  }
468  }
469 }

491 void LowerCHIRRTLPass::visitStmt(ConnectOp connect) {
492  // Check if we are writing to a memory and, if we are, replace the
493  // destination.
494  auto writeIt = wdataValues.find(connect.dest());
495  if (writeIt != wdataValues.end()) {
496  auto writeData = writeIt->second;
497  connect.destMutable().assign(writeData.data);
498  // Assign the write mask.
499  ImplicitLocOpBuilder builder(connect.getLoc(), connect);
500  connectLeafsTo(builder, writeData.mask, getConst(1));
501  // Only ReadWrite memories have a write mode.
502  if (writeData.mode)
503  emitConnect(builder, writeData.mode, getConst(1));
504  }
505  // Check if we are reading from a memory and, if we are, replace the
506  // source.
507  auto readIt = rdataValues.find(connect.src());
508  if (readIt != rdataValues.end()) {
509  auto newSource = readIt->second;
510  connect.srcMutable().assign(newSource);
511  }
512 }

537 /// This function will create clones of subaccess, subindex, and subfield
538 /// operations which are indexing a CHIRRTL memory ports that will index into
539 /// the new memory's data field. If a subfield result is used to read from a
540 /// memory port, it will be cloned to read from the memory's rdata field. If
541 /// the subfield is used to write to a memory port, it will be cloned twice to
542 /// write to both the wdata and wmask fields. Users of this subfield operation
543 /// will be redirected to the appropriate clone when they are visited.
544 template <typename OpType, typename... T>
545 void LowerCHIRRTLPass::cloneSubindexOpForMemory(OpType op, Value input,
546  T... operands) {
547  // If the subaccess operation has no direction recorded, then it does not
548  // index a CHIRRTL memory and will be left alone.
549  auto it = subfieldDirs.find(op);
550  if (it == subfieldDirs.end())
551  return;
552 
553  // All uses of this op will be updated to use the appropriate clone. If the
554  // recorded direction of this subfield is Infer, then the value is not
555  // actually used to read or write from a memory port, and it will be just
556  // removed.
557  opsToDelete.push_back(op);
558 
559  auto direction = it->second;
560  ImplicitLocOpBuilder builder(op->getLoc(), op);
561 
562  // If the subaccess operation is used to read from a memory port, we need to
563  // clone it to read from the rdata field.
564  if (direction == MemDirAttr::Read || direction == MemDirAttr::ReadWrite) {
565  rdataValues[op] = builder.create<OpType>(rdataValues[input], operands...);
566  }
567 
568  // If the subaccess operation is used to write to the memory, we need to clone
569  // it to write to the the wdata and the wmask fields.
570  if (direction == MemDirAttr::Write || direction == MemDirAttr::ReadWrite) {
571  auto writeData = wdataValues[input];
572  auto write = builder.create<OpType>(writeData.data, operands...);
573  auto mask = builder.create<OpType>(writeData.mask, operands...);
574  wdataValues[op] = {write, mask, writeData.mode};
575  }
576 }

578 void LowerCHIRRTLPass::visitExpr(SubaccessOp subaccess) {
579  // Check if the subaccess reads from a memory for
580  // the index.
581  auto readIt = rdataValues.find(subaccess.index());

610 void LowerCHIRRTLPass::runOnOperation() {
611  // Walk the entire body of the module and dispatch the visitor on each
612  // function. This will replace all CHIRRTL memories and ports, and update all
613  // uses.
614  getOperation().getBody()->walk(
615  [&](Operation *op) { dispatchCHIRRTLVisitor(op); });
616 
617  // If there are no operations to delete, then we didn't find any CHIRRTL
618  // memories.
619  if (opsToDelete.empty())
620  markAllAnalysesPreserved();
621 
622  // Remove the old memories and their ports.
623  while (!opsToDelete.empty())
624  opsToDelete.pop_back_val()->erase();

626  // Clear out any cached data.
