49 // This is the main entrypoint for the conversion pass.
50 void LowerIntmodulesPass::runOnOperation() {
51  auto &ig = getAnalysis<InstanceGraph>();
52 
53  // Convert to int ops.
54  for (auto op :
55  llvm::make_early_inc_range(getOperation().getOps<FIntModuleOp>())) {
56  auto *node = ig.lookup(op);
57 
58  if (failed(checkModForAnnotations(op, op.getIntrinsic())))
59  return signalPassFailure();
60 
61  for (auto *use : llvm::make_early_inc_range(node->uses())) {
62  auto inst = use->getInstance<InstanceOp>();
63  if (failed(checkInstForAnnotations(inst, op.getIntrinsic())))
64  return signalPassFailure();
65 
66  // Replace the instance of this intmodule with firrtl.int.generic.
67  // Inputs become operands, outputs are the result (if any).
68  ImplicitLocOpBuilder builder(op.getLoc(), inst);
69 
70  SmallVector<Value> inputs;
71  struct OutputInfo {
72  Value result;
73  BundleType::BundleElement element;
74  };
75  SmallVector<OutputInfo> outputs;
76  for (auto [idx, result] : llvm::enumerate(inst.getResults())) {
77  // Replace inputs with wires that will be used as operands.
78  if (inst.getPortDirection(idx) != Direction::Out) {
79  auto w = builder.create<WireOp>(result.getLoc(), result.getType())
80  .getResult();
81  result.replaceAllUsesWith(w);
82  inputs.push_back(w);
83  continue;
84  }
85 
86  // Gather outputs. This will become a bundle if more than one, but
87  // typically there are zero or one.
88  auto ftype = dyn_cast<FIRRTLBaseType>(inst.getType(idx));
89  if (!ftype) {
90  inst.emitError("intrinsic has non-FIRRTL or non-base port type")
91  << inst.getType(idx);
92  signalPassFailure();
93  return;
94  }
95  outputs.push_back(
96  OutputInfo{inst.getResult(idx),
97  BundleType::BundleElement(inst.getPortName(idx),
98  /*isFlip=*/false, ftype)});
99  }
100 
101  // Create the replacement operation.
102  if (outputs.empty()) {
103  // If no outputs, just create the operation.
104  builder.create<GenericIntrinsicOp>(/*result=*/Type(),
105  op.getIntrinsicAttr(), inputs,
106  op.getParameters());
107 
108  } else if (outputs.size() == 1) {
109  // For single output, the result is the output.
110  auto resultType = outputs.front().element.type;
111  auto intop = builder.create<GenericIntrinsicOp>(
112  resultType, op.getIntrinsicAttr(), inputs, op.getParameters());
113  outputs.front().result.replaceAllUsesWith(intop.getResult());
114  } else {
115  // For multiple outputs, create a bundle with fields for each output
116  // and replace users with subfields.
117  auto resultType = builder.getType<BundleType>(llvm::map_to_vector(
118  outputs, [](const auto &info) { return info.element; }));
119  auto intop = builder.create<GenericIntrinsicOp>(
120  resultType, op.getIntrinsicAttr(), inputs, op.getParameters());
121  for (auto &output : outputs)
122  output.result.replaceAllUsesWith(builder.create<SubfieldOp>(
123  intop.getResult(), output.element.name));
124  }
125  // Remove instance from IR and instance graph.
126  use->erase();
127  inst.erase();
128  }
129  // Remove intmodule from IR and instance graph.
130  ig.erase(node);
131  op.erase();
132  }
133 
134  // Special handling for magic EICG wrapper extmodule. Deprecate and remove.
135  if (fixupEICGWrapper) {
136  constexpr StringRef eicgName = "EICG_wrapper";
137  for (auto op :
138  llvm::make_early_inc_range(getOperation().getOps<FExtModuleOp>())) {
139  if (op.getDefname() != eicgName)
140  continue;
141  if (failed(checkModForAnnotations(op, eicgName)))
142  return signalPassFailure();
143 
144  auto *node = ig.lookup(op);
145  for (auto *use : llvm::make_early_inc_range(node->uses())) {
146  auto inst = use->getInstance<InstanceOp>();
147  if (failed(checkInstForAnnotations(inst, eicgName)))
148  return signalPassFailure();
149 
150  ImplicitLocOpBuilder builder(op.getLoc(), inst);
151  auto replaceResults = [](OpBuilder &b, auto &&range) {
152  return llvm::map_to_vector(range, [&b](auto v) {
153  auto w = b.create<WireOp>(v.getLoc(), v.getType()).getResult();
154  v.replaceAllUsesWith(w);
155  return w;
156  });
157  };
158 
159  auto inputs = replaceResults(builder, inst.getResults().drop_back());
160  auto intop = builder.create<GenericIntrinsicOp>(
161  builder.getType<ClockType>(), "circt_clock_gate", inputs,
162  op.getParameters());
163  inst.getResults().back().replaceAllUsesWith(intop.getResult());
164 
165  // Remove instance from IR and instance graph.
166  use->erase();
167  inst.erase();
168  }
169  // Remove extmodule from IR and instance graph.
170  ig.erase(node);
171  op.erase();
172  }
173  }
174 
175  markAnalysesPreserved<InstanceGraph>();
176 }
