32 /// Lower CompRegOp to `sv.reg` and `sv.alwaysff`. Use a posedge clock and
33 /// synchronous reset.
34 struct CompRegLower : public OpConversionPattern<CompRegOp> {
35 public:
36  using OpConversionPattern::OpConversionPattern;
37 
38  LogicalResult
39  matchAndRewrite(CompRegOp reg, OpAdaptor adaptor,
40  ConversionPatternRewriter &rewriter) const final {
41  Location loc = reg.getLoc();
42 
43  auto svReg = rewriter.create<sv::RegOp>(loc, reg.getResult().getType(),
44  reg.nameAttr());
45  svReg->setDialectAttrs(reg->getDialectAttrs());
46 
47  // If the seq::CompRegOp has an inner_sym attribute, set this for the
48  // sv::RegOp inner_sym attribute.
49  if (reg.sym_name().hasValue())
50  svReg.inner_symAttr(reg.sym_nameAttr());
51 
52  if (reg.svAttributes())
53  svReg.svAttributesAttr(reg.svAttributesAttr());
54 
55  auto regVal = rewriter.create<sv::ReadInOutOp>(loc, svReg);
56  if (reg.reset() && reg.resetValue()) {
57  rewriter.create<sv::AlwaysFFOp>(
58  loc, sv::EventControl::AtPosEdge, reg.clk(), ResetType::SyncReset,
59  sv::EventControl::AtPosEdge, reg.reset(),
60  [&]() { rewriter.create<sv::PAssignOp>(loc, svReg, reg.input()); },
61  [&]() {
62  rewriter.create<sv::PAssignOp>(loc, svReg, reg.resetValue());
63  });
64  } else {
65  rewriter.create<sv::AlwaysFFOp>(
66  loc, sv::EventControl::AtPosEdge, reg.clk(),
67  [&]() { rewriter.create<sv::PAssignOp>(loc, svReg, reg.input()); });
68  }
69 
70  rewriter.replaceOp(reg, {regVal});
71  return success();
72  }
73 };
74 } // namespace

76 void SeqToSVPass::runOnOperation() {
77  ModuleOp top = getOperation();
78  MLIRContext &ctxt = getContext();
79 
80  ConversionTarget target(ctxt);
81  target.addIllegalDialect<SeqDialect>();
82  target.addLegalDialect<sv::SVDialect>();
83  RewritePatternSet patterns(&ctxt);
84  patterns.add<CompRegLower>(&ctxt);
85 
86  if (failed(applyPartialConversion(top, target, std::move(patterns))))
87  signalPassFailure();
88 }
