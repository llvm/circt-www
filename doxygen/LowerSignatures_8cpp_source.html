71 template <typename T>
72 class FieldIDSearch {
73  using E = typename T::ElementType;
74  using V = SmallVector<E>;
75 
76 public:
77  using const_iterator = typename V::const_iterator;
78 
79  template <typename Container>
80  FieldIDSearch(const Container &src) {
81  if constexpr (std::is_convertible_v<Container, Attribute>)
82  if (!src)
83  return;
84  for (auto attr : src)
85  vals.push_back(attr);
86  std::sort(vals.begin(), vals.end(), fieldComp);
87  }
88 
89  std::pair<const_iterator, const_iterator> find(uint64_t low,
90  uint64_t high) const {
91  return {std::lower_bound(vals.begin(), vals.end(), low, fieldCompInt2),
92  std::upper_bound(vals.begin(), vals.end(), high, fieldCompInt1)};
93  }
94 
95  bool empty(uint64_t low, uint64_t high) const {
96  auto [b, e] = find(low, high);
97  return b == e;
98  }
99 
100 private:
101  static constexpr auto fieldComp = [](const E &lhs, const E &rhs) {
102  return lhs.getFieldID() < rhs.getFieldID();
103  };
104  static constexpr auto fieldCompInt2 = [](const E &lhs, uint64_t rhs) {
105  return lhs.getFieldID() < rhs;
106  };
107  static constexpr auto fieldCompInt1 = [](uint64_t lhs, const E &rhs) {
108  return lhs < rhs.getFieldID();
109  };
110 
111  V vals;
112 };

116 static hw::InnerSymAttr
117 symbolsForFieldIDRange(MLIRContext *ctx,
118  const FieldIDSearch<hw::InnerSymAttr> &syms,
119  uint64_t low, uint64_t high) {
120  auto [b, e] = syms.find(low, high);
121  SmallVector<hw::InnerSymPropertiesAttr, 4> newSyms(b, e);
122  if (newSyms.empty())
123  return {};
124  for (auto &sym : newSyms)
125  sym = hw::InnerSymPropertiesAttr::get(
126  ctx, sym.getName(), sym.getFieldID() - low, sym.getSymVisibility());
127  return hw::InnerSymAttr::get(ctx, newSyms);

141 static LogicalResult
142 computeLoweringImpl(FModuleLike mod, PortConversion &newPorts, Convention conv,
143  size_t portID, const PortInfo &port, bool isFlip,
144  Twine name, FIRRTLType type, uint64_t fieldID,
145  const FieldIDSearch<hw::InnerSymAttr> &syms,
146  const FieldIDSearch<AnnotationSet> &annos) {
147  auto *ctx = type.getContext();
148  return FIRRTLTypeSwitch<FIRRTLType, LogicalResult>(type)
149  .Case<BundleType>([&](BundleType bundle) -> LogicalResult {
150  // This should be enhanced to be able to handle bundle<all flips of
151  // passive>, or this should be a canonicalizer
152  if (conv != Convention::Scalarized && bundle.isPassive()) {
153  auto lastId = fieldID + bundle.getMaxFieldID();
154  newPorts.push_back(
155  {{StringAttr::get(ctx, name), type,
156  isFlip ? Direction::Out : Direction::In,
157  symbolsForFieldIDRange(ctx, syms, fieldID, lastId), port.loc,
158  annosForFieldIDRange(ctx, annos, fieldID, lastId)},
159  portID,
160  newPorts.size(),
161  fieldID});
162  } else {
163  for (auto [idx, elem] : llvm::enumerate(bundle.getElements())) {
164  if (failed(computeLoweringImpl(
165  mod, newPorts, conv, portID, port, isFlip ^ elem.isFlip,
166  name + "_" + elem.name.getValue(), elem.type,
167  fieldID + bundle.getFieldID(idx), syms, annos)))
168  return failure();
169  if (!syms.empty(fieldID, fieldID))
170  return mod.emitError("Port [")
171  << port.name
172  << "] should be subdivided, but cannot be because of "
173  "symbol ["
174  << port.sym.getSymIfExists(fieldID) << "] on a bundle";
175  if (!annos.empty(fieldID, fieldID)) {
176  auto err = mod.emitError("Port [")
177  << port.name
178  << "] should be subdivided, but cannot be because of "
179  "annotations [";
180  auto [b, e] = annos.find(fieldID, fieldID);
181  err << b->getClass();
182  b++;
183  for (; b != e; ++b)
184  err << ", " << b->getClass();
185  err << "] on a bundle";
186  return err;
187  }
188  }
189  }
190  return success();
191  })
192  .Case<FVectorType>([&](FVectorType vector) -> LogicalResult {
193  if (conv != Convention::Scalarized &&
194  vector.getElementType().isPassive()) {
195  auto lastId = fieldID + vector.getMaxFieldID();
196  newPorts.push_back(
197  {{StringAttr::get(ctx, name), type,
198  isFlip ? Direction::Out : Direction::In,
199  symbolsForFieldIDRange(ctx, syms, fieldID, lastId), port.loc,
200  annosForFieldIDRange(ctx, annos, fieldID, lastId)},
201  portID,
202  newPorts.size(),
203  fieldID});
204  } else {
205  for (size_t i = 0, e = vector.getNumElements(); i < e; ++i) {
206  if (failed(computeLoweringImpl(
207  mod, newPorts, conv, portID, port, isFlip,
208  name + "_" + Twine(i), vector.getElementType(),
209  fieldID + vector.getFieldID(i), syms, annos)))
210  return failure();
211  if (!syms.empty(fieldID, fieldID))
212  return mod.emitError("Port [")
213  << port.name
214  << "] should be subdivided, but cannot be because of "
215  "symbol ["
216  << port.sym.getSymIfExists(fieldID) << "] on a vector";
217  if (!annos.empty(fieldID, fieldID)) {
218  auto err = mod.emitError("Port [")
219  << port.name
220  << "] should be subdivided, but cannot be because of "
221  "annotations [";
222  auto [b, e] = annos.find(fieldID, fieldID);
223  err << b->getClass();
224  ++b;
225  for (; b != e; ++b)
226  err << ", " << b->getClass();
227  err << "] on a vector";
228  return err;
229  }
230  }
231  }
232  return success();
233  })
234  .Case<FEnumType>([&](FEnumType fenum) { return failure(); })
235  .Default([&](FIRRTLType type) {
236  // Properties and other types wind up here.
237  newPorts.push_back(
238  {{StringAttr::get(ctx, name), type,
239  isFlip ? Direction::Out : Direction::In,
240  symbolsForFieldIDRange(ctx, syms, fieldID, fieldID), port.loc,
241  annosForFieldIDRange(ctx, annos, fieldID, fieldID)},
242  portID,
243  newPorts.size(),
244  fieldID});
245  return success();
246  });
247 }

249 // compute a new moduletype from an old module type and lowering convention.
250 // Also compute a fieldID map from port, fieldID -> port
251 static LogicalResult computeLowering(FModuleLike mod, Convention conv,
252  PortConversion &newPorts) {
253  for (auto [idx, port] : llvm::enumerate(mod.getPorts())) {
254  if (computeLoweringImpl(
255  mod, newPorts, conv, idx, port, port.direction == Direction::Out,
256  port.name.getValue(), type_cast<FIRRTLType>(port.type), 0,
257  FieldIDSearch<hw::InnerSymAttr>(port.sym),
258  FieldIDSearch<AnnotationSet>(port.annotations))
259  .failed())
260  return failure();
261  }
262  return success();
263 }

265 static LogicalResult lowerModuleSignature(FModuleLike module, Convention conv,
266  AttrCache &cache,
267  PortConversion &newPorts) {
268  ImplicitLocOpBuilder theBuilder(module.getLoc(), module.getContext());
269  if (computeLowering(module, conv, newPorts).failed())
270  return failure();
271  if (auto mod = dyn_cast<FModuleOp>(module.getOperation())) {
272  Block *body = mod.getBodyBlock();
273  theBuilder.setInsertionPointToStart(body);
274  auto oldNumArgs = body->getNumArguments();
275 
276  // Compute the replacement value for old arguments
277  // This creates all the new arguments and produces bounce wires when
278  // necessary
279  SmallVector<Value> bounceWires(oldNumArgs);
280  for (auto &p : newPorts) {
281  auto newArg = body->addArgument(p.type, p.loc);
282  // Get or create a bounce wire for changed ports
283  // For unmodified ports, move the uses to the replacement port
284  if (p.fieldID != 0) {
285  auto &wire = bounceWires[p.portID];
286  if (!wire)
287  wire = theBuilder
288  .create<WireOp>(module.getPortType(p.portID),
289  module.getPortNameAttr(p.portID),
290  NameKindEnum::InterestingName)
291  .getResult();
292  } else {
293  bounceWires[p.portID] = newArg;
294  }
295  }
296  // replace old arguments. Somethings get dropped completely, like
297  // zero-length vectors.
298  for (auto idx = 0U; idx < oldNumArgs; ++idx) {
299  if (!bounceWires[idx]) {
300  bounceWires[idx] = theBuilder
301  .create<WireOp>(module.getPortType(idx),
302  module.getPortNameAttr(idx))
303  .getResult();
304  }
305  body->getArgument(idx).replaceAllUsesWith(bounceWires[idx]);
306  }
307 
308  // Goodbye old ports, now ResultID in the PortInfo is correct.
309  body->eraseArguments(0, oldNumArgs);
310 
311  // Connect the bounce wires to the new arguments
312  for (auto &p : newPorts) {
313  if (isa<BlockArgument>(bounceWires[p.portID]))
314  continue;
315  if (p.isOutput())
316  emitConnect(
317  theBuilder, body->getArgument(p.resultID),
318  getValueByFieldID(theBuilder, bounceWires[p.portID], p.fieldID));
319  else
320  emitConnect(
321  theBuilder,
322  getValueByFieldID(theBuilder, bounceWires[p.portID], p.fieldID),
323  body->getArgument(p.resultID));
324  }
325  }
326 
327  SmallVector<NamedAttribute, 8> newModuleAttrs;
328 
329  // Copy over any attributes that weren't original argument attributes.
330  for (auto attr : module->getAttrDictionary())
331  // Drop old "portNames", directions, and argument attributes. These are
332  // handled differently below.
333  if (attr.getName() != "portNames" && attr.getName() != "portDirections" &&
334  attr.getName() != "portTypes" && attr.getName() != "portAnnotations" &&
335  attr.getName() != "portSyms" && attr.getName() != "portLocations" &&
336  attr.getName() != "internalPaths")
337  newModuleAttrs.push_back(attr);
338 
339  SmallVector<Direction> newPortDirections;
340  SmallVector<Attribute> newPortNames;
341  SmallVector<Attribute> newPortTypes;
342  SmallVector<Attribute> newPortSyms;
343  SmallVector<Attribute> newPortLocations;
344  SmallVector<Attribute, 8> newPortAnnotations;
345  SmallVector<Attribute> newInternalPaths;
346 
347  bool hasInternalPaths = false;
348  auto internalPaths = module->getAttrOfType<ArrayAttr>("internalPaths");
349  for (auto p : newPorts) {
350  newPortTypes.push_back(TypeAttr::get(p.type));
351  newPortNames.push_back(p.name);
352  newPortDirections.push_back(p.direction);
353  newPortSyms.push_back(p.sym);
354  newPortLocations.push_back(p.loc);
355  newPortAnnotations.push_back(p.annotations.getArrayAttr());
356  if (internalPaths) {
357  auto internalPath = internalPaths[p.portID].cast<InternalPathAttr>();
358  newInternalPaths.push_back(internalPath);
359  if (internalPath.getPath())
360  hasInternalPaths = true;
361  }
362  }
363 
364  newModuleAttrs.push_back(NamedAttribute(
365  cache.sPortDirections,
366  direction::packAttribute(module.getContext(), newPortDirections)));
367 
368  newModuleAttrs.push_back(
369  NamedAttribute(cache.sPortNames, theBuilder.getArrayAttr(newPortNames)));
370 
371  newModuleAttrs.push_back(
372  NamedAttribute(cache.sPortTypes, theBuilder.getArrayAttr(newPortTypes)));
373 
374  newModuleAttrs.push_back(NamedAttribute(
375  cache.sPortLocations, theBuilder.getArrayAttr(newPortLocations)));
376 
377  newModuleAttrs.push_back(NamedAttribute(
378  cache.sPortAnnotations, theBuilder.getArrayAttr(newPortAnnotations)));
379 
380  assert(newInternalPaths.empty() ||
381  newInternalPaths.size() == newPorts.size());
382  if (hasInternalPaths) {
383  newModuleAttrs.emplace_back(cache.sInternalPaths,
384  theBuilder.getArrayAttr(newInternalPaths));
385  }
386 
387  // Update the module's attributes.
388  module->setAttrs(newModuleAttrs);
389  module.setPortSymbols(newPortSyms);
390  return success();
391 }

393 static void lowerModuleBody(FModuleOp mod,
394  const DenseMap<StringAttr, PortConversion> &ports) {
395  mod->walk([&](InstanceOp inst) -> void {
396  ImplicitLocOpBuilder theBuilder(inst.getLoc(), inst);
397  const auto &modPorts = ports.at(inst.getModuleNameAttr().getAttr());
398 
399  // Fix up the Instance
400  SmallVector<PortInfo> instPorts; // Oh I wish ArrayRef was polymorphic.
401  for (auto p : modPorts) {
402  p.sym = {};
403  // Might need to partially copy stuff from the old instance.
404  p.annotations = AnnotationSet{mod.getContext()};
405  instPorts.push_back(p);
406  }
407  auto annos = inst.getAnnotations();
408  auto newOp = theBuilder.create<InstanceOp>(
409  instPorts, inst.getModuleName(), inst.getName(), inst.getNameKind(),
410  annos.getValue(), inst.getLayers(), inst.getLowerToBind(),
411  inst.getInnerSymAttr());
412 
413  auto oldDict = inst->getDiscardableAttrDictionary();
414  auto newDict = newOp->getDiscardableAttrDictionary();
415  auto oldNames = inst.getPortNamesAttr();
416  SmallVector<NamedAttribute> newAttrs;
417  for (auto na : oldDict)
418  if (!newDict.contains(na.getName()))
419  newOp->setDiscardableAttr(na.getName(), na.getValue());
420 
421  // Connect up the old instance users to the new instance
422  SmallVector<WireOp> bounce(inst.getNumResults());
423  for (auto p : modPorts) {
424  // No change? No bounce wire.
425  if (p.fieldID == 0) {
426  inst.getResult(p.portID).replaceAllUsesWith(
427  newOp.getResult(p.resultID));
428  continue;
429  }
430  if (!bounce[p.portID]) {
431  bounce[p.portID] = theBuilder.create<WireOp>(
432  inst.getResult(p.portID).getType(),
433  theBuilder.getStringAttr(
434  inst.getName() + "." +
435  oldNames[p.portID].cast<StringAttr>().getValue()));
436  inst.getResult(p.portID).replaceAllUsesWith(
437  bounce[p.portID].getResult());
438  }
439  // Connect up the Instance to the bounce wires
440  if (p.isInput())
441  emitConnect(theBuilder, newOp.getResult(p.resultID),
442  getValueByFieldID(theBuilder, bounce[p.portID].getResult(),
443  p.fieldID));
444  else
445  emitConnect(theBuilder,
446  getValueByFieldID(theBuilder, bounce[p.portID].getResult(),
447  p.fieldID),
448  newOp.getResult(p.resultID));
449  }
450  // Zero Width ports may have dangling connects since they are not preserved
451  // and do not have bounce wires.
452  for (auto *use : llvm::make_early_inc_range(inst->getUsers())) {
453  assert(isa<StrictConnectOp>(use) || isa<ConnectOp>(use));
454  use->erase();
455  }
456  inst->erase();
457  return;
458  });
459 }

471 // This is the main entrypoint for the lowering pass.
472 void LowerSignaturesPass::runOnOperation() {
473  LLVM_DEBUG(debugPassHeader(this) << "\n");
474  // Cached attr
475  AttrCache cache(&getContext());
476 
477  DenseMap<StringAttr, PortConversion> portMap;
478  auto circuit = getOperation();
479 
480  for (auto mod : circuit.getOps<FModuleLike>()) {
481  if (lowerModuleSignature(mod, mod.getConvention(), cache,
482  portMap[mod.getNameAttr()])
483  .failed())
484  return signalPassFailure();
485  }
486  parallelForEach(&getContext(), circuit.getOps<FModuleOp>(),
487  [&portMap](FModuleOp mod) { lowerModuleBody(mod, portMap); });
488 }
