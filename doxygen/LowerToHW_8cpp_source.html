50 /// Given a FIRRTL type, return the corresponding type for the HW dialect.
51 /// This returns a null type if it cannot be lowered.
52 static Type lowerType(Type type) {
53  auto firType = type.dyn_cast<FIRRTLType>();
54  if (!firType)
55  return {};
56 
57  // Ignore flip types.
58  firType = firType.getPassiveType();
59 
60  if (BundleType bundle = firType.dyn_cast<BundleType>()) {
61  mlir::SmallVector<hw::StructType::FieldInfo, 8> hwfields;
62  for (auto element : bundle) {
63  Type etype = lowerType(element.type);
64  if (!etype)
65  return {};
66  hwfields.push_back(hw::StructType::FieldInfo{element.name, etype});
67  }
68  return hw::StructType::get(type.getContext(), hwfields);
69  }
70  if (FVectorType vec = firType.dyn_cast<FVectorType>()) {
71  auto elemTy = lowerType(vec.getElementType());
72  if (!elemTy)
73  return {};
74  return hw::ArrayType::get(elemTy, vec.getNumElements());
75  }
76 
77  auto width = firType.getBitWidthOrSentinel();
78  if (width >= 0) // IntType, analog with known width, clock, etc.
79  return IntegerType::get(type.getContext(), width);
80 
81  return {};
82 }

84 /// This verifies that the target operation has been lowered to a legal
85 /// operation. This checks that the operation recursively has no FIRRTL
86 /// operations or types.
87 static LogicalResult verifyOpLegality(Operation *op) {
88  auto checkTypes = [](Operation *op) -> WalkResult {
89  // Check that this operation is not a FIRRTL op.
90  if (isa_and_nonnull<FIRRTLDialect>(op->getDialect()))
91  return op->emitError("Found unhandled FIRRTL operation '")
92  << op->getName() << "'";
93 
94  // Helper to check a TypeRange for any FIRRTL types.
95  auto checkTypeRange = [&](TypeRange types) -> LogicalResult {
96  if (llvm::any_of(types, [](Type type) {
97  return isa<FIRRTLDialect>(type.getDialect());
98  }))
99  return op->emitOpError("found unhandled FIRRTL type");
100  return success();
101  };
102 
103  // Check operand and result types.
104  if (failed(checkTypeRange(op->getOperandTypes())) ||
105  failed(checkTypeRange(op->getResultTypes())))
106  return WalkResult::interrupt();
107 
108  // Check the block argument types.
109  for (auto &region : op->getRegions())
110  for (auto &block : region)
111  if (failed(checkTypeRange(block.getArgumentTypes())))
112  return WalkResult::interrupt();
113 
114  // Continue to the next operation.
115  return WalkResult::advance();
116  };
117 
118  if (checkTypes(op).wasInterrupted() || op->walk(checkTypes).wasInterrupted())
119  return failure();
120  return success();
121 }

130 static Value castToFIRRTLType(Value val, Type type,
131  ImplicitLocOpBuilder &builder) {

201  if (rhs.empty())
202  return lhs;

369 void CircuitLoweringState::processRemainingAnnotations(
370  Operation *op, const AnnotationSet &annoSet) {
371  if (!enableAnnotationWarning || annoSet.empty())
372  return;
373  std::lock_guard<std::mutex> lock(annotationPrintingMtx);
374 
375  for (auto a : annoSet) {
376  auto inserted = pendingAnnotations.insert(a.getClass());
377  if (!inserted.second)
378  continue;
379 
380  // The following annotations are okay to be silently dropped at this point.
381  // This can occur for example if an annotation marks something in the IR as
382  // not to be processed by a pass, but that pass hasn't run anyway.
383  if (a.isClass(
384  // If the class is `circt.nonlocal`, it's not really an annotation,
385  // but part of a path specifier for another annotation which is
386  // non-local. We can ignore these path specifiers since there will
387  // be a warning produced for the real annotation.
388  "circt.nonlocal",
389  // The following are either consumed by a pass running before
390  // LowerToHW, or they have no effect if the pass doesn't run at all.
391  // If the accompanying pass runs on the HW dialect, then LowerToHW
392  // should have consumed and processed these into an attribute on the
393  // output.
394  "sifive.enterprise.firrtl.DontObfuscateModuleAnnotation",
395  noDedupAnnoClass,
396  // The following are inspected (but not consumed) by FIRRTL/GCT
397  // passes that have all run by now. Since no one is responsible for
398  // consuming these, they will linger around and can be ignored.
399  "sifive.enterprise.firrtl.ScalaClassAnnotation", dutAnnoClass,
400  metadataDirectoryAttrName,
401  "sifive.enterprise.firrtl.ElaborationArtefactsDirectory",
402  testBenchDirAnnoClass,
403  "sifive.enterprise.grandcentral.phases.SubCircuitsTargetDirectory",
404  // This annotation is used to mark which external modules are
405  // imported blackboxes from the BlackBoxReader pass.
406  "firrtl.transforms.BlackBox",
407  // This annotation is used by several GrandCentral passes.
408  extractGrandCentralClass,
409  // The following will be handled while lowering the verification
410  // ops.
411  extractAssertAnnoClass, extractAssumeAnnoClass,
412  extractCoverageAnnoClass,
413  // The following will be handled after lowering FModule ops, since
414  // they are still needed on the circuit until after lowering
415  // FModules.
416  moduleHierAnnoClass, testHarnessHierAnnoClass,
417  blackBoxTargetDirAnnoClass))
418  continue;
419 
420  mlir::emitWarning(op->getLoc(), "unprocessed annotation:'" + a.getClass() +
421  "' still remaining after LowerToHW");
422  }

467 /// Run on the firrtl.circuit operation, lowering any firrtl.module operations
468 /// it contains.
469 void FIRRTLModuleLowering::runOnOperation() {
470 
471  // We run on the top level modules in the IR blob. Start by finding the
472  // firrtl.circuit within it. If there is none, then there is nothing to do.
473  auto *topLevelModule = getOperation().getBody();
474 
475  // Find the single firrtl.circuit in the module.
476  CircuitOp circuit;
477  for (auto &op : *topLevelModule) {
478  if ((circuit = dyn_cast<CircuitOp>(&op)))
479  break;
480  }
481 
482  if (!circuit)
483  return;
484 
485  auto *circuitBody = circuit.getBody();
486 
487  // Keep track of the mapping from old to new modules. The result may be null
488  // if lowering failed.
489  CircuitLoweringState state(circuit, enableAnnotationWarning,
490  &getAnalysis<InstanceGraph>(),
491  &getAnalysis<NLATable>());
492 
493  SmallVector<FModuleOp, 32> modulesToProcess;
494 
495  AnnotationSet circuitAnno(circuit);
496  moveVerifAnno(getOperation(), circuitAnno, extractAssertAnnoClass,
497  "firrtl.extract.assert");
498  moveVerifAnno(getOperation(), circuitAnno, extractAssumeAnnoClass,
499  "firrtl.extract.assume");
500  moveVerifAnno(getOperation(), circuitAnno, extractCoverageAnnoClass,
501  "firrtl.extract.cover");
502  circuitAnno.removeAnnotationsWithClass(
503  extractAssertAnnoClass, extractAssumeAnnoClass, extractCoverageAnnoClass);
504 
505  state.processRemainingAnnotations(circuit, circuitAnno);
506  // Iterate through each operation in the circuit body, transforming any
507  // FModule's we come across. If any module fails to lower, return early.
508  for (auto &op : make_early_inc_range(circuitBody->getOperations())) {
509  TypeSwitch<Operation *>(&op)
510  .Case<FModuleOp>([&](auto module) {
511  auto loweredMod = lowerModule(module, topLevelModule, state);
512  if (!loweredMod)
513  return signalPassFailure();
514 
515  state.oldToNewModuleMap[&op] = loweredMod;
516  modulesToProcess.push_back(module);
517  })
518  .Case<FExtModuleOp>([&](auto extModule) {
519  auto loweredMod = lowerExtModule(extModule, topLevelModule, state);
520  if (!loweredMod)
521  return signalPassFailure();
522  state.oldToNewModuleMap[&op] = loweredMod;
523  })
524  .Case<FMemModuleOp>([&](auto memModule) {
525  auto loweredMod = lowerMemModule(memModule, topLevelModule, state);
526  if (!loweredMod)
527  return signalPassFailure();
528  state.oldToNewModuleMap[&op] = loweredMod;
529  })
530  .Case<HierPathOp>([&](auto nla) {
531  // Just drop it.
532  })
533  .Default([&](Operation *op) {
534  // We don't know what this op is. If it has no illegal FIRRTL types,
535  // we can forward the operation. Otherwise, we emit an error and drop
536  // the operation from the circuit.
537  if (succeeded(verifyOpLegality(op)))
538  op->moveBefore(topLevelModule, topLevelModule->end());
539  else
540  return signalPassFailure();
541  });
542  }
543  // Handle the creation of the module hierarchy metadata.
544 
545  // Collect the two sets of hierarchy files from the circuit. Some of them will
546  // be rooted at the test harness, the others will be rooted at the DUT.
547  SmallVector<Attribute> dutHierarchyFiles;
548  SmallVector<Attribute> testHarnessHierarchyFiles;
549  circuitAnno.removeAnnotations([&](Annotation annotation) {
550  if (annotation.isClass(moduleHierAnnoClass)) {
551  auto file = hw::OutputFileAttr::getFromFilename(
552  &getContext(),
553  annotation.getMember<StringAttr>("filename").getValue(),
554  /*excludeFromFileList=*/true);
555  dutHierarchyFiles.push_back(file);
556  return true;
557  }
558  if (annotation.isClass(testHarnessHierAnnoClass)) {
559  auto file = hw::OutputFileAttr::getFromFilename(
560  &getContext(),
561  annotation.getMember<StringAttr>("filename").getValue(),
562  /*excludeFromFileList=*/true);
563  // If there is no testHarness, we print the hiearchy for this file
564  // starting at the DUT.
565  if (state.getTestHarness())
566  testHarnessHierarchyFiles.push_back(file);
567  else
568  dutHierarchyFiles.push_back(file);
569  return true;
570  }
571  return false;
572  });
573  // Attach the lowered form of these annotations.
574  if (!dutHierarchyFiles.empty())
575  state.oldToNewModuleMap[state.getDut()]->setAttr(
576  moduleHierarchyFileAttrName,
577  ArrayAttr::get(&getContext(), dutHierarchyFiles));
578  if (!testHarnessHierarchyFiles.empty())
579  state.oldToNewModuleMap[state.getTestHarness()]->setAttr(
580  moduleHierarchyFileAttrName,
581  ArrayAttr::get(&getContext(), testHarnessHierarchyFiles));
582 
583  // Collect all the memories in the module. Construct the FirMemory, set the
584  // DUT flag and also record the Memop.
585  auto collectFIRRTLMemories =
586  [&state](FModuleOp module) -> SmallVector<FirMemory> {
587  SmallVector<FirMemory> retval;
588  // Check if this module is in the DUT hierarchy.
589  bool isInDut = state.isInDUT(module);
590  for (auto op : module.getBody()->getOps<MemOp>()) {
591  auto sum = op.getSummary();
592  sum.isInDut = isInDut;
593  sum.op = op;
594  retval.push_back(sum);
595  }
596  return retval;
597  };

599  // 1. First collect the memories as FirMemory, set the DUT flag and also
600  // record the MemOp.
601  // 2. Then Dedup the memories using the mergeFIRRTLMemories routine. This uses
602  // the memory parameters to merge memories with the exactly same parameters.
603  // 3. Then For each of the deduped memories, create the HWModuleGeneratedOp,
604  // and assign a unique name to the wrapper module based on the MemOp name. It
605  // is important to use the MemOp name because appropriate prefixes have to be
606  // respected, also the correct output directory needs to be set based on the
607  // DUT or testbench hierarchy.
608  SmallVector<FirMemory> memories;
609  if (getContext().isMultithreadingEnabled()) {
610  memories = llvm::parallelTransformReduce(
611  modulesToProcess.begin(), modulesToProcess.end(),
612  SmallVector<FirMemory>(), mergeFIRRTLMemories, collectFIRRTLMemories);
613  } else {
614  for (auto m : modulesToProcess)
615  memories = mergeFIRRTLMemories(memories, collectFIRRTLMemories(m));
616  }
617  if (!memories.empty())
618  lowerMemoryDecls(memories, state);
619 
620  // Now that we've lowered all of the modules, move the bodies over and
621  // update any instances that refer to the old modules.
622  auto result = mlir::failableParallelForEachN(
623  &getContext(), 0, modulesToProcess.size(), [&](auto index) {
624  return lowerModuleBody(modulesToProcess[index], state);
625  });
626 
627  // If any module bodies failed to lower, return early.
628  if (failed(result))
629  return signalPassFailure();
630 
631  // Move binds from inside modules to outside modules.
632  for (auto bind : state.binds) {
633  bind->moveBefore(bind->getParentOfType<hw::HWModuleOp>());
634  }
635 
636  // Finally delete all the old modules.
637  for (auto oldNew : state.oldToNewModuleMap)
638  oldNew.first->erase();
639 
640  // Emit all the macros and preprocessor gunk at the start of the file.
641  lowerFileHeader(circuit, state);
642 
643  // Now that the modules are moved over, remove the Circuit.
644  circuit.erase();
645 }

647 void FIRRTLModuleLowering::lowerMemoryDecls(ArrayRef<FirMemory> mems,
648  CircuitLoweringState &state) {
649  assert(!mems.empty());
650  auto namesp = CircuitNamespace(state.circuitOp);
651  state.used_RANDOMIZE_MEM_INIT = 1;
652  // Insert memories at the bottom of the file.
653  OpBuilder b(state.circuitOp);
654  b.setInsertionPointAfter(state.circuitOp);
655  std::array<StringRef, 11> schemaFields = {
656  "depth", "numReadPorts", "numWritePorts", "numReadWritePorts",
657  "readLatency", "writeLatency", "width", "maskGran",
658  "readUnderWrite", "writeUnderWrite", "writeClockIDs"};
659  auto schemaFieldsAttr = b.getStrArrayAttr(schemaFields);
660  auto schema = b.create<hw::HWGeneratorSchemaOp>(
661  mems.front().loc, "FIRRTLMem", "FIRRTL_Memory", schemaFieldsAttr);
662  auto memorySchema = SymbolRefAttr::get(schema);
663 
664  Type b1Type = IntegerType::get(&getContext(), 1);
665 
666  for (auto &mem : mems) {
667  SmallVector<hw::PortInfo> ports;
668  size_t inputPin = 0;
669  size_t outputPin = 0;
670  // We don't need a single bit mask, it can be combined with enable. Create
671  // an unmasked memory if maskBits = 1.
672 
673  auto makePortCommon = [&](StringRef prefix, size_t idx, Type bAddrType) {
674  ports.push_back({b.getStringAttr(prefix + Twine(idx) + "_addr"),
675  PortDirection::INPUT, bAddrType, inputPin++});
676  ports.push_back({b.getStringAttr(prefix + Twine(idx) + "_en"),
677  PortDirection::INPUT, b1Type, inputPin++});
678  ports.push_back({b.getStringAttr(prefix + Twine(idx) + "_clk"),
679  PortDirection::INPUT, b1Type, inputPin++});
680  };
681 
682  Type bDataType =
683  IntegerType::get(&getContext(), std::max((size_t)1, mem.dataWidth));
684  Type maskType = IntegerType::get(&getContext(), mem.maskBits);
685 
686  Type bAddrType = IntegerType::get(
687  &getContext(), std::max(1U, llvm::Log2_64_Ceil(mem.depth)));
688 
689  for (size_t i = 0, e = mem.numReadPorts; i != e; ++i) {
690  makePortCommon("R", i, bAddrType);
691  ports.push_back({b.getStringAttr("R" + Twine(i) + "_data"),
692  PortDirection::OUTPUT, bDataType, outputPin++});
693  }
694  for (size_t i = 0, e = mem.numReadWritePorts; i != e; ++i) {
695  makePortCommon("RW", i, bAddrType);
696  ports.push_back({b.getStringAttr("RW" + Twine(i) + "_wmode"),
697  PortDirection::INPUT, b1Type, inputPin++});
698  ports.push_back({b.getStringAttr("RW" + Twine(i) + "_wdata"),
699  PortDirection::INPUT, bDataType, inputPin++});
700  ports.push_back({b.getStringAttr("RW" + Twine(i) + "_rdata"),
701  PortDirection::OUTPUT, bDataType, outputPin++});
702  // Ignore mask port, if maskBits =1
703  if (mem.isMasked)
704  ports.push_back({b.getStringAttr("RW" + Twine(i) + "_wmask"),
705  PortDirection::INPUT, maskType, inputPin++});
706  }
707 
708  for (size_t i = 0, e = mem.numWritePorts; i != e; ++i) {
709  makePortCommon("W", i, bAddrType);
710  ports.push_back({b.getStringAttr("W" + Twine(i) + "_data"),
711  PortDirection::INPUT, bDataType, inputPin++});
712  // Ignore mask port, if maskBits =1
713  if (mem.isMasked)
714  ports.push_back({b.getStringAttr("W" + Twine(i) + "_mask"),
715  PortDirection::INPUT, maskType, inputPin++});
716  }
717 
718  // Mask granularity is the number of data bits that each mask bit can
719  // guard. By default it is equal to the data bitwidth.
720  auto maskGran = mem.isMasked ? mem.dataWidth / mem.maskBits : mem.dataWidth;
721  NamedAttribute genAttrs[] = {
722  b.getNamedAttr("depth", b.getI64IntegerAttr(mem.depth)),
723  b.getNamedAttr("numReadPorts", b.getUI32IntegerAttr(mem.numReadPorts)),
724  b.getNamedAttr("numWritePorts",
725  b.getUI32IntegerAttr(mem.numWritePorts)),
726  b.getNamedAttr("numReadWritePorts",
727  b.getUI32IntegerAttr(mem.numReadWritePorts)),
728  b.getNamedAttr("readLatency", b.getUI32IntegerAttr(mem.readLatency)),
729  b.getNamedAttr("writeLatency", b.getUI32IntegerAttr(mem.writeLatency)),
730  b.getNamedAttr("width", b.getUI32IntegerAttr(mem.dataWidth)),
731  b.getNamedAttr("maskGran", b.getUI32IntegerAttr(maskGran)),
732  b.getNamedAttr("readUnderWrite",
733  b.getUI32IntegerAttr(mem.readUnderWrite)),
734  b.getNamedAttr("writeUnderWrite",
735  hw::WUWAttr::get(b.getContext(), mem.writeUnderWrite)),
736  b.getNamedAttr("writeClockIDs", b.getI32ArrayAttr(mem.writeClockIDs))};
737 
738  // Make the global module for the memory
739  // Set a name for the memory wrapper module, the combMem is an arbitrary
740  // suffix. It is important to derive the name from the original MemOp name,
741  // to respect the corresponding prefixes..
742  auto memoryName = b.getStringAttr(
743  namesp.newName(static_cast<MemOp>(mem.op).name() + "_combMem"));
744  // Now record this generated name for the corresponding FirMemory name,
745  // because all the memories that have the same FirMemory name, will now use
746  // this new generated name at the Instance Op. Basically this is the name
747  // used for all the deduped memories.
748  state.memoryNameMap[mem.getFirMemoryName()] = memoryName;
749  auto genOp = b.create<hw::HWModuleGeneratedOp>(
750  mem.loc, memorySchema, memoryName, ports, StringRef(), ArrayAttr(),
751  genAttrs);
752  // Also set the appropriate directory.
753  if (!mem.isInDut)
754  if (auto testBenchDir = state.getTestBenchDirectory())
755  genOp->setAttr("output_file", testBenchDir);
756  }

759 /// Emit the file header that defines a bunch of macros.
760 void FIRRTLModuleLowering::lowerFileHeader(CircuitOp op,
761  CircuitLoweringState &state) {
762  // Intentionally pass an UnknownLoc here so we don't get line number
763  // comments on the output of this boilerplate in generated Verilog.
764  ImplicitLocOpBuilder b(UnknownLoc::get(&getContext()), op);
765 
766  // TODO: We could have an operation for macros and uses of them, and
767  // even turn them into symbols so we can DCE unused macro definitions.
768  auto emitString = [&](StringRef verilogString) {
769  b.create<sv::VerbatimOp>(verilogString);
770  };
771 
772  // Helper function to emit a "#ifdef guard" with a `define in the then and
773  // optionally in the else branch.
774  auto emitGuardedDefine = [&](const char *guard, const char *defineTrue,
775  const char *defineFalse = nullptr) {
776  std::string define = "`define ";
777  if (!defineFalse) {
778  assert(defineTrue && "didn't define anything");
779  b.create<sv::IfDefOp>(guard, [&]() { emitString(define + defineTrue); });
780  } else {
781  b.create<sv::IfDefOp>(
782  guard,
783  [&]() {
784  if (defineTrue)
785  emitString(define + defineTrue);
786  },
787  [&]() { emitString(define + defineFalse); });
788  }
789  };
790 
791  // If none of the macros are needed, then don't emit any header at all, not
792  // even the header comment.
793  if (!state.used_RANDOMIZE_GARBAGE_ASSIGN && !state.used_RANDOMIZE_REG_INIT &&
794  !state.used_RANDOMIZE_MEM_INIT && !state.used_PRINTF_COND &&
795  !state.used_ASSERT_VERBOSE_COND && !state.used_STOP_COND)
796  return;
797 
798  emitString("// Standard header to adapt well known macros to our needs.");
799 
800  bool needRandom = false;
801  if (state.used_RANDOMIZE_GARBAGE_ASSIGN) {
802  emitGuardedDefine("RANDOMIZE_GARBAGE_ASSIGN", "RANDOMIZE");
803  needRandom = true;
804  }
805  if (state.used_RANDOMIZE_REG_INIT) {
806  emitGuardedDefine("RANDOMIZE_REG_INIT", "RANDOMIZE");
807  needRandom = true;
808  }
809  if (state.used_RANDOMIZE_MEM_INIT) {
810  emitGuardedDefine("RANDOMIZE_MEM_INIT", "RANDOMIZE");
811  needRandom = true;
812  }
813 
814  if (needRandom) {
815  emitString("\n// RANDOM may be set to an expression that produces a 32-bit "
816  "random unsigned value.");
817  emitGuardedDefine("RANDOM", nullptr, "RANDOM $random");
818  }
819 
820  if (state.used_PRINTF_COND) {
821  emitString("\n// Users can define 'PRINTF_COND' to add an extra gate to "
822  "prints.");
823  emitGuardedDefine("PRINTF_COND", "PRINTF_COND_ (`PRINTF_COND)",
824  "PRINTF_COND_ 1");
825  }
826 
827  if (state.used_ASSERT_VERBOSE_COND) {
828  emitString("\n// Users can define 'ASSERT_VERBOSE_COND' to add an extra "
829  "gate to assert error printing.");
830  emitGuardedDefine("ASSERT_VERBOSE_COND",
831  "ASSERT_VERBOSE_COND_ (`ASSERT_VERBOSE_COND)",
832  "ASSERT_VERBOSE_COND_ 1");
833  }
834 
835  if (state.used_STOP_COND) {
836  emitString("\n// Users can define 'STOP_COND' to add an extra gate "
837  "to stop conditions.");
838  emitGuardedDefine("STOP_COND", "STOP_COND_ (`STOP_COND)", "STOP_COND_ 1");
839  }
840 
841  if (needRandom) {
842  emitString("\n// Users can define INIT_RANDOM as general code that gets "
843  "injected "
844  "into the\n// initializer block for modules with registers.");
845  emitGuardedDefine("INIT_RANDOM", nullptr, "INIT_RANDOM");
846 
847  emitString(
848  "\n// If using random initialization, you can also define "
849  "RANDOMIZE_DELAY to\n// customize the delay used, otherwise 0.002 "
850  "is used.");
851  emitGuardedDefine("RANDOMIZE_DELAY", nullptr, "RANDOMIZE_DELAY 0.002");
852 
853  emitString("\n// Define INIT_RANDOM_PROLOG_ for use in our modules below.");
854  b.create<sv::IfDefOp>(
855  "RANDOMIZE",
856  [&]() {
857  emitGuardedDefine(
858  "VERILATOR", "INIT_RANDOM_PROLOG_ `INIT_RANDOM",
859  "INIT_RANDOM_PROLOG_ `INIT_RANDOM #`RANDOMIZE_DELAY begin end");
860  },
861  [&]() { emitString("`define INIT_RANDOM_PROLOG_"); });
862  }
863 
864  if (state.used_RANDOMIZE_GARBAGE_ASSIGN) {
865  emitString("\n// RANDOMIZE_GARBAGE_ASSIGN enable range checks for mem "
866  "assignments.");
867  b.create<sv::IfDefOp>(
868  "RANDOMIZE_GARBAGE_ASSIGN",
869  [&]() {
870  emitString(
871  "`define RANDOMIZE_GARBAGE_ASSIGN_BOUND_CHECK(INDEX, VALUE, "
872  "SIZE) \\");
873  emitString(" ((INDEX) < (SIZE) ? (VALUE) : {`RANDOM})");
874  },
875  [&]() {
876  emitString("`define RANDOMIZE_GARBAGE_ASSIGN_BOUND_CHECK(INDEX, "
877  "VALUE, SIZE) (VALUE)");
878  });
879  }
880 
881  // Blank line to separate the header from the modules.
882  emitString("");
883 }

885 LogicalResult FIRRTLModuleLowering::lowerPorts(
886  ArrayRef<PortInfo> firrtlPorts, SmallVectorImpl<hw::PortInfo> &ports,
887  Operation *moduleOp, CircuitLoweringState &loweringState) {
888  ports.reserve(firrtlPorts.size());
889  size_t numArgs = 0;
890  size_t numResults = 0;
891  for (auto firrtlPort : firrtlPorts) {
892  hw::PortInfo hwPort;
893  hwPort.name = firrtlPort.name;
894  hwPort.type = lowerType(firrtlPort.type);
895  hwPort.sym = firrtlPort.sym;
896 
897  // We can't lower all types, so make sure to cleanly reject them.
898  if (!hwPort.type) {
899  moduleOp->emitError("cannot lower this port type to HW");
900  return failure();
901  }
902 
903  // If this is a zero bit port, just drop it. It doesn't matter if it is
904  // input, output, or inout. We don't want these at the HW level.
905  if (hwPort.type.isInteger(0))
906  continue;
907 
908  // Figure out the direction of the port.
909  if (firrtlPort.isOutput()) {
910  hwPort.direction = hw::PortDirection::OUTPUT;
911  hwPort.argNum = numResults++;
912  } else if (firrtlPort.isInput()) {
913  hwPort.direction = hw::PortDirection::INPUT;
914  hwPort.argNum = numArgs++;
915  } else {
916  // If the port is an inout bundle or contains an analog type, then it is
917  // implicitly inout.
918  hwPort.type = hw::InOutType::get(hwPort.type);
919  hwPort.direction = hw::PortDirection::INOUT;
920  hwPort.argNum = numArgs++;
921  }
922  ports.push_back(hwPort);
923  loweringState.processRemainingAnnotations(moduleOp, firrtlPort.annotations);
924  }
925  return success();
926 }

928 /// Map the parameter specifier on the specified extmodule into the HWModule
929 /// representation for parameters. If `ignoreValues` is true, all the values
930 /// are dropped.
931 static ArrayAttr getHWParameters(FExtModuleOp module, bool ignoreValues) {
932  auto params = llvm::map_range(
933  module.parameters(), [](Attribute a) { return a.cast<ParamDeclAttr>(); });
934  if (params.empty())
935  return {};
936 
937  Builder builder(module);
938 
939  // Map the attributes over from firrtl attributes to HW attributes
940  // directly. MLIR's DictionaryAttr always stores keys in the dictionary
941  // in sorted order which is nicely stable.
942  SmallVector<Attribute> newParams;
943  for (const ParamDeclAttr &entry : params) {
944  auto name = entry.getName();
945  auto type = TypeAttr::get(entry.getValue().getType());
946  auto value = ignoreValues ? Attribute() : entry.getValue();
947  auto paramAttr =
948  hw::ParamDeclAttr::get(builder.getContext(), name, type, value);
949  newParams.push_back(paramAttr);
950  }
951  return builder.getArrayAttr(newParams);
952 }

1010 /// Run on each firrtl.module, transforming it from an firrtl.module into an
1011 /// hw.module, then deleting the old one.
1012 hw::HWModuleOp
1013 FIRRTLModuleLowering::lowerModule(FModuleOp oldModule, Block *topLevelModule,
1014  CircuitLoweringState &loweringState) {
1015  // Map the ports over, lowering their types as we go.
1016  SmallVector<PortInfo> firrtlPorts = oldModule.getPorts();
1017  SmallVector<hw::PortInfo, 8> ports;
1018  if (failed(lowerPorts(firrtlPorts, ports, oldModule, loweringState)))
1019  return {};
1020 
1021  // Build the new hw.module op.
1022  auto builder = OpBuilder::atBlockEnd(topLevelModule);
1023  auto nameAttr = builder.getStringAttr(oldModule.getName());
1024  auto newModule =
1025  builder.create<hw::HWModuleOp>(oldModule.getLoc(), nameAttr, ports);
1026  if (auto outputFile = oldModule->getAttr("output_file"))
1027  newModule->setAttr("output_file", outputFile);
1028 
1029  // If the circuit has an entry point, set all other modules private.
1030  // Otherwise, mark all modules as public.
1031  SymbolTable::setSymbolVisibility(newModule,
1032  SymbolTable::getSymbolVisibility(oldModule));
1033 
1034  // Transform module annotations
1035  AnnotationSet annos(oldModule);
1036 
1037  if (annos.removeAnnotation(verifBlackBoxAnnoClass))
1038  newModule->setAttr("firrtl.extract.cover.extra", builder.getUnitAttr());
1039 
1040  // If this is in the test harness, make sure it goes to the test directory.
1041  if (auto testBenchDir = loweringState.getTestBenchDirectory())
1042  if (loweringState.isInTestHarness(oldModule)) {
1043  newModule->setAttr("output_file", testBenchDir);
1044  newModule->setAttr("firrtl.extract.do_not_extract",
1045  builder.getUnitAttr());
1046  newModule.commentAttr(builder.getStringAttr("VCS coverage exclude_file"));
1047  }
1048 
1049  bool failed = false;
1050  // Remove ForceNameAnnotations by generating verilogNames on instances.
1051  annos.removeAnnotations([&](Annotation anno) {
1052  if (!anno.isClass(forceNameAnnoClass))
1053  return false;
1054 
1055  auto sym = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal");
1056  // This must be a non-local annotation due to how the Chisel API is
1057  // implemented.
1058  //
1059  // TODO: handle this in some sensible way based on what the SFC does with
1060  // a local annotation.
1061  if (!sym) {
1062  auto diag = oldModule.emitOpError()
1063  << "contains a '" << forceNameAnnoClass
1064  << "' that is not a non-local annotation";
1065  diag.attachNote() << "the erroneous annotation is '" << anno.getDict()
1066  << "'\n";
1067  failed = true;
1068  return false;
1069  }
1070 
1071  auto nla = loweringState.nlaTable->getNLA(sym.getAttr());
1072  // The non-local anchor must exist.
1073  //
1074  // TODO: handle this with annotation verification.
1075  if (!nla) {
1076  auto diag = oldModule.emitOpError()
1077  << "contains a '" << forceNameAnnoClass
1078  << "' whose non-local symbol, '" << sym
1079  << "' does not exist in the circuit";
1080  diag.attachNote() << "the erroneous annotation is '" << anno.getDict();
1081  failed = true;
1082  return false;
1083  }
1084 
1085  // Add the forced name to global state (keyed by a pseudo-inner name ref).
1086  // Error out if this key is alredy in use.
1087  //
1088  // TODO: this error behavior can be relaxed to always overwrite with the
1089  // new forced name (the bug-compatible behavior of the Chisel
1090  // implementation) or fixed to duplicate modules such that the naming can
1091  // be applied.
1092  auto inst =
1093  nla.namepath().getValue().take_back(2)[0].cast<hw::InnerRefAttr>();
1094  auto inserted = loweringState.instanceForceNames.insert(
1095  {{inst.getModule(), inst.getName()}, anno.getMember("name")});
1096  if (!inserted.second &&
1097  (anno.getMember("name") != (inserted.first->second))) {
1098  auto diag = oldModule.emitError()
1099  << "contained multiple '" << forceNameAnnoClass
1100  << "' with different names: " << inserted.first->second
1101  << " was not " << anno.getMember("name");
1102  diag.attachNote() << "the erroneous annotation is '" << anno.getDict()
1103  << "'";
1104  failed = true;
1105  return false;
1106  }
1107  return true;
1108  });
1109 
1110  if (failed)
1111  return {};
1112 
1113  loweringState.processRemainingAnnotations(oldModule, annos);
1114  return newModule;
1115 }

1117 /// Given a value of analog type, check to see the only use of it is an
1118 /// attach. If so, remove the attach and return the value being attached to
1119 /// it, converted to an HW inout type. If this isn't a situation we can
1120 /// handle, just return null.
1121 static Value tryEliminatingAttachesToAnalogValue(Value value,
1122  Operation *insertPoint) {
1123  if (!value.hasOneUse())
1124  return {};
1125 
1126  auto attach = dyn_cast<AttachOp>(*value.user_begin());
1127  if (!attach || attach.getNumOperands() != 2)
1128  return {};
1129 
1130  // Don't optimize zero bit analogs.
1131  auto loweredType = lowerType(value.getType());
1132  if (loweredType.isInteger(0))
1133  return {};
1134 
1135  // Check to see if the attached value dominates the insertion point. If
1136  // not, just fail.
1137  auto attachedValue = attach.getOperand(attach.getOperand(0) == value);
1138  auto *op = attachedValue.getDefiningOp();
1139  if (op && op->getBlock() == insertPoint->getBlock() &&
1140  !op->isBeforeInBlock(insertPoint))
1141  return {};
1142 
1143  attach.erase();
1144 
1145  ImplicitLocOpBuilder builder(insertPoint->getLoc(), insertPoint);
1146  return castFromFIRRTLType(attachedValue, hw::InOutType::get(loweredType),
1147  builder);

1150 /// Given a value of flip type, check to see if all of the uses of it are
1151 /// connects. If so, remove the connects and return the value being connected
1152 /// to it, converted to an HW type. If this isn't a situation we can handle,
1153 /// just return null.
1154 ///
1155 /// This can happen when there are no connects to the value. The 'mergePoint'
1156 /// location is where a 'hw.merge' operation should be inserted if needed.
1157 static Value tryEliminatingConnectsToValue(Value flipValue,
1158  Operation *insertPoint) {
1159  // Handle analog's separately.
1160  if (flipValue.getType().isa<AnalogType>())
1161  return tryEliminatingAttachesToAnalogValue(flipValue, insertPoint);
1162 
1163  Operation *connectOp = nullptr;
1164  for (auto &use : flipValue.getUses()) {
1165  // We only know how to deal with connects where this value is the
1166  // destination.
1167  if (use.getOperandNumber() != 0)
1168  return {};
1169  if (!isa<ConnectOp, StrictConnectOp>(use.getOwner()))
1170  return {};
1171 
1172  // We only support things with a single connect.
1173  if (connectOp)
1174  return {};
1175  connectOp = use.getOwner();
1176  }
1177 
1178  // We don't have an HW equivalent of "poison" so just don't special case
1179  // the case where there are no connects other uses of an output.
1180  if (!connectOp)
1181  return {}; // TODO: Emit an sv.constant here since it is unconnected.
1182 
1183  // Don't special case zero-bit results.
1184  auto loweredType = lowerType(flipValue.getType());
1185  if (loweredType.isInteger(0))
1186  return {};
1187 
1188  // Convert each connect into an extended version of its operand being
1189  // output.
1190  ImplicitLocOpBuilder builder(insertPoint->getLoc(), insertPoint);
1191 
1192  auto connectSrc = connectOp->getOperand(1);
1193 
1194  // Convert fliped sources to passive sources.
1195  if (!connectSrc.getType().cast<FIRRTLType>().isPassive())
1196  connectSrc =
1197  builder
1198  .create<mlir::UnrealizedConversionCastOp>(
1199  connectSrc.getType().cast<FIRRTLType>().getPassiveType(),
1200  connectSrc)
1201  .getResult(0);
1202 
1203  // We know it must be the destination operand due to the types, but the
1204  // source may not match the destination width.
1205  auto destTy = flipValue.getType().cast<FIRRTLType>().getPassiveType();
1206 
1207  if (!destTy.isGround()) {
1208  // If types are not ground type and they don't match, we give up.
1209  if (destTy != connectSrc.getType().cast<FIRRTLType>())
1210  return {};
1211  } else if (destTy.getBitWidthOrSentinel() !=
1212  connectSrc.getType().cast<FIRRTLType>().getBitWidthOrSentinel()) {
1213  // The only type mismatchs we care about is due to integer width
1214  // differences.
1215  auto destWidth = destTy.getBitWidthOrSentinel();
1216  assert(destWidth != -1 && "must know integer widths");
1217  connectSrc = builder.createOrFold<PadPrimOp>(destTy, connectSrc, destWidth);
1218  }
1219 
1220  // Remove the connect and use its source as the value for the output.
1221  connectOp->erase();
1222 
1223  // Convert from FIRRTL type to builtin type.
1224  return castFromFIRRTLType(connectSrc, loweredType, builder);
1225 }

1230  for (auto op : structValue.getUsers()) {

1243 /// Now that we have the operations for the hw.module's corresponding to the
1244 /// firrtl.module's, we can go through and move the bodies over, updating the
1245 /// ports and instances.
1246 LogicalResult
1247 FIRRTLModuleLowering::lowerModuleBody(FModuleOp oldModule,
1248  CircuitLoweringState &loweringState) {
1249  auto newModule =
1250  dyn_cast_or_null<hw::HWModuleOp>(loweringState.getNewModule(oldModule));
1251  // Don't touch modules if we failed to lower ports.
1252  if (!newModule)
1253  return success();
1254 
1255  ImplicitLocOpBuilder bodyBuilder(oldModule.getLoc(), newModule.body());
1256 
1257  // Use a placeholder instruction be a cursor that indicates where we want to
1258  // move the new function body to. This is important because we insert some
1259  // ops at the start of the function and some at the end, and the body is
1260  // currently empty to avoid iterator invalidation.
1261  auto cursor = bodyBuilder.create<hw::ConstantOp>(APInt(1, 1));
1262  bodyBuilder.setInsertionPoint(cursor);
1263 
1264  // Insert argument casts, and re-vector users in the old body to use them.
1265  SmallVector<PortInfo> ports = oldModule.getPorts();
1266  assert(oldModule.body().getNumArguments() == ports.size() &&
1267  "port count mismatch");
1268 
1269  size_t nextNewArg = 0;
1270  size_t firrtlArg = 0;
1271  SmallVector<Value, 4> outputs;
1272 
1273  // This is the terminator in the new module.
1274  auto outputOp = newModule.getBodyBlock()->getTerminator();
1275  ImplicitLocOpBuilder outputBuilder(oldModule.getLoc(), outputOp);
1276 
1277  for (auto &port : ports) {
1278  // Inputs and outputs are both modeled as arguments in the FIRRTL level.
1279  auto oldArg = oldModule.body().getArgument(firrtlArg++);
1280 
1281  bool isZeroWidth =
1282  port.type.cast<FIRRTLType>().getBitWidthOrSentinel() == 0;
1283 
1284  if (!port.isOutput() && !isZeroWidth) {
1285  // Inputs and InOuts are modeled as arguments in the result, so we can
1286  // just map them over. We model zero bit outputs as inouts.
1287  Value newArg = newModule.body().getArgument(nextNewArg++);
1288 
1289  // Cast the argument to the old type, reintroducing sign information in
1290  // the hw.module body.
1291  newArg = castToFIRRTLType(newArg, oldArg.getType(), bodyBuilder);
1292  // Switch all uses of the old operands to the new ones.
1293  oldArg.replaceAllUsesWith(newArg);
1294  continue;
1295  }
1296 
1297  // We lower zero width inout and outputs to a wire that isn't connected to
1298  // anything outside the module. Inputs are lowered to zero.
1299  if (isZeroWidth && port.isInput()) {
1300  Value newArg = bodyBuilder.create<WireOp>(
1301  port.type, "." + port.getName().str() + ".0width_input");
1302  oldArg.replaceAllUsesWith(newArg);
1303  continue;
1304  }
1305 
1306  if (auto value = tryEliminatingConnectsToValue(oldArg, outputOp)) {
1307  // If we were able to find the value being connected to the output,
1308  // directly use it!
1309  outputs.push_back(value);
1310  assert(oldArg.use_empty() && "should have removed all uses of oldArg");
1311  continue;
1312  }
1313 
1314  // Outputs need a temporary wire so they can be connect'd to, which we
1315  // then return.
1316  Value newArg = bodyBuilder.create<WireOp>(
1317  port.type, "." + port.getName().str() + ".output");
1318  // Switch all uses of the old operands to the new ones.
1319  oldArg.replaceAllUsesWith(newArg);
1320 
1321  // Don't output zero bit results or inouts.
1322  auto resultHWType = lowerType(port.type);
1323  if (!resultHWType.isInteger(0)) {
1324  auto output = castFromFIRRTLType(newArg, resultHWType, outputBuilder);
1325  outputs.push_back(output);
1326  }
1327  }
1328 
1329  // Update the hw.output terminator with the list of outputs we have.
1330  outputOp->setOperands(outputs);
1331 
1332  // Finally splice the body over, don't move the old terminator over though.
1333  auto &oldBlockInstList = oldModule.getBody()->getOperations();
1334  auto &newBlockInstList = newModule.getBodyBlock()->getOperations();
1335  newBlockInstList.splice(Block::iterator(cursor), oldBlockInstList,
1336  oldBlockInstList.begin(), oldBlockInstList.end());
1337 
1338  // We are done with our cursor op.
1339  cursor.erase();
1340 
1341  // Lower all of the other operations.
1342  return lowerModuleOperations(newModule, loweringState);
1343 }

1630 // This is the main entrypoint for the lowering pass.
1631 LogicalResult FIRRTLLowering::run() {
1632  // FIRRTL FModule is a single block because FIRRTL ops are a DAG. Walk
1633  // through each operation, lowering each in turn if we can, introducing
1634  // casts if we cannot.
1635  auto &body = theModule.getBody();
1636  randomizePrologEmitted = false;
1637  areFIRRTLBeforeAndAfterInitialEmitted = false;
1638 
1639  SmallVector<Operation *, 16> opsToRemove;
1640 
1641  // Iterate through each operation in the module body, attempting to lower
1642  // each of them. We maintain 'builder' for each invocation.
1643  for (auto &op : body.front().getOperations()) {
1644  builder.setInsertionPoint(&op);
1645  builder.setLoc(op.getLoc());
1646  auto done = succeeded(dispatchVisitor(&op));
1647  circuitState.processRemainingAnnotations(&op, AnnotationSet(&op));
1648  if (done)
1649  opsToRemove.push_back(&op);
1650  else {
1651  switch (handleUnloweredOp(&op)) {
1652  case AlreadyLowered:
1653  break; // Something like hw.output, which is already lowered.
1654  case NowLowered: // Something handleUnloweredOp removed.
1655  opsToRemove.push_back(&op);
1656  break;
1657  case LoweringFailure:
1658  backedgeBuilder.abandon();
1659  return failure();
1660  }
1661  }
1662  }
1663 
1664  // Now that all of the operations that can be lowered are, remove the
1665  // original values. We know that any lowered operations will be dead (if
1666  // removed in reverse order) at this point - any users of them from
1667  // unremapped operations will be changed to use the newly lowered ops.
1668  while (!opsToRemove.empty()) {
1669  assert(opsToRemove.back()->use_empty() &&
1670  "Should remove ops in reverse order of visitation");
1671  opsToRemove.pop_back_val()->erase();
1672  }
1673 
1674  // Now that the IR is in a stable form, try to eliminate temporary wires
1675  // inserted by MemOp insertions.
1676  for (auto wire : tmpWiresToOptimize)
1677  optimizeTemporaryWire(wire);
1678 
1679  return backedgeBuilder.clearOrEmitError();
1680 }

1682 // Try to optimize out temporary wires introduced during lowering.
1683 void FIRRTLLowering::optimizeTemporaryWire(sv::WireOp wire) {
1684  // Wires have inout type, so they'll have connects and read_inout operations
1685  // that work on them. If anything unexpected is found then leave it alone.
1686  SmallVector<sv::ReadInOutOp> reads;
1687  sv::AssignOp write;
1688 
1689  for (auto *user : wire->getUsers()) {
1690  if (auto read = dyn_cast<sv::ReadInOutOp>(user)) {
1691  reads.push_back(read);
1692  continue;
1693  }
1694 
1695  // Otherwise must be a connect, and we must not have seen a write yet.
1696  auto assign = dyn_cast<sv::AssignOp>(user);

3766 /// Template for lowering verification statements from type A to
3767 /// type B.
3768 ///
3769 /// For example, lowering the "foo" op to the "bar" op would start
3770 /// with:
3771 ///
3772 /// foo(clock, condition, enable, "message")
3773 ///
3774 /// This becomes a Verilog clocking block with the "bar" op guarded
3775 /// by an if enable:
3776 ///
3777 /// always @(posedge clock) begin
3778 /// if (enable) begin
3779 /// bar(condition);
3780 /// end
3781 /// end
3782 /// The above can also be reduced into a concurrent verification statement
3783 /// sv.assert.concurrent posedge %clock (condition && enable)
3784 LogicalResult FIRRTLLowering::lowerVerificationStatement(
3785  Operation *op, StringRef labelPrefix, Value opClock, Value opPredicate,
3786  Value opEnable, StringAttr opMessageAttr, ValueRange opOperands,
3787  StringAttr opNameAttr, bool isConcurrent, EventControl opEventControl) {
3788  StringRef opName = op->getName().stripDialect();
3789  auto isAssert = opName == "assert";
3790  auto isCover = opName == "cover";
3791 
3792  auto clock = getLoweredValue(opClock);
3793  auto enable = getLoweredValue(opEnable);
3794  auto predicate = getLoweredValue(opPredicate);
3795  if (!clock || !enable || !predicate)
3796  return failure();
3797 
3798  StringAttr label;
3799  if (opNameAttr && !opNameAttr.getValue().empty())
3800  label = opNameAttr;
3801  StringAttr prefixedLabel;
3802  if (label)
3803  prefixedLabel =
3804  StringAttr::get(builder.getContext(), labelPrefix + label.getValue());
3805 
3806  StringAttr message;
3807  SmallVector<Value> messageOps;
3808  if (!isCover && opMessageAttr && !opMessageAttr.getValue().empty()) {
3809  message = opMessageAttr;
3810  for (auto operand : opOperands) {
3811  auto loweredValue = getLoweredValue(operand);
3812  // Wrap any message ops in $sampled() to guarantee that these will print
3813  // with the same value as when the assertion triggers. (See SystemVerilog
3814  // 2017 spec section 16.9.3 for more information.) The custom
3815  // "ifElseFatal" variant is special cased because this isn't actually a
3816  // concurrent assertion.
3817  auto format = op->getAttrOfType<StringAttr>("format");
3818  if (isConcurrent && (!format || format.getValue() != "ifElseFatal"))
3819  loweredValue = builder.create<sv::SampledOp>(loweredValue);
3820  messageOps.push_back(loweredValue);
3821  }
3822  }
3823 
3824  auto emit = [&]() {
3825  // Handle the purely procedural flavor of the operation.
3826  if (!isConcurrent) {
3827  auto deferImmediate = circt::sv::DeferAssertAttr::get(
3828  builder.getContext(), circt::sv::DeferAssert::Immediate);
3829  addToAlwaysBlock(clock, [&]() {
3830  addIfProceduralBlock(enable, [&]() {
3831  buildImmediateVerifOp(builder, opName, predicate, deferImmediate,
3832  prefixedLabel, message, messageOps);
3833  });
3834  });
3835  return;
3836  }
3837 
3838  auto boolType = IntegerType::get(builder.getContext(), 1);
3839 
3840  // Handle the `ifElseFatal` format, which does not emit an SVA but
3841  // rather a process that uses $error and $fatal to perform the checks.
3842  // TODO: This should *not* be part of the op, but rather a lowering
3843  // option that the user of this pass can choose.
3844  auto format = op->template getAttrOfType<StringAttr>("format");
3845  if (format && format.getValue() == "ifElseFatal") {
3846  predicate = comb::createOrFoldNot(predicate, builder);
3847  predicate = builder.createOrFold<comb::AndOp>(enable, predicate);
3848  addToIfDefBlock("SYNTHESIS", {}, [&]() {
3849  addToAlwaysBlock(clock, [&]() {
3850  addIfProceduralBlock(predicate, [&]() {
3851  circuitState.used_ASSERT_VERBOSE_COND = true;
3852  circuitState.used_STOP_COND = true;
3853  addIfProceduralBlock(
3854  builder.create<sv::MacroRefExprOp>(boolType,
3855  "ASSERT_VERBOSE_COND_"),
3856  [&]() { builder.create<sv::ErrorOp>(message, messageOps); });
3857  addIfProceduralBlock(
3858  builder.create<sv::MacroRefExprOp>(boolType, "STOP_COND_"),
3859  [&]() { builder.create<sv::FatalOp>(); });
3860  });
3861  });
3862  });
3863  return;
3864  }

3867  // Except for covers, combine them: enable & predicate
3868  if (!isCover) {
3869  auto notEnable = comb::createOrFoldNot(enable, builder);
3870  predicate = builder.createOrFold<comb::OrOp>(notEnable, predicate);
3871  } else {
3872  predicate = builder.createOrFold<comb::AndOp>(enable, predicate);
3873  }

3878  case EventControl::AtPosEdge:
3879  event = circt::sv::EventControl::AtPosEdge;
3880  break;
3881  case EventControl::AtEdge:
3882  event = circt::sv::EventControl::AtEdge;
3883  break;
3884  case EventControl::AtNegEdge:
3885  event = circt::sv::EventControl::AtNegEdge;
3886  break;
3887  }
3888 
3889  buildConcurrentVerifOp(
3890  builder, opName,
3891  circt::sv::EventControlAttr::get(builder.getContext(), event), clock,
3892  predicate, prefixedLabel, message, messageOps);
3893 
3894  // Assertions gain a companion `assume` behind a
3895  // `USE_PROPERTY_AS_CONSTRAINT` guard.
3896  if (isAssert) {
3897  StringAttr assumeLabel;
3898  if (label)
3899  assumeLabel = StringAttr::get(builder.getContext(),
3900  "assume__" + label.getValue());

3929  };
