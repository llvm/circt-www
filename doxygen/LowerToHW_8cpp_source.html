83 /// This verifies that the target operation has been lowered to a legal
84 /// operation. This checks that the operation recursively has no FIRRTL
85 /// operations or types.
86 static LogicalResult verifyOpLegality(Operation *op) {
87  auto checkTypes = [](Operation *op) -> WalkResult {
88  // Check that this operation is not a FIRRTL op.
89  if (isa_and_nonnull<FIRRTLDialect>(op->getDialect()))
90  return op->emitError("Found unhandled FIRRTL operation '")
91  << op->getName() << "'";
92 
93  // Helper to check a TypeRange for any FIRRTL types.
94  auto checkTypeRange = [&](TypeRange types) -> LogicalResult {
95  if (llvm::any_of(types, [](Type type) {
96  return isa<FIRRTLDialect>(type.getDialect());
97  }))
98  return op->emitOpError("found unhandled FIRRTL type");
99  return success();
100  };
101 
102  // Check operand and result types.
103  if (failed(checkTypeRange(op->getOperandTypes())) ||
104  failed(checkTypeRange(op->getResultTypes())))
105  return WalkResult::interrupt();
106 
107  // Check the block argument types.
108  for (auto &region : op->getRegions())
109  for (auto &block : region)
110  if (failed(checkTypeRange(block.getArgumentTypes())))
111  return WalkResult::interrupt();
112 
113  // Continue to the next operation.
114  return WalkResult::advance();
115  };
116 
117  if (checkTypes(op).wasInterrupted() || op->walk(checkTypes).wasInterrupted())
118  return failure();
119  return success();
120 }

129 /// unrealized conversion casts as necessary.
130 static Value castToFIRRTLType(Value val, Type type,

493 void CircuitLoweringState::processRemainingAnnotations(
494  Operation *op, const AnnotationSet &annoSet) {
495  if (!enableAnnotationWarning || annoSet.empty())
496  return;
497  std::lock_guard<std::mutex> lock(annotationPrintingMtx);
498 
499  for (auto a : annoSet) {
500  auto inserted = pendingAnnotations.insert(a.getClass());
501  if (!inserted.second)
502  continue;
503 
504  // The following annotations are okay to be silently dropped at this point.
505  // This can occur for example if an annotation marks something in the IR as
506  // not to be processed by a pass, but that pass hasn't run anyway.
507  if (a.isClass(
508  // If the class is `circt.nonlocal`, it's not really an annotation,
509  // but part of a path specifier for another annotation which is
510  // non-local. We can ignore these path specifiers since there will
511  // be a warning produced for the real annotation.
512  "circt.nonlocal",
513  // The following are either consumed by a pass running before
514  // LowerToHW, or they have no effect if the pass doesn't run at all.
515  // If the accompanying pass runs on the HW dialect, then LowerToHW
516  // should have consumed and processed these into an attribute on the
517  // output.
518  dontObfuscateModuleAnnoClass, noDedupAnnoClass,
519  // The following are inspected (but not consumed) by FIRRTL/GCT
520  // passes that have all run by now. Since no one is responsible for
521  // consuming these, they will linger around and can be ignored.
522  scalaClassAnnoClass, dutAnnoClass, metadataDirectoryAttrName,
523  elaborationArtefactsDirectoryAnnoClass, testBenchDirAnnoClass,
524  subCircuitsTargetDirectoryAnnoClass,
525  // This annotation is used to mark which external modules are
526  // imported blackboxes from the BlackBoxReader pass.
527  blackBoxAnnoClass,
528  // This annotation is used by several GrandCentral passes.
529  extractGrandCentralClass,
530  // The following will be handled while lowering the verification
531  // ops.
532  extractAssertAnnoClass, extractAssumeAnnoClass,
533  extractCoverageAnnoClass,
534  // The following will be handled after lowering FModule ops, since
535  // they are still needed on the circuit until after lowering
536  // FModules.
537  moduleHierAnnoClass, testHarnessHierAnnoClass,
538  blackBoxTargetDirAnnoClass))
539  continue;
540 
541  mlir::emitWarning(op->getLoc(), "unprocessed annotation:'" + a.getClass() +
542  "' still remaining after LowerToHW");
543  }

596  // We run on the top level modules in the IR blob. Start by finding the
597  // firrtl.circuit within it. If there is none, then there is nothing to do.
598  auto *topLevelModule = getOperation().getBody();
599 
600  // Find the single firrtl.circuit in the module.
601  CircuitOp circuit;
602  for (auto &op : *topLevelModule) {
603  if ((circuit = dyn_cast<CircuitOp>(&op)))
604  break;
605  }
606 
607  if (!circuit)
608  return;
609 
610  auto *circuitBody = circuit.getBodyBlock();
611 
612  // Keep track of the mapping from old to new modules. The result may be null
613  // if lowering failed.
614  CircuitLoweringState state(
615  circuit, enableAnnotationWarning, emitChiselAssertsAsSVA,
616  getAnalysis<InstanceGraph>(), &getAnalysis<NLATable>());
617 
618  SmallVector<hw::HWModuleOp, 32> modulesToProcess;
619 
620  AnnotationSet circuitAnno(circuit);
621  moveVerifAnno(getOperation(), circuitAnno, extractAssertAnnoClass,
622  "firrtl.extract.assert");
623  moveVerifAnno(getOperation(), circuitAnno, extractAssumeAnnoClass,
624  "firrtl.extract.assume");
625  moveVerifAnno(getOperation(), circuitAnno, extractCoverageAnnoClass,
626  "firrtl.extract.cover");
627  circuitAnno.removeAnnotationsWithClass(
628  extractAssertAnnoClass, extractAssumeAnnoClass, extractCoverageAnnoClass);
629 
630  state.processRemainingAnnotations(circuit, circuitAnno);
631  // Iterate through each operation in the circuit body, transforming any
632  // FModule's we come across. If any module fails to lower, return early.
633  for (auto &op : make_early_inc_range(circuitBody->getOperations())) {
634  auto result =
635  TypeSwitch<Operation *, LogicalResult>(&op)
636  .Case<FModuleOp>([&](auto module) {
637  auto loweredMod = lowerModule(module, topLevelModule, state);
638  if (!loweredMod)
639  return failure();
640 
641  state.recordModuleMapping(&op, loweredMod);
642  modulesToProcess.push_back(loweredMod);
643  // Lower all the alias types.
644  module.walk([&](Operation *op) {
645  for (auto res : op->getResults()) {
646  if (auto aliasType =
647  type_dyn_cast<BaseTypeAliasType>(res.getType()))
648  state.lowerType(aliasType, op->getLoc());
649  }
650  });
651  return lowerModulePortsAndMoveBody(module, loweredMod, state);
652  })
653  .Case<FExtModuleOp>([&](auto extModule) {
654  auto loweredMod =
655  lowerExtModule(extModule, topLevelModule, state);
656  if (!loweredMod)
657  return failure();
658  state.recordModuleMapping(&op, loweredMod);
659  return success();
660  })
661  .Case<FMemModuleOp>([&](auto memModule) {
662  auto loweredMod =
663  lowerMemModule(memModule, topLevelModule, state);
664  if (!loweredMod)
665  return failure();
666  state.recordModuleMapping(&op, loweredMod);
667  return success();
668  })
669  .Default([&](Operation *op) {
670  // We don't know what this op is. If it has no illegal FIRRTL
671  // types, we can forward the operation. Otherwise, we emit an
672  // error and drop the operation from the circuit.
673  if (succeeded(verifyOpLegality(op)))
674  op->moveBefore(topLevelModule, topLevelModule->end());
675  else
676  return failure();
677  return success();
678  });
679  if (failed(result))
680  return signalPassFailure();
681  }
682  // Ensure no more TypeDecl can be added to the global TypeScope.
683  state.typeAliases.freeze();
684  // Handle the creation of the module hierarchy metadata.
685 
686  // Collect the two sets of hierarchy files from the circuit. Some of them will
687  // be rooted at the test harness, the others will be rooted at the DUT.
688  SmallVector<Attribute> dutHierarchyFiles;
689  SmallVector<Attribute> testHarnessHierarchyFiles;
690  circuitAnno.removeAnnotations([&](Annotation annotation) {
691  if (annotation.isClass(moduleHierAnnoClass)) {
692  auto file = hw::OutputFileAttr::getFromFilename(
693  &getContext(),
694  annotation.getMember<StringAttr>("filename").getValue(),
695  /*excludeFromFileList=*/true);
696  dutHierarchyFiles.push_back(file);
697  return true;
698  }
699  if (annotation.isClass(testHarnessHierAnnoClass)) {
700  auto file = hw::OutputFileAttr::getFromFilename(
701  &getContext(),
702  annotation.getMember<StringAttr>("filename").getValue(),
703  /*excludeFromFileList=*/true);
704  // If there is no testHarness, we print the hiearchy for this file
705  // starting at the DUT.
706  if (state.getTestHarness())
707  testHarnessHierarchyFiles.push_back(file);
708  else
709  dutHierarchyFiles.push_back(file);
710  return true;
711  }
712  return false;
713  });
714  // Attach the lowered form of these annotations.
715  if (!dutHierarchyFiles.empty())
716  state.getNewModule(state.getDut())
717  ->setAttr(moduleHierarchyFileAttrName,
718  ArrayAttr::get(&getContext(), dutHierarchyFiles));
719  if (!testHarnessHierarchyFiles.empty())
720  state.getNewModule(state.getTestHarness())
721  ->setAttr(moduleHierarchyFileAttrName,
722  ArrayAttr::get(&getContext(), testHarnessHierarchyFiles));
723 
724  // Finally, lower all operations.
725  auto result = mlir::failableParallelForEachN(
726  &getContext(), 0, modulesToProcess.size(), [&](auto index) {
727  return lowerModuleOperations(modulesToProcess[index], state);
728  });
729 
730  // If any module bodies failed to lower, return early.
731  if (failed(result))
732  return signalPassFailure();
733 
734  // Move binds from inside modules to outside modules.
735  for (auto bind : state.binds) {
736  bind->moveBefore(bind->getParentOfType<hw::HWModuleOp>());
737  }
738 
739  // Fix up fragment attributes.
740  for (auto &[module, fragments] : state.fragments)
741  module->setAttr(emit::getFragmentsAttrName(),
742  ArrayAttr::get(&getContext(), fragments.getArrayRef()));
743 
744  // Finally delete all the old modules.
745  for (auto oldNew : state.oldToNewModuleMap)
746  oldNew.first->erase();
747 
748  if (!state.macroDeclNames.empty()) {
749  ImplicitLocOpBuilder b(UnknownLoc::get(&getContext()), circuit);
750  for (auto name : state.macroDeclNames) {
751  b.create<sv::MacroDeclOp>(name);
752  }
753  }
754 
755  // Emit all the macros and preprocessor gunk at the start of the file.
756  lowerFileHeader(circuit, state);
757 
758  // Now that the modules are moved over, remove the Circuit.
759  circuit.erase();
760 }

762 /// Emit the file header that defines a bunch of macros.
763 void FIRRTLModuleLowering::lowerFileHeader(CircuitOp op,
764  CircuitLoweringState &state) {
765  // Intentionally pass an UnknownLoc here so we don't get line number
766  // comments on the output of this boilerplate in generated Verilog.
767  ImplicitLocOpBuilder b(UnknownLoc::get(&getContext()), op);
768 
769  // Helper function to emit a "#ifdef guard" with a `define in the then and
770  // optionally in the else branch.
771  auto emitGuardedDefine = [&](StringRef guard, StringRef defName,
772  StringRef defineTrue = "",
773  StringRef defineFalse = StringRef()) {
774  if (!defineFalse.data()) {
775  assert(defineTrue.data() && "didn't define anything");
776  b.create<sv::IfDefOp>(
777  guard, [&]() { b.create<sv::MacroDefOp>(defName, defineTrue); });
778  } else {
779  b.create<sv::IfDefOp>(
780  guard,
781  [&]() {
782  if (defineTrue.data())
783  b.create<sv::MacroDefOp>(defName, defineTrue);
784  },
785  [&]() { b.create<sv::MacroDefOp>(defName, defineFalse); });
786  }
787  };
788 
789  // Helper function to emit #ifndef guard.
790  auto emitGuard = [&](const char *guard, llvm::function_ref<void(void)> body) {
791  b.create<sv::IfDefOp>(
792  guard, []() {}, body);
793  };
794 
795  if (state.usedPrintfCond) {
796  b.create<sv::MacroDeclOp>("PRINTF_COND");
797  b.create<sv::MacroDeclOp>("PRINTF_COND_");
798  b.create<emit::FragmentOp>("PRINTF_COND_FRAGMENT", [&] {
799  b.create<sv::VerbatimOp>(
800  "\n// Users can define 'PRINTF_COND' to add an extra gate to "
801  "prints.");
802  emitGuard("PRINTF_COND_", [&]() {
803  emitGuardedDefine("PRINTF_COND", "PRINTF_COND_", "(`PRINTF_COND)", "1");
804  });
805  });
806  }
807 
808  if (state.usedAssertVerboseCond) {
809  b.create<sv::MacroDeclOp>("ASSERT_VERBOSE_COND");
810  b.create<sv::MacroDeclOp>("ASSERT_VERBOSE_COND_");
811  b.create<emit::FragmentOp>("ASSERT_VERBOSE_COND_FRAGMENT", [&] {
812  b.create<sv::VerbatimOp>(
813  "\n// Users can define 'ASSERT_VERBOSE_COND' to add an extra "
814  "gate to assert error printing.");
815  emitGuard("ASSERT_VERBOSE_COND_", [&]() {
816  emitGuardedDefine("ASSERT_VERBOSE_COND", "ASSERT_VERBOSE_COND_",
817  "(`ASSERT_VERBOSE_COND)", "1");
818  });
819  });
820  }
821 
822  if (state.usedStopCond) {
823  b.create<sv::MacroDeclOp>("STOP_COND");
824  b.create<sv::MacroDeclOp>("STOP_COND_");
825  b.create<emit::FragmentOp>("STOP_COND_FRAGMENT", [&] {
826  b.create<sv::VerbatimOp>(
827  "\n// Users can define 'STOP_COND' to add an extra gate "
828  "to stop conditions.");
829  emitGuard("STOP_COND_", [&]() {
830  emitGuardedDefine("STOP_COND", "STOP_COND_", "(`STOP_COND)", "1");
831  });
832  });
833  }

836 LogicalResult
837 FIRRTLModuleLowering::lowerPorts(ArrayRef<PortInfo> firrtlPorts,
838  SmallVectorImpl<hw::PortInfo> &ports,
839  Operation *moduleOp, StringRef moduleName,
840  CircuitLoweringState &loweringState) {
841  ports.reserve(firrtlPorts.size());
842  size_t numArgs = 0;
843  size_t numResults = 0;
844  for (auto e : llvm::enumerate(firrtlPorts)) {
845  PortInfo firrtlPort = e.value();
846  size_t portNo = e.index();
847  hw::PortInfo hwPort;
848  hwPort.name = firrtlPort.name;
849  hwPort.type = loweringState.lowerType(firrtlPort.type, firrtlPort.loc);
850  if (firrtlPort.sym)
851  if (firrtlPort.sym.size() > 1 ||
852  (firrtlPort.sym.size() == 1 && !firrtlPort.sym.getSymName()))
853  return emitError(firrtlPort.loc)
854  << "cannot lower aggregate port " << firrtlPort.name
855  << " with field sensitive symbols, HW dialect does not support "
856  "per field symbols yet.";
857  hwPort.setSym(firrtlPort.sym, moduleOp->getContext());
858  bool hadDontTouch = firrtlPort.annotations.removeDontTouch();
859  if (hadDontTouch && !hwPort.getSym()) {
860  if (hwPort.type.isInteger(0)) {
861  if (enableAnnotationWarning) {
862  mlir::emitWarning(firrtlPort.loc)
863  << "zero width port " << hwPort.name
864  << " has dontTouch annotation, removing anyway";
865  }
866  continue;
867  }
868 
869  hwPort.setSym(
870  hw::InnerSymAttr::get(StringAttr::get(
871  moduleOp->getContext(),
872  Twine("__") + moduleName + Twine("__DONTTOUCH__") +
873  Twine(portNo) + Twine("__") + firrtlPort.name.strref())),
874  moduleOp->getContext());
875  }
876 
877  // We can't lower all types, so make sure to cleanly reject them.
878  if (!hwPort.type) {
879  moduleOp->emitError("cannot lower this port type to HW");
880  return failure();
881  }
882 
883  // If this is a zero bit port, just drop it. It doesn't matter if it is
884  // input, output, or inout. We don't want these at the HW level.
885  if (hwPort.type.isInteger(0)) {
886  auto sym = hwPort.getSym();
887  if (sym && !sym.empty()) {
888  return mlir::emitError(firrtlPort.loc)
889  << "zero width port " << hwPort.name
890  << " is referenced by name [" << sym
891  << "] (e.g. in an XMR) but must be removed";
892  }
893  continue;
894  }
895 
896  // Figure out the direction of the port.
897  if (firrtlPort.isOutput()) {
898  hwPort.dir = hw::ModulePort::Direction::Output;
899  hwPort.argNum = numResults++;
900  } else if (firrtlPort.isInput()) {
901  hwPort.dir = hw::ModulePort::Direction::Input;
902  hwPort.argNum = numArgs++;
903  } else {
904  // If the port is an inout bundle or contains an analog type, then it is
905  // implicitly inout.
906  hwPort.type = hw::InOutType::get(hwPort.type);
907  hwPort.dir = hw::ModulePort::Direction::InOut;
908  hwPort.argNum = numArgs++;
909  }
910  hwPort.loc = firrtlPort.loc;
911  ports.push_back(hwPort);
912  loweringState.processRemainingAnnotations(moduleOp, firrtlPort.annotations);
913  }
914  return success();
915 }

917 /// Map the parameter specifier on the specified extmodule into the HWModule
918 /// representation for parameters. If `ignoreValues` is true, all the values
919 /// are dropped.
920 static ArrayAttr getHWParameters(FExtModuleOp module, bool ignoreValues) {
921  auto params = llvm::map_range(module.getParameters(), [](Attribute a) {
922  return cast<ParamDeclAttr>(a);
923  });
924  if (params.empty())
925  return {};
926 
927  Builder builder(module);
928 
929  // Map the attributes over from firrtl attributes to HW attributes
930  // directly. MLIR's DictionaryAttr always stores keys in the dictionary
931  // in sorted order which is nicely stable.
932  SmallVector<Attribute> newParams;
933  for (const ParamDeclAttr &entry : params) {
934  auto name = entry.getName();
935  auto type = entry.getType();
936  auto value = ignoreValues ? Attribute() : entry.getValue();
937  auto paramAttr =
938  hw::ParamDeclAttr::get(builder.getContext(), name, type, value);
939  newParams.push_back(paramAttr);
940  }
941  return builder.getArrayAttr(newParams);
942 }

944 bool FIRRTLModuleLowering::handleForceNameAnnos(
945  FModuleLike oldModule, AnnotationSet &annos,
946  CircuitLoweringState &loweringState) {
947  bool failed = false;
948  // Remove ForceNameAnnotations by generating verilogNames on instances.
949  annos.removeAnnotations([&](Annotation anno) {
950  if (!anno.isClass(forceNameAnnoClass))
951  return false;
952 
953  auto sym = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal");
954  // This must be a non-local annotation due to how the Chisel API is
955  // implemented.
956  //
957  // TODO: handle this in some sensible way based on what the SFC does with
958  // a local annotation.
959  if (!sym) {
960  auto diag = oldModule.emitOpError()
961  << "contains a '" << forceNameAnnoClass
962  << "' that is not a non-local annotation";
963  diag.attachNote() << "the erroneous annotation is '" << anno.getDict()
964  << "'\n";
965  failed = true;
966  return false;
967  }
968 
969  auto nla = loweringState.nlaTable->getNLA(sym.getAttr());
970  // The non-local anchor must exist.
971  //
972  // TODO: handle this with annotation verification.
973  if (!nla) {
974  auto diag = oldModule.emitOpError()
975  << "contains a '" << forceNameAnnoClass
976  << "' whose non-local symbol, '" << sym
977  << "' does not exist in the circuit";
978  diag.attachNote() << "the erroneous annotation is '" << anno.getDict();
979  failed = true;
980  return false;
981  }
982 
983  // Add the forced name to global state (keyed by a pseudo-inner name ref).
984  // Error out if this key is alredy in use.
985  //
986  // TODO: this error behavior can be relaxed to always overwrite with the
987  // new forced name (the bug-compatible behavior of the Chisel
988  // implementation) or fixed to duplicate modules such that the naming can
989  // be applied.
990  auto inst =
991  cast<hw::InnerRefAttr>(nla.getNamepath().getValue().take_back(2)[0]);
992  auto inserted = loweringState.instanceForceNames.insert(
993  {{inst.getModule(), inst.getName()}, anno.getMember("name")});
994  if (!inserted.second &&
995  (anno.getMember("name") != (inserted.first->second))) {
996  auto diag = oldModule.emitError()
997  << "contained multiple '" << forceNameAnnoClass
998  << "' with different names: " << inserted.first->second
999  << " was not " << anno.getMember("name");
1000  diag.attachNote() << "the erroneous annotation is '" << anno.getDict()
1001  << "'";
1002  failed = true;
1003  return false;
1004  }
1005  return true;
1006  });
1007  return failed;
1008 }

1141 /// Given a value of analog type, check to see the only use of it is an
1142 /// attach. If so, remove the attach and return the value being attached to
1143 /// it, converted to an HW inout type. If this isn't a situation we can
1144 /// handle, just return null.
1145 static Value tryEliminatingAttachesToAnalogValue(Value value,
1146  Operation *insertPoint) {
1147  if (!value.hasOneUse())
1148  return {};
1149 
1150  auto attach = dyn_cast<AttachOp>(*value.user_begin());
1151  if (!attach || attach.getNumOperands() != 2)
1152  return {};
1153 
1154  // Don't optimize zero bit analogs.
1155  auto loweredType = lowerType(value.getType());
1156  if (loweredType.isInteger(0))
1157  return {};
1158 
1159  // Check to see if the attached value dominates the insertion point. If
1160  // not, just fail.
1161  auto attachedValue = attach.getOperand(attach.getOperand(0) == value);
1162  auto *op = attachedValue.getDefiningOp();
1163  if (op && op->getBlock() == insertPoint->getBlock() &&
1164  !op->isBeforeInBlock(insertPoint))
1165  return {};
1166 
1167  attach.erase();
1168 
1169  ImplicitLocOpBuilder builder(insertPoint->getLoc(), insertPoint);
1170  return castFromFIRRTLType(attachedValue, hw::InOutType::get(loweredType),
1171  builder);
1172 }

1174 /// Given a value of flip type, check to see if all of the uses of it are
1175 /// connects. If so, remove the connects and return the value being connected
1176 /// to it, converted to an HW type. If this isn't a situation we can handle,
1177 /// just return null.
1178 ///
1179 /// This can happen when there are no connects to the value. The 'mergePoint'
1180 /// location is where a 'hw.merge' operation should be inserted if needed.
1181 static Value
1182 tryEliminatingConnectsToValue(Value flipValue, Operation *insertPoint,
1183  CircuitLoweringState &loweringState) {
1184  // Handle analog's separately.
1185  if (type_isa<AnalogType>(flipValue.getType()))
1186  return tryEliminatingAttachesToAnalogValue(flipValue, insertPoint);
1187 
1188  Operation *connectOp = nullptr;
1189  for (auto &use : flipValue.getUses()) {
1190  // We only know how to deal with connects where this value is the
1191  // destination.
1192  if (use.getOperandNumber() != 0)
1193  return {};
1194  if (!isa<ConnectOp, StrictConnectOp>(use.getOwner()))
1195  return {};
1196 
1197  // We only support things with a single connect.
1198  if (connectOp)
1199  return {};
1200  connectOp = use.getOwner();
1201  }
1202 
1203  // We don't have an HW equivalent of "poison" so just don't special case
1204  // the case where there are no connects other uses of an output.
1205  if (!connectOp)
1206  return {}; // TODO: Emit an sv.constant here since it is unconnected.
1207 
1208  // Don't special case zero-bit results.
1209  auto loweredType =
1210  loweringState.lowerType(flipValue.getType(), flipValue.getLoc());
1211  if (loweredType.isInteger(0))
1212  return {};
1213 
1214  // Convert each connect into an extended version of its operand being
1215  // output.
1216  ImplicitLocOpBuilder builder(insertPoint->getLoc(), insertPoint);
1217 
1218  auto connectSrc = connectOp->getOperand(1);
1219 
1220  // Directly forward foreign types.
1221  if (!connectSrc.getType().isa<FIRRTLType>()) {
1222  connectOp->erase();
1223  return connectSrc;
1224  }
1225 
1226  // Convert fliped sources to passive sources.
1227  if (!type_cast<FIRRTLBaseType>(connectSrc.getType()).isPassive())
1228  connectSrc = builder
1229  .create<mlir::UnrealizedConversionCastOp>(
1230  type_cast<FIRRTLBaseType>(connectSrc.getType())
1231  .getPassiveType(),
1232  connectSrc)
1233  .getResult(0);
1234 
1235  // We know it must be the destination operand due to the types, but the
1236  // source may not match the destination width.
1237  auto destTy = type_cast<FIRRTLBaseType>(flipValue.getType()).getPassiveType();
1238 
1239  if (destTy != connectSrc.getType() &&
1240  (isa<BaseTypeAliasType>(connectSrc.getType()) ||
1241  isa<BaseTypeAliasType>(destTy))) {
1242  connectSrc =
1243  builder.createOrFold<BitCastOp>(flipValue.getType(), connectSrc);
1244  }
1245  if (!destTy.isGround()) {
1246  // If types are not ground type and they don't match, we give up.
1247  if (destTy != type_cast<FIRRTLType>(connectSrc.getType()))
1248  return {};
1249  } else if (destTy.getBitWidthOrSentinel() !=
1250  type_cast<FIRRTLBaseType>(connectSrc.getType())
1251  .getBitWidthOrSentinel()) {
1252  // The only type mismatchs we care about is due to integer width
1253  // differences.
1254  auto destWidth = destTy.getBitWidthOrSentinel();
1255  assert(destWidth != -1 && "must know integer widths");
1256  connectSrc = builder.createOrFold<PadPrimOp>(destTy, connectSrc, destWidth);
1257  }
1258 
1259  // Remove the connect and use its source as the value for the output.
1260  connectOp->erase();
1261 
1262  // Convert from FIRRTL type to builtin type.
1263  return castFromFIRRTLType(connectSrc, loweredType, builder);
1264 }

1267  StringRef field) {
1268  SmallVector<SubfieldOp> accesses;
1269  for (auto *op : structValue.getUsers()) {
1270  assert(isa<SubfieldOp>(op));
1271  auto fieldAccess = cast<SubfieldOp>(op);

1273  fieldAccess.getInput().getType().base().getElementIndex(field);

1280 /// Now that we have the operations for the hw.module's corresponding to the
1281 /// firrtl.module's, we can go through and move the bodies over, updating the
1282 /// ports and output op.
1283 LogicalResult FIRRTLModuleLowering::lowerModulePortsAndMoveBody(
1284  FModuleOp oldModule, hw::HWModuleOp newModule,
1285  CircuitLoweringState &loweringState) {
1286  ImplicitLocOpBuilder bodyBuilder(oldModule.getLoc(), newModule.getBody());
1287 
1288  // Use a placeholder instruction be a cursor that indicates where we want to
1289  // move the new function body to. This is important because we insert some
1290  // ops at the start of the function and some at the end, and the body is
1291  // currently empty to avoid iterator invalidation.
1292  auto cursor = bodyBuilder.create<hw::ConstantOp>(APInt(1, 1));
1293  bodyBuilder.setInsertionPoint(cursor);
1294 
1295  // Insert argument casts, and re-vector users in the old body to use them.
1296  SmallVector<PortInfo> firrtlPorts = oldModule.getPorts();
1297  assert(oldModule.getBody().getNumArguments() == firrtlPorts.size() &&
1298  "port count mismatch");
1299 
1300  SmallVector<Value, 4> outputs;
1301 
1302  // This is the terminator in the new module.
1303  auto outputOp = newModule.getBodyBlock()->getTerminator();
1304  ImplicitLocOpBuilder outputBuilder(oldModule.getLoc(), outputOp);
1305 
1306  unsigned nextHWInputArg = 0;
1307  int hwPortIndex = -1;
1308  for (auto [firrtlPortIndex, port] : llvm::enumerate(firrtlPorts)) {
1309  // Inputs and outputs are both modeled as arguments in the FIRRTL level.
1310  auto oldArg = oldModule.getBody().getArgument(firrtlPortIndex);
1311 
1312  bool isZeroWidth =
1313  type_isa<FIRRTLBaseType>(port.type) &&
1314  type_cast<FIRRTLBaseType>(port.type).getBitWidthOrSentinel() == 0;
1315  if (!isZeroWidth)
1316  ++hwPortIndex;
1317 
1318  if (!port.isOutput() && !isZeroWidth) {
1319  // Inputs and InOuts are modeled as arguments in the result, so we can
1320  // just map them over. We model zero bit outputs as inouts.
1321  Value newArg = newModule.getBody().getArgument(nextHWInputArg++);
1322 
1323  // Cast the argument to the old type, reintroducing sign information in
1324  // the hw.module body.
1325  newArg = castToFIRRTLType(newArg, oldArg.getType(), bodyBuilder);
1326  // Switch all uses of the old operands to the new ones.
1327  oldArg.replaceAllUsesWith(newArg);
1328  continue;
1329  }
1330 
1331  // We lower zero width inout and outputs to a wire that isn't connected to
1332  // anything outside the module. Inputs are lowered to zero.
1333  if (isZeroWidth && port.isInput()) {
1334  Value newArg = bodyBuilder
1335  .create<WireOp>(port.type, "." + port.getName().str() +
1336  ".0width_input")
1337  .getResult();
1338  oldArg.replaceAllUsesWith(newArg);
1339  continue;
1340  }
1341 
1342  if (auto value =
1343  tryEliminatingConnectsToValue(oldArg, outputOp, loweringState)) {
1344  // If we were able to find the value being connected to the output,
1345  // directly use it!
1346  outputs.push_back(value);
1347  assert(oldArg.use_empty() && "should have removed all uses of oldArg");
1348  continue;
1349  }
1350 
1351  // Outputs need a temporary wire so they can be connect'd to, which we
1352  // then return.
1353  auto newArg = bodyBuilder.create<WireOp>(
1354  port.type, "." + port.getName().str() + ".output");
1355 
1356  // Switch all uses of the old operands to the new ones.
1357  oldArg.replaceAllUsesWith(newArg.getResult());
1358 
1359  // Don't output zero bit results or inouts.
1360  auto resultHWType = loweringState.lowerType(port.type, port.loc);
1361  if (!resultHWType.isInteger(0)) {
1362  auto output =
1363  castFromFIRRTLType(newArg.getResult(), resultHWType, outputBuilder);
1364  outputs.push_back(output);
1365 
1366  // If output port has symbol, move it to this wire.
1367  if (auto sym = newModule.getPort(hwPortIndex).getSym()) {
1368  newArg.setInnerSymAttr(sym);
1369  newModule.setPortSymbolAttr(hwPortIndex, {});
1370  }
1371  }
1372  }
1373 
1374  // Update the hw.output terminator with the list of outputs we have.
1375  outputOp->setOperands(outputs);
1376 
1377  // Finally splice the body over, don't move the old terminator over though.
1378  auto &oldBlockInstList = oldModule.getBodyBlock()->getOperations();
1379  auto &newBlockInstList = newModule.getBodyBlock()->getOperations();
1380  newBlockInstList.splice(Block::iterator(cursor), oldBlockInstList,
1381  oldBlockInstList.begin(), oldBlockInstList.end());
1382 
1383  // We are done with our cursor op.
1384  cursor.erase();
1385 
1386  return success();
1387 }

1767 // This is the main entrypoint for the lowering pass.
1768 LogicalResult FIRRTLLowering::run() {
1769  // FIRRTL FModule is a single block because FIRRTL ops are a DAG. Walk
1770  // through each operation, lowering each in turn if we can, introducing
1771  // casts if we cannot.
1772  auto &body = theModule.getBody();
1773 
1774  SmallVector<Operation *, 16> opsToRemove;
1775 
1776  // Iterate through each operation in the module body, attempting to lower
1777  // each of them. We maintain 'builder' for each invocation.
1778  for (auto &op : body.front().getOperations()) {
1779  builder.setInsertionPoint(&op);
1780  builder.setLoc(op.getLoc());
1781  auto done = succeeded(dispatchVisitor(&op));
1782  circuitState.processRemainingAnnotations(&op, AnnotationSet(&op));
1783  if (done)
1784  opsToRemove.push_back(&op);
1785  else {
1786  switch (handleUnloweredOp(&op)) {
1787  case AlreadyLowered:
1788  break; // Something like hw.output, which is already lowered.
1789  case NowLowered: // Something handleUnloweredOp removed.
1790  opsToRemove.push_back(&op);
1791  break;
1792  case LoweringFailure:
1793  backedgeBuilder.abandon();
1794  return failure();
1795  }
1796  }
1797  }
1798 
1799  // Replace all backedges with uses of their regular values. We process them
1800  // after the module body since the lowering table is too hard to keep up to
1801  // date. Multiple operations may be lowered to the same backedge when values
1802  // are folded, which means we would have to scan the entire lowering table to
1803  // safely replace a backedge.
1804  for (auto &[backedge, value] : backedges) {
1805  SmallVector<Location> driverLocs;
1806  // In the case where we have backedges connected to other backedges, we have
1807  // to find the value that actually drives the group.
1808  while (true) {
1809  // If we find the original backedge we have some undriven logic or
1810  // a combinatorial loop. Bail out and provide information on the nodes.
1811  if (backedge == value) {
1812  Location edgeLoc = backedge.getLoc();
1813  if (driverLocs.empty()) {
1814  mlir::emitError(edgeLoc, "sink does not have a driver");
1815  } else {
1816  auto diag = mlir::emitError(edgeLoc, "sink in combinational loop");
1817  for (auto loc : driverLocs)
1818  diag.attachNote(loc) << "through driver here";
1819  }
1820  backedgeBuilder.abandon();
1821  return failure();
1822  }
1823  // If the value is not another backedge, we have found the driver.
1824  auto it = backedges.find(value);
1825  if (it == backedges.end())
1826  break;
1827  // Find what is driving the next backedge.
1828  driverLocs.push_back(value.getLoc());
1829  value = it->second;
1830  }
1831  if (auto *defOp = backedge.getDefiningOp())
1832  maybeUnusedValues.erase(defOp);
1833  backedge.replaceAllUsesWith(value);
1834  }
1835 
1836  // Now that all of the operations that can be lowered are, remove th
1837  // original values. We know that any lowered operations will be dead (if
1838  // removed in reverse order) at this point - any users of them from
1839  // unremapped operations will be changed to use the newly lowered ops.
1840  hw::ConstantOp zeroI0;
1841  while (!opsToRemove.empty()) {
1842  auto *op = opsToRemove.pop_back_val();
1843 
1844  // We remove zero-width values when lowering FIRRTL ops. We can't remove
1845  // such a value if it escapes to a foreign op. In that case, create an
1846  // `hw.constant 0 : i0` to pass along.
1847  for (auto result : op->getResults()) {
1848  if (!isZeroBitFIRRTLType(result.getType()))
1849  continue;
1850  if (!zeroI0) {
1851  auto builder = OpBuilder::atBlockBegin(&body.front());
1852  zeroI0 = builder.create<hw::ConstantOp>(op->getLoc(),
1853  builder.getIntegerType(0), 0);
1854  maybeUnusedValues.insert(zeroI0);
1855  }
1856  result.replaceAllUsesWith(zeroI0);
1857  }
1858 
1859  if (!op->use_empty()) {
1860  auto d = op->emitOpError(
1861  "still has uses; should remove ops in reverse order of visitation");
1862  SmallPtrSet<Operation *, 2> visited;
1863  for (auto *user : op->getUsers())
1864  if (visited.insert(user).second)
1865  d.attachNote(user->getLoc())
1866  << "used by " << user->getName() << " op";
1867  return d;
1868  }
1869  maybeUnusedValues.erase(op);
1870  op->erase();
1871  }
1872 
1873  // Prune operations that may have become unused throughout the lowering.
1874  while (!maybeUnusedValues.empty()) {
1875  auto it = maybeUnusedValues.begin();
1876  auto *op = *it;
1877  maybeUnusedValues.erase(it);
1878  if (!isOpTriviallyDead(op))
1879  continue;
1880  for (auto operand : op->getOperands())
1881  if (auto *defOp = operand.getDefiningOp())
1882  maybeUnusedValues.insert(defOp);
1883  op->erase();
1884  }
1885 
1886  // Determine the actual types of lowered LTL operations and remove any
1887  // intermediate wires among them.
1888  if (failed(fixupLTLOps()))
1889  return failure();
1890 
1891  return backedgeBuilder.clearOrEmitError();
1892 }

2595 /// lowering.

2600  // lowered. This allows us to handled partially lowered inputs, and also allow
2601  // other FIRRTL operations to spawn additional already-lowered operations,

2617  // primitive operations that produce a zero bit result to be ignored by the

4569  // Don't emit anything for a zero or one operand attach.
4570  if (op.getAttached().size() < 2)
4571  return success();
4572 
4573  SmallVector<Value, 4> inoutValues;
4574  for (auto v : op.getAttached()) {
4575  inoutValues.push_back(getPossiblyInoutLoweredValue(v));
4576  if (!inoutValues.back()) {
4577  // Ignore zero bit values.
4578  if (!isZeroBitFIRRTLType(v.getType()))
4579  return failure();
4580  inoutValues.pop_back();
4581  continue;
4582  }
4583 
4584  if (!inoutValues.back().getType().isa<hw::InOutType>())
4585  return op.emitError("operand isn't an inout type");
4586  }
4587 
4588  if (inoutValues.size() < 2)
4589  return success();
4590 
4591  // If the op has a single source value, the value is used as a lowering result
4592  // of other values. Therefore we can delete the attach op here.
4593  if (getSingleNonInstanceOperand(op))
4594  return success();
4595 
4596  // If all operands of the attach are internal to this module (none of them
4597  // are ports), then they can all be replaced with a single wire, and we can
4598  // delete the attach op.
4599  bool isAttachInternalOnly =
4600  llvm::none_of(inoutValues, [](auto v) { return isa<BlockArgument>(v); });
4601 
4602  if (isAttachInternalOnly) {
4603  auto v0 = inoutValues.front();
4604  for (auto v : inoutValues) {
4605  if (v == v0)
4606  continue;
4607  v.replaceAllUsesWith(v0);
4608  }
4609  return success();
4610  }
4611 
4612  // If the attach operands contain a port, then we can't do anything to
4613  // simplify the attach operation.
4614  circuitState.addMacroDecl(builder.getStringAttr("SYNTHESIS"));
4615  circuitState.addMacroDecl(builder.getStringAttr("VERILATOR"));
4616  addToIfDefBlock(
4617  "SYNTHESIS",
4618  // If we're doing synthesis, we emit an all-pairs assign complex.
4619  [&]() {
4620  SmallVector<Value, 4> values;
4621  for (size_t i = 0, e = inoutValues.size(); i != e; ++i)
4622  values.push_back(getReadValue(inoutValues[i]));
4623 
4624  for (size_t i1 = 0, e = inoutValues.size(); i1 != e; ++i1) {
4625  for (size_t i2 = 0; i2 != e; ++i2)
4626  if (i1 != i2)
4627  builder.create<sv::AssignOp>(inoutValues[i1], values[i2]);
4628  }
4629  },
4630  // In the non-synthesis case, we emit a SystemVerilog alias
4631  // statement.

4634  "VERILATOR",
4635  [&]() {
4636  builder.create<sv::VerbatimOp>(
4637  "`error \"Verilator does not support alias and thus "
4638  "cannot "
4639  "arbitrarily connect bidirectional wires and ports\"");
4640  },

4642  });
4643

4647 LogicalResult FIRRTLLowering::fixupLTLOps() {
4648  if (ltlOpFixupWorklist.empty())
4649  return success();
4650  LLVM_DEBUG(llvm::dbgs() << "Fixing up " << ltlOpFixupWorklist.size()
4651  << " LTL ops\n");
4652 
4653  // Add wire users into the worklist.
4654  for (unsigned i = 0, e = ltlOpFixupWorklist.size(); i != e; ++i)
4655  for (auto *user : ltlOpFixupWorklist[i]->getUsers())
4656  if (isa<hw::WireOp>(user))
4657  ltlOpFixupWorklist.insert(user);
4658 
4659  // Re-infer LTL op types and remove wires.
4660  while (!ltlOpFixupWorklist.empty()) {
4661  auto *op = ltlOpFixupWorklist.pop_back_val();
4662 
4663  // Update the operation's return type by re-running type inference.
4664  if (auto opIntf = dyn_cast_or_null<mlir::InferTypeOpInterface>(op)) {
4665  LLVM_DEBUG(llvm::dbgs() << "- Update " << *op << "\n");
4666  SmallVector<Type, 2> types;
4667  auto result = opIntf.inferReturnTypes(
4668  op->getContext(), op->getLoc(), op->getOperands(),
4669  op->getAttrDictionary(), op->getPropertiesStorage(), op->getRegions(),
4670  types);
4671  if (failed(result))
4672  return failure();
4673  assert(types.size() == op->getNumResults());
4674 
4675  // Update the result types and add the dependent ops into the worklist if
4676  // the type changed.
4677  for (auto [result, type] : llvm::zip(op->getResults(), types)) {
4678  if (result.getType() == type)
4679  continue;
4680  LLVM_DEBUG(llvm::dbgs()
4681  << " - Result #" << result.getResultNumber() << " from "
4682  << result.getType() << " to " << type << "\n");
4683  result.setType(type);
4684  for (auto *user : result.getUsers())
4685  if (user != op)
4686  ltlOpFixupWorklist.insert(user);
4687  }
4688  }
4689 
4690  // Remove LTL-typed wires.
4691  if (auto wireOp = dyn_cast<hw::WireOp>(op)) {
4692  if (isa<ltl::SequenceType, ltl::PropertyType>(wireOp.getType())) {
4693  wireOp.replaceAllUsesWith(wireOp.getInput());
4694  LLVM_DEBUG(llvm::dbgs() << "- Remove " << wireOp << "\n");
4695  if (wireOp.use_empty())
4696  wireOp.erase();
4697  }
4698  continue;
4699  }
4700 
4701  // Ensure that the operation has no users outside of LTL operations.
4702  SmallPtrSet<Operation *, 4> usersReported;
4703  for (auto *user : op->getUsers()) {
4704  if (!usersReported.insert(user).second)
4705  continue;
4706  if (isa<ltl::LTLDialect, verif::VerifDialect>(user->getDialect()))
4707  continue;
4708  if (isa<hw::WireOp>(user))
4709  continue;
4710  auto d = op->emitError(
4711  "verification operation used in a non-verification context");
4712  d.attachNote(user->getLoc())
4713  << "leaking outside verification context here";
4714  return d;
4715  }
4716  }
4717 
4718  return success();
4719 }
