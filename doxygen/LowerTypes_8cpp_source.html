9 // This file defines the LowerTypes pass. This pass replaces aggregate types
10 // with expanded values.
11 //
12 // This pass walks the operations in reverse order. This lets it visit users
13 // before defs. Users can usually be expanded out to multiple operations (think
14 // mux of a bundle to muxes of each field) with a temporary subWhatever op
15 // inserted. When processing an aggregate producer, we blow out the op as
16 // appropriate, then walk the users, often those are subWhatever ops which can
17 // be bypassed and deleted. Function arguments are logically last on the
18 // operation visit order and walked left to right, being peeled one layer at a
19 // time with replacements inserted to the right of the original argument.
20 //
21 // Each processing of an op peels one layer of aggregate type off. Because new
22 // ops are inserted immediately above the current up, the walk will visit them
23 // next, effectively recusing on the aggregate types, without recusing. These
24 // potentially temporary ops(if the aggregate is complex) effectively serve as
25 // the worklist. Often aggregates are shallow, so the new ops are the final
26 // ones.
27 //

81 /// Return true if the type has more than zero bitwidth.
82 static bool hasZeroBitWidth(FIRRTLType type) {
83  return TypeSwitch<FIRRTLType, bool>(type)
84  .Case<BundleType>([&](auto bundle) {
85  for (size_t i = 0, e = bundle.getNumElements(); i < e; ++i) {
86  auto elt = bundle.getElement(i);
87  if (hasZeroBitWidth(elt.type))
88  return true;
89  }
90  return bundle.getNumElements() == 0;
91  })
92  .Case<FVectorType>([&](auto vector) {
93  if (vector.getNumElements() == 0)
94  return true;
95  return hasZeroBitWidth(vector.getElementType());
96  })
97  .Default([](auto groundType) {
98  return firrtl::getBitWidth(groundType).getValueOr(0) == 0;
99  });
100 }

111 /// Peel one layer of an aggregate type into its components. Type may be
112 /// complex, but empty, in which case fields is empty, but the return is true.
113 static bool peelType(Type type, SmallVectorImpl<FlatBundleFieldEntry> &fields,
114  bool allowedToPreserveAggregate = false) {
115  // If the aggregate preservation is enabled and the type is preservable,
116  // then just return.
117  if (allowedToPreserveAggregate && isPreservableAggregateType(type))
118  return false;
119 
120  return TypeSwitch<Type, bool>(type)
121  .Case<BundleType>([&](auto bundle) {
122  SmallString<16> tmpSuffix;
123  // Otherwise, we have a bundle type. Break it down.
124  for (size_t i = 0, e = bundle.getNumElements(); i < e; ++i) {
125  auto elt = bundle.getElement(i);
126  // Construct the suffix to pass down.
127  tmpSuffix.resize(0);
128  tmpSuffix.push_back('_');
129  tmpSuffix.append(elt.name.getValue());
130  fields.emplace_back(elt.type, i, bundle.getFieldID(i), tmpSuffix,
131  elt.isFlip);
132  }
133  return true;
134  })
135  .Case<FVectorType>([&](auto vector) {
136  // Increment the field ID to point to the first element.
137  for (size_t i = 0, e = vector.getNumElements(); i != e; ++i) {
138  fields.emplace_back(vector.getElementType(), i, vector.getFieldID(i),
139  "_" + std::to_string(i), false);
140  }
141  return true;
142  })
143  .Default([](auto op) { return false; });
144 }

146 /// Return if something is not a normal subaccess. Non-normal includes
147 /// zero-length vectors and constant indexes (which are really subindexes).
148 static bool isNotSubAccess(Operation *op) {
149  SubaccessOp sao = dyn_cast<SubaccessOp>(op);
150  if (!sao)
151  return true;

158 /// Look through and collect subfields leading to a subaccess.
159 static SmallVector<Operation *> getSAWritePath(Operation *op) {
160  SmallVector<Operation *> retval;
161  auto defOp = op->getOperand(0).getDefiningOp();
162  while (isa_and_nonnull<SubfieldOp, SubindexOp, SubaccessOp>(defOp)) {
163  retval.push_back(defOp);
164  defOp = defOp->getOperand(0).getDefiningOp();
165  }
166  // Trim to the subaccess
167  while (!retval.empty() && isNotSubAccess(retval.back()))
168  retval.pop_back();
169  return retval;
170 }

178 /// If an annotation on one operation is replicated across multiple IR
179 /// operations as a result of type lowering, the replicated annotations may want
180 /// to track which field ID they were applied to. This function adds a fieldID
181 /// to such a replicated operation, if the annotation in question requires it.
182 static Attribute updateAnnotationFieldID(MLIRContext *ctxt, Attribute attr,
183  unsigned fieldID, Type i64ty) {
184  DictionaryAttr dict = attr.cast<DictionaryAttr>();
185 
186  // No need to do anything if the annotation applies to the entire field.
187  if (fieldID == 0)
188  return attr;
189 
190  // Only certain annotations require precise tracking of field IDs.
191  Annotation anno(dict);
192  if (!isAnnotationSensitiveToFieldID(anno))
193  return attr;
194 
195  // Add the new ID to the existing field ID in the annotation.
196  if (auto existingFieldID = anno.getMember<IntegerAttr>("fieldID"))
197  fieldID += existingFieldID.getValue().getZExtValue();
198  NamedAttrList fields(dict);
199  fields.set("fieldID", IntegerAttr::get(i64ty, fieldID));
200  return DictionaryAttr::get(ctxt, fields);
201 }

203 static MemOp cloneMemWithNewType(ImplicitLocOpBuilder *b, MemOp op,
204  FlatBundleFieldEntry field) {
205  SmallVector<Type, 8> ports;
206  SmallVector<Attribute, 8> portNames;
207 
208  auto oldPorts = op.getPorts();
209  for (size_t portIdx = 0, e = oldPorts.size(); portIdx < e; ++portIdx) {
210  auto port = oldPorts[portIdx];
211  ports.push_back(MemOp::getTypeForPort(op.depth(), field.type, port.second));
212  portNames.push_back(port.first);
213  }
214 
215  // It's easier to duplicate the old annotations, then fix and filter them.
216  auto newMem =
217  b->create<MemOp>(ports, op.readLatency(), op.writeLatency(), op.depth(),
218  op.ruw(), portNames, (op.name() + field.suffix).str(),
219  op.nameKind(), op.annotations().getValue(),
220  op.portAnnotations().getValue(), op.inner_symAttr());
221  if (auto oldName = getInnerSymName(op))
222  newMem.inner_symAttr(InnerSymAttr::get(StringAttr::get(
223  b->getContext(), oldName.getValue() + (op.name() + field.suffix))));
224 
225  SmallVector<Attribute> newAnnotations;
226  for (size_t portIdx = 0, e = newMem.getNumResults(); portIdx < e; ++portIdx) {
227  auto portType = newMem.getResult(portIdx).getType().cast<BundleType>();
228  auto oldPortType = op.getResult(portIdx).getType().cast<BundleType>();
229  SmallVector<Attribute> portAnno;
230  for (auto attr : newMem.getPortAnnotation(portIdx)) {
231  Annotation anno(attr);
232  if (auto annoFieldID = anno.getFieldID()) {
233  auto targetIndex = oldPortType.getIndexForFieldID(annoFieldID);
234 
235  // Apply annotations to all elements if the target is the whole
236  // sub-field.
237  if (annoFieldID == oldPortType.getFieldID(targetIndex)) {
238  anno.setMember(
239  "circt.fieldID",
240  b->getI32IntegerAttr(portType.getFieldID(targetIndex)));
241  portAnno.push_back(anno.getDict());
242  continue;
243  }
244 
245  // Handle aggregate sub-fields, including `(r/w)data` and `(w)mask`.
246  if (oldPortType.getElement(targetIndex).type.isa<BundleType>()) {
247  // Check whether the annotation falls into the range of the current
248  // field. Note that the `field` here is peeled from the `data`
249  // sub-field of the memory port, thus we need to add the fieldID of
250  // `data` or `mask` sub-field to get the "real" fieldID.
251  auto fieldID = field.fieldID + oldPortType.getFieldID(targetIndex);
252  if (annoFieldID >= fieldID &&
253  annoFieldID <= fieldID + field.type.getMaxFieldID()) {
254  // Set the field ID of the new annotation.
255  auto newFieldID =
256  annoFieldID - fieldID + portType.getFieldID(targetIndex);
257  anno.setMember("circt.fieldID", b->getI32IntegerAttr(newFieldID));
258  portAnno.push_back(anno.getDict());
259  }
260  }
261  } else
262  portAnno.push_back(attr);
263  }
264  newAnnotations.push_back(b->getArrayAttr(portAnno));
265  }
266  newMem.setAllPortAnnotations(newAnnotations);
267  return newMem;
268 }

446 /// Conditionally expand a subaccessop write path
447 bool TypeLoweringVisitor::processSAPath(Operation *op) {
448  // Does this LHS have a subaccessop?
449  SmallVector<Operation *> writePath = getSAWritePath(op);
450  if (writePath.empty())
451  return false;
452 
453  lowerSAWritePath(op, writePath);
454  // Unhook the writePath from the connect. This isn't the right type, but we
455  // are deleting the op anyway.
456  op->eraseOperands(0, 2);
457  // See how far up the tree we can delete things.
458  for (size_t i = 0; i < writePath.size(); ++i) {

460  writePath[i]->erase();
461  } else {
462  break;
463  }
464  }
465  return true;

468 void TypeLoweringVisitor::lowerBlock(Block *block) {
469  // Lower the operations bottom up.
470  for (auto it = block->rbegin(), e = block->rend(); it != e;) {
471  auto &iop = *it;
472  builder->setInsertionPoint(&iop);
473  builder->setLoc(iop.getLoc());
474  bool removeOp = dispatchVisitor(&iop);

478  if (removeOp)

483 ArrayAttr TypeLoweringVisitor::filterAnnotations(
484  MLIRContext *ctxt, ArrayAttr annotations, FIRRTLType srcType,
485  FlatBundleFieldEntry field, bool &needsSym, StringRef sym) {
486  SmallVector<Attribute> retval;
487  if (!annotations || annotations.empty())
488  return ArrayAttr::get(ctxt, retval);
489  bool isGroundType = field.type.isGround();
490  for (auto opAttr : annotations) {
491  Optional<int64_t> maybeFieldID = None;
492  DictionaryAttr annotation;
493  annotation = opAttr.dyn_cast<DictionaryAttr>();
494  if (annotations)
495  // Erase the circt.fieldID. If this is needed later, it will be re-added.
496  if (auto id = annotation.getAs<IntegerAttr>("circt.fieldID")) {
497  maybeFieldID = id.getInt();
498  Annotation anno(annotation);
499  anno.removeMember("circt.fieldID");
500  annotation = anno.getDict();
501  }
502  if (!maybeFieldID) {
503  retval.push_back(
504  updateAnnotationFieldID(ctxt, opAttr, field.fieldID, cache.i64ty));
505  continue;
506  }
507  auto fieldID = maybeFieldID.getValue();
508  // Check whether the annotation falls into the range of the current field.
509  if (fieldID != 0 &&
510  !(fieldID >= field.fieldID &&
511  fieldID <= field.fieldID + field.type.getMaxFieldID()))
512  continue;
513 
514  // Apply annotations to all elements if fieldID is equal to zero.
515  if (fieldID == 0) {
516  retval.push_back(annotation);
517  continue;
518  }
519 
520  if (auto newFieldID = fieldID - field.fieldID) {
521  // If the target is a subfield/subindex of the current field, create a
522  // new annotation with the correct circt.fieldID.
523  Annotation newAnno(annotation);
524  newAnno.setMember("circt.fieldID",
525  builder->getI32IntegerAttr(newFieldID));
526  retval.push_back(newAnno.getDict());
527  continue;
528  }
529  if (Annotation(opAttr).getClass() ==
530  "firrtl.transforms.DontTouchAnnotation") {
531  // This is intended to cover the case of a non-local DontTouchAnnotation
532  // (which is represented as an annotation) being converted to a symbol on
533  // a ground type. This code will, however, also lower any local
534  // DontTouchAnnotation (even though this should not exist at this point).
535  needsSym = true;
536  continue;
537  }
538  // We are keeping the annotation. If the anotation is non-local and this is
539  // a ground type (this won't be further lowered) then generate a symbol.
540  needsSym =
541  isGroundType && annotation.getAs<FlatSymbolRefAttr>("circt.nonlocal");
542  retval.push_back(annotation);
543  }
544  return ArrayAttr::get(ctxt, retval);
545 }

547 bool TypeLoweringVisitor::lowerProducer(
548  Operation *op,
549  llvm::function_ref<Operation *(const FlatBundleFieldEntry &, ArrayAttr)>
550  clone) {
551  // If this is not a bundle, there is nothing to do.
552  auto srcType = op->getResult(0).getType().cast<FIRRTLType>();
553  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
554 
555  if (!peelType(srcType, fieldTypes, preserveAggregate))
556  return false;
557 
558  SmallVector<Value> lowered;
559  // Loop over the leaf aggregates.
560  SmallString<16> loweredName;
561  SmallString<16> loweredSymName;
562  auto nameKindAttr = op->getAttrOfType<NameKindEnumAttr>(cache.nameKindAttr);
563 
564  auto innerSymAttr = getInnerSymName(op);
565  if (innerSymAttr)
566  loweredSymName = innerSymAttr.getValue();
567  if (auto nameAttr = op->getAttrOfType<StringAttr>(cache.nameAttr))
568  loweredName = nameAttr.getValue();
569  if (loweredSymName.empty())
570  loweredSymName = loweredName;
571  if (loweredSymName.empty())
572  loweredSymName = uniqueName();
573  auto baseNameLen = loweredName.size();
574  auto baseSymNameLen = loweredSymName.size();
575  auto oldAnno = op->getAttr("annotations").dyn_cast_or_null<ArrayAttr>();
576 
577  for (auto field : fieldTypes) {
578  if (!loweredName.empty()) {
579  loweredName.resize(baseNameLen);
580  loweredName += field.suffix;
581  }
582  if (!loweredSymName.empty()) {
583  loweredSymName.resize(baseSymNameLen);
584  loweredSymName += field.suffix;
585  }
586  bool needsSym = false;
587 
588  // For all annotations on the parent op, filter them based on the target
589  // attribute.
590  ArrayAttr loweredAttrs = filterAnnotations(context, oldAnno, srcType, field,
591  needsSym, loweredSymName);
592  auto *newOp = clone(field, loweredAttrs);
593 
594  // Carry over the name, if present.
595  if (!loweredName.empty())
596  newOp->setAttr(cache.nameAttr, StringAttr::get(context, loweredName));
597  if (nameKindAttr)
598  newOp->setAttr(cache.nameKindAttr, nameKindAttr);
599  // Carry over the inner_sym name, if present.
600  if (needsSym || op->hasAttr(cache.innerSymAttr)) {
601  auto newName = StringAttr::get(context, loweredSymName);
602  newOp->setAttr(cache.innerSymAttr, InnerSymAttr::get(newName));
603  assert(!loweredSymName.empty());
604 
605  // If this operation has an inner symbol, then update the origSymbols
606  // disjoint set to make sure that all derived symbols are associated with
607  // the original symbol.
608  if (innerSymAttr) {
609  origSymbols.unionSets(innerSymAttr.getValue(), newName.getValue());
610  if (field.type.isGround()) {
611  auto module = op->getParentOfType<FModuleOp>();
612  auto key = origSymbols.findLeader(innerSymAttr.getValue());
613  StringAttr keyAttr = StringAttr::get(module.getContext(), *key);
614  innerRefRenames[hw::InnerRefAttr::get(module.getNameAttr(), keyAttr)]
615  .push_back(OpAnnoTarget(newOp));
616  }
617  }
618  }
619  lowered.push_back(newOp->getResult(0));
620  }
621 
622  processUsers(op->getResult(0), lowered);
623  return true;
624 }

653 // module. This also maintains the name attribute for the new argument,
654 // possibly with a new suffix appended.
655 std::pair<Value, PortInfo>
656 TypeLoweringVisitor::addArg(Operation *module, unsigned insertPt,
657  unsigned insertPtOffset, FIRRTLType srcType,
658  FlatBundleFieldEntry field, PortInfo &oldArg) {
659  Value newValue;
660  if (auto mod = dyn_cast<FModuleOp>(module)) {
661  Block *body = mod.getBody();
662  // Append the new argument.
663  newValue = body->insertArgument(insertPt, field.type, oldArg.loc);
664  }
665 
666  // Save the name attribute for the new argument.
667  auto name = builder->getStringAttr(oldArg.name.getValue() + field.suffix);
668 
669  SmallString<16> symtmp;
670  StringRef sym;
671  bool oldArgHadSym = oldArg.sym && !oldArg.sym.getValue().empty();
672  if (oldArgHadSym) {
673  symtmp = (oldArg.sym.getValue() + field.suffix).str();
674  sym = symtmp;
675  } else
676  sym = name.getValue();
677 
678  bool needsSym = false;
679  // Populate the new arg attributes.
680  auto newAnnotations =
681  filterAnnotations(context, oldArg.annotations.getArrayAttr(), srcType,
682  field, needsSym, sym);
683  // Flip the direction if the field is an output.
684  auto direction = (Direction)((unsigned)oldArg.direction ^ field.isOutput);
685

699  keyAttr)]

701  }

828 /// Lower memory operations. A new memory is created for every leaf
829 /// element in a memory's data type.
830 bool TypeLoweringVisitor::visitDecl(MemOp op) {
831  // Attempt to get the bundle types.
832  SmallVector<FlatBundleFieldEntry> fields;
833 
834  // MemOp should have ground types so we can't preserve aggregates.
835  if (!peelType(op.getDataType(), fields, false))
836  return false;
837 
838  SmallVector<MemOp> newMemories;
839  SmallVector<WireOp> oldPorts;
840 
841  // Wires for old ports
842  for (unsigned int index = 0, end = op.getNumResults(); index < end; ++index) {
843  auto result = op.getResult(index);
844  auto wire = builder->create<WireOp>(
845  result.getType(),
846  (op.name() + "_" + op.getPortName(index).getValue()).str());
847  oldPorts.push_back(wire);
848  result.replaceAllUsesWith(wire.getResult());
849  }
850  // If annotations targeting fields of an aggregate are present, we cannot
851  // flatten the memory. It must be split into one memory per aggregate field.
852  // Do not overwrite the pass flag!
853 
854  // Memory for each field
855  for (const auto &field : fields)
856  newMemories.push_back(cloneMemWithNewType(builder, op, field));
857  // Hook up the new memories to the wires the old memory was replaced with.
858  for (size_t index = 0, rend = op.getNumResults(); index < rend; ++index) {
859  auto result = oldPorts[index];
860  auto rType = result.getType().cast<BundleType>();
861  for (size_t fieldIndex = 0, fend = rType.getNumElements();
862  fieldIndex != fend; ++fieldIndex) {
863  auto name = rType.getElement(fieldIndex).name.getValue();
864  auto oldField = builder->create<SubfieldOp>(result, fieldIndex);
865  // data and mask depend on the memory type which was split. They can also
866  // go both directions, depending on the port direction.
867  if (name == "data" || name == "mask" || name == "wdata" ||
868  name == "wmask" || name == "rdata") {
869  for (const auto &field : fields) {
870  auto realOldField = getSubWhatever(oldField, field.index);
871  auto newField = getSubWhatever(
872  newMemories[field.index].getResult(index), fieldIndex);
873  if (rType.getElement(fieldIndex).isFlip)
874  std::swap(realOldField, newField);
875  emitConnect(*builder, newField, realOldField);
876  }
877  } else {
878  for (auto mem : newMemories) {
879  auto newField =
880  builder->create<SubfieldOp>(mem.getResult(index), fieldIndex);
881  emitConnect(*builder, newField, oldField);
882  }
883  }
884  }
885  }
886  return true;
887 }

1107 // Expand BitCastOp of aggregates
1108 bool TypeLoweringVisitor::visitExpr(BitCastOp op) {
1109  Value srcLoweredVal = op.input();
1110  // If the input is of aggregate type, then cat all the leaf fields to form a
1111  // UInt type result. That is, first bitcast the aggregate type to a UInt.
1112  // Attempt to get the bundle types.
1113  SmallVector<FlatBundleFieldEntry> fields;
1114  if (peelType(op.input().getType(), fields,
1115  /* allowedToPreserveAggregate */ false)) {
1116  size_t uptoBits = 0;
1117  // Loop over the leaf aggregates and concat each of them to get a UInt.
1118  // Bitcast the fields to handle nested aggregate types.
1119  for (const auto &field : llvm::enumerate(fields)) {
1120  auto fieldBitwidth = getBitWidth(field.value().type).getValue();
1121  // Ignore zero width fields, like empty bundles.
1122  if (fieldBitwidth == 0)
1123  continue;
1124  Value src = getSubWhatever(op.input(), field.index());
1125  // The src could be an aggregate type, bitcast it to a UInt type.
1126  src = builder->createOrFold<BitCastOp>(
1127  UIntType::get(context, fieldBitwidth), src);
1128  // Take the first field, or else Cat the previous fields with this field.
1129  if (uptoBits == 0)
1130  srcLoweredVal = src;
1131  else
1132  srcLoweredVal = builder->create<CatPrimOp>(src, srcLoweredVal);
1133  // Record the total bits already accumulated.
1134  uptoBits += fieldBitwidth;
1135  }
1136  } else {
1137  srcLoweredVal = builder->createOrFold<AsUIntPrimOp>(srcLoweredVal);
1138  }
1139  // Now the input has been cast to srcLoweredVal, which is of UInt type.
1140  // If the result is an aggregate type, then use lowerProducer.
1141  if (op.getResult().getType().isa<BundleType, FVectorType>()) {
1142  // uptoBits is used to keep track of the bits that have been extracted.
1143  size_t uptoBits = 0;
1144  auto clone = [&](const FlatBundleFieldEntry &field,
1145  ArrayAttr attrs) -> Operation * {
1146  // All the fields must have valid bitwidth, a requirement for BitCastOp.
1147  auto fieldBits = getBitWidth(field.type).getValue();
1148  // If empty field, then it doesnot have any use, so replace it with an
1149  // invalid op, which should be trivially removed.
1150  if (fieldBits == 0)
1151  return builder->create<InvalidValueOp>(field.type);
1152 
1153  // Assign the field to the corresponding bits from the input.
1154  // Bitcast the field, incase its an aggregate type.
1155  auto extractBits = builder->create<BitsPrimOp>(
1156  srcLoweredVal, uptoBits + fieldBits - 1, uptoBits);
1157  uptoBits += fieldBits;
1158  return builder->create<BitCastOp>(field.type, extractBits);
1159  };
1160  return lowerProducer(op, clone);
1161  }
1162 
1163  // If ground type, then replace the result.
1164  if (op.getType().dyn_cast<SIntType>())
1165  srcLoweredVal = builder->create<AsSIntPrimOp>(srcLoweredVal);
1166  op.getResult().replaceAllUsesWith(srcLoweredVal);
1167  return true;
1168 }
