9 // This file defines the LowerTypes pass. This pass replaces aggregate types
10 // with expanded values.
11 //
12 // This pass walks the operations in reverse order. This lets it visit users
13 // before defs. Users can usually be expanded out to multiple operations (think
14 // mux of a bundle to muxes of each field) with a temporary subWhatever op
15 // inserted. When processing an aggregate producer, we blow out the op as
16 // appropriate, then walk the users, often those are subWhatever ops which can
17 // be bypassed and deleted. Function arguments are logically last on the
18 // operation visit order and walked left to right, being peeled one layer at a
19 // time with replacements inserted to the right of the original argument.
20 //
21 // Each processing of an op peels one layer of aggregate type off. Because new
22 // ops are inserted immediately above the current up, the walk will visit them
23 // next, effectively recusing on the aggregate types, without recusing. These
24 // potentially temporary ops(if the aggregate is complex) effectively serve as
25 // the worklist. Often aggregates are shallow, so the new ops are the final
26 // ones.
27 //

72 /// Peel one layer of an aggregate type into its components. Type may be
73 /// complex, but empty, in which case fields is empty, but the return is true.
74 static bool peelType(Type type, SmallVectorImpl<FlatBundleFieldEntry> &fields) {
75  return TypeSwitch<Type, bool>(type)
76  .Case<BundleType>([&](auto bundle) {
77  SmallString<16> tmpSuffix;
78  // Otherwise, we have a bundle type. Break it down.
79  for (size_t i = 0, e = bundle.getNumElements(); i < e; ++i) {
80  auto elt = bundle.getElement(i);
81  // Construct the suffix to pass down.
82  tmpSuffix.resize(0);
83  tmpSuffix.push_back('_');
84  tmpSuffix.append(elt.name.getValue());
85  fields.emplace_back(elt.type, i, bundle.getFieldID(i),
86  bundle.getFieldID(i) + elt.type.getMaxFieldID(),
87  tmpSuffix, elt.isFlip);
88  }
89  return true;
90  })
91  .Case<FVectorType>([&](auto vector) {
92  // Increment the field ID to point to the first element.
93  auto width = vector.getElementType().getMaxFieldID() + 1;
94  for (size_t i = 0, e = vector.getNumElements(); i != e; ++i) {
95  fields.emplace_back(vector.getElementType(), i, vector.getFieldID(i),
96  vector.getFieldID(i) + width,
97  "_" + std::to_string(i), false);
98  }
99  return true;
100  })
101  .Default([](auto op) { return false; });
102 }

104 /// Look through and collect subfields leading to a subaccess.
105 static SmallVector<Operation *> getSAWritePath(Operation *op) {
106  SmallVector<Operation *> retval;
107  auto defOp = op->getOperand(0).getDefiningOp();
108  while (defOp && isa<SubfieldOp, SubindexOp, SubaccessOp>(defOp)) {
109  retval.push_back(defOp);
110  defOp = defOp->getOperand(0).getDefiningOp();
111  }
112  // Trim to the subaccess
113  while (!retval.empty() && !isa<SubaccessOp>(retval.back()))
114  retval.pop_back();
115  return retval;
116 }

118 /// Copy annotations from \p annotations to \p loweredAttrs, except annotations
119 /// with "target" key, that do not match the field suffix.
120 static ArrayAttr filterAnnotations(MLIRContext *ctxt, ArrayAttr annotations,
121  FIRRTLType srcType,
122  FlatBundleFieldEntry field) {
123  SmallVector<Attribute> retval;
124  if (!annotations || annotations.empty())
125  return ArrayAttr::get(ctxt, retval);
126  for (auto opAttr : annotations) {
127  if (auto subAnno = opAttr.dyn_cast<SubAnnotationAttr>()) {
128  // Check for overlap of the two ranges
129  if ((field.minFieldID >= subAnno.getMinFieldID() &&
130  field.minFieldID <= subAnno.getMaxFieldID()) ||
131  (field.maxFieldID >= subAnno.getMinFieldID() &&
132  field.minFieldID <= subAnno.getMinFieldID())) {
133  auto newMin =
134  srcType.rootChildFieldID(subAnno.getMinFieldID(), field.index);
135  auto newMax =
136  srcType.rootChildFieldID(subAnno.getMinFieldID(), field.index);
137  if (newMin.first == 0 && newMax.first == 0)
138  retval.push_back(subAnno.getAnnotations());
139  else
140  retval.push_back(SubAnnotationAttr::get(
141  ctxt, newMin.first, newMax.first, subAnno.getAnnotations()));
142  }
143  } else
144  retval.push_back(opAttr);
145  }
146  return ArrayAttr::get(ctxt, retval);
147 }

149 static MemOp cloneMemWithNewType(ImplicitLocOpBuilder *b, MemOp op,
150  FlatBundleFieldEntry field) {
151  SmallVector<Type, 8> ports;
152  SmallVector<Attribute, 8> portNames;
153 
154  auto oldPorts = op.getPorts();
155  for (size_t portIdx = 0, e = oldPorts.size(); portIdx < e; ++portIdx) {
156  auto port = oldPorts[portIdx];
157  ports.push_back(MemOp::getTypeForPort(op.depth(), field.type, port.second));
158  portNames.push_back(port.first);
159  }
160 
161  // It's easier to duplicate the old annotations, then fix and filter them.
162  auto newMem = b->create<MemOp>(
163  ports, op.readLatency(), op.writeLatency(), op.depth(), op.ruw(),
164  portNames, (op.name() + field.suffix).str(), op.annotations().getValue(),
165  op.portAnnotations().getValue());
166 
167  SmallVector<Attribute> newAnnotations;
168  for (size_t portIdx = 0, e = newMem.getNumResults(); portIdx < e; ++portIdx) {
169  auto portType = newMem.getResult(portIdx).getType().cast<BundleType>();
170  auto oldPortType = op.getResult(portIdx).getType().cast<BundleType>();
171  SmallVector<Attribute> portAnno;
172  for (auto attr : newMem.getPortAnnotation(portIdx)) {
173  if (auto subAnno = attr.dyn_cast<SubAnnotationAttr>()) {
174  auto minIdx = oldPortType.getIndexForFieldID(subAnno.getMinFieldID());
175  auto maxIdx = oldPortType.getIndexForFieldID(subAnno.getMaxFieldID());
176  if (minIdx != maxIdx) {
177  op.emitError("Cannot handle annotation ranges spanning fields");
178  continue;
179  }
180  if (minIdx >= 3) { // data or mask
181  // is the annotation in the filtered field?
182  // FieldID of the annotation in the data field.
183  auto minID =
184  oldPortType.rootChildFieldID(subAnno.getMinFieldID(), minIdx);
185  // FieldID of the annotation in filtered data type.
186  minID = op.getDataType().rootChildFieldID(minID.first, field.index);
187  if (minID.second) { // contained!
188  portAnno.push_back(SubAnnotationAttr::get(
189  b->getContext(), portType.getFieldID(minIdx) + minID.first,
190  portType.getFieldID(maxIdx) + minID.first,
191  subAnno.getAnnotations()));
192  }
193 
194  } else {
195  portAnno.push_back(SubAnnotationAttr::get(
196  b->getContext(), portType.getFieldID(minIdx),
197  portType.getFieldID(maxIdx), subAnno.getAnnotations()));
198  }
199  } else {
200  portAnno.push_back(attr);
201  }
202  }
203  newAnnotations.push_back(b->getArrayAttr(portAnno));
204  }
205  newMem.setAllPortAnnotations(newAnnotations);
206  return newMem;
207 }

281 /// Conditionally expand a subaccessop write path
282 bool TypeLoweringVisitor::processSAPath(Operation *op) {
283  // Does this LHS have a subaccessop?
284  SmallVector<Operation *> writePath = getSAWritePath(op);
285  if (!writePath.empty()) {
286  lowerSAWritePath(op, writePath);
287  // Unhook the writePath from the connect. This isn't the right type, but we
288  // are deleting the op anyway.
289  op->eraseOperands(0, 2);
290  // See how far up the tree we can delete things.
291  for (size_t i = 0; i < writePath.size(); ++i) {
292  if (writePath[i]->use_empty()) {
293  writePath[i]->erase();
294  } else {
295  break;
296  }
297  }
298  opsToRemove.push_back(op);
299  return true;
300  }
301  return false;
302 }

304 void TypeLoweringVisitor::lowerBlock(Block *block) {
305  // Lower the operations.
306  for (auto &iop : llvm::reverse(*block)) {
307  // We erase old ops eagerly so we don't have dangling uses we've already
308  // lowered.
309  for (auto *op : opsToRemove)
310  op->erase();
311  opsToRemove.clear();
312 
313  builder->setInsertionPoint(&iop);
314  builder->setLoc(iop.getLoc());
315  dispatchVisitor(&iop);
316  }
317 
318  for (auto *op : opsToRemove)
319  op->erase();
320  opsToRemove.clear();
321 }

323 void TypeLoweringVisitor::lowerProducer(
324  Operation *op,
325  llvm::function_ref<Operation *(FlatBundleFieldEntry, StringRef, ArrayAttr)>
326  clone) {
327  // If this is not a bundle, there is nothing to do.
328  auto srcType = op->getResult(0).getType().cast<FIRRTLType>();
329  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
330  if (!peelType(srcType, fieldTypes))
331  return;
332 
333  SmallVector<Value> lowered;
334  // Loop over the leaf aggregates.
335  SmallString<16> loweredName;
336  if (auto nameAttr = op->getAttr("name"))
337  if (auto nameStrAttr = nameAttr.dyn_cast<StringAttr>())
338  loweredName = nameStrAttr.getValue();
339  auto baseNameLen = loweredName.size();
340  auto oldAnno = op->getAttr("annotations").dyn_cast_or_null<ArrayAttr>();
341 
342  for (auto field : fieldTypes) {
343  if (!loweredName.empty()) {
344  loweredName.resize(baseNameLen);
345  loweredName += field.suffix;
346  }
347  // For all annotations on the parent op, filter them based on the target
348  // attribute.
349  ArrayAttr loweredAttrs =
350  filterAnnotations(context, oldAnno, srcType, field);
351  auto newOp = clone(field, loweredName, loweredAttrs);
352  lowered.push_back(newOp->getResult(0));
353  }
354 
355  processUsers(op->getResult(0), lowered);
356  opsToRemove.push_back(op);
357 }

359 void TypeLoweringVisitor::processUsers(Value val, ArrayRef<Value> mapping) {
360  for (auto user : llvm::make_early_inc_range(val.getUsers())) {
361  if (SubindexOp sio = dyn_cast<SubindexOp>(user)) {
362  Value repl = mapping[sio.index()];
363  sio.replaceAllUsesWith(repl);
364  sio.erase();
365  } else if (SubfieldOp sfo = dyn_cast<SubfieldOp>(user)) {
366  // Get the input bundle type.
367  Value repl = mapping[sfo.fieldIndex()];
368  sfo.replaceAllUsesWith(repl);
369  sfo.erase();
370  } else {
371  llvm_unreachable("Unknown aggregate user");
372  }
373  }
374 }

383 // Creates and returns a new block argument of the specified type to the
384 // module. This also maintains the name attribute for the new argument,
385 // possibly with a new suffix appended.
386 std::pair<Value, firrtl::ModulePortInfo>
387 TypeLoweringVisitor::addArg(Operation *module, unsigned insertPt,
388  FIRRTLType srcType, FlatBundleFieldEntry field,
389  ModulePortInfo &oldArg) {
390  Value newValue;
391  if (auto mod = dyn_cast<FModuleOp>(module)) {
392  Block *body = mod.getBodyBlock();
393  // Append the new argument.
394  newValue = body->insertArgument(insertPt, field.type);
395  }
396 
397  // Save the name attribute for the new argument.
398  auto name =
399  builder->getStringAttr(oldArg.name.getValue().str() + field.suffix.str());
400 
401  // Populate the new arg attributes.
402  auto newAnnotations = filterAnnotations(
403  context, oldArg.annotations.getArrayAttr(), srcType, field);
404 
405  // Flip the direction if the field is an output.
406  auto direction = (Direction)((unsigned)oldArg.direction ^ field.isOutput);
407 
408  return std::make_pair(
409  newValue, firrtl::ModulePortInfo{name, field.type, direction, oldArg.loc,
410  AnnotationSet(newAnnotations)});
411 }

413 // Lower arguments with bundle type by flattening them.
414 bool TypeLoweringVisitor::lowerArg(
415  Operation *module, size_t argIndex,
416  SmallVectorImpl<std::pair<ModulePortInfo, SmallVector<NamedAttribute>>>
417  &newArgs,
418  SmallVectorImpl<Value> &lowering) {
419 
420  // Flatten any bundle types.
421  SmallVector<FlatBundleFieldEntry> fieldTypes;
422  auto srcType = newArgs[argIndex].first.type.cast<FIRRTLType>();
423  if (!peelType(srcType, fieldTypes))
424  return false;
425 
426  for (auto field : llvm::enumerate(fieldTypes)) {
427  auto newValue = addArg(module, 1 + argIndex + field.index(), srcType,
428  field.value(), newArgs[argIndex].first);
429  newArgs.insert(newArgs.begin() + 1 + argIndex + field.index(),
430  std::make_pair(newValue.second, newArgs[argIndex].second));
431  // Lower any other arguments by copying them to keep the relative order.
432  lowering.push_back(newValue.first);
433  }
434  return true;
435 }

449 void TypeLoweringVisitor::lowerSAWritePath(Operation *op,
450  ArrayRef<Operation *> writePath) {
451  SubaccessOp sao = cast<SubaccessOp>(writePath.back());
452  auto saoType = sao.input().getType().cast<FVectorType>();
453  auto selectWidth =
454  sao.index().getType().cast<FIRRTLType>().getBitWidthOrSentinel();
455 
456  for (size_t index = 0, e = saoType.getNumElements(); index < e; ++index) {
457  auto cond = builder->create<EQPrimOp>(
458  sao.index(),
459  builder->createOrFold<ConstantOp>(UIntType::get(context, selectWidth),
460  APInt(selectWidth, index)));
461  builder->create<WhenOp>(cond, false, [&]() {
462  // Recreate the write Path
463  Value leaf = builder->create<SubindexOp>(sao.input(), index);
464  for (int i = writePath.size() - 2; i >= 0; --i)
465  leaf = cloneAccess(builder, writePath[i], leaf);
466 
467  if (isa<ConnectOp>(op))
468  builder->create<ConnectOp>(leaf, op->getOperand(1));
469  else
470  builder->create<PartialConnectOp>(leaf, op->getOperand(1));
471  });
472  }
473 }

475 // Expand connects of aggregates
476 void TypeLoweringVisitor::visitStmt(ConnectOp op) {
477  if (processSAPath(op))
478  return;
479 
480  // Attempt to get the bundle types.
481  SmallVector<FlatBundleFieldEntry> fields;
482  if (!peelType(op.dest().getType(), fields))
483  return;
484 
485  // Loop over the leaf aggregates.
486  for (auto field : llvm::enumerate(fields)) {
487  Value src = getSubWhatever(op.src(), field.index());
488  Value dest = getSubWhatever(op.dest(), field.index());
489  if (field.value().isOutput)
490  std::swap(src, dest);
491  if (src.getType().isa<AnalogType>())
492  builder->create<AttachOp>(ArrayRef<Value>{dest, src});
493  else
494  builder->create<ConnectOp>(dest, src);
495  }
496  opsToRemove.push_back(op);
497 }

499 void TypeLoweringVisitor::visitStmt(PartialConnectOp op) {
500  if (processSAPath(op))
501  return;
502 
503  SmallVector<FlatBundleFieldEntry> srcFields, destFields;
504  peelType(op.src().getType(), srcFields);
505  bool dValid = peelType(op.dest().getType(), destFields);
506 
507  // Ground Type
508  if (!dValid) {
509  // check for truncation
510  Value src = op.src();
511  Value dest = op.dest();
512  auto srcType = src.getType().cast<FIRRTLType>();
513  auto destType = dest.getType().cast<FIRRTLType>();
514  auto srcWidth = srcType.getBitWidthOrSentinel();
515  auto destWidth = destType.getBitWidthOrSentinel();
516 
517  if (destType == srcType) {
518  builder->create<ConnectOp>(dest, src);
519  opsToRemove.push_back(op);
520  } else if (destType.isa<IntType>() && srcType.isa<IntType>() &&
521  destWidth >= 0) {
522  if (destWidth < srcWidth) {
523  // firrtl.tail always returns uint even for sint operands.
524  IntType tmpType = destType.cast<IntType>();
525  if (tmpType.isSigned())
526  tmpType = UIntType::get(destType.getContext(), destWidth);
527  src = builder->create<TailPrimOp>(tmpType, src, srcWidth - destWidth);
528  // Insert the cast back to signed if needed.
529  if (tmpType != destType)
530  src = builder->create<AsSIntPrimOp>(destType, src);
531  } else {
532  // Need to extend arg
533  src = builder->create<PadPrimOp>(src, destWidth);
534  }
535  builder->create<ConnectOp>(dest, src);
536  opsToRemove.push_back(op);
537  }
538  return;
539  }
540 
541  // Aggregates
542  if (FVectorType fvector = op.src().getType().dyn_cast<FVectorType>()) {
543  for (int index = 0, e = std::min(srcFields.size(), destFields.size());
544  index != e; ++index) {
545  Value src = builder->create<SubindexOp>(op.src(), index);
546  Value dest = builder->create<SubindexOp>(op.dest(), index);
547  if (src.getType() == dest.getType())
548  builder->create<ConnectOp>(dest, src);
549  else
550  builder->create<PartialConnectOp>(dest, src);
551  }
552  } else if (BundleType srcBundle = op.src().getType().dyn_cast<BundleType>()) {
553  // Pairwise connect on matching field names
554  BundleType destBundle = op.dest().getType().cast<BundleType>();
555  for (int srcIndex = 0, srcEnd = srcBundle.getNumElements();
556  srcIndex < srcEnd; ++srcIndex) {
557  auto srcName = srcBundle.getElement(srcIndex).name;
558  for (int destIndex = 0, destEnd = destBundle.getNumElements();
559  destIndex < destEnd; ++destIndex) {
560  auto destName = destBundle.getElement(destIndex).name;
561  if (srcName == destName) {
562  Value src = builder->create<SubfieldOp>(op.src(), srcIndex);
563  Value dest = builder->create<SubfieldOp>(op.dest(), destIndex);
564  if (destFields[destIndex].isOutput)
565  std::swap(src, dest);
566  if (src.getType().isa<AnalogType>())
567  builder->create<AttachOp>(ArrayRef<Value>{dest, src});
568  else if (src.getType() == dest.getType())
569  builder->create<ConnectOp>(dest, src);
570  else
571  builder->create<PartialConnectOp>(dest, src);
572  break;
573  }
574  }
575  }
576  } else {
577  op.emitError("Unknown aggregate type");
578  }
579 
580  opsToRemove.push_back(op);
581 }

599 /// Lower memory operations. A new memory is created for every leaf
600 /// element in a memory's data type.
601 void TypeLoweringVisitor::visitDecl(MemOp op) {
602  // Attempt to get the bundle types.
603  SmallVector<FlatBundleFieldEntry> fields;
604  if (!peelType(op.getDataType(), fields))
605  return;
606 
607  SmallVector<MemOp> newMemories;
608  SmallVector<Value> wireToOldResult;
609  SmallVector<WireOp> oldPorts;
610 
611  // Wires for old ports
612  for (unsigned int index = 0, end = op.getNumResults(); index < end; ++index) {
613  auto result = op.getResult(index);
614  auto wire = builder->create<WireOp>(
615  result.getType(),
616  (op.name() + "_" + op.getPortName(index).getValue()).str());
617  oldPorts.push_back(wire);
618  result.replaceAllUsesWith(wire.getResult());
619  }
620 
621  // Memory for each field
622  for (auto field : fields)
623  newMemories.push_back(cloneMemWithNewType(builder, op, field));
624 
625  // Hook up the new memories to the wires the old memory was replaced with.
626  for (size_t index = 0, rend = op.getNumResults(); index < rend; ++index) {
627  auto result = oldPorts[index];
628  auto rType = result.getType().cast<BundleType>();
629  for (size_t fieldIndex = 0, fend = rType.getNumElements();
630  fieldIndex != fend; ++fieldIndex) {
631  auto name = rType.getElement(fieldIndex).name.getValue();
632  auto oldField = builder->create<SubfieldOp>(result, fieldIndex);
633  // data and mask depend on the memory type which was split. They can also
634  // go both directions, depending on the port direction.
635  if (name == "data" || name == "mask" || name == "wdata" ||
636  name == "wmask" || name == "rdata") {
637  for (auto field : fields) {
638  auto realOldField = getSubWhatever(oldField, field.index);
639  auto newField = getSubWhatever(
640  newMemories[field.index].getResult(index), fieldIndex);
641  if (rType.getElement(fieldIndex).isFlip)
642  std::swap(realOldField, newField);
643  builder->create<ConnectOp>(newField, realOldField);
644  }
645  } else {
646  for (auto mem : newMemories) {
647  auto newField =
648  builder->create<SubfieldOp>(mem.getResult(index), fieldIndex);
649  builder->create<ConnectOp>(newField, oldField);
650  }
651  }
652  }
653  }
654  opsToRemove.push_back(op);
655 }

659  builder = &theBuilder;

662  OpBuilder builder(context);

665  SmallVector<unsigned> argsToRemove;

667  SmallVector<std::pair<ModulePortInfo, SmallVector<NamedAttribute>>> newArgs;
668  for (auto port : llvm::enumerate(extModule.getPorts())) {

698  SmallVector<Attribute, 8> newArgAttrs;

709  builder.getArrayAttr(newArgNames)));

712  direction::packAttribute(newArgDirections, context)));

715  newModuleAttrs.push_back(NamedAttribute(

718

720  extModule->setAttrs(newModuleAttrs);
721 
722  // Keep the module's type up-to-date.
723  extModule.setType(builder.getFunctionType(inputTypes, {}));

730  builder = &theBuilder;

733  lowerBlock(body);
734

737  // First get all the info for existing ports
738  SmallVector<std::pair<ModulePortInfo, SmallVector<NamedAttribute>>> newArgs;
739  for (auto port : llvm::enumerate(module.getPorts())) {

763  for (auto attr : module->getAttrDictionary())

765  // handled differently below.
766  if (attr.first != "portNames" && attr.first != direction::attrKey &&
767  attr.first != mlir::function_like_impl::getArgDictAttrName())
768  newModuleAttrs.push_back(attr);
769 
770  SmallVector<Attribute> newArgNames;
771  SmallVector<Direction> newArgDirections;
772  SmallVector<Attribute, 8> newArgAttrs;
773  for (auto &port : newArgs) {

780  builder->getArrayAttr(newArgNames)));

784

786  newModuleAttrs.push_back(NamedAttribute(
787  builder->getIdentifier(mlir::function_like_impl::getArgDictAttrName()),

867 void TypeLoweringVisitor::visitDecl(InstanceOp op) {
868  SmallVector<Type, 8> resultTypes;
869  SmallVector<int64_t, 8> endFields; // Compressed sparse row encoding
870  SmallVector<StringAttr, 8> resultNames;
871  bool skip = true;
872  auto oldPortAnno = op.portAnnotations();
873  SmallVector<Attribute> newPortAnno;
874 
875  endFields.push_back(0);
876  for (size_t i = 0, e = op.getNumResults(); i != e; ++i) {
877  auto srcType = op.getType(i).cast<FIRRTLType>();
878 
879  // Flatten any nested bundle types the usual way.
880  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
881  if (!peelType(srcType, fieldTypes)) {
882  resultTypes.push_back(srcType);
883  newPortAnno.push_back(oldPortAnno[i]);
884  } else {
885  skip = false;
886  // Store the flat type for the new bundle type.
887  for (auto field : fieldTypes) {
888  resultTypes.push_back(field.type);
889  newPortAnno.push_back(filterAnnotations(
890  context, oldPortAnno[i].dyn_cast_or_null<ArrayAttr>(), srcType,
891  field));
892  }
893  }
894  endFields.push_back(resultTypes.size());
895  }
896 
897  if (skip)
898  return;
899 
900  // FIXME: annotation update
901  auto newInstance = builder->create<InstanceOp>(
902  resultTypes, op.moduleNameAttr(), op.nameAttr(), op.annotations(),
903  builder->getArrayAttr(newPortAnno), op.lowerToBindAttr());
904 
905  SmallVector<Value> lowered;
906  for (size_t aggIndex = 0, eAgg = op.getNumResults(); aggIndex != eAgg;
907  ++aggIndex) {
908  lowered.clear();
909  for (size_t fieldIndex = endFields[aggIndex],
910  eField = endFields[aggIndex + 1];
911  fieldIndex < eField; ++fieldIndex)
912  lowered.push_back(newInstance.getResult(fieldIndex));
913  if (lowered.size() != 1 ||
914  op.getType(aggIndex) != resultTypes[endFields[aggIndex]])
915  processUsers(op.getResult(aggIndex), lowered);
916  else
917  op.getResult(aggIndex).replaceAllUsesWith(lowered[0]);
918  }
919  opsToRemove.push_back(op);
920 }

922 void TypeLoweringVisitor::visitExpr(SubaccessOp op) {
923  // Reads. All writes have been eliminated before now
924 
925  auto input = op.input();
926  auto vType = input.getType().cast<FVectorType>();
927  auto selectWidth =
928  op.index().getType().cast<FIRRTLType>().getBitWidthOrSentinel();
929 
930  Value mux = builder->create<InvalidValueOp>(vType.getElementType());
931  // Build up the mux
932  for (size_t index = 0, e = vType.getNumElements(); index < e; ++index) {
933  auto cond = builder->create<EQPrimOp>(
934  op.index(), builder->createOrFold<ConstantOp>(
935  UIntType::get(op.getContext(), selectWidth),
936  APInt(selectWidth, index)));
937  auto access = builder->create<SubindexOp>(input, index);
938  mux = builder->create<MuxPrimOp>(cond, access, mux);
939  }
940  op.replaceAllUsesWith(mux);
941  opsToRemove.push_back(op);
942 }

957 void LowerTypesPass::runParallel() {
958  // Collect the operations to iterate in a vector. We can't use parallelFor
959  // with the regular op list, since it requires a RandomAccessIterator. This
960  // also lets us use parallelForEachN, which means we don't have to
961  // llvm::enumerate the ops with their index. TODO(mlir): There should really
962  // be a way to do this without collecting the operations first.
963  std::deque<Operation *> ops;
964  llvm::for_each(getOperation().getBody()->getOperations(),
965  [&](Operation &op) { ops.push_back(&op); });
966 
967  mlir::ParallelDiagnosticHandler diagHandler(&getContext());
968  llvm::parallelForEachN(0, ops.size(), [&](auto index) {
969  // Notify the handler the op index and then perform lowering.
970  diagHandler.setOrderIDForThread(index);
971  TypeLoweringVisitor(&getContext()).lowerModule(ops[index]);
972  diagHandler.eraseOrderIDForThread();
973  });
974 }
