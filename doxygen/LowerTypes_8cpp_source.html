9 // This file defines the LowerTypes pass. This pass replaces aggregate types
10 // with expanded values.
11 //
12 // This pass walks the operations in reverse order. This lets it visit users
13 // before defs. Users can usually be expanded out to multiple operations (think
14 // mux of a bundle to muxes of each field) with a temporary subWhatever op
15 // inserted. When processing an aggregate producer, we blow out the op as
16 // appropriate, then walk the users, often those are subWhatever ops which can
17 // be bypassed and deleted. Function arguments are logically last on the
18 // operation visit order and walked left to right, being peeled one layer at a
19 // time with replacements inserted to the right of the original argument.
20 //
21 // Each processing of an op peels one layer of aggregate type off. Because new
22 // ops are inserted immediately above the current up, the walk will visit them
23 // next, effectively recusing on the aggregate types, without recusing. These
24 // potentially temporary ops(if the aggregate is complex) effectively serve as
25 // the worklist. Often aggregates are shallow, so the new ops are the final
26 // ones.
27 //

129 /// Return true if we can preserve the type.
130 static bool isPreservableAggregateType(Type type,
131  PreserveAggregate::PreserveMode mode) {
132  if (auto refType = type_dyn_cast<RefType>(type)) {
133  // Always preserve rwprobe's.
134  if (refType.getForceable())
135  return true;
136  // FIXME: Don't preserve read-only RefType for now. This is workaround for
137  // MemTap which causes type mismatches (issue 4479).
138  return false;
139  }
140 
141  // Return false if no aggregate value is preserved.
142  if (mode == PreserveAggregate::None)
143  return false;
144 
145  auto firrtlType = type_dyn_cast<FIRRTLBaseType>(type);
146  if (!firrtlType)
147  return false;
148 
149  // We can a preserve the type iff (i) the type is not passive, (ii) the type
150  // doesn't contain analog and (iii) type don't contain zero bitwidth.
151  if (!firrtlType.isPassive() || firrtlType.containsAnalog() ||
152  hasZeroBitWidth(firrtlType))
153  return false;
154 
155  switch (mode) {
156  case PreserveAggregate::All:
157  return true;
158  case PreserveAggregate::OneDimVec:
159  return isOneDimVectorType(firrtlType);
160  case PreserveAggregate::Vec:
161  return !containsBundleType(firrtlType);
162  default:
163  llvm_unreachable("unexpected mode");
164  }
165 }

167 /// Peel one layer of an aggregate type into its components. Type may be
168 /// complex, but empty, in which case fields is empty, but the return is true.
169 static bool peelType(Type type, SmallVectorImpl<FlatBundleFieldEntry> &fields,
170  PreserveAggregate::PreserveMode mode) {
171  // If the aggregate preservation is enabled and the type is preservable,
172  // then just return.
173  if (isPreservableAggregateType(type, mode))
174  return false;
175 
176  if (auto refType = type_dyn_cast<RefType>(type))
177  type = refType.getType();
178  return FIRRTLTypeSwitch<Type, bool>(type)
179  .Case<BundleType>([&](auto bundle) {
180  SmallString<16> tmpSuffix;
181  // Otherwise, we have a bundle type. Break it down.
182  for (size_t i = 0, e = bundle.getNumElements(); i < e; ++i) {
183  auto elt = bundle.getElement(i);
184  // Construct the suffix to pass down.
185  tmpSuffix.resize(0);
186  tmpSuffix.push_back('_');
187  tmpSuffix.append(elt.name.getValue());
188  fields.emplace_back(elt.type, i, bundle.getFieldID(i), tmpSuffix,
189  elt.isFlip);
190  }
191  return true;
192  })
193  .Case<FVectorType>([&](auto vector) {
194  // Increment the field ID to point to the first element.
195  for (size_t i = 0, e = vector.getNumElements(); i != e; ++i) {
196  fields.emplace_back(vector.getElementType(), i, vector.getFieldID(i),
197  "_" + std::to_string(i), false);
198  }
199  return true;
200  })
201  .Default([](auto op) { return false; });
202 }

204 /// Return if something is not a normal subaccess. Non-normal includes
205 /// zero-length vectors and constant indexes (which are really subindexes).
206 static bool isNotSubAccess(Operation *op) {
207  SubaccessOp sao = llvm::dyn_cast<SubaccessOp>(op);
208  if (!sao)
209  return true;

211  llvm::dyn_cast_or_null<ConstantOp>(sao.getIndex().getDefiningOp());

215 /// Look through and collect subfields leading to a subaccess.
216 static SmallVector<Operation *> getSAWritePath(Operation *op) {
217  SmallVector<Operation *> retval;
218  auto defOp = op->getOperand(0).getDefiningOp();
219  while (isa_and_nonnull<SubfieldOp, SubindexOp, SubaccessOp>(defOp)) {
220  retval.push_back(defOp);
221  defOp = defOp->getOperand(0).getDefiningOp();
222  }
223  // Trim to the subaccess
224  while (!retval.empty() && isNotSubAccess(retval.back()))
225  retval.pop_back();
226  return retval;
227 }

229 /// Clone memory for the specified field. Returns null op on error.
230 static MemOp cloneMemWithNewType(ImplicitLocOpBuilder *b, MemOp op,
231  FlatBundleFieldEntry field) {
232  SmallVector<Type, 8> ports;
233  SmallVector<Attribute, 8> portNames;
234  SmallVector<Attribute, 8> portLocations;
235 
236  auto oldPorts = op.getPorts();
237  for (size_t portIdx = 0, e = oldPorts.size(); portIdx < e; ++portIdx) {
238  auto port = oldPorts[portIdx];
239  ports.push_back(
240  MemOp::getTypeForPort(op.getDepth(), field.type, port.second));
241  portNames.push_back(port.first);
242  }
243 
244  // It's easier to duplicate the old annotations, then fix and filter them.
245  auto newMem = b->create<MemOp>(
246  ports, op.getReadLatency(), op.getWriteLatency(), op.getDepth(),
247  op.getRuw(), portNames, (op.getName() + field.suffix).str(),
248  op.getNameKind(), op.getAnnotations().getValue(),
249  op.getPortAnnotations().getValue(), op.getInnerSymAttr());
250 
251  if (op.getInnerSym()) {
252  op.emitError("cannot split memory with symbol present");
253  return {};
254  }
255 
256  SmallVector<Attribute> newAnnotations;
257  for (size_t portIdx = 0, e = newMem.getNumResults(); portIdx < e; ++portIdx) {
258  auto portType = type_cast<BundleType>(newMem.getResult(portIdx).getType());
259  auto oldPortType = type_cast<BundleType>(op.getResult(portIdx).getType());
260  SmallVector<Attribute> portAnno;
261  for (auto attr : newMem.getPortAnnotation(portIdx)) {
262  Annotation anno(attr);
263  if (auto annoFieldID = anno.getFieldID()) {
264  auto targetIndex = oldPortType.getIndexForFieldID(annoFieldID);
265 
266  // Apply annotations to all elements if the target is the whole
267  // sub-field.
268  if (annoFieldID == oldPortType.getFieldID(targetIndex)) {
269  anno.setMember(
270  "circt.fieldID",
271  b->getI32IntegerAttr(portType.getFieldID(targetIndex)));
272  portAnno.push_back(anno.getDict());
273  continue;
274  }
275 
276  // Handle aggregate sub-fields, including `(r/w)data` and `(w)mask`.
277  if (type_isa<BundleType>(oldPortType.getElement(targetIndex).type)) {
278  // Check whether the annotation falls into the range of the current
279  // field. Note that the `field` here is peeled from the `data`
280  // sub-field of the memory port, thus we need to add the fieldID of
281  // `data` or `mask` sub-field to get the "real" fieldID.
282  auto fieldID = field.fieldID + oldPortType.getFieldID(targetIndex);
283  if (annoFieldID >= fieldID &&
284  annoFieldID <=
285  fieldID + hw::FieldIdImpl::getMaxFieldID(field.type)) {
286  // Set the field ID of the new annotation.
287  auto newFieldID =
288  annoFieldID - fieldID + portType.getFieldID(targetIndex);
289  anno.setMember("circt.fieldID", b->getI32IntegerAttr(newFieldID));
290  portAnno.push_back(anno.getDict());
291  }
292  }
293  } else
294  portAnno.push_back(attr);
295  }
296  newAnnotations.push_back(b->getArrayAttr(portAnno));
297  }
298  newMem.setAllPortAnnotations(newAnnotations);
299  return newMem;
300 }

476 /// Conditionally expand a subaccessop write path
477 bool TypeLoweringVisitor::processSAPath(Operation *op) {
478  // Does this LHS have a subaccessop?
479  SmallVector<Operation *> writePath = getSAWritePath(op);
480  if (writePath.empty())
481  return false;
482 
483  lowerSAWritePath(op, writePath);
484  // Unhook the writePath from the connect. This isn't the right type, but we
485  // are deleting the op anyway.
486  op->eraseOperands(0, 2);
487  // See how far up the tree we can delete things.
488  for (size_t i = 0; i < writePath.size(); ++i) {
489  if (writePath[i]->use_empty()) {
490  writePath[i]->erase();
491  } else {
492  break;
493  }
494  }
495  return true;
496 }

498 void TypeLoweringVisitor::lowerBlock(Block *block) {
499  // Lower the operations bottom up.
500  for (auto it = block->rbegin(), e = block->rend(); it != e;) {
501  auto &iop = *it;
502  builder->setInsertionPoint(&iop);
503  builder->setLoc(iop.getLoc());
504  bool removeOp = dispatchVisitor(&iop);
505  ++it;
506  // Erase old ops eagerly so we don't have dangling uses we've already
507  // lowered.
508  if (removeOp)
509  iop.erase();
510  }

513 ArrayAttr TypeLoweringVisitor::filterAnnotations(MLIRContext *ctxt,
514  ArrayAttr annotations,
515  FIRRTLType srcType,
516  FlatBundleFieldEntry field) {
517  SmallVector<Attribute> retval;
518  if (!annotations || annotations.empty())
519  return ArrayAttr::get(ctxt, retval);
520  for (auto opAttr : annotations) {
521  Annotation anno(opAttr);
522  auto fieldID = anno.getFieldID();
523  anno.removeMember("circt.fieldID");
524 
525  // If no fieldID set, or points to root, forward the annotation without the
526  // fieldID field (which was removed above).
527  if (fieldID == 0) {
528  retval.push_back(anno.getAttr());
529  continue;
530  }
531  // Check whether the annotation falls into the range of the current field.
532 
533  if (fieldID < field.fieldID ||
534  fieldID > field.fieldID + hw::FieldIdImpl::getMaxFieldID(field.type))
535  continue;
536 
537  // Add fieldID back if non-zero relative to this field.
538  if (auto newFieldID = fieldID - field.fieldID) {
539  // If the target is a subfield/subindex of the current field, create a
540  // new annotation with the correct circt.fieldID.
541  anno.setMember("circt.fieldID", builder->getI32IntegerAttr(newFieldID));
542  }
543 
544  retval.push_back(anno.getAttr());
545  }
546  return ArrayAttr::get(ctxt, retval);
547 }

549 LogicalResult TypeLoweringVisitor::partitionSymbols(
550  hw::InnerSymAttr sym, FIRRTLType parentType,
551  SmallVectorImpl<hw::InnerSymAttr> &newSyms, Location errorLoc) {
552 
553  // No symbol, nothing to partition.
554  if (!sym || sym.empty())
555  return success();
556 
557  auto *context = sym.getContext();
558 
559  auto baseType = getBaseType(parentType);
560  if (!baseType)
561  return mlir::emitError(errorLoc,
562  "unable to partition symbol on unsupported type ")
563  << parentType;
564 
565  return TypeSwitch<FIRRTLType, LogicalResult>(baseType)
566  .Case<BundleType, FVectorType>([&](auto aggType) -> LogicalResult {
567  struct BinningInfo {
568  uint64_t index;
569  uint64_t relFieldID;
570  hw::InnerSymPropertiesAttr prop;
571  };
572 
573  // Walk each inner symbol, compute binning information/assignment.
574  SmallVector<BinningInfo> binning;
575  for (auto prop : sym) {
576  auto fieldID = prop.getFieldID();
577  // Special-case fieldID == 0, helper methods require non-zero fieldID.
578  if (fieldID == 0)
579  return mlir::emitError(errorLoc, "unable to lower due to symbol ")
580  << prop.getName()
581  << " with target not preserved by lowering";
582  auto [index, relFieldID] = aggType.getIndexAndSubfieldID(fieldID);
583  binning.push_back({index, relFieldID, prop});
584  }
585 
586  // Sort by index, fieldID.
587  llvm::stable_sort(binning, [&](auto &lhs, auto &rhs) {
588  return std::tuple(lhs.index, lhs.relFieldID) <
589  std::tuple(rhs.index, rhs.relFieldID);
590  });
591  assert(!binning.empty());
592 
593  // Populate newSyms, group all symbols on same index.
594  newSyms.resize(aggType.getNumElements());
595  for (auto binIt = binning.begin(), binEnd = binning.end();
596  binIt != binEnd;) {
597  auto curIndex = binIt->index;

608  newSyms[curIndex] = hw::InnerSymAttr::get(context, propsForIndex);
609  }

614  errorLoc, "unable to partition symbol on unsupported type ")

620  Operation *op,
621  llvm::function_ref<Value(const FlatBundleFieldEntry &, ArrayAttr)> clone,
622  Type srcType) {
623

625  srcType = op->getResult(0).getType();
626  auto srcFType = type_dyn_cast<FIRRTLType>(srcType);
627  if (!srcFType)
628  return false;
629  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
630 
631  if (!peelType(srcFType, fieldTypes, aggregatePreservationMode))
632  return false;
633 
634  SmallVector<Value> lowered;
635  // Loop over the leaf aggregates.
636  SmallString<16> loweredName;
637  auto nameKindAttr = op->getAttrOfType<NameKindEnumAttr>(cache.nameKindAttr);
638 
639  if (auto nameAttr = op->getAttrOfType<StringAttr>(cache.nameAttr))
640  loweredName = nameAttr.getValue();
641  auto baseNameLen = loweredName.size();
642  auto oldAnno = op->getAttr("annotations").dyn_cast_or_null<ArrayAttr>();
643

645  if (auto symOp = dyn_cast<hw::InnerSymbolOpInterface>(op)) {
646  if (failed(partitionSymbols(symOp.getInnerSymAttr(), srcFType, fieldSyms,
647  symOp.getLoc()))) {
648  encounteredError = true;
649  return false;
650  }
651  }
652 
653  for (const auto &[field, sym] : llvm::zip_equal(fieldTypes, fieldSyms)) {
654  if (!loweredName.empty()) {
655  loweredName.resize(baseNameLen);
656  loweredName += field.suffix;
657  }
658 
659  // For all annotations on the parent op, filter them based on the target
660  // attribute.
661  ArrayAttr loweredAttrs =
662  filterAnnotations(context, oldAnno, srcFType, field);
663  auto newVal = clone(field, loweredAttrs);
664 
665  // If inner symbols on this field, add to new op.
666  if (sym) {
667  // Splitting up something with symbols on it should lower to ops
668  // that also can have symbols on them.
669  auto newSymOp = newVal.getDefiningOp<hw::InnerSymbolOpInterface>();
670  assert(
671  newSymOp &&
672  "op with inner symbol lowered to op that cannot take inner symbol");
673  newSymOp.setInnerSymbolAttr(sym);
674  }
675 
676  // Carry over the name, if present.
677  if (auto *newOp = newVal.getDefiningOp()) {
678  if (!loweredName.empty())
679  newOp->setAttr(cache.nameAttr, StringAttr::get(context, loweredName));
680  if (nameKindAttr)
681  newOp->setAttr(cache.nameKindAttr, nameKindAttr);
682  }
683  lowered.push_back(newVal);
684  }
685

687  return true;

690 void TypeLoweringVisitor::processUsers(Value val, ArrayRef<Value> mapping) {
691  for (auto *user : llvm::make_early_inc_range(val.getUsers())) {
692  TypeSwitch<Operation *, void>(user)
693  .Case<SubindexOp>([mapping](SubindexOp sio) {
694  Value repl = mapping[sio.getIndex()];
695  sio.replaceAllUsesWith(repl);
696  sio.erase();
697  })
698  .Case<SubfieldOp>([mapping](SubfieldOp sfo) {
699  // Get the input bundle type.
700  Value repl = mapping[sfo.getFieldIndex()];
701  sfo.replaceAllUsesWith(repl);
702  sfo.erase();
703  })
704  .Case<RefSubOp>([mapping](RefSubOp refSub) {
705  Value repl = mapping[refSub.getIndex()];
706  refSub.replaceAllUsesWith(repl);
707  refSub.erase();
708  })
709  .Default([&](auto op) {
710  // This means we have already processed the user, and it didn't lower
711  // its inputs. This is an opaque user, which will continue to have
712  // aggregate type as input, even after LowerTypes. So, construct the
713  // vector/bundle back from the lowered elements to ensure a valid
714  // input into the opaque op. This only supports Bundles and Vectors.
715 
716  // This builder ensures that the aggregate construction happens at the
717  // user location, and the LowerTypes algorithm will not touch them any
718  // more, because LowerTypes was reverse iterating on the block and the
719  // user has already been processed.
720  ImplicitLocOpBuilder b(user->getLoc(), user);
721 
722  // This shouldn't happen (non-FIRRTLBaseType's in lowered types, or
723  // refs), check explicitly here for clarity/early detection.
724  assert(llvm::none_of(mapping, [](auto v) {
725  auto fbasetype = type_dyn_cast<FIRRTLBaseType>(v.getType());
726  return !fbasetype || fbasetype.containsReference();
727  }));
728 
729  Value input =
730  TypeSwitch<Type, Value>(val.getType())
731  .template Case<FVectorType>([&](auto vecType) {
732  return b.createOrFold<VectorCreateOp>(vecType, mapping);
733  })
734  .template Case<BundleType>([&](auto bundleType) {
735  return b.createOrFold<BundleCreateOp>(bundleType, mapping);
736  })
737  .Default([&](auto _) -> Value { return {}; });
738  if (!input) {
739  user->emitError("unable to reconstruct source of type ")
740  << val.getType();
741  encounteredError = true;
742  return;
743  }
744  user->replaceUsesOfWith(val, input);
745  });
746  }

788 // Lower arguments with bundle type by flattening them.
789 bool TypeLoweringVisitor::lowerArg(FModuleLike module, size_t argIndex,
790  size_t argsRemoved,
791  SmallVectorImpl<PortInfoWithIP> &newArgs,
792  SmallVectorImpl<Value> &lowering) {
793

795  SmallVector<FlatBundleFieldEntry> fieldTypes;
796  auto srcType = type_cast<FIRRTLType>(newArgs[argIndex].pi.type);
797  if (!peelType(srcType, fieldTypes, getPreservationModeForModule(module)))
798  return false;
799

801  if (auto ip = newArgs[argIndex].internalPath; ip && ip->getPath()) {

805  return false;

808  SmallVector<hw::InnerSymAttr> fieldSyms(fieldTypes.size());

821  lowering.push_back(newValue.first);

950 /// Lower memory operations. A new memory is created for every leaf
951 /// element in a memory's data type.
952 bool TypeLoweringVisitor::visitDecl(MemOp op) {
953  // Attempt to get the bundle types.
954  SmallVector<FlatBundleFieldEntry> fields;
955 
956  // MemOp should have ground types so we can't preserve aggregates.
957  if (!peelType(op.getDataType(), fields, memoryPreservationMode))
958  return false;
959 
960  if (op.getInnerSym()) {
961  op->emitError() << "has a symbol, but no symbols may exist on aggregates "
962  "passed through LowerTypes";
963  encounteredError = true;
964  return false;
965  }
966 
967  SmallVector<MemOp> newMemories;
968  SmallVector<WireOp> oldPorts;
969 
970  // Wires for old ports
971  for (unsigned int index = 0, end = op.getNumResults(); index < end; ++index) {
972  auto result = op.getResult(index);
973  if (op.getPortKind(index) == MemOp::PortKind::Debug) {
974  op.emitOpError("cannot lower memory with debug port");
975  encounteredError = true;
976  return false;
977  }
978  auto wire = builder->create<WireOp>(
979  result.getType(),
980  (op.getName() + "_" + op.getPortName(index).getValue()).str());
981  oldPorts.push_back(wire);
982  result.replaceAllUsesWith(wire.getResult());
983  }
984  // If annotations targeting fields of an aggregate are present, we cannot
985  // flatten the memory. It must be split into one memory per aggregate field.
986  // Do not overwrite the pass flag!
987 
988  // Memory for each field
989  for (const auto &field : fields) {
990  auto newMemForField = cloneMemWithNewType(builder, op, field);
991  if (!newMemForField) {
992  op.emitError("failed cloning memory for field");
993  encounteredError = true;
994  return false;
995  }
996  newMemories.push_back(newMemForField);
997  }
998  // Hook up the new memories to the wires the old memory was replaced with.
999  for (size_t index = 0, rend = op.getNumResults(); index < rend; ++index) {
1000  auto result = oldPorts[index].getResult();
1001  auto rType = type_cast<BundleType>(result.getType());
1002  for (size_t fieldIndex = 0, fend = rType.getNumElements();
1003  fieldIndex != fend; ++fieldIndex) {
1004  auto name = rType.getElement(fieldIndex).name.getValue();
1005  auto oldField = builder->create<SubfieldOp>(result, fieldIndex);
1006  // data and mask depend on the memory type which was split. They can also
1007  // go both directions, depending on the port direction.
1008  if (name == "data" || name == "mask" || name == "wdata" ||
1009  name == "wmask" || name == "rdata") {
1010  for (const auto &field : fields) {
1011  auto realOldField = getSubWhatever(oldField, field.index);
1012  auto newField = getSubWhatever(
1013  newMemories[field.index].getResult(index), fieldIndex);
1014  if (rType.getElement(fieldIndex).isFlip)
1015  std::swap(realOldField, newField);
1016  emitConnect(*builder, newField, realOldField);
1017  }
1018  } else {
1019  for (auto mem : newMemories) {
1020  auto newField =
1021  builder->create<SubfieldOp>(mem.getResult(index), fieldIndex);
1022  emitConnect(*builder, newField, oldField);
1023  }
1024  }
1025  }
1026  }
1027  return true;
1028 }

1321 // Expand BitCastOp of aggregates
1322 bool TypeLoweringVisitor::visitExpr(BitCastOp op) {
1323  Value srcLoweredVal = op.getInput();
1324  // If the input is of aggregate type, then cat all the leaf fields to form a
1325  // UInt type result. That is, first bitcast the aggregate type to a UInt.
1326  // Attempt to get the bundle types.
1327  SmallVector<FlatBundleFieldEntry> fields;
1328  if (peelType(op.getInput().getType(), fields, PreserveAggregate::None)) {
1329  size_t uptoBits = 0;
1330  // Loop over the leaf aggregates and concat each of them to get a UInt.
1331  // Bitcast the fields to handle nested aggregate types.
1332  for (const auto &field : llvm::enumerate(fields)) {
1333  auto fieldBitwidth = *getBitWidth(field.value().type);
1334  // Ignore zero width fields, like empty bundles.
1335  if (fieldBitwidth == 0)
1336  continue;
1337  Value src = getSubWhatever(op.getInput(), field.index());
1338  // The src could be an aggregate type, bitcast it to a UInt type.
1339  src = builder->createOrFold<BitCastOp>(
1340  UIntType::get(context, fieldBitwidth), src);
1341  // Take the first field, or else Cat the previous fields with this field.
1342  if (uptoBits == 0)
1343  srcLoweredVal = src;
1344  else
1345  srcLoweredVal = builder->create<CatPrimOp>(src, srcLoweredVal);
1346  // Record the total bits already accumulated.
1347  uptoBits += fieldBitwidth;
1348  }
1349  } else {
1350  srcLoweredVal = builder->createOrFold<AsUIntPrimOp>(srcLoweredVal);
1351  }
1352  // Now the input has been cast to srcLoweredVal, which is of UInt type.
1353  // If the result is an aggregate type, then use lowerProducer.
1354  if (type_isa<BundleType, FVectorType>(op.getResult().getType())) {
1355  // uptoBits is used to keep track of the bits that have been extracted.
1356  size_t uptoBits = 0;
1357  auto clone = [&](const FlatBundleFieldEntry &field,
1358  ArrayAttr attrs) -> Value {
1359  // All the fields must have valid bitwidth, a requirement for BitCastOp.
1360  auto fieldBits = *getBitWidth(field.type);
1361  // If empty field, then it doesnot have any use, so replace it with an
1362  // invalid op, which should be trivially removed.
1363  if (fieldBits == 0)
1364  return builder->create<InvalidValueOp>(field.type);
1365 
1366  // Assign the field to the corresponding bits from the input.
1367  // Bitcast the field, incase its an aggregate type.
1368  auto extractBits = builder->create<BitsPrimOp>(
1369  srcLoweredVal, uptoBits + fieldBits - 1, uptoBits);
1370  uptoBits += fieldBits;
1371  return builder->create<BitCastOp>(field.type, extractBits);
1372  };
1373  return lowerProducer(op, clone);
1374  }
1375 
1376  // If ground type, then replace the result.
1377  if (type_isa<SIntType>(op.getType()))
1378  srcLoweredVal = builder->create<AsSIntPrimOp>(srcLoweredVal);
1379  op.getResult().replaceAllUsesWith(srcLoweredVal);
1380  return true;
1381 }
