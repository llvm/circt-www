34 void LoweringOptions::parse(StringRef text, ErrorHandlerT errorHandler) {
35  while (!text.empty()) {
36  // Remove the first option from the text.
37  auto split = text.split(",");
38  auto option = split.first.trim();
39  text = split.second;
40  if (option == "") {
41  // Empty options are fine.
42  } else if (option == "alwaysFF") {
43  useAlwaysFF = true;
44  } else if (option.startswith("emittedLineLength=")) {
45  option = option.drop_front(strlen("emittedLineLength="));
46  if (option.getAsInteger(10, emittedLineLength)) {
47  errorHandler("expected integer source width");
48  emittedLineLength = 90;
49  }
50  } else {
51  errorHandler(llvm::Twine("unknown style option \'") + option + "\'");
52  // We continue parsing options after a failure.
53  }
54  }
55 }

57 std::string LoweringOptions::toString() const {
58  std::string options = "";
59  // All options should add a trailing comma to simplify the code.
60  if (useAlwaysFF)
61  options += "alwaysFF,";
62  if (emittedLineLength != 90)
63  options += "emittedLineLength=" + std::to_string(emittedLineLength) + ',';
64 
65  // Remove a trailing comma if present.
66  if (!options.empty()) {
67  assert(options.back() == ',' && "all options should add a trailing comma");
68  options.pop_back();
69  }
70  return options;
71 }
