44 void LoweringOptions::parse(StringRef text, ErrorHandlerT errorHandler) {
45  while (!text.empty()) {
46  // Remove the first option from the text.
47  auto split = text.split(",");
48  auto option = split.first.trim();
49  text = split.second;
50  if (option == "") {
51  // Empty options are fine.
52  } else if (option == "noAlwaysComb") {
53  noAlwaysComb = true;
54  } else if (option == "exprInEventControl") {
55  allowExprInEventControl = true;
56  } else if (option == "disallowPackedArrays") {
57  disallowPackedArrays = true;
58  } else if (option == "disallowLocalVariables") {
59  disallowLocalVariables = true;
60  } else if (option == "verifLabels") {
61  enforceVerifLabels = true;
62  } else if (option.consume_front("emittedLineLength=")) {
63  if (option.getAsInteger(10, emittedLineLength)) {
64  errorHandler("expected integer source width");
65  emittedLineLength = DEFAULT_LINE_LENGTH;
66  }
67  } else if (option == "explicitBitcast") {
68  explicitBitcast = true;
69  } else if (option == "emitReplicatedOpsToHeader") {
70  emitReplicatedOpsToHeader = true;
71  } else if (option.consume_front("maximumNumberOfTermsPerExpression=")) {
72  if (option.getAsInteger(10, maximumNumberOfTermsPerExpression)) {
73  errorHandler("expected integer source width");
74  maximumNumberOfTermsPerExpression = DEFAULT_TERM_LIMIT;
75  }
76  } else if (option.consume_front("maximumNumberOfTermsInConcat=")) {
77  if (option.getAsInteger(10, maximumNumberOfTermsInConcat)) {
78  errorHandler("expected integer source width");
79  maximumNumberOfTermsInConcat = DEFAULT_CONCAT_TERM_LIMIT;
80  }
81  } else if (option.consume_front("locationInfoStyle=")) {
82  if (auto style = parseLocationInfoStyle(option)) {
83  locationInfoStyle = *style;
84  } else {
85  errorHandler("expected 'plain', 'wrapInAtSquareBracket', or 'none'");
86  }
87  } else if (option == "disallowPortDeclSharing") {
88  disallowPortDeclSharing = true;
89  } else if (option == "printDebugInfo") {
90  printDebugInfo = true;
91  } else if (option == "spillWiresAtPrepare") {
92  spillWiresAtPrepare = true;
93  } else {
94  errorHandler(llvm::Twine("unknown style option \'") + option + "\'");
95  // We continue parsing options after a failure.
96  }
97  }
98 
99  if (spillWiresAtPrepare && !disallowLocalVariables) {
100  // FIXME: Currently we cannot spill temporaries in procedural regions. Hence
101  // `spillWiresAtPrepare` must be used together with
102  // `disallowLocalVariables`.
103  errorHandler(llvm::Twine(
104  "`spillWiresAtPrepare` must be used with `disallowLocalVariables`"));
105  }
106 }

108 std::string LoweringOptions::toString() const {
109  std::string options = "";
110  // All options should add a trailing comma to simplify the code.
111  if (noAlwaysComb)
112  options += "noAlwaysComb,";
113  if (allowExprInEventControl)
114  options += "exprInEventControl,";
115  if (disallowPackedArrays)
116  options += "disallowPackedArrays,";
117  if (disallowLocalVariables)
118  options += "disallowLocalVariables,";
119  if (enforceVerifLabels)
120  options += "verifLabels,";
121  if (explicitBitcast)
122  options += "explicitBitcast,";
123  if (emitReplicatedOpsToHeader)
124  options += "emitReplicatedOpsToHeader,";
125  if (locationInfoStyle == LocationInfoStyle::WrapInAtSquareBracket)
126  options += "locationInfoStyle=wrapInAtSquareBracket,";
127  if (locationInfoStyle == LocationInfoStyle::None)
128  options += "locationInfoStyle=none,";
129  if (disallowPortDeclSharing)
130  options += "disallowPortDeclSharing,";
131  if (printDebugInfo)
132  options += "printDebugInfo,";
133  if (spillWiresAtPrepare)
134  options += "spillWiresAtPrepare,";
135 
136  if (emittedLineLength != DEFAULT_LINE_LENGTH)
137  options += "emittedLineLength=" + std::to_string(emittedLineLength) + ',';
138  if (maximumNumberOfTermsPerExpression != DEFAULT_TERM_LIMIT)
139  options += "maximumNumberOfTermsPerExpression=" +
140  std::to_string(maximumNumberOfTermsPerExpression) + ',';
141  if (maximumNumberOfTermsInConcat != DEFAULT_CONCAT_TERM_LIMIT)
142  options += "maximumNumberOfTermsInConcat=" +
143  std::to_string(maximumNumberOfTermsInConcat) + ',';
144 
145  // Remove a trailing comma if present.
146  if (!options.empty()) {
147  assert(options.back() == ',' && "all options should add a trailing comma");
148  options.pop_back();
149  }
150  return options;
151 }

210 void circt::applyLoweringCLOptions(ModuleOp module) {
211  // If the command line options were not registered in the first place, there
212  // is nothing to parse.
213  if (!clOptions.isConstructed())
214  return;
215 
216  // If an output style is applied on the command line, all previous options are
217  // discarded.
218  if (clOptions->loweringOptions.getNumOccurrences()) {
219  clOptions->loweringOptions.setAsAttribute(module);
220  }
221 }

223 LoweringOptions
224 circt::getLoweringCLIOption(mlir::ModuleOp module,
225  LoweringOptions::ErrorHandlerT errorHandler) {
226  // If the command line options were not registered in the first place, use the
227  // lowering option associated with module op.
228  if (!clOptions.isConstructed() ||
229  !clOptions->loweringOptions.getNumOccurrences()) {
230  if (auto styleAttr = LoweringOptions::getAttributeFrom(module))
231  return LoweringOptions(styleAttr, errorHandler);
232  // If the module doesn't have a lowering option, then use the default value.
233  return LoweringOptions();
234  }
235 
236  return clOptions->loweringOptions.getValue();
237 }
