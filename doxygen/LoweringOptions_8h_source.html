25 /// Options which control the emission from CIRCT to Verilog.
26 struct LoweringOptions {
27  /// Error callback type used to indicate errors parsing the options string.
28  using ErrorHandlerT = function_ref<void(llvm::Twine)>;
29 
30  /// Create a LoweringOptions with the default values.
31  LoweringOptions() = default;
32 
33  /// Create a LoweringOptions and read in options from a string,
34  /// overriding only the set options in the string.
35  LoweringOptions(StringRef options, ErrorHandlerT errorHandler);
36 
37  /// Create a LoweringOptions with values loaded from an MLIR ModuleOp. This
38  /// loads a string attribute with the key `circt.loweringOptions`. If there is
39  /// an error parsing the attribute this will print an error using the
40  /// ModuleOp.
41  LoweringOptions(mlir::ModuleOp module);
42 
43  /// Return the value of the `circt.loweringOptions` in the specified module
44  /// if present, or a null attribute if not.
45  static StringAttr getAttributeFrom(ModuleOp module);
46 
47  /// Read in options from a string, overriding only the set options in the
48  /// string.
49  void parse(StringRef options, ErrorHandlerT callback);
50 
51  /// Returns a string representation of the options.
52  std::string toString() const;
53 
54  /// Write the verilog emitter options to a module's attributes.
55  void setAsAttribute(mlir::ModuleOp module);
56 
57  /// Load any emitter options from the module. If there is an error validating
58  /// the attribute, this will print an error using the ModuleOp.
59  void parseFromAttribute(mlir::ModuleOp module);
60 
61  /// If true, emits `sv.alwayscomb` as Verilog `always @(*)` statements.
62  /// Otherwise, print them as `always_comb`.
63  bool noAlwaysComb = false;
64 
65  /// If true, expressions are allowed in the sensitivity list of `always`
66  /// statements, otherwise they are forced to be simple wires. Some EDA
67  /// tools rely on these being simple wires.
68  bool allowExprInEventControl = false;
69 
70  /// If true, eliminate packed arrays for tools that don't support them (e.g.
71  /// Yosys).
72  bool disallowPackedArrays = false;
73 
74  /// If true, do not emit SystemVerilog locally scoped "automatic" or logic
75  /// declarations - emit top level wire and reg's instead.
76  bool disallowLocalVariables = false;
77 
78  /// If true, verification statements like `assert`, `assume`, and `cover` will
79  /// always be emitted with a label. If the statement has no label in the IR, a
80  /// generic one will be created. Some EDA tools require verification
81  /// statements to be labeled.
82  bool enforceVerifLabels = false;
83 
84  /// This is the maximum number of terms in an expression before that
85  /// expression spills a wire.
86  enum { DEFAULT_TERM_LIMIT = 256 };
87  unsigned maximumNumberOfTermsPerExpression = DEFAULT_TERM_LIMIT;
88 
89  /// This is the maximum number of terms in an expression used in a concat
90  /// before that expression spills a wire.
91  enum { DEFAULT_CONCAT_TERM_LIMIT = 10 };
92  unsigned maximumNumberOfTermsInConcat = DEFAULT_CONCAT_TERM_LIMIT;
93 
94  /// This is the target width of lines in an emitted Verilog source file in
95  /// columns.
96  enum { DEFAULT_LINE_LENGTH = 90 };
97  unsigned emittedLineLength = DEFAULT_LINE_LENGTH;
98 
99  /// Add an explicit bitcast for avoiding bitwidth mismatch LINT errors.
100  bool explicitBitcast = false;
101 
102  /// If true, replicated ops are emitted to a header file.
103  bool emitReplicatedOpsToHeader = false;
104 
105  /// This option controls emitted location information style.
106  enum LocationInfoStyle {
107  Plain, // Default.
108  WrapInAtSquareBracket, // Wrap location info in @[..].
109  None, // No location info comment.
110  } locationInfoStyle = Plain;
111 
112  /// If true, every port is declared separately
113  /// (each includes direction and type (e.g., `input [3:0]`)).
114  /// When false (default), ports share declarations when possible.
115  bool disallowPortDeclSharing = false;
116 
117  /// Print debug info.
118  bool printDebugInfo = false;
119 
120  /// If true, verilog expression legalization is performed at
121  /// PrepareForEmission via IR mutation. This means temporary wires are
122  /// explicitly spilled to wire ops instead of implicilty being spilled at
123  /// emission. Once ExportVerilog simplification is completed, this feature
124  /// shoiuld be enabled by default.
125  bool spillWiresAtPrepare = false;
126 };

139 LoweringOptions getLoweringCLIOption(ModuleOp module,
140  LoweringOptions::ErrorHandlerT);
