40 LogicalResult OpGenerator::runOnOperation(mlir::Operation *op,
41  GeneratorSet generatorSet) {
42  if (generators.size() == 0)
43  return failure();
44  if (generators.size() > 1 && generatorSet.size() < 1)
45  return op->emitError("at least one generator must be selected");
46 
47  // Check if any of the generators were selected in the generator set. If more
48  // than one candidate is present in the generator set, raise an error.
49  GeneratorCallback gen;
50  Attribute parameters = op->getAttr("parameters");
51  for (auto &generatorPair : generators) {
52  if (generatorSet.contains(generatorPair.first())) {
53  if (gen)
54  return op->emitError("multiple generators selected");
55  auto callbackPair = generatorPair.second.find(parameters);
56  if (callbackPair != generatorPair.second.end())
57  gen = callbackPair->second;
58  }
59  }
60 
61  // If no generator was selected by the generator set, and there is just one
62  // generator, default to using that. Otherwise raise an error.
63  if (!gen) {
64  if (generators.size() == 1) {
65  auto generatorMap = generators.begin()->second;
66  auto callbackPair = generatorMap.find(parameters);
67  if (callbackPair != generatorMap.end())
68  gen = callbackPair->second;
69  } else {
70  return op->emitError("unable to select a generator");
71  }
72  }
73 
74  mlir::IRRewriter rewriter(op->getContext());
75  Operation *replacement = gen(op);
76  if (replacement == nullptr)
77  return op->emitError("Failed generator on ") << op->getName();
78  replacement->setLoc(op->getLoc());
79  rewriter.replaceOp(op, replacement->getResults());
80  return success();
81 }

122 void RunGeneratorsPass::runOnOperation() {
123  GeneratorSet generatorSet;
124  for (auto &generator : generators)
125  if (!generator.empty())
126  generatorSet.insert(StringRef(generator));
127  MLIRContext *ctxt = &getContext();
128  MSFTDialect *msft = ctxt->getLoadedDialect<MSFTDialect>();
129  if (!msft)
130  return;
131 
132  Operation *top = getOperation();
133  top->walk([this, msft, generatorSet](Operation *op) {
134  if (failed(msft->generators->runOnOperation(op, generatorSet)))
135  signalPassFailure();
136  });
137 }
