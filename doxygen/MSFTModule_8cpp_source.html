80  void walkPlacements(
81  py::function pycb,
82  std::tuple<py::object, py::object, py::object, py::object> bounds,
83  py::object prim, py::object walkOrder) {
84 
85  auto handleNone = [](py::object o) {
86  return o.is_none() ? -1 : o.cast<int64_t>();
87  };
88  int64_t cBounds[4] = {
89  handleNone(std::get<0>(bounds)), handleNone(std::get<1>(bounds)),
90  handleNone(std::get<2>(bounds)), handleNone(std::get<3>(bounds))};
91  CirctMSFTPrimitiveType cPrim;
92  if (prim.is_none())
93  cPrim = -1;
94  else
95  cPrim = prim.cast<CirctMSFTPrimitiveType>();
96 
97  CirctMSFTWalkOrder cWalkOrder;
98  if (!walkOrder.is_none())
99  cWalkOrder = walkOrder.cast<CirctMSFTWalkOrder>();
100  else
101  cWalkOrder = CirctMSFTWalkOrder{CirctMSFTDirection::NONE,
102  CirctMSFTDirection::NONE};
103 
104  circtMSFTPlacementDBWalkPlacements(
105  db,
106  [](MlirAttribute loc, MlirOperation locOp, void *userData) {
107  py::gil_scoped_acquire gil;
108  py::function pycb = *((py::function *)(userData));
109  pycb(loc, locOp);
110  },
111  cBounds, cPrim, cWalkOrder, &pycb);
112  }
113 
114 private:
115  CirctMSFTPlacementDB db;
116 };

118 class PyLocationVecIterator {
119 public:
120  /// Get item at the specified position, translating a nullptr to None.
121  static py::handle getItem(MlirAttribute locVec, intptr_t pos) {
122  MlirAttribute loc = circtMSFTLocationVectorAttrGetElement(locVec, pos);
123  if (loc.ptr == nullptr)
124  return py::none();
125  return py::detail::type_caster<MlirAttribute>().cast(
126  loc, py::return_value_policy::automatic, py::handle());
127  }
128 
129  PyLocationVecIterator(MlirAttribute attr) : attr(attr) {}
130  PyLocationVecIterator &dunderIter() { return *this; }
131 
132  py::handle dunderNext() {
133  if (nextIndex >= circtMSFTLocationVectorAttrGetNumElements(attr)) {
134  throw py::stop_iteration();
135  }
136  return getItem(attr, nextIndex++);
137  }
138 
139  static void bind(py::module &m) {
140  py::class_<PyLocationVecIterator>(m, "LocationVectorAttrIterator",
141  py::module_local())
142  .def("__iter__", &PyLocationVecIterator::dunderIter)
143  .def("__next__", &PyLocationVecIterator::dunderNext);
144  }
145 
146 private:
147  MlirAttribute attr;
148  intptr_t nextIndex = 0;
149 };
