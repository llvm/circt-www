121 /// Walk all the placements within 'bounds' ([xmin, xmax, ymin, ymax], inclusive
122 /// on all sides), with -1 meaning unbounded.
123 MLIR_CAPI_EXPORTED void circtMSFTPlacementDBWalkPlacements(
124  CirctMSFTPlacementDB cdb, CirctMSFTPlacementCallback ccb, int64_t bounds[4],
125  CirctMSFTPrimitiveType cPrimTypeFilter, CirctMSFTWalkOrder cWalkOrder,
126  void *userData) {
127 
128  PlacementDB *db = unwrap(cdb);
129  auto cb = [ccb, userData](PhysLocationAttr loc,
130  DynInstDataOpInterface locOp) {
131  ccb(wrap(loc), wrap(locOp), userData);
132  };
133  Optional<PrimitiveType> primTypeFilter;
134  if (cPrimTypeFilter >= 0)
135  primTypeFilter = static_cast<PrimitiveType>(cPrimTypeFilter);
136 
137  Optional<PlacementDB::WalkOrder> walkOrder;
138  if (cWalkOrder.columns != CirctMSFTDirection::NONE ||
139  cWalkOrder.rows != CirctMSFTDirection::NONE)
140  walkOrder = PlacementDB::WalkOrder{
141  static_cast<PlacementDB::Direction>(cWalkOrder.columns),
142  static_cast<PlacementDB::Direction>(cWalkOrder.rows)};
143 
144  db->walkPlacements(
145  cb, std::make_tuple(bounds[0], bounds[1], bounds[2], bounds[3]),
146  primTypeFilter, walkOrder);
147 }
