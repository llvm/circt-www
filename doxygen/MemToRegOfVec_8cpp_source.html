37  void runOnOperation() override {
38  auto circtOp = getOperation();
39  DenseSet<Operation *> dutModuleSet;
40  if (!AnnotationSet::removeAnnotations(circtOp,
41  convertMemToRegOfVecAnnoClass))
42  return;
43  auto *body = circtOp.getBody();
44 
45  // Find the device under test and create a set of all modules underneath it.
46  auto it = llvm::find_if(*body, [&](Operation &op) -> bool {
47  return AnnotationSet(&op).hasAnnotation(dutAnnoClass);
48  });
49  if (it != body->end()) {
50  auto &instanceGraph = getAnalysis<InstanceGraph>();
51  auto *node = instanceGraph.lookup(&(*it));
52  llvm::for_each(llvm::depth_first(node), [&](hw::InstanceGraphNode *node) {
53  dutModuleSet.insert(node->getModule());
54  });
55  }
56 
57  mlir::parallelForEach(circtOp.getContext(), dutModuleSet,
58  [&](Operation *op) {
59  if (auto mod = dyn_cast<FModuleOp>(op))
60  runOnModule(mod);
61  });
62  }

64  void runOnModule(FModuleOp mod) {
65 
66  mod.getBody()->walk([&](MemOp memOp) {
67  LLVM_DEBUG(llvm::dbgs() << "\n Memory op:" << memOp);
68  if (AnnotationSet::removeAnnotations(memOp, excludeMemToRegAnnoClass))
69  return;
70 
71  auto firMem = memOp.getSummary();
72  // Ignore if the memory is candidate for macro replacement.
73  // The requirements for macro replacement:
74  // 1. read latency and write latency of one.
75  // 2. only one readwrite port or write port.
76  // 3. zero or one read port.
77  // 4. undefined read-under-write behavior.
78  if (replSeqMem &&
79  ((firMem.readLatency == 1 && firMem.writeLatency == 1) &&
80  (firMem.numWritePorts + firMem.numReadWritePorts == 1) &&
81  (firMem.numReadPorts <= 1) && firMem.dataWidth > 0))
82  return;
83 
84  generateMemory(memOp, firMem);
85  memOp.erase();
86  });
87  }
88  Value addPipelineStages(ImplicitLocOpBuilder &b, size_t stages, Value clock,
89  Value pipeInput, StringRef name, Value gate = {}) {
90  if (!stages)
91  return pipeInput;
92 
93  while (stages--) {
94  auto reg = b.create<RegOp>(pipeInput.getType(), clock,
95  moduleNamespace.newName(name));
96  if (gate) {
97  b.create<WhenOp>(gate, /*withElseRegion*/ false,
98  [&]() { b.create<StrictConnectOp>(reg, pipeInput); });
99  } else
100  b.create<StrictConnectOp>(reg, pipeInput);
101 
102  pipeInput = reg;
103  }
104 
105  return pipeInput;
106  }

141  void generateRead(const FirMemory &firMem, Value clock, Value addr,
142  Value enable, Value data, Value regOfVec,
143  ImplicitLocOpBuilder &builder) {
144  if (ignoreReadEnable) {
145  // If read enable is ignored, then guard the address update with read
146  // enable.
147  for (size_t j = 0, e = firMem.readLatency; j != e; ++j) {
148  auto enLast = enable;
149  if (j < e - 1)
150  enable = addPipelineStages(builder, 1, clock, enable, "en");
151  addr = addPipelineStages(builder, 1, clock, addr, "addr", enLast);
152  }
153  } else {
154  // Add pipeline stages to respect the read latency. One register for each
155  // latency cycle.
156  enable =
157  addPipelineStages(builder, firMem.readLatency, clock, enable, "en");
158  addr =
159  addPipelineStages(builder, firMem.readLatency, clock, addr, "addr");
160  }
161 
162  // Read the register[address] into a temporary.
163  Value rdata = builder.create<SubaccessOp>(regOfVec, addr);
164  if (!ignoreReadEnable) {
165  // Initialize read data out with invalid.
166  builder.create<StrictConnectOp>(
167  data, builder.create<InvalidValueOp>(data.getType()));
168  // If enable is true, then connect the data read from memory register.
169  builder.create<WhenOp>(enable, /*withElseRegion*/ false, [&]() {
170  builder.create<StrictConnectOp>(data, rdata);
171  });
172  } else {
173  // Ignore read enable signal.
174  builder.create<StrictConnectOp>(data, rdata);
175  }
176  }

178  void generateWrite(const FirMemory &firMem, Value clock, Value addr,
179  Value enable, Value maskBits, Value wdataIn,
180  Value regOfVec, ImplicitLocOpBuilder &builder) {
181 
182  auto numStages = firMem.writeLatency - 1;
183  // Add pipeline stages to respect the write latency. Intermediate registers
184  // for each stage.
185  addr = addPipelineStages(builder, numStages, clock, addr, "addr");
186  enable = addPipelineStages(builder, numStages, clock, enable, "en");
187  wdataIn = addPipelineStages(builder, numStages, clock, wdataIn, "wdata");
188  maskBits = addPipelineStages(builder, numStages, clock, maskBits, "wmask");
189  // Create the register access.
190  auto rdata = builder.create<SubaccessOp>(regOfVec, addr);
191 
192  // The tuple for the access to individual fields of an aggregate data type.
193  // Tuple::<register, data, mask>
194  // The logic:
195  // if (mask)
196  // register = data
197  SmallVector<std::tuple<Value, Value, Value>, 8> loweredRegDataMaskFields;
198 
199  // Write to each aggregate data field is guarded by the corresponding mask
200  // field. This means we have to generate read and write access for each
201  // individual field of the aggregate type.
202  // There are two options to handle this,
203  // 1. FlattenMemory: cast the aggregate data into a UInt and generate
204  // appropriate mask logic.
205  // 2. Create access for each individual field of the aggregate type.
206  // Here we implement the option 2 using getFields.
207  // getFields, creates an access to each individual field of the data and
208  // mask, and the corresponding field into the register. It populates
209  // the loweredRegDataMaskFields vector.
210  // This is similar to what happens in LowerTypes.
211  //
212  if (!getFields(rdata, wdataIn, maskBits, loweredRegDataMaskFields,
213  builder)) {
214  wdataIn.getDefiningOp()->emitOpError(
215  "Cannot convert memory to bank of registers");
216  return;
217  }
218  // If enable:
219  builder.create<WhenOp>(enable, /*withElseRegion*/ false, [&]() {
220  // For each data field. Only one field if not aggregate.
221  for (auto regDataMask : loweredRegDataMaskFields) {
222  auto regField = std::get<0>(regDataMask);
223  auto dataField = std::get<1>(regDataMask);
224  auto maskField = std::get<2>(regDataMask);
225  // If mask, then update the register field.
226  builder.create<WhenOp>(maskField, /*withElseRegion*/ false, [&]() {
227  builder.create<StrictConnectOp>(regField, dataField);
228  });
229  }
230  });
231  }

233  void generateReadWrite(const FirMemory &firMem, Value clock, Value addr,
234  Value enable, Value maskBits, Value wdataIn,
235  Value rdataOut, Value wmode, Value regOfVec,
236  ImplicitLocOpBuilder &builder) {
237 
238  // Add pipeline stages to respect the write latency. Intermediate registers
239  // for each stage. Number of pipeline stages, max of read/write latency.
240  auto numStages = std::max(firMem.readLatency, firMem.writeLatency) - 1;
241  addr = addPipelineStages(builder, numStages, clock, addr, "addr");
242  enable = addPipelineStages(builder, numStages, clock, enable, "en");
243  wdataIn = addPipelineStages(builder, numStages, clock, wdataIn, "wdata");
244  maskBits = addPipelineStages(builder, numStages, clock, maskBits, "wmask");
245 
246  // Read the register[address] into a temporary.
247  Value rdata = builder.create<SubaccessOp>(regOfVec, addr);
248 
249  SmallVector<std::tuple<Value, Value, Value>, 8> loweredRegDataMaskFields;
250  if (!getFields(rdata, wdataIn, maskBits, loweredRegDataMaskFields,
251  builder)) {
252  wdataIn.getDefiningOp()->emitOpError(
253  "Cannot convert memory to bank of registers");
254  return;
255  }
256  // Initialize read data out with invalid.
257  builder.create<StrictConnectOp>(
258  rdataOut, builder.create<InvalidValueOp>(rdataOut.getType()));
259  // If enable:
260  builder.create<WhenOp>(enable, /*withElseRegion*/ false, [&]() {
261  // If write mode:
262  builder.create<WhenOp>(
263  wmode, true,
264  // Write block:
265  [&]() {
266  // For each data field. Only one field if not aggregate.
267  for (auto regDataMask : loweredRegDataMaskFields) {
268  auto regField = std::get<0>(regDataMask);
269  auto dataField = std::get<1>(regDataMask);
270  auto maskField = std::get<2>(regDataMask);
271  // If mask true, then set the field.
272  builder.create<WhenOp>(
273  maskField, /*withElseRegion*/ false, [&]() {
274  builder.create<StrictConnectOp>(regField, dataField);
275  });
276  }
277  },
278  // Read block:
279  [&]() { builder.create<StrictConnectOp>(rdataOut, rdata); });
280  });
281  }

283  // Generate individual field accesses for an aggregate type. Return false if
284  // it fails. Which can happen if invalid fields are present of the mask and
285  // input types donot match. The assumption is that, \p reg and \p input have
286  // exactly the same type. And \p mask has the same bundle fields, but each
287  // field is of type UInt<1> So, populate the \p results with each field
288  // access. For example, the first entry should be access to first field of \p
289  // reg, first field of \p input and first field of \p mask.
290  bool getFields(Value reg, Value input, Value mask,
291  SmallVectorImpl<std::tuple<Value, Value, Value>> &results,
292  ImplicitLocOpBuilder &builder) {
293 
294  // Check if the number of fields of mask and input type match.
295  auto isValidMask = [&](FIRRTLType inType, FIRRTLType maskType) -> bool {
296  if (auto bundle = inType.dyn_cast<BundleType>()) {
297  if (auto mBundle = maskType.dyn_cast<BundleType>())
298  return mBundle.getNumElements() == bundle.getNumElements();
299  } else if (auto vec = inType.dyn_cast<FVectorType>()) {
300  if (auto mVec = maskType.dyn_cast<FVectorType>())
301  return mVec.getNumElements() == vec.getNumElements();
302  } else
303  return true;
304  return false;
305  };
306 
307  std::function<bool(Value, Value, Value)> flatAccess =
308  [&](Value reg, Value input, Value mask) -> bool {
309  FIRRTLType inType = input.getType().cast<FIRRTLType>();
310  if (!isValidMask(inType, mask.getType().cast<FIRRTLType>())) {
311  input.getDefiningOp()->emitOpError("Mask type is not valid");
312  return false;
313  }
314  return TypeSwitch<FIRRTLType, bool>(inType)
315  .Case<BundleType>([&](BundleType bundle) {
316  for (size_t i = 0, e = bundle.getNumElements(); i != e; ++i) {
317  auto regField = builder.create<SubfieldOp>(reg, i);
318  auto inputField = builder.create<SubfieldOp>(input, i);
319  auto maskField = builder.create<SubfieldOp>(mask, i);
320  if (!flatAccess(regField, inputField, maskField))
321  return false;
322  }
323  return true;
324  })
325  .Case<FVectorType>([&](auto vector) {
326  for (size_t i = 0, e = vector.getNumElements(); i != e; ++i) {
327  auto regField = builder.create<SubindexOp>(reg, i);
328  auto inputField = builder.create<SubindexOp>(input, i);
329  auto maskField = builder.create<SubindexOp>(mask, i);
330  if (!flatAccess(regField, inputField, maskField))
331  return false;
332  }
333  return true;
334  })
335  .Case<IntType>([&](auto iType) {
336  results.push_back({reg, input, mask});
337  return iType.getWidth().hasValue();
338  })
339  .Default([&](auto) { return false; });
340  };
341  if (flatAccess(reg, input, mask))
342  return true;
343  return false;
344  }

346  void scatterMemTapAnno(RegOp op, ArrayAttr attr,
347  ImplicitLocOpBuilder &builder) {
348  AnnotationSet annos(attr);
349  SmallVector<Attribute> regAnnotations;
350  auto vecType = op.getType().cast<FVectorType>();
351  for (auto anno : annos) {
352  if (anno.isClass(memTapSourceClass)) {
353  for (size_t i = 0,
354  e = op.getType().cast<FVectorType>().getNumElements();
355  i != e; ++i) {
356  NamedAttrList newAnno;
357  newAnno.append("class", anno.getMember("class"));
358  newAnno.append("circt.fieldID",
359  builder.getI64IntegerAttr(vecType.getFieldID(i)));
360  newAnno.append("id", anno.getMember("id"));
361  if (auto nla = anno.getMember("circt.nonlocal"))
362  newAnno.append("circt.nonlocal", nla);
363  newAnno.append(
364  "portID",
365  IntegerAttr::get(IntegerType::get(builder.getContext(), 64), i));
366 
367  regAnnotations.push_back(builder.getDictionaryAttr(newAnno));
368  }
369  } else
370  regAnnotations.push_back(anno.getAttr());
371  }
372  op.annotationsAttr(builder.getArrayAttr(regAnnotations));
373  }

375  /// Generate the logic for implementing the memory using Registers.
376  void generateMemory(MemOp memOp, FirMemory &firMem) {
377  ImplicitLocOpBuilder builder(memOp.getLoc(), memOp);
378  moduleNamespace.add(memOp->getParentOfType<FModuleOp>());
379  auto dataType = memOp.getDataType();
380 
381  auto innerSym = memOp.inner_sym();
382 
383  RegOp regOfVec = {};
384  for (size_t index = 0, rend = memOp.getNumResults(); index < rend;
385  ++index) {
386  auto result = memOp.getResult(index);
387  // Create a temporary wire to replace the memory port. This makes it
388  // simpler to delete the memOp.
389  auto wire = builder.create<WireOp>(
390  result.getType(),
391  (memOp.name() + "_" + memOp.getPortName(index).getValue()).str(),
392  memOp.nameKind());
393  result.replaceAllUsesWith(wire.getResult());
394  result = wire;
395  // Create an access to all the common subfields.
396  auto adr = getAddr(builder, result);
397  auto enb = getEnable(builder, result);
398  auto clk = getClock(builder, result);
399  auto dta = getData(builder, result);
400  // IF the register is not yet created.
401  if (!regOfVec) {
402  // Create the register corresponding to the memory.
403  regOfVec = builder.create<RegOp>(
404  FVectorType::get(dataType, firMem.depth), clk, memOp.nameAttr());
405 
406  // Copy all the memory annotations.
407  if (!memOp.annotationsAttr().empty())
408  scatterMemTapAnno(regOfVec, memOp.annotationsAttr(), builder);
409  if (innerSym)
410  regOfVec.inner_symAttr(memOp.inner_symAttr());
411  }
412  auto portKind = memOp.getPortKind(index);
413  if (portKind == MemOp::PortKind::Read) {
414  generateRead(firMem, clk, adr, enb, dta, regOfVec, builder);
415  } else if (portKind == MemOp::PortKind::Write) {
416  auto mask = getMask(builder, result);
417  generateWrite(firMem, clk, adr, enb, mask, dta, regOfVec, builder);
418  } else {
419  auto wmode = getWmode(builder, result);
420  auto wDta = getData(builder, result, true);
421  auto mask = getMask(builder, result);
422  generateReadWrite(firMem, clk, adr, enb, mask, wDta, dta, wmode,
423  regOfVec, builder);
424  }
425  }
426  }
