35 /// Parse a function result list.
36 ///
37 /// function-result-list ::= function-result-list-parens
38 /// function-result-list-parens ::= `(` `)`
39 /// | `(` function-result-list-no-parens `)`
40 /// function-result-list-no-parens ::= function-result (`,` function-result)*
41 /// function-result ::= (percent-identifier `:`) type attribute-dict?
42 ///
43 ParseResult module_like_impl::parseFunctionResultList(
44  OpAsmParser &parser, SmallVectorImpl<Type> &resultTypes,
45  SmallVectorImpl<DictionaryAttr> &resultAttrs,
46  SmallVectorImpl<Attribute> &resultNames) {
47 
48  auto parseElt = [&]() -> ParseResult {
49  std::string portName;
50  if (parser.parseKeywordOrString(&portName))
51  return failure();
52  resultNames.push_back(StringAttr::get(parser.getContext(), portName));
53 
54  resultTypes.emplace_back();
55  resultAttrs.emplace_back();
56 
57  NamedAttrList attrs;
58  if (parser.parseColonType(resultTypes.back()) ||
59  parser.parseOptionalAttrDict(attrs))
60  return failure();
61 
62  resultAttrs.back() = attrs.getDictionary(parser.getContext());
63 
64  return success();
65  };
66 
67  return parser.parseCommaSeparatedList(OpAsmParser::Delimiter::Paren,
68  parseElt);
69 }
