60 ParseResult ConstantOp::parse(OpAsmParser &parser, OperationState &result) {
61  // Parse the constant value without bit width.
62  APInt value;
63  auto valueLoc = parser.getCurrentLocation();
64 
65  if (parser.parseInteger(value) ||
66  parser.parseOptionalAttrDict(result.attributes) || parser.parseColon())
67  return failure();
68 
69  // Parse the result type..
70  UnpackedType type;
71  auto typeLoc = parser.getCurrentLocation();
72  if (parser.parseType(type))
73  return failure();
74 
75  // Ensure that the result type is a simple bit vector type.
76  auto sbvt = type.getSimpleBitVectorOrNull();
77  if (!sbvt)
78  return parser.emitError(typeLoc, "expected simple bit vector type");
79 
80  // Extend or truncate the constant value to match the size of the type.
81  if (sbvt.size > value.getBitWidth()) {
82  // sext is always safe here, even for unsigned values, because the
83  // parseOptionalInteger method will return something with a zero in the
84  // top bits if it is a positive number.
85  value = value.sext(sbvt.size);
86  } else if (sbvt.size < value.getBitWidth()) {
87  // The parser can return an unnecessarily wide result with leading
88  // zeros. This isn't a problem, but truncating off bits is bad.
89  unsigned neededBits =
90  value.isNegative() ? value.getSignificantBits() : value.getActiveBits();
91  if (sbvt.size < neededBits)
92  return parser.emitError(valueLoc,
93  "constant out of range for result type ")
94  << type;
95  value = value.trunc(sbvt.size);
96  }
97 
98  // Build the attribute and op.
99  auto attrType = IntegerType::get(parser.getContext(), sbvt.size);
100  auto attrValue = IntegerAttr::get(attrType, value);
101 
102  result.addAttribute("value", attrValue);
103  result.addTypes(type);
104  return success();
105 }

125 /// This builder allows construction of small signed integers like 0, 1, -1

142  DictionaryAttr attrs, mlir::OpaqueProperties properties,

151  }
152  results.push_back(
