31 /// Returns the passed value if the integer width is already correct.
32 /// Zero-extends if it is too narrow.
33 /// Truncates if the integer is too wide and the truncated part is zero, if it
34 /// is not zero it returns the max value integer of target-width.
35 static Value adjustIntegerWidth(OpBuilder &builder, Value value,
36  uint32_t targetWidth, Location loc) {
37  uint32_t intWidth = value.getType().getIntOrFloatBitWidth();
38  if (intWidth == targetWidth)
39  return value;
40 
41  if (intWidth < targetWidth) {
42  Value zeroExt = builder.create<hw::ConstantOp>(
43  loc, builder.getIntegerType(targetWidth - intWidth), 0);
44  return builder.create<comb::ConcatOp>(loc, ValueRange{zeroExt, value});
45  }
46 
47  Value hi = builder.create<comb::ExtractOp>(loc, value, targetWidth,
48  intWidth - targetWidth);
49  Value zero = builder.create<hw::ConstantOp>(
50  loc, builder.getIntegerType(intWidth - targetWidth), 0);
51  Value isZero = builder.create<comb::ICmpOp>(loc, comb::ICmpPredicate::eq, hi,
52  zero, false);
53  Value lo = builder.create<comb::ExtractOp>(loc, value, 0, targetWidth);
54  Value max = builder.create<hw::ConstantOp>(
55  loc, builder.getIntegerType(targetWidth), -1);
56  return builder.create<comb::MuxOp>(loc, isZero, lo, max, false);
57 }

148  if (typeConverter->convertTypes(op.getResultTypes(), convResTypes).failed())
149  return failure();
150 
151  // Drop the cast if the operand and result types agree after type
152  // conversion.

154  rewriter.replaceOp(op, adaptor.getOperands());
155  return success();

235 static void addGenericLegality(ConversionTarget &target) {
236  target.addDynamicallyLegalOp<Op>([](Op op) {
237  return !hasMooreType(op->getOperands()) && !hasMooreType(op->getResults());
238  });

241 static void populateLegality(ConversionTarget &target) {
242  target.addIllegalDialect<MooreDialect>();
243  target.addLegalDialect<mlir::BuiltinDialect>();
244  target.addLegalDialect<hw::HWDialect>();
245  target.addLegalDialect<llhd::LLHDDialect>();
246  target.addLegalDialect<comb::CombDialect>();
247 
248  addGenericLegality<cf::CondBranchOp>(target);
249  addGenericLegality<cf::BranchOp>(target);
250  addGenericLegality<func::CallOp>(target);
251  addGenericLegality<func::ReturnOp>(target);
252  addGenericLegality<UnrealizedConversionCastOp>(target);
253 
254  target.addDynamicallyLegalOp<func::FuncOp>([](func::FuncOp op) {
255  auto argsConverted = llvm::none_of(op.getBlocks(), [](auto &block) {
256  return hasMooreType(block.getArguments());
257  });
258  auto resultsConverted = !hasMooreType(op.getResultTypes());
259  return argsConverted && resultsConverted;
260  });
261 }

263 static void populateTypeConversion(TypeConverter &typeConverter) {
264  typeConverter.addConversion([&](IntType type) {
265  return mlir::IntegerType::get(type.getContext(), type.getBitSize());
266  });
267 
268  // Directly map simple bit vector types to a compact integer type. This needs
269  // to be added after all of the other conversions above, such that SBVs
270  // conversion gets tried first before any of the others.
271  typeConverter.addConversion([&](UnpackedType type) -> std::optional<Type> {
272  if (auto sbv = type.getSimpleBitVectorOrNull())
273  return mlir::IntegerType::get(type.getContext(), sbv.size);
274  return std::nullopt;
275  });
276 
277  // Valid target types.
278  typeConverter.addConversion([](mlir::IntegerType type) { return type; });
279 }

317 /// This is the main entrypoint for the Moore to Core conversion pass.
318 void MooreToCorePass::runOnOperation() {
319  MLIRContext &context = getContext();
320  ModuleOp module = getOperation();
321 
322  ConversionTarget target(context);
323  TypeConverter typeConverter;
324  RewritePatternSet patterns(&context);
325  populateLegality(target);
326  populateTypeConversion(typeConverter);
327  populateOpConversion(patterns, typeConverter);
328 
329  if (failed(applyFullConversion(module, target, std::move(patterns))))
330  signalPassFailure();
331 }
