72 PackedType SimpleBitVectorType::getType(MLIRContext *context) const {
73  if (!*this)
74  return {};
75  Optional<Sign> maybeSign;
76  if (explicitSign)
77  maybeSign = sign;
78 
79  // If the type originally used an integer atom, try to reconstruct that.
80  if (usedAtom)
81  if (auto kind = IntType::getKindFromDomainAndSize(domain, size))
82  return IntType::get(context, *kind, maybeSign);
83 
84  // Build the core integer bit type.
85  auto kind = domain == Domain::TwoValued ? IntType::Bit : IntType::Logic;
86  auto intType = IntType::get(context, kind, maybeSign);
87 
88  // If the vector is wider than a single bit, or the dimension was explicit in
89  // the original type, add a dimension around the bit type.
90  if (size > 1 || explicitSize)
91  return PackedRangeDim::get(intType, size);
92  return intType;
93 }

160 SimpleBitVectorType UnpackedType::getSimpleBitVectorOrNull() const {
161  return TypeSwitch<UnpackedType, SimpleBitVectorType>(fullyResolved())
162  .Case<IntType>([](auto type) {
163  // Integer types trivially map to SBVTs.
164  return getSimpleBitVectorFromIntType(type);
165  })
166  .Case<PackedRangeDim>([](auto rangeType) {
167  // Inner type must be an integer.
168  auto innerType =
169  rangeType.getInner().fullyResolved().template dyn_cast<IntType>();
170  if (!innerType)
171  return SimpleBitVectorType{};
172 
173  // Inner type must be a single-bit integer. Cannot have integer atom
174  // vectors like `int [31:0]`.
175  auto sbv = getSimpleBitVectorFromIntType(innerType);
176  if (sbv.usedAtom)
177  return SimpleBitVectorType{};
178 
179  // Range must be have non-zero size, and go downwards to zero.
180  auto range = rangeType.getRange();
181  if (range.size == 0 || range.offset != 0 || range.dir != RangeDir::Down)
182  return SimpleBitVectorType{};
183  sbv.explicitSize = true;
184  sbv.size = range.size;
185  return sbv;
186  })
187  .Default([](auto) { return SimpleBitVectorType{}; });
188 }

190 SimpleBitVectorType UnpackedType::castToSimpleBitVectorOrNull() const {
191  // If the type is already a valid SBVT, return that immediately without
192  // casting.
193  if (auto sbv = getSimpleBitVectorOrNull())
194  return sbv;
195 
196  // All packed types with a known size (i.e., with no `[]` dimensions) can be
197  // cast to an SBVT.
198  auto packed = fullyResolved().dyn_cast<PackedType>();
199  if (!packed)
200  return {};
201  auto bitSize = packed.getBitSize();
202  if (!bitSize || *bitSize == 0)
203  return {};
204 
205  return SimpleBitVectorType(packed.getDomain(), packed.getSign(), *bitSize,
206  /*usedAtom=*/false, /*explicitSign=*/false,
207  /*explicitSize=*/false);
208 }

210 void UnpackedType::format(
211  llvm::raw_ostream &os,
212  llvm::function_ref<void(llvm::raw_ostream &os)> around) const {
213  TypeSwitch<UnpackedType>(*this)
214  .Case<StringType>([&](auto) { os << "string"; })
215  .Case<ChandleType>([&](auto) { os << "chandle"; })
216  .Case<EventType>([&](auto) { os << "event"; })
217  .Case<RealType>([&](auto type) { os << type.getKeyword(); })
218  .Case<PackedType, UnpackedStructType>([&](auto type) { type.format(os); })
219  .Case<UnpackedDim>([&](auto type) { type.format(os, around); })
220  .Case<UnpackedNamedType>(
221  [&](auto type) { os << type.getName().getValue(); })
222  .Case<UnpackedRefType>(
223  [&](auto type) { os << "type(" << type.getInner() << ")"; })
224  .Default([](auto) { llvm_unreachable("all types should be handled"); });
225 
226  // In case there were no unpacked dimensions, the `around` function was never
227  // called. However, callers expect us to be able to format things like `bit
228  // [7:0] fieldName`, where `fieldName` would be printed by `around`. So in
229  // case `around` is non-null, but no unpacked dimension had a chance to print
230  // it, simply print it now.
231  if (!isa<UnpackedDim>() && around) {
232  os << " ";
233  around(os);
234  }
235 }

326 struct IntTypeStorage : TypeStorage {
327  using KeyTy = unsigned;
328  using Kind = IntType::Kind;

330  IntTypeStorage(KeyTy key)

672  /// `finalize` method constructs the these resolved types by resolving the
673  /// inner type appropriately and wrapping it in the dimension type. These
674  /// wrapped types, which are equivalent to the `dim` itself but with the inner
675  /// type resolved, are passed to `DimStorage::mutate` which fills in the
676  /// `resolved` and `fullyResolved` fields behind a storage lock in the
677  /// MLIRContext.
678  ///
679  /// This has been inspired by https://reviews.llvm.org/D84171.
680  template <class ConcreteDim, typename... Args>
681  void finalize(ConcreteDim dim, Args... args) const {
682  if (resolved && fullyResolved)
683  return;
684  auto inner = dim.getInner();
685  auto newResolved = dim;
686  auto newFullyResolved = dim;
687  if (inner != inner.resolved())
688  newResolved = ConcreteDim::get(inner.resolved(), args...);
689  if (inner != inner.fullyResolved())
690  newFullyResolved = ConcreteDim::get(inner.fullyResolved(), args...);
691  auto result = dim.mutate(newResolved, newFullyResolved);
692  (void)result; // Supress warning
693  assert(succeeded(result));
694  }
695 
696  UnpackedType inner;
697  UnpackedType resolved;
698  UnpackedType fullyResolved;
699 };

1031  loc(loc) {

1033  // Otherwise it is four-valued.

1035  [](auto &member) {

1046  *bitSize += *memberSize;

1162 /// Parse a type with custom syntax.
1163 static OptionalParseResult customTypeParser(DialectAsmParser &parser,
1164  StringRef mnemonic, Subset subset,
1165  llvm::SMLoc loc, Type &type) {
1166  auto *context = parser.getContext();
1167 
1168  auto yieldPacked = [&](PackedType x) {
1169  type = x;
1170  return success();
1171  };
1172  auto yieldUnpacked = [&](UnpackedType x) {
1173  if (!subset.allowUnpacked) {
1174  parser.emitError(loc)
1175  << "unpacked type " << x << " where only packed types are allowed";
1176  return failure();
1177  }
1178  type = x;
1179  return success();
1180  };
1181  auto yieldImplied =
1182  [&](llvm::function_ref<PackedType()> ifPacked,
1183  llvm::function_ref<UnpackedType()> ifUnpacked) {
1184  if (subset.implied == Subset::Packed)
1185  return yieldPacked(ifPacked());
1186  if (subset.implied == Subset::Unpacked)
1187  return yieldUnpacked(ifUnpacked());
1188  parser.emitError(loc)
1189  << "ambiguous packing; wrap `" << mnemonic
1190  << "` in `packed<...>` or `unpacked<...>` to disambiguate";
1191  return failure();
1192  };
1193 
1194  // Explicit packing indicators, like `unpacked.named`.
1195  if (mnemonic == "unpacked") {
1196  UnpackedType inner;
1197  if (parser.parseLess() ||
1198  parseMooreType(parser, {Subset::Unpacked, true}, inner) ||
1199  parser.parseGreater())
1200  return failure();
1201  return yieldUnpacked(inner);
1202  }
1203  if (mnemonic == "packed") {
1204  PackedType inner;
1205  if (parser.parseLess() ||
1206  parseMooreType(parser, {Subset::Packed, false}, inner) ||
1207  parser.parseGreater())
1208  return failure();
1209  return yieldPacked(inner);
1210  }
1211 
1212  // Packed primary types.
1213  if (mnemonic == "void")
1214  return yieldPacked(VoidType::get(context));
1215 
1216  if (auto kind = IntType::getKindFromKeyword(mnemonic)) {
1217  Optional<Sign> sign;
1218  if (succeeded(parser.parseOptionalLess())) {
1219  StringRef signKeyword;
1220  if (parser.parseKeyword(&signKeyword) || parser.parseGreater())
1221  return failure();
1222  sign = getSignFromKeyword(signKeyword);
1223  if (!sign) {
1224  parser.emitError(parser.getCurrentLocation())
1225  << "expected keyword `unsigned` or `signed`";
1226  return failure();
1227  }
1228  }
1229  return yieldPacked(IntType::get(context, *kind, sign));
1230  }
1231 
1232  // Unpacked primary types.
1233  if (mnemonic == "string")
1234  return yieldUnpacked(StringType::get(context));
1235  if (mnemonic == "chandle")
1236  return yieldUnpacked(ChandleType::get(context));
1237  if (mnemonic == "event")
1238  return yieldUnpacked(EventType::get(context));
1239  if (auto kind = RealType::getKindFromKeyword(mnemonic))
1240  return yieldUnpacked(RealType::get(context, *kind));
1241 
1242  // Enums
1243  if (mnemonic == "enum") {
1244  if (parser.parseLess())
1245  return failure();
1246  StringAttr name;
1247  auto result = parser.parseOptionalAttribute(name);
1248  if (result.hasValue())
1249  if (*result || parser.parseComma())
1250  return failure();
1251  LocationAttr loc;
1252  PackedType base;
1253  result = parser.parseOptionalAttribute(loc);
1254  if (result.hasValue()) {
1255  if (*result)
1256  return failure();
1257  } else {
1258  if (parseMooreType(parser, {Subset::Packed, false}, base) ||
1259  parser.parseComma() || parser.parseAttribute(loc))
1260  return failure();
1261  }
1262  if (parser.parseGreater())
1263  return failure();
1264  return yieldPacked(EnumType::get(name, loc, base));
1265  }
1266 
1267  // Everything that follows can be packed or unpacked. The packing is inferred
1268  // from the last `packed<...>` or `unpacked<...>` that we've seen. The
1269  // `yieldImplied` function will call the first lambda to construct a packed
1270  // type, or the second lambda to construct an unpacked type. If the
1271  // `subset.implied` field is not set, which means there hasn't been any prior
1272  // `packed` or `unpacked`, the function will emit an error properly.
1273 
1274  // Packed and unpacked type indirections.
1275  if (mnemonic == "named") {
1276  UnpackedType inner;
1277  StringAttr name;
1278  LocationAttr loc;
1279  if (parser.parseLess() || parser.parseAttribute(name) ||
1280  parser.parseComma() || parseMooreType(parser, subset, inner) ||
1281  parser.parseComma() || parser.parseAttribute(loc) ||
1282  parser.parseGreater())
1283  return failure();
1284  return yieldImplied(
1285  [&]() {
1286  return PackedNamedType::get(inner.cast<PackedType>(), name, loc);
1287  },
1288  [&]() { return UnpackedNamedType::get(inner, name, loc); });
1289  }
1290  if (mnemonic == "ref") {
1291  UnpackedType inner;
1292  LocationAttr loc;
1293  if (parser.parseLess() || parseMooreType(parser, subset, inner) ||
1294  parser.parseComma() || parser.parseAttribute(loc) ||
1295  parser.parseGreater())
1296  return failure();
1297  return yieldImplied(
1298  [&]() { return PackedRefType::get(inner.cast<PackedType>(), loc); },
1299  [&]() { return UnpackedRefType::get(inner, loc); });
1300  }
1301 
1302  // Packed and unpacked ranges.
1303  if (mnemonic == "unsized") {
1304  UnpackedType inner;
1305  if (parser.parseLess() || parseMooreType(parser, subset, inner) ||
1306  parser.parseGreater())
1307  return failure();
1308  return yieldImplied(
1309  [&]() { return PackedUnsizedDim::get(inner.cast<PackedType>()); },
1310  [&]() { return UnpackedUnsizedDim::get(inner); });
1311  }
1312  if (mnemonic == "range") {
1313  UnpackedType inner;
1314  int left, right;
1315  if (parser.parseLess() || parseMooreType(parser, subset, inner) ||
1316  parser.parseComma() || parser.parseInteger(left) ||
1317  parser.parseColon() || parser.parseInteger(right) ||
1318  parser.parseGreater())
1319  return failure();
1320  return yieldImplied(
1321  [&]() {
1322  return PackedRangeDim::get(inner.cast<PackedType>(), left, right);
1323  },
