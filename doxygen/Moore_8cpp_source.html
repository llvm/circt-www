29 static Optional<Sign> convertMooreSign(enum MooreSign sign) {
30  switch (sign) {
31  case MooreSign::MooreSigned:
32  return Sign::Signed;
33  case MooreSign::MooreUnsigned:
34  return Sign::Unsigned;
35  case MooreSign::MooreNone:
36  return {};
37  }
38  llvm_unreachable("All cases should be covered.");
39 }

41 static IntType::Kind convertMooreIntKind(enum MooreIntKind kind) {
42  switch (kind) {
43  case MooreIntKind::MooreBit:
44  return IntType::Kind::Bit;
45  case MooreIntKind::MooreLogic:
46  return IntType::Kind::Logic;
47  case MooreIntKind::MooreReg:
48  return IntType::Kind::Reg;

50  case MooreIntKind::MooreByte:

54  case MooreIntKind::MooreInt:
55  return IntType::Kind::Int;
56  case MooreIntKind::MooreLongInt:
57  return IntType::Kind::LongInt;
58  case MooreIntKind::MooreInteger:
59  return IntType::Kind::Integer;
60  case MooreIntKind::MooreTime:
