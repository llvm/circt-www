24 /// This table tracks nlas and what modules participate in them.
25 ///
26 /// To use this class, retrieve a cached copy from the analysis manager:
27 /// auto &nlaTable = getAnalysis<NLATable>(getOperation());
28 class NLATable {
29 
30 public:
31  /// Create a new NLA table of a circuit. This must be called on a FIRRTL
32  /// CircuitOp or MLIR ModuleOp. To ensure that the analysis does not return
33  /// stale data while a pass is running, it should be kept up-to-date when
34  /// modules are added or renamed and NLAs are updated.
35  explicit NLATable(Operation *operation);
36 
37  /// Lookup all NLAs an operation participates in. This returns a reference to
38  /// the internal record, so make a copy before making any update to the
39  /// NLATable.
40  ArrayRef<HierPathOp> lookup(Operation *op);
41 
42  /// Lookup all NLAs an operation participates in. This returns a reference to
43  /// the internal record, so make a copy before making any update to the
44  /// NLATable.
45  ArrayRef<HierPathOp> lookup(StringAttr name);
46 
47  /// Resolve a symbol to an NLA.
48  HierPathOp getNLA(StringAttr name);
49 
50  /// Resolve a symbol to a Module.
51  FModuleLike getModule(StringAttr name);
52 
53  /// Compute the NLAs that are common between the two modules, `mod1` and
54  /// `mod2` and insert them into the set `common`.
55  /// The set of NLAs that an instance op participates in is the set of common
56  /// NLAs between the parent module and the instance target. This can be used
57  /// to get the set of NLAs that an InstanceOp participates in, instead of
58  /// recording them on the op in the IR.
59  void commonNLAs(StringAttr mod1, StringAttr mod2,
60  DenseSet<HierPathOp> &common) {
61  auto mod1NLAs = lookup(mod1);
62  auto mod2NLAs = lookup(mod2);
63  common.insert(mod1NLAs.begin(), mod1NLAs.end());
64  DenseSet<HierPathOp> set2(mod2NLAs.begin(), mod2NLAs.end());
65  llvm::set_intersect(common, set2);
66  }
67 
68  /// Get the NLAs that the InstanceOp participates in, insert it to the
69  /// DenseSet `nlas`.
70  void getInstanceNLAs(InstanceOp inst, DenseSet<HierPathOp> &nlas) {
71  auto instSym = getInnerSymName(inst);
72  // If there is no inner sym on the InstanceOp, then it does not participate
73  // in any NLA.
74  if (!instSym)
75  return;
76  auto mod = inst->getParentOfType<FModuleOp>().getNameAttr();
77  // Get the NLAs that are common between the parent module and the target
78  // module. This should contain the NLAs that this InstanceOp participates
79  // in.
80  commonNLAs(inst->getParentOfType<FModuleOp>().getNameAttr(),
81  inst.moduleNameAttr().getAttr(), nlas);
82  // Handle the case when there are more than one Instances for the same
83  // target module. Getting the `commonNLA`, in that case is not enough,
84  // remove the NLAs that donot have the InstanceOp as the innerSym.
85  for (auto nla : llvm::make_early_inc_range(nlas)) {
86  if (!nla.hasInnerSym(mod, instSym))
87  nlas.erase(nla);
88  }
89  }
90 
91  /// Get the NLAs that the module `modName` particiaptes in, and insert them
92  /// into the DenseSet `nlas`.
93  void getNLAsInModule(StringAttr modName, DenseSet<HierPathOp> &nlas) {
94  for (auto nla : lookup(modName))
95  nlas.insert(nla);
96  }

98  //===-------------------------------------------------------------------------
99  // Methods to keep an NLATable up to date.
100  //
101  // These methods are not thread safe. Make sure that modifications are
102  // properly synchronized or performed in a serial context. When the
103  // NLATable is used as an analysis, this is only safe when the pass is
104  // on a CircuitOp.
105 
106  /// Insert a new NLA. This updates two internal records,
107  /// 1. Update the map for the `nlaOp` name to the Operation.
108  /// 2. For each module in the NLA namepath, insert the NLA into the list of
109  /// HierPathOps that participate in the corresponding module. This does
110  /// not update the module name to module op map, if any potentially new module
111  /// in the namepath does not already exist in the record.
112  void addNLA(HierPathOp nla);
113 
114  /// Remove the NLA from the analysis. This updates two internal records,
115  /// 1. Remove the NLA name to the operation map entry.
116  /// 2. For each module in the namepath of the NLA, remove the entry from the
117  /// list of NLAs that the module participates in.
118  /// Note that this invalidates any reference to the NLA list returned by
119  /// 'lookup'.
120  void erase(HierPathOp nlaOp, SymbolTable *symbolTable = nullptr);
121 
122  /// Record a new FModuleLike operation. This updates the Module name to Module
123  /// operation map.
124  void addModule(FModuleLike mod) { symToOp[mod.moduleNameAttr()] = mod; }
125 
126  /// Stop tracking a module. Remove the module from two internal records,
127  /// 1. Module name to Module op map.
128  /// 2. Module name to list of NLAs that the module participates in.
129  void eraseModule(StringAttr name) {
130  symToOp.erase(name);
131  nodeMap.erase(name);
132  }
133 
134  /// Replace the module `oldModule` with `newModule` in the namepath of the nla
135  /// `nlaName`. This moves the nla from from the list of `oldModule` to
136  /// `newModule`. Move `nlaName` from the list of NLAs that `oldModule`
137  /// participates in to `newModule`. This can delete and invalidate any
138  /// reference returned by `lookup`.
139  void updateModuleInNLA(StringAttr nlaName, StringAttr oldModule,
140  StringAttr newModule);
141 
142  /// Replace the module `oldModule` with `newModule` in the namepath of the nla
143  /// `nlaName`. This moves the nla from from the list of `oldModule` to
144  /// `newModule`. Move `nlaName` from the list of NLAs that `oldModule`
145  /// participates in to `newModule`. This can delete and invalidate any
146  /// reference returned by `lookup`.
147  void updateModuleInNLA(HierPathOp nlaOp, StringAttr oldModule,
148  StringAttr newModule);
149 
150  /// Rename a module, this updates the name to module tracking and the name to
151  /// NLA tracking. This moves all the NLAs that `oldModName` is participating
152  /// in to the `newModName`. The `oldModName` must exist in the name to module
153  /// record. This also removes all the entries for `oldModName`.
154  void renameModule(StringAttr oldModName, StringAttr newModName);
155 
156  /// Replace the module `oldModName` with `newModName` in the namepath of any
157  /// NLA. Since the module is being updated, the symbols inside the module
158  /// should also be renamed. Use the rename map `innerSymRenameMap` to update
159  /// the inner_sym names in the namepath.
160  void renameModuleAndInnerRef(
161  StringAttr newModName, StringAttr oldModName,
162  const DenseMap<StringAttr, StringAttr> &innerSymRenameMap);
163 
164  /// Remove the NLA from the Module. This updates the module name to NLA
165  /// tracking.
166  void removeNLAfromModule(HierPathOp nla, StringAttr mod) {
167  llvm::erase_value(nodeMap[mod], nla);
168  }
169 
170  /// Remove all the nlas in the set `nlas` from the module. This updates the
171  /// module name to NLA tracking.
172  void removeNLAsfromModule(const DenseSet<HierPathOp> &nlas, StringAttr mod) {
173  llvm::erase_if(nodeMap[mod],
174  [&nlas](const auto &nla) { return nlas.count(nla); });
175  }
176 
177  /// Add the nla to the module. This ensures that the list of NLAs that the
178  /// module participates in is updated. This will be required if `mod` is added
179  /// to the namepath of `nla`.
180  void addNLAtoModule(HierPathOp nla, StringAttr mod) {
181  nodeMap[mod].push_back(nla);
182  }
183 
184 private:
185  NLATable(const NLATable &) = delete;
186 
187  /// Map modules to the NLA's that target them.
188  llvm::DenseMap<StringAttr, SmallVector<HierPathOp, 4>> nodeMap;
189 
190  /// Map symbol names to module and NLA operations.
191  llvm::DenseMap<StringAttr, Operation *> symToOp;
192 };
