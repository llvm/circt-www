31 /// Iterates over the handshake::FuncOp's in the program to build an instance
32 /// graph. In doing so, we detect whether there are any cycles in this graph, as
33 /// well as infer a top module for the design by performing a topological sort
34 /// of the instance graph. The result of this sort is placed in sortedFuncs.
35 LogicalResult resolveInstanceGraph(ModuleOp moduleOp,
36  InstanceGraph &instanceGraph,
37  std::string &topLevel,
38  SmallVectorImpl<std::string> &sortedFuncs) {
39  // Create use graph
40  auto walkFuncOps = [&](handshake::FuncOp funcOp) {
41  auto &funcUses = instanceGraph[funcOp.getName().str()];
42  funcOp.walk([&](handshake::InstanceOp instanceOp) {
43  funcUses.insert(instanceOp.getModule().str());
44  });
45  };
46  moduleOp.walk(walkFuncOps);
47 
48  // find top-level (and cycles) using a topological sort. Initialize all
49  // instances as candidate top level modules; these will be pruned whenever
50  // they are referenced by another module.
51  std::set<std::string> visited, marked, candidateTopLevel;
52  SmallVector<std::string> cycleTrace;
53  bool cyclic = false;
54  llvm::transform(instanceGraph,
55  std::inserter(candidateTopLevel, candidateTopLevel.begin()),
56  [](auto it) { return it.first; });
57  std::function<void(const std::string &, SmallVector<std::string>)> cycleUtil =
58  [&](const std::string &node, SmallVector<std::string> trace) {
59  if (cyclic || visited.count(node))
60  return;
61  trace.push_back(node);
62  if (marked.count(node)) {
63  cyclic = true;
64  cycleTrace = trace;
65  return;
66  }
67  marked.insert(node);
68  for (auto use : instanceGraph[node]) {
69  candidateTopLevel.erase(use);
70  cycleUtil(use, trace);
71  }
72  marked.erase(node);
73  visited.insert(node);
74  sortedFuncs.insert(sortedFuncs.begin(), node);
75  };
76  for (auto it : instanceGraph) {
77  if (visited.count(it.first) == 0)
78  cycleUtil(it.first, {});
79  if (cyclic)
80  break;
81  }
82 
83  if (cyclic) {
84  auto err = moduleOp.emitOpError();
85  err << "cannot deduce top level function - cycle "
86  "detected in instance graph (";
87  llvm::interleave(
88  cycleTrace, err, [&](auto node) { err << node; }, "->");
89  err << ").";
90  return err;
91  }
92  assert(!candidateTopLevel.empty() &&
93  "if non-cyclic, there should be at least 1 candidate top level");
94 
95  if (candidateTopLevel.size() > 1) {
96  auto err = moduleOp.emitOpError();
97  err << "multiple candidate top-level modules detected (";
98  llvm::interleaveComma(candidateTopLevel, err,
99  [&](auto topLevel) { err << topLevel; });
100  err << "). Please remove one of these from the source program.";
101  return err;
102  }
103  topLevel = *candidateTopLevel.begin();
104  return success();
105 }

107 LogicalResult verifyAllValuesHasOneUse(handshake::FuncOp funcOp) {
108  for (auto &subOp : funcOp.getOps()) {
109  for (auto res : llvm::enumerate(subOp.getResults())) {
110  if (!res.value().hasOneUse())
111  return subOp.emitOpError()
112  << "result " << res.index() << " has multiple uses.";
113  }
114  }
115 
116  Block &entryBlock = funcOp.front();
117  for (auto barg : enumerate(entryBlock.getArguments())) {
118  if (!barg.value().hasOneUse())
119  return funcOp.emitOpError()
120  << "argument " << barg.index() << " has multiple uses.";
121  }
122  return success();
123 }
