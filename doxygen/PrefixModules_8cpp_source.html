29 /// This maps a FModuleOp to a list of all prefixes that need to be applied.
30 /// When a module has multiple prefixes, it will be cloned for each one. Usually
31 /// there is only a single prefix applied to each module, although there could
32 /// be many.
33 using PrefixMap = llvm::DenseMap<StringRef, std::vector<std::string>>;

35 /// Insert a string into the end of vector if the string is not already present.
36 static void recordPrefix(PrefixMap &prefixMap, StringRef moduleName,
37  std::string prefix) {
38  auto &modulePrefixes = prefixMap[moduleName];
39  if (llvm::find(modulePrefixes, prefix) == modulePrefixes.end())
40  modulePrefixes.push_back(prefix);
41 }

56 /// Get the PrefixInfo for a module from a NestedPrefixModulesAnnotation on a
57 /// module. If the module is not annotated, the prefix returned will be empty.
58 static PrefixInfo getPrefixInfo(Operation *module) {
59  AnnotationSet annotations(module);
60 
61  // Get the annotation from the module.
62  auto anno = annotations.getAnnotation(
63  "sifive.enterprise.firrtl.NestedPrefixModulesAnnotation");
64  if (!anno)
65  return {"", false};
66 
67  // Get the prefix from the annotation.
68  StringRef prefix = "";
69  if (auto prefixAttr = anno.getMember<StringAttr>("prefix"))
70  prefix = prefixAttr.getValue();
71 
72  // Get the inclusive flag from the annotation.
73  bool inclusive = false;
74  if (auto inclusiveAttr = anno.getMember<BoolAttr>("inclusive"))
75  inclusive = inclusiveAttr.getValue();
76 
77  return {prefix, inclusive};
78 }

80 /// If there is an inclusive prefix attached to the module, return it.
81 static StringRef getPrefix(Operation *module) {
82  auto prefixInfo = getPrefixInfo(module);
83  if (prefixInfo.inclusive)
84  return prefixInfo.prefix;
85  return "";
86 }

129 /// When a module is cloned, it carries with it all non-local annotations. This
130 /// function will remove all non-local annotations from the clone with a path
131 /// that doesn't match.
132 void PrefixModulesPass::removeDeadAnnotations(StringAttr moduleName,
133  Operation *op) {
134  // A predicate to check if an annotation can be removed. If there is a
135  // reference to a NLA, the NLA should either contain this module in its path,
136  // if its an InstanceOp. Else, it must exist at the leaf of the NLA. Otherwise
137  // the NLA reference can be removed, since its a spurious annotation, result
138  // of cloning the original module.
139  auto canRemoveAnno = [&](Annotation anno, Operation *op) -> bool {
140  auto nla = anno.getMember("circt.nonlocal");
141  if (!nla)
142  return false;
143  auto nlaName = nla.cast<FlatSymbolRefAttr>().getAttr();
144  auto nlaOp = nlaTable->getNLA(nlaName);
145  if (!nlaOp) {
146  op->emitError("cannot find HierPathOp :" + nlaName.getValue());
147  signalPassFailure();
148  return false;
149  }
150 
151  bool isValid = false;
152  if (isa<InstanceOp>(op))
153  isValid = nlaOp.hasModule(moduleName);
154  else
155  isValid = nlaOp.leafMod() == moduleName;
156  return !isValid;
157  };
158  AnnotationSet::removePortAnnotations(
159  op, std::bind(canRemoveAnno, std::placeholders::_2, op));
160  AnnotationSet::removeAnnotations(
161  op, std::bind(canRemoveAnno, std::placeholders::_1, op));
162 }

164 /// Applies the prefix to the module. This will update the required prefixes of
165 /// any referenced module in the prefix map.
166 void PrefixModulesPass::renameModuleBody(std::string prefix, FModuleOp module) {
167  auto *context = module.getContext();
168  uint32_t groupID = 0;
169  auto iter = prefixIdMap.find(prefix);
170  if (iter == prefixIdMap.end()) {
171  groupID = prefixIdMap.size() + 1;
172  prefixIdMap[prefix] = groupID;
173  } else
174  groupID = iter->second;
175 
176  // If we are renaming the body of this module, we need to mark that we have
177  // changed the IR. If we are prefixing with the empty string, then nothing has
178  // changed yet.
179  if (!prefix.empty())
180  anythingChanged = true;
181  StringAttr thisMod = module.getNameAttr();
182 
183  // Remove spurious NLA references from the module ports and the module itself.
184  // Some of the NLA references become invalid after a module is cloned, based
185  // on the instance.
186  removeDeadAnnotations(thisMod, module);
187 
188  module.body().walk([&](Operation *op) {
189  // Remove spurious NLA references either on a leaf op, or the InstanceOp.
190  removeDeadAnnotations(thisMod, op);
191 
192  if (auto memOp = dyn_cast<MemOp>(op)) {
193  // Memories will be turned into modules and should be prefixed.
194  memOp.nameAttr(StringAttr::get(context, prefix + memOp.name()));
195  memOp.groupIDAttr(IntegerAttr::get(
196  IntegerType::get(context, 32, IntegerType::Unsigned), groupID));
197  } else if (auto instanceOp = dyn_cast<InstanceOp>(op)) {
198  auto target = dyn_cast<FModuleLike>(
199  *instanceGraph->getReferencedModule(instanceOp));
200 
201  // Skip all external modules, unless one of the following conditions
202  // is true:
203  // - This is a Grand Central Data Tap
204  // - This is a Grand Central Mem Tap
205  if (auto *extModule = dyn_cast_or_null<FExtModuleOp>(&target)) {
206  auto isDataTap =
207  AnnotationSet(*extModule).hasAnnotation(dataTapsBlackboxClass);
208  auto isMemTap = AnnotationSet::forPort(*extModule, 0)
209  .hasAnnotation(memTapPortClass);
210  if (!isDataTap && !isMemTap)
211  return;
212  }
213 
214  // Record that we must prefix the target module with the current prefix.
215  recordPrefix(prefixMap, target.moduleName(), prefix);
216 
217  // Fixup this instance op to use the prefixed module name. Note that the
218  // referenced FModuleOp will be renamed later.
219  auto newTarget = StringAttr::get(context, prefix + getPrefix(target) +
220  target.moduleName());
221  AnnotationSet instAnnos(instanceOp);
222  // If the instance has HierPathOp, then update its module name also.
223  // There can be multiple HierPathOps attached to the instance op.
224 
225  StringAttr oldModName = instanceOp.moduleNameAttr().getAttr();
226  // Update the NLAs that apply on this InstanceOp.
227  for (Annotation anno : instAnnos) {
228  if (auto nla = anno.getMember("circt.nonlocal")) {
229  auto nlaName = nla.cast<FlatSymbolRefAttr>().getAttr();
230  nlaTable->updateModuleInNLA(nlaName, oldModName, newTarget);
231  }
232  }
233  // Now get the NLAs that pass through the InstanceOp and update them also.
234  DenseSet<HierPathOp> instNLAs;
235  nlaTable->getInstanceNLAs(instanceOp, instNLAs);
236  for (auto nla : instNLAs)
237  nlaTable->updateModuleInNLA(nla, oldModName, newTarget);
238 
239  instanceOp.moduleNameAttr(FlatSymbolRefAttr::get(context, newTarget));
240  }
241  });
242 }

244 /// Apply all required renames to the current module. This will update the
245 /// prefix map for any referenced module.
246 void PrefixModulesPass::renameModule(FModuleOp module) {
247  // If the module is annotated to have a prefix, it will be applied after the
248  // parent's prefix.
249  auto prefixInfo = getPrefixInfo(module);
250  auto innerPrefix = prefixInfo.prefix;
251 
252  // Remove the annotation from the module.
253  AnnotationSet::removeAnnotations(
254  module, "sifive.enterprise.firrtl.NestedPrefixModulesAnnotation");
255 
256  // We only add the annotated prefix to the module name if it is inclusive.
257  auto moduleName = module.getName().str();
258  if (prefixInfo.inclusive)
259  moduleName = (innerPrefix + moduleName).str();
260 
261  auto &prefixes = prefixMap[module.getName()];
262 
263  // If there are no required prefixes of this module, then this module is a
264  // top-level module, and there is an implicit requirement that it has an empty
265  // prefix. This empty prefix will be applied to to all modules instantiated by
266  // this module.
267  if (prefixes.empty())
268  prefixes.push_back("");
269 
270  auto &firstPrefix = prefixes.front();
271 
272  auto fixNLAsRootedAt = [&](StringAttr oldModName, StringAttr newModuleName) {
273  DenseSet<HierPathOp> nlas;
274  nlaTable->getNLAsInModule(oldModName, nlas);
275  for (auto n : nlas)
276  if (n.root() == oldModName)
277  nlaTable->updateModuleInNLA(n, oldModName, newModuleName);
278  };
279  // Rename the module for each required prefix. This will clone the module
280  // once for each prefix but the first.
281  OpBuilder builder(module);
282  builder.setInsertionPointAfter(module);
283  auto oldModName = module.getNameAttr();
284  for (auto &outerPrefix : llvm::drop_begin(prefixes)) {
285  auto moduleClone = cast<FModuleOp>(builder.clone(*module));
286  auto newModuleName = (outerPrefix + moduleName);
287  auto newModNameAttr = StringAttr::get(module.getContext(), newModuleName);
288  moduleClone.setName(newModuleName);
289  // It is critical to add the new module to the NLATable, otherwise the
290  // rename operation would fail.
291  nlaTable->addModule(moduleClone);
292  fixNLAsRootedAt(oldModName, newModNameAttr);
293  // Each call to this function could invalidate the `prefixes` reference.
294  renameModuleBody((outerPrefix + innerPrefix).str(), moduleClone);
295  }
296 
297  auto prefixFull = (firstPrefix + innerPrefix).str();
298  auto newModuleName = firstPrefix + moduleName;
299  // The first prefix renames the module in place. There is always at least 1
300  // prefix.
301  module.setName(newModuleName);
302  nlaTable->addModule(module);
303  fixNLAsRootedAt(oldModName,
304  StringAttr::get(module.getContext(), newModuleName));
305  renameModuleBody(prefixFull, module);
306 
307  // If this module contains a Grand Central interface, then also apply renames
308  // to that, but only if there are prefixes to apply.
309  AnnotationSet annotations(module);
310  if (!annotations.hasAnnotation(parentAnnoClass) &&
311  !annotations.hasAnnotation(companionAnnoClass))
312  return;
313  SmallVector<Attribute> newAnnotations;
314  for (auto anno : annotations) {
315  if (!anno.isClass(parentAnnoClass) && !anno.isClass(companionAnnoClass)) {
316  newAnnotations.push_back(anno.getDict());
317  continue;
318  }
319 
320  NamedAttrList newAnno;
321  for (auto attr : anno) {
322  if (attr.getName() == "name") {
323  newAnno.append(attr.getName(),
324  (Attribute)builder.getStringAttr(
325  Twine(prefixFull) +
326  attr.getValue().cast<StringAttr>().getValue()));
327  continue;
328  }
329  newAnno.append(attr.getName(), attr.getValue());
330  }
331  newAnnotations.push_back(
332  DictionaryAttr::getWithSorted(builder.getContext(), newAnno));
333 
334  // Record that we need to apply this prefix to the interface definition.
335  if (anno.getMember<StringAttr>("type").getValue() == "parent")
336  interfacePrefixMap[anno.getMember<IntegerAttr>("id")] = prefixFull;
337  }
338  AnnotationSet(newAnnotations, builder.getContext()).applyToOperation(module);
339 }

341 /// Apply prefixes from the `prefixMap` to an external module. No modifications
342 /// are made if there are no prefixes for this external module. If one prefix
343 /// exists, then the external module will be updated in place. If multiple
344 /// prefixes exist, then the original external module will be updated in place
345 /// and prefixes _after_ the first will cause the module to be cloned
346 /// ("duplicated" in Scala FIRRTL Compiler terminology). The logic of this
347 /// member function is the same as `renameModule` except that there is no module
348 /// body to recursively update.
349 void PrefixModulesPass::renameExtModule(FExtModuleOp extModule) {
350  // Lookup prefixes for this module. If none exist, bail out.
351  auto &prefixes = prefixMap[extModule.getName()];
352  if (prefixes.empty())
353  return;
354 
355  OpBuilder builder(extModule);
356  builder.setInsertionPointAfter(extModule);
357 
358  // Function to apply an outer prefix to an external module. If the module has
359  // an optional "defname" (a name that will be used to generate Verilog), also
360  // update the defname.
361  auto applyPrefixToNameAndDefName = [&](FExtModuleOp &extModule,
362  StringRef prefix) {
363  extModule.setName((prefix + extModule.getName()).str());
364  if (auto defname = extModule.defname())
365  extModule->setAttr("defname",
366  builder.getStringAttr(prefix + defname.getValue()));
367  };
368 
369  // Duplicate the external module if there is more than one prefix.
370  for (auto &prefix : llvm::drop_begin(prefixes)) {
371  auto duplicate = cast<FExtModuleOp>(builder.clone(*extModule));
372  applyPrefixToNameAndDefName(duplicate, prefix);
373  }
374 
375  // Update the original module with a new prefix.
376  applyPrefixToNameAndDefName(extModule, prefixes.front());
377 }

379 /// Apply prefixes from the `prefixMap` to a memory module.
380 void PrefixModulesPass::renameMemModule(FMemModuleOp memModule) {
381  // Lookup prefixes for this module. If none exist, bail out.
382  auto &prefixes = prefixMap[memModule.getName()];
383  if (prefixes.empty())
384  return;
385 
386  OpBuilder builder(memModule);
387  builder.setInsertionPointAfter(memModule);
388 
389  // Duplicate the external module if there is more than one prefix.
390  auto originalName = memModule.getName();
391  for (auto &prefix : llvm::drop_begin(prefixes)) {
392  auto duplicate = cast<FMemModuleOp>(builder.clone(*memModule));
393  duplicate.setName((prefix + originalName).str());
394  removeDeadAnnotations(duplicate.getNameAttr(), duplicate);
395  }
396 
397  // Update the original module with a new prefix.
398  memModule.setName((prefixes.front() + originalName).str());
399  removeDeadAnnotations(memModule.getNameAttr(), memModule);
400 }

402 /// Mutate circuit-level annotations to add prefix information to Grand Central
403 /// (SystemVerilog) interfaces. Add a "prefix" field to each interface
404 /// definition (an annotation with class "AugmentedBundleType") that holds the
405 /// prefix that was determined during runOnModule. It is assumed that this
406 /// field did not exist before.
407 void PrefixModulesPass::prefixGrandCentralInterfaces() {
408  // Early exit if no interfaces need prefixes.
409  if (interfacePrefixMap.empty())
410  return;
411 
412  auto circuit = getOperation();
413  OpBuilder builder(circuit);
414 
415  SmallVector<Attribute> newCircuitAnnotations;
416  for (auto anno : AnnotationSet(circuit)) {
417  // Only mutate this annotation if it is an AugmentedBundleType and
418  // interfacePrefixMap has prefix information for it.
419  StringRef prefix;
420  if (anno.isClass("sifive.enterprise.grandcentral.AugmentedBundleType")) {
421  if (auto id = anno.getMember<IntegerAttr>("id"))
422  prefix = interfacePrefixMap[id];
423  }
424 
425  // Nothing to do. Copy the annotation.
426  if (prefix.empty()) {
427  newCircuitAnnotations.push_back(anno.getDict());
428  continue;
429  }
430 
431  // Add a "prefix" field with the prefix for this interface. This is safe to
432  // put at the back and do a `getWithSorted` because the last field is
433  // conveniently called "name".
434  NamedAttrList newAnno(anno.getDict().getValue());
435  newAnno.append("prefix", builder.getStringAttr(prefix));
436  newCircuitAnnotations.push_back(
437  DictionaryAttr::getWithSorted(builder.getContext(), newAnno));
438  }
439 
440  // Overwrite the old circuit annotation with the new one created here.
441  AnnotationSet(newCircuitAnnotations, builder.getContext())
442  .applyToOperation(circuit);
443 }

445 void PrefixModulesPass::runOnOperation() {
446  auto *context = &getContext();
447  instanceGraph = &getAnalysis<InstanceGraph>();
448  nlaTable = &getAnalysis<NLATable>();
449  auto circuitOp = getOperation();
450 
451  // If the main module is prefixed, we have to update the CircuitOp.
452  auto mainModule = instanceGraph->getTopLevelModule();
453  auto prefix = getPrefix(mainModule);
454  if (!prefix.empty()) {
455  auto oldModName = mainModule.moduleNameAttr();
456  auto newMainModuleName =
457  StringAttr::get(context, (prefix + circuitOp.name()).str());
458  circuitOp.nameAttr(newMainModuleName);
459 
460  // Now update all the NLAs that have the top level module symbol.
461  nlaTable->renameModule(oldModName, newMainModuleName);
462  for (auto n : nlaTable->lookup(oldModName))
463  if (n.root() == oldModName)
464  nlaTable->updateModuleInNLA(n, oldModName, newMainModuleName);
465  }
466 
467  // Walk all Modules in a top-down order. For each module, look at the list of
468  // required prefixes to be applied.
469  DenseSet<InstanceGraphNode *> visited;
470  for (auto *current : *instanceGraph) {
471  for (auto &node : llvm::inverse_post_order_ext(current, visited)) {
472  if (auto module = dyn_cast<FModuleOp>(*node->getModule()))
473  renameModule(module);
474  if (auto extModule = dyn_cast<FExtModuleOp>(*node->getModule()))
475  renameExtModule(extModule);
476  if (auto memModule = dyn_cast<FMemModuleOp>(*node->getModule()))
477  renameMemModule(memModule);
478  }
479  }
480 
481  // Update any interface definitions if needed.
482  prefixGrandCentralInterfaces();
483 
484  prefixMap.clear();
485  prefixIdMap.clear();
486  interfacePrefixMap.clear();
487  if (!anythingChanged)
488  markAllAnalysesPreserved();
489 }
