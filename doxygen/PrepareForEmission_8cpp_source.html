35 // Check if the value is from read of a wire or reg or is a port.
36 bool ExportVerilog::isSimpleReadOrPort(Value v) {
37  if (v.isa<BlockArgument>())
38  return true;
39  auto vOp = v.getDefiningOp();
40  if (!vOp)
41  return false;
42  auto read = dyn_cast<ReadInOutOp>(vOp);
43  if (!read)
44  return false;
45  auto readSrc = read.input().getDefiningOp();
46  if (!readSrc)
47  return false;
48  return isa<WireOp, RegOp>(readSrc);
49 }

51 // Check if the value is deemed worth spilling into a wire.
52 static bool shouldSpillWire(Operation &op, const LoweringOptions &options) {
53  auto isAssign = [](Operation *op) {
54  return isa<AssignOp, PAssignOp, BPAssignOp, OutputOp>(op);
55  };
56  auto isConcat = [](Operation *op) { return isa<ConcatOp>(op); };
57 
58  if (!isVerilogExpression(&op))
59  return false;
60 
61  // If there are more than the maximum number of terms in this single result
62  // expression, and it hasn't already been spilled, this should spill.
63  if (op.getNumOperands() > options.maximumNumberOfTermsPerExpression &&
64  op.getNumResults() == 1 &&
65  llvm::none_of(op.getResult(0).getUsers(), isAssign))
66  return true;
67 
68  // Work around Verilator #3405. Large expressions inside a concat is worst
69  // case O(n^2) in a certain Verilator optimization, and can effectively hang
70  // Verilator on large designs. Verilator 4.224+ works around this by having a
71  // hard limit on its recursion. Here we break large expressions inside concats
72  // according to a configurable limit to work around the same issue.
73  // See https://github.com/verilator/verilator/issues/3405.
74  if (op.getNumOperands() > options.maximumNumberOfTermsInConcat &&
75  op.getNumResults() == 1 &&
76  llvm::any_of(op.getResult(0).getUsers(), isConcat))
77  return true;
78 
79  // When `spillWiresAtPrepare` is true, spill temporary wires if necessary.
80  if (options.spillWiresAtPrepare &&
81  !ExportVerilog::isExpressionEmittedInline(&op))
82  return true;
83 
84  return false;
85 }

87 // Given an invisible instance, make sure all inputs are driven from
88 // wires or ports.
89 static void lowerBoundInstance(InstanceOp op) {
90  if (!op->hasAttr("doNotPrint"))
91  return;
92  Block *block = op->getParentOfType<HWModuleOp>().getBodyBlock();
93  auto builder = ImplicitLocOpBuilder::atBlockBegin(op.getLoc(), block);
94 
95  SmallString<32> nameTmp{"_", op.instanceName(), "_"};
96  auto namePrefixSize = nameTmp.size();
97 
98  size_t nextOpNo = 0;
99  for (auto &port : getModulePortInfo(op).inputs) {
100  auto src = op.getOperand(nextOpNo);
101  ++nextOpNo;
102 
103  if (isSimpleReadOrPort(src))
104  continue;
105 
106  nameTmp.resize(namePrefixSize);
107  if (port.name)
108  nameTmp += port.name.getValue().str();
109  else
110  nameTmp += std::to_string(nextOpNo - 1);
111 
112  auto newWire = builder.create<WireOp>(src.getType(), nameTmp);
113  auto newWireRead = builder.create<ReadInOutOp>(newWire);
114  auto connect = builder.create<AssignOp>(newWire, src);
115  newWireRead->moveBefore(op);
116  connect->moveBefore(op);
117  op.setOperand(nextOpNo - 1, newWireRead);
118  }
119 }

121 // Ensure that each output of an instance are used only by a wire
122 static void lowerInstanceResults(InstanceOp op) {
123  Block *block = op->getParentOfType<HWModuleOp>().getBodyBlock();
124  auto builder = ImplicitLocOpBuilder::atBlockBegin(op.getLoc(), block);
125 
126  SmallString<32> nameTmp{"_", op.instanceName(), "_"};
127  auto namePrefixSize = nameTmp.size();
128 
129  size_t nextResultNo = 0;
130  for (auto &port : getModulePortInfo(op).outputs) {
131  auto result = op.getResult(nextResultNo);
132  ++nextResultNo;
133 
134  if (result.hasOneUse()) {
135  OpOperand &use = *result.getUses().begin();
136  if (dyn_cast_or_null<OutputOp>(use.getOwner()))
137  continue;
138  if (auto assign = dyn_cast_or_null<AssignOp>(use.getOwner())) {
139  // Move assign op after instance to resolve cyclic dependencies.
140  assign->moveAfter(op);
141  continue;
142  }
143  }
144 
145  nameTmp.resize(namePrefixSize);
146  if (port.name)
147  nameTmp += port.name.getValue().str();
148  else
149  nameTmp += std::to_string(nextResultNo - 1);
150  Value newWire = builder.create<WireOp>(result.getType(), nameTmp);
151 
152  while (!result.use_empty()) {
153  auto newWireRead = builder.create<ReadInOutOp>(newWire);
154  OpOperand &use = *result.getUses().begin();
155  use.set(newWireRead);
156  newWireRead->moveBefore(use.getOwner());
157  }
158 
159  auto connect = builder.create<AssignOp>(newWire, result);
160  connect->moveAfter(op);
161  }
162 }

164 // Given a side effect free "always inline" operation, make sure that it
165 // exists in the same block as its users and that it has one use for each one.
166 static void lowerAlwaysInlineOperation(Operation *op) {
167  assert(op->getNumResults() == 1 &&
168  "only support 'always inline' ops with one result");
169 
170  // Nuke use-less operations.
171  if (op->use_empty()) {
172  op->erase();
173  return;
174  }
175 
176  // Moving/cloning an op should pull along its operand tree with it if they
177  // are always inline. This happens when an array index has a constant
178  // operand for example.
179  auto recursivelyHandleOperands = [](Operation *op) {
180  for (auto operand : op->getOperands()) {
181  if (auto *operandOp = operand.getDefiningOp())
182  if (isExpressionAlwaysInline(operandOp))
183  lowerAlwaysInlineOperation(operandOp);
184  }
185  };
186 
187  // If this operation has multiple uses, duplicate it into N-1 of them in
188  // turn.
189  while (!op->hasOneUse()) {
190  OpOperand &use = *op->getUses().begin();
191  Operation *user = use.getOwner();
192 
193  // Clone the op before the user.
194  auto *newOp = op->clone();
195  user->getBlock()->getOperations().insert(Block::iterator(user), newOp);
196  // Change the user to use the new op.
197  use.set(newOp->getResult(0));
198 
199  // If any of the operations of the moved op are always inline, recursively
200  // handle them too.
201  recursivelyHandleOperands(newOp);
202  }
203 
204  // Finally, ensures the op is in the same block as its user so it can be
205  // inlined.
206  Operation *user = *op->getUsers().begin();
207  if (op->getBlock() != user->getBlock()) {
208  op->moveBefore(user);
209 
210  // If any of the operations of the moved op are always inline, recursively
211  // move/clone them too.
212  recursivelyHandleOperands(op);
213  }
214  return;
215 }

217 /// Lower a variadic fully-associative operation into an expression tree. This
218 /// enables long-line splitting to work with them.
219 static Value lowerFullyAssociativeOp(Operation &op, OperandRange operands,
220  SmallVector<Operation *> &newOps) {
221  // save the top level name
222  auto name = op.getAttr("sv.namehint");
223  if (name)
224  op.removeAttr("sv.namehint");
225  Value lhs, rhs;
226  switch (operands.size()) {
227  case 0:
228  assert(0 && "cannot be called with empty operand range");
229  break;
230  case 1:
231  return operands[0];
232  case 2:
233  lhs = operands[0];
234  rhs = operands[1];
235  break;
236  default:
237  auto firstHalf = operands.size() / 2;
238  lhs = lowerFullyAssociativeOp(op, operands.take_front(firstHalf), newOps);
239  rhs = lowerFullyAssociativeOp(op, operands.drop_front(firstHalf), newOps);
240  break;
241  }
242 
243  OperationState state(op.getLoc(), op.getName());
244  state.addOperands(ValueRange{lhs, rhs});
245  state.addTypes(op.getResult(0).getType());
246  auto *newOp = Operation::create(state);
247  op.getBlock()->getOperations().insert(Block::iterator(&op), newOp);
248  newOps.push_back(newOp);
249  if (name)
250  newOp->setAttr("sv.namehint", name);
251  return newOp->getResult(0);
252 }

254 /// When we find that an operation is used before it is defined in a graph
255 /// region, we emit an explicit wire to resolve the issue.
256 static void lowerUsersToTemporaryWire(Operation &op) {
257  Block *block = op.getBlock();
258  auto builder = ImplicitLocOpBuilder::atBlockBegin(op.getLoc(), block);
259 
260  auto createWireForResult = [&](Value result, StringAttr name) {
261  Value newWire;
262  if (name)
263  newWire = builder.create<WireOp>(result.getType(), name);
264  else
265  newWire = builder.create<WireOp>(result.getType());
266 
267  while (!result.use_empty()) {
268  auto newWireRead = builder.create<ReadInOutOp>(newWire);
269  OpOperand &use = *result.getUses().begin();
270  use.set(newWireRead);
271  newWireRead->moveBefore(use.getOwner());
272  }
273  auto connect = builder.create<AssignOp>(newWire, result);
274  connect->moveAfter(&op);
275  };
276 
277  // If the op has a single result and a namehint, give the name to its
278  // temporary wire.
279  if (op.getNumResults() == 1) {
280  auto namehint = op.getAttrOfType<StringAttr>("sv.namehint");
281  // Remove a namehint from the op because the name is moved to the wire.
282  if (namehint)
283  op.removeAttr("sv.namehint");
284  createWireForResult(op.getResult(0), namehint);
285  return;
286  }
287 
288  // If the op has multiple results, create wires for each result.
289  for (auto result : op.getResults())
290  createWireForResult(result, StringAttr());
291 }

293 /// Transform "a + -cst" ==> "a - cst" for prettier output. This returns the
294 /// first operation emitted.
295 static Operation *rewriteAddWithNegativeConstant(comb::AddOp add,
296  hw::ConstantOp rhsCst) {
297  ImplicitLocOpBuilder builder(add.getLoc(), add);
298 
299  // Get the positive constant.
300  auto negCst = builder.create<hw::ConstantOp>(-rhsCst.getValue());
301  auto sub = builder.create<comb::SubOp>(add.getOperand(0), negCst);
302  add.getResult().replaceAllUsesWith(sub);
303  add.erase();
304  if (rhsCst.use_empty())
305  rhsCst.erase();
306  return negCst;
307 }

309 /// Given an operation in a procedural region, scan up the region tree to find
310 /// the first operation in a graph region (typically an always or initial op).
311 ///
312 /// By looking for a graph region, we will stop at graph-region #ifdef's that
313 /// may enclose this operation.
314 static Operation *findParentInNonProceduralRegion(Operation *op) {
315  Operation *parentOp = op->getParentOp();
316  assert(parentOp->hasTrait<ProceduralRegion>() &&
317  "we should only be hoisting from procedural");
318  while (parentOp->getParentOp()->hasTrait<ProceduralRegion>())
319  parentOp = parentOp->getParentOp();
320  return parentOp;
321 }

323 /// This function is invoked on side effecting Verilog expressions when we're in
324 /// 'disallowLocalVariables' mode for old Verilog clients. This ensures that
325 /// any side effecting expressions are only used by a single BPAssign to a
326 /// sv.reg operation. This ensures that the verilog emitter doesn't have to
327 /// worry about spilling them.
328 ///
329 /// This returns true if the op was rewritten, false otherwise.
330 static bool rewriteSideEffectingExpr(Operation *op) {
331  assert(op->getNumResults() == 1 && "isn't a verilog expression");
332 
333  // Check to see if this is already rewritten.
334  if (op->hasOneUse()) {
335  if (auto assign = dyn_cast<BPAssignOp>(*op->user_begin()))
336  if (assign.dest().getDefiningOp<RegOp>())
337  return false;
338  }
339 
340  // Otherwise, we have to transform it. Insert a reg at the top level, make
341  // everything using the side effecting expression read the reg, then assign to
342  // it after the side effecting operation.
343  Value opValue = op->getResult(0);
344 
345  // Scan to the top of the region tree to find out where to insert the reg.
346  Operation *parentOp = findParentInNonProceduralRegion(op);
347  OpBuilder builder(parentOp);
348  auto reg = builder.create<RegOp>(op->getLoc(), opValue.getType());
349 
350  // Everything using the expr now uses a read_inout of the reg.
351  auto value = builder.create<ReadInOutOp>(op->getLoc(), reg);
352  opValue.replaceAllUsesWith(value);
353 
354  // We assign the side effect expr to the reg immediately after that expression
355  // is computed.
356  builder.setInsertionPointAfter(op);
357  builder.create<BPAssignOp>(op->getLoc(), reg, opValue);
358  return true;
359 }

361 /// This function is called for non-side-effecting Verilog expressions when
362 /// we're in 'disallowLocalVariables' mode for old Verilog clients. It hoists
363 /// non-constant expressions out to the top level so they don't turn into local
364 /// variable declarations.
365 static bool hoistNonSideEffectExpr(Operation *op) {
366  // Never hoist "always inline" expressions except for inout stuffs - they will
367  // never generate a temporary and in fact must always be emitted inline.
368  if (isExpressionAlwaysInline(op) &&
369  !(isa<sv::ReadInOutOp>(op) ||
370  op->getResult(0).getType().isa<hw::InOutType>()))
371  return false;
372 
373  // Scan to the top of the region tree to find out where to move the op.
374  Operation *parentOp = findParentInNonProceduralRegion(op);
375 
376  // We can typically hoist all the way out to the top level in one step, but
377  // there may be intermediate operands that aren't hoistable. If so, just
378  // hoist one level.
379  bool cantHoist = false;
380  if (llvm::any_of(op->getOperands(), [&](Value operand) -> bool {
381  // The operand value dominates the original operation, but may be
382  // defined in one of the procedural regions between the operation and
383  // the top level of the module. We can tell this quite efficiently by
384  // looking for ops in a procedural region - because procedural regions
385  // live in graph regions but not visa-versa.
386  Operation *operandOp = operand.getDefiningOp();
387  if (!operandOp) // References to ports are always ok.
388  return false;
389 
390  if (operandOp->getParentOp()->hasTrait<ProceduralRegion>()) {
391  cantHoist |= operandOp->getBlock() == op->getBlock();
392  return true;
393  }
394  return false;
395  })) {
396 
397  // If the operand is in the same block as the expression then we can't hoist
398  // this out at all.
399  if (cantHoist)
400  return false;
401 
402  // Otherwise, we can hoist it, but not all the way out in one step. Just
403  // hoist one level out.
404  parentOp = op->getParentOp();
405  }
406 
407  op->moveBefore(parentOp);
408  return true;
409 }

418 /// If exactly one use of this op is an assign, replace the other uses with a
419 /// read from the assigned wire or reg. This assumes the preconditions for doing
420 /// so are met: op must be an expression in a non-procedural region.
421 static void reuseExistingInOut(Operation *op) {
422  // Try to collect a single assign and all the other uses of op.
423  sv::AssignOp assign;
424  SmallVector<OpOperand *> uses;
425 
426  // Look at each use.
427  for (OpOperand &use : op->getUses()) {
428  // If it's an assign, try to save it.
429  if (auto assignUse = dyn_cast<AssignOp>(use.getOwner())) {
430  // If there are multiple assigns, bail out.
431  if (assign)
432  return;
433 
434  // Remember this assign for later.
435  assign = assignUse;
436  continue;
437  }
438 
439  // If not an assign, remember this use for later.
440  uses.push_back(&use);
441  }
442 
443  // If we didn't find anything, bail out.
444  if (!assign || uses.empty())
445  return;
446 
447  if (auto *cop = assign.src().getDefiningOp())
448  if (isa<ConstantOp>(cop))
449  return;
450 
451  // Replace all saved uses with a read from the assigned destination.
452  ImplicitLocOpBuilder builder(assign.dest().getLoc(), op->getContext());
453  for (OpOperand *use : uses) {
454  builder.setInsertionPoint(use->getOwner());
455  auto read = builder.create<ReadInOutOp>(assign.dest());
456  use->set(read);
457  }
458 }

460 /// For each module we emit, do a prepass over the structure, pre-lowering and
461 /// otherwise rewriting operations we don't want to emit.
462 void ExportVerilog::prepareHWModule(Block &block,
463  const LoweringOptions &options) {
464 
465  // First step, check any nested blocks that exist in this region. This walk
466  // can pull things out to our level of the hierarchy.
467  for (auto &op : block) {
468  // If the operations has regions, prepare each of the region bodies.
469  for (auto &region : op.getRegions()) {
470  if (!region.empty())
471  prepareHWModule(region.front(), options);
472  }
473  }
474 
475  // Next, walk all of the operations at this level.
476 
477  // True if these operations are in a procedural region.
478  bool isProceduralRegion = block.getParentOp()->hasTrait<ProceduralRegion>();
479  for (Block::iterator opIterator = block.begin(), e = block.end();
480  opIterator != e;) {
481  auto &op = *opIterator++;
482 
483  // Name legalization should have happened in a different pass for these sv
484  // elements and we don't want to change their name through re-legalization
485  // (e.g. letting a temporary take the name of an unvisited wire). Adding
486  // them now ensures any temporary generated will not use one of the names
487  // previously declared.
488  if (auto instance = dyn_cast<InstanceOp>(op)) {
489  // Anchor return values to wires early
490  lowerInstanceResults(instance);
491  // Anchor ports of bound instances
492  lowerBoundInstance(instance);
493  }
494 
495  // Force any expression used in the event control of an always process to be
496  // a trivial wire, if the corresponding option is set.
497  if (!options.allowExprInEventControl) {
498  auto enforceWire = [&](Value expr) {
499  // Direct port uses are fine.
500  if (isSimpleReadOrPort(expr))
501  return;
502  if (auto inst = expr.getDefiningOp<InstanceOp>())
503  return;
504  auto builder = ImplicitLocOpBuilder::atBlockBegin(
505  op.getLoc(), &op.getParentOfType<HWModuleOp>().front());
506  auto newWire = builder.create<WireOp>(expr.getType());
507  builder.setInsertionPoint(&op);
508  auto newWireRead = builder.create<ReadInOutOp>(newWire);
509  // For simplicity, replace all uses with the read first. This lets us
510  // recursive root out all uses of the expression.
511  expr.replaceAllUsesWith(newWireRead);
512  builder.setInsertionPoint(&op);
513  builder.create<AssignOp>(newWire, expr);
514  // To get the output correct, given that reads are always inline,
515  // duplicate them for each use.
516  lowerAlwaysInlineOperation(newWireRead);
517  };
518  if (auto always = dyn_cast<AlwaysOp>(op)) {
519  for (auto clock : always.clocks())
520  enforceWire(clock);
521  continue;
522  }
523  if (auto always = dyn_cast<AlwaysFFOp>(op)) {
524  enforceWire(always.clock());
525  if (auto reset = always.reset())
526  enforceWire(reset);
527  continue;
528  }
529  }
530 
531  // If the target doesn't support local variables, hoist all the expressions
532  // out to the nearest non-procedural region.
533  if (options.disallowLocalVariables && isVerilogExpression(&op) &&
534  isProceduralRegion) {
535 
536  // Force any side-effecting expressions in nested regions into a sv.reg
537  // if we aren't allowing local variable declarations. The Verilog emitter
538  // doesn't want to have to have to know how to synthesize a reg in the
539  // case they have to be spilled for whatever reason.
540  if (!mlir::MemoryEffectOpInterface::hasNoEffect(&op)) {
541  if (rewriteSideEffectingExpr(&op))
542  continue;
543  }
544 
545  // Hoist other expressions out to the parent region.
546  //
547  // NOTE: This effectively disables inlining of expressions into if
548  // conditions, $fwrite statements, and instance inputs. We could be
549  // smarter in ExportVerilog itself, but we'd have to teach it to put
550  // spilled expressions (due to line length, multiple-uses, and
551  // non-inlinable expressions) in the outer scope.
552  if (hoistNonSideEffectExpr(&op))
553  continue;
554  }
555 
556  // Duplicate "always inline" expression for each of their users and move
557  // them to be next to their users.
558  if (isExpressionAlwaysInline(&op)) {
559  lowerAlwaysInlineOperation(&op);
560  continue;
561  }
562 
563  // If this expression is deemed worth spilling into a wire, do it here.
564  if (shouldSpillWire(op, options)) {
565  // If we're not in a procedural region, or we are, but we can hoist out of
566  // it, we are good to generate a wire.
567  if (!isProceduralRegion ||
568  (isProceduralRegion && hoistNonSideEffectExpr(&op))) {
569  lowerUsersToTemporaryWire(op);
570 
571  // If we're in a procedural region, we move on to the next op in the
572  // block. The expression splitting and canonicalization below will
573  // happen after we recurse back up. If we're not in a procedural region,
574  // the expression can continue being worked on.
575  if (isProceduralRegion)
576  continue;
577  }
578  }
579 
580  // Lower variadic fully-associative operations with more than two operands
581  // into balanced operand trees so we can split long lines across multiple
582  // statements.
583  // TODO: This is checking the Commutative property, which doesn't seem
584  // right in general. MLIR doesn't have a "fully associative" property.
585  if (op.getNumOperands() > 2 && op.getNumResults() == 1 &&
586  op.hasTrait<mlir::OpTrait::IsCommutative>() &&
587  mlir::MemoryEffectOpInterface::hasNoEffect(&op) &&
588  op.getNumRegions() == 0 && op.getNumSuccessors() == 0 &&
589  (op.getAttrs().empty() ||
590  (op.getAttrs().size() == 1 && op.hasAttr("sv.namehint")))) {
591  // Lower this operation to a balanced binary tree of the same operation.
592  SmallVector<Operation *> newOps;
593  auto result = lowerFullyAssociativeOp(op, op.getOperands(), newOps);
594  op.getResult(0).replaceAllUsesWith(result);
595  op.erase();
596 
597  // Make sure we revisit the newly inserted operations.
598  opIterator = Block::iterator(newOps.front());
599  continue;
600  }
601 
602  // Turn a + -cst ==> a - cst
603  if (auto addOp = dyn_cast<comb::AddOp>(op)) {
604  if (auto cst = addOp.getOperand(1).getDefiningOp<hw::ConstantOp>()) {
605  assert(addOp.getNumOperands() == 2 && "commutative lowering is done");
606  if (cst.getValue().isNegative()) {
607  Operation *firstOp = rewriteAddWithNegativeConstant(addOp, cst);
608  opIterator = Block::iterator(firstOp);
609  continue;
610  }
611  }
612  }

614  // Try to anticipate expressions that ExportVerilog may spill to a temporary
615  // inout, and re-use an existing inout when possible. This is legal when op
616  // is an expression in a non-procedural region.
617  if (!isProceduralRegion && isVerilogExpression(&op))
618  reuseExistingInOut(&op);
619  }
620 
621  // Now that all the basic ops are settled, check for any use-before def issues
622  // in graph regions. Lower these into explicit wires to keep the emitter
623  // simple.
624  if (!isProceduralRegion) {
625  SmallPtrSet<Operation *, 32> seenOperations;
626 
627  for (auto &op : llvm::make_early_inc_range(block)) {
628  // Check the users of any expressions to see if they are
629  // lexically below the operation itself. If so, it is being used out
630  // of order.
631  bool haveAnyOutOfOrderUses = false;
632  for (auto *userOp : op.getUsers()) {
633  // If the user is in a suboperation like an always block, then zip up
634  // to the operation that uses it.
635  while (&block != &userOp->getParentRegion()->front())
636  userOp = userOp->getParentOp();
637 
638  if (seenOperations.count(userOp)) {
639  haveAnyOutOfOrderUses = true;
640  break;
641  }
642  }
643 
644  // Remember that we've seen this operation.
645  seenOperations.insert(&op);
646 
647  // If all the uses of the operation are below this, then we're ok.
648  if (!haveAnyOutOfOrderUses)
649  continue;
650 
651  // If this is a reg/wire declaration, then we move it to the top of the
652  // block. We can't abstract the inout result.
653  if (isMovableDeclaration(&op)) {
654  op.moveBefore(&block.front());
655  continue;
656  }
657 
658  // If this is a constant, then we move it to the top of the block.
659  if (isConstantExpression(&op)) {
660  op.moveBefore(&block.front());
661  continue;
662  }
663 
664  // If this is a an operation reading from a declaration, move it up,
665  // along with the corresponding declaration.
666  if (auto readInOut = dyn_cast<ReadInOutOp>(op)) {
667  auto *def = readInOut.input().getDefiningOp();
668  if (isMovableDeclaration(def)) {
669  op.moveBefore(&block.front());
670  def->moveBefore(&block.front());
671  continue;
672  }
673  }
674 
675  // Otherwise, we need to lower this to a wire to resolve this.
676  lowerUsersToTemporaryWire(op);
677  }
678  }
679 }
