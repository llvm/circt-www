44 /// Return true if this is something that will get printed as a unary operator
45 /// by the Verilog printer.
46 static bool isVerilogUnaryOperator(Operation *op) {
47  if (isa<comb::ParityOp>(op))
48  return true;
49 
50  if (auto xorOp = dyn_cast<comb::XorOp>(op))
51  return xorOp.isBinaryNot();
52 
53  if (auto icmpOp = dyn_cast<comb::ICmpOp>(op))
54  return icmpOp.isEqualAllOnes() || icmpOp.isNotEqualZero();
55 
56  return false;
57 }

59 /// Sink an operation into the same block where it is used. This will clone the
60 /// operation so it can be sunk into multiple blocks. If there are no more uses
61 /// in the current block, the op will be removed.
62 void PrettifyVerilogPass::sinkOpToUses(Operation *op) {
63  assert(mlir::MemoryEffectOpInterface::hasNoEffect(op) &&
64  "Op with side effects cannot be sunk to its uses.");
65  auto block = op->getBlock();
66  // This maps a block to the block local instance of the op.
67  SmallDenseMap<Block *, Value, 8> blockLocalValues;
68  for (auto &use : llvm::make_early_inc_range(op->getUses())) {
69  // If the current use is in the same block as the operation, there is
70  // nothing to do.
71  auto localBlock = use.getOwner()->getBlock();
72  if (block == localBlock)
73  continue;
74  // Find the block local clone of the operation. If there is not one already,
75  // the op will be cloned in to the block.
76  auto &localValue = blockLocalValues[localBlock];
77  if (!localValue) {
78  // Clone the operation and insert it to the beginning of the block.
79  localValue = OpBuilder::atBlockBegin(localBlock).clone(*op)->getResult(0);
80  }
81  // Replace the current use, removing it from the use list.
82  use.set(localValue);
83  anythingChanged = true;
84  }
85  // If this op is no longer used, drop it.
86  if (op->use_empty()) {
87  op->erase();
88  anythingChanged = true;
89  }
90 }

92 /// This is called on unary operators.
93 void PrettifyVerilogPass::prettifyUnaryOperator(Operation *op) {
94  // If this is a multiple use unary operator, duplicate it and move it into the
95  // block corresponding to the user. This avoids emitting a temporary just for
96  // a unary operator. Instead of:
97  //
98  // tmp1 = ^(thing+thing);
99  // = tmp1 + 42
100  //
101  // we get:
102  //
103  // tmp2 = thing+thing;
104  // = ^tmp2 + 42
105  //
106  // This is particularly helpful when the operand of the unary op has multiple
107  // uses as well.
108  if (op->use_empty() || op->hasOneUse())
109  return;
110 
111  while (!op->hasOneUse()) {
112  OpOperand &use = *op->use_begin();
113  Operation *user = use.getOwner();
114 
115  // Clone the operation and insert before this user.
116  auto *cloned = op->clone();
117  user->getBlock()->getOperations().insert(Block::iterator(user), cloned);
118 
119  // Update user's operand to the new value.
120  use.set(cloned->getResult(0));
121  }
122 
123  // There is exactly one user left, so move this before it.
124  Operation *user = *op->user_begin();
125  op->moveBefore(user);
126  anythingChanged = true;
127 }

129 /// Transform "a + -cst" ==> "a - cst" for prettier output.
130 static void rewriteAddWithNegativeConstant(comb::AddOp add,
131  hw::ConstantOp rhsCst) {
132  ImplicitLocOpBuilder builder(add.getLoc(), add);
133 
134  // Get the positive constant.
135  auto negCst = builder.create<hw::ConstantOp>(-rhsCst.getValue());
136  auto sub = builder.create<comb::SubOp>(add.getOperand(0), negCst);
137  add.getResult().replaceAllUsesWith(sub);
138  add.erase();
139  if (rhsCst.use_empty())
140  rhsCst.erase();
141 }

143 void PrettifyVerilogPass::runOnOperation() {
144  // Keeps track if anything changed during this pass, used to determine if
145  // the analyses were preserved.
146  anythingChanged = false;
147 
148  // Walk the operations in post-order, transforming any that are interesting.
149  getOperation()->walk([&](Operation *op) {
150  if (isVerilogUnaryOperator(op))
151  return prettifyUnaryOperator(op);
152  // Sink or duplicate constant ops into the same block as their use. This
153  // will allow the verilog emitter to inline constant expressions.
154  if (matchPattern(op, mlir::m_Constant()))
155  return sinkOpToUses(op);
156 
157  // Sink "free" operations which make Verilog prettier.
158  if (isa<sv::ReadInOutOp>(op))
159  return sinkOpToUses(op);
160 
161  // Turn a + -cst ==> a - cst
162  if (auto addOp = dyn_cast<comb::AddOp>(op))
163  if (auto cst = addOp.getOperand(1).getDefiningOp<hw::ConstantOp>())
164  if (addOp.getNumOperands() == 2 && cst.getValue().isNegative())
165  return rewriteAddWithNegativeConstant(addOp, cst);
166  });
167 
168  // If we did not change anything in the graph mark all analysis as
169  // preserved.
170  if (!anythingChanged)
171  markAllAnalysesPreserved();
172 }
