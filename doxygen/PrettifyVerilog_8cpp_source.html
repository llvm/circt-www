9 // This pass contains elective transformations that improve the quality of
10 // SystemVerilog generated by the ExportVerilog library. This pass is not
11 // compulsory: things that are required for ExportVerilog to be correct should

50 /// Return true if this is something that will get printed as a unary operator
51 /// by the Verilog printer.
52 static bool isVerilogUnaryOperator(Operation *op) {
53  if (isa<comb::ParityOp>(op))
54  return true;
55 
56  if (auto xorOp = dyn_cast<comb::XorOp>(op))
57  return xorOp.isBinaryNot();
58 
59  if (auto icmpOp = dyn_cast<comb::ICmpOp>(op))
60  return icmpOp.isEqualAllOnes() || icmpOp.isNotEqualZero();
61 
62  return false;
63 }

65 /// Sink an operation into the same block where it is used. This will clone the
66 /// operation so it can be sunk into multiple blocks. If there are no more uses
67 /// in the current block, the op will be removed.
68 void PrettifyVerilogPass::sinkOrCloneOpToUses(Operation *op) {
69  assert(mlir::MemoryEffectOpInterface::hasNoEffect(op) &&
70  "Op with side effects cannot be sunk to its uses.");
71  auto block = op->getBlock();
72  // This maps a block to the block local instance of the op.
73  SmallDenseMap<Block *, Value, 8> blockLocalValues;
74  for (auto &use : llvm::make_early_inc_range(op->getUses())) {
75  // If the current use is in the same block as the operation, there is
76  // nothing to do.
77  auto localBlock = use.getOwner()->getBlock();
78  if (block == localBlock)
79  continue;
80  // Find the block local clone of the operation. If there is not one already,
81  // the op will be cloned in to the block.
82  auto &localValue = blockLocalValues[localBlock];
83  if (!localValue) {
84  // Clone the operation and insert it to the beginning of the block.
85  localValue = OpBuilder::atBlockBegin(localBlock).clone(*op)->getResult(0);
86  }
87  // Replace the current use, removing it from the use list.
88  use.set(localValue);
89  anythingChanged = true;
90  }
91  // If this op is no longer used, drop it.
92  if (op->use_empty()) {
93  op->erase();
94  anythingChanged = true;
95  }
96 }

98 /// This is called on unary operators. This returns true if the operator is
99 /// moved.
100 bool PrettifyVerilogPass::prettifyUnaryOperator(Operation *op) {
101  // If this is a multiple use unary operator, duplicate it and move it into the
102  // block corresponding to the user. This avoids emitting a temporary just for
103  // a unary operator. Instead of:
104  //
105  // tmp1 = ^(thing+thing);
106  // = tmp1 + 42
107  //
108  // we get:
109  //
110  // tmp2 = thing+thing;
111  // = ^tmp2 + 42
112  //
113  // This is particularly helpful when the operand of the unary op has multiple
114  // uses as well.
115  if (op->use_empty() || op->hasOneUse())
116  return false;
117 
118  // If this operation has any users that cannot inline the operation, then
119  // don't duplicate any of them.
120  for (auto *user : op->getUsers()) {
121  if (isa<comb::ExtractOp, hw::ArraySliceOp>(user))
122  return false;
123  }
124 
125  // Duplicating unary operations can move them across blocks (down the region
126  // tree). Make sure to keep referenced constants local.
127  auto cloneConstantOperandsIfNeeded = [&](Operation *op) {
128  for (auto &operand : op->getOpOperands()) {
129  auto constant = operand.get().getDefiningOp<hw::ConstantOp>();
130  if (!constant)
131  continue;
132 
133  // If the constant is in a different block, clone or move it into the
134  // block.
135  if (constant->getBlock() != op->getBlock())
136  operand.set(OpBuilder(op).clone(*constant)->getResult(0));
137  }
138  };
139 
140  while (!op->hasOneUse()) {
141  OpOperand &use = *op->use_begin();
142  Operation *user = use.getOwner();
143 
144  // Clone the operation and insert before this user.
145  auto *cloned = OpBuilder(user).clone(*op);
146  cloneConstantOperandsIfNeeded(cloned);
147 
148  // Update user's operand to the new value.
149  use.set(cloned->getResult(0));
150  }
151 
152  // There is exactly one user left, so move this before it.
153  Operation *user = *op->user_begin();
154  op->moveBefore(user);
155  cloneConstantOperandsIfNeeded(op);
156 
157  anythingChanged = true;
158  return true;
159 }

161 // Return the depth of the specified block in the region tree, stopping at
162 // 'topBlock'.
163 static unsigned getBlockDepth(Block *block, Block *topBlock) {
164  unsigned result = 0;
165  while (block != topBlock) {
166  block = block->getParentOp()->getBlock();
167  ++result;
168  }
169  return result;
170 }

172 /// This method is called on expressions to see if we can sink them down the
173 /// region tree. This is a good thing to do to reduce scope of the expression.
174 ///
175 /// This is called on expressions that may have side effects, and filters out
176 /// the side effecting cases late for efficiency.
177 void PrettifyVerilogPass::sinkExpression(Operation *op) {
178  // Ignore expressions with no users.
179  if (op->use_empty())
180  return;
181 
182  Block *curOpBlock = op->getBlock();
183 
184  // Single-used expressions are the most common and simple to handle.
185  if (op->hasOneUse()) {
186  if (curOpBlock != op->user_begin()->getBlock()) {
187  // Ok, we're about to make a change, ensure that there are no side
188  // effects.
189  if (!mlir::MemoryEffectOpInterface::hasNoEffect(op))
190  return;
191 
192  op->moveBefore(*op->user_begin());
193  anythingChanged = true;
194  }
195  return;
196  }
197 
198  // Find the nearest common ancestor of all the users.
199  auto userIt = op->user_begin();
200  Block *ncaBlock = userIt->getBlock();
201  ++userIt;
202  unsigned ncaBlockDepth = getBlockDepth(ncaBlock, curOpBlock);
203  if (ncaBlockDepth == 0)
204  return; // Have a user in the current block.
205 
206  for (auto e = op->user_end(); userIt != e; ++userIt) {
207  auto *userBlock = userIt->getBlock();
208  if (userBlock == curOpBlock)
209  return; // Op has a user in it own block, can't sink it.
210  if (userBlock == ncaBlock)
211  continue;
212 
213  // Get the region depth of the user block so we can march up the region tree
214  // to a common ancestor.
215  unsigned userBlockDepth = getBlockDepth(userBlock, curOpBlock);
216  while (userBlock != ncaBlock) {
217  if (ncaBlockDepth < userBlockDepth) {
218  userBlock = userBlock->getParentOp()->getBlock();
219  --userBlockDepth;
220  } else if (userBlockDepth < ncaBlockDepth) {
221  ncaBlock = ncaBlock->getParentOp()->getBlock();
222  --ncaBlockDepth;
223  } else {
224  userBlock = userBlock->getParentOp()->getBlock();
225  --userBlockDepth;
226  ncaBlock = ncaBlock->getParentOp()->getBlock();
227  --ncaBlockDepth;
228  }
229  }
230 
231  if (ncaBlockDepth == 0)
232  return; // Have a user in the current block.
233  }
234 
235  // Ok, we're about to make a change, ensure that there are no side
236  // effects.
237  if (!mlir::MemoryEffectOpInterface::hasNoEffect(op))
238  return;
239 
240  // Ok, we found a common ancestor between all the users that is deeper than
241  // the current op. Sink it into the start of that block.
242  assert(ncaBlock != curOpBlock && "should have bailed out earlier");
243  op->moveBefore(&ncaBlock->front());
244  anythingChanged = true;
245 }

247 void PrettifyVerilogPass::processPostOrder(Block &body) {
248  SmallVector<Operation *> instances;
249 
250  // Walk the block bottom-up, processing the region tree inside out.
251  for (auto &op :
252  llvm::make_early_inc_range(llvm::reverse(body.getOperations()))) {
253  if (op.getNumRegions()) {
254  for (auto &region : op.getRegions())
255  for (auto &regionBlock : region.getBlocks())
256  processPostOrder(regionBlock);
257  }
258 
259  // Sink and duplicate unary operators.
260  if (isVerilogUnaryOperator(&op) && prettifyUnaryOperator(&op))
261  continue;
262 
263  // Sink or duplicate constant ops and invisible "free" ops into the same
264  // block as their use. This will allow the verilog emitter to inline
265  // constant expressions and avoids ReadInOutOp from preventing motion.
266  if (matchPattern(&op, mlir::m_Constant()) ||
267  isa<sv::ReadInOutOp, sv::ArrayIndexInOutOp, sv::StructFieldInOutOp,
268  sv::IndexedPartSelectInOutOp, hw::ParamValueOp>(op)) {
269  sinkOrCloneOpToUses(&op);
270  continue;
271  }
272 
273  // Sink normal expressions down the region tree if they aren't used within
274  // their current block. This allows them to be folded into the using
275  // expression inline in the best case, and better scopes the temporary wire
276  // they generate in the worst case. Our overall traversal order is
277  // post-order here which means all users will already be sunk.
278  if (hw::isCombinational(&op) || sv::isExpression(&op)) {
279  sinkExpression(&op);
280  continue;
281  }
282 
283  if (isa<hw::InstanceOp>(op))
284  instances.push_back(&op);
285  }
286 
287  // If we have any instances, keep their relative order but shift them to the
288  // end of the module. Any outputs will be writing to a wire or an output port
289  // of the enclosing module anyway, and this allows inputs to be inlined into
290  // the operand list as parameters.
291  if (!instances.empty()) {
292  for (Operation *instance : llvm::reverse(instances)) {
293  if (instance != &body.back())
294  instance->moveBefore(&body.back());
295  }
296  }
297 }

299 void PrettifyVerilogPass::runOnOperation() {
300  hw::HWModuleOp thisModule = getOperation();
301 
302  // Keeps track if anything changed during this pass, used to determine if
303  // the analyses were preserved.
304  anythingChanged = false;
305 
306  // Walk the operations in post-order, transforming any that are interesting.
307  processPostOrder(*thisModule.getBodyBlock());
308 
309  // If we did not change anything in the graph mark all analysis as
310  // preserved.
311  if (!anythingChanged)
312  markAllAnalysesPreserved();
313 }
