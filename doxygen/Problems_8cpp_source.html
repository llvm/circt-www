26 LogicalResult Problem::insertDependence(Dependence dep) {
27  Operation *src = dep.getSource();
28  Operation *dst = dep.getDestination();
29 
30  // Fail early on invalid dependences (src == dst == null), and def-use
31  // dependences that cannot be added because the source value is not the result
32  // of an operation (e.g., a BlockArgument).
33  if (!src || !dst)
34  return failure();
35 
36  // record auxiliary dependences explicitly
37  if (dep.isAuxiliary())
38  auxDependences[dst].insert(src);
39 
40  // auto-register the endpoints
41  operations.insert(src);
42  operations.insert(dst);
43 
44  return success();
45 }

102 LogicalResult Problem::verifyDependence(Dependence dep) {
103  Operation *i = dep.getSource();
104  Operation *j = dep.getDestination();
105 
106  unsigned stI = *getStartTime(i);
107  unsigned latI = *getLatency(*getLinkedOperatorType(i));
108  unsigned stJ = *getStartTime(j);
109 
110  // check if i's result is available before j starts
111  if (!(stI + latI <= stJ))
112  return containingOp->emitError()
113  << "Precedence violated for dependence."
114  << "\n from: " << *i << ", result available in t=" << (stI + latI)
115  << "\n to: " << *j << ", starts in t=" << stJ;
116 
117  return success();
118 }

183 DependenceIterator::DependenceIterator(Problem &problem, Operation *op,
184  bool end)
185  : problem(problem), op(op), operandIdx(0), auxPredIdx(0), auxPreds(nullptr),
186  dep() {
187  if (!end) {
188  if (problem.auxDependences.count(op))
189  auxPreds = &problem.auxDependences[op];
190 
191  findNextDependence();
192  }
193 }
194 
195 void DependenceIterator::findNextDependence() {
196  // Yield dependences corresponding to values used by `op`'s operands...
197  while (operandIdx < op->getNumOperands()) {
198  dep = Dependence(&op->getOpOperand(operandIdx++));
199  Operation *src = dep.getSource();
200 
201  // ... but only if they are outgoing from operations that are registered in
202  // the scheduling problem.
203  if (src && problem.hasOperation(src))
204  return;
205  }
206 
207  // Then, yield auxiliary dependences, if present.
208  if (auxPreds && auxPredIdx < auxPreds->size()) {
209  dep = Dependence((*auxPreds)[auxPredIdx++], op);
210  return;
211  }
212 
213  // An invalid dependence signals the end of iteration.
214  dep = Dependence();
215 }
