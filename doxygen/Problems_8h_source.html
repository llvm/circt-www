32 /// This class models the most basic scheduling problem.
33 ///
34 /// A problem instance is comprised of:
35 ///
36 /// - *Operations*: The vertices in the data-flow graph to be scheduled.
37 /// - *Dependences*: The edges in the data-flow graph to be scheduled, modeling
38 /// a precedence relation between the involved operations.
39 /// - *Operator types*: An abstraction of the characteristics of the target
40 /// representation, e.g. modules in the HLS component library, available
41 /// functional units, etc. -- operations are executed on instances/units of
42 /// an operator type.
43 ///
44 /// Operations and operator types are stored explicitly. The registered
45 /// operations induce a subgraph of the SSA graph. We implicitly include the
46 /// dependences corresponding to its *def-use* relationships in the problem,
47 /// e.g. if operation *y*'s second operand uses the first result produced by
48 /// *x*, we'd have a dependence `x:0 --> y:1`. Clients can additionally register
49 /// explicit, *auxiliary* dependence between operations, e.g. to encode memory
50 /// dependencies or other ordering constraints. Auxiliary dependences do not
51 /// distinguish specific operands/results. The differences between the flavors
52 /// are transparent to concrete algorithms.
53 ///
54 /// All components of the problem (operations, dependences, operator types, as
55 /// well as the problem itself) can be annotated with *properties*. In this
56 /// basic problem, we model
57 ///
58 /// - `linkedOperatorType` maps operations to their operator types.
59 /// - `latency`, an operator type-property denoting the number of time steps
60 /// after which the operator's result is available.
61 /// - `startTime`, an operation-property for the time step in which an operation
62 /// is started. Together, the start times for all operations represent the
63 /// problem's solution, i.e. the schedule.
64 ///
65 /// Subclasses, i.e. corresponding to more complex scheduling problems, can
66 /// declare additional properties as needed. The `clearSolution` method must be
67 /// overridden if the new properties are part of the solution.
68 //
69 /// The `check...` methods perform validity checks before scheduling, e.g. that
70 /// all operations have an associated operator type, etc.
71 ///
72 /// The `verify...` methods check the correctness of the solution determined by
73 /// a concrete scheduling algorithm, e.g. that there are start times available
74 /// for each registered operation, and the precedence constraints as modeled by
75 /// the dependences are satisfied.
76 class Problem {
77 public:
78  /// Initialize a scheduling problem corresponding to \p containingOp.
79  Problem(Operation *containingOp) : containingOp(containingOp) {}
80  virtual ~Problem() = default;
81 
82  friend detail::DependenceIterator;
83 
84  //===--------------------------------------------------------------------===//
85  // Aliases for the problem components
86  //===--------------------------------------------------------------------===//
87 public:
88  /// A thin wrapper to allow a uniform handling of def-use and auxiliary
89  /// dependences.
90  using Dependence = detail::Dependence;
91 
92  /// Operator types are distinguished by name (chosen by the client).
93  using OperatorType = mlir::Identifier;
94 
95  //===--------------------------------------------------------------------===//
96  // Aliases for containers storing the problem components and properties
97  //===--------------------------------------------------------------------===//
98 public:
99  using OperationSet = llvm::SetVector<Operation *>;
100  using DependenceRange = llvm::iterator_range<detail::DependenceIterator>;
101  using OperatorTypeSet = llvm::SetVector<OperatorType>;
102 
103 protected:
104  using AuxDependenceMap =
105  llvm::DenseMap<Operation *, llvm::SmallSetVector<Operation *, 4>>;
106 
107  template <typename T>
108  using OperationProperty = llvm::DenseMap<Operation *, Optional<T>>;
109  template <typename T>
110  using DependenceProperty = llvm::DenseMap<Dependence, Optional<T>>;
111  template <typename T>
112  using OperatorTypeProperty = llvm::DenseMap<OperatorType, Optional<T>>;
113  template <typename T>
114  using ProblemProperty = Optional<T>;
115 
116  //===--------------------------------------------------------------------===//
117  // Containers for problem components and properties
118  //===--------------------------------------------------------------------===//
119 private:
120  // Operation containing the ops for this scheduling problem. Used for its
121  // MLIRContext and to emit diagnostics.
122  Operation *containingOp;
123 
124  // Problem components
125  OperationSet operations;
126  AuxDependenceMap auxDependences;
127  OperatorTypeSet operatorTypes;
128 
129  // Operation properties
130  OperationProperty<OperatorType> linkedOperatorType;
131  OperationProperty<unsigned> startTime;
132 
133  // Pperator type properties
134  OperatorTypeProperty<unsigned> latency;
135 
136  //===--------------------------------------------------------------------===//
137  // Problem construction
138  //===--------------------------------------------------------------------===//
139 public:
140  /// Include \p op in this scheduling problem.
141  void insertOperation(Operation *op) { operations.insert(op); }
142 
143  /// Include \p dep in the scheduling problem. Return failure if \p dep does
144  /// not represent a valid def-use or auxiliary dependence between operations.
145  /// The endpoints become registered operations w.r.t. the problem.
146  LogicalResult insertDependence(Dependence dep);
147 
148  /// Include \p opr in this scheduling problem.
149  void insertOperatorType(OperatorType opr) { operatorTypes.insert(opr); }
150 
151  /// Retrieves the operator type identified by the client-specific \p name. The
152  /// operator type is automatically registered in the scheduling problem.
153  OperatorType getOrInsertOperatorType(StringRef name);
154 
155  //===--------------------------------------------------------------------===//
156  // Access to problem components
157  //===--------------------------------------------------------------------===//
158 public:
159  /// Return the operation containing this problem, e.g. to emit diagnostics.
160  Operation *getContainingOp() { return containingOp; }
161 
162  /// Return true if \p op is part of this problem.
163  bool hasOperation(Operation *op) { return operations.contains(op); }
164  /// Return the set of operations.
165  const OperationSet &getOperations() { return operations; }
166 
167  /// Return a range object to transparently iterate over \p op's *incoming*
168  /// 1) implicit def-use dependences (backed by the SSA graph), and then
169  /// 2) explictly added auxiliary dependences.
170  ///
171  /// In other words, this yields dependences whose destination operation is
172  /// \p op, and whose source operations are \p op's predecessors in the problem
173  /// graph.
174  ///
175  /// To iterate over all of the scheduling problem's dependences, simply
176  /// process the ranges for all registered operations.
177  DependenceRange getDependences(Operation *op);
178 
179  /// Return true if \p opr is part of this problem.
180  bool hasOperatorType(OperatorType opr) { return operatorTypes.contains(opr); }
181  /// Return the set of operator types.
182  const OperatorTypeSet &getOperatorTypes() { return operatorTypes; }
183 
184  //===--------------------------------------------------------------------===//
185  // Access to properties
186  //===--------------------------------------------------------------------===//
187 public:
188  /// The linked operator type provides the runtime characteristics for \p op.
189  Optional<OperatorType> getLinkedOperatorType(Operation *op) {
190  return linkedOperatorType.lookup(op);
191  }
192  void setLinkedOperatorType(Operation *op, OperatorType opr) {
193  linkedOperatorType[op] = opr;
194  }
195 
196  /// The latency is the number of cycles \p opr needs to compute its result.
197  Optional<unsigned> getLatency(OperatorType opr) {
198  return latency.lookup(opr);
199  }
200  void setLatency(OperatorType opr, unsigned val) { latency[opr] = val; }
201 
202  /// Return the start time for \p op, as computed by the scheduler.
203  /// These start times comprise the basic problem's solution, i.e. the
204  /// *schedule*.
205  Optional<unsigned> getStartTime(Operation *op) {
206  return startTime.lookup(op);
207  }
208  void setStartTime(Operation *op, unsigned val) { startTime[op] = val; }
209 
210  /// Clear all properties that are part of the solution.
211  virtual void clearSolution() { startTime.clear(); }
212 
213  //===--------------------------------------------------------------------===//
214  // Hooks to check/verify the different problem components
215  //===--------------------------------------------------------------------===//
216 protected:
217  virtual LogicalResult checkOperation(Operation *op);
218  virtual LogicalResult checkDependence(Dependence dep);
219  virtual LogicalResult checkOperatorType(OperatorType opr);
220  virtual LogicalResult checkProblem();
221 
222  virtual LogicalResult verifyOperation(Operation *op);
223  virtual LogicalResult verifyDependence(Dependence dep);
224  virtual LogicalResult verifyOperatorType(OperatorType opr);
225  virtual LogicalResult verifyProblem();
226 
227 public:
228  /// Return success if the constructed scheduling problem is valid.
229  LogicalResult check();
230  /// Return success if the computed solution is valid.
231  LogicalResult verify();
232 };
