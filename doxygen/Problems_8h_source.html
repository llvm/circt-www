42 /// This class models the most basic scheduling problem.
43 ///
44 /// A problem instance is comprised of:
45 ///
46 /// - *Operations*: The vertices in the data-flow graph to be scheduled.
47 /// - *Dependences*: The edges in the data-flow graph to be scheduled, modeling
48 /// a precedence relation between the involved operations.
49 /// - *Operator types*: An abstraction of the characteristics of the target
50 /// representation, e.g. modules in the HLS component library, available
51 /// functional units, etc. -- operations are executed on instances/units of
52 /// an operator type.
53 ///
54 /// Operations and operator types are stored explicitly. The registered
55 /// operations induce a subgraph of the SSA graph. We implicitly include the
56 /// dependences corresponding to its *def-use* relationships in the problem,
57 /// e.g. if operation *y*'s second operand uses the first result produced by
58 /// *x*, we'd have a dependence `x:0 --> y:1`. Clients can additionally register
59 /// explicit, *auxiliary* dependence between operations, e.g. to encode memory
60 /// dependencies or other ordering constraints. Auxiliary dependences do not
61 /// distinguish specific operands/results. The differences between the flavors
62 /// are transparent to concrete algorithms.
63 ///
64 /// All components of the problem (operations, dependences, operator types, as
65 /// well as the problem itself) can be annotated with *properties*. In this
66 /// basic problem, we model
67 ///
68 /// - `linkedOperatorType` maps operations to their operator types.
69 /// - `latency`, an operator type-property denoting the number of time steps
70 /// after which the operator's result is available.
71 /// - `startTime`, an operation-property for the time step in which an operation
72 /// is started. Together, the start times for all operations represent the
73 /// problem's solution, i.e. the schedule.
74 ///
75 /// Subclasses, i.e. corresponding to more complex scheduling problems, can
76 /// declare additional properties as needed.
77 //
78 /// The `check...` methods perform validity checks before scheduling, e.g. that
79 /// all operations have an associated operator type, etc.
80 ///
81 /// The `verify...` methods check the correctness of the solution determined by
82 /// a concrete scheduling algorithm, e.g. that there are start times available
83 /// for each registered operation, and the precedence constraints as modeled by
84 /// the dependences are satisfied.
85 class Problem {
86  DEFINE_FACTORY_METHOD(Problem)
87 
88 public:
89  virtual ~Problem() = default;

91  friend detail::DependenceIterator;

93  //===--------------------------------------------------------------------===//
94  // Aliases for the problem components
95  //===--------------------------------------------------------------------===//
96 public:
97  /// A thin wrapper to allow a uniform handling of def-use and auxiliary
98  /// dependences.
99  using Dependence = detail::Dependence;
100 
101  /// Operator types are distinguished by name (chosen by the client).
102  using OperatorType = mlir::StringAttr;
103 
104  //===--------------------------------------------------------------------===//
105  // Aliases for containers storing the problem components and properties
106  //===--------------------------------------------------------------------===//
107 public:
108  using OperationSet = llvm::SetVector<Operation *>;
109  using DependenceRange = llvm::iterator_range<detail::DependenceIterator>;
110  using OperatorTypeSet = llvm::SetVector<OperatorType>;
111 
112 protected:
113  using AuxDependenceMap =
114  llvm::DenseMap<Operation *, llvm::SmallSetVector<Operation *, 4>>;

125  //===--------------------------------------------------------------------===//
126  // Containers for problem components and properties
127  //===--------------------------------------------------------------------===//
128 private:
129  // Operation containing the ops for this scheduling problem. Used for its

132

136  OperatorTypeSet operatorTypes;

146  // Problem construction
147  //===--------------------------------------------------------------------===//
148 public:

151 
152  /// Include \p dep in the scheduling problem. Return failure if \p dep does

159 
160  /// Retrieves the operator type identified by the client-specific \p name. The

164  //===--------------------------------------------------------------------===//
165  // Access to problem components
166  //===--------------------------------------------------------------------===//
167 public:
168  /// Return the operation containing this problem, e.g. to emit diagnostics.
169  Operation *getContainingOp() { return containingOp; }
170  /// Set the operation containing this problem, e.g. to emit diagnostics.
171  void setContainingOp(Operation *op) { containingOp = op; }
172 
173  /// Return true if \p op is part of this problem.
174  bool hasOperation(Operation *op) { return operations.contains(op); }
175  /// Return the set of operations.
176  const OperationSet &getOperations() { return operations; }
177 
178  /// Return a range object to transparently iterate over \p op's *incoming*
179  /// 1) implicit def-use dependences (backed by the SSA graph), and then
180  /// 2) explictly added auxiliary dependences.
181  ///
182  /// In other words, this yields dependences whose destination operation is
183  /// \p op, and whose source operations are \p op's predecessors in the problem
184  /// graph.
185  ///
186  /// To iterate over all of the scheduling problem's dependences, simply
187  /// process the ranges for all registered operations.
188  DependenceRange getDependences(Operation *op);
189 
190  /// Return true if \p opr is part of this problem.
191  bool hasOperatorType(OperatorType opr) { return operatorTypes.contains(opr); }
192  /// Return the set of operator types.
193  const OperatorTypeSet &getOperatorTypes() { return operatorTypes; }
194 
195  //===--------------------------------------------------------------------===//
196  // Access to properties
197  //===--------------------------------------------------------------------===//
198 public:
199  /// The linked operator type provides the runtime characteristics for \p op.
200  Optional<OperatorType> getLinkedOperatorType(Operation *op) {
201  return linkedOperatorType.lookup(op);
202  }
203  void setLinkedOperatorType(Operation *op, OperatorType opr) {
204  linkedOperatorType[op] = opr;
205  }
206 
207  /// The latency is the number of cycles \p opr needs to compute its result.
208  Optional<unsigned> getLatency(OperatorType opr) {
209  return latency.lookup(opr);
210  }
211  void setLatency(OperatorType opr, unsigned val) { latency[opr] = val; }
212 
213  /// Return the start time for \p op, as computed by the scheduler.
214  /// These start times comprise the basic problem's solution, i.e. the
215  /// *schedule*.
216  Optional<unsigned> getStartTime(Operation *op) {
217  return startTime.lookup(op);
218  }
219  void setStartTime(Operation *op, unsigned val) { startTime[op] = val; }
220 
221  //===--------------------------------------------------------------------===//
222  // Properties as string key-value pairs (e.g. for DOT graphs)
223  //===--------------------------------------------------------------------===//
224 public:
225  using PropertyStringVector =
226  llvm::SmallVector<std::pair<std::string, std::string>, 2>;
227 
228  virtual PropertyStringVector getProperties(Operation *op);
229  virtual PropertyStringVector getProperties(Dependence dep);
230  virtual PropertyStringVector getProperties(OperatorType opr);
231  virtual PropertyStringVector getProperties();
232 
233  //===--------------------------------------------------------------------===//
234  // Property-specific validators
235  //===--------------------------------------------------------------------===//
236 protected:
237  /// \p op is linked to a registered operator type.
238  virtual LogicalResult checkLinkedOperatorType(Operation *op);
239  /// \p opr has a latency.
240  virtual LogicalResult checkLatency(OperatorType opr);
241  /// \p op has a start time.
242  virtual LogicalResult verifyStartTime(Operation *op);
243  /// \p dep's source operation is available before \p dep's destination
244  /// operation starts.
245  virtual LogicalResult verifyPrecedence(Dependence dep);
246 
247  //===--------------------------------------------------------------------===//
248  // Client API for problem validation
249  //===--------------------------------------------------------------------===//
250 public:
251  /// Return success if the constructed scheduling problem is valid.
252  virtual LogicalResult check();
253  /// Return success if the computed solution is valid.
254  virtual LogicalResult verify();
255 };

257 /// This class models a cyclic scheduling problem. Its solution solution can be
258 /// used to construct a pipelined datapath with a fixed, integer initiation
259 /// interval, in which the execution of multiple iterations/samples/etc. may
260 /// overlap.
261 class CyclicProblem : public virtual Problem {
262  DEFINE_FACTORY_METHOD(CyclicProblem)
263 
264 private:
265  DependenceProperty<unsigned> distance;
266  ProblemProperty<unsigned> initiationInterval;
267 
268 public:
269  /// The distance determines whether a dependence has to be satisfied in the
270  /// same iteration (distance=0 or not set), or distance-many iterations later.
271  Optional<unsigned> getDistance(Dependence dep) {
272  return distance.lookup(dep);
273  }
274  void setDistance(Dependence dep, unsigned val) { distance[dep] = val; }
275 
276  /// The initiation interval (II) is the number of time steps between
277  /// subsequent iterations, i.e. a new iteration is started every II time
278  /// steps. The best possible value is 1, meaning that a corresponding pipeline
279  /// accepts new data every cycle. This property is part of the cyclic
280  /// problem's solution.
281  Optional<unsigned> getInitiationInterval() { return initiationInterval; }
282  void setInitiationInterval(unsigned val) { initiationInterval = val; }
283 
284  virtual PropertyStringVector getProperties(Dependence dep) override;
285  virtual PropertyStringVector getProperties() override;
286 
287 protected:
288  /// \p dep's source operation is available before \p dep's destination
289  /// operation starts (\p dep's distance iterations later).
290  virtual LogicalResult verifyPrecedence(Dependence dep) override;
291  /// This problem has a non-zero II.
292  virtual LogicalResult verifyInitiationInterval();
293 
294 public:
295  virtual LogicalResult verify() override;
296 };

298 /// This class models the accumulation of physical propagation delays on
299 /// combinational paths along SSA dependences.
300 ///
301 /// Each operator type is annotated with estimated values for incoming and
302 /// outgoing delays. Combinational operators (zero-latency, no internal
303 /// registers) have only a single delay; this important special case is modeled
304 /// by setting the incoming and outgoing delays to the same value.
305 ///
306 /// A solution to this problem comprises per-operation start times in a
307 /// continuous unit, e.g. in nanoseconds, inside the discrete time steps/cycles
308 /// determined by the underlying scheduling problem.
309 class ChainingProblem : public virtual Problem {
310  DEFINE_FACTORY_METHOD(ChainingProblem)
311 
312 private:
313  OperatorTypeProperty<float> incomingDelay, outgoingDelay;
314  OperationProperty<float> startTimeInCycle;
315 
316 public:
317  /// The incoming delay denotes the propagation time from the operand inputs to
318  /// either the result outputs (combinational operators) or the first internal
319  /// register stage.
320  Optional<float> getIncomingDelay(OperatorType opr) {
321  return incomingDelay.lookup(opr);
322  }
323  void setIncomingDelay(OperatorType opr, float delay) {
324  incomingDelay[opr] = delay;
325  }
326 
327  /// The outgoing delay denotes the propagation time from either the operand
328  /// inputs (combinational operators) or the last internal register stage to
329  /// the result outputs.
330  Optional<float> getOutgoingDelay(OperatorType opr) {
331  return outgoingDelay.lookup(opr);
332  }
333  void setOutgoingDelay(OperatorType opr, float delay) {
334  outgoingDelay[opr] = delay;
335  }
336 
337  /// Computed by the scheduler, this start time is relative to the beginning of
338  /// the cycle that \p op starts in.
339  Optional<float> getStartTimeInCycle(Operation *op) {
340  return startTimeInCycle.lookup(op);
341  }
342  void setStartTimeInCycle(Operation *op, float time) {
343  startTimeInCycle[op] = time;
344  }
345 
346  virtual PropertyStringVector getProperties(Operation *op) override;
347  virtual PropertyStringVector getProperties(OperatorType opr) override;
348 
349 protected:
350  /// Incoming/outgoing delays are set for \p opr and non-negative. The delays
351  /// are equal if \p opr is a zero-latency operator type.
352  virtual LogicalResult checkDelays(OperatorType opr);
353 
354  /// \p op has a non-negative start time in its cycle.
355  virtual LogicalResult verifyStartTimeInCycle(Operation *op);
356  /// If \p dep is an SSA edge and its source operation finishes in the same
357  /// time step as the destination operation, the source's result is available
358  /// before the destination starts in that cycle.
359  virtual LogicalResult verifyPrecedenceInCycle(Dependence dep);
360 
361 public:
362  virtual LogicalResult check() override;
363  virtual LogicalResult verify() override;
364 };

366 /// This class models a resource-constrained scheduling problem. An optional,
367 /// non-zero *limit* marks operator types to be *shared* by the operations using
368 /// them. In an HLS setting, this corresponds to multiplexing multiple
369 /// operations onto a pre-allocated number of operator instances. These
370 /// instances are assumed to be *fully pipelined*, meaning each instance can
371 /// accept new operands (coming from a distinct operation) in each time step.
372 ///
373 /// A solution to this problem is feasible iff the number of operations that use
374 /// a certain limited operator type, and start in the same time step, does not
375 /// exceed the operator type's limit. These constraints do not apply to operator
376 /// types without a limit (not set, or 0).
377 class SharedOperatorsProblem : public virtual Problem {
378  DEFINE_FACTORY_METHOD(SharedOperatorsProblem)
379 
380 private:
381  OperatorTypeProperty<unsigned> limit;
382 
383 public:
384  /// The limit is the maximum number of operations using \p opr that are
385  /// allowed to start in the same time step.
386  Optional<unsigned> getLimit(OperatorType opr) { return limit.lookup(opr); }
387  void setLimit(OperatorType opr, unsigned val) { limit[opr] = val; }
388 
389  virtual PropertyStringVector getProperties(OperatorType opr) override;
390 
391 protected:
392  /// If \p opr is limited, it has a non-zero latency.
393  virtual LogicalResult checkLatency(OperatorType opr) override;
394  /// \p opr is not oversubscribed in any time step.
395  virtual LogicalResult verifyUtilization(OperatorType opr);
396 
397 public:
398  virtual LogicalResult verify() override;
399 };

401 /// This class models the modulo scheduling problem as the composition of the
402 /// cyclic problem and the resource-constrained problem with fully-pipelined
403 /// shared operators.
404 ///
405 /// A solution to this problem comprises an integer II and integer start times
406 /// for all registered operations, and is feasible iff:
407 /// (1) The precedence constraints implied by the `CyclicProblem`'s dependence
408 /// edges are satisfied, and
409 /// (2) The number of operations that use a certain limited operator type,
410 /// and start in the same congruence class (= start time *mod* II), does
411 /// not exceed the operator type's limit.
412 class ModuloProblem : public virtual CyclicProblem,
413  public virtual SharedOperatorsProblem {
414  DEFINE_FACTORY_METHOD(ModuloProblem)
415 
416 protected:
417  /// \p opr is not oversubscribed in any congruence class modulo II.
418  virtual LogicalResult verifyUtilization(OperatorType opr) override;
419 
420 public:
421  virtual LogicalResult verify() override;
422 };
