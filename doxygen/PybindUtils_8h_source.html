33 /// Accumulates into a python file-like object, either writing text (default)
34 /// or binary.
35 class PyFileAccumulator {
36 public:
37  PyFileAccumulator(pybind11::object fileObject, bool binary)
38  : pyWriteFunction(fileObject.attr("write")), binary(binary) {}
39 
40  void *getUserData() { return this; }
41 
42  MlirStringCallback getCallback() {
43  return [](MlirStringRef part, void *userData) {
44  pybind11::gil_scoped_acquire();
45  PyFileAccumulator *accum = static_cast<PyFileAccumulator *>(userData);
46  if (accum->binary) {
47  // Note: Still has to copy and not avoidable with this API.
48  pybind11::bytes pyBytes(part.data, part.length);
49  accum->pyWriteFunction(pyBytes);
50  } else {
51  pybind11::str pyStr(part.data,
52  part.length); // Decodes as UTF-8 by default.
53  accum->pyWriteFunction(pyStr);
54  }
55  };
56  }
57 
58 private:
59  pybind11::object pyWriteFunction;
60  bool binary;
61 };
