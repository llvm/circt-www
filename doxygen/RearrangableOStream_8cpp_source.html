43 /// Split the segment pointed to by the specified cursor, returning a new
44 /// cursor which is guaranteed to be at the start of the segment (offset=0).
45 auto RearrangableOStream::splitSegment(Cursor position) -> Cursor {
46  assert(!position.isInvalid() && "invalid position");
47 
48  // If we're already pointing to the start of a segment, then there is nothing
49  // to do.
50  if (position.offset == 0)
51  return position;
52 
53  // The size of the current segment may not be updated if we're looking at the
54  // last node we're building into.
55  bool isLastSegment = false;
56  size_t segmentSize = position.node->size();
57  char *segmentPtr = const_cast<char *>(position.node->data());
58  if (segmentSize == 0 && position.node == std::prev(segments.end())) {
59  segmentSize = remainingChunkPtr - segmentPtr;
60  isLastSegment = true;
61  }
62 
63  assert(position.offset <= segmentSize &&
64  "cannot insert into an invalid position");
65 
66  // If we're splitting at the end of the segment, we don't need to do anything,
67  // but we need to refer to the one after this to get an offset of zero.
68  if (position.offset == segmentSize) {
69  if (!isLastSegment)
70  return Cursor(std::next(position.node), 0);
71 
72  // If we were working on the last segment, then close it off and create a
73  // new one to continue at the end.
74  *position.node = StringRef(segmentPtr, segmentSize);
75  segments.push_back(StringRef(remainingChunkPtr, 0));
76  return Cursor(std::prev(segments.end()), 0);
77  }
78 
79  // Keep the first part of the segment at the cursor.
80  *position.node = StringRef(segmentPtr, position.offset);
81 
82  // Add a new segment after this segment for rest of it.
83  segments.insert(std::next(position.node),
84  StringRef(segmentPtr + position.offset,
85  isLastSegment ? 0 : segmentSize - position.offset));
86  return Cursor(std::next(position.node), 0);
87 }

89 /// Insert the specified string literal text into the buffer at the specified
90 /// cursor position. This can split and rearrange segments to make room for
91 /// it.
92 void RearrangableOStream::insertLiteral(Cursor position, StringRef what) {
93  assert(!position.isInvalid() && "invalid position");
94  flush();
95 
96  // If there is space in the current chunk and we just need to memcpy a
97  // small amount of text, do so to avoid creating new segments.
98  if (position.node->empty() && position.node == std::prev(segments.end()) &&
99  what.size() <= remainingChunkSize) {
100  char *segmentPtr = const_cast<char *>(position.node->data());
101  size_t segmentSize = remainingChunkPtr - segmentPtr;
102 
103  assert(position.offset <= segmentSize &&
104  "cannot insert into an invalid position");
105 
106  size_t sizeToCopy = segmentSize - position.offset;
107  if (sizeToCopy < 128) {
108  // Move over the existing text to make room, then plop "what" in the hole.
109  char *cursorPtr = segmentPtr + position.offset;
110  memmove(cursorPtr + what.size(), cursorPtr, sizeToCopy);
111  memcpy(cursorPtr, what.data(), what.size());
112  remainingChunkPtr += what.size();
113  remainingChunkSize -= what.size();
114  return;
115  }
116  }
117 
118  // Split the segment at the cursor point.
119  position = splitSegment(position);
120 
121  // Add the string literal in between as a new segment.
122  segments.insert(position.node, what);
123 }

125 /// Rearrange the block of text in the "[fromBegin, fromEnd)" region to appear
126 /// before the position cursor. This returns a new position Cursor reflecting
127 /// a new Cursor after the inserted text.
128 RearrangableOStream::Cursor
129 RearrangableOStream::moveRangeBefore(Cursor position, Cursor fromBegin,
130  Cursor fromEnd) {
131  // Make sure nothing is buffered in raw_ostream, it is all in the segments.
132  flush();
133  // Split segments at the places we need to move things around. This ensures
134  // that the offsets of each cursor are all zero.
135  fromEnd = splitSegment(fromEnd);
136  fromBegin = splitSegment(fromBegin);
137  position = splitSegment(position);
138 
139  // Just move the segments into place now.
140  segments.splice(position.node, segments, fromBegin.node, fromEnd.node);
141  return position;
142 }

154 /// Split and close off the currently emitted segment, and start a new one.
155 void RearrangableOStream::splitCurrentSegment() {
156  auto &curSegment = segments.back();
157  assert(curSegment.size() == 0 && "segment shouldn't be finished yet");
158 
159  // If nothing is in flight, we are done.
160  if (curSegment.data() == remainingChunkPtr)
161  return;
162 
163  // Set the length of the last segment correctly.
164  curSegment =
165  StringRef(curSegment.data(), remainingChunkPtr - curSegment.data());
166  // Start the next segment.
167  segments.push_back(StringRef(remainingChunkPtr, 0));
168 }

170 void RearrangableOStream::write_impl(const char *ptr, size_t size) {
171  if (size == 0)
172  return;
173 
174  // If we are out of space, allocate another chunk, which is at least twice
175  // as big as the last one.
176  if (size > remainingChunkSize) {
177  splitCurrentSegment();
178  remainingChunkSize = lastChunkSize = std::max(lastChunkSize * 2, size);
179  remainingChunkPtr = (char *)malloc(lastChunkSize);
180  chunks.push_back(remainingChunkPtr);
181  segments.back() = StringRef(remainingChunkPtr, 0);
182  }
183 
184  // Copy the data in and remember we did.
185  memcpy(remainingChunkPtr, ptr, size);
186  remainingChunkPtr += size;
187  remainingChunkSize -= size;
188 }
