19 /// This class is a raw_ostream that supports streaming a wide variety of text
20 /// data into it, but explicitly manages the orders of chunks. It is used by
21 /// ExportVerilog to support efficient insertion of text into text regions that
22 /// have already been generated. There are three major concepts here: chunks,
23 /// segments, and cursors.
24 ///
25 /// A "chunk" is a slab of memory that we throw text into. We use a simple
26 /// size-doubling policy for memory allocation and just insert into the back of
27 /// the chunk. This is what makes this a fancy raw_ostream.
28 ///
29 /// "Segments" are slices of chunks represented as StringRef's. These segments
30 /// are stored in a list and can be reordered to move text around within the
31 /// generated output. This is what makes this "Rearrangable".
32 ///
33 /// "Cursors" are like iterators into the segment stream, which is the unit of
34 /// reordering. They are free to create, but you need to be a bit careful about
35 /// invalidation: they are invalidated when the segment they refer to is split
36 /// before their offset point. Assertions in the code should catch the most
37 /// common problems here.
38 class RearrangableOStream : public raw_ostream {
39 public:
40  explicit RearrangableOStream() {
41  lastChunkSize = 128; // First allocation is 256 bytes.
42  remainingChunkPtr = nullptr;
43  remainingChunkSize = 0;
44  // We always have the unfinished segment on the end of the list.
45  segments.push_back(StringRef(nullptr, 0));
46  }
47  ~RearrangableOStream() override {
48  // Free all the data chunks allocated.
49  for (char *ptr : chunks)
50  free(ptr);
51  }
52 
53  /// A Cursor represents a position in the raw_ostream. This is represented
54  /// with a segment+offset pair.
55  struct Cursor {
56  // Default construct a cursor in "invalid state".
57  Cursor() : node(std::list<StringRef>::iterator()), offset(~size_t()) {}
58 
59  bool isInvalid() const { return offset == ~size_t(); }
60 
61  void dump(RearrangableOStream &os) const;
62 
63  private:
64  Cursor(std::list<StringRef>::iterator node, size_t offset)
65  : node(node), offset(offset) {}
66  std::list<StringRef>::iterator node;
67  size_t offset;
68  friend class RearrangableOStream;
69  };

71  /// Return a cursor for the current insertion point.
72  Cursor getCursor() {
73  // Make sure nothing is buffered anywhere in the stream.
74  flush();
75  return Cursor(std::prev(segments.end()),
76  size_t(remainingChunkPtr - segments.back().data()));
77  }

115  /// This keeps track of the allocated blocks of memory so we can deallocate
116  /// the memory when done.
117  SmallVector<char *> chunks;
118  /// This is the total size of the most recent chunk.
119  size_t lastChunkSize;
120  /// This is the remaining space in the current chunk;
121  char *remainingChunkPtr;
122  size_t remainingChunkSize;
123 
124  /// This keeps track of the output segments being built up. Each segment
125  /// refers to a contiguous piece of text in chunks or might refer to static
126  /// data known to outlive the ostream (e.g. a string literal). The back() of
127  /// the list contains the current segment we are building.
128  ///
129  /// This is a std::list because we need to be able to splice segments of the
130  /// list around efficiently, and want stable node iterators.
131  std::list<StringRef> segments;
132 };
