39 void RemoveUnusedPortsPass::runOnOperation() {
40  auto &instanceGraph = getAnalysis<InstanceGraph>();
41  LLVM_DEBUG(llvm::dbgs() << "===----- Remove unused ports -----==="
42  << "\n");
43  // Iterate in the reverse order of instance graph iterator, i.e. from leaves
44  // to top.
45  for (auto *node : llvm::post_order(&instanceGraph))
46  if (auto module = dyn_cast<FModuleOp>(*node->getModule()))
47  // Don't prune the main module.
48  if (!module.isPublic())
49  removeUnusedModulePorts(module, node);
50 }

52 void RemoveUnusedPortsPass::removeUnusedModulePorts(
53  FModuleOp module, InstanceGraphNode *instanceGraphNode) {
54  LLVM_DEBUG(llvm::dbgs() << "Prune ports of module: " << module.getName()
55  << "\n");
56  // This tracks port indexes that can be erased.
57  SmallVector<unsigned> removalPortIndexes;
58  // This tracks constant values of output ports. None indicates an invalid
59  // value.
60  SmallVector<llvm::Optional<APSInt>> outputPortConstants;
61  auto ports = module.getPorts();
62 
63  for (const auto &e : llvm::enumerate(ports)) {
64  unsigned index = e.index();
65  auto port = e.value();
66  auto arg = module.getArgument(index);
67 
68  // If the port is don't touch or has unprocessed annotations, we cannot
69  // remove the port. Maybe we can allow annotations though.
70  if ((hasDontTouch(arg) || !port.annotations.empty()) && !ignoreDontTouch)
71  continue;
72 
73  // TODO: Handle inout ports.
74  if (port.isInOut())
75  continue;
76 
77  // If the port is input and has an user, we cannot remove the
78  // port.
79  if (port.isInput() && !arg.use_empty())
80  continue;
81 
82  auto portIsUnused = [&](InstanceRecord *a) -> bool {
83  auto port = a->getInstance()->getResult(arg.getArgNumber());
84  return port.getUses().empty();
85  };
86 
87  // Output port.
88  if (port.isOutput()) {
89  if (arg.use_empty()) {
90  // Sometimes the connection is already removed possibly by IMCP.
91  // In that case, regard the port value as an invalid value.
92  outputPortConstants.push_back(None);
93  } else if (llvm::all_of(instanceGraphNode->uses(), portIsUnused)) {
94  // Replace the port with a wire if it is unused.
95  auto builder =
96  ImplicitLocOpBuilder::atBlockBegin(arg.getLoc(), module.getBody());
97  auto wire = builder.create<WireOp>(arg.getType());
98  arg.replaceAllUsesWith(wire);
99  outputPortConstants.push_back(None);
100  } else if (arg.hasOneUse()) {
101  // If the port has a single use, check the port is only connected to
102  // invalid or constant
103  Operation *op = arg.use_begin().getUser();
104  auto connectLike = dyn_cast<FConnectLike>(op);
105  if (!connectLike)
106  continue;
107  auto *srcOp = connectLike.src().getDefiningOp();
108  if (!isa_and_nonnull<InvalidValueOp, ConstantOp>(srcOp))
109  continue;
110 
111  if (auto constant = dyn_cast<ConstantOp>(srcOp))
112  outputPortConstants.push_back(constant.value());
113  else {
114  assert(isa<InvalidValueOp>(srcOp) && "only expect invalid");
115  outputPortConstants.push_back(None);
116  }
117 
118  // Erase connect op because we are going to remove this output ports.
119  op->erase();
120 
121  if (srcOp->use_empty())
122  srcOp->erase();
123  } else {
124  // Otherwise, we cannot remove the port.
125  continue;
126  }
127  }
128 
129  removalPortIndexes.push_back(index);
130  }
131 
132  // If there is nothing to remove, abort.
133  if (removalPortIndexes.empty())
134  return;
135 
136  // Delete ports from the module.
137  module.erasePorts(removalPortIndexes);
138  LLVM_DEBUG(llvm::for_each(removalPortIndexes, [&](unsigned index) {
139  llvm::dbgs() << "Delete port: " << ports[index].name << "\n";
140  }););
141 
142  // Rewrite all uses.
143  for (auto *use : instanceGraphNode->uses()) {
144  auto instance = ::cast<InstanceOp>(*use->getInstance());
145  ImplicitLocOpBuilder builder(instance.getLoc(), instance);
146  unsigned outputPortIndex = 0;
147  for (auto index : removalPortIndexes) {
148  auto result = instance.getResult(index);
149  assert(!ports[index].isInOut() && "don't expect inout ports");
150 
151  // If the port is input, replace the port with an unwritten wire
152  // so that we can remove use-chains in SV dialect canonicalization.
153  if (ports[index].isInput()) {
154  WireOp wire = builder.create<WireOp>(result.getType());
155 
156  // Check that the input port is only written. Sometimes input ports are
157  // used as temporary wires. In that case, we cannot erase connections.
158  bool onlyWritten = llvm::all_of(result.getUsers(), [&](Operation *op) {
159  if (auto connect = dyn_cast<FConnectLike>(op))
160  return connect.dest() == result;
161  return false;
162  });
163 
164  result.replaceUsesWithIf(wire, [&](OpOperand &op) -> bool {
165  // Connects can be deleted directly.
166  if (onlyWritten && isa<FConnectLike>(op.getOwner())) {
167  op.getOwner()->erase();
168  return false;
169  }
170  return true;
171  });
172 
173  // If the wire doesn't have an user, just erase it.
174  if (wire.use_empty())
175  wire.erase();
176 
177  continue;
178  }
179 
180  // Output port. Replace with the output port with an invalid or constant
181  // value.
182  auto portConstant = outputPortConstants[outputPortIndex++];
183  Value value;
184  if (portConstant)
185  value = builder.create<ConstantOp>(*portConstant);
186  else
187  value = builder.create<InvalidValueOp>(result.getType());
188 
189  result.replaceAllUsesWith(value);
190  }
191 
192  // Create a new instance op without unused ports.
193  instance.erasePorts(builder, removalPortIndexes);
194  // Remove old one.
195  instance.erase();
196  }
197 
198  numRemovedPorts += removalPortIndexes.size();
199 }
