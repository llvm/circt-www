8 // This pass makes a number of updates to the circuit that are required to match
9 // the behavior of the Scala FIRRTL Compiler (SFC). This pass removes invalid
10 // values from the circuit. This is a combination of the Scala FIRRTL
11 // Compiler's RemoveRests pass and RemoveValidIf. This is done to remove two
12 // "interpretations" of invalid. Namely: (1) registers that are initialized to
13 // an invalid value (module scoped and looking through wires and connects only)
14 // are converted to an unitialized register and (2) invalid values are converted
15 // to zero (after rule 1 is applied). Additionally, this pass checks and
16 // disallows async reset registers that are not driven with a constant when
17 // looking through wires, connects, and nodes.
18 //
19 //===----------------------------------------------------------------------===//

38 void SFCCompatPass::runOnOperation() {
39  LLVM_DEBUG(
40  llvm::dbgs() << "==----- Running SFCCompat "
41  "---------------------------------------------------===\n"
42  << "Module: '" << getOperation().getName() << "'\n";);
43 
44  bool madeModifications = false;
45  SmallVector<InvalidValueOp> invalidOps;
46  for (auto &op : llvm::make_early_inc_range(getOperation().getOps())) {
47  // Populate invalidOps for later handling.
48  if (auto inv = dyn_cast<InvalidValueOp>(op)) {
49  invalidOps.push_back(inv);
50  continue;
51  }
52  auto reg = dyn_cast<RegResetOp>(op);
53  if (!reg)
54  continue;
55 
56  // If the `RegResetOp` has an invalidated initialization, then replace it
57  // with a `RegOp`.
58  if (isModuleScopedDrivenBy<InvalidValueOp>(reg.resetValue(), true, false,
59  false)) {
60  LLVM_DEBUG(llvm::dbgs() << " - RegResetOp '" << reg.name()
61  << "' will be replaced with a RegOp\n");
62  ImplicitLocOpBuilder builder(reg.getLoc(), reg);
63  RegOp newReg = builder.create<RegOp>(
64  reg.getType(), reg.clockVal(), reg.name(), reg.nameKind(),
65  reg.annotations(), reg.inner_symAttr());
66  reg.replaceAllUsesWith(newReg.getResult());
67  reg.erase();
68  madeModifications = true;
69  continue;
70  }
71 
72  // If the `RegResetOp` has an asynchronous reset and the reset value is not
73  // a module-scoped constant when looking through wires and nodes, then
74  // generate an error. This implements the SFC's CheckResets pass.
75  if (!reg.resetSignal().getType().isa<AsyncResetType>())
76  continue;
77  if (isModuleScopedDrivenBy<ConstantOp, InvalidValueOp, SpecialConstantOp>(
78  reg.resetValue(), true, true, true))
79  continue;
80  auto resetDriver =
81  getModuleScopedDriver(reg.resetValue(), true, true, true);
82  auto diag = reg.emitOpError()
83  << "has an async reset, but its reset value is not driven with "
84  "a constant value through wires, nodes, or connects";
85  diag.attachNote(resetDriver.getLoc()) << "reset driver is here";
86  return signalPassFailure();
87  }
88 
89  // Convert all invalid values to zero.
90  for (auto inv : invalidOps) {
91  // Delete invalids which have no uses.
92  if (inv->getUses().empty()) {
93  inv->erase();
94  continue;
95  }
96  ImplicitLocOpBuilder builder(inv.getLoc(), inv);
97  Value replacement =
98  TypeSwitch<FIRRTLType, Value>(inv.getType())
99  .Case<ClockType, AsyncResetType, ResetType>(
100  [&](auto type) -> Value {
101  return builder.create<SpecialConstantOp>(
102  type, builder.getBoolAttr(false));
103  })
104  .Case<IntType>([&](IntType type) -> Value {
105  return builder.create<ConstantOp>(type, getIntZerosAttr(type));
106  })
107  .Case<BundleType, FVectorType>([&](auto type) -> Value {
108  auto width = circt::firrtl::getBitWidth(type);
109  assert(width && "width must be inferred");
110  auto zero = builder.create<ConstantOp>(APSInt(*width));
111  return builder.create<BitCastOp>(type, zero);
112  })
113  .Default([&](auto) {
114  llvm_unreachable("all types are supported");
115  return Value();
116  });
117  inv.replaceAllUsesWith(replacement);
118  inv.erase();
119  madeModifications = true;
120  }
121 
122  if (!madeModifications)
123  return markAllAnalysesPreserved();
124 }
