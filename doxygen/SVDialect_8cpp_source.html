51 /// Return a StringSet that contains all of the reserved names (e.g. Verilog
52 /// keywords) that we need to avoid for fear of name conflicts.
53 struct ReservedWordsCreator {
54  static void *call() {
55  auto set = std::make_unique<StringSet<>>();
56  static const char *const reservedWords[] = {
57 #include "ReservedWords.def"
58  };
59  for (auto *word : reservedWords)
60  set->insert(word);
61  return set.release();
62  }
63 };

69 /// Given string \p origName, generate a new name if it conflicts with any
70 /// keyword or any other name in the set \p recordNames. Use the int \p
71 /// nextGeneratedNameID as a counter for suffix. Update the \p recordNames with
72 /// the generated name and return the StringRef.
73 StringRef circt::sv::resolveKeywordConflict(StringRef origName,
74  llvm::StringSet<> &recordNames,
75  size_t &nextGeneratedNameID) {
76  // Get the list of reserved words we need to avoid. We could prepopulate this
77  // into the used words cache, but it is large and immutable, so we just query
78  // it when needed.
79 
80  // Fast path: name is valid
81  if (!reservedWords->count(origName)) {
82  auto itAndInserted = recordNames.insert(origName);
83  if (itAndInserted.second)
84  return itAndInserted.first->getKey();
85  }
86 
87  // We need to mutate the name, get the copy ready.
88  SmallString<16> nameBuffer(origName.begin(), origName.end());
89  nameBuffer.push_back('_');
90  auto baseSize = nameBuffer.size();
91 
92  while (1) {
93  // We need to auto-unique it.
94  auto suffix = llvm::utostr(nextGeneratedNameID++);
95  nameBuffer.append(suffix.begin(), suffix.end());
96 
97  // The name may be unique. No keywords have an underscore followed by a
98  // number, so don't check that again.
99  auto itAndInserted = recordNames.insert(nameBuffer);
100  if (itAndInserted.second)
101  return itAndInserted.first->getKey();
102 
103  // Chop off the suffix and try again until we get a unique name..
104  nameBuffer.resize(baseSize);
105  }
106 }

116 /// Legalize the specified name for use in SV output. Auto-uniquifies the name
117 /// through \c resolveKeywordConflict if required. If the name is empty, a
118 /// unique temp name is created.
119 StringRef circt::sv::legalizeName(StringRef name,
120  llvm::StringSet<> &recordNames,
121  size_t &nextGeneratedNameID) {
122  // Fastest path: empty name.
123  if (name.empty())
124  return resolveKeywordConflict("_GEN", recordNames, nextGeneratedNameID);
125 
126  // Check that the name is valid as the semi-fast path.
127  if (llvm::all_of(name, isValidVerilogCharacter) &&
128  isValidVerilogCharacterFirst(name.front()))
129  return resolveKeywordConflict(name, recordNames, nextGeneratedNameID);
130 
131  // The name consists of at least one invalid character. Escape it.
132  SmallString<16> tmpName;
133  if (!isValidVerilogCharacterFirst(name.front()) && name.front() != ' ' &&
134  name.front() != '.')
135  tmpName += '_';
136  for (char ch : name) {
137  if (isValidVerilogCharacter(ch))
138  tmpName += ch;
139  else if (ch == ' ' || ch == '.')
140  tmpName += '_';
141  else {
142  tmpName += llvm::utohexstr((unsigned char)ch);
143  }
144  }
145 
146  // Make sure the new valid name does not conflict with any existing names.
147  return resolveKeywordConflict(tmpName, recordNames, nextGeneratedNameID);
148 }

150 /// Check if a name is valid for use in SV output by only containing characters
151 /// allowed in SV identifiers.
152 ///
153 /// Call \c legalizeName() to obtain a legalized version of the name.
154 bool circt::sv::isNameValid(StringRef name) {
155  if (name.empty())
156  return false;
157  if (!isValidVerilogCharacterFirst(name.front()))
158  return false;
159  for (char ch : name) {
160  if (!isValidVerilogCharacter(ch))
161  return false;
162  }
163  return reservedWords->count(name) == 0;
164 }
