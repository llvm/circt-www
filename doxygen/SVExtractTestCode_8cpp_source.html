35 // Reimplemented from SliceAnalysis to use a worklist rather than recursion and
36 // non-insert ordered set.
37 static void getBackwardSliceSimple(Operation *rootOp,
38  SetVector<Operation *> &backwardSlice,
39  std::function<bool(Operation *)> filter) {
40  SmallVector<Operation *> worklist;
41  worklist.push_back(rootOp);
42 
43  while (!worklist.empty()) {
44  Operation *op = worklist.back();
45  worklist.pop_back();
46 
47  if (!op || op->hasTrait<mlir::OpTrait::IsIsolatedFromAbove>())
48  continue;
49 
50  // Evaluate whether we should keep this def.
51  // This is useful in particular to implement scoping; i.e. return the
52  // transitive backwardSlice in the current scope.
53  if (filter && !filter(op))
54  continue;
55 
56  for (auto en : llvm::enumerate(op->getOperands())) {
57  auto operand = en.value();
58  if (auto *definingOp = operand.getDefiningOp()) {
59  if (!backwardSlice.contains(definingOp))
60  worklist.push_back(definingOp);
61  } else if (auto blockArg = operand.dyn_cast<BlockArgument>()) {
62  Block *block = blockArg.getOwner();
63  Operation *parentOp = block->getParentOp();
64  // TODO: determine whether we want to recurse backward into the other
65  // blocks of parentOp, which are not technically backward unless they
66  // flow into us. For now, just bail.
67  assert(parentOp->getNumRegions() == 1 &&
68  parentOp->getRegion(0).getBlocks().size() == 1);
69  if (!backwardSlice.contains(parentOp))
70  worklist.push_back(parentOp);
71  } else {
72  llvm_unreachable("No definingOp and not a block argument.");
73  }
74  }
75 
76  backwardSlice.insert(op);
77  }
78 
79  // Don't insert the top level operation, we just queried on it and don't
80  // want it in the results.
81  backwardSlice.remove(rootOp);
82 }

84 // Compute the dataflow for a set of ops.
85 static void dataflowSlice(SetVector<Operation *> &ops,
86  SetVector<Operation *> &results) {
87  for (auto op : ops) {
88  getBackwardSliceSimple(op, results, [](Operation *testOp) -> bool {
89  return !isa<sv::ReadInOutOp>(testOp) && !isa<hw::InstanceOp>(testOp) &&
90  !isa<sv::PAssignOp>(testOp) && !isa<sv::BPAssignOp>(testOp);
91  });
92  }
93 }

95 // Compute the ops defining the blocks a set of ops are in.
96 static void blockSlice(SetVector<Operation *> &ops,
97  SetVector<Operation *> &blocks) {
98  for (auto op : ops) {
99  while (!isa<hw::HWModuleOp>(op->getParentOp())) {
100  op = op->getParentOp();
101  blocks.insert(op);
102  }
103  }
104 }

106 // Aggressively mark operations to be moved to the new module. This leaves
107 // maximum flexibility for optimization after removal of the nodes from the
108 // old module.
109 static SetVector<Operation *> computeCloneSet(SetVector<Operation *> &roots) {
110  SetVector<Operation *> results;
111  // Get Dataflow for roots
112  dataflowSlice(roots, results);
113 
114  // Get Blocks
115  SetVector<Operation *> blocks;
116  blockSlice(roots, blocks);
117  blockSlice(results, blocks);
118 
119  // Make sure dataflow to block args (if conds, etc) is included
120  dataflowSlice(blocks, results);
121 
122  // include the blocks and roots to clone
123  results.insert(roots.begin(), roots.end());
124  results.insert(blocks.begin(), blocks.end());
125 
126  return results;
127 }

129 static StringRef getNameForPort(Value val, ArrayAttr modulePorts) {
130  if (auto readinout = dyn_cast_or_null<ReadInOutOp>(val.getDefiningOp())) {
131  if (auto wire = dyn_cast<WireOp>(readinout.input().getDefiningOp()))
132  return wire.name();
133  if (auto reg = dyn_cast<RegOp>(readinout.input().getDefiningOp()))
134  return reg.name();
135  } else if (auto bv = val.dyn_cast<BlockArgument>()) {
136  return modulePorts[bv.getArgNumber()].cast<StringAttr>().getValue();
137  }
138  return "";

141 // Given a set of values, construct a module and bind instance of that module
142 // that passes those values through. Returns the new module and the instance
143 // pointing to it.
144 static hw::HWModuleOp createModuleForCut(hw::HWModuleOp op,
145  SetVector<Value> &inputs,
146  BlockAndValueMapping &cutMap,
147  StringRef suffix, Attribute path,
148  Attribute fileName) {
149  // Create the extracted module right next to the original one.
150  OpBuilder b(op);
151 
152  // Construct the ports, this is just the input Values
153  SmallVector<hw::PortInfo> ports;
154  {
155  auto srcPorts = op.argNames();
156  for (auto port : llvm::enumerate(inputs)) {
157  auto name = getNameForPort(port.value(), srcPorts);
158  ports.push_back({b.getStringAttr(name), hw::PortDirection::INPUT,
159  port.value().getType(), port.index()});
160  }
161  }
162 
163  // Create the module, setting the output path if indicated.
164  auto newMod = b.create<hw::HWModuleOp>(
165  op.getLoc(),
166  b.getStringAttr(getVerilogModuleNameAttr(op).getValue() + suffix), ports);
167  if (path)
168  newMod->setAttr("output_file", path);
169  newMod.commentAttr(b.getStringAttr("VCS coverage exclude_file"));
170 
171  // Update the mapping from old values to cloned values
172  for (auto port : llvm::enumerate(inputs))
173  cutMap.map(port.value(), newMod.body().getArgument(port.index()));
174  cutMap.map(op.getBodyBlock(), newMod.getBodyBlock());
175 
176  // Add an instance in the old module for the extracted module
177  b = OpBuilder::atBlockTerminator(op.getBodyBlock());
178  auto inst = b.create<hw::InstanceOp>(
179  op.getLoc(), newMod, newMod.getName(), inputs.getArrayRef(), ArrayAttr(),
180  b.getStringAttr(
181  ("__ETC_" + getVerilogModuleNameAttr(op).getValue() + suffix).str()));
182  inst->setAttr("doNotPrint", b.getBoolAttr(true));
183  b = OpBuilder::atBlockEnd(
184  &op->getParentOfType<mlir::ModuleOp>()->getRegion(0).front());
185 
186  auto bindOp =
187  b.create<sv::BindOp>(op.getLoc(), op.getNameAttr(), inst.inner_symAttr());
188  if (fileName)
189  bindOp->setAttr("output_file", fileName);
190  return newMod;
191 }

201 // Shallow clone, which we use to not clone the content of blocks, doesn't
202 // clone the regions, so create all the blocks we need and update the mapping.
203 static void addBlockMapping(BlockAndValueMapping &cutMap, Operation *oldOp,
204  Operation *newOp) {
205  assert(oldOp->getNumRegions() == newOp->getNumRegions());
206  for (size_t i = 0, e = oldOp->getNumRegions(); i != e; ++i) {
207  auto &oldRegion = oldOp->getRegion(i);
208  auto &newRegion = newOp->getRegion(i);
209  for (auto oi = oldRegion.begin(), oe = oldRegion.end(); oi != oe; ++oi) {
210  cutMap.map(&*oi, &newRegion.emplaceBlock());
211  }
212  }
213 }

215 static bool hasOoOArgs(hw::HWModuleOp newMod, Operation *op) {
216  for (auto arg : op->getOperands()) {
217  auto argOp = arg.getDefiningOp(); // may be null
218  if (!argOp)
219  continue;
220  if (argOp->getParentOfType<hw::HWModuleOp>() != newMod)
221  return true;
222  }
223  return false;
224 }

226 // Do the cloning, which is just a pre-order traversal over the module looking
227 // for marked ops.
228 static void migrateOps(hw::HWModuleOp oldMod, hw::HWModuleOp newMod,
229  SetVector<Operation *> &depOps,
230  BlockAndValueMapping &cutMap) {
231  SmallVector<Operation *, 16> lateBoundOps;
232  OpBuilder b = OpBuilder::atBlockBegin(newMod.getBodyBlock());
233  oldMod.walk<WalkOrder::PreOrder>([&](Operation *op) {
234  if (depOps.count(op)) {
235  setInsertPointToEndOrTerminator(b, cutMap.lookup(op->getBlock()));
236  auto newOp = b.cloneWithoutRegions(*op, cutMap);
237  addBlockMapping(cutMap, op, newOp);
238  if (hasOoOArgs(newMod, newOp))
239  lateBoundOps.push_back(newOp);
240  }
241  });
242  // update any operand which was emitted before it's defining op was.
243  for (auto op : lateBoundOps)
244  for (unsigned argidx = 0, e = op->getNumOperands(); argidx < e; ++argidx) {
245  Value arg = op->getOperand(argidx);
246  if (cutMap.contains(arg))
247  op->setOperand(argidx, cutMap.lookup(arg));
248  }
249 }

257 struct SVExtractTestCodeImplPass
258  : public SVExtractTestCodeBase<SVExtractTestCodeImplPass> {
259  void runOnOperation() override;
260 
261 private:
262  void doModule(hw::HWModuleOp module, std::function<bool(Operation *)> fn,
263  StringRef suffix, Attribute path, Attribute bindFile) {
264  bool hasError = false;
265  // Find Operations of interest.
266  SetVector<Operation *> roots;
267  module->walk([&fn, &roots, &hasError](Operation *op) {
268  if (fn(op)) {
269  roots.insert(op);
270  if (op->getNumResults()) {
271  op->emitError("Extracting op with result");
272  hasError = true;
273  }
274  }
275  });
276  if (hasError) {
277  signalPassFailure();
278  return;
279  }
280  // No Ops? No problem.
281  if (roots.empty())
282  return;
283 
284  // Find the data-flow and structural ops to clone. Result includes roots.
285  auto opsToClone = computeCloneSet(roots);
286  // Find the dataflow into the clone set
287  SetVector<Value> inputs;
288  for (auto op : opsToClone)
289  for (auto arg : op->getOperands()) {
290  auto argOp = arg.getDefiningOp(); // may be null
291  if (!opsToClone.count(argOp))
292  inputs.insert(arg);
293  }
294 
295  // Make a module to contain the clone set, with arguments being the cut
296  BlockAndValueMapping cutMap;
297  auto bmod =
298  createModuleForCut(module, inputs, cutMap, suffix, path, bindFile);
299  // do the clone
300  migrateOps(module, bmod, opsToClone, cutMap);
301  // erase old operations of interest
302  for (auto op : roots)
303  op->erase();
304  }
305 };

309 void SVExtractTestCodeImplPass::runOnOperation() {
310  auto top = getOperation();
311  auto *topLevelModule = top.getBody();
312  auto assertDir =
313  top->getAttrOfType<hw::OutputFileAttr>("firrtl.extract.assert");
314  auto assumeDir =
315  top->getAttrOfType<hw::OutputFileAttr>("firrtl.extract.assume");
316  auto coverDir =
317  top->getAttrOfType<hw::OutputFileAttr>("firrtl.extract.cover");
318  auto assertBindFile =
319  top->getAttrOfType<hw::OutputFileAttr>("firrtl.extract.assert.bindfile");
320  auto assumeBindFile =
321  top->getAttrOfType<hw::OutputFileAttr>("firrtl.extract.assume.bindfile");
322  auto coverBindFile =
323  top->getAttrOfType<hw::OutputFileAttr>("firrtl.extract.cover.bindfile");
324 
325  hw::HWSymbolCache symCache;
326  symCache.addDefinitions(top);
327  symCache.freeze();
328 
329  // Symbols not in the cache will only be fore instances added by an extract
330  // phase and are not instances that could possibly have extract flags on them.
331  auto isAssert = [&symCache](Operation *op) -> bool {
332  if (auto inst = dyn_cast<hw::InstanceOp>(op))
333  if (auto mod = symCache.getDefinition(inst.moduleNameAttr()))
334  if (mod->getAttr("firrtl.extract.assert.extra"))
335  return true;
336 
337  // If the format of assert is "ifElseFatal", PrintOp is lowered into
338  // ErrorOp. So we have to check message contents whether they encode
339  // verifications. See FIRParserAsserts for more details.
340  if (auto error = dyn_cast<ErrorOp>(op)) {
341  if (auto message = error.message())
342  return message.getValue().startswith("assert:") ||
343  message.getValue().startswith("Assertion failed") ||
344  message.getValue().startswith("assertNotX:") ||
345  message.getValue().contains("[verif-library-assert]");
346  return false;
347  }
348 
349  return isa<AssertOp>(op) || isa<FinishOp>(op) || isa<FWriteOp>(op) ||
350  isa<AssertConcurrentOp>(op) || isa<FatalOp>(op);
351  };
352  auto isAssume = [&symCache](Operation *op) -> bool {
353  if (auto inst = dyn_cast<hw::InstanceOp>(op))
354  if (auto mod = symCache.getDefinition(inst.moduleNameAttr()))
355  if (mod->getAttr("firrtl.extract.assume.extra"))
356  return true;
357  return isa<AssumeOp>(op) || isa<AssumeConcurrentOp>(op);
358  };
359  auto isCover = [&symCache](Operation *op) -> bool {
360  if (auto inst = dyn_cast<hw::InstanceOp>(op))
361  if (auto mod = symCache.getDefinition(inst.moduleNameAttr()))
362  if (mod->getAttr("firrtl.extract.cover.extra"))
363  return true;
364  return isa<CoverOp>(op) || isa<CoverConcurrentOp>(op);
365  };
366 
367  auto &instanceGraph = getAnalysis<circt::hw::InstanceGraph>();
368  auto isBound = [&instanceGraph](hw::HWModuleLike op) -> bool {
369  auto *node = instanceGraph.lookup(op);
370  return llvm::any_of(node->uses(), [](hw::InstanceRecord *a) {
371  auto inst = a->getInstance();
372  if (!inst)
373  return false;
374  return inst->hasAttr("doNotPrint");
375  });
376  };
377 
378  for (auto &op : topLevelModule->getOperations()) {
379  if (auto rtlmod = dyn_cast<hw::HWModuleOp>(op)) {
380  // Extract two sets of ops to different modules. This will add modules,
381  // but not affect modules in the symbol table. If any instance of the
382  // module is bound, then extraction is skipped. This avoids problems
383  // where certain simulators dislike having binds that target bound
384  // modules.
385  if (isBound(rtlmod))
386  continue;
387 
388  // In the module is in test harness, we don't have to extract from it.
389  if (rtlmod->hasAttr("firrtl.extract.do_not_extract")) {
390  rtlmod->removeAttr("firrtl.extract.do_not_extract");
391  continue;
392  }
393 
394  doModule(rtlmod, isAssert, "_assert", assertDir, assertBindFile);
395  doModule(rtlmod, isAssume, "_assume", assumeDir, assumeBindFile);
396  doModule(rtlmod, isCover, "_cover", coverDir, coverBindFile);
397  }
398  }
399  // We have to wait until all the instances are processed to clean up the
400  // annotations.
401  for (auto &op : topLevelModule->getOperations())
402  if (isa<hw::HWModuleOp, hw::HWModuleExternOp>(op)) {
403  op.removeAttr("firrtl.extract.assert.extra");
404  op.removeAttr("firrtl.extract.cover.extra");
405  op.removeAttr("firrtl.extract.assume.extra");
406  }
407 }
