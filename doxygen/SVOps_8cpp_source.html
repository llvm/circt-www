50 /// Returns the operation registered with the given symbol name with the regions
51 /// of 'symbolTableOp'. recurse through nested regions which don't contain the
52 /// symboltable trait. Returns nullptr if no valid symbol was found.
53 static Operation *lookupSymbolInNested(Operation *symbolTableOp,
54  StringRef symbol) {
55  Region &region = symbolTableOp->getRegion(0);
56  if (region.empty())
57  return nullptr;
58 
59  // Look for a symbol with the given name.
60  StringAttr symbolNameId = StringAttr::get(symbolTableOp->getContext(),
61  SymbolTable::getSymbolAttrName());
62  for (Block &block : region)
63  for (Operation &nestedOp : block) {
64  auto nameAttr = nestedOp.getAttrOfType<StringAttr>(symbolNameId);
65  if (nameAttr && nameAttr.getValue() == symbol)
66  return &nestedOp;
67  if (!nestedOp.hasTrait<OpTrait::SymbolTable>() &&
68  nestedOp.getNumRegions()) {
69  if (auto *nop = lookupSymbolInNested(&nestedOp, symbol))
70  return nop;
71  }
72  }
73  return nullptr;
74 }

80 static ParseResult parseImplicitSSAName(OpAsmParser &parser,
81  NamedAttrList &resultAttrs) {
82 
83  if (parser.parseOptionalAttrDict(resultAttrs))
84  return failure();
85 
86  // If the attribute dictionary contains no 'name' attribute, infer it from
87  // the SSA name (if specified).
88  bool hadName = llvm::any_of(resultAttrs, [](NamedAttribute attr) {
89  return attr.getName() == "name";
90  });
91 
92  // If there was no name specified, check to see if there was a useful name
93  // specified in the asm file.
94  if (hadName)
95  return success();
96 
97  // If there is no explicit name attribute, get it from the SSA result name.
98  // If numeric, just use an empty name.
99  auto resultName = parser.getResultName(0).first;
100  if (!resultName.empty() && isdigit(resultName[0]))
101  resultName = "";
102  auto nameAttr = parser.getBuilder().getStringAttr(resultName);
103  auto *context = parser.getBuilder().getContext();
104  resultAttrs.push_back({StringAttr::get(context, "name"), nameAttr});
105  return success();
106 }

108 static void printImplicitSSAName(OpAsmPrinter &p, Operation *op,
109  DictionaryAttr attr) {
110  // Note that we only need to print the "name" attribute if the asmprinter
111  // result name disagrees with it. This can happen in strange cases, e.g.
112  // when there are conflicts.
113  bool namesDisagree = false;
114 
115  SmallString<32> resultNameStr;
116  llvm::raw_svector_ostream tmpStream(resultNameStr);
117  p.printOperand(op->getResult(0), tmpStream);
118  auto expectedName = op->getAttrOfType<StringAttr>("name").getValue();
119  auto actualName = tmpStream.str().drop_front();
120  if (actualName != expectedName) {
121  // Anonymous names are printed as digits, which is fine.
122  if (!expectedName.empty() || !isdigit(actualName[0]))
123  namesDisagree = true;
124  }
125 
126  if (namesDisagree)
127  p.printOptionalAttrDict(op->getAttrs(),
128  {SymbolTable::getSymbolAttrName(),
129  hw::InnerName::getInnerNameAttrName(),
130  "svAttributes"});
131  else
132  p.printOptionalAttrDict(op->getAttrs(),
133  {"name", SymbolTable::getSymbolAttrName(),
134  hw::InnerName::getInnerNameAttrName(),
135  "svAttributes"});
136 }

400 static void replaceOpWithRegion(PatternRewriter &rewriter, Operation *op,

402  assert(llvm::hasSingleElement(region) && "expected single-region block");
403  Block *fromBlock = &region.front();

409 LogicalResult IfOp::canonicalize(IfOp op, PatternRewriter &rewriter) {
410  if (auto constant = op.cond().getDefiningOp<hw::ConstantOp>()) {
411 
412  if (constant.getValue().isAllOnesValue())
413  replaceOpWithRegion(rewriter, op, op.thenRegion());
414  else if (!op.elseRegion().empty())
415  replaceOpWithRegion(rewriter, op, op.elseRegion());
416 
417  rewriter.eraseOp(op);
418 
419  return success();
420  }
421 
422  // Erase empty if's.
423 
424  // If there is stuff in the then block, leave this operation alone.
425  if (!op.getThenBlock()->empty())
426  return failure();
427 
428  // If not and there is no else, then this operation is just useless.
429  if (!op.hasElse() || op.getElseBlock()->empty()) {
430  rewriter.eraseOp(op);
431  return success();
432  }
433 
434  // Otherwise, invert the condition and move the 'else' block to the 'then'
435  // region.
436  auto cond = comb::createOrFoldNot(op.getLoc(), op.cond(), rewriter);
437  op.setOperand(cond);
438 
439  auto *thenBlock = op.getThenBlock(), *elseBlock = op.getElseBlock();
440 
441  // Move the body of the then block over to the else.
442  thenBlock->getOperations().splice(thenBlock->end(),
443  elseBlock->getOperations());
444  rewriter.eraseBlock(elseBlock);
445  return success();
446 }

486 static ParseResult parseEventList(
487  OpAsmParser &p, Attribute &eventsAttr,
488  SmallVectorImpl<OpAsmParser::UnresolvedOperand> &clocksOperands) {
489 
490  // Parse zero or more conditions intoevents and clocksOperands.
491  SmallVector<Attribute> events;
492 
493  auto loc = p.getCurrentLocation();
494  StringRef keyword;
495  if (!p.parseOptionalKeyword(&keyword)) {
496  while (1) {
497  auto kind = symbolizeEventControl(keyword);
498  if (!kind.hasValue())
499  return p.emitError(loc, "expected 'posedge', 'negedge', or 'edge'");
500  auto eventEnum = static_cast<int32_t>(kind.getValue());
501  events.push_back(p.getBuilder().getI32IntegerAttr(eventEnum));
502 
503  clocksOperands.push_back({});
504  if (p.parseOperand(clocksOperands.back()))
505  return failure();
506 
507  if (failed(p.parseOptionalComma()))
508  break;
509  if (p.parseKeyword(&keyword))
510  return failure();
511  }
512  }
513  eventsAttr = p.getBuilder().getArrayAttr(events);
514  return success();
515 }

676  }

678  intAttr = IntegerAttr::get(patternType, pattern);

712 /// Parse case op.
713 /// case op ::= `sv.case` case-style? validation-qualifier? cond `:` type
714 /// attr-dict case-pattern^*
715 /// case-style ::= `case` | `casex` | `casez`
716 /// validation-qualifier (see SV Spec 12.5.3) ::= `unique` | `unique0`
717 /// | `priority`
718 /// case-pattern ::= `case` bit-pattern `:` region
719 ParseResult CaseOp::parse(OpAsmParser &parser, OperationState &result) {
720  auto &builder = parser.getBuilder();
721 
722  OpAsmParser::UnresolvedOperand condOperand;
723  Type condType;
724 
725  auto loc = parser.getCurrentLocation();
726 
727  StringRef keyword;
728  if (!parser.parseOptionalKeyword(&keyword, {"case", "casex", "casez"})) {
729  auto kind = symbolizeCaseStmtType(keyword);
730  auto caseEnum = static_cast<int32_t>(kind.getValue());
731  result.addAttribute("caseStyle", builder.getI32IntegerAttr(caseEnum));
732  }
733 
734  // Parse validation qualifier.
735  if (!parser.parseOptionalKeyword(
736  &keyword, {"plain", "priority", "unique", "unique0"})) {
737  auto kind = symbolizeValidationQualifierTypeEnum(keyword);
738  result.addAttribute("validationQualifier",
739  ValidationQualifierTypeEnumAttr::get(
740  builder.getContext(), kind.getValue()));
741  }
742 
743  if (parser.parseOperand(condOperand) || parser.parseColonType(condType) ||
744  parser.parseOptionalAttrDict(result.attributes) ||
745  parser.resolveOperand(condOperand, condType, result.operands))
746  return failure();
747 
748  // Check the integer type.
749  hw::EnumType enumType = condType.dyn_cast<hw::EnumType>();
750  unsigned condWidth = 0;
751  if (!enumType) {
752  if (!result.operands[0].getType().isSignlessInteger())
753  return parser.emitError(loc, "condition must have signless integer type");
754  condWidth = condType.getIntOrFloatBitWidth();
755  }
756 
757  // Parse all the cases.
758  SmallVector<Attribute> casePatterns;
759  SmallVector<CasePatternBit, 16> caseBits;
760  while (1) {
761  if (succeeded(parser.parseOptionalKeyword("default"))) {
762  casePatterns.push_back(CaseDefaultPattern(parser.getContext()).attr());
763  } else if (failed(parser.parseOptionalKeyword("case"))) {
764  // Not default or case, must be the end of the cases.
765  break;
766  } else if (enumType) {
767  // Enumerated case; parse the case value.
768  StringRef caseVal;
769 
770  if (parser.parseKeyword(&caseVal))
771  return failure();
772 
773  if (!enumType.contains(caseVal))
774  return parser.emitError(loc)
775  << "case value '" + caseVal + "' is not a member of enum type "
776  << enumType;
777  casePatterns.push_back(
778  hw::EnumFieldAttr::get(parser.getEncodedSourceLoc(loc),
779  builder.getStringAttr(caseVal), enumType));
780  } else {
781  // Parse the pattern. It always starts with b, so it is an MLIR
782  // keyword.
783  StringRef caseVal;
784  loc = parser.getCurrentLocation();
785  if (parser.parseKeyword(&caseVal))
786  return failure();
787 
788  if (caseVal.front() != 'b')
789  return parser.emitError(loc, "expected case value starting with 'b'");
790  caseVal = caseVal.drop_front();
791 
792  // Parse and decode each bit, we reverse the list later for MSB->LSB.
793  for (; !caseVal.empty(); caseVal = caseVal.drop_front()) {
794  CasePatternBit bit;
795  switch (caseVal.front()) {
796  case '0':
797  bit = CasePatternBit::Zero;
798  break;
799  case '1':
800  bit = CasePatternBit::One;
801  break;
802  case 'x':
803  bit = CasePatternBit::AnyX;
804  break;
805  case 'z':
806  bit = CasePatternBit::AnyZ;
807  break;
808  default:
809  return parser.emitError(loc, "unexpected case bit '")
810  << caseVal.front() << "'";
811  }
812  caseBits.push_back(bit);
813  }
814 
815  if (caseVal.size() > condWidth)
816  return parser.emitError(loc, "too many bits specified in pattern");
817  std::reverse(caseBits.begin(), caseBits.end());
818 
819  // High zeros may be missing.
820  if (caseBits.size() < condWidth)
821  caseBits.append(condWidth - caseBits.size(), CasePatternBit::Zero);
822 
823  auto resultPattern = CaseBitPattern(caseBits, builder.getContext());
824  casePatterns.push_back(resultPattern.attr());
825  caseBits.clear();
826  }
827 
828  // Parse the case body.
829  auto caseRegion = std::make_unique<Region>();
830  if (parser.parseColon() || parser.parseRegion(*caseRegion))
831  return failure();
832  result.addRegion(std::move(caseRegion));
833  }
834 
835  result.addAttribute("casePatterns", builder.getArrayAttr(casePatterns));
836  return success();
837 }

914 // Strength reduce case styles based on the bit patterns.
915 LogicalResult CaseOp::canonicalize(CaseOp op, PatternRewriter &rewriter) {
916  if (op.caseStyle() == CaseStmtType::CaseStmt)
917  return failure();
918  if (op.cond().getType().isa<hw::EnumType>())
919  return failure();
920 
921  auto caseInfo = op.getCases();
922  bool noXZ = llvm::all_of(caseInfo, [](const CaseInfo &ci) {
923  return !ci.pattern.get()->hasX() && !ci.pattern.get()->hasZ();
924  });
925  bool noX = llvm::all_of(caseInfo, [](const CaseInfo &ci) {
926  if (isa<CaseDefaultPattern>(ci.pattern))
927  return true;
928  return !ci.pattern.get()->hasX();
929  });
930  bool noZ = llvm::all_of(caseInfo, [](const CaseInfo &ci) {
931  if (isa<CaseDefaultPattern>(ci.pattern))
932  return true;
933  return !ci.pattern.get()->hasZ();
934  });
935 
936  if (op.caseStyle() == CaseStmtType::CaseXStmt) {
937  if (noXZ) {
938  rewriter.updateRootInPlace(op, [&]() {
939  op.caseStyleAttr(
940  CaseStmtTypeAttr::get(op.getContext(), CaseStmtType::CaseStmt));
941  });
942  return success();
943  }
944  if (noX) {
945  rewriter.updateRootInPlace(op, [&]() {
946  op.caseStyleAttr(
947  CaseStmtTypeAttr::get(op.getContext(), CaseStmtType::CaseZStmt));
948  });
949  return success();
950  }
951  }
952 
953  if (op.caseStyle() == CaseStmtType::CaseZStmt && noZ) {
954  rewriter.updateRootInPlace(op, [&]() {
955  op.caseStyleAttr(
956  CaseStmtTypeAttr::get(op.getContext(), CaseStmtType::CaseStmt));
957  });
958  return success();
959  }
960 
961  return failure();
962 }

1248 // If this wire is only written to, delete the wire and all writers.
1249 LogicalResult WireOp::canonicalize(WireOp wire, PatternRewriter &rewriter) {
1250  // If the wire has a symbol, then we can't delete it.
1251  if (wire.inner_symAttr())
1252  return failure();
1253 
1254  // Wires have inout type, so they'll have assigns and read_inout operations
1255  // that work on them. If anything unexpected is found then leave it alone.
1256  SmallVector<sv::ReadInOutOp> reads;
1257  sv::AssignOp write;
1258 
1259  for (auto *user : wire->getUsers()) {
1260  if (auto read = dyn_cast<sv::ReadInOutOp>(user)) {
1261  reads.push_back(read);
1262  continue;
1263  }
1264 
1265  // Otherwise must be an assign, and we must not have seen a write yet.
1266  auto assign = dyn_cast<sv::AssignOp>(user);
1267  // Either the wire has more than one write or another kind of Op (other than
1268  // AssignOp and ReadInOutOp), then can't optimize.
1269  if (!assign || write)
1270  return failure();
1271 
1272  // If the assign op has SV attributes, we don't want to delete the
1273  // assignment.
1274  if (assign.svAttributesAttr())
1275  return failure();
1276 
1277  write = assign;
1278  }
1279 
1280  Value connected;
1281  if (!write) {
1282  // If no write and only reads, then replace with XOp.
1283  connected = rewriter.create<ConstantXOp>(
1284  wire.getLoc(),
1285  wire.getResult().getType().cast<InOutType>().getElementType());
1286  } else if (isa<hw::HWModuleOp>(write->getParentOp()))
1287  connected = write.src();
1288  else
1289  // If the write is happening at the module level then we don't have any
1290  // use-before-def checking to do, so we only handle that for now.
1291  return failure();
1292

1457 // reg s <= cond ? val : s simplification.
1458 // Don't assign a register's value to itself, conditionally assign the new value
1459 // instead.
1460 LogicalResult PAssignOp::canonicalize(PAssignOp op, PatternRewriter &rewriter) {
1461  auto mux = op.src().getDefiningOp<comb::MuxOp>();
1462  if (!mux)
1463  return failure();
1464 
1465  auto reg = dyn_cast<sv::RegOp>(op.dest().getDefiningOp());
1466  if (!reg)
1467  return failure();
1468 
1469  bool trueBranch; // did we find the register on the true branch?
1470  auto tvread = mux.trueValue().getDefiningOp<sv::ReadInOutOp>();
1471  auto fvread = mux.falseValue().getDefiningOp<sv::ReadInOutOp>();
1472  if (tvread && reg == tvread.input().getDefiningOp<sv::RegOp>())
1473  trueBranch = true;
1474  else if (fvread && reg == fvread.input().getDefiningOp<sv::RegOp>())
1475  trueBranch = false;
1476  else
1477  return failure();
1478 
1479  // Check that this is the only write of the register
1480  for (auto &use : reg->getUses()) {
1481  if (isa<ReadInOutOp>(use.getOwner()))
1482  continue;
1483  if (use.getOwner() == op)
1484  continue;
1485  return failure();
1486  }
1487 
1488  // Replace a non-blocking procedural assign in a procedural region with a
1489  // conditional procedural assign. We've ensured that this is the only write
1490  // of the register.
1491  if (trueBranch) {
1492  auto cond = comb::createOrFoldNot(mux.getLoc(), mux.cond(), rewriter);
1493  rewriter.create<sv::IfOp>(mux.getLoc(), cond, [&]() {
1494  rewriter.create<PAssignOp>(op.getLoc(), reg, mux.falseValue());
1495  });
1496  } else {
1497  rewriter.create<sv::IfOp>(mux.getLoc(), mux.cond(), [&]() {
1498  rewriter.create<PAssignOp>(op.getLoc(), reg, mux.trueValue());
1499  });
1500  }
1501 
1502  // Remove the wire.
1503  rewriter.eraseOp(op);
1504  return success();
1505 }
