46 /// Returns the operation registered with the given symbol name with the regions
47 /// of 'symbolTableOp'. recurse through nested regions which don't contain the
48 /// symboltable trait. Returns nullptr if no valid symbol was found.
49 static Operation *lookupSymbolInNested(Operation *symbolTableOp,
50  StringRef symbol) {
51  Region &region = symbolTableOp->getRegion(0);
52  if (region.empty())
53  return nullptr;
54 
55  // Look for a symbol with the given name.
56  Identifier symbolNameId = Identifier::get(SymbolTable::getSymbolAttrName(),
57  symbolTableOp->getContext());
58  for (Block &block : region)
59  for (Operation &nestedOp : block) {
60  auto nameAttr = nestedOp.getAttrOfType<StringAttr>(symbolNameId);
61  if (nameAttr && nameAttr.getValue() == symbol)
62  return &nestedOp;
63  if (!nestedOp.hasTrait<OpTrait::SymbolTable>() &&
64  nestedOp.getNumRegions()) {
65  if (auto *nop = lookupSymbolInNested(&nestedOp, symbol))
66  return nop;
67  }
68  }
69  return nullptr;
70 }

76 static ParseResult parseImplicitSSAName(OpAsmParser &parser,
77  NamedAttrList &resultAttrs) {
78 
79  if (parser.parseOptionalAttrDict(resultAttrs))
80  return failure();
81 
82  // If the attribute dictionary contains no 'name' attribute, infer it from
83  // the SSA name (if specified).
84  bool hadName = llvm::any_of(
85  resultAttrs, [](NamedAttribute attr) { return attr.first == "name"; });
86 
87  // If there was no name specified, check to see if there was a useful name
88  // specified in the asm file.
89  if (hadName)
90  return success();
91 
92  // If there is no explicit name attribute, get it from the SSA result name.
93  // If numeric, just use an empty name.
94  auto resultName = parser.getResultName(0).first;
95  if (!resultName.empty() && isdigit(resultName[0]))
96  resultName = "";
97  auto nameAttr = parser.getBuilder().getStringAttr(resultName);
98  auto *context = parser.getBuilder().getContext();
99  resultAttrs.push_back({Identifier::get("name", context), nameAttr});
100  return success();
101 }

103 static void printImplicitSSAName(OpAsmPrinter &p, Operation *op,
104  DictionaryAttr attr) {
105  // Note that we only need to print the "name" attribute if the asmprinter
106  // result name disagrees with it. This can happen in strange cases, e.g.
107  // when there are conflicts.
108  bool namesDisagree = false;
109 
110  SmallString<32> resultNameStr;
111  llvm::raw_svector_ostream tmpStream(resultNameStr);
112  p.printOperand(op->getResult(0), tmpStream);
113  auto expectedName = op->getAttrOfType<StringAttr>("name").getValue();
114  auto actualName = tmpStream.str().drop_front();
115  if (actualName != expectedName) {
116  // Anonymous names are printed as digits, which is fine.
117  if (!expectedName.empty() || !isdigit(actualName[0]))
118  namesDisagree = true;
119  }
120 
121  if (namesDisagree)
122  p.printOptionalAttrDict(op->getAttrs(), {"sym_name"});
123  else
124  p.printOptionalAttrDict(op->getAttrs(), {"name", "sym_name"});
125 }

132 static void

134  function_ref<void(Value, StringRef)> setNameFn) {
135  // If the string is macro like, then use a pretty name. We only take the

144  if (!name.empty())
145  setNameFn(op->getResult(0), name);
146 }

201 // If this reg is only written to, delete the reg and all writers.
202 LogicalResult RegOp::canonicalize(RegOp op, PatternRewriter &rewriter) {
203  // If the reg has a symbol, then we can't delete it.
204  if (op.sym_nameAttr())
205  return failure();
206  // Check that all operations on the wire are sv.assigns. All other wire
207  // operations will have been handled by other canonicalization.
208  for (auto &use : op.getResult().getUses())
209  if (!isa<AssignOp>(use.getOwner()))
210  return failure();
211 
212  // Remove all uses of the wire.
213  for (auto &use : op.getResult().getUses())
214  rewriter.eraseOp(use.getOwner());
215 
216  // Remove the wire.
217  rewriter.eraseOp(op);
218  return success();
219 }

304 /// Replaces the given op with the contents of the given single-block region.
305 static void replaceOpWithRegion(PatternRewriter &rewriter, Operation *op,
306  Region &region) {
307  assert(llvm::hasSingleElement(region) && "expected single-region block");
308  Block *fromBlock = &region.front();
309  // Merge it in above the specified operation.
310  op->getBlock()->getOperations().splice(Block::iterator(op),
311  fromBlock->getOperations());
312 }

314 LogicalResult IfOp::canonicalize(IfOp op, PatternRewriter &rewriter) {
315  if (auto constant = op.cond().getDefiningOp<hw::ConstantOp>()) {
316 
317  if (constant.getValue().isAllOnesValue())
318  replaceOpWithRegion(rewriter, op, op.thenRegion());
319  else if (!op.elseRegion().empty())
320  replaceOpWithRegion(rewriter, op, op.elseRegion());
321 
322  rewriter.eraseOp(op);
323 
324  return success();
325  }
326 
327  // Erase empty if's.
328 
329  // If there is stuff in the then block, leave this operation alone.
330  if (!op.getThenBlock()->empty())
331  return failure();
332 
333  // If not and there is no else, then this operation is just useless.
334  if (!op.hasElse() || op.getElseBlock()->empty()) {
335  rewriter.eraseOp(op);
336  return success();
337  }
338 
339  // Otherwise, invert the condition and move the 'else' block to the 'then'
340  // region.
341  auto full =
342  rewriter.create<hw::ConstantOp>(op.getLoc(), op.cond().getType(), -1);
343  Value ops[] = {full, op.cond()};
344  auto cond =
345  rewriter.createOrFold<comb::XorOp>(op.getLoc(), op.cond().getType(), ops);
346  op.setOperand(cond);
347 
348  auto *thenBlock = op.getThenBlock(), *elseBlock = op.getElseBlock();
349 
350  // Move the body of the then block over to the else.
351  thenBlock->getOperations().splice(thenBlock->end(),
352  elseBlock->getOperations());
353  rewriter.eraseBlock(elseBlock);
354  return success();
355 }

394 static ParseResult
395 parseEventList(OpAsmParser &p, Attribute &eventsAttr,
396  SmallVectorImpl<OpAsmParser::OperandType> &clocksOperands) {
397 
398  // Parse zero or more conditions intoevents and clocksOperands.
399  SmallVector<Attribute> events;
400 
401  auto loc = p.getCurrentLocation();
402  StringRef keyword;
403  if (!p.parseOptionalKeyword(&keyword)) {
404  while (1) {
405  auto kind = symbolizeEventControl(keyword);
406  if (!kind.hasValue())
407  return p.emitError(loc, "expected 'posedge', 'negedge', or 'edge'");
408  auto eventEnum = static_cast<int32_t>(kind.getValue());
409  events.push_back(p.getBuilder().getI32IntegerAttr(eventEnum));
410 
411  clocksOperands.push_back({});
412  if (p.parseOperand(clocksOperands.back()))
413  return failure();
414 
415  if (failed(p.parseOptionalComma()))
416  break;
417  if (p.parseKeyword(&keyword))
418  return failure();
419  }
420  }
421  eventsAttr = p.getBuilder().getArrayAttr(events);
422  return success();
423 }

525 // significant bit.
526 CaseZOp::CasePattern::CasePattern(ArrayRef<PatternBit> bits,
527  MLIRContext *context) {

550 static ParseResult parseCaseZOp(OpAsmParser &parser, OperationState &result) {
551  auto &builder = parser.getBuilder();
552 
553  OpAsmParser::OperandType condOperand;
554  Type condType;
555 
556  auto loc = parser.getCurrentLocation();
557  if (parser.parseOperand(condOperand) || parser.parseColonType(condType) ||
558  parser.parseOptionalAttrDict(result.attributes) ||
559  parser.resolveOperand(condOperand, condType, result.operands))
560  return failure();
561 
562  // Check the integer type.
563  if (!result.operands[0].getType().isSignlessInteger())
564  return parser.emitError(loc, "condition must have signless integer type");
565  auto condWidth = condType.getIntOrFloatBitWidth();
566 
567  // Parse all the cases.
568  SmallVector<Attribute> casePatterns;
569  SmallVector<CaseZOp::PatternBit, 16> caseBits;
570  while (1) {
571  if (succeeded(parser.parseOptionalKeyword("default"))) {
572  // Fill the pattern with Any.
573  caseBits.assign(condWidth, CaseZOp::PatternAny);
574  } else if (failed(parser.parseOptionalKeyword("case"))) {
575  // Not default or case, must be the end of the cases.
576  break;
577  } else {
578  // Parse the pattern. It always starts with b, so it is an MLIR keyword.
579  StringRef caseVal;
580  loc = parser.getCurrentLocation();
581  if (parser.parseKeyword(&caseVal))
582  return failure();
583 
584  if (caseVal.front() != 'b')
585  return parser.emitError(loc, "expected case value starting with 'b'");
586  caseVal = caseVal.drop_front();
587 
588  // Parse and decode each bit, we reverse the list later for MSB->LSB.
589  for (; !caseVal.empty(); caseVal = caseVal.drop_front()) {
590  CaseZOp::PatternBit bit;
591  switch (caseVal.front()) {
592  case '0':
593  bit = CaseZOp::PatternZero;
594  break;
595  case '1':
596  bit = CaseZOp::PatternOne;
597  break;
598  case 'x':
599  bit = CaseZOp::PatternAny;
600  break;
601  default:
602  return parser.emitError(loc, "unexpected case bit '")
603  << caseVal.front() << "'";
604  }
605  caseBits.push_back(bit);
606  }
607 
608  if (caseVal.size() > condWidth)
609  return parser.emitError(loc, "too many bits specified in pattern");
610  std::reverse(caseBits.begin(), caseBits.end());
611 
612  // High zeros may be missing.
613  if (caseBits.size() < condWidth)
614  caseBits.append(condWidth - caseBits.size(), CaseZOp::PatternZero);
615  }
616 
617  auto resultPattern = CaseZOp::CasePattern(caseBits, builder.getContext());
618  casePatterns.push_back(resultPattern.attr);
619  caseBits.clear();
620 
621  // Parse the case body.
622  auto caseRegion = std::make_unique<Region>();
623  if (parser.parseColon() || parser.parseRegion(*caseRegion))
624  return failure();
625  result.addRegion(std::move(caseRegion));
626  }
627 
628  result.addAttribute("casePatterns", builder.getArrayAttr(casePatterns));
629  return success();
630 }

632 static void printCaseZOp(OpAsmPrinter &p, CaseZOp op) {
633  p << "sv.casez" << ' ' << op.cond() << " : " << op.cond().getType();
634  p.printOptionalAttrDict(op->getAttrs(), /*elidedAttrs=*/{"casePatterns"});
635 
636  for (auto caseInfo : op.getCases()) {
637  p.printNewline();
638  auto pattern = caseInfo.pattern;
639  if (pattern.isDefault()) {
640  p << "default";
641  } else {
642  p << "case b";
643  for (size_t i = 0, e = pattern.getWidth(); i != e; ++i)
644  p << CaseZOp::getLetter(pattern.getBit(e - i - 1),
645  /*isVerilog=*/false);
646  }
647 
648  p << ':';
649  p.printRegion(*caseInfo.block->getParent(), /*printEntryBlockArgs=*/false,
650  /*printBlockTerminators=*/true);
651  }
652 }

877 // If this wire is only written to, delete the wire and all writers.
878 LogicalResult WireOp::canonicalize(WireOp wire, PatternRewriter &rewriter) {
879  // If the wire has a symbol, then we can't delete it.
880  if (wire.sym_nameAttr())
881  return failure();
882 
883  // Wires have inout type, so they'll have assigns and read_inout operations
884  // that work on them. If anything unexpected is found then leave it alone.
885  SmallVector<sv::ReadInOutOp> reads;
886  sv::AssignOp write;
887 
888  for (auto *user : wire->getUsers()) {
889  if (auto read = dyn_cast<sv::ReadInOutOp>(user)) {
890  reads.push_back(read);
891  continue;
892  }
893 
894  // Otherwise must be an assign, and we must not have seen a write yet.
895  auto assign = dyn_cast<sv::AssignOp>(user);
896  // Either the wire has more than one write or another kind of Op (other than
897  // AssignOp and ReadInOutOp), then can't optimize.
898  if (!assign || write)
899  return failure();
900  write = assign;
901  }
902 
903  Value connected;
904  if (!write) {
905  // If no write and only reads, then replace with XOp.
906  connected = rewriter.create<ConstantXOp>(
907  wire.getLoc(),
908  wire.getResult().getType().cast<InOutType>().getElementType());
909  } else if (isa<hw::HWModuleOp>(write->getParentOp()))
910  connected = write.src();
911  else
912  // If the write is happening at the module level then we don't have any
913  // use-before-def checking to do, so we only handle that for now.
914  return failure();
915 
916  // Ok, we can do this. Replace all the reads with the connected value.
917  for (auto read : reads)
918  rewriter.replaceOp(read, connected);
919 
920  // And remove the write and wire itself.
921  if (write)
922  rewriter.eraseOp(write);
923  rewriter.eraseOp(wire);
924  return success();
925 }

970 //===----------------------------------------------------------------------===//
971 
972 // reg s <= cond ? val : s simplification.
973 // Don't assign a register's value to itself, conditionally assign the new value
974 // instead.
975 LogicalResult PAssignOp::canonicalize(PAssignOp op, PatternRewriter &rewriter) {
976  auto mux = op.src().getDefiningOp<comb::MuxOp>();
977  if (!mux)
978  return failure();
979 
980  auto reg = dyn_cast<sv::RegOp>(op.dest().getDefiningOp());
981  if (!reg)
982  return failure();
983 
984  bool trueBranch; // did we find the register on the true branch?
985  auto tvread = mux.trueValue().getDefiningOp<sv::ReadInOutOp>();
986  auto fvread = mux.falseValue().getDefiningOp<sv::ReadInOutOp>();
987  if (tvread && reg == tvread.input().getDefiningOp<sv::RegOp>())
988  trueBranch = true;
989  else if (fvread && reg == fvread.input().getDefiningOp<sv::RegOp>())
990  trueBranch = false;
991  else
992  return failure();
993 
994  // Check that this is the only write of the register
995  for (auto &use : reg->getUses()) {
996  if (isa<ReadInOutOp>(use.getOwner()))
997  continue;
998  if (use.getOwner() == op)
999  continue;
1000  return failure();
1001  }
1002 
1003  // Replace a non-blocking procedural assign in a procedural region with a
1004  // conditional procedural assign. We've ensured that this is the only write
1005  // of the register.
1006  if (trueBranch) {
1007  auto one =
1008  rewriter.create<hw::ConstantOp>(mux.getLoc(), mux.cond().getType(), -1);
1009  Value ops[] = {mux.cond(), one};
1010  auto cond = rewriter.createOrFold<comb::XorOp>(mux.getLoc(),
1011  mux.cond().getType(), ops);
1012  rewriter.create<sv::IfOp>(mux.getLoc(), cond, [&]() {
1013  rewriter.create<PAssignOp>(op.getLoc(), reg, mux.falseValue());
1014  });
1015  } else {
1016  rewriter.create<sv::IfOp>(mux.getLoc(), mux.cond(), [&]() {
1017  rewriter.create<PAssignOp>(op.getLoc(), reg, mux.trueValue());
1018  });
1019  }
1020 
1021  // Remove the wire.
1022  rewriter.eraseOp(op);
1023  return success();
1024 }
