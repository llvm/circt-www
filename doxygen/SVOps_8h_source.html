47 // This is provides convenient access to encode and decode a pattern.
48 class CasePattern {
49 
50 public:
51  enum CasePatternKind { CPK_bit, CPK_enum, CPK_default };
52  CasePattern(CasePatternKind kind) : kind(kind) {}
53  virtual ~CasePattern() {}
54  CasePatternKind getKind() const { return kind; }
55 
56  /// Return true if this pattern has an X.
57  virtual bool hasX() const { return false; }
58 
59  /// Return true if this pattern has an Z.
60  virtual bool hasZ() const { return false; }
61 
62  virtual Attribute attr() const = 0;
63 
64 private:
65  const CasePatternKind kind;
66 };

68 class CaseDefaultPattern : public CasePattern {
69 public:
70  using AttrType = mlir::UnitAttr;
71  CaseDefaultPattern(MLIRContext *ctx)
72  : CasePattern(CasePatternKind::CPK_default) {
73  unitAttr = AttrType::get(ctx);
74  }
75 
76  Attribute attr() const override { return unitAttr; }
77 
78  static bool classof(const CasePattern *S) {
79  return S->getKind() == CPK_default;
80  }
81 
82 private:
83  // The default pattern is recognized by a UnitAttr. This is needed since we
84  // need to be able to determine the pattern type of a case based on an
85  // attribute attached to the sv.case op.
86  UnitAttr unitAttr;
87 };

100  Attribute attr() const override { return intAttr; }
101 
102  /// Get a CasePattern from a specified list of CasePatternBit. Bits are
103  /// specified in most least significant order - element zero is the least

105  CaseBitPattern(ArrayRef<CasePatternBit> bits, MLIRContext *context);
106 
107  /// Get a CasePattern for the specified constant value.

152 template <typename ConcreteType>
153 class ProceduralRegion
154  : public mlir::OpTrait::TraitBase<ConcreteType, ProceduralRegion> {
155  static LogicalResult verifyTrait(Operation *op) {
156  return mlir::OpTrait::impl::verifyAtLeastNRegions(op, 1);
