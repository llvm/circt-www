81 /// Parses a type registered to this dialect
82 Type SVDialect::parseType(DialectAsmParser &parser) const {
83  llvm::StringRef mnemonic;
84  auto loc = parser.getCurrentLocation();
85  if (parser.parseKeyword(&mnemonic))
86  return Type();
87  Type type;
88  auto parseResult = generatedTypeParser(getContext(), parser, mnemonic, type);
89  if (parseResult.hasValue())
90  return type;
91  parser.emitError(loc, "Failed to parse type sv.") << mnemonic << "\n";
92  return Type();
93 }
