22 template <typename ConcreteType, typename ResultType = void,
23  typename... ExtraArgs>
24 class Visitor {
25 public:
26  ResultType dispatchSVVisitor(Operation *op, ExtraArgs... args) {
27  auto *thisCast = static_cast<ConcreteType *>(this);
28  return TypeSwitch<Operation *, ResultType>(op)
29  .template Case<
30  // Expressions
31  ReadInOutOp, ArrayIndexInOutOp, VerbatimExprOp, VerbatimExprSEOp,
32  IndexedPartSelectInOutOp, IndexedPartSelectOp, StructFieldInOutOp,
33  ConstantXOp, ConstantZOp, MacroRefExprOp,
34  // Declarations.
35  RegOp, WireOp, LogicOp, LocalParamOp, XMROp,
36  // Control flow.
37  OrderedOutputOp, IfDefOp, IfDefProceduralOp, IfOp, AlwaysOp,
38  AlwaysCombOp, AlwaysFFOp, InitialOp, CaseOp,
39  // Other Statements.
40  AssignOp, BPAssignOp, PAssignOp, ForceOp, ReleaseOp, AliasOp,
41  FWriteOp, VerbatimOp,
42  // Type declarations.
43  InterfaceOp, InterfaceSignalOp, InterfaceModportOp,
44  InterfaceInstanceOp, GetModportOp, AssignInterfaceSignalOp,
45  ReadInterfaceSignalOp,
46  // Verification statements.
47  AssertOp, AssumeOp, CoverOp, AssertConcurrentOp, AssumeConcurrentOp,
48  CoverConcurrentOp,
49  // Bind Statements
50  BindOp,
51  // Simulator control tasks
52  StopOp, FinishOp, ExitOp,
53  // Severity message tasks
54  FatalOp, ErrorOp, WarningOp, InfoOp,
55  // Generate statements
56  GenerateOp, GenerateCaseOp,
57  // Sampled value functiions
58  SampledOp>([&](auto expr) -> ResultType {
59  return thisCast->visitSV(expr, args...);
60  })
61  .Default([&](auto expr) -> ResultType {
62  return thisCast->visitInvalidSV(op, args...);
63  });
64  }

66  /// This callback is invoked on any invalid operations.
67  ResultType visitInvalidSV(Operation *op, ExtraArgs... args) {
68  op->emitOpError("unknown SV node");
69  abort();
70  }

72  /// This callback is invoked on any SV operations that are not handled by the
73  /// concrete visitor.
74  ResultType visitUnhandledSV(Operation *op, ExtraArgs... args) {
75  return ResultType();
76  }
