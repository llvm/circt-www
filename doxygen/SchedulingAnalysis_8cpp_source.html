27 /// CyclicSchedulingAnalysis constructs a CyclicProblem for each AffineForOp by
28 /// performing a memory dependence analysis and inserting dependences into the
29 /// problem. The client should retrieve the partially complete problem to add
30 /// and associate operator types.
31 circt::analysis::CyclicSchedulingAnalysis::CyclicSchedulingAnalysis(
32  Operation *op, AnalysisManager &am) {
33  auto funcOp = cast<func::FuncOp>(op);
34 
35  MemoryDependenceAnalysis &memoryAnalysis =
36  am.getAnalysis<MemoryDependenceAnalysis>();
37 
38  // Only consider innermost loops of perfectly nested AffineForOps.
39  for (auto root : funcOp.getOps<AffineForOp>()) {
40  SmallVector<AffineForOp> nestedLoops;
41  getPerfectlyNestedLoops(nestedLoops, root);
42  analyzeForOp(nestedLoops.back(), memoryAnalysis);
43  }
44 }

46 void circt::analysis::CyclicSchedulingAnalysis::analyzeForOp(
47  AffineForOp forOp, MemoryDependenceAnalysis memoryAnalysis) {
48  // Create a cyclic scheduling problem.
49  CyclicProblem problem = CyclicProblem::get(forOp);
50 
51  // Insert memory dependences into the problem.
52  forOp.getBody()->walk([&](Operation *op) {
53  // Insert every operation into the problem.
54  problem.insertOperation(op);
55 
56  ArrayRef<MemoryDependence> dependences = memoryAnalysis.getDependences(op);
57  if (dependences.empty())
58  return;
59 
60  for (MemoryDependence memoryDep : dependences) {
61  // Don't insert a dependence into the problem if there is no dependence.
62  if (!hasDependence(memoryDep.dependenceType))
63  continue;
64 
65  // Insert a dependence into the problem.
66  Problem::Dependence dep(memoryDep.source, op);
67  auto depInserted = problem.insertDependence(dep);
68  assert(succeeded(depInserted));
69  (void)depInserted;
70 
71  // Use the lower bound of the innermost loop for this dependence. This
72  // assumes outer loops execute sequentially, i.e. one iteration of the
73  // inner loop completes before the next iteration is initiated. With
74  // proper analysis and lowerings, this can be relaxed.
75  unsigned distance = memoryDep.dependenceComponents.back().lb.getValue();
76  if (distance > 0)
77  problem.setDistance(dep, distance);
78  }
79  });
80 
81  // Insert conditional dependences into the problem.
82  forOp.getBody()->walk([&](Operation *op) {
83  Block *thenBlock = nullptr;
84  Block *elseBlock = nullptr;
85  if (auto ifOp = dyn_cast<scf::IfOp>(op)) {
86  thenBlock = ifOp.thenBlock();
87  elseBlock = ifOp.elseBlock();
88  } else if (auto ifOp = dyn_cast<AffineIfOp>(op)) {
89  thenBlock = ifOp.getThenBlock();
90  if (ifOp.hasElse())
91  elseBlock = ifOp.getElseBlock();
92  } else {
93  return WalkResult::advance();
94  }
95 
96  // No special handling required for control-only `if`s.
97  if (op->getNumResults() == 0)
98  return WalkResult::skip();
99 
100  // Model the implicit value flow from the `yield` to the `if`'s result(s).
101  Problem::Dependence depThen(thenBlock->getTerminator(), op);
102  auto depInserted = problem.insertDependence(depThen);
103  assert(succeeded(depInserted));
104  (void)depInserted;
105 
106  if (elseBlock) {
107  Problem::Dependence depElse(elseBlock->getTerminator(), op);
108  depInserted = problem.insertDependence(depElse);
109  assert(succeeded(depInserted));
110  (void)depInserted;
111  }
112 
113  return WalkResult::advance();
114  });
115 
116  // Set the anchor for scheduling. Insert dependences from all stores to the
117  // terminator to ensure the problem schedules them before the terminator.
118  auto *anchor = forOp.getBody()->getTerminator();
119  forOp.getBody()->walk([&](Operation *op) {
120  if (!isa<mlir::AffineStoreOp, memref::StoreOp>(op))
121  return;
122  Problem::Dependence dep(op, anchor);
123  auto depInserted = problem.insertDependence(dep);
124  assert(succeeded(depInserted));
125  (void)depInserted;
126  });
127 
128  // Handle explicitly computed loop-carried values, i.e. excluding the
129  // induction variable. Insert inter-iteration dependences from the definers of
130  // "iter_args" to their users.
131  if (unsigned nIterArgs = anchor->getNumOperands(); nIterArgs > 0) {
132  auto iterArgs = forOp.getRegionIterArgs();
133  for (unsigned i = 0; i < nIterArgs; ++i) {
134  Operation *iterArgDefiner = anchor->getOperand(i).getDefiningOp();
135  // If it's not an operation, we don't need to model the dependence.
136  if (!iterArgDefiner)
137  continue;
138 
139  for (Operation *iterArgUser : iterArgs[i].getUsers()) {
140  Problem::Dependence dep(iterArgDefiner, iterArgUser);
141  auto depInserted = problem.insertDependence(dep);
142  assert(succeeded(depInserted));
143  (void)depInserted;
144 
145  // Values always flow between subsequent iterations.
146  problem.setDistance(dep, 1);
147  }
148  }
149  }
150 
151  // Store the partially complete problem.
152  problems.insert(std::pair<Operation *, CyclicProblem>(forOp, problem));
153 }
