135 /// Return the encoding value for the size of this type (from the encoding
136 /// spec): 0 = 0 bits, 1 = 1 bit, 2 = 1 byte, 3 = 2 bytes, 4 = 4 bytes, 5 = 8
137 /// bytes (non-pointer), 6 = 8 bytes (pointer).
138 static size_t bitsEncoding(::capnp::schema::Type::Reader type) {
139  using ty = ::capnp::schema::Type;
140  switch (type.which()) {
141  case ty::VOID:
142  return 0;
143  case ty::BOOL:
144  return 1;
145  case ty::UINT8:
146  case ty::INT8:
147  return 2;
148  case ty::UINT16:
149  case ty::INT16:
150  return 3;
151  case ty::UINT32:
152  case ty::INT32:
153  return 4;
154  case ty::UINT64:
155  case ty::INT64:
156  return 5;
157  case ty::ANY_POINTER:
158  case ty::DATA:
159  case ty::INTERFACE:
160  case ty::LIST:
161  case ty::STRUCT:
162  case ty::TEXT:
163  return 6;
164  default:
165  assert(false && "Type not yet supported");
166  }
167 }

169 /// Return the number of bits used by a Capnp type.
170 static size_t bits(::capnp::schema::Type::Reader type) {
171  size_t enc = bitsEncoding(type);
172  if (enc <= 1)
173  return enc;
174  if (enc == 6)
175  return 64;
176  return 1 << (enc + 1);

211 /// Write a valid capnp schema to memory, then parse it out of memory using the
212 /// capnp library. Writing and parsing text within a single process is ugly, but
213 /// this is by far the easiest way to do this. This isn't the use case for which
214 /// Cap'nProto was designed.
215 ::capnp::ParsedSchema TypeSchemaImpl::getSchema() const {
216  if (rootSchema != ::capnp::ParsedSchema())
217  return rootSchema;
218 
219  // Write the schema to `schemaText`.
220  std::string schemaText;
221  llvm::raw_string_ostream os(schemaText);
222  emitId(os, 0xFFFFFFFFFFFFFFFF) << ";\n";
223  auto rc = write(os);
224  assert(succeeded(rc) && "Failed schema text output.");
225  os.str();
226 
227  // Write `schemaText` to an in-memory filesystem then parse it. Yes, this is
228  // the only way to do this.
229  kj::Own<kj::Filesystem> fs = kj::newDiskFilesystem();
230  kj::Own<kj::Directory> dir = kj::newInMemoryDirectory(kj::nullClock());
231  kj::Path fakePath = kj::Path::parse("schema.capnp");
232  { // Ensure that 'fakeFile' has flushed.
233  auto fakeFile = dir->openFile(fakePath, kj::WriteMode::CREATE);
234  fakeFile->writeAll(schemaText);
235  }
236  rootSchema = parser.parseFromDirectory(*dir, std::move(fakePath), nullptr);
237  return rootSchema;
238 }

254 // We compute a deterministic hash based on the type. Since llvm::hash_value
255 // changes from execution to execution, we don't use it.
256 uint64_t TypeSchemaImpl::capnpTypeID() const {
257  if (cachedID)
258  return *cachedID;
259 
260  // Get the MLIR asm type, padded to a multiple of 64 bytes.
261  std::string typeName;
262  llvm::raw_string_ostream osName(typeName);
263  osName << type;
264  size_t overhang = osName.tell() % 64;
265  if (overhang != 0)
266  osName.indent(64 - overhang);
267  osName.flush();
268  const char *typeNameC = typeName.c_str();
269 
270  uint64_t hash = esiCosimSchemaVersion;
271  for (size_t i = 0, e = typeName.length() / 64; i < e; ++i)
272  hash =
273  llvm::hashing::detail::hash_33to64_bytes(&typeNameC[i * 64], 64, hash);
274 
275  // Capnp IDs always have a '1' high bit.
276  cachedID = hash | 0x8000000000000000;
277  return *cachedID;
278 }

280 /// Returns true if the type is currently supported.
281 static bool isSupported(Type type, bool outer = false) {
282  return llvm::TypeSwitch<::mlir::Type, bool>(type)
283  .Case([](IntegerType t) { return t.getWidth() <= 64; })
284  .Case([](hw::ArrayType t) { return isSupported(t.getElementType()); })
285  .Case([outer](hw::StructType t) {
286  // We don't yet support structs containing structs.
287  if (!outer)
288  return false;
289  // A struct is supported if all of its elements are.
290  for (auto field : t.getElements()) {
291  if (!isSupported(field.type))
292  return false;
293  }
294  return true;
295  })
296  .Default([](Type) { return false; });
297 }

302 /// Returns the expected size of an array (capnp list) in 64-bit words.
303 static int64_t size(hw::ArrayType mType, capnp::schema::Field::Reader cField) {
304  assert(cField.isSlot());
305  auto cType = cField.getSlot().getType();
306  assert(cType.isList());
307  size_t elementBits = bits(cType.getList().getElementType());
308  int64_t listBits = mType.getSize() * elementBits;
309  return llvm::divideCeil(listBits, 64);
310 }

312 /// Compute the size of a capnp struct, in 64-bit words.
313 static int64_t size(capnp::schema::Node::Struct::Reader cStruct,
314  ArrayRef<hw::StructType::FieldInfo> mFields) {
315  using namespace capnp::schema;
316  int64_t size = (1 + // Header
317  cStruct.getDataWordCount() + cStruct.getPointerCount());
318  auto cFields = cStruct.getFields();
319  for (Field::Reader cField : cFields) {
320  assert(!cField.isGroup() && "Capnp groups are not supported");
321  // Capnp code order is the index in the the MLIR fields array.
322  assert(cField.getCodeOrder() < mFields.size());
323 
324  // The size of the thing to which the pointer is pointing, not the size of
325  // the pointer itself.
326  int64_t pointedToSize =
327  TypeSwitch<mlir::Type, int64_t>(mFields[cField.getCodeOrder()].type)
328  .Case([](IntegerType) { return 0; })
329  .Case([cField](hw::ArrayType mType) {
330  return ::size(mType, cField);
331  });
332  size += pointedToSize;
333  }
334  return size; // Convert from 64-bit words to bits.
335 }

344 /// Write a valid Capnp name for 'type'.
345 static void emitName(Type type, uint64_t id, llvm::raw_ostream &os) {
346  llvm::TypeSwitch<Type>(type)
347  .Case([&os](IntegerType intTy) {
348  std::string intName;
349  llvm::raw_string_ostream(intName) << intTy;
350  // Capnp struct names must start with an uppercase character.
351  intName[0] = toupper(intName[0]);
352  os << intName;
353  })
354  .Case([&os](hw::ArrayType arrTy) {
355  os << "ArrayOf" << arrTy.getSize() << 'x';
356  emitName(arrTy.getElementType(), 0, os);
357  })
358  .Case([&os, id](hw::StructType t) { os << "Struct" << id; })
359  .Default([](Type) {
360  assert(false && "Type not supported. Please check support first with "
361  "isSupported()");
362  });
363 }

376 /// Write a valid Capnp type.
377 static void emitCapnpType(Type type, IndentingOStream &os) {
378  llvm::TypeSwitch<Type>(type)
379  .Case([&os](IntegerType intTy) {
380  auto w = intTy.getWidth();
381  if (w == 0) {
382  os << "Void";
383  } else if (w == 1) {
384  os << "Bool";
385  } else {
386  if (intTy.isSigned())
387  os << "Int";
388  else
389  os << "UInt";
390 
391  // Round up.
392  if (w <= 8)
393  os << "8";
394  else if (w <= 16)
395  os << "16";
396  else if (w <= 32)
397  os << "32";
398  else if (w <= 64)
399  os << "64";
400  else
401  assert(false && "Type not supported. Integer too wide. Please "
402  "check support first with isSupported()");
403  }
404  })
405  .Case([&os](hw::ArrayType arrTy) {
406  os << "List(";
407  emitCapnpType(arrTy.getElementType(), os);
408  os << ')';
409  })
410  .Case([](hw::StructType structTy) {
411  assert(false && "Struct containing structs not supported");
412  })
413  .Default([](Type) {
414  assert(false && "Type not supported. Please check support first with "
415  "isSupported()");
416  });
417 }

419 /// This function is essentially a placeholder which only supports ints. It'll
420 /// need to be re-worked when we start supporting structs, arrays, unions,
421 /// enums, etc.
422 LogicalResult TypeSchemaImpl::write(llvm::raw_ostream &rawOS) const {
423  IndentingOStream os(rawOS);
424 
425  // Since capnp requires messages to be structs, emit a wrapper struct.
426  os.indent() << "struct ";
427  writeMetadata(rawOS);
428  os << " {\n";
429  os.addIndent();
430 
431  size_t counter = 0;
432  size_t maxNameLength = 0;
433  for (auto field : fieldTypes)
434  maxNameLength = std::max(maxNameLength, field.name.size());
435 
436  for (auto field : fieldTypes) {
437  // Specify the actual type, followed by the capnp field.
438  os.indent() << field.name.getValue();
439  os.pad(maxNameLength - field.name.size()) << " @" << counter++ << " :";
440  emitCapnpType(field.type, os);
441  os << "; # Actual type is " << field.type << ".\n";
442  }
443 
444  os.reduceIndent();
445  os.indent() << "}\n\n";
446  return success();
447 }

611  /// Create an op to slice the array from lsb to lsb + size. Return a new slice
612  /// with that op.
613  Slice slice(int64_t lsb, int64_t size) const {
614  hw::ArrayType dstTy = hw::ArrayType::get(type.getElementType(), size);
615  IntegerType idxTy =
616  builder->getIntegerType(llvm::Log2_64_Ceil(type.getSize()));
617  Value lsbConst = builder->create<hw::ConstantOp>(loc(), idxTy, lsb);

620  return Slice(this, lsb, newSlice);

623  /// Create an op to slice the array from lsb to lsb + size. Return a new slice
624  /// with that op. If lsb is greater width thn necessary, lop off the high
625  /// bits.
626  Slice slice(Value lsb, int64_t size) const {
627  assert(lsb.getType().isa<IntegerType>());
628 
629  unsigned expIdxWidth = llvm::Log2_64_Ceil(type.getSize());
630  int64_t lsbWidth = lsb.getType().getIntOrFloatBitWidth();
631  if (lsbWidth > expIdxWidth)
632  lsb = builder->create<comb::ExtractOp>(
633  loc(), builder->getIntegerType(expIdxWidth), lsb, 0);
634  else if (lsbWidth < expIdxWidth)
635  assert(false && "LSB Value must not be smaller than expected.");
636  auto dstTy = hw::ArrayType::get(type.getElementType(), size);
637  Value newSlice = builder->create<hw::ArraySliceOp>(loc(), dstTy, s, lsb);
638  return Slice(this, llvm::Optional<int64_t>(), newSlice);
639  }

777 /// Helps build capnp message DAGs, which are stored in 'segments'. To better
778 /// reason about something which is more memory-like than wire-like, this class
779 /// contains a data structure to efficiently model memory and map it to Values
780 /// (wires).
781 class CapnpSegmentBuilder : public GasketBuilder {
782 public:
783  CapnpSegmentBuilder(OpBuilder &b, Location loc, uint64_t expectedSize)
784  : GasketBuilder(b, loc), segmentValues(allocator), messageSize(0),
785  expectedSize(expectedSize) {}
786  CapnpSegmentBuilder(const CapnpSegmentBuilder &) = delete;
787  ~CapnpSegmentBuilder() {}
788 
789  GasketComponent build(::capnp::schema::Node::Struct::Reader cStruct,
790  ArrayRef<GasketComponent> mFieldValues);
791 
792 private:
793  /// Allocate and build a struct. Return the address of the data section as an
794  /// offset into the 'memory' map.
795  GasketComponent encodeStructAt(uint64_t ptrLoc,
796  ::capnp::schema::Node::Struct::Reader cStruct,
797  ArrayRef<GasketComponent> mFieldValues);
798  /// Build a value from the 'memory' map. Concatenates all the values in the
799  /// 'memory' map, filling in the blank addresses with padding.
800  GasketComponent compile() const;
801 
802  /// Encode 'val' and place the value at the specified 'memory' offset.
803  void encodeFieldAt(uint64_t offset, GasketComponent val,
804  ::capnp::schema::Type::Reader type);
805  /// Allocate and build a list, returning the address which was allocated.
806  uint64_t buildList(Slice val, ::capnp::schema::Type::Reader type);
807 
808  /// Insert 'val' into the 'memory' map.
809  void insert(uint64_t offset, GasketComponent val) {
810  uint64_t valSize = val.size();
811  assert(!segmentValues.overlaps(offset, offset + valSize - 1));
812  assert(offset + valSize - 1 < expectedSize &&
813  "Tried to insert above the max expected size of the message.");
814  segmentValues.insert(offset, offset + valSize - 1, val);
815  }
816 
817  /// This is where the magic lives. An IntervalMap allows us to efficiently
818  /// model segment 'memory' and to place Values at any address. We can then
819  /// manage 'memory allocations' (figuring out where to place pointed to
820  /// objects) separately from the data contained in those values, some of which
821  /// are pointers themselves.
822  llvm::IntervalMap<uint64_t, GasketComponent>::Allocator allocator;
823  llvm::IntervalMap<uint64_t, GasketComponent> segmentValues;
824 
825  /// Track the allocated message size. Increase to 'alloc' more.
826  uint64_t messageSize;
827  uint64_t alloc(size_t bits) {
828  uint64_t ptr = messageSize;
829  messageSize += bits;
830  return ptr;
831  }
832 
833  /// The expected maximum size of the message.
834  uint64_t expectedSize;
835 };

838 void CapnpSegmentBuilder::encodeFieldAt(uint64_t offset, GasketComponent val,
839  ::capnp::schema::Type::Reader type) {
840  TypeSwitch<Type>(val.getValue().getType())
841  .Case([&](IntegerType it) { insert(offset, val); })
842  .Case([&](hw::ArrayType arrTy) {
843  uint64_t listOffset = buildList(Slice(val), type);
844  int32_t relativeOffset = (listOffset - offset - 64) / 64;
845  insert(offset,
846  GasketComponent::concat(
847  {constant(2, 1), constant(30, relativeOffset),
848  constant(3, bitsEncoding(type.getList().getElementType())),
849  constant(29, arrTy.getSize())}));
850  });
851 }

853 uint64_t CapnpSegmentBuilder::buildList(Slice val,
854  ::capnp::schema::Type::Reader type) {
855  hw::ArrayType arrTy = val.getValue().getType().cast<hw::ArrayType>();
856  auto elemType = type.getList().getElementType();
857  size_t elemWidth = bits(elemType);
858  uint64_t listSize = elemWidth * arrTy.getSize();
859  uint64_t m;
860  if ((m = listSize % 64) != 0)
861  listSize += (64 - m);
862  uint64_t listOffset = alloc(listSize);
863 
864  for (size_t i = 0, e = arrTy.getSize(); i < e; ++i) {
865  size_t elemNum = e - i - 1;
866  encodeFieldAt(listOffset + (elemNum * elemWidth), val[i], elemType);
867  }
868  return listOffset;
869 }

871 GasketComponent CapnpSegmentBuilder::encodeStructAt(
872  uint64_t ptrLoc, ::capnp::schema::Node::Struct::Reader cStruct,
873  ArrayRef<GasketComponent> mFieldValues) {
874 
875  assert(ptrLoc % 64 == 0);
876  size_t structSize =
877  (cStruct.getDataWordCount() + cStruct.getPointerCount()) * 64;
878  uint64_t structDataSectionOffset = alloc(structSize);
879  uint64_t structPointerSectionOffset =
880  structDataSectionOffset + (cStruct.getDataWordCount() * 64);
881  assert(structDataSectionOffset % 64 == 0);
882  int64_t relativeStructDataOffsetWords =
883  ((structDataSectionOffset - ptrLoc) / 64) -
884  /*offset from end of pointer.*/ 1;
885  GasketComponent structPtr = {constant(2, 0),
886  constant(30, relativeStructDataOffsetWords),
887  constant(16, cStruct.getDataWordCount()),
888  constant(16, cStruct.getPointerCount())};
889 
890  // Loop through data fields.
891  for (auto field : cStruct.getFields()) {
892  uint16_t idx = field.getCodeOrder();
893  assert(idx < mFieldValues.size() &&
894  "Capnp struct longer than fieldValues.");
895  auto cFieldType = field.getSlot().getType();
896  uint64_t fieldOffset =
897  (isPointerType(cFieldType) ? structPointerSectionOffset
898  : structDataSectionOffset) +
899  field.getSlot().getOffset() * bits(cFieldType);
900  encodeFieldAt(fieldOffset, mFieldValues[idx], cFieldType);
901  }
902 
903  return structPtr;
904 }

906 GasketComponent CapnpSegmentBuilder::compile() const {
907  // Fill in missing bits.
908  SmallVector<GasketComponent, 16> segmentValuesPlusPadding;
909  uint64_t lastStop = 0;
910  for (auto it = segmentValues.begin(), e = segmentValues.end(); it != e;
911  ++it) {
912  auto value = it.value();
913  int64_t padBits = it.start() - lastStop;
914  assert(padBits >= 0 && "Overlap not allowed");
915  if (padBits)
916  segmentValuesPlusPadding.push_back(padding(padBits));
917  segmentValuesPlusPadding.push_back(value.castBitArray());
918  // IntervalMap has inclusive ranges, but we want to reason about [,) regions
919  // to make the math work.
920  lastStop = it.stop() + 1;
921  }
922  assert(expectedSize >= lastStop);
923  if (lastStop != expectedSize)
924  segmentValuesPlusPadding.push_back(padding(expectedSize - lastStop));
925 
926  return GasketComponent::concat(segmentValuesPlusPadding);
927 }

1001 /// Construct the proper operations to decode a capnp list. This only works for
1002 /// arrays of ints or bools. Will need to be updated for structs and lists of
1003 /// lists.
1004 static GasketComponent decodeList(hw::ArrayType type,
1005  capnp::schema::Field::Reader field,
1006  Slice ptrSection, AssertBuilder &asserts) {
1007  capnp::schema::Type::Reader capnpType = field.getSlot().getType();
1008  assert(capnpType.isList());
1009  assert(capnpType.getList().hasElementType());
1010 
1011  auto loc = ptrSection.loc();
1012  OpBuilder &b = ptrSection.b();
1013  GasketBuilder gb(b, loc);
1014 
1015  // Get the list pointer and break out its parts.
1016  auto ptr = ptrSection.slice(field.getSlot().getOffset() * 64, 64)
1017  .name(field.getName(), "_ptr");
1018  auto ptrType = ptr.slice(0, 2).name(field.getName(), "_ptrType");
1019  auto offset = ptr.slice(2, 30)
1020  .cast(b.getIntegerType(30))
1021  .name(field.getName(), "_offset");
1022  auto elemSize = ptr.slice(32, 3).name(field.getName(), "_elemSize");
1023  auto length = ptr.slice(35, 29).name(field.getName(), "_listLength");
1024 
1025  // Assert that ptr type == list type;
1026  asserts.assertEqual(ptrType, 1);
1027 
1028  // Assert that the element size in the message matches our expectation.
1029  auto expectedElemSizeBits = bits(capnpType.getList().getElementType());
1030  unsigned expectedElemSizeField;
1031  switch (expectedElemSizeBits) {
1032  case 0:
1033  expectedElemSizeField = 0;
1034  break;
1035  case 1:
1036  expectedElemSizeField = 1;
1037  break;
1038  case 8:
1039  expectedElemSizeField = 2;
1040  break;
1041  case 16:
1042  expectedElemSizeField = 3;
1043  break;
1044  case 32:
1045  expectedElemSizeField = 4;
1046  break;
1047  case 64:
1048  expectedElemSizeField = 5;
1049  break;
1050  default:
1051  assert(false && "bits() returned unexpected value");
1052  }
1053  asserts.assertEqual(elemSize, expectedElemSizeField);
1054 
1055  // Assert that the length of the list (array) is at most the length of the
1056  // array.
1057  asserts.assertPred(length, ICmpPredicate::ule, type.getSize());
1058 
1059  // Get the entire message slice, compute the offset into the list, then get
1060  // the list data in an ArrayType.
1061  auto msg = ptr.getRootSlice();
1062  auto ptrOffset = ptr.getOffsetFromRoot();
1063  assert(ptrOffset);
1064  GasketComponent offsetInBits(
1065  b, b.create<comb::ConcatOp>(loc, offset, gb.zero(6)));
1066  GasketComponent listOffset(
1067  b, b.create<comb::AddOp>(loc, offsetInBits,
1068  gb.constant(36, *ptrOffset + 64)));
1069  listOffset.name(field.getName(), "_listOffset");
1070  auto listSlice =
1071  msg.slice(listOffset, type.getSize() * expectedElemSizeBits).name("list");
1072 
1073  // Cast to an array of capnp int elements.
1074  assert(type.getElementType().isa<IntegerType>() &&
1075  "DecodeList() only works on arrays of ints currently");
1076  Type capnpElemTy =
1077  b.getIntegerType(expectedElemSizeBits, IntegerType::Signless);
1078  auto arrayOfElements = listSlice.castToSlice(capnpElemTy, type.getSize());
1079  if (arrayOfElements.getValue().getType() == type)
1080  return arrayOfElements;
1081 
1082  // Collect the reduced elements.
1083  SmallVector<Value, 64> arrayValues;
1084  for (size_t i = 0, e = type.getSize(); i < e; ++i) {
1085  auto capnpElem = arrayOfElements[i].name(field.getName(), "_capnp_elem");
1086  auto esiElem = capnpElem.downcast(type.getElementType().cast<IntegerType>())
1087  .name(field.getName(), "_elem");
1088  arrayValues.push_back(esiElem);
1089  }
1090  auto array = b.create<hw::ArrayCreateOp>(loc, arrayValues);
1091  return GasketComponent(b, array);
1092 }

1126  ports.push_back(hw::PortInfo{b.getStringAttr("clk"), hw::PortDirection::INPUT,

1129  hw::PortDirection::INPUT, valid.getType(), 1});
1130  ports.push_back(hw::PortInfo{b.getStringAttr("encodedInput"),
1131  hw::PortDirection::INPUT, operandVal.getType(),
1132  2});
1133  ports.push_back(hw::PortInfo{b.getStringAttr("decoded"),
1134  hw::PortDirection::OUTPUT, getType(), 0});

1136  operandVal.getLoc(), b.getStringAttr(modName), ports);
1137 
1138  Block *innerBlock = retMod.getBodyBlock();
1139  b.setInsertionPointToStart(innerBlock);
1140  clk = innerBlock->getArgument(0);
1141  valid = innerBlock->getArgument(1);
1142  operandVal = innerBlock->getArgument(2);
1143 
1144  // Various useful integer types.
1145  auto i16 = b.getIntegerType(16);
1146 
1147  size_t size = this->size();
1148  hw::ArrayType operandType = operandVal.getType().dyn_cast<hw::ArrayType>();
1149  assert(operandType && operandType.getSize() == size &&
1150  "Operand type and length must match the type's capnp size.");

1153  operand.setLoc(loc);
1154 
1155  auto alwaysAt = b.create<sv::AlwaysOp>(loc, sv::EventControl::AtPosEdge, clk);
1156  auto ifValid =
1157  OpBuilder(alwaysAt.getBodyRegion()).create<sv::IfOp>(loc, valid);
1158  AssertBuilder asserts(loc, ifValid.getBodyRegion());
1159 
1160  // The next 64-bits of a capnp message is the root struct pointer.
1161  ::capnp::schema::Node::Reader rootProto = getTypeSchema().getProto();
1162  auto ptr = operand.slice(0, 64).name("rootPointer");
1163 
1164  // Since this is the root, we _expect_ the offset to be zero but that's only
1165  // guaranteed to be the case with canonically-encoded messages.
1166  // TODO: support cases where the pointer offset is non-zero.
1167  Slice assertPtr(ptr);
1168  auto typeAndOffset = assertPtr.slice(0, 32).name("typeAndOffset");
1169  asserts.assertEqual(typeAndOffset, 0);
1170 
1171  // We expect the data section to be equal to the computed data section size.
1172  auto dataSectionSize =
1173  assertPtr.slice(32, 16).cast(i16).name("dataSectionSize");
1174  asserts.assertEqual(dataSectionSize,
1175  rootProto.getStruct().getDataWordCount());
1176 
1177  // We expect the pointer section to be equal to the computed pointer section
1178  // size.
1179  auto ptrSectionSize =
1180  assertPtr.slice(48, 16).cast(i16).name("ptrSectionSize");
1181  asserts.assertEqual(ptrSectionSize, rootProto.getStruct().getPointerCount());
1182 
1183  // Get pointers to the data and pointer sections.
1184  auto st = rootProto.getStruct();
1185  auto dataSection =
1186  operand.slice(64, st.getDataWordCount() * 64).name("dataSection");
1187  auto ptrSection = operand
1188  .slice(64 + (st.getDataWordCount() * 64),
1189  rootProto.getStruct().getPointerCount() * 64)
1190  .name("ptrSection");
1191 
1192  // Loop through fields.
1193  SmallVector<GasketComponent, 64> fieldValues;
1194  for (auto field : st.getFields()) {
1195  uint16_t idx = field.getCodeOrder();
1196  assert(idx < fieldTypes.size() && "Capnp struct longer than fieldTypes.");
1197  fieldValues.push_back(decodeField(fieldTypes[idx].type, field, dataSection,
1198  ptrSection, asserts));
1199  }
1200 
1201  // What to return depends on the type. (e.g. structs have to be constructed
1202  // from the field values.)
1203  GasketComponent ret =
1204  TypeSwitch<Type, GasketComponent>(type)
1205  .Case([&fieldValues](IntegerType) { return fieldValues[0]; })
1206  .Case([&fieldValues](hw::ArrayType) { return fieldValues[0]; })
1207  .Case([&](hw::StructType) {
1208  SmallVector<Value, 8> rawValues(llvm::map_range(
1209  fieldValues, [](GasketComponent c) { return c.getValue(); }));
1210  return GasketComponent(
1211  b, b.create<hw::StructCreateOp>(loc, type, rawValues));
1212  });
1213  ret.name(name());
1214
