23 ParseResult CompRegOp::parse(OpAsmParser &parser, OperationState &result) {
24  llvm::SMLoc loc = parser.getCurrentLocation();
25 
26  if (succeeded(parser.parseOptionalKeyword("sym"))) {
27  StringAttr symName;
28  if (parser.parseSymbolName(symName, "sym_name", result.attributes))
29  return failure();
30  }
31 
32  SmallVector<OpAsmParser::UnresolvedOperand, 4> operands;
33  if (parser.parseOperandList(operands))
34  return failure();
35  switch (operands.size()) {
36  case 0:
37  return parser.emitError(loc, "expected operands");
38  case 1:
39  return parser.emitError(loc, "expected clock operand");
40  case 2:
41  // No reset.
42  break;
43  case 3:
44  return parser.emitError(loc, "expected resetValue operand");
45  case 4:
46  // reset and reset value included.
47  break;
48  default:
49  return parser.emitError(loc, "too many operands");
50  }
51 
52  if (succeeded(parser.parseOptionalKeyword("svattrs"))) {
53  ArrayAttr svattrs;
54  if (parser.parseAttribute(svattrs, "svAttributes", result.attributes))
55  return failure();
56  }
57 
58  Type ty;
59  if (parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
60  parser.parseType(ty))
61  return failure();
62  Type i1 = IntegerType::get(result.getContext(), 1);
63 
64  // If there was no name specified, check to see if there was a useful name
65  // specified in the asm file.
66  if (!result.attributes.getNamed("name")) {
67  // If there is no explicit name attribute, get it from the SSA result name.
68  // If numeric, just use an empty name.
69  StringRef resultName = parser.getResultName(0).first;
70  if (!resultName.empty() && isdigit(resultName[0]))
71  resultName = "";
72  result.addAttribute("name", parser.getBuilder().getStringAttr(resultName));
73  }
74 
75  result.addTypes({ty});
76  if (operands.size() == 2)
77  return parser.resolveOperands(operands, {ty, i1}, loc, result.operands);
78  else
79  return parser.resolveOperands(operands, {ty, i1, i1, ty}, loc,
80  result.operands);
81 }

83 void CompRegOp::print(::mlir::OpAsmPrinter &p) {
84  SmallVector<StringRef> elidedAttrs;
85  if (sym_name().hasValue()) {
86  elidedAttrs.push_back("sym_name");
87  p << ' ' << "sym ";
88  p.printSymbolName(*sym_name());
89  }
90 
91  p << ' ' << input() << ", " << clk();
92  if (reset())
93  p << ", " << reset() << ", " << resetValue() << ' ';
94 
95  if (svAttributes()) {
96  p << " svattrs " << svAttributesAttr();
97  elidedAttrs.push_back("svAttributes");
98  }
99 
100  // Determine if 'name' can be elided.
101  if (name().empty()) {
102  elidedAttrs.push_back("name");
103  } else {
104  SmallString<32> resultNameStr;
105  llvm::raw_svector_ostream tmpStream(resultNameStr);
106  p.printOperand(data(), tmpStream);
107  auto actualName = tmpStream.str().drop_front();
108  if (actualName == name())
109  elidedAttrs.push_back("name");
110  }
111 
112  p.printOptionalAttrDict((*this)->getAttrs(), elidedAttrs);
113  p << " : " << input().getType();
114 }
