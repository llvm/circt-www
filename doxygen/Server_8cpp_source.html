79 /// This is the client polling for a message. If one is available, send it.
80 /// TODO: implement a blocking call with a timeout.
81 kj::Promise<void> EndpointServer::recv(RecvContext context) {
82  KJ_REQUIRE(open, "EndPoint closed already");
83  KJ_REQUIRE(!context.getParams().getBlock(),
84  "Blocking recv() not supported yet");
85 
86  // Try to pop a message.
87  Endpoint::BlobPtr blob;
88  auto msgPresent = endpoint.getMessageToClient(blob);
89  context.getResults().setHasData(msgPresent);
90  if (msgPresent) {
91  KJ_REQUIRE(blob->size() % 8 == 0,
92  "Response msg was malformed. Size of response was not a "
93  "multiple of 8 bytes.");
94  // Copy the blob into a single segment.
95  auto segment =
96  kj::ArrayPtr<capnp::word>((word *)blob->data(), blob->size() / 8)
97  .asConst();
98  // Create a single-element array of segments.
99  kj::Array<kj::ArrayPtr<const capnp::word>> segments =
100  kj::heapArray({segment});
101  // Create an object which will read the segments into a message on send.
102  std::unique_ptr<SegmentArrayMessageReader> msgReader =
103  std::make_unique<SegmentArrayMessageReader>(segments);
104  // Send.
105  context.getResults().getResp().set(msgReader->getRoot<AnyPointer>());
106  }
107  return kj::READY_NOW;
108 }

110 /// 'Send' is from the client perspective, so this is a message we are
111 /// recieving. The only way I could figure out to copy the raw message is a
112 /// double copy. I was have issues getting libkj's arrays to play nice with
113 /// others.
114 kj::Promise<void> EndpointServer::send(SendContext context) {
115  KJ_REQUIRE(open, "EndPoint closed already");
116  auto capnpMsgPointer = context.getParams().getMsg();
117  KJ_REQUIRE(capnpMsgPointer.isStruct(),
118  "Only messages can go in the 'msg' parameter");
119 
120  // Copy the incoming message into a flat, single segment buffer.
121  auto msgSize = capnpMsgPointer.targetSize();
122  auto builder = std::make_unique<MallocMessageBuilder>(
123  msgSize.wordCount + 1, AllocationStrategy::FIXED_SIZE);
124  builder->setRoot(capnpMsgPointer);
125  auto segments = builder->getSegmentsForOutput();
126  KJ_ASSERT(segments.size() == 1);
127 
128  // Now copy it into a blob and queue it.
129  auto fstSegmentData = segments[0].asBytes();
130  auto blob = std::make_shared<Endpoint::Blob>(fstSegmentData.begin(),
131  fstSegmentData.end());
132  endpoint.pushMessageToSim(blob);
133  return kj::READY_NOW;
134 }

159 kj::Promise<void> CosimServer::open(OpenContext ctxt) {
160  Endpoint *ep = reg[ctxt.getParams().getIface().getEndpointID()];
161  KJ_REQUIRE(ep != nullptr, "Could not find endpoint");
162 
163  auto gotLock = ep->setInUse();
164  KJ_REQUIRE(gotLock, "Endpoint in use");
165 
166  ctxt.getResults().setIface(EsiDpiEndpoint<AnyPointer, AnyPointer>::Client(
167  kj::heap<EndpointServer>(*ep)));

186 void RpcServer::mainLoop(uint16_t port) {
187  capnp::EzRpcServer rpcServer(kj::heap<CosimServer>(endpoints),
188  /* bindAddress */ "*", port);
189  auto &waitScope = rpcServer.getWaitScope();
190  // If port is 0, ExRpcSever selects one and we have to wait to get the port.

192  auto portPromise = rpcServer.getPort();

195  writePort(port);
196  printf("[COSIM] Listening on port: %u\n", (unsigned int)port);
