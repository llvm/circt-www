36 /// This class provides a framework to model certain scheduling problems as
37 /// lexico-parametric linear programs (LP), which are then solved with an
38 /// extended version of the dual simplex algorithm.
39 ///
40 /// The approach is described in:
41 /// [1] B. D. de Dinechin, "Simplex Scheduling: More than Lifetime-Sensitive
42 /// Instruction Scheduling", PRISM 1994.22, 1994.
43 /// [2] B. D. de Dinechin, "Fast Modulo Scheduling Under the Simplex Scheduling
44 /// Framework", PRISM 1995.01, 1995.
45 ///
46 /// Resource-free scheduling problems (called "central problems" in the papers)
47 /// have an *integer* linear programming formulation with a totally unimodular
48 /// constraint matrix. Such ILPs can however be solved optimally in polynomial
49 /// time with a (non-integer) LP solver (such as the simplex algorithm), as the
50 /// LP solution is guaranteed to be integer. Note that this is the same idea as
51 /// used by SDC-based schedulers.
52 class SimplexSchedulerBase {
53 protected:
54  /// The objective is to minimize the start time of this operation.
55  Operation *lastOp;
56 
57  /// S is part of a mechanism to assign fixed values to the LP variables.
58  int parameterS;
59 
60  /// T represents the initiation interval (II). Its minimally-feasible value is
61  /// computed by the algorithm.
62  int parameterT;
63 
64  /// The simplex tableau is the algorithm's main data structure.
65  /// The dashed parts always contain the zero respectively the identity matrix,
66  /// and therefore are not stored explicitly.
67  ///
68  /// ◀───nColumns────▶
69  /// nParameters────┐
70  /// ◀─┴─▶
71  /// ┌─────┬───────────┬ ─ ─ ─ ─ ┐
72  /// ▲│. . .│. . ... . .│ 0 ▲
73  /// nObjectives││. . .│. . ... . .│ │ │
74  /// ▼│. . .│. . ... . .│ │
75  /// ├─────┼───────────┼ ─ ─ ─ ─ ┤ │
76  /// firstConstraintRow > │. . .│. . ... . .│1 │nRows
77  /// │. . .│. . ... . .│ 1 │ │
78  /// │. . .│. . ... . .│ 1 │
79  /// │. . .│. . ... . .│ 1 │ │
80  /// │. . .│. . ... . .│ 1 ▼
81  /// └─────┴───────────┴ ─ ─ ─ ─ ┘
82  /// parameter1Column ^
83  /// parameterSColumn ^
84  /// parameterTColumn ^
85  /// firstNonBasicVariableColumn ^
86  /// ─────────── ──────────
87  /// nonBasicVariables basicVariables
88  SmallVector<SmallVector<int>> tableau;
89 
90  /// During the pivot operation, one column in the elided part of the tableau
91  /// is modified; this vector temporarily catches the changes.
92  SmallVector<int> implicitBasicVariableColumnVector;
93 
94  /// The linear program models the operations' start times as variables, which
95  /// we identify here as 0, ..., |ops|-1.
96  /// Additionally, for each dependence (precisely, the inequality modeling the
97  /// precedence constraint), a slack variable is required; these are identified
98  /// as |ops|, ..., |ops|+|deps|-1.
99  ///
100  /// This vector stores the numeric IDs of non-basic variables. A variable's
101  /// index *i* in this vector corresponds to the tableau *column*
102  /// `firstNonBasicVariableColumn`+*i*.
103  SmallVector<unsigned> nonBasicVariables;
104 
105  /// This vector store the numeric IDs of basic variables. A variable's index
106  /// *i* in this vector corresponds to the tableau *row*
107  /// `firstConstraintRow`+*i*.
108  SmallVector<unsigned> basicVariables;
109 
110  /// Used to conveniently retrieve an operation's start time variable. The
111  /// alternative would be to find the op's index in the problem's list of
112  /// operations.
113  DenseMap<Operation *, unsigned> startTimeVariables;
114 
115  /// This vector keeps track of the current locations (i.e. row or column) of
116  /// a start time variable in the tableau. We encode column numbers as positive
117  /// integers, and row numbers as negative integers. We do not track the slack
118  /// variables.
119  SmallVector<int> startTimeLocations;
120 
121  /// Non-basic variables can be "frozen" to a specific value, which prevents
122  /// them from being pivoted into basis again.
123  DenseMap<unsigned, unsigned> frozenVariables;
124 
125  /// Number of rows in the tableau = |obj| + |deps|.
126  unsigned nRows;
127  /// Number of explicitly stored columns in the tableau = |params| + |ops|.
128  unsigned nColumns;
129 
130  // Number of objective rows.
131  unsigned nObjectives;
132  /// All other rows encode linear constraints.
133  unsigned &firstConstraintRow = nObjectives;
134 
135  // Number of parameters (fixed for now).
136  static constexpr unsigned nParameters = 3;
137  /// The first column corresponds to the always-one "parameter" in u = (1,S,T).
138  static constexpr unsigned parameter1Column = 0;
139  /// The second column corresponds to the variable-freezing parameter S.
140  static constexpr unsigned parameterSColumn = 1;
141  /// The third column corresponds to the parameter T, i.e. the current II.
142  static constexpr unsigned parameterTColumn = 2;
143  /// All other (explicitly stored) columns represent non-basic variables.
144  static constexpr unsigned firstNonBasicVariableColumn = nParameters;
145 
146  /// Allow subclasses to collect additional constraints that are not part of
147  /// the input problem, but should be modeled in the linear problem.
148  SmallVector<Problem::Dependence> additionalConstraints;
149 
150  virtual Problem &getProblem() = 0;
151  virtual bool fillObjectiveRow(SmallVector<int> &row, unsigned obj);
152  virtual void fillConstraintRow(SmallVector<int> &row,
153  Problem::Dependence dep);
154  virtual void fillAdditionalConstraintRow(SmallVector<int> &row,
155  Problem::Dependence dep);
156  void buildTableau();
157 
158  int getParametricConstant(unsigned row);
159  SmallVector<int> getObjectiveVector(unsigned column);
160  Optional<unsigned> findDualPivotRow();
161  Optional<unsigned> findDualPivotColumn(unsigned pivotRow,
162  bool allowPositive = false);
163  Optional<unsigned> findPrimalPivotColumn();
164  Optional<unsigned> findPrimalPivotRow(unsigned pivotColumn);
165  void multiplyRow(unsigned row, int factor);
166  void addMultipleOfRow(unsigned sourceRow, int factor, unsigned targetRow);
167  void pivot(unsigned pivotRow, unsigned pivotColumn);
168  LogicalResult solveTableau();
169  LogicalResult restoreDualFeasibility();
170  bool isInBasis(unsigned startTimeVariable);
171  unsigned freeze(unsigned startTimeVariable, unsigned timeStep);
172  void translate(unsigned column, int factor1, int factorS, int factorT);
173  LogicalResult scheduleAt(unsigned startTimeVariable, unsigned timeStep);
174  void moveBy(unsigned startTimeVariable, unsigned amount);
175  unsigned getStartTime(unsigned startTimeVariable);
176 
177  LogicalResult checkLastOp();
178  void dumpTableau();
179 
180 public:
181  explicit SimplexSchedulerBase(Operation *lastOp) : lastOp(lastOp) {}
182  virtual ~SimplexSchedulerBase() = default;
183  virtual LogicalResult schedule() = 0;
184 };

298 bool SimplexSchedulerBase::fillObjectiveRow(SmallVector<int> &row,
299  unsigned obj) {
300  assert(obj == 0);
301  // Minimize start time of user-specified last operation.
302  row[startTimeLocations[startTimeVariables[lastOp]]] = 1;
303  return false;
304 }

306 void SimplexSchedulerBase::fillConstraintRow(SmallVector<int> &row,
307  Problem::Dependence dep) {
308  auto &prob = getProblem();
309  Operation *src = dep.getSource();
310  Operation *dst = dep.getDestination();
311  unsigned latency = *prob.getLatency(*prob.getLinkedOperatorType(src));
312  row[parameter1Column] = -latency; // note the negation
313  if (src != dst) { // note that these coefficients just zero out in self-arcs.
314  row[startTimeLocations[startTimeVariables[src]]] = 1;
315  row[startTimeLocations[startTimeVariables[dst]]] = -1;
316  }
317 }

326 void SimplexSchedulerBase::buildTableau() {
327  auto &prob = getProblem();
328 
329  // The initial tableau is constructed so that operations' start time variables
330  // are out of basis, whereas all slack variables are in basis. We will number
331  // them accordingly.
332  unsigned var = 0;
333 
334  // Assign column and variable numbers to the operations' start times.
335  for (auto *op : prob.getOperations()) {
336  nonBasicVariables.push_back(var);
337  startTimeVariables[op] = var;
338  startTimeLocations.push_back(firstNonBasicVariableColumn + var);
339  ++var;
340  }
341 
342  // one column for each parameter (1,S,T), and for all operations
343  nColumns = nParameters + nonBasicVariables.size();
344 
345  // Helper to grow both the tableau and the implicit column vector.
346  auto addRow = [&]() -> SmallVector<int> & {
347  implicitBasicVariableColumnVector.push_back(0);
348  return tableau.emplace_back(nColumns, 0);
349  };
350 
351  // Set up the objective rows.
352  nObjectives = 0;
353  bool hasMoreObjectives;
354  do {
355  auto &objRowVec = addRow();
356  hasMoreObjectives = fillObjectiveRow(objRowVec, nObjectives);
357  ++nObjectives;
358  } while (hasMoreObjectives);
359 
360  // Now set up rows/constraints for the dependences.
361  for (auto *op : prob.getOperations()) {
362  for (auto &dep : prob.getDependences(op)) {
363  auto &consRowVec = addRow();
364  fillConstraintRow(consRowVec, dep);
365  basicVariables.push_back(var);
366  ++var;
367  }
368  }
369  for (auto &dep : additionalConstraints) {
370  auto &consRowVec = addRow();
371  fillAdditionalConstraintRow(consRowVec, dep);
372  basicVariables.push_back(var);
373  ++var;
374  }
375 
376  // one row per objective + one row per dependence
377  nRows = tableau.size();
378 }

380 int SimplexSchedulerBase::getParametricConstant(unsigned row) {
381  auto &rowVec = tableau[row];
382  // Compute the dot-product ~B[row] * u between the constant matrix and the
383  // parameter vector.
384  return rowVec[parameter1Column] + rowVec[parameterSColumn] * parameterS +
385  rowVec[parameterTColumn] * parameterT;
386 }

388 SmallVector<int> SimplexSchedulerBase::getObjectiveVector(unsigned column) {
389  SmallVector<int> objVec;
390  // Extract the column vector C^T[column] from the cost matrix.
391  for (unsigned obj = 0; obj < nObjectives; ++obj)
392  objVec.push_back(tableau[obj][column]);
393  return objVec;
394 }

396 Optional<unsigned> SimplexSchedulerBase::findDualPivotRow() {
397  // Find the first row in which the parametric constant is negative.
398  for (unsigned row = firstConstraintRow; row < nRows; ++row)
399  if (getParametricConstant(row) < 0)
400  return row;
401 
402  return None;
403 }

405 Optional<unsigned>
406 SimplexSchedulerBase::findDualPivotColumn(unsigned pivotRow,
407  bool allowPositive) {
408  SmallVector<int> maxQuot(nObjectives, std::numeric_limits<int>::min());
409  Optional<unsigned> pivotCol;
410 
411  // Look for non-zero entries in the constraint matrix (~A part of the
412  // tableau). If multiple candidates exist, take the one corresponding to the
413  // lexicographical maximum (over the objective rows) of the quotients:
414  // tableau[<objective row>][col] / pivotCand
415  for (unsigned col = firstNonBasicVariableColumn; col < nColumns; ++col) {
416  if (frozenVariables.count(
417  nonBasicVariables[col - firstNonBasicVariableColumn]))
418  continue;
419 
420  int pivotCand = tableau[pivotRow][col];
421  // Only negative candidates bring us closer to the optimal solution.
422  // However, when freezing variables to a certain value, we accept that the
423  // value of the objective function degrades.
424  if (pivotCand < 0 || (allowPositive && pivotCand > 0)) {
425  // The constraint matrix has only {-1, 0, 1} entries by construction.
426  assert(pivotCand * pivotCand == 1);
427 
428  SmallVector<int> quot;
429  for (unsigned obj = 0; obj < nObjectives; ++obj)
430  quot.push_back(tableau[obj][col] / pivotCand);
431 
432  if (std::lexicographical_compare(maxQuot.begin(), maxQuot.end(),
433  quot.begin(), quot.end())) {
434  maxQuot = quot;
435  pivotCol = col;
436  }
437  }
438  }
439 
440  return pivotCol;
441 }

443 Optional<unsigned> SimplexSchedulerBase::findPrimalPivotColumn() {
444  // Find the first lexico-negative column in the cost matrix.
445  SmallVector<int> zeroVec(nObjectives, 0);
446  for (unsigned col = firstNonBasicVariableColumn; col < nColumns; ++col) {
447  if (frozenVariables.count(
448  nonBasicVariables[col - firstNonBasicVariableColumn]))
449  continue;
450 
451  SmallVector<int> objVec = getObjectiveVector(col);
452  if (std::lexicographical_compare(objVec.begin(), objVec.end(),
453  zeroVec.begin(), zeroVec.end()))
454  return col;
455  }
456 
457  return None;
458 }

460 Optional<unsigned>
461 SimplexSchedulerBase::findPrimalPivotRow(unsigned pivotColumn) {
462  int minQuot = std::numeric_limits<int>::max();
463  Optional<unsigned> pivotRow;
464 
465  // Look for positive entries in the constraint matrix (~A part of the
466  // tableau). If multiple candidates exist, take the one corresponding to the
467  // minimum of the quotient:
468  // parametricConstant(row) / pivotCand
469  for (unsigned row = firstConstraintRow; row < nRows; ++row) {
470  int pivotCand = tableau[row][pivotColumn];
471  if (pivotCand > 0) {
472  // The constraint matrix has only {-1, 0, 1} entries by construction.
473  assert(pivotCand == 1);
474  int quot = getParametricConstant(row) / pivotCand;
475  if (quot < minQuot) {
476  minQuot = quot;
477  pivotRow = row;
478  }
479  }
480  }
481 
482  return pivotRow;
483 }

486  assert(factor != 0);

493 void SimplexSchedulerBase::addMultipleOfRow(unsigned sourceRow, int factor,
494  unsigned targetRow) {
495  assert(factor != 0 && sourceRow != targetRow);
496  for (unsigned col = 0; col < nColumns; ++col)
497  tableau[targetRow][col] += tableau[sourceRow][col] * factor;
498  // Again, perform row operation on the temporary column vector as well.
499  implicitBasicVariableColumnVector[targetRow] +=
500  implicitBasicVariableColumnVector[sourceRow] * factor;
501 }

503 /// The pivot operation applies elementary row operations to the tableau in
504 /// order to make the \p pivotColumn (corresponding to a non-basic variable) a
505 /// unit vector (only the \p pivotRow'th entry is 1). Then, a basis exchange is
506 /// performed: the non-basic variable is swapped with the basic variable
507 /// associated with the pivot row.
508 void SimplexSchedulerBase::pivot(unsigned pivotRow, unsigned pivotColumn) {
509  // The implicit columns are part of an identity matrix.
510  implicitBasicVariableColumnVector[pivotRow] = 1;
511 
512  int pivotElem = tableau[pivotRow][pivotColumn];
513  // The constraint matrix has only {-1, 0, 1} entries by construction.
514  assert(pivotElem * pivotElem == 1);
515  // Make `tableau[pivotRow][pivotColumn]` := 1
516  multiplyRow(pivotRow, 1 / pivotElem);
517 
518  for (unsigned row = 0; row < nRows; ++row) {
519  if (row == pivotRow)
520  continue;
521 
522  int elem = tableau[row][pivotColumn];
523  if (elem == 0)
524  continue; // nothing to do
525 
526  // Make `tableau[row][pivotColumn]` := 0.
527  addMultipleOfRow(pivotRow, -elem, row);
528  }
529 
530  // Swap the pivot column with the implicitly constructed column vector.
531  // We really only need to copy in one direction here, as the former pivot
532  // column is a unit vector, which is not stored explicitly.
533  for (unsigned row = 0; row < nRows; ++row) {
534  tableau[row][pivotColumn] = implicitBasicVariableColumnVector[row];
535  implicitBasicVariableColumnVector[row] = 0; // Reset for next pivot step.
536  }
537 
538  // Look up numeric IDs of variables involved in this pivot operation.
539  unsigned &nonBasicVar =
540  nonBasicVariables[pivotColumn - firstNonBasicVariableColumn];
541  unsigned &basicVar = basicVariables[pivotRow - firstConstraintRow];
542 
543  // Keep track of where start time variables are; ignore slack variables.
544  if (nonBasicVar < startTimeLocations.size())
545  startTimeLocations[nonBasicVar] = -pivotRow; // ...going into basis.
546  if (basicVar < startTimeLocations.size())
547  startTimeLocations[basicVar] = pivotColumn; // ...going out of basis.
548 
549  // Record the swap in the variable lists.
550  std::swap(nonBasicVar, basicVar);
551 }

553 LogicalResult SimplexSchedulerBase::solveTableau() {
554  // "Solving" technically means perfoming dual pivot steps until primal
555  // feasibility is reached, i.e. the parametric constants in all constraint
556  // rows are non-negative.
557  while (auto pivotRow = findDualPivotRow()) {
558  // Look for pivot elements.
559  if (auto pivotCol = findDualPivotColumn(*pivotRow)) {
560  pivot(*pivotRow, *pivotCol);
561  continue;
562  }
563 
564  // If we did not find a pivot column, then the entire row contained only
565  // positive entries, and the problem is in principle infeasible. However, if
566  // the entry in the `parameterTColumn` is positive, we can make the LP
567  // feasible again by increasing the II.
568  int entry1Col = tableau[*pivotRow][parameter1Column];
569  int entryTCol = tableau[*pivotRow][parameterTColumn];
570  if (entryTCol > 0) {
571  // The negation of `entry1Col` is not in the paper. I think this is an
572  // oversight, because `entry1Col` certainly is negative (otherwise the row
573  // would not have been a valid pivot row), and without the negation, the
574  // new II would be negative.
575  assert(entry1Col < 0);
576  parameterT = (-entry1Col - 1) / entryTCol + 1;
577 
578  LLVM_DEBUG(dbgs() << "Increased II to " << parameterT << '\n');
579 
580  continue;
581  }
582 
583  // Otherwise, the linear program is infeasible.
584  return failure();
585  }
586 
587  // Optimal solution found!
588  return success();
589 }

591 LogicalResult SimplexSchedulerBase::restoreDualFeasibility() {
592  // Dual feasibility requires that all columns in the cost matrix are
593  // non-lexico-negative. This property may be violated after changing the order
594  // of the objective rows, and can be restored by performing primal pivot
595  // steps.
596  while (auto pivotCol = findPrimalPivotColumn()) {
597  // Look for pivot elements.
598  if (auto pivotRow = findPrimalPivotRow(*pivotCol)) {
599  pivot(*pivotRow, *pivotCol);
600  continue;
601  }
602 
603  // Otherwise, the linear program is unbounded.
604  return failure();
605  }
606 
607  // Optimal solution found!
608  return success();
609 }

611 bool SimplexSchedulerBase::isInBasis(unsigned startTimeVariable) {
612  assert(startTimeVariable < startTimeLocations.size());
613  int loc = startTimeLocations[startTimeVariable];

621 unsigned SimplexSchedulerBase::freeze(unsigned startTimeVariable,
622  unsigned timeStep) {
623  assert(startTimeVariable < startTimeLocations.size());
624  assert(!frozenVariables.count(startTimeVariable));
625 
626  // Mark variable.
627  frozenVariables[startTimeVariable] = timeStep;
628 
629  if (!isInBasis(startTimeVariable))
630  // That's all for non-basic variables.
631  return startTimeLocations[startTimeVariable];
632 
633  // We need to pivot this variable one out of basis.
634  unsigned pivotRow = -startTimeLocations[startTimeVariable];
635 
636  // Here, positive pivot elements can be considered as well, hence finding a
637  // suitable column should not fail.
638  auto pivotCol = findDualPivotColumn(pivotRow, /* allowPositive= */ true);
639  assert(pivotCol);
640 
641  // Perform the exchange.
642  pivot(pivotRow, *pivotCol);
643

662 LogicalResult SimplexSchedulerBase::scheduleAt(unsigned startTimeVariable,
663  unsigned timeStep) {
664  assert(startTimeVariable < startTimeLocations.size());
665  assert(!frozenVariables.count(startTimeVariable));
666 
667  // Freeze variable and translate its column by parameter S.
668  unsigned frozenCol = freeze(startTimeVariable, timeStep);
669  translate(frozenCol, /* factor1= */ 0, /* factorS= */ 1, /* factorT= */ 0);
670 
671  // Temporarily set S to the desired value, and attempt to solve.
672  parameterS = timeStep;
673  auto solved = solveTableau();
674  parameterS = 0;
675 
676  if (failed(solved)) {
677  // The LP is infeasible with the new constraint. We could try other values
678  // for S, but for now, we just roll back and signal failure to the driver.
679  translate(frozenCol, /* factor1= */ 0, /* factorS= */ -1, /* factorT= */ 0);
680  frozenVariables.erase(startTimeVariable);
681  auto solvedAfterRollback = solveTableau();
682  assert(succeeded(solvedAfterRollback));
683  (void)solvedAfterRollback;
684  return failure();
685  }
686 
687  // Translate S by the other parameter(s). For acyclic problems, this means
688  // setting `factor1` to `timeStep`. For cyclic problems, we perform a modulo
689  // decomposition: S = `factor1` + `factorT` * T, with `factor1` < T.
690  //
691  // This translation does not change the values of the parametric constants,
692  // hence we do not need to solve the tableau again.
693  //
694  // Note: I added a negation of the factors here, which is not mentioned in the
695  // paper's text, but apparently used in the example. Without it, the intended
696  // effect, i.e. making the S-column all-zero again, is not achieved.
697  if (parameterT == 0)
698  translate(parameterSColumn, /* factor1= */ -timeStep, /* factorS= */ 1,
699  /* factorT= */ 0);
700  else
701  translate(parameterSColumn, /* factor1= */ -(timeStep % parameterT),
702  /* factorS= */ 1,
703  /* factorT= */ -(timeStep / parameterT));
704 
705  return success();
706 }

851 LogicalResult SharedOperatorsSimplexScheduler::schedule() {
852  if (failed(checkLastOp()))
853  return failure();
854 
855  parameterS = 0;
856  parameterT = 0;
857  buildTableau();
858 
859  LLVM_DEBUG(dbgs() << "Initial tableau:\n"; dumpTableau());
860 
861  if (failed(solveTableau()))
862  return prob.getContainingOp()->emitError() << "problem is infeasible";
863 
864  LLVM_DEBUG(dbgs() << "After solving resource-free problem:\n"; dumpTableau());
865 
866  // The *heuristic* part of this scheduler starts here:
867  // We will now *choose* start times for operations using a shared operator
868  // type, in a way that respects the allocation limits, and consecutively solve
869  // the LP with these added constraints. The individual LPs are still solved to
870  // optimality (meaning: the start times of the "last" operation is still
871  // optimal w.r.t. the already fixed operations), however the heuristic choice
872  // means we cannot guarantee the optimality for the overall problem.
873 
874  // Determine which operations are subject to resource constraints.
875  auto &ops = prob.getOperations();
876  SmallVector<Operation *> limitedOps;
877  for (auto *op : ops)
878  if (isLimited(op, prob))
879  limitedOps.push_back(op);
880 
881  // Build a priority list of the limited operations.
882  //
883  // We sort by the resource-free start times to produce a topological order of
884  // the operations. Better priority functions are known, but require computing
885  // additional properties, e.g. ASAP and ALAP times for mobility, or graph
886  // analysis for height. Assigning operators (=resources) in this order at
887  // least ensures that the (acyclic!) problem remains feasible throughout the
888  // process.
889  //
890  // TODO: Implement more sophisticated priority function.
891  std::stable_sort(limitedOps.begin(), limitedOps.end(),
892  [&](Operation *a, Operation *b) {
893  return getStartTime(startTimeVariables[a]) <
894  getStartTime(startTimeVariables[b]);
895  });
896 
897  // Store the number of operations using an operator type in a particular time
898  // step.
899  SmallDenseMap<Problem::OperatorType, SmallDenseMap<unsigned, unsigned>>
900  reservationTable;
901 
902  for (auto *op : limitedOps) {
903  auto opr = *prob.getLinkedOperatorType(op);
904  unsigned limit = prob.getLimit(opr).getValueOr(0);
905  assert(limit > 0);
906 
907  // Find the first time step (beginning at the current start time in the
908  // partial schedule) in which an operator instance is available.
909  unsigned startTimeVar = startTimeVariables[op];
910  unsigned candTime = getStartTime(startTimeVar);
911  while (reservationTable[opr].lookup(candTime) == limit)
912  ++candTime;
913 
914  // Fix the start time. As explained above, this cannot make the problem
915  // infeasible.
916  auto fixed = scheduleAt(startTimeVar, candTime);
917  assert(succeeded(fixed));
918  (void)fixed;
919 
920  // Record the operator use.
921  ++reservationTable[opr][candTime];
922 
923  LLVM_DEBUG(dbgs() << "After scheduling " << startTimeVar
924  << " to t=" << candTime << ":\n";
925  dumpTableau());
926  }
927 
928  assert(parameterT == 0);
929  LLVM_DEBUG(
930  dbgs() << "Final tableau:\n"; dumpTableau();
931  dbgs() << "Feasible solution found with start time of last operation = "
932  << -getParametricConstant(0) << '\n');
933 
934  for (auto *op : ops)
935  prob.setStartTime(op, getStartTime(startTimeVariables[op]));
936 
937  return success();
938 }

972 bool ModuloSimplexScheduler::fillObjectiveRow(SmallVector<int> &row,
973  unsigned obj) {

979  case OBJ_AXAP:

990 void ModuloSimplexScheduler::updateMargins() {
991  // Assumption: current secondary objective is "ASAP".
992  // Negate the objective row once to effectively maximize the sum of start
993  // times, which yields the "ALAP" times after solving the tableau. Then,
994  // negate it again to restore the "ASAP" objective, and store these times as
995  // well.
996  for (auto *axapTimes : {&alapTimes, &asapTimes}) {
997  multiplyRow(OBJ_AXAP, -1);
998  // This should not fail for a feasible tableau.
999  auto dualFeasRestored = restoreDualFeasibility();
1000  auto solved = solveTableau();
1001  assert(succeeded(dualFeasRestored) && succeeded(solved));
1002  (void)dualFeasRestored, (void)solved;
1003 
1004  for (unsigned stv = 0; stv < startTimeLocations.size(); ++stv)
1005  (*axapTimes)[stv] = getStartTime(stv);
1006  }

1009 void ModuloSimplexScheduler::incrementII() {
1010  // Account for the shift in the frozen start times that will be caused by
1011  // increasing `parameterT`: Assuming decompositions of
1012  // t = phi * II + tau and t' = phi * (II + 1) + tau,
1013  // so the required shift is
1014  // t' - t = phi = floordiv(t / II)
1015  for (auto &kv : frozenVariables) {
1016  unsigned &frozenTime = kv.getSecond();
1017  frozenTime += frozenTime / parameterT;
1018  }
1019 
1020  // Increment the parameter.
1021  ++parameterT;
1022 }

1024 void ModuloSimplexScheduler::scheduleOperation(Operation *n) {
1025  unsigned stvN = startTimeVariables[n];
1026 
1027  // Get current state of the LP, and determine range of alternative times
1028  // guaranteed to be feasible.
1029  unsigned stN = getStartTime(stvN);
1030  unsigned lbN = (unsigned)std::max<int>(asapTimes[stvN], stN - parameterT + 1);
1031  unsigned ubN = (unsigned)std::min<int>(alapTimes[stvN], lbN + parameterT - 1);
1032 
1033  LLVM_DEBUG(dbgs() << "Attempting to schedule at t=" << stN << ", or in ["
1034  << lbN << ", " << ubN << "]: " << *n << '\n');
1035 
1036  SmallVector<unsigned> candTimes;
1037  candTimes.push_back(stN);
1038  for (unsigned ct = lbN; ct <= ubN; ++ct)
1039  if (ct != stN)
1040  candTimes.push_back(ct);
1041 
1042  for (unsigned ct : candTimes)
1043  if (succeeded(mrt.enter(n, ct))) {
1044  auto fixedN = scheduleAt(stvN, stN);
1045  assert(succeeded(fixedN));
1046  (void)fixedN;
1047  LLVM_DEBUG(dbgs() << "Success at t=" << stN << " " << *n << '\n');
1048  return;
1049  }
1050 
1051  // As a last resort, increase II to make room for the op. De Dinechin's
1052  // Theorem 1 lays out conditions/guidelines to transform the current partial
1053  // schedule for II to a valid one for a larger II'.
1054 
1055  LLVM_DEBUG(dbgs() << "Incrementing II to " << (parameterT + 1)
1056  << " to resolve resource conflict for " << *n << '\n');
1057 
1058  // Note that the approach below is much simpler than in the paper
1059  // because of the fully-pipelined operators. In our case, it's always
1060  // sufficient to increment the II by one.
1061 
1062  // Decompose start time.
1063  unsigned phiN = stN / parameterT;
1064  unsigned tauN = stN % parameterT;
1065 
1066  // Keep track whether the following moves free at least one operator
1067  // instance in the slot desired by the current op - then it can stay there.
1068  unsigned deltaN = 1;
1069 
1070  // We're going to revisit the current partial schedule.
1071  SmallVector<Operation *> moved;
1072  for (Operation *j : scheduled) {
1073  unsigned stvJ = startTimeVariables[j];
1074  unsigned stJ = getStartTime(stvJ);
1075  unsigned phiJ = stJ / parameterT;
1076  unsigned tauJ = stJ % parameterT;
1077  unsigned deltaJ = 0;
1078 
1079  // To actually resolve the resource conflicts, we move operations that are
1080  // "preceded" (cf. de Dinechin's ≺ relation) one slot to the right.
1081  if (tauN < tauJ || (tauN == tauJ && phiN > phiJ) ||
1082  (tauN == tauJ && phiN == phiJ && stvN < stvJ)) {
1083  // TODO: Replace the last condition with a proper graph analysis.
1084 
1085  deltaJ = 1;
1086  moved.push_back(j);
1087  if (tauN == tauJ)
1088  deltaN = 0;
1089  }
1090 
1091  // Apply the move to the tableau.
1092  moveBy(stvJ, deltaJ);
1093  }
1094 
1095  // Finally, increment the II.
1096  incrementII();
1097  auto solved = solveTableau();
1098  assert(succeeded(solved));
1099  (void)solved;
1100 
1101  // Re-enter moved operations into their new slots.
1102  for (auto *m : moved)
1103  mrt.release(m);
1104  for (auto *m : moved) {
1105  auto enteredM = mrt.enter(m, getStartTime(startTimeVariables[m]));
1106  assert(succeeded(enteredM));
1107  (void)enteredM;
1108  }
1109 
1110  // Finally, schedule the operation. Adding `phiN` accounts for the implicit
1111  // shift caused by incrementing the II; cf. `incrementII()`.
1112  auto fixedN = scheduleAt(stvN, stN + phiN + deltaN);
1113  auto enteredN = mrt.enter(n, tauN + deltaN);
1114  assert(succeeded(fixedN) && succeeded(enteredN));
1115  (void)fixedN, (void)enteredN;
1116 }

1118 LogicalResult ModuloSimplexScheduler::schedule() {
1119  if (failed(checkLastOp()))
1120  return failure();
1121 
1122  parameterS = 0;
1123  parameterT = 1;
1124  buildTableau();
1125  asapTimes.resize(startTimeLocations.size());
1126  alapTimes.resize(startTimeLocations.size());
1127 
1128  LLVM_DEBUG(dbgs() << "Initial tableau:\n"; dumpTableau());
1129 
1130  if (failed(solveTableau()))
1131  return prob.getContainingOp()->emitError() << "problem is infeasible";
1132 
1133  // Determine which operations are subject to resource constraints.
1134  auto &ops = prob.getOperations();
1135  for (auto *op : ops)
1136  if (isLimited(op, prob))
1137  unscheduled.push_back(op);
1138 
1139  // Main loop: Iteratively fix limited operations to time steps.
1140  while (!unscheduled.empty()) {
1141  // Update ASAP/ALAP times.
1142  updateMargins();
1143 
1144  // Heuristically (here: least amount of slack) pick the next operation to
1145  // schedule.
1146  auto *opIt =
1147  std::min_element(unscheduled.begin(), unscheduled.end(),
1148  [&](Operation *opA, Operation *opB) {
1149  auto stvA = startTimeVariables[opA];
1150  auto stvB = startTimeVariables[opB];
1151  auto slackA = alapTimes[stvA] - asapTimes[stvA];
1152  auto slackB = alapTimes[stvB] - asapTimes[stvB];
1153  return slackA < slackB;
1154  });
1155  Operation *op = *opIt;
1156  unscheduled.erase(opIt);
1157 
1158  scheduleOperation(op);
1159  scheduled.push_back(op);
1160  }
1161 
1162  LLVM_DEBUG(dbgs() << "Final tableau:\n"; dumpTableau();
1163  dbgs() << "Solution found with II = " << parameterT
1164  << " and start time of last operation = "
1165  << -getParametricConstant(0) << '\n');
1166 
1167  prob.setInitiationInterval(parameterT);
1168  for (auto *op : ops)
1169  prob.setStartTime(op, getStartTime(startTimeVariables[op]));
1170 
1171  return success();
1172 }
