39 /// A value imported into a split.
40 struct ImportedValue {
41  /// Indicates where this value originates from. If true, the value is an input
42  /// of the original arc. If false the value is exported from a different
43  /// split.
44  unsigned isInput : 1;
45  /// The original arc's input number, or the result number of the split that
46  /// exports this value.
47  unsigned index : 15;
48  /// If this value is exported from a different split, this is that split's
49  /// index.
50  unsigned split : 16;
51 };

53 /// A single arc split out from another arc.
54 struct Split {
55  Split(MLIRContext *context, unsigned index, const APInt &color)
56  : index(index), color(color), block(std::make_unique<Block>()),
57  builder(context) {
58  builder.setInsertionPointToStart(block.get());
59  }
60 
61  /// Map an input of the original arc into this split.
62  void importInput(BlockArgument arg) {
63  importedValues.push_back({true, arg.getArgNumber(), 0});
64  mapping.map(arg, block->addArgument(arg.getType(), arg.getLoc()));
65  }
66 
67  /// Map a value in a different split into this split.
68  void importFromOtherSplit(Value value, Split &otherSplit) {
69  auto resultIdx = otherSplit.exportValue(value);
70  importedValues.push_back({false, resultIdx, otherSplit.index});
71  mapping.map(value, block->addArgument(value.getType(), value.getLoc()));
72  }
73 
74  /// Export a value in this split as an output. Returns result number this
75  /// value will have.
76  unsigned exportValue(Value value) {
77  value = mapping.lookup(value);
78  auto result = exportedValueIndices.insert({value, exportedValues.size()});
79  if (result.second)
80  exportedValues.push_back(value);
81  return result.first->second;
82  }
83 
84  unsigned index;
85  APInt color;
86 
87  std::unique_ptr<Block> block;
88  OpBuilder builder;
89  IRMapping mapping;
90 
91  /// Where each value mapped to a block argument is coming from.
92  SmallVector<ImportedValue> importedValues;
93  /// Which values of this split are exposed as outputs.
94  SmallVector<Value> exportedValues;
95  SmallDenseMap<Value, unsigned> exportedValueIndices;
96 };

98 /// Helper structure to split one arc into multiple ones.
99 struct Splitter {
100  Splitter(MLIRContext *context, Location loc) : context(context), loc(loc) {}
101  void run(Block &block, DenseMap<Operation *, APInt> &coloring);
102  Split &getSplit(const APInt &color);
103 
104  MLIRContext *context;
105  Location loc;
106 
107  /// A split for each distinct operation coloring in the original arc.
108  SmallVector<Split *> splits;
109  SmallDenseMap<APInt, std::unique_ptr<Split>> splitsByColor;
110 
111  /// Where each of the original arc's outputs come from after splitting.
112  SmallVector<ImportedValue> outputs;
113 };

116 /// Create a separate arc body block for every unique coloring of operations.
117 void Splitter::run(Block &block, DenseMap<Operation *, APInt> &coloring) {
118  for (auto &op : block.without_terminator()) {
119  auto color = coloring.lookup(&op);
120  auto &split = getSplit(color);
121 
122  // Collect the operands of the current operation.
123  SmallSetVector<Value, 4> operands;
124  op.walk([&](Operation *op) {
125  for (auto operand : op->getOperands())
126  if (operand.getParentBlock() == &block)
127  operands.insert(operand);
128  });
129 
130  // Each operand that is either an input of the original arc or that is
131  // defined by an operation that got moved to a different split, create an
132  // input to the current split for that value.
133  for (auto operand : operands) {
134  if (split.mapping.contains(operand))
135  continue;
136  if (auto blockArg = dyn_cast<BlockArgument>(operand)) {
137  split.importInput(blockArg);
138  continue;
139  }
140  auto *operandOp = operand.getDefiningOp();
141  auto operandColor = coloring.lookup(operandOp);
142  assert(operandOp && color != operandColor);
143  auto &operandSplit = getSplit(operandColor);
144  split.importFromOtherSplit(operand, operandSplit);
145  }
146 
147  // Move the operation into the split.
148  split.builder.clone(op, split.mapping);
149  }
150 
151  // Reconstruct where each of the original arc outputs got mapped to.
152  for (auto operand : block.getTerminator()->getOperands()) {
153  if (auto blockArg = dyn_cast<BlockArgument>(operand)) {
154  outputs.push_back({true, blockArg.getArgNumber(), 0});
155  continue;
156  }
157  auto &operandSplit = getSplit(coloring.lookup(operand.getDefiningOp()));
158  auto resultIdx = operandSplit.exportValue(operand);
159  outputs.push_back({false, resultIdx, operandSplit.index});
160  }
161 
162  // Create the final `arc.output` op for each of the splits.
163  for (auto &split : splits)
164  split->builder.create<arc::OutputOp>(loc, split->exportedValues);
165 }

167 /// Get or create the split for a given operation color.
168 Split &Splitter::getSplit(const APInt &color) {
169  auto &split = splitsByColor[color];
170  if (!split) {
171  auto index = splits.size();
172  LLVM_DEBUG(llvm::dbgs()
173  << "- Creating split " << index << " for " << color << "\n");
174  split = std::make_unique<Split>(context, index, color);
175  splits.push_back(split.get());
176  }
177  return *split;
178 }

197 void SplitLoopsPass::runOnOperation() {
198  auto module = getOperation();
199  allArcUses.clear();
200 
201  // Collect all arc definitions.
202  Namespace arcNamespace;
203  DenseMap<StringAttr, DefineOp> arcDefs;
204  for (auto arcDef : module.getOps<DefineOp>()) {
205  arcNamespace.newName(arcDef.getSymName());
206  arcDefs[arcDef.getSymNameAttr()] = arcDef;
207  }
208 
209  // Collect all arc uses and determine which arcs we should split.
210  SetVector<DefineOp> arcsToSplit;
211  DenseMap<DefineOp, SmallVector<CallOpInterface>> arcUses;
212  SetVector<CallOpInterface> allArcUses;
213 
214  auto result = module.walk([&](CallOpInterface callOp) -> WalkResult {
215  auto refSym = callOp.getCallableForCallee().dyn_cast<SymbolRefAttr>();
216  if (!refSym)
217  return callOp->emitOpError("found unsupported call to an arc");
218 
219  auto defOp = arcDefs.lookup(refSym.getLeafReference());
220  arcUses[defOp].push_back(callOp);
221  allArcUses.insert(callOp);
222 
223  auto clockedOp = dyn_cast<ClockedOpInterface>(callOp.getOperation());
224  if ((!clockedOp || clockedOp.getLatency() == 0) &&
225  callOp->getNumResults() > 1)
226  arcsToSplit.insert(defOp);
227 
228  return WalkResult::advance();
229  });
230 
231  if (result.wasInterrupted())
232  return signalPassFailure();
233 
234  // Split all arcs with more than one result.
235  // TODO: This is ugly and we should only split arcs that are truly involved in
236  // a loop. But detecting the minimal split among the arcs is fairly
237  // non-trivial and needs a dedicated implementation effort.
238  for (auto defOp : arcsToSplit)
239  splitArc(arcNamespace, defOp, arcUses[defOp]);
240 
241  // Ensure that there are no loops through arcs remaining.
242  if (failed(ensureNoLoops()))
243  return signalPassFailure();
244 }

246 /// Split a single arc into a separate arc for each result.
247 void SplitLoopsPass::splitArc(Namespace &arcNamespace, DefineOp defOp,
248  ArrayRef<CallOpInterface> arcUses) {
249  LLVM_DEBUG(llvm::dbgs() << "Splitting arc " << defOp.getSymNameAttr()
250  << "\n");
251 
252  // Mark the operations in the arc according to which result they contribute
253  // to.
254  auto numResults = defOp.getNumResults();
255  DenseMap<Value, APInt> valueColoring;
256  DenseMap<Operation *, APInt> opColoring;
257 
258  for (auto &operand : defOp.getBodyBlock().getTerminator()->getOpOperands())
259  valueColoring.insert(
260  {operand.get(),
261  APInt::getOneBitSet(numResults, operand.getOperandNumber())});
262 
263  for (auto &op : llvm::reverse(defOp.getBodyBlock().without_terminator())) {
264  auto coloring = APInt::getZero(numResults);
265  for (auto result : op.getResults())
266  if (auto it = valueColoring.find(result); it != valueColoring.end())
267  coloring |= it->second;
268  opColoring.insert({&op, coloring});
269  op.walk([&](Operation *op) {
270  for (auto &operand : op->getOpOperands())
271  valueColoring.try_emplace(operand.get(), numResults, 0).first->second |=
272  coloring;
273  });
274  }
275 
276  // Determine the splits for this arc.
277  Splitter splitter(&getContext(), defOp.getLoc());
278  splitter.run(defOp.getBodyBlock(), opColoring);
279 
280  // Materialize the split arc definitions.
281  ImplicitLocOpBuilder builder(defOp.getLoc(), defOp);
282  SmallVector<DefineOp> splitArcs;
283  splitArcs.reserve(splitter.splits.size());
284  for (auto &split : splitter.splits) {
285  auto splitName = defOp.getSymName();
286  if (splitter.splits.size() > 1)
287  splitName = arcNamespace.newName(defOp.getSymName() + "_split_" +
288  Twine(split->index));
289  auto splitArc = builder.create<DefineOp>(
290  splitName, builder.getFunctionType(
291  split->block->getArgumentTypes(),
292  split->block->getTerminator()->getOperandTypes()));
293  splitArc.getBody().push_back(split->block.release());
294  splitArcs.push_back(splitArc);
295  }
296 
297  // Replace all uses with the new splits and remove the old definition.
298  for (auto arcUse : arcUses)
299  replaceArcUse(arcUse, splitArcs, splitter.splits, splitter.outputs);
300  defOp.erase();
301 }

303 /// Replace a use of the original arc with new uses for the splits.
304 void SplitLoopsPass::replaceArcUse(CallOpInterface arcUse,
305  ArrayRef<DefineOp> splitDefs,
306  ArrayRef<Split *> splits,
307  ArrayRef<ImportedValue> outputs) {
308  ImplicitLocOpBuilder builder(arcUse.getLoc(), arcUse);
309  SmallVector<CallOp> newUses(splits.size());
310 
311  // Resolve an `ImportedValue` to either an operand of the original arc or the
312  // result of another split.
313  auto getMappedValue = [&](ImportedValue value) {
314  if (value.isInput)
315  return arcUse.getArgOperands()[value.index];
316  return newUses[value.split].getResult(value.index);
317  };
318 
319  // Collect the operands for each split and create a new use for each. These
320  // are either operands of the original arc, or values from other splits
321  // exported as results.
322  DenseMap<unsigned, unsigned> splitIdxMap;
323  for (auto [i, split] : llvm::enumerate(splits))
324  splitIdxMap[split->index] = i;
325 
326  DenseSet<unsigned> splitsDone;
327  SmallVector<std::pair<const DefineOp, const Split *>> worklist;
328 
329  auto getMappedValuesOrSchedule = [&](ArrayRef<ImportedValue> importedValues,
330  SmallVector<Value> &operands) {
331  for (auto importedValue : importedValues) {
332  if (!importedValue.isInput && !splitsDone.contains(importedValue.split)) {
333  unsigned idx = splitIdxMap[importedValue.split];
334  worklist.push_back({splitDefs[idx], splits[idx]});
335  return false;
336  }
337 
338  operands.push_back(getMappedValue(importedValue));
339  }
340 
341  return true;
342  };
343 
344  // Initialize worklist
345  for (auto [splitDef, split] : llvm::reverse(llvm::zip(splitDefs, splits)))
346  worklist.push_back({splitDef, split});
347 
348  // Process worklist
349  while (!worklist.empty()) {
350  auto [splitDef, split] = worklist.back();
351 
352  if (splitsDone.contains(split->index)) {
353  worklist.pop_back();
354  continue;
355  }
356 
357  SmallVector<Value> operands;
358  if (!getMappedValuesOrSchedule(split->importedValues, operands))
359  continue;
360 
361  auto newUse = builder.create<CallOp>(splitDef, operands);
362  allArcUses.insert(newUse);
363  newUses[split->index] = newUse;
364 
365  splitsDone.insert(split->index);
366  worklist.pop_back();
367  }
368 
369  // Update the users of the original arc results.
370  for (auto [result, importedValue] : llvm::zip(arcUse->getResults(), outputs))
371  result.replaceAllUsesWith(getMappedValue(importedValue));
372  allArcUses.erase(arcUse);
373  arcUse.erase();
374 }

376 /// Check that there are no more zero-latency loops through arcs.
377 LogicalResult SplitLoopsPass::ensureNoLoops() {
378  SmallVector<std::pair<Operation *, unsigned>, 0> worklist;
379  DenseSet<Operation *> finished;
380  DenseSet<Operation *> seen;
381  for (auto op : allArcUses) {
382  if (finished.contains(op))
383  continue;
384  assert(seen.empty());
385  worklist.push_back({op, 0});
386  while (!worklist.empty()) {
387  auto [op, idx] = worklist.back();
388  ++worklist.back().second;
389  if (idx == op->getNumOperands()) {
390  seen.erase(op);
391  finished.insert(op);
392  worklist.pop_back();
393  continue;
394  }
395  auto operand = op->getOperand(idx);
396  auto *def = operand.getDefiningOp();
397  if (!def || finished.contains(def))
398  continue;
399  if (auto clockedOp = dyn_cast<ClockedOpInterface>(def);
400  clockedOp && clockedOp.getLatency() > 0)
401  continue;
402  if (!seen.insert(def).second) {
403  auto d = def->emitError(
404  "loop splitting did not eliminate all loops; loop detected");
405  for (auto [op, idx] : llvm::reverse(worklist)) {
406  d.attachNote(op->getLoc())
407  << "through operand " << (idx - 1) << " here:";
408  if (op == def)
409  break;
410  }
411  return failure();
412  }
413  worklist.push_back({def, 0});
414  }
415  }
416  return success();
417 }
