77 /// Default function for partial lowering of handshake::FuncOp. Lowering is
78 /// achieved by a provided partial lowering function.
79 ///
80 /// A partial lowering function may only replace a subset of the operations
81 /// within the funcOp currently being lowered. However, the dialect conversion
82 /// scheme requires the matched root operation to be replaced/updated, if the
83 /// match was successful. To facilitate this, rewriter.updateRootInPlace
84 /// wraps the partial update function.
85 /// Next, the function operation is expected to go from illegal to legalized,
86 /// after matchAndRewrite returned true. To work around this,
87 /// LowerFuncOpTarget::loweredFuncs is used to communicate between the target
88 /// and the conversion, to indicate that the partial lowering was completed.
89 template <typename TOp>
90 struct PartialLowerOp : public ConversionPattern {
91  using PartialLoweringFunc =
92  std::function<LogicalResult(TOp, ConversionPatternRewriter &)>;
93 
94 public:
95  PartialLowerOp(LowerOpTarget<TOp> &target, MLIRContext *context,
96  LogicalResult &loweringResRef, const PartialLoweringFunc &fun)
97  : ConversionPattern(TOp::getOperationName(), 1, context), target(target),
98  loweringRes(loweringResRef), fun(fun) {}
99  using ConversionPattern::ConversionPattern;
100  LogicalResult
101  matchAndRewrite(Operation *op, ArrayRef<Value> /*operands*/,
102  ConversionPatternRewriter &rewriter) const override {
103  assert(isa<TOp>(op));
104  rewriter.updateRootInPlace(
105  op, [&] { loweringRes = fun(dyn_cast<TOp>(op), rewriter); });
106  target.loweredOps[op] = true;
107  return loweringRes;
108  };
109 
110 private:
111  LowerOpTarget<TOp> &target;
112  LogicalResult &loweringRes;
113  // NOTE: this is basically the rewrite function
114  PartialLoweringFunc fun;
115 };

219 void handshake::removeBasicBlocks(Region &r) {
220  auto &entryBlock = r.front().getOperations();
221 
222  // Erase all TerminatorOp, and move ReturnOp to the end of entry block.
223  for (auto &block : r) {
224  Operation &termOp = block.back();
225  if (isa<handshake::TerminatorOp>(termOp))
226  termOp.erase();
227  else if (isa<handshake::ReturnOp>(termOp))
228  entryBlock.splice(entryBlock.end(), block.getOperations(), termOp);
229  }
230 
231  // Move all operations to entry block and erase other blocks.
232  for (auto &block : llvm::make_early_inc_range(llvm::drop_begin(r, 1))) {
233  entryBlock.splice(--entryBlock.end(), block.getOperations());
234  }
235  for (auto &block : llvm::make_early_inc_range(llvm::drop_begin(r, 1))) {
236  block.clear();
237  block.dropAllDefinedValueUses();
238  for (size_t i = 0; i < block.getNumArguments(); i++) {
239  block.eraseArgument(i);
240  }
241  block.erase();
242  }
243 }

252 static HandshakeLowering::BlockValues getBlockUses(Region &f) {
253  // Returns map of values used in block but defined outside of block
254  // For liveness analysis
255  HandshakeLowering::BlockValues uses;
256  for (Block &block : f) {
257  // Operands of operations in b which do not originate from operations or
258  // arguments of b
259  for (Operation &op : block) {
260  for (int i = 0, e = op.getNumOperands(); i < e; ++i) {
261  Block *operandBlock;
262  if (op.getOperand(i).isa<BlockArgument>()) {
263  // Operand is block argument, get its owner block
264  operandBlock = op.getOperand(i).cast<BlockArgument>().getOwner();
265  } else {
266  // Operand comes from operation, get the block of its defining op
267  auto *operand = op.getOperand(i).getDefiningOp();
268  assert(operand != NULL);
269  operandBlock = operand->getBlock();
270  }
271  // If operand defined in some other block, add to uses
272  if (operandBlock != &block)
273  // Add only unique uses
274  if (std::find(uses[&block].begin(), uses[&block].end(),
275  op.getOperand(i)) == uses[&block].end())
276  uses[&block].push_back(op.getOperand(i));
277  }
278  }
279  }
280  return uses;
281 }

326 static HandshakeLowering::BlockValues livenessAnalysis(Region &f) {
327  // Liveness analysis algorithm adapted from:
328  // https://suif.stanford.edu/~courses/cs243/lectures/l2.pdf
329  // See slide 19 (Liveness: Iterative Algorithm)
330 
331  // blockUses: values used in block but not defined in block
332  HandshakeLowering::BlockValues blockUses = getBlockUses(f);
333 
334  // blockDefs: values defined in block
335  HandshakeLowering::BlockValues blockDefs = getBlockDefs(f);
336 
337  HandshakeLowering::BlockValues blockLiveIns;
338 
339  bool change = true;
340  // Iterate while there are any changes to any of the livein sets
341  while (change) {
342  change = false;
343 
344  // liveOuts(b): all liveins of all successors of b
345  // liveOuts(b) = U (blockLiveIns(s)) forall successors s of b
346  for (Block &block : f) {
347  std::vector<Value> liveOuts;
348  for (int i = 0, e = block.getNumSuccessors(); i < e; ++i) {
349  Block *succ = block.getSuccessor(i);
350  liveOuts = vectorUnion(liveOuts, blockLiveIns[succ]);
351  }
352 
353  // diff(b): liveouts of b which are not defined in b
354  // diff(b) = liveOuts(b) - blockDefs(b)
355  auto diff = vectorDiff(liveOuts, blockDefs[&block]);
356 
357  // liveIns(b) = blockUses(b) U diff(b)
358  auto tmpLiveIns = vectorUnion(blockUses[&block], diff);
359 
360  // Update blockLiveIns if new liveins found
361  if (tmpLiveIns.size() > blockLiveIns[&block].size()) {
362  blockLiveIns[&block] = tmpLiveIns;
363  change = true;
364  }
365  }
366  }
367  return blockLiveIns;
368 }

376 // Insert appropriate type of Merge CMerge for control-only path,
377 // Merge for single-successor blocks, Mux otherwise
378 Operation *HandshakeLowering::insertMerge(Block *block, Value val,
379  ConversionPatternRewriter &rewriter) {
380  unsigned numPredecessors = getBlockPredecessorCount(block);
381 
382  // Control-only path originates from StartOp
383  if (!val.isa<BlockArgument>()) {
384  if (isa<StartOp>(val.getDefiningOp())) {
385  auto cmerge = rewriter.create<handshake::ControlMergeOp>(
386  block->front().getLoc(), val, numPredecessors);
387  setBlockEntryControl(block, cmerge.result());
388  return cmerge;
389  }
390  }
391 
392  // If there are no block predecessors (i.e., entry block)
393  if (numPredecessors <= 1) {
394  SmallVector<Value> mergeOperands;
395  mergeOperands.append(2, val); // 2 dummy values used here due to
396  // hard-coded logic of reconnectMergeOps.
397  return rewriter.create<handshake::MergeOp>(block->front().getLoc(),
398  mergeOperands);
399  }
400 
401  return rewriter.create<handshake::MuxOp>(block->front().getLoc(), val,
402  numPredecessors);
403 }

444 // Get value from predBlock which will be set as operand of op (merge)
445 static Value getMergeOperand(Operation *op, Block *predBlock,
446  HandshakeLowering::BlockOps blockMerges) {
447  // Helper value (defining value of merge) to identify Merges which propagate
448  // the same defining value
449  Value srcVal = op->getOperand(0);
450  Block *block = op->getBlock();
451 
452  // Value comes from predecessor block (i.e., not an argument of this block)
453  if (std::find(block->getArguments().begin(), block->getArguments().end(),
454  srcVal) == block->getArguments().end()) {
455  // Value is not defined by operation in predBlock
456  if (!blockHasSrcOp(srcVal, predBlock)) {
457  // Find the corresponding Merge
458  for (Operation *predOp : blockMerges[predBlock])
459  if (predOp->getOperand(0) == srcVal)
460  return predOp->getResult(0);
461  } else
462  return srcVal;
463  }
464 
465  // Value is argument of this block
466  // Operand of terminator in predecessor block should be input to Merge
467  else {
468  unsigned index = srcVal.cast<BlockArgument>().getArgNumber();
469  Operation *termOp = predBlock->getTerminator();
470  if (mlir::cf::CondBranchOp br = dyn_cast<mlir::cf::CondBranchOp>(termOp)) {
471  if (block == br.getTrueDest())
472  return br.getTrueOperand(index);
473  else {
474  assert(block == br.getFalseDest());
475  return br.getFalseOperand(index);
476  }
477  } else if (isa<mlir::cf::BranchOp>(termOp))
478  return termOp->getOperand(index);
479  }
480  return nullptr;
481 }

517 static void reconnectMergeOps(Region &f,
518  HandshakeLowering::BlockOps blockMerges,
519  HandshakeLowering::blockArgPairs &mergePairs) {
520  // All merge operands are initially set to original (defining) value
521  // We here replace defining value with appropriate value from predecessor
522  // block The predecessor can either be a merge, the original defining value,
523  // or a branch Operand Operand(0) is helper defining value for identifying
524  // matching merges, it does not correspond to any predecessor block
525 
526  for (Block &block : f) {
527  for (Operation *op : blockMerges[&block]) {
528  int count = 1;
529  // Set appropriate operand from predecessor block
530  for (auto *predBlock : block.getPredecessors()) {
531  Value mgOperand = getMergeOperand(op, predBlock, blockMerges);
532  assert(mgOperand != nullptr);
533  if (!mgOperand.getDefiningOp()) {
534  assert(mergePairs.count(mgOperand));
535  mgOperand = mergePairs[mgOperand]->getResult(0);
536  }
537  op->setOperand(count, mgOperand);
538  count++;
539  }
540  // Reconnect all operands originating from livein defining value through
541  // corresponding merge of that block
542  for (Operation &opp : block)
543  if (!isa<MergeLikeOpInterface>(opp))
544  opp.replaceUsesOfWith(op->getOperand(0), op->getResult(0));
545  }
546  }
547 
548  // Disconnect original value (Operand(0), used as helper) from all merges
549  // If the original value must be a merge operand, it is still set as some
550  // subsequent operand
551  // If block has multiple predecessors, connect Muxes to ControlMerge
552  for (Block &block : f) {
553  unsigned numPredecessors = getBlockPredecessorCount(&block);
554 
555  if (numPredecessors <= 1) {
556  for (Operation *op : blockMerges[&block])
557  op->eraseOperand(0);
558  } else {
559  Operation *cntrlMg = getControlMerge(&block);
560  assert(cntrlMg != nullptr);
561  cntrlMg->eraseOperand(0);
562 
563  for (Operation *op : blockMerges[&block])
564  if (op != cntrlMg)
565  op->setOperand(0, cntrlMg->getResult(1));
566  }
567  }
568 
569  removeBlockOperands(f);
570 }

625 static void blockBFS(Block *start, BFSCallback callback) {
626  DenseSet<Block *> visited;
627  SmallVector<Block *> queue = {start};
628  while (!queue.empty()) {
629  Block *currBlock = queue.front();
630  queue.erase(queue.begin());
631  visited.insert(currBlock);
632 
633  switch (callback(currBlock)) {
634  case BFSNextState::Halt:
635  return;
636  case BFSNextState::Continue: {
637  llvm::copy_if(currBlock->getSuccessors(), std::back_inserter(queue),
638  [&](Block *block) { return visited.count(block) == 0; });
639  break;
640  }
641  case BFSNextState::SkipSuccessors:
642  for (auto *succ : currBlock->getSuccessors())
643  visited.insert(succ);
644  continue;
645  }
646  }
647 }

655  isReachable = true;
656  return BFSNextState::Halt;
657  }
658  return BFSNextState::Continue;

674 /// Helper that checks if entry is a loop header. If it is, it collects
675 /// additional information about the loop for further processing.
676 static LogicalResult collectLoopInfo(Block *entry, DominanceInfo &domInfo,
677  LoopInfo &loopInfo) {
678  for (auto *backedge : entry->getPredecessors()) {
679  bool dominates = domInfo.dominates(entry, backedge);
680  bool formsLoop = isReachable(entry, backedge);
681  if (formsLoop) {
682  if (dominates)
683  loopInfo.loopLatches.insert(backedge);
684  else
685  return entry->getParentOp()->emitError()
686  << "Non-canonical loop structures detected; a potential "
687  "loop header has backedges not dominated by the loop "
688  "header. This indicates that the loop has multiple entry "
689  "points. Handshake lowering does not yet support this "
690  "form of control flow, exiting.";
691  }
692  }
693 
694  loopInfo.isValid = !loopInfo.loopLatches.empty();
695  if (loopInfo.isValid) {
696  loopInfo.loopHeader = entry;
697 
698  // Exit blocks are the blocks that control is transfered to after exiting
699  // the loop. This is essentially determining the strongly connected
700  // components with the loop header. We perform a BFS from the loop header,
701  // and if the loop header is reachable from the block, it is within the
702  // loop.
703  blockBFS(entry, [&](Block *currBlock) {
704  if (isReachable(currBlock, entry)) {
705  loopInfo.inLoop.insert(currBlock);
706  return BFSNextState::Continue;
707  }
708  loopInfo.exitBlocks.insert(currBlock);
709  return BFSNextState::SkipSuccessors;
710  });
711 
712  assert(loopInfo.inLoop.size() >= 2 && "A loop must have at least 2 blocks");
713  assert(loopInfo.exitBlocks.size() != 0 &&
714  "A loop must have an exit block...?");
715  }
716 
717  return success();
718 }

774 LogicalResult
775 LoopNetworkRewriter::processRegion(Region &r,
776  ConversionPatternRewriter &rewriter) {
777  this->rewriter = &rewriter;
778 
779  Operation *op = r.getParentOp();
780 
781  // Initialize the BFS queue with the entry block.
782  queue = {&r.front()};
783  domInfo = DominanceInfo(op);
784 
785  while (!queue.empty()) {
786  Block *currBlock = queue.front();
787  queue.pop_front();
788  visited.insert(currBlock);
789 
790  if (!hasMultiplePredecessors(currBlock)) {
791  // Not a loop header;
792  llvm::copy_if(currBlock->getSuccessors(), std::back_inserter(queue),
793  [&](Block *block) { return !visited.count(block); });
794  } else {
795  // Check if currBlock starts a loop
796  LoopInfo loopInfo;
797  if (failed(collectLoopInfo(currBlock, domInfo, loopInfo)))
798  return failure();
799 
800  if (loopInfo.isValid) {
801  if (loopInfo.loopLatches.size() > 1)
802  return emitError(op->getLoc()) << "Multiple loop latches detected "
803  "(backedges from within the loop "
804  "to the loop header). Loop task "
805  "pipelining is only supported for "
806  "loops with unified loop latches.";
807 
808  // This is the start of an outer loop - go process!
809  if (failed(processOuterLoop(op->getLoc(), loopInfo)))
810  return failure();
811  }
812  }
813  }
814 
815  return success();

832 static std::vector<Value> getSortedInputs(ControlMergeOp cmerge, MuxOp mux) {

852 BufferOp LoopNetworkRewriter::buildContinueNetwork(Block *loopHeader,
853  Block *loopLatch,
854  Backedge &loopPrimingInput) {
855  // Gather the muxes to replace before modifying block internals; it's been
856  // found that if this is not done, we have determinism issues wrt. generating
857  // the same order of replaced muxes on repeated runs of an identical
858  // conversion.
859  llvm::SmallVector<MuxOp> muxesToReplace;
860  llvm::copy(loopHeader->getOps<MuxOp>(), std::back_inserter(muxesToReplace));
861 
862  // Fetch the control merge of the block; it is assumed that, at this point of
863  // lowering, no other form of control can be used for the loop header block
864  // than a control merge.
865  auto *cmerge = getControlMerge(loopHeader);
866  assert(hl.getBlockEntryControl(loopHeader) == cmerge->getResult(0) &&
867  "Expected control merge to be the control component of a loop header");
868  auto loc = cmerge->getLoc();
869 
870  // sanity check: cmerge should have >1 input to actually be a loop
871  assert(cmerge->getNumOperands() > 1 && "This cannot be a loop header");
872 
873  // Partition the control merge inputs into those originating from backedges,
874  // and those originating elsewhere.
875  SmallVector<Value> externalCtrls, loopCtrls;
876  for (auto cval : cmerge->getOperands()) {
877  if (cval.getParentBlock() == loopLatch)
878  loopCtrls.push_back(cval);
879  else
880  externalCtrls.push_back(cval);
881  }
882  assert(loopCtrls.size() == 1 &&
883  "Expected a single loop control value to match the single loop latch");
884  Value loopCtrl = loopCtrls.front();
885 
886  // Merge all of the controls in each partition
887  rewriter->setInsertionPointToStart(loopHeader);
888  auto externalCtrlMerge = rewriter->create<ControlMergeOp>(loc, externalCtrls);
889 
890  // Create loop mux and the loop priming register. The loop mux will on select
891  // "0" select external control, and internal control at "1". This convention
892  // must be followed by the loop exit network.
893  auto primingRegister = rewriter->create<BufferOp>(
894  loc, rewriter->getI1Type(), /*size=*/1, loopPrimingInput,
895  /*bufferType=*/BufferTypeEnum::seq);
896  // Initialize the priming register to path 0.
897  primingRegister->setAttr("initValues", rewriter->getI64ArrayAttr({0}));
898 
899  // The loop control mux will deterministically select between control entering
900  // the loop from any external block or the single loop backedge.
901  auto loopCtrlMux = rewriter->create<MuxOp>(
902  loc, primingRegister.getResult(),
903  llvm::SmallVector<Value>{externalCtrlMerge.result(), loopCtrl});
904 
905  // Replace the existing control merge 'result' output with the loop control
906  // mux.
907  cmerge->getResult(0).replaceAllUsesWith(loopCtrlMux.getResult());
908 
909  // Register the new block entry control value
910  hl.setBlockEntryControl(loopHeader, loopCtrlMux.getResult());
911 
912  // Next, we need to consider how to replace the control merge 'index' output,
913  // used to drive input selection to the block.
914 
915  // Inputs to the loop header will be sourced from muxes with inputs from both
916  // the loop latch as well as external blocks. Iterate through these and sort
917  // based on the input ordering to the external/internal control merge.
918  // We do this by maintaining a mapping between the external and loop data
919  // inputs for each data mux in the design. The key of these maps is the
920  // original mux (that is to be replaced).
921  DenseMap<MuxOp, std::vector<Value>> externalDataInputs;
922  DenseMap<MuxOp, Value> loopDataInputs;
923  for (auto muxOp : muxesToReplace) {
924  if (muxOp == loopCtrlMux)
925  continue;
926 
927  externalDataInputs[muxOp] = getSortedInputs(externalCtrlMerge, muxOp);
928  loopDataInputs[muxOp] = getOperandFromBlock(muxOp, loopLatch);
929  assert(/*loop latch input*/ 1 + externalDataInputs[muxOp].size() ==
930  muxOp.dataOperands().size() &&
931  "Expected all mux operands to be partitioned between loop and "
932  "external data inputs");
933  }
934 
935  // With this, we now replace each of the data input muxes in the loop header.
936  // We instantiate a single mux driven by the external control merge.
937  // This, as well as the corresponding data input coming from within the single
938  // loop latch, will then be selected between by a 3rd mux, based on the
939  // priming register.
940  for (MuxOp mux : muxesToReplace) {
941  auto externalDataMux = rewriter->create<MuxOp>(
942  loc, externalCtrlMerge.index(), externalDataInputs[mux]);
943 
944  rewriter->replaceOp(
945  mux, rewriter
946  ->create<MuxOp>(loc, primingRegister,
947  llvm::SmallVector<Value>{externalDataMux,
948  loopDataInputs[mux]})
949  .getResult());
950  }
951 
952  // Now all values defined by the original cmerge should have been replaced,
953  // and it may be erased.
954  rewriter->eraseOp(cmerge);
955 
956  // Return the priming register to be referenced by the exit network builder.
957  return primingRegister;
958 }

960 void LoopNetworkRewriter::buildExitNetwork(
961  Block *loopHeader, const llvm::SmallSet<ExitPair, 2> &exitPairs,
962  BufferOp loopPrimingRegister, Backedge &loopPrimingInput) {
963  auto loc = loopPrimingRegister.getLoc();
964 
965  // Iterate over the exit pairs to gather up the condition signals that need to
966  // be connected to the exit network. In doing so, we parity-correct these
967  // condition values based on the convention used in buildContinueNetwork - The
968  // loop mux will on select "0" select external control, and internal control
969  // at "1". This convention which must be followed by the loop exit network.
970  // External control must be selected when exiting the loop (to reprime the
971  // register).
972  SmallVector<Value> parityCorrectedConds;
973  for (auto &[condBlock, exitBlock] : exitPairs) {
974  auto condBr = getControlCondBranch(condBlock);
975  assert(
976  condBr &&
977  "Expected a conditional control branch op in the loop condition block");
978  Operation *trueUser = *condBr.trueResult().getUsers().begin();
979  bool isTrueParity = trueUser->getBlock() == exitBlock;
980  assert(isTrueParity ^
981  ((*condBr.falseResult().getUsers().begin())->getBlock() ==
982  exitBlock) &&
983  "The user of either the true or the false result should be in the "
984  "exit block");
985 
986  Value condValue = condBr.conditionOperand();
987  if (isTrueParity) {
988  // This goes against the convention, and we have to invert the condition
989  // value before connecting it to the exit network.
990  rewriter->setInsertionPoint(condBr);
991  condValue = rewriter->create<arith::XOrIOp>(
992  loc, condValue.getType(), condValue,
993  rewriter->create<arith::ConstantOp>(
994  loc, rewriter->getIntegerAttr(rewriter->getI1Type(), 1)));
995  }
996  parityCorrectedConds.push_back(condValue);
997  }
998 
999  // Merge all of the parity-corrected exit conditions and assign them
1000  // to the loop priming input.
1001  auto exitMerge = rewriter->create<MergeOp>(loc, parityCorrectedConds);
1002  loopPrimingInput.setValue(exitMerge);
1003 }

1005 LogicalResult LoopNetworkRewriter::processOuterLoop(Location loc,
1006  LoopInfo &loopInfo) {
1007  // We determine the exit pairs of the loop; this is the in-loop nodes
1008  // which branch off to the exit nodes.
1009  llvm::SmallSet<ExitPair, 2> exitPairs;
1010  for (auto *exitNode : loopInfo.exitBlocks) {
1011  for (auto *pred : exitNode->getPredecessors()) {
1012  // is the predecessor inside the loop?
1013  if (!loopInfo.inLoop.contains(pred))
1014  continue;
1015 
1016  ExitPair condPair = {pred, exitNode};
1017  assert(!exitPairs.count(condPair) &&
1018  "identical condition pairs should never be possible");
1019  exitPairs.insert(condPair);
1020  }
1021  }
1022  assert(exitPairs.size() > 0 && "No exits from loop?");
1023 
1024  // The first precondition to our loop transformation is that only a single
1025  // exit pair exists in the loop.
1026  if (exitPairs.size() > 1)
1027  return emitError(loc)
1028  << "Multiple exits detected within a loop. Loop task pipelining is "
1029  "only supported for loops with unified loop exit blocks.";
1030 
1031  BackedgeBuilder bebuilder(*rewriter, loopInfo.loopHeader->front().getLoc());
1032 
1033  // Build the loop continue network. Loop continuation is triggered solely by
1034  // backedges to the header.
1035  auto loopPrimingRegisterInput = bebuilder.get(rewriter->getI1Type());
1036  auto loopPrimingRegister =
1037  buildContinueNetwork(loopInfo.loopHeader, *loopInfo.loopLatches.begin(),
1038  loopPrimingRegisterInput);
1039 
1040  // Build the loop exit network. Loop exiting is driven solely by exit pairs
1041  // from the loop.
1042  buildExitNetwork(loopInfo.loopHeader, exitPairs, loopPrimingRegister,
1043  loopPrimingRegisterInput);
1044 
1045  // Finally, we must reprime the BFS in processFunction; it must skip all
1046  // of the inLoop nodes - since they have now been captured within the loop
1047  // network that we just built- and restart from the exitBlocks - since they
1048  // are guaranteed to be outside of the loop, but more loops may be
1049  // downstream of these blocks.
1050  llvm::copy(loopInfo.inLoop, std::inserter(visited, visited.begin()));
1051  llvm::copy_if(loopInfo.exitBlocks, std::back_inserter(queue),
1052  [&](Block *exitBlock) { return !visited.count(exitBlock); });
1053  return success();
1054 }

1056 // Return the appropriate branch result based on successor block which uses it
1057 static Value getSuccResult(Operation *termOp, Operation *newOp,
1058  Block *succBlock) {

1065  return dyn_cast<handshake::ConditionalBranchOp>(newOp).falseResult();

1090  // Count the number of branches which the liveout needs

1092

1109

1112  u.getOwner()->replaceUsesOfWith(val, res);
1113  break;
1114  }
1115  }
1116  }
1117  }
1118  }
1119  }
1120 
1121  // Remove StandardOp branch terminators and place new terminator

1126  if (!(isa<mlir::cf::CondBranchOp>(termOp) ||

1335  Value memref;

1341  .Case<memref::LoadOp>([&](auto loadOp) {

1344  SmallVector<Value, 8> operands(loadOp.getIndices());

1356  newOp = rewriter.create<handshake::StoreOp>(
1357  op.getLoc(), storeOp.getValueToStore(), operands);
1358  })

1362  MemRefAccess access(&op);

1364  // of an affine map, which is a linear combination of constants
1365  // and parameters. Therefore, we should extract the affine map of
1366  // each address and expand it into proper expressions that
1367  // calculate the result.

1369  if (auto loadOp = dyn_cast<mlir::AffineReadOpInterface>(op))
1370  map = loadOp.getAffineMap();
1371  else

1376  // actual address indices that can be used as operands for
1377  // handshake LoadOp/StoreOp. The following processing requires it
1378  // to be a valid result.

1381  assert(operands && "Address operands of affine memref access "

1386  op.getLoc(), access.memref, *operands);
1387  newOp = loadOp;
1388  op.getResult(0).replaceAllUsesWith(loadOp.dataResult());
1389  } else {
1390  newOp = rewriter.create<handshake::StoreOp>(
1391  op.getLoc(), op.getOperand(0), *operands);
1392  }
1393  })
1394  .Default([&](auto) {
1395  op.emitOpError("Load/store operation cannot be handled.");
1396  });
1397 
1398  memRefOps[memref].push_back(newOp);
1399  opsToErase.push_back(&op);
1400  }

1410  }
1411

1413 }

1601 LogicalResult
1602 HandshakeLowering::connectToMemory(ConversionPatternRewriter &rewriter,
1603  MemRefToMemoryAccessOp memRefOps, bool lsq) {
1604  // Add MemoryOps which represent the memory interface
1605  // Connect memory operations and control appropriately
1606  int mem_count = 0;
1607  for (auto memory : memRefOps) {
1608  // First operand corresponds to memref (alloca or function argument)
1609  Value memrefOperand = memory.first;
1610 
1611  // A memory is external if the memref that defines it is provided as a
1612  // function (block) argument.
1613  bool isExternalMemory = memrefOperand.isa<BlockArgument>();
1614 
1615  mlir::MemRefType memrefType =
1616  memrefOperand.getType().cast<mlir::MemRefType>();
1617  if (memrefType.getNumDynamicDims() != 0 ||
1618  memrefType.getShape().size() != 1)
1619  return emitError(memrefOperand.getLoc())
1620  << "memref's must be both statically sized and "
1621  "unidimensional.";
1622 
1623  std::vector<Value> operands;
1624 
1625  // Get control values which need to connect to memory
1626  std::vector<Value> controlVals = getControlValues(memory.second);
1627 
1628  // In case of LSQ interface, set control values as inputs (used to
1629  // trigger allocation to LSQ)
1630  if (lsq)
1631  operands.insert(operands.end(), controlVals.begin(), controlVals.end());
1632 
1633  // Add load indices and store data+indices to memory operands
1634  // Count number of loads so that we can generate appropriate number of
1635  // memory outputs (data to load ops)
1636 
1637  // memory.second is in program order
1638  // Enforce MemoryOp port ordering as follows:
1639  // Operands: all stores then all loads (stdata1, staddr1, stdata2,...,
1640  // ldaddr1, ldaddr2,....) Outputs: all load outputs, ordered the same as
1641  // load addresses (lddata1, lddata2, ...), followed by all none outputs,
1642  // ordered as operands (stnone1, stnone2,...ldnone1, ldnone2,...)
1643  std::vector<int> newInd(memory.second.size(), 0);
1644  int ind = 0;
1645  for (int i = 0, e = memory.second.size(); i < e; ++i) {
1646  auto *op = memory.second[i];
1647  if (isa<handshake::StoreOp>(op)) {
1648  SmallVector<Value, 8> results = getResultsToMemory(op);
1649  operands.insert(operands.end(), results.begin(), results.end());
1650  newInd[i] = ind++;
1651  }
1652  }
1653 
1654  int ld_count = 0;
1655 
1656  for (int i = 0, e = memory.second.size(); i < e; ++i) {
1657  auto *op = memory.second[i];
1658  if (isa<handshake::LoadOp>(op)) {
1659  SmallVector<Value, 8> results = getResultsToMemory(op);
1660  operands.insert(operands.end(), results.begin(), results.end());
1661 
1662  ld_count++;
1663  newInd[i] = ind++;
1664  }
1665  }
1666 
1667  // control-only outputs for each access (indicate access completion)
1668  int cntrl_count = lsq ? 0 : memory.second.size();
1669 
1670  Block *entryBlock = &r.front();
1671  rewriter.setInsertionPointToStart(entryBlock);
1672 
1673  // Place memory op next to the alloc op
1674  Operation *newOp = nullptr;
1675  if (isExternalMemory)
1676  newOp = rewriter.create<ExternalMemoryOp>(
1677  entryBlock->front().getLoc(), memrefOperand, operands, ld_count,
1678  cntrl_count - ld_count, mem_count++);
1679  else
1680  newOp = rewriter.create<MemoryOp>(entryBlock->front().getLoc(), operands,
1681  ld_count, cntrl_count, lsq, mem_count++,
1682  memrefOperand);
1683 
1684  setLoadDataInputs(memory.second, newOp);
1685 
1686  if (!lsq) {
1687  // Create Joins which join done signals from memory with the
1688  // control-only network
1689  addJoinOps(rewriter, controlVals);
1690 
1691  // Connect all load/store done signals to the join of their block
1692  // Ensure that the block terminates only after all its accesses have
1693  // completed
1694  // True is default. When no sync needed, set to false (for now,
1695  // user-determined)
1696  bool control = true;
1697 
1698  if (control) {
1699  if (setJoinControlInputs(memory.second, newOp, ld_count, newInd)
1700  .failed())
1701  return failure();
1702 
1703  } else {
1704  for (int i = 0, e = cntrl_count; i < e; ++i) {
1705  rewriter.setInsertionPointAfter(newOp);
1706  rewriter.create<SinkOp>(newOp->getLoc(),
1707  newOp->getResult(ld_count + i));
1708  }
1709  }
1710 
1711  // Set control-only inputs to each memory op
1712  // Ensure that op starts only after prior blocks have completed
1713  // Ensure that op starts only after predecessor ops (with RAW, WAR, or
1714  // WAW) have completed
1715  setMemOpControlInputs(rewriter, memory.second, newOp, ld_count, newInd);
1716  }
1717  }
1718 
1719  if (lsq)
1720  addLazyForks(r, rewriter);
1721  else
1722  addMemOpForks(r, rewriter);
1723 
1724  removeAllocOps(r, rewriter);
1725 
1726  // Loads and stores have some sinks which are no longer needed now that
1727  // they connect to MemoryOp
1728  removeRedundantSinks(r, rewriter);
1729  return success();
1730 }
