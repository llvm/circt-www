62  template <typename TTerm>
63  LogicalResult setControlOnlyPath(ConversionPatternRewriter &rewriter) {
64  // Creates start and end points of the control-only path
65 
66  // Temporary start node (removed in later steps) in entry block
67  Block *entryBlock = &r.front();
68  rewriter.setInsertionPointToStart(entryBlock);
69  Operation *startOp = rewriter.create<StartOp>(entryBlock->front().getLoc());
70  setBlockEntryControl(entryBlock, startOp->getResult(0));
71 
72  // Replace original return ops with new returns with additional control
73  // input
74  for (auto retOp : llvm::make_early_inc_range(r.getOps<TTerm>())) {
75  rewriter.setInsertionPoint(retOp);
76  SmallVector<Value, 8> operands(retOp->getOperands());
77  operands.push_back(startOp->getResult(0));
78  rewriter.replaceOpWithNewOp<handshake::ReturnOp>(retOp, operands);
79  }
80  return success();
81  }

126 // Driver for the HandshakeLowering class.
127 // Note: using two different vararg template names due to potantial references
128 // that would cause a type mismatch
129 template <typename T, typename... TArgs, typename... TArgs2>
130 LogicalResult runPartialLowering(
131  T &instance,
132  LogicalResult (T::*memberFunc)(ConversionPatternRewriter &, TArgs2...),
133  TArgs &...args) {
134  return partiallyLowerRegion(
135  [&](Region &, ConversionPatternRewriter &rewriter) -> LogicalResult {

144 // Driver that applies the partial lowerings expressed in HandshakeLowering to
145 // the region encapsulated in it. The region is assumed to have a terminator of
146 // type TTerm. See HandshakeLowering for the different lowering steps.
147 template <typename TTerm>
148 LogicalResult lowerRegion(HandshakeLowering &hl, bool sourceConstants,
149  bool disableTaskPipelining) {
150  // Perform initial dataflow conversion. This process allows for the use of
151  // non-deterministic merge-like operations.
152  HandshakeLowering::MemRefToMemoryAccessOp memOps;
153 
154  if (failed(
155  runPartialLowering(hl, &HandshakeLowering::replaceMemoryOps, memOps)))
156  return failure();
157  if (failed(runPartialLowering(hl,
158  &HandshakeLowering::setControlOnlyPath<TTerm>)))
159  return failure();
160  if (failed(runPartialLowering(hl, &HandshakeLowering::addMergeOps)))
161  return failure();
162  if (failed(runPartialLowering(hl, &HandshakeLowering::replaceCallOps)))
163  return failure();
164  if (failed(runPartialLowering(hl, &HandshakeLowering::addBranchOps)))
165  return failure();
166 
167  // The following passes modifies the dataflow graph to being safe for task
168  // pipelining. In doing so, non-deterministic merge structures are replaced
169  // for deterministic structures.
170  if (!disableTaskPipelining) {
171  if (failed(
172  runPartialLowering(hl, &HandshakeLowering::loopNetworkRewriting)))
173  return failure();
174  }
175 
176  // Fork/sink materialization. @todo: this should be removed and
177  // materialization should be run as a separate pass afterward initial dataflow
178  // conversion! However, connectToMemory has some hard-coded assumptions on the
179  // existence of fork/sink operations...
180  if (failed(partiallyLowerRegion(addSinkOps, hl.getContext(), hl.getRegion())))
181  return failure();
182 
183  if (failed(runPartialLowering(
184  hl, &HandshakeLowering::connectConstantsToControl, sourceConstants)))
185  return failure();
186 
187  if (failed(partiallyLowerRegion(addForkOps, hl.getContext(), hl.getRegion())))
188  return failure();
189  if (failed(checkDataflowConversion(hl.getRegion(), disableTaskPipelining)))
190  return failure();
191 
192  bool lsq = false;
193  if (failed(runPartialLowering(hl, &HandshakeLowering::connectToMemory, memOps,
194  lsq)))
195  return failure();
196 
197  // Add control argument to entry block, replace references to the
198  // temporary handshake::StartOp operation, and finally remove the start
199  // op.
200  if (failed(runPartialLowering(hl, &HandshakeLowering::finalize)))
201  return failure();
202  return success();
203 }
