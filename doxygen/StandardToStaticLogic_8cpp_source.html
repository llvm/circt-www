24 valueVector getPipelineArgs(Block &block) {
25  valueVector arguments;
26  for (auto &op : block) {
27  if (!op.mightHaveTrait<OpTrait::IsTerminator>()) {
28  for (auto operand : op.getOperands()) {
29  if (operand.isa<BlockArgument>()) {
30  // Add only unique uses
31  if (std::find(arguments.begin(), arguments.end(), operand) ==
32  arguments.end())
33  arguments.push_back(operand);
34  } else if (operand.getDefiningOp()->getBlock() != &block) {
35  // Add only unique uses
36  if (std::find(arguments.begin(), arguments.end(), operand) ==
37  arguments.end())
38  arguments.push_back(operand);
39  }
40  }
41  }
42  }
43  return arguments;
44 }

46 valueVector getPipelineResults(Block &block) {
47  SmallVector<Value, 4> results;
48  for (auto &op : block) {
49  for (auto result : op.getResults()) {
50  bool isResult = false;
51  for (auto user : result.getUsers()) {
52  if (user->getBlock() != &block ||
53  user->hasTrait<OpTrait::IsTerminator>()) {
54  isResult = true;
55  break;
56  }
57  }
58  if (isResult)
59  results.push_back(result);
60  }
61  }
62  return results;
63 }

65 static void createPipeline(mlir::func::FuncOp f, OpBuilder &builder) {
66  for (Block &block : f) {
67  if (!block.front().mightHaveTrait<OpTrait::IsTerminator>()) {
68 
69  auto arguments = getPipelineArgs(block);
70  auto results = getPipelineResults(block);
71  builder.setInsertionPoint(&block.back());
72  builder.create<staticlogic::ReturnOp>(f.getLoc(), ValueRange(results));
73 
74  // Create pipeline operation, and move all operations except terminator
75  // into the pipeline.
76  builder.setInsertionPoint(&block.front());
77  auto pipeline = builder.create<staticlogic::PipelineOp>(
78  f.getLoc(), ValueRange(arguments), ValueRange(results));
79 
80  auto &body = pipeline.getRegion().front();
81  body.getOperations().splice(body.getOperations().begin(),
82  block.getOperations(), ++block.begin(),
83  --block.end());
84 
85  // Reconnect arguments of the pipeline operation.
86  unsigned argIdx = 0;
87  for (auto value : arguments) {
88  value.replaceUsesWithIf(
89  body.getArgument(argIdx),
90  function_ref<bool(OpOperand &)>([&body](OpOperand &use) -> bool {
91  return use.getOwner()->getBlock() == &body;
92  }));
93  argIdx += 1;
94  }
95 
96  // Reconnect results of the pipeline operation.
97  unsigned resultIdx = 0;
98  for (auto value : results) {
99  value.replaceUsesWithIf(
100  pipeline.getResult(resultIdx),
101  function_ref<bool(OpOperand &)>([&body](OpOperand &use) -> bool {
102  return use.getOwner()->getBlock() != &body;
103  }));
104  resultIdx += 1;
105  }
106  }
107  }
108 }
