67 void Slot::insertChange(int index, int bitOffset, uint8_t *bytes,
68  unsigned width) {
69  // Get the amount of 64 bit words required to store the value in an APInt.
70  auto size = llvm::divideCeil(width, 64);
71 
72  if (changesSize >= buffers.size()) {
73  // Create a new change buffer if we don't have any unused one available for
74  // reuse.
75  buffers.push_back(std::make_pair(
76  bitOffset,
77  APInt(width, makeArrayRef(reinterpret_cast<uint64_t *>(bytes), size))));
78  } else {
79  // Reuse the first available buffer.
80  buffers[changesSize] = std::make_pair(
81  bitOffset,
82  APInt(width, makeArrayRef(reinterpret_cast<uint64_t *>(bytes), size)));
83  }
84 
85  // Map the signal index to the change buffer so we can retrieve
86  // it after sorting.
87  changes.push_back(std::make_pair(index, changesSize));
88  ++changesSize;
89 }

107 Slot &UpdateQueue::getOrCreateSlot(Time time) {
108  auto &top = begin()[topSlot];
109 
110  // Directly add to top slot.
111  if (!top.unused && time == top.time) {
112  return top;
113  }
114 
115  // We need to search through the queue for an existing slot only if we're
116  // spawning an event later than the top slot. Adding to an existing slot
117  // scheduled earlier than the top slot should never happens, as then it should
118  // be the top.
119  if (events > 0 && top.time < time) {
120  for (size_t i = 0, e = size(); i < e; ++i) {
121  if (time == begin()[i].time) {
122  return begin()[i];
123  }
124  }
125  }
126 
127  // Spawn new event using an existing slot.
128  if (!unused.empty()) {
129  auto firstUnused = unused.pop_back_val();
130  auto &newSlot = begin()[firstUnused];
131  newSlot.unused = false;
132  newSlot.time = time;
133 
134  // Update the top of the queue either if it is currently unused or the new
135  // timestamp is earlier than it.
136  if (top.unused || time < top.time)
137  topSlot = firstUnused;
138 
139  ++events;
140  return newSlot;
141  }
142 
143  // We do not have pre-allocated slots available, generate a new one.
144  push_back(Slot(time));
145 
146  // Update the top of the queue either if it is currently unused or the new
147  // timestamp is earlier than it.
148  if (top.unused || time < top.time)
149  topSlot = size() - 1;
150 
151  ++events;
152  return back();
153 }

155 const Slot &UpdateQueue::top() {
156  assert(topSlot < size() && "top is pointing out of bounds!");
157 
158  // Sort the changes of the top slot such that all changes to the same signal
159  // are in succession.
160  auto &top = begin()[topSlot];
161  llvm::sort(top.changes.begin(), top.changes.begin() + top.changesSize);
162  return top;
163 }

165 void UpdateQueue::pop() {
166  // Reset internal structures and decrease the event counter.
167  auto &curr = begin()[topSlot];
168  curr.unused = true;
169  curr.changesSize = 0;
170  curr.scheduled.clear();
171  curr.changes.clear();
172  curr.time = Time();
173  --events;
174 
175  // Add to unused slots list for easy retrieval.
176  unused.push_back(topSlot);
177 
178  // Update the current top of the queue.
179  topSlot = std::distance(
180  begin(),
181  std::min_element(begin(), end(), [](const auto &a, const auto &b) {
182  // a is "smaller" than b if either a's timestamp is earlier than b's, or
183  // b is unused (i.e. b has no actual meaning).
184  return !a.unused && (a < b || b.unused);
185  }));
186 }

237 int State::addSignalData(int index, std::string owner, uint8_t *value,
238  uint64_t size) {
239  auto it = getInstanceIterator(owner);
240 
241  uint64_t globalIdx = (*it).sensitivityList[index + (*it).nArgs].globalIndex;
242  auto &sig = signals[globalIdx];
243 
244  // Add pointer and size to global signal table entry.
245  sig.store(value, size);
246 
247  // Add the value pointer to the signal detail struct for each instance this
248  // signal appears in.
249  for (auto inst : signals[globalIdx].getTriggeredInstanceIndices()) {
250  for (auto &detail : instances[inst].sensitivityList) {
251  if (detail.globalIndex == globalIdx) {
252  detail.value = sig.getValue();
253  }
254  }
255  }
256  return globalIdx;
