110 void Slot::insertChange(int index, int bitOffset, uint8_t *bytes,
111  unsigned width) {
112  // Get the amount of 64 bit words required to store the value in an APInt.
113  auto size = llvm::divideCeil(width, 64);
114 
115  if (changesSize >= buffers.size()) {
116  // Create a new change buffer if we don't have any unused one available for
117  // reuse.
118  buffers.push_back(std::make_pair(
119  bitOffset,
120  APInt(width, makeArrayRef(reinterpret_cast<uint64_t *>(bytes), size))));
121  } else {
122  // Reuse the first available buffer.
123  buffers[changesSize] = std::make_pair(
124  bitOffset,
125  APInt(width, makeArrayRef(reinterpret_cast<uint64_t *>(bytes), size)));
126  }
127 
128  // Map the signal index to the change buffer so we can retrieve
129  // it after sorting.
130  changes.push_back(std::make_pair(index, changesSize));
131  ++changesSize;
132 }

150 Slot &UpdateQueue::getOrCreateSlot(Time time) {
151  auto &top = begin()[topSlot];
152 
153  // Directly add to top slot.
154  if (!top.unused && time == top.time) {
155  return top;
156  }
157 
158  // We need to search through the queue for an existing slot only if we're
159  // spawning an event later than the top slot. Adding to an existing slot
160  // scheduled earlier than the top slot should never happens, as then it should
161  // be the top.
162  if (events > 0 && top.time < time) {
163  for (size_t i = 0, e = size(); i < e; ++i) {
164  if (time == begin()[i].time) {
165  return begin()[i];
166  }
167  }
168  }
169 
170  // Spawn new event using an existing slot.
171  if (!unused.empty()) {
172  auto firstUnused = unused.pop_back_val();
173  auto &newSlot = begin()[firstUnused];
174  newSlot.unused = false;
175  newSlot.time = time;
176 
177  // Update the top of the queue either if it is currently unused or the new
178  // timestamp is earlier than it.
179  if (top.unused || time < top.time)
180  topSlot = firstUnused;
181 
182  ++events;
183  return newSlot;
184  }
185 
186  // We do not have pre-allocated slots available, generate a new one.
187  push_back(Slot(time));
188 
189  // Update the top of the queue either if it is currently unused or the new
190  // timestamp is earlier than it.
191  if (top.unused || time < top.time)
192  topSlot = size() - 1;
193 
194  ++events;
195  return back();
196 }

198 const Slot &UpdateQueue::top() {
199  assert(topSlot < size() && "top is pointing out of bounds!");
200 
201  // Sort the changes of the top slot such that all changes to the same signal
202  // are in succession.
203  auto &top = begin()[topSlot];
204  llvm::sort(top.changes.begin(), top.changes.begin() + top.changesSize);
205  return top;
206 }

208 void UpdateQueue::pop() {
209  // Reset internal structures and decrease the event counter.
210  auto &curr = begin()[topSlot];
211  curr.unused = true;
212  curr.changesSize = 0;
213  curr.scheduled.clear();
214  curr.changes.clear();
215  curr.time = Time();
216  --events;
217 
218  // Add to unused slots list for easy retrieval.
219  unused.push_back(topSlot);
220 
221  // Update the current top of the queue.
222  topSlot = std::distance(
223  begin(),
224  std::min_element(begin(), end(), [](const auto &a, const auto &b) {
225  // a is "smaller" than b if either a's timestamp is earlier than b's, or
226  // b is unused (i.e. b has no actual meaning).
227  return !a.unused && (a < b || b.unused);
228  }));
229 }

280 int State::addSignalData(int index, std::string owner, uint8_t *value,
281  uint64_t size) {
282  auto it = getInstanceIterator(owner);
283 
284  uint64_t globalIdx = (*it).sensitivityList[index + (*it).nArgs].globalIndex;
285  auto &sig = signals[globalIdx];
286 
287  // Add pointer and size to global signal table entry.
288  sig.value = std::unique_ptr<uint8_t>(value);
289  sig.size = size;
290 
291  // Add the value pointer to the signal detail struct for each instance this
292  // signal appears in.
293  for (auto inst : signals[globalIdx].triggers) {
294  for (auto &detail : instances[inst].sensitivityList) {
295  if (detail.globalIndex == globalIdx) {
296  detail.value = sig.value.get();
297  }
298  }
299  }
300  return globalIdx;
301 }
