29 /// The simulator's internal representation of time.
30 class Time {
31 public:
32  /// Empty (zero) time constructor. All the time values are defaulted to 0.
33  Time() = default;
34 
35  /// Construct with given time values.
36  Time(uint64_t time, uint64_t delta, uint64_t eps)
37  : time(time), delta(delta), eps(eps) {}
38 
39  /// Compare the time values in order of time, delta, eps.
40  bool operator<(const Time &rhs) const {
41  return time < rhs.time || (time == rhs.time && delta < rhs.delta) ||
42  (time == rhs.time && delta == rhs.delta && eps < rhs.eps);
43  }
44 
45  /// Return true if all the time values are equal.
46  bool operator==(const Time &rhs) const {
47  return time == rhs.time && delta == rhs.delta && eps == rhs.eps;
48  }
49 
50  /// Add two time values.
51  Time operator+(const Time &rhs) const {
52  return Time(time + rhs.time, delta + rhs.delta, eps + rhs.eps);
53  }
54 
55  /// Get the stored time in a printable format.
56  std::string toString() const;
57 
58  uint64_t getTime() const { return time; }
59 
60 private:
61  /// Simulation real time.
62  uint64_t time;
63  uint64_t delta;
64  uint64_t eps;
65 };

75 /// The simulator's internal representation of a signal.
76 class Signal {
77 public:
78  /// Construct an "empty" signal.
79  Signal(std::string name, std::string owner)
80  : name(name), owner(owner), size(0), value(nullptr) {}
81 
82  /// Construct a signal with the given name, owner and initial value.
83  Signal(std::string name, std::string owner, uint8_t *value, uint64_t size)
84  : name(name), owner(owner), size(size), value(value) {}
85 
86  /// Default move constructor.
87  Signal(Signal &&) = default;
88 
89  /// Default signal destructor.
90  ~Signal() = default;
91 
92  /// Returns true if the signals match in name, owner, size and value.
93  bool operator==(const Signal &rhs) const {
94  if (owner != rhs.owner || name != rhs.name || size != rhs.size)
95  return false;
96  return std::memcmp(value, rhs.value, size);
97  }
98 
99  /// Returns true if the owner name is lexically smaller than rhs's owner, or
100  /// the name is lexically smaller than rhs's name, in case they share the same
101  /// owner.
102  bool operator<(const Signal &rhs) const {
103  return (owner < rhs.owner || (owner == rhs.owner && name < rhs.name));
104  }
105 
106  bool isOwner(const std::string &rhs) const { return owner == rhs; };
107 
108  std::string getOwner() const { return owner; }
109 
110  bool isValidSigName() const {
111  return std::regex_match(name, std::regex("(sig)?[0-9]*"));
112  }
113 
114  std::string getName() const { return name; }
115 
116  uint64_t getSize() const { return size; }
117 
118  uint8_t *getValue() const { return value; }
119 
120  const std::vector<unsigned> &getTriggeredInstanceIndices() const {
121  return instanceIndices;
122  }
123 
124  void pushInstanceIndex(unsigned i) { instanceIndices.push_back(i); }
125 
126  bool hasElement() const { return elements.size() > 0; }
127 
128  size_t getElementSize() const { return elements.size(); }
129 
130  void pushElement(std::pair<unsigned, unsigned> val) {
131  elements.push_back(val);
132  }
133 
134  /// Store JIT allocated signal pointer and size.
135  void store(uint8_t *v, uint64_t s) {
136  value = v;
137  size = s;
138  }
139 
140  /// Update signal value when it is changed, the width of incoming signal
141  /// value and the stored signal value are identical.
142  /// As majority signals are smaller than 64 bits, this implementation
143  /// is much faster as it avoided memcpy in most cases.
144  /// @param v Pointer to signal value
145  /// @return return true when signal is updated, false when not
146  bool updateWhenChanged(const uint64_t *v) {
147  switch (size) {
148  case 1: {
149  const uint8_t *newVal = reinterpret_cast<const uint8_t *>(v);
150  if (*value == *newVal)
151  return false;
152  *value = *newVal;
153  break;
154  }
155  case 2: {
156  const uint16_t *newVal = reinterpret_cast<const uint16_t *>(v);
157  if (*(uint16_t *)value == *newVal)
158  return false;
159  *(uint16_t *)value = *newVal;
160  break;
161  }
162  case 4: {
163  const uint32_t *newVal = reinterpret_cast<const uint32_t *>(v);
164  if (*(uint32_t *)value == *newVal)
165  return false;
166  *(uint32_t *)value = *newVal;
167  break;
168  }
169  case 8: {
170  if (*(uint64_t *)value == *v)
171  return false;
172  *(uint64_t *)value = *v;
173  break;
174  }
175  default: {
176  if (std::memcmp(value, v, size) == 0)
177  return false;
178  std::memcpy(value, v, size);
179  break;
180  }
181  }
182 
183  return true;
184  }
185 
186  /// Return the value of the signal in hexadecimal string format.
187  std::string toHexString() const;
188 
189  /// Return the value of the i-th element of the signal in hexadecimal string
190  /// format.
191  std::string toHexString(unsigned) const;
192 
193 private:
194  std::string name;
195  std::string owner;
196  // The list of instances this signal triggers.
197  std::vector<unsigned> instanceIndices;
198  uint64_t size;
199  uint8_t *value;
200  std::vector<std::pair<unsigned, unsigned>> elements;
201 };
