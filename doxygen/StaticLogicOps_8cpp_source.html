29 ParseResult PipelineWhileOp::parse(OpAsmParser &parser,
30  OperationState &result) {
31  // Parse initiation interval.
32  IntegerAttr ii;
33  if (parser.parseKeyword("II") || parser.parseEqual() ||
34  parser.parseAttribute(ii))
35  return failure();
36  result.addAttribute("II", ii);
37 
38  // Parse optional trip count.
39  if (succeeded(parser.parseOptionalKeyword("trip_count"))) {
40  IntegerAttr tripCount;
41  if (parser.parseEqual() || parser.parseAttribute(tripCount))
42  return failure();
43  result.addAttribute("tripCount", tripCount);
44  }
45 
46  // Parse iter_args assignment list.
47  SmallVector<OpAsmParser::Argument> regionArgs;
48  SmallVector<OpAsmParser::UnresolvedOperand> operands;
49  if (succeeded(parser.parseOptionalKeyword("iter_args"))) {
50  if (parser.parseAssignmentList(regionArgs, operands))
51  return failure();
52  }
53 
54  // Parse function type from iter_args to results.
55  FunctionType type;
56  if (parser.parseColon() || parser.parseType(type))
57  return failure();
58 
59  // Function result type is the pipeline result type.
60  result.addTypes(type.getResults());
61 
62  // Resolve iter_args operands.
63  for (auto [regionArg, operand, type] :
64  llvm::zip(regionArgs, operands, type.getInputs())) {
65  regionArg.type = type;
66  if (parser.resolveOperand(operand, type, result.operands))
67  return failure();
68  }
69 
70  // Parse condition region.
71  Region *condition = result.addRegion();
72  if (parser.parseRegion(*condition, regionArgs))
73  return failure();
74 
75  // Parse stages region.
76  if (parser.parseKeyword("do"))
77  return failure();
78  Region *stages = result.addRegion();
79  if (parser.parseRegion(*stages, regionArgs))
80  return failure();
81 
82  return success();
83 }

86  // Print the initiation interval.
87  p << " II = " << ' ' << II();

98  p << ") : ";

102  getResultTypes());
103  p.printType(type);
104 
105  // Print condition region.
106  p << ' ';
107  p.printRegion(condition(), /*printEntryBlockArgs=*/false);
108  p << " do";
109 
110  // Print stages region.
111  p << ' ';
112  p.printRegion(stages(), /*printEntryBlockArgs=*/false);

115 LogicalResult PipelineWhileOp::verify() {
116  // Verify trip count is not negative.
117  if (tripCount() && *tripCount() < 0)
118  return emitOpError("trip count must not be negative, found ")
119  << *tripCount();
120 
121  // Verify the condition block is "combinational" based on an allowlist of
122  // Arithmetic ops.
123  Block &conditionBlock = condition().front();
124  Operation *nonCombinational;
125  WalkResult conditionWalk = conditionBlock.walk([&](Operation *op) {
126  if (isa<StaticLogicDialect>(op->getDialect()))
127  return WalkResult::advance();
128 
129  if (!isa<arith::AddIOp, arith::AndIOp, arith::BitcastOp, arith::CmpIOp,
130  arith::ConstantOp, arith::IndexCastOp, arith::MulIOp, arith::OrIOp,
131  arith::SelectOp, arith::ShLIOp, arith::ExtSIOp, arith::CeilDivSIOp,
132  arith::DivSIOp, arith::FloorDivSIOp, arith::RemSIOp,
133  arith::ShRSIOp, arith::SubIOp, arith::TruncIOp, arith::DivUIOp,
134  arith::RemUIOp, arith::ShRUIOp, arith::XOrIOp, arith::ExtUIOp>(
135  op)) {
136  nonCombinational = op;
137  return WalkResult::interrupt();
138  }
139 
140  return WalkResult::advance();
141  });
142 
143  if (conditionWalk.wasInterrupted())
144  return emitOpError("condition must have a combinational body, found ")
145  << *nonCombinational;
146 
147  // Verify the condition block terminates with a value of type i1.
148  TypeRange conditionResults =
149  conditionBlock.getTerminator()->getOperandTypes();
150  if (conditionResults.size() != 1)
151  return emitOpError("condition must terminate with a single result, found ")
152  << conditionResults;
153 
154  if (conditionResults.front() != IntegerType::get(getContext(), 1))
155  return emitOpError("condition must terminate with an i1 result, found ")
156  << conditionResults.front();
157 
158  // Verify the stages block contains at least one stage and a terminator.
159  Block &stagesBlock = stages().front();
160  if (stagesBlock.getOperations().size() < 2)
161  return emitOpError("stages must contain at least one stage");
162 
163  int64_t lastStartTime = -1;
164  for (Operation &inner : stagesBlock) {
165  // Verify the stages block contains only `staticlogic.pipeline.stage` and
166  // `staticlogic.pipeline.terminator` ops.
167  if (!isa<PipelineStageOp, PipelineTerminatorOp>(inner))
168  return emitOpError(
169  "stages may only contain 'staticlogic.pipeline.stage' or "
170  "'staticlogic.pipeline.terminator' ops, found ")
171  << inner;
172 
173  // Verify the stage start times are monotonically increasing.
174  if (auto stage = dyn_cast<PipelineStageOp>(inner)) {
175  if (lastStartTime == -1) {
176  lastStartTime = stage.start();
177  continue;
178  }
179 
180  if (lastStartTime >= stage.start())
181  return stage.emitOpError("'start' must be after previous 'start' (")
182  << lastStartTime << ')';
183 
184  lastStartTime = stage.start();
185  }
186  }
187 
188  return success();
189 }

191 void PipelineWhileOp::build(OpBuilder &builder, OperationState &state,
192  TypeRange resultTypes, IntegerAttr ii,
193  Optional<IntegerAttr> tripCount,
194  ValueRange iterArgs) {
195  OpBuilder::InsertionGuard g(builder);
196 
197  state.addTypes(resultTypes);
198  state.addAttribute("II", ii);
199  if (tripCount)
200  state.addAttribute("tripCount", *tripCount);
201  state.addOperands(iterArgs);
202 
203  Region *condRegion = state.addRegion();
204  Block &condBlock = condRegion->emplaceBlock();
205 
206  SmallVector<Location, 4> argLocs;
207  for (auto arg : iterArgs)
208  argLocs.push_back(arg.getLoc());
209  condBlock.addArguments(iterArgs.getTypes(), argLocs);
210  builder.setInsertionPointToEnd(&condBlock);
211  builder.create<PipelineRegisterOp>(builder.getUnknownLoc(), ValueRange());
212 
213  Region *stagesRegion = state.addRegion();
214  Block &stagesBlock = stagesRegion->emplaceBlock();
215  stagesBlock.addArguments(iterArgs.getTypes(), argLocs);
216  builder.setInsertionPointToEnd(&stagesBlock);
217  builder.create<PipelineTerminatorOp>(builder.getUnknownLoc(), ValueRange(),
218  ValueRange());
219 }

249 LogicalResult PipelineRegisterOp::verify() {
250  PipelineStageOp stage = (*this)->getParentOfType<PipelineStageOp>();
251 
252  // If this doesn't terminate a stage, it is terminating the condition.
253  if (stage == nullptr)
254  return success();
255 
256  // Verify stage terminates with the same types as the result types.
257  TypeRange registerTypes = getOperandTypes();
258  TypeRange resultTypes = stage.getResultTypes();
259  if (registerTypes != resultTypes)
260  return emitOpError("operand types (")
261  << registerTypes << ") must match result types (" << resultTypes
262  << ")";
263 
264  return success();
265 }

267 unsigned PipelineStageOp::getStageNumber() {
268  unsigned number = 0;
269  auto *op = getOperation();
270  auto parent = op->getParentOfType<PipelineWhileOp>();
271  Operation *stage = &parent.getStagesBlock().front();
272  while (stage != op && stage->getNextNode()) {
273  ++number;
274  stage = stage->getNextNode();
275  }
276  return number;
277 }

283 LogicalResult PipelineTerminatorOp::verify() {
284  PipelineWhileOp pipeline = (*this)->getParentOfType<PipelineWhileOp>();
285 
286  // Verify pipeline terminates with the same `iter_args` types as the pipeline.
287  auto iterArgs = iter_args();
288  TypeRange terminatorArgTypes = iterArgs.getTypes();
289  TypeRange pipelineArgTypes = pipeline.iterArgs().getTypes();
290  if (terminatorArgTypes != pipelineArgTypes)
291  return emitOpError("'iter_args' types (")
292  << terminatorArgTypes << ") must match pipeline 'iter_args' types ("
293  << pipelineArgTypes << ")";
294 
295  // Verify `iter_args` are defined by a pipeline stage.
296  for (auto iterArg : iterArgs)
297  if (iterArg.getDefiningOp<PipelineStageOp>() == nullptr)
298  return emitOpError(
299  "'iter_args' must be defined by a 'staticlogic.pipeline.stage'");
300 
301  // Verify pipeline terminates with the same result types as the pipeline.
302  auto opResults = results();
303  TypeRange terminatorResultTypes = opResults.getTypes();
304  TypeRange pipelineResultTypes = pipeline.getResultTypes();
305  if (terminatorResultTypes != pipelineResultTypes)
306  return emitOpError("'results' types (")
307  << terminatorResultTypes << ") must match pipeline result types ("
308  << pipelineResultTypes << ")";
309 
310  // Verify `results` are defined by a pipeline stage.
311  for (auto result : opResults)
312  if (result.getDefiningOp<PipelineStageOp>() == nullptr)
313  return emitOpError(
314  "'results' must be defined by a 'staticlogic.pipeline.stage'");
315 
316  return success();
317 }
