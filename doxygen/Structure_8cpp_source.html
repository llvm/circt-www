19 static moore::ProcedureKind
20 convertProcedureKind(slang::ast::ProceduralBlockKind kind) {
21  switch (kind) {
22  case slang::ast::ProceduralBlockKind::Always:
23  return moore::ProcedureKind::Always;
24  case slang::ast::ProceduralBlockKind::AlwaysComb:
25  return moore::ProcedureKind::AlwaysComb;
26  case slang::ast::ProceduralBlockKind::AlwaysLatch:
27  return moore::ProcedureKind::AlwaysLatch;
28  case slang::ast::ProceduralBlockKind::AlwaysFF:
29  return moore::ProcedureKind::AlwaysFF;
30  case slang::ast::ProceduralBlockKind::Initial:
31  return moore::ProcedureKind::Initial;
32  case slang::ast::ProceduralBlockKind::Final:
33  return moore::ProcedureKind::Final;
34  }
35  llvm_unreachable("all procedure kinds handled");
36 }

39 struct MemberVisitor {
40  Context &context;
41  Location loc;
42  OpBuilder &builder;
43 
44  MemberVisitor(Context &context, Location loc)
45  : context(context), loc(loc), builder(context.builder) {}
46 
47  // Skip empty members (stray semicolons).
48  LogicalResult visit(const slang::ast::EmptyMemberSymbol &) {
49  return success();
50  }
51 
52  // Skip members that are implicitly imported from some other scope for the
53  // sake of name resolution, such as enum variant names.
54  LogicalResult visit(const slang::ast::TransparentMemberSymbol &) {
55  return success();
56  }
57 
58  // Skip typedefs.
59  LogicalResult visit(const slang::ast::TypeAliasType &) { return success(); }
60 
61  // Handle instances.
62  LogicalResult visit(const slang::ast::InstanceSymbol &instNode) {
63  auto targetModule = context.convertModuleHeader(&instNode.body);
64  if (!targetModule)
65  return failure();
66 
67  builder.create<moore::InstanceOp>(
68  loc, builder.getStringAttr(instNode.name),
69  FlatSymbolRefAttr::get(targetModule.getSymNameAttr()));
70 
71  return success();
72  }
73 
74  // Handle variables.
75  LogicalResult visit(const slang::ast::VariableSymbol &varNode) {
76  auto type = context.convertType(*varNode.getDeclaredType());
77  if (!type)
78  return failure();
79 
80  Value initial;
81  if (const auto *init = varNode.getInitializer()) {
82  initial = context.convertExpression(*init);
83  if (!initial)
84  return failure();
85  }
86 
87  auto varOp = builder.create<moore::VariableOp>(
88  loc, type, builder.getStringAttr(varNode.name), initial);
89  context.valueSymbols.insert(&varNode, varOp);
90  return success();
91  }
92 
93  // Handle continuous assignments.
94  LogicalResult visit(const slang::ast::ContinuousAssignSymbol &assignNode) {
95  if (const auto *delay = assignNode.getDelay()) {
96  auto loc = context.convertLocation(delay->sourceRange);
97  return mlir::emitError(loc,
98  "delayed continuous assignments not supported");
99  }
100 
101  const auto &expr =
102  assignNode.getAssignment().as<slang::ast::AssignmentExpression>();
103 
104  auto lhs = context.convertExpression(expr.left());
105  auto rhs = context.convertExpression(expr.right());
106  if (!lhs || !rhs)
107  return failure();
108 
109  if (lhs.getType() != rhs.getType())
110  rhs = builder.create<moore::ConversionOp>(loc, lhs.getType(), rhs);
111 
112  builder.create<moore::ContinuousAssignOp>(loc, lhs, rhs);
113  return success();
114  }
115 
116  // Handle procedures.
117  LogicalResult visit(const slang::ast::ProceduralBlockSymbol &procNode) {
118  auto procOp = builder.create<moore::ProcedureOp>(
119  loc, convertProcedureKind(procNode.procedureKind));
120  procOp.getBodyRegion().emplaceBlock();
121  OpBuilder::InsertionGuard guard(builder);
122  builder.setInsertionPointToEnd(procOp.getBody());
123  Context::ValueSymbolScope scope(context.valueSymbols);
124  return context.convertStatement(procNode.getBody());
125  }
126 
127  // Ignore statement block symbols. These get generated by Slang for blocks
128  // with variables and other declarations. For example, having an initial
129  // procedure with a variable declaration, such as `initial begin int x; end`,
130  // will create the procedure with a block and variable declaration as
131  // expected, but will also create a `StatementBlockSymbol` with just the
132  // variable layout _next to_ the initial procedure.
133  LogicalResult visit(const slang::ast::StatementBlockSymbol &) {
134  return success();
135  }
136 
137  /// Emit an error for all other members.
138  template <typename T>
139  LogicalResult visit(T &&node) {
140  mlir::emitError(loc, "unsupported construct: ")
141  << slang::ast::toString(node.kind);
142  return failure();
143  }
144 };

151 /// Convert an entire Slang compilation to MLIR ops. This is the main entry
152 /// point for the conversion.
153 LogicalResult
154 Context::convertCompilation(slang::ast::Compilation &compilation) {
155  const auto &root = compilation.getRoot();
156 
157  // Visit all top-level declarations in all compilation units. This does not
158  // include instantiable constructs like modules, interfaces, and programs,
159  // which are listed separately as top instances.
160  for (auto *unit : root.compilationUnits) {
161  for (const auto &member : unit->members()) {
162  // Error out on all top-level declarations.
163  auto loc = convertLocation(member.location);
164  return mlir::emitError(loc, "unsupported construct: ")
165  << slang::ast::toString(member.kind);
166  }
167  }
168 
169  // Prime the root definition worklist by adding all the top-level modules.
170  SmallVector<const slang::ast::InstanceSymbol *> topInstances;
171  for (auto *inst : root.topInstances)
172  convertModuleHeader(&inst->body);
173 
174  // Convert all the root module definitions.
175  while (!moduleWorklist.empty()) {
176  auto *module = moduleWorklist.front();
177  moduleWorklist.pop();
178  if (failed(convertModuleBody(module)))
179  return failure();
180  }
181 
182  return success();
183 }

185 /// Convert a module and its ports to an empty module op in the IR. Also adds
186 /// the op to the worklist of module bodies to be lowered. This acts like a
187 /// module "declaration", allowing instances to already refer to a module even
188 /// before its body has been lowered.
189 moore::SVModuleOp
190 Context::convertModuleHeader(const slang::ast::InstanceBodySymbol *module) {
191  if (auto op = moduleOps.lookup(module))
192  return op;
193  auto loc = convertLocation(module->location);
194  OpBuilder::InsertionGuard g(builder);
195 
196  // We only support modules for now. Extension to interfaces and programs
197  // should be trivial though, since they are essentially the same thing with
198  // only minor differences in semantics.
199  if (module->getDefinition().definitionKind !=
200  slang::ast::DefinitionKind::Module) {
201  mlir::emitError(loc, "unsupported construct: ")
202  << module->getDefinition().getKindString();
203  return {};
204  }
205 
206  // Handle the port list.
207  for (auto *symbol : module->getPortList()) {
208  auto portLoc = convertLocation(symbol->location);
209  mlir::emitError(portLoc, "unsupported module port: ")
210  << slang::ast::toString(symbol->kind);
211  return {};
212  }
213 
214  // Pick an insertion point for this module according to the source file
215  // location.
216  auto it = orderedRootOps.lower_bound(module->location);
217  if (it == orderedRootOps.end())
218  builder.setInsertionPointToEnd(intoModuleOp.getBody());
219  else
220  builder.setInsertionPoint(it->second);
221 
222  // Create an empty module that corresponds to this module.
223  auto moduleOp = builder.create<moore::SVModuleOp>(loc, module->name);
224  orderedRootOps.insert(it, {module->location, moduleOp});
225  moduleOp.getBodyRegion().emplaceBlock();
226 
227  // Add the module to the symbol table of the MLIR module, which uniquifies its
228  // name as we'd expect.
229  symbolTable.insert(moduleOp);
230 
231  // Schedule the body to be lowered.
232  moduleWorklist.push(module);
233  moduleOps.insert({module, moduleOp});
234  return moduleOp;
235 }

237 /// Convert a module's body to the corresponding IR ops. The module op must have
238 /// already been created earlier through a `convertModuleHeader` call.
239 LogicalResult
240 Context::convertModuleBody(const slang::ast::InstanceBodySymbol *module) {
241  auto moduleOp = moduleOps.lookup(module);
242  assert(moduleOp);
243  OpBuilder::InsertionGuard g(builder);
244  builder.setInsertionPointToEnd(moduleOp.getBody());
245 
246  ValueSymbolScope scope(valueSymbols);
247  for (auto &member : module->members()) {
248  auto loc = convertLocation(member.location);
249  if (failed(member.visit(MemberVisitor(*this, loc))))
250  return failure();
251  }
252 
253  return success();
254 }
