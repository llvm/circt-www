25 /// A namespace that is used to store existing names and generate new names in
26 /// some scope within the IR. This exists to work around limitations of
27 /// SymbolTables. This acts as a base class providing facilities common to all
28 /// namespaces implementations.
29 class Namespace {
30 public:
31  Namespace() {}
32  Namespace(const Namespace &other) = default;
33  Namespace(Namespace &&other) : nextIndex(std::move(other.nextIndex)) {}
34 
35  Namespace &operator=(const Namespace &other) = default;
36  Namespace &operator=(Namespace &&other) {
37  nextIndex = std::move(other.nextIndex);
38  return *this;
39  }

41  /// SymbolCache initializer; initialize from every key that is convertible to
42  /// a StringAttr in the SymbolCache.
43  void add(SymbolCache &symCache) {
44  for (auto &&[attr, _] : symCache)
45  if (auto strAttr = attr.dyn_cast<StringAttr>())
46  nextIndex.insert({strAttr.getValue(), 0});
47  }
48 
49  /// Empty the namespace.
50  void clear() { nextIndex.clear(); }
51 
52  /// Return a unique name, derived from the input `name`, and add the new name
53  /// to the internal namespace. There are two possible outcomes for the
54  /// returned name:
55  ///
56  /// 1. The original name is returned.
57  /// 2. The name is given a `_<n>` suffix where `<n>` is a number starting from
58  /// `_0` and incrementing by one each time.
59  StringRef newName(const Twine &name) {
60  // Special case the situation where there is no name collision to avoid
61  // messing with the SmallString allocation below.
62  llvm::SmallString<64> tryName;
63  auto inserted = nextIndex.insert({name.toStringRef(tryName), 0});
64  if (inserted.second)
65  return inserted.first->getKey();
66 
67  // Try different suffixes until we get a collision-free one.
68  if (tryName.empty())
69  name.toVector(tryName); // toStringRef may leave tryName unfilled
70 
71  // Indexes less than nextIndex[tryName] are lready used, so skip them.
72  // Indexes larger than nextIndex[tryName] may be used in another name.
73  size_t &i = nextIndex[tryName];
74  tryName.push_back('_');
75  size_t baseLength = tryName.size();
76  do {
77  tryName.resize(baseLength);
78  Twine(i++).toVector(tryName); // append integer to tryName
79  inserted = nextIndex.insert({tryName, 0});
80  } while (!inserted.second);
81 
82  return inserted.first->getKey();
83  }
84 
85  /// Return a unique name, derived from the input `name` and ensure the
86  /// returned name has the input `suffix`. Also add the new name to the
87  /// internal namespace.
88  /// There are two possible outcomes for the returned name:
89  /// 1. The original name + suffix is returned.
90  /// 2. The name is given a suffix `_<n>_<suffix>` where `<n>` is a number
91  /// starting from `_0` and incrementing by one each time.
92  StringRef newName(const Twine &name, const Twine &suffix) {
93  // Special case the situation where there is no name collision to avoid
94  // messing with the SmallString allocation below.
95  llvm::SmallString<64> tryName;
96  auto inserted =
97  nextIndex.insert({name.concat(suffix).toStringRef(tryName), 0});
98  if (inserted.second)
99  return inserted.first->getKey();
100 
101  // Try different suffixes until we get a collision-free one.
102  tryName.clear();
103  name.toVector(tryName); // toStringRef may leave tryName unfilled
104  tryName.push_back('_');
105  size_t baseLength = tryName.size();
106 
107  // Get the initial number to start from. Since `:` is not a valid character
108  // in a verilog identifier, we use it separate the name and suffix.
109  tryName.push_back(':');
110  suffix.toVector(tryName);
111 
112  // Indexes less than nextIndex[tryName] are already used, so skip them.
113  // Indexes larger than nextIndex[tryName] may be used in another name.
114  size_t &i = nextIndex[tryName];
115  do {
116  tryName.resize(baseLength);
117  Twine(i++).toVector(tryName); // append integer to tryName
118  suffix.toVector(tryName);
119  inserted = nextIndex.insert({tryName, 0});
120  } while (!inserted.second);
121 
122  return inserted.first->getKey();
123  }
124 
125 protected:
126  // The "next index" that will be tried when trying to unique a string within a
127  // namespace. It follows that all values less than the "next index" value are
128  // already used.
129  llvm::StringMap<size_t> nextIndex;
130 };
