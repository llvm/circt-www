26 static LogicalResult constructProblem(Problem &prob, FuncOp func) {
27  // set up catch-all operator type with unit latency
28  auto unitOpr = prob.getOrInsertOperatorType("unit");
29  prob.setLatency(unitOpr, 1);
30 
31  // parse additional operator type information attached to the test case
32  if (auto attr = func->getAttrOfType<ArrayAttr>("operatortypes")) {
33  for (auto oprAttr : attr.getAsRange<DictionaryAttr>()) {
34  auto name = oprAttr.getAs<StringAttr>("name");
35  auto latency = oprAttr.getAs<IntegerAttr>("latency");
36  if (!(name && latency))
37  continue;
38 
39  auto opr = prob.getOrInsertOperatorType(name.getValue());
40  prob.setLatency(opr, latency.getInt());
41  }
42  }
43 
44  // construct problem (consider only the first block)
45  for (auto &op : func.getBlocks().front().getOperations()) {
46  prob.insertOperation(&op);
47 
48  if (auto oprRefAttr = op.getAttrOfType<StringAttr>("opr")) {
49  auto opr = prob.getOrInsertOperatorType(oprRefAttr.getValue());
50  prob.setLinkedOperatorType(&op, opr);
51  } else {
52  prob.setLinkedOperatorType(&op, unitOpr);
53  }
54  }
55 
56  // parse auxiliary dependences in the testcase, encoded as an array of
57  // 2-element arrays of integer attributes (see `test_asap.mlir`)
58  if (auto attr = func->getAttrOfType<ArrayAttr>("auxdeps")) {
59  auto &ops = prob.getOperations();
60  for (auto auxDepAttr : attr.getAsRange<ArrayAttr>()) {
61  if (auxDepAttr.size() != 2)
62  continue;
63  auto fromIdxAttr = auxDepAttr[0].dyn_cast<IntegerAttr>();
64  auto toIdxAttr = auxDepAttr[1].dyn_cast<IntegerAttr>();
65  if (!fromIdxAttr || !toIdxAttr)
66  continue;
67  unsigned fromIdx = fromIdxAttr.getInt();
68  unsigned toIdx = toIdxAttr.getInt();
69  if (fromIdx >= ops.size() || toIdx >= ops.size())
70  continue;
71 
72  // finally, we have two integer indices in range of the operations list
73  if (failed(prob.insertDependence(
74  std::make_pair(ops[fromIdx], ops[toIdx])))) {
75  return func->emitError("inserting aux dependence failed");
76  }
77  }
78  }
79 
80  return success();
81 }

93 void TestProblemPass::runOnFunction() {
94  auto func = getFunction();
95 
96  Problem prob(func);
97  if (failed(constructProblem(prob, func))) {
98  func->emitError("problem construction failed");
99  return signalPassFailure();
100  }
101 
102  if (failed(prob.check())) {
103  func->emitError("problem check failed");
104  return signalPassFailure();
105  }
106 
107  // get schedule from the test case
108  for (auto *op : prob.getOperations())
109  if (auto startTimeAttr = op->getAttrOfType<IntegerAttr>("problemStartTime"))
110  prob.setStartTime(op, startTimeAttr.getInt());
111 
112  if (failed(prob.verify())) {
113  func->emitError("problem verification failed");
114  return signalPassFailure();
115  }
116 }

131 
132  Problem prob(func);

134  func->emitError("problem construction failed");

142

144  func->emitError("scheduling failed");
145  return signalPassFailure();

147
