71 /// times as inline expression (when they have multiple uses) rather than having
72 /// a temporary wire.

75 /// subtrees arbitrarily.

77  // We don't want wires that are just constants aesthetically.
78  if (isConstantExpression(op))
79  return true;

85  return true;

87 
88  return false;

101 /// Return the width of the specified type in bits or -1 if it isn't
102 /// supported.
103 static int getBitWidthOrSentinel(Type type) {
104  return TypeSwitch<Type, int>(type)
105  .Case<IntegerType>([](IntegerType integerType) {
106  // Verilog doesn't support zero bit integers. We only support them in
107  // limited cases.
108  return integerType.getWidth();
109  })
110  .Case<InOutType>([](InOutType inoutType) {
111  return getBitWidthOrSentinel(inoutType.getElementType());
112  })
113  .Case<TypeAliasType>([](TypeAliasType alias) {
114  return getBitWidthOrSentinel(alias.getInnerType());
115  })
116  .Default([](Type) { return -1; });
117 }

119 /// Push this type's dimension into a vector.
120 static void getTypeDims(SmallVectorImpl<int64_t> &dims, Type type,
121  Location loc) {
122  if (auto inout = type.dyn_cast<hw::InOutType>())
123  return getTypeDims(dims, inout.getElementType(), loc);
124  if (auto uarray = type.dyn_cast<hw::UnpackedArrayType>())
125  return getTypeDims(dims, uarray.getElementType(), loc);
126  if (type.isa<InterfaceType>())
127  return;
128  if (hw::type_isa<StructType>(type))
129  return;
130 
131  int width;
132  if (auto arrayType = hw::type_dyn_cast<hw::ArrayType>(type)) {
133  width = arrayType.getSize();
134  } else {
135  width = getBitWidthOrSentinel(type);
136  }
137  if (width == -1)
138  mlir::emitError(loc, "value has an unsupported verilog type ") << type;
139 
140  if (width != 1) // Width 1 is implicit.
141  dims.push_back(width);
142 
143  if (auto arrayType = type.dyn_cast<hw::ArrayType>()) {
144  getTypeDims(dims, arrayType.getElementType(), loc);
145  }
146 }

148 /// Emit a list of dimensions.
149 static void emitDims(ArrayRef<int64_t> dims, raw_ostream &os) {
150  for (int64_t width : dims)
151  switch (width) {
152  case -1: // -1 is an invalid type.
153  os << "<<invalid type>>";
154  return;
155  case 0:
156  os << "/*Zero Width*/";
157  break;
158  default:
159  os << '[' << (width - 1) << ":0]";
160  break;
161  }
162 }

182 /// Return true if this is a zero bit type, e.g. a zero bit integer or array

186  return intType.getWidth() == 0;

212 /// Output the basic type that consists of packed and primitive types. This is
213 /// those to the left of the name in verilog. implicitIntType controls whether
214 /// to print a base type for (logic) for inteters or whether the caller will
215 /// have handled this (with logic, wire, reg, etc).
216 /// Returns whether anything was printed out
217 static bool printPackedTypeImpl(Type type, raw_ostream &os, Operation *op,
218  SmallVectorImpl<int64_t> &dims,
219  bool implicitIntType) {
220  return TypeSwitch<Type, bool>(type)
221  .Case<IntegerType>([&](IntegerType integerType) {
222  if (!implicitIntType)
223  os << "logic";
224  if (integerType.getWidth() != 1)
225  dims.push_back(integerType.getWidth());
226  if (!dims.empty() && !implicitIntType)
227  os << ' ';
228 
229  emitDims(dims, os);
230  return !dims.empty() || !implicitIntType;
231  })
232  .Case<InOutType>([&](InOutType inoutType) {
233  return printPackedTypeImpl(inoutType.getElementType(), os, op, dims,
234  implicitIntType);
235  })
236  .Case<StructType>([&](StructType structType) {
237  os << "struct packed {";
238  for (auto &element : structType.getElements()) {
239  SmallVector<int64_t, 8> structDims;
240  printPackedTypeImpl(stripUnpackedTypes(element.type), os, op,
241  structDims, /*implicitIntType=*/false);
242  os << ' ' << element.name << "; ";
243  }
244  os << '}';
245  emitDims(dims, os);
246  return true;
247  })
248  .Case<ArrayType>([&](ArrayType arrayType) {
249  dims.push_back(arrayType.getSize());
250  return printPackedTypeImpl(arrayType.getElementType(), os, op, dims,
251  implicitIntType);
252  })
253  .Case<InterfaceType>([](InterfaceType ifaceType) { return false; })
254  .Case<UnpackedArrayType>([&](UnpackedArrayType arrayType) {
255  os << "<<unexpected unpacked array>>";
256  op->emitError("Unexpected unpacked array in packed type ") << arrayType;
257  return true;
258  })
259  .Case<TypeAliasType>([&](TypeAliasType typeRef) {
260  auto typedecl = typeRef.getDecl(op);
261  if (!typedecl.hasValue())
262  return false;
263 
264  os << typedecl.getValue().getPreferredName();
265  emitDims(dims, os);
266  return true;
267  })
268  .Default([&](Type type) {
269  os << "<<invalid type>>";
270  op->emitError("value has an unsupported verilog type ") << type;
271  return true;
272  });
273 }

294 /// Return the word (e.g. "reg") in Verilog to declare the specified thing.
295 static StringRef getVerilogDeclWord(Operation *op) {
296  if (isa<RegOp>(op)) {
297  // Check if the type stored in this register is a struct or array of
298  // structs. In this case, according to spec section 6.8, the "reg" prefix
299  // should be left off.
300  auto elementType =
301  op->getResult(0).getType().cast<InOutType>().getElementType();
302  if (elementType.isa<StructType>())
303  return "";
304  if (auto innerType = elementType.dyn_cast<ArrayType>()) {
305  while (innerType.getElementType().isa<ArrayType>())
306  innerType = innerType.getElementType().cast<ArrayType>();
307  if (innerType.getElementType().isa<StructType>() ||
308  innerType.getElementType().isa<TypeAliasType>())
309  return "";
310  }
311  if (elementType.isa<TypeAliasType>())
312  return "";
313 
314  return "reg";
315  }
316  if (isa<WireOp>(op))
317  return "wire";
318  if (isa<ConstantOp>(op))
319  return "localparam";
320 
321  // Interfaces instances use the name of the declared interface.
322  if (auto interface = dyn_cast<InterfaceInstanceOp>(op))
323  return interface.getInterfaceType().getInterface().getValue();
324 
325  // If 'op' is in a module, output 'wire'. If 'op' is in a procedural block,
326  // fall through to default.
327  bool isProcedural = op->getParentOp()->hasTrait<ProceduralRegion>();
328  return isProcedural ? "automatic logic" : "wire";
329 }
330 /// Return the name of a value without using the name map. This is needed when
331 /// looking into an instance from a different module as happens with bind. It
332 /// may return "" when unable to determine a name. This works in situations
333 /// where names are pre-legalized during prepare.
334 static StringRef getNameRemotely(Value &value,
335  ArrayRef<ModulePortInfo> modulePorts) {
336  if (auto barg = value.dyn_cast<BlockArgument>()) {
337  return modulePorts[barg.getArgNumber()].getName();
338  }
339  if (auto readinout = dyn_cast<ReadInOutOp>(value.getDefiningOp())) {
340  if (auto wire = dyn_cast<WireOp>(readinout.input().getDefiningOp())) {
341  return wire.name();
342  }
343  if (auto reg = dyn_cast<RegOp>(readinout.input().getDefiningOp())) {
344  return reg.name();
345  }
346  }
347  return {};
348 }

374 /// Return the location information as a (potentially empty) string.
375 static std::string
376 getLocationInfoAsString(const SmallPtrSet<Operation *, 8> &ops) {
377  std::string resultStr;
378  llvm::raw_string_ostream sstr(resultStr);
379 
380  // Multiple operations may come from the same location or may not have useful
381  // location info. Unique it now.
382  SmallPtrSet<Attribute, 8> locations;
383  for (auto *op : ops) {
384  if (auto loc = op->getLoc().dyn_cast<FileLineColLoc>())
385  locations.insert(loc);
386  }
387 
388  auto printLoc = [&](FileLineColLoc loc) {
389  sstr << loc.getFilename();
390  if (auto line = loc.getLine()) {
391  sstr << ':' << line;
392  if (auto col = loc.getColumn())
393  sstr << ':' << col;
394  }
395  };
396 
397  switch (locations.size()) {
398  case 1:
399  printLoc((*locations.begin()).cast<FileLineColLoc>());
400  LLVM_FALLTHROUGH;
401  case 0:
402  return sstr.str();
403  default:
404  break;
405  }
406 
407  // Sort the entries.
408  SmallVector<FileLineColLoc, 8> locVector;
409  locVector.reserve(locations.size());
410  for (auto loc : locations)
411  locVector.push_back(loc.cast<FileLineColLoc>());
412 
413  llvm::array_pod_sort(
414  locVector.begin(), locVector.end(),
415  [](const FileLineColLoc *lhs, const FileLineColLoc *rhs) -> int {
416  if (auto fn = lhs->getFilename().compare(rhs->getFilename()))
417  return fn;
418  if (lhs->getLine() != rhs->getLine())
419  return lhs->getLine() < rhs->getLine() ? -1 : 1;
420  return lhs->getColumn() < rhs->getColumn() ? -1 : 1;
421  });
422 
423  // The entries are sorted by filename, line, col. Try to merge together
424  // entries to reduce verbosity on the column info.
425  StringRef lastFileName;
426  for (size_t i = 0, e = locVector.size(); i != e;) {
427  if (i != 0)
428  sstr << ", ";
429 
430  // Print the filename if it changed.
431  auto first = locVector[i];
432  if (first.getFilename() != lastFileName) {
433  lastFileName = first.getFilename();
434  sstr << lastFileName;
435  }
436 
437  // Scan for entries with the same file/line.
438  size_t end = i + 1;
439  while (end != e && first.getFilename() == locVector[end].getFilename() &&
440  first.getLine() == locVector[end].getLine())
441  ++end;
442 
443  // If we have one entry, print it normally.
444  if (end == i + 1) {
445  if (auto line = first.getLine()) {
446  sstr << ':' << line;
447  if (auto col = first.getColumn())
448  sstr << ':' << col;
449  }
450  ++i;
451  continue;
452  }
453 
454  // Otherwise print a brace enclosed list.
455  sstr << ':' << first.getLine() << ":{";
456  while (i != end) {
457  sstr << locVector[i++].getColumn();
458 
459  if (i != end)
460  sstr << ',';
461  }
462  sstr << '}';
463  }
464 
465  return sstr.str();
466 }

667 void EmitterBase::emitTextWithSubstitutions(
668  StringRef string, Operation *op,
669  std::function<void(Value)> operandEmitter) {
670  // Perform operand substitions as we emit the line string. We turn {{42}}
671  // into the value of operand 42.
672 
673  // Scan 'line' for a substitution, emitting any non-substitution prefix,
674  // then the mentioned operand, chopping the relevant text off 'line' and
675  // returning true. This returns false if no substitution is found.
676  auto emitUntilSubstitution = [&](size_t next = 0) -> bool {
677  size_t start = 0;
678  while (1) {
679  next = string.find("{{", next);
680  if (next == StringRef::npos)
681  return false;
682 
683  // Check to make sure we have a number followed by }}. If not, we
684  // ignore the {{ sequence as something that could happen in Verilog.
685  next += 2;
686  start = next;
687  while (next < string.size() && isdigit(string[next]))
688  ++next;
689  // We need at least one digit.
690  if (start == next)
691  continue;
692 
693  // We must have a }} right after the digits.
694  if (!string.substr(next).startswith("}}"))
695  continue;
696 
697  // We must be able to decode the integer into an unsigned.
698  unsigned operandNo = 0;
699  if (string.drop_front(start)
700  .take_front(next - start)
701  .getAsInteger(10, operandNo)) {
702  emitError(op, "operand substitution too large");
703  continue;
704  }
705  next += 2;
706 
707  if (operandNo >= op->getNumOperands()) {
708  emitError(op, "operand " + llvm::utostr(operandNo) + " isn't valid");
709  continue;
710  }
711 
712  // Emit any text before the substitution.
713  os << string.take_front(start - 2);
714 
715  // Emit the operand.
716  operandEmitter(op->getOperand(operandNo));
717 
718  // Forget about the part we emitted.
719  string = string.drop_front(next);
720  return true;
721  }
722  };
723 
724  // Emit all the substitutions.
725  while (emitUntilSubstitution())
726  ;
727 
728  // Emit any text after the last substitution.
729  os << string;
730 }

972 SubExprInfo ExprEmitter::emitBinary(Operation *op, VerilogPrecedence prec,
973  const char *syntax,
974  SubExprSignRequirement operandSignReq) {
975  auto lhsInfo =
976  emitSubExpr(op->getOperand(0), prec, OOLBinary, operandSignReq);
977  os << ' ' << syntax << ' ';
978 
979  // Right associative operators are already generally variadic, we need to
980  // handle things like: (a<4> == b<4>) == (c<3> == d<3>).
981  // When processing the top operation of the tree, the rhs needs parens.
982  auto rhsPrec = VerilogPrecedence(prec - 1);
983  auto rhsInfo =
984  emitSubExpr(op->getOperand(1), rhsPrec, OOLBinary, operandSignReq);
985 
986  // SystemVerilog 11.8.1 says that the result of a binary expression is signed
987  // only if both operands are signed.
988  SubExprSignResult signedness = IsUnsigned;
989  if (lhsInfo.signedness == IsSigned && rhsInfo.signedness == IsSigned)
990  signedness = IsSigned;
991 
992  return {prec, signedness};
993 }

996  const char *syntax) {

1002  if (emitSubExpr(v1, prec, OOLBinary).signedness != IsSigned)
1003  sign = IsUnsigned;
1004  },
1005  [&] { os << ' ' << syntax << ' '; });
1006 
1007  return {prec, sign};

1019 /// up to the point where they turn into massively long source lines of Verilog.
1020 /// At that point, we retroactively break the huge expression by inserting
1021 /// temporaries. This handles the bookkeeping.
1022 void ExprEmitter::retroactivelyEmitExpressionIntoTemporary(Operation *op) {
1023  assert(isVerilogExpression(op) && !emitter.outOfLineExpressions.count(op) &&
1024  "Should only be called on expressions though to be inlined");
1025 
1026  emitter.outOfLineExpressions.insert(op);
1027  names.addName(op->getResult(0), "_tmp");
1028

1030  tooLargeSubExpressions.push_back(op);

1033 /// Emit the specified value as a subexpression to the stream.
1034 SubExprInfo ExprEmitter::emitSubExpr(Value exp,
1035  VerilogPrecedence parenthesizeIfLooserThan,
1036  SubExprOutOfLineBehavior outOfLineBehavior,
1037  SubExprSignRequirement signRequirement) {
1038  auto *op = exp.getDefiningOp();
1039  bool shouldEmitInlineExpr = op && isVerilogExpression(op);
1040 
1041  // Don't emit this expression inline if it has multiple uses.
1042  if (shouldEmitInlineExpr && parenthesizeIfLooserThan != ForceEmitMultiUse &&
1043  emitter.outOfLineExpressions.count(op))
1044  shouldEmitInlineExpr = false;
1045 
1046  // If this is a non-expr or shouldn't be done inline, just refer to its name.
1047  if (!shouldEmitInlineExpr) {
1048  // All wires are declared as unsigned, so if the client needed it signed,
1049  // emit a conversion.
1050  if (signRequirement == RequireSigned) {
1051  os << "$signed(" << names.getName(exp) << ')';
1052  return {Symbol, IsSigned};
1053  }
1054 
1055  os << names.getName(exp);
1056  return {Symbol, IsUnsigned};
1057  }
1058 
1059  unsigned subExprStartIndex = outBuffer.size();
1060 
1061  // Inform the visit method about the preferred sign we want from the result.
1062  // It may choose to ignore this, but some emitters can change behavior based
1063  // on contextual desired sign.
1064  signPreference = signRequirement;
1065 
1066  // Okay, this is an expression we should emit inline. Do this through our
1067  // visitor.
1068  auto expInfo = dispatchCombinationalVisitor(exp.getDefiningOp());
1069 
1070  // Check cases where we have to insert things before the expression now that
1071  // we know things about it.
1072  auto addPrefix = [&](StringRef prefix) {
1073  outBuffer.insert(outBuffer.begin() + subExprStartIndex, prefix.begin(),
1074  prefix.end());
1075  };
1076  if (signRequirement == RequireSigned && expInfo.signedness == IsUnsigned) {
1077  addPrefix("$signed(");
1078  os << ')';
1079  expInfo.signedness = IsSigned;
1080  } else if (signRequirement == RequireUnsigned &&
1081  expInfo.signedness == IsSigned) {
1082  addPrefix("$unsigned(");
1083  os << ')';
1084  expInfo.signedness = IsUnsigned;
1085  } else if (expInfo.precedence > parenthesizeIfLooserThan) {
1086  // If this subexpression would bind looser than the expression it is bound
1087  // into, then we need to parenthesize it. Insert the parentheses
1088  // retroactively.
1089  addPrefix("(");
1090  os << ')';
1091  // Reset the precedence to the () level.
1092  expInfo.precedence = Selection;
1093  }
1094 
1095  // If we emitted this subexpression and it resulted in something very large,
1096  // then we may be in the process of making super huge lines. Back off to
1097  // emitting this as its own temporary on its own line.
1098  unsigned threshold;
1099  switch (outOfLineBehavior) {
1100  case OOLTopLevel:
1101  threshold = ~0U;
1102  break;
1103  case OOLUnary:
1104  threshold = std::max(state.options.emittedLineLength - 20, 10U);
1105  break;
1106  case OOLBinary:
1107  threshold = std::max(state.options.emittedLineLength / 2, 10U);
1108  break;
1109  }
1110 
1111  if (outBuffer.size() - subExprStartIndex > threshold &&
1112  parenthesizeIfLooserThan != ForceEmitMultiUse &&
1113  !isExpressionAlwaysInline(op)) {
1114  // Inform the module emitter that this expression needs a temporary
1115  // wire/logic declaration and set it up so it will be referenced instead of
1116  // emitted inline.
1117  retroactivelyEmitExpressionIntoTemporary(op);
1118 
1119  // Lop this off the buffer we emitted.
1120  outBuffer.resize(subExprStartIndex);
1121 
1122  // Try again, now it will get emitted as a out-of-line leaf.
1123  return emitSubExpr(exp, parenthesizeIfLooserThan, outOfLineBehavior,
1124  signRequirement);
1125  }
1126 
1127  // Remember that we emitted this.
1128  emittedExprs.insert(exp.getDefiningOp());
1129  return expInfo;
1130 }

1132 SubExprInfo ExprEmitter::visitComb(SExtOp op) {
1133  auto inWidth = op.getOperand().getType().getIntOrFloatBitWidth();
1134  auto destWidth = op.getType().getIntOrFloatBitWidth();
1135 
1136  // Handle sign extend from a single bit in a pretty way.
1137  if (inWidth == 1) {
1138  os << '{' << destWidth << '{';
1139  emitSubExpr(op.getOperand(), LowestPrecedence, OOLUnary);
1140  os << "}}";
1141  return {Symbol, IsUnsigned};
1142  }
1143 
1144  // Otherwise, this is a sign extension of a general expression.
1145  os << '{';
1146  if (destWidth - inWidth == 1) {
1147  // Special pattern for single bit extension, where we just need the bit.
1148  emitSubExpr(op.getOperand(), Unary, OOLUnary);
1149  os << '[' << (inWidth - 1) << ']';
1150  } else {
1151  // General pattern for multi-bit extension: splat the bit.
1152  os << '{' << (destWidth - inWidth) << '{';
1153  emitSubExpr(op.getOperand(), Unary, OOLUnary);
1154  os << '[' << (inWidth - 1) << "]}}";
1155  }
1156  os << ", ";
1157  emitSubExpr(op.getOperand(), LowestPrecedence, OOLUnary);
1158  os << '}';

1199 SubExprInfo ExprEmitter::visitComb(ICmpOp op) {
1200  const char *symop[] = {"==", "!=", "<", "<=", ">",
1201  ">=", "<", "<=", ">", ">="};
1202  SubExprSignRequirement signop[] = {
1203  // Equality
1204  NoRequirement, NoRequirement,
1205  // Signed Comparisons
1206  RequireSigned, RequireSigned, RequireSigned, RequireSigned,
1207  // Unsigned Comparisons
1208  RequireUnsigned, RequireUnsigned, RequireUnsigned, RequireUnsigned};
1209 
1210  auto pred = static_cast<uint64_t>(op.predicate());
1211  assert(pred < sizeof(symop) / sizeof(symop[0]));
1212 
1213  // Lower "== -1" to Reduction And.
1214  if (op.isEqualAllOnes())
1215  return emitUnary(op, "&", true);
1216 
1217  // Lower "!= 0" to Reduction Or.
1218  if (op.isNotEqualZero())
1219  return emitUnary(op, "|", true);
1220 
1221  auto result = emitBinary(op, Comparison, symop[pred], signop[pred]);
1222 
1223  // SystemVerilog 11.8.1: "Comparison... operator results are unsigned,
1224  // regardless of the operands".
1225  result.signedness = IsUnsigned;
1226  return result;

1430 /// Return true if we are unable to ever inline the specified operation. This
1431 /// happens because not all Verilog expressions are composable, notably you
1432 /// can only use bit selects like x[4:6] on simple expressions, you cannot use
1433 /// expressions in the sensitivity list of always blocks, etc.
1434 static bool isExpressionUnableToInline(Operation *op) {
1435  if (auto cast = dyn_cast<BitcastOp>(op))
1436  if (!haveMatchingDims(cast.input().getType(), cast.result().getType(),
1437  op->getLoc()))
1438  // Bitcasts rely on the type being assigned to, so we cannot inline.
1439  return true;
1440 
1441  // StructCreateOp needs to be assigning to a named temporary so that types
1442  // are inferred properly by verilog
1443  if (isa<StructCreateOp>(op))
1444  return true;
1445 
1446  auto *opBlock = op->getBlock();
1447 
1448  // Scan the users of the operation to see if any of them need this to be
1449  // emitted out-of-line.
1450  for (auto user : op->getUsers()) {
1451  // If the user is in a different block and the op shouldn't be inlined, then
1452  // we emit this as an out-of-line declaration into its block and the user
1453  // can refer to it.
1454  if (user->getBlock() != opBlock)
1455  return true;
1456 
1457  // Verilog bit selection is required by the standard to be:
1458  // "a vector, packed array, packed structure, parameter or concatenation".
1459  // It cannot be an arbitrary expression.
1460  if (isa<ExtractOp>(user))
1461  if (!isOkToBitSelectFrom(op->getResult(0)))
1462  return true;
1463 
1464  // Sign extend (when the operand isn't a single bit) requires a bitselect
1465  // syntactically.
1466  if (auto sext = dyn_cast<SExtOp>(user)) {
1467  auto sextOperandType = sext.getOperand().getType().cast<IntegerType>();
1468  if (sextOperandType.getWidth() != 1 &&
1469  !isOkToBitSelectFrom(op->getResult(0)))
1470  return true;
1471  }
1472  // ArraySliceOp uses its operand twice, so we want to assign it first then
1473  // use that variable in the ArraySliceOp expression.
1474  if (isa<ArraySliceOp>(user) && !isa<ConstantOp>(op))
1475  return true;
1476 
1477  // Always blocks must have a name in their sensitivity list, not an expr.
1478  if (isa<AlwaysOp>(user) || isa<AlwaysFFOp>(user))
1479  return true;
1480  }
1481  return false;
1482 }

1484 /// Return true if this expression should be emitted inline into any statement
1485 /// that uses it.
1486 static bool isExpressionEmittedInline(Operation *op) {
1487  // If it isn't structurally possible to inline this expression, emit it out
1488  // of line.
1489  if (isExpressionUnableToInline(op))
1490  return false;
1491 
1492  // If it has a single use, emit it inline.
1493  if (op->getResult(0).hasOneUse())
1494  return true;
1495 
1496  // If it is nullary and duplicable, then we can emit it inline.
1497  return op->getNumOperands() == 0 && isDuplicatableNullaryExpression(op);
1498 }

1531 void NameCollector::collectNames(Block &block) {
1532  bool isBlockProcedural = block.getParentOp()->hasTrait<ProceduralRegion>();
1533 
1534  SmallString<32> nameTmp;
1535 
1536  // Loop over all of the results of all of the ops. Anything that defines a
1537  // value needs to be noticed.
1538  for (auto &op : block) {
1539  bool isExpr = isVerilogExpression(&op);
1540 
1541  // Instances are handled in prepareHWModule
1542  if (isa<InstanceOp>(op))
1543  continue;
1544 
1545  for (auto result : op.getResults()) {
1546  // If this is an expression emitted inline or unused, it doesn't need a
1547  // name.
1548  if (isExpr) {
1549  // If this expression is dead, or can be emitted inline, ignore it.
1550  if (result.use_empty() || isExpressionEmittedInline(&op))
1551  continue;
1552 
1553  // Remember that this expression should be emitted out of line.
1554  moduleEmitter.outOfLineExpressions.insert(&op);
1555  }
1556 
1557  // Otherwise, it must be an expression or a declaration like a
1558  // RegOp/WireOp.
1559  if (!names.hasName(result))
1560  names.addName(result, op.getAttrOfType<StringAttr>("name"));
1561 
1562  // Don't measure or emit wires that are emitted inline (i.e. the wire
1563  // definition is emitted on the line of the expression instead of a
1564  // block at the top of the module).
1565  if (isExpr) {
1566  // Procedural blocks always emit out of line variable declarations,
1567  // because Verilog requires that they all be at the top of a block.
1568  // TODO: Improve this, at least in the simple cases.
1569  if (!isBlockProcedural)
1570  continue;
1571  }
1572 
1573  // Emit this value.
1574  valuesToEmit.push_back(ValuesToEmitRecord{result, {}});
1575  auto &typeString = valuesToEmit.back().typeString;
1576 
1577  maxDeclNameWidth =
1578  std::max(getVerilogDeclWord(&op).size(), maxDeclNameWidth);
1579 
1580  // Convert the port's type to a string and measure it.
1581  {
1582  llvm::raw_svector_ostream stringStream(typeString);
1583  printPackedType(stripUnpackedTypes(result.getType()), stringStream,
1584  &op);
1585  }
1586  maxTypeWidth = std::max(typeString.size(), maxTypeWidth);
1587  }
1588 
1589  // Recursively process any regions under the op iff this is a procedural
1590  // #ifdef region: we need to emit automatic logic values at the top of the
1591  // enclosing region.
1592  if (isa<IfDefProceduralOp>(op)) {
1593  for (auto &region : op.getRegions()) {
1594  if (!region.empty())
1595  collectNames(region.front());
1596  }
1597  }
1598  }
1599 }

1783 /// Emit the specified value as an expression. If this is an inline-emitted
1784 /// expression, we emit that expression, otherwise we emit a reference to the
1785 /// already computed name.
1786 ///
1787 void StmtEmitter::emitExpression(Value exp,
1788  SmallPtrSet<Operation *, 8> &emittedExprs,
1789  VerilogPrecedence parenthesizeIfLooserThan) {
1790  SmallVector<Operation *> tooLargeSubExpressions;
1791  ExprEmitter(emitter, outBuffer, emittedExprs, tooLargeSubExpressions, names)
1792  .emitExpression(exp, parenthesizeIfLooserThan);
1793 
1794  // It is possible that the emitted expression was too large to fit on a line
1795  // and needs to be split. If so, the new subexpressions that need emitting
1796  // are put out into the the 'tooLargeSubExpressions' list. Re-emit these at
1797  // the start of the current statement as their own stmt expressions.
1798  if (tooLargeSubExpressions.empty())
1799  return;
1800 
1801  // Pop this statement off and save it to the side.
1802  std::string thisStmt(outBuffer.begin() + statementBeginningIndex,
1803  outBuffer.end());
1804  outBuffer.resize(statementBeginningIndex);
1805 
1806  // Emit each stmt expression in turn.
1807  for (auto *expr : tooLargeSubExpressions) {
1808  statementBeginningIndex = outBuffer.size();
1809  ++numStatementsEmitted;
1810  emitStatementExpression(expr);
1811  }
1812 
1813  // Re-add this statement now that all the preceeding ones are out.
1814  outBuffer.append(thisStmt.begin(), thisStmt.end());
1815 
1816  // If we are working on a procedural statement, we need to emit the
1817  // declarations for each variable separately from the assignments to them.
1818  // We do this after inserting the assign statements because we don't want to
1819  // invalidate the index.
1820  if (tooLargeSubExpressions[0]->getParentOp()->hasTrait<ProceduralRegion>()) {
1821  thisStmt.assign(outBuffer.begin() + blockDeclarationInsertPointIndex,
1822  outBuffer.end());
1823  outBuffer.resize(blockDeclarationInsertPointIndex);
1824  for (auto *expr : tooLargeSubExpressions) {
1825  if (!emitDeclarationForTemporary(expr))
1826  os << ";\n";
1827  }
1828  blockDeclarationInsertPointIndex = outBuffer.size();
1829  outBuffer.append(thisStmt.begin(), thisStmt.end());
1830  }

1834  // This is invoked for expressions that have a non-single use. This could
1835  // either be because they are dead or because they have multiple uses.
1836  if (op->getResult(0).use_empty()) {
1837  indent() << "// Unused: ";

1840  indent() << "// Zero width: ";

2928 // exists in the same block as its users and that it has one use for each one.
