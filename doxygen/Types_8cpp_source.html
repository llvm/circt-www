158  // Collect the members in a struct or union.
159  LogicalResult collectMembers(const slang::ast::Scope &structType,
160  SmallVectorImpl<moore::StructMember> &members,
161  bool enforcePacked) {
162  for (auto &field : structType.membersOfType<slang::ast::FieldSymbol>()) {
163  auto loc = context.convertLocation(field.location);
164  auto name = StringAttr::get(context.getContext(), field.name);
165  auto innerType = context.convertType(*field.getDeclaredType());
166  if (!innerType)
167  return failure();
168  // The Slang frontend guarantees the inner type to be packed if the struct
169  // is packed.
170  assert(!enforcePacked || isa<moore::PackedType>(innerType));
171  members.push_back({name, loc, cast<moore::UnpackedType>(innerType)});
172  }
173  return success();
174  }

176  // Handle packed and unpacked structs.
177  Type visit(const slang::ast::PackedStructType &type) {
178  auto loc = context.convertLocation(type.location);
179  SmallVector<moore::StructMember> members;
180  if (failed(collectMembers(type, members, true)))
181  return {};
182  return moore::PackedStructType::get(moore::StructKind::Struct, members,
183  StringAttr{}, loc);

188  SmallVector<moore::StructMember> members;
189  if (failed(collectMembers(type, members, false)))
190  return {};
191  return moore::UnpackedStructType::get(moore::StructKind::Struct, members,
192  StringAttr{}, loc);
