24 /// HandshakeVisitor is a visitor for handshake nodes.
25 template <typename ConcreteType, typename ResultType = void,
26  typename... ExtraArgs>
27 class HandshakeVisitor {
28 public:
29  ResultType dispatchHandshakeVisitor(Operation *op, ExtraArgs... args) {
30  auto *thisCast = static_cast<ConcreteType *>(this);
31  return TypeSwitch<Operation *, ResultType>(op)
32  .template Case<
33  // Handshake nodes.
34  BranchOp, BufferOp, ConditionalBranchOp, ConstantOp, ControlMergeOp,
35  EndOp, ForkOp, FuncOp, InstanceOp, JoinOp, LazyForkOp, LoadOp,
36  MemoryOp, ExternalMemoryOp, MergeOp, MuxOp, ReturnOp, SinkOp,
37  handshake::SelectOp, SourceOp, StartOp, StoreOp, TerminatorOp,
38  PackOp, UnpackOp>([&](auto opNode) -> ResultType {
39  return thisCast->visitHandshake(opNode, args...);
40  })
41  .Default([&](auto opNode) -> ResultType {
42  return thisCast->visitInvalidOp(op, args...);
43  });
44  }
45 
46  /// This callback is invoked on any invalid operations.
47  ResultType visitInvalidOp(Operation *op, ExtraArgs... args) {
48  op->emitOpError("is unsupported operation");
49  abort();
50  }
51 
52  /// This callback is invoked on any operations that are not handled by the
53  /// concrete visitor.
54  ResultType visitUnhandledOp(Operation *op, ExtraArgs... args) {
55  return ResultType();
56  }
57 
58 #define HANDLE(OPTYPE) \
59  ResultType visitHandshake(OPTYPE op, ExtraArgs... args) { \
60  return static_cast<ConcreteType *>(this)->visitUnhandledOp(op, args...); \
61  }
62 
63  // Handshake nodes.
64  HANDLE(BranchOp);
65  HANDLE(BufferOp);
66  HANDLE(ConditionalBranchOp);
67  HANDLE(ConstantOp);
68  HANDLE(ControlMergeOp);
69  HANDLE(EndOp);
70  HANDLE(ForkOp);
71  HANDLE(FuncOp);
72  HANDLE(InstanceOp);
73  HANDLE(JoinOp);
74  HANDLE(LazyForkOp);
75  HANDLE(LoadOp);
76  HANDLE(MemoryOp);
77  HANDLE(handshake::SelectOp);
78  HANDLE(ExternalMemoryOp);
79  HANDLE(MergeOp);
80  HANDLE(MuxOp);
81  HANDLE(ReturnOp);
82  HANDLE(SinkOp);
83  HANDLE(SourceOp);
84  HANDLE(StartOp);
85  HANDLE(StoreOp);
86  HANDLE(TerminatorOp);
87  HANDLE(PackOp);
88  HANDLE(UnpackOp);
89 #undef HANDLE
90 };
