27 /// This calculates the lowest common ancestor of the instance operations. It
28 /// supports finding the LCA of an arbitrary number of nodes (instead of the
29 /// usual 2 at a time), and incremental target node discovery.
30 static InstanceGraphNode *
31 lowestCommonAncestor(InstanceGraphNode *top,
32  llvm::function_ref<bool(InstanceRecord *)> predicate) {
33  struct StackElement {
34  StackElement(InstanceGraphNode *node)
35  : node(node), it(node->begin()), count(0) {}
36  // The node itself.
37  InstanceGraphNode *node;
38  // The current iterator of this node's children.
39  InstanceGraphNode::iterator it;
40  // The number of interesting nodes under this one.
41  unsigned count;
42  };
43 
44  // Since we support incremental discovery of the interesting modules, we keep
45  // track of node which has the most children under it so far.
46  InstanceGraphNode *currentLCA = nullptr;
47  unsigned currentCount = 0;
48 
49  // This is used to pass the count of a child back to its parent.
50  unsigned childCount = 0;
51 
52  // The workstack for a depth-first walk.
53  SmallVector<StackElement> stack;
54  stack.emplace_back(top);
55  while (!stack.empty()) {
56  auto &element = stack.back();
57  auto &node = element.node;
58  auto &it = element.it;
59 
60  // Add the count of the just-processed child node. If we are coming from
61  // the parent node, childCount will be 0.
62  element.count += childCount;
63 
64  // Check if we're done processing this nodes children.
65  if (it == node->end()) {
66  // Store the current count in the childCount, so that we may return the
67  // count to this node's parent op.
68  childCount = element.count;
69 
70  // If this node has more children than any other node, it is the best LCA
71  // of all the nodes we have found *so far*.
72  if (childCount > currentCount) {
73  currentLCA = node;
74  currentCount = element.count;
75  }
76 
77  // Pop back to the parent op.
78  stack.pop_back();
79  continue;
80  }
81 
82  // If the current node is interesting, increase this node's count.
83  auto *instanceNode = *it++;
84  if (predicate(instanceNode))
85  ++element.count;
86 
87  // Set up to iterate the child node.
88  stack.emplace_back(instanceNode->getTarget());
89  childCount = 0;
90  }
91  return currentLCA;
92 }

104 void WireDFTPass::runOnOperation() {
105  auto circuit = getOperation();
106 
107  // This is the module marked as the device under test.
108  Operation *dut = nullptr;
109 
110  // This is the signal marked as the DFT enable, a 1-bit signal to be wired to
111  // the EICG modules.
112  Value enableSignal;
113  FModuleOp enableModule;
114 
115  // Walk all modules looking for the DUT module and the annotated enable
116  // signal.
117  for (auto &op : *circuit.getBody()) {
118  auto module = dyn_cast<FModuleOp>(op);
119 
120  // If this isn't a regular module, continue.
121  if (!module)
122  continue;
123 
124  // Check if this module is the DUT.
125  AnnotationSet annos(module);
126  if (annos.hasAnnotation(dutClass)) {
127  // Check if we already found the DUT.
128  if (dut) {
129  auto diag = module->emitError("more than one module marked DUT");
130  diag.attachNote(dut->getLoc()) << "first module here";
131  signalPassFailure();
132  return;
133  }
134  dut = module;
135  }
136 
137  // See if this module has any port marked as the DFT enable.
138  bool error = false;
139  AnnotationSet::removePortAnnotations(module, [&](unsigned i,
140  Annotation anno) {
141  // If we have already encountered an error or this is not the right
142  // annotation kind, continue.
143  if (error || !anno.isClass(dftEnableClass))
144  return false;
145  // If we have already found a DFT enable, emit an error.
146  if (enableSignal) {
147  auto diag =
148  module->emitError("more than one thing marked as a DFT enable");
149  diag.attachNote(enableSignal.getLoc()) << "first thing defined here";
150  error = true;
151  return false;
152  }
153  // Grab the enable value and remove the annotation.
154  enableSignal =
155  getValueByFieldID(ImplicitLocOpBuilder::atBlockBegin(
156  module->getLoc(), module.getBody()),
157  module.getArgument(i), anno.getFieldID());
158  enableModule = module;
159  return true;
160  });
161  if (error)
162  return signalPassFailure();
163 
164  // Walk the module body looking for any operation marked as the DFT enable.
165  auto walkResult = module->walk([&](Operation *op) {
166  AnnotationSet::removeAnnotations(op, [&](Annotation anno) {
167  // If we have already encountered an error or this is not the right
168  // annotation kind, continue.
169  if (error || !anno.isClass(dftEnableClass))
170  return false;
171  if (enableSignal) {
172  auto diag =
173  op->emitError("more than one thing marked as a DFT enable");
174  diag.attachNote(enableSignal.getLoc()) << "first thing defined here";
175  error = true;
176  return false;
177  }
178  // Grab the enable value and remove the annotation.
179  enableSignal = getValueByFieldID(
180  ImplicitLocOpBuilder::atBlockEnd(op->getLoc(), op->getBlock()),
181  op->getResult(0), anno.getFieldID());
182  enableModule = module;
183  return true;
184  });
185  if (error)
186  return WalkResult::interrupt();
187  return WalkResult::advance();
188  });
189  if (walkResult.wasInterrupted())
190  return signalPassFailure();
191  }
192 
193  // No enable signal means we have no work to do.
194  if (!enableSignal)
195  return markAllAnalysesPreserved();
196 
197  // This pass requires a DUT.
198  if (!dut) {
199  circuit->emitError("no DUT module found.");
200  return signalPassFailure();
201  }
202 
203  auto &instanceGraph = getAnalysis<InstanceGraph>();
204 
205  // Find the LowestCommonAncestor node of all the nodes to be wired together,
206  // and collect all the ClockGate modules.
207  llvm::SetVector<InstanceRecord *> clockGates;
208  auto *lca = lowestCommonAncestor(
209  instanceGraph.lookup(dut), [&](InstanceRecord *node) {
210  auto module = node->getTarget()->getModule();
211  // If this is a clock gate, record the module and return true.
212  if (module.moduleName().startswith("EICG_wrapper")) {
213  clockGates.insert(node);
214  return true;
215  }
216  // Return true if this is the module with the enable signal.
217  return node->getParent()->getModule() == enableModule;
218  });
219 
220  // If there are no clock gates under the DUT, we can stop now.
221  if (!clockGates.size())
222  return;
223 
224  // Stash some useful things.
225  auto *context = &getContext();
226  auto uint1Type = enableSignal.getType().cast<FIRRTLType>();
227  auto loc = lca->getModule().getLoc();
228  auto portName = StringAttr::get(context, "test_en");
229 
230  // This maps an enable signal to each module.
231  DenseMap<InstanceGraphNode *, Value> signals;
232 
233  // Helper to insert a port into an instance op. We have to replace the whole
234  // op and then keep the instance graph updated.
235  auto insertPortIntoInstance =
236  [&](InstanceRecord *instanceNode,
237  std::pair<unsigned, PortInfo> port) -> InstanceOp {
238  auto instance = cast<InstanceOp>(*instanceNode->getInstance());
239  auto clone = instance.cloneAndInsertPorts({port});
240  instanceGraph.replaceInstance(instance, clone);
241  instance->replaceAllUsesWith(clone.getResults().drop_back());
242  instance->erase();
243  return clone;
244  };
245 
246  // At this point we have found the the enable signal, all important clock
247  // gates, and the ancestor of these. From here we need wire the enable signal
248  // upward to the LCA, and then wire the enable signal down to all clock
249  // gates.
250 
251  // This first part wires the enable signal up ward to the LCA module.
252  auto *node = instanceGraph.lookup(enableModule);
253  auto signal = enableSignal;
254  PortInfo portInfo = {portName, uint1Type, Direction::Out, {}, loc};
255  while (node != lca) {
256  // If there is more than one parent the we are in trouble. We can't handle
257  // more than one enable signal to wire everywhere else.
258  if (!node->hasOneUse()) {
259  auto diag = emitError(enableSignal.getLoc(),
260  "mutliple instantiations of the DFT enable signal");
261  auto it = node->usesBegin();
262  diag.attachNote((*it++)->getInstance()->getLoc())
263  << "first instance here";
264  diag.attachNote((*it)->getInstance()->getLoc()) << "second instance here";
265  return signalPassFailure();
266  }
267 
268  // Record the signal for this module.
269  signals[node] = signal;
270 
271  // Create an output port to this module.
272  auto module = cast<FModuleOp>(*node->getModule());
273  unsigned portNo = module.getNumPorts();
274  module.insertPorts({{portNo, portInfo}});
275  auto builder =
276  ImplicitLocOpBuilder::atBlockEnd(module.getLoc(), module.getBody());
277  builder.create<ConnectOp>(module.getArgument(portNo), signal);
278 
279  // Add an output port to the instance of this module.
280  auto *instanceNode = (*node->usesBegin());
281  auto clone = insertPortIntoInstance(instanceNode, {portNo, portInfo});
282 
283  // Set up for the next iteration.
284  signal = clone.getResult(portNo);
285  node = instanceNode->getParent();
286  }
287 
288  // Record the enable signal in the LCA.
289  signals[node] = signal;
290 
291  // Drill the enable signal to each of the leaf clock gates. We do this
292  // searching upward in the hiearchy until we find a module with the signal.
293  // This is a recursive function due to lazyness.
294  portInfo = {portName, uint1Type, Direction::In, {}, loc};
295  std::function<Value(InstanceGraphNode *)> getSignal =
296  [&](InstanceGraphNode *node) -> Value {
297  // Mutable signal reference.
298  auto &signal = signals[node];
299 
300  // Early break if this module has already been wired.
301  if (signal)
302  return signal;
303 
304  // Add an input signal to this module.
305  auto module = cast<FModuleOp>(*node->getModule());
306  unsigned portNo = module.getNumPorts();
307  module.insertPorts({{portNo, portInfo}});
308  auto arg = module.getArgument(portNo);
309 
310  // Record the new signal.
311  signal = arg;
312 
313  // Attach the input signal to each instance of this module.
314  for (auto *instanceNode : node->uses()) {
315  // Add an input signal to this instance op.
316  auto clone = insertPortIntoInstance(instanceNode, {portNo, portInfo});
317 
318  // Wire the parent signal to the instance op.
319  auto *parent = instanceNode->getParent();
320  auto module = cast<FModuleOp>(*parent->getModule());
321  auto signal = getSignal(parent);
322  auto builder =
323  ImplicitLocOpBuilder::atBlockEnd(module->getLoc(), module.getBody());
324  builder.create<ConnectOp>(clone.getResult(portNo), signal);
325  }
326 
327  return arg;
328  };
329 
330  // Wire the signal to each clock gate using the helper above.
331  for (auto *instance : clockGates) {
332  auto *parent = instance->getParent();
333  auto module = cast<FModuleOp>(*parent->getModule());
334  auto builder =
335  ImplicitLocOpBuilder::atBlockEnd(module->getLoc(), module.getBody());
336  // Hard coded port result number; the clock gate test_en port is 1.
337  auto testEnPortNo = 1;
338  builder.create<ConnectOp>(
339  cast<InstanceOp>(*instance->getInstance()).getResult(testEnPortNo),
340  getSignal(parent));
341  }
342 
343  // And we're done!
344  markAnalysesPreserved<InstanceGraph>();
345 }
