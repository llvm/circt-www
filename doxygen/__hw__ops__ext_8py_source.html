13 def create_parameters(parameters: dict[str, _ir.Attribute], module: ModuleLike):
14  # Compute mapping from parameter name to index, and initialize array.
15  mod_param_decls = module.parameters
16  mod_param_decls_idxs = {
17  decl.name: idx for (idx, decl) in enumerate(mod_param_decls)
18  }
19  inst_param_array = [None] * len(module.parameters)
20 
21  # Fill in all the parameters specified.
22  if isinstance(parameters, DictAttr):
23  parameters = {i.name: i.attr for i in parameters}
24  for (pname, pval) in parameters.items():
25  if pname not in mod_param_decls_idxs:
26  raise ValueError(
27  f"Could not find parameter '{pname}' in module parameter decls")
28  idx = mod_param_decls_idxs[pname]
29  param_decl = mod_param_decls[idx]
30  inst_param_array[idx] = hw.ParamDeclAttr.get(pname, param_decl.param_type,
31  pval)
32 
33  # Fill in the defaults from the module param decl.
34  for (idx, pval) in enumerate(inst_param_array):
35  if pval is not None:
36  continue
37  inst_param_array[idx] = mod_param_decls[idx]
38 
39  return inst_param_array

216 def _create_output_op(cls_name, output_ports, entry_block, bb_ret):
217  """Create the hw.OutputOp from the body_builder return."""
218 
219  # Determine if the body already has an output op.
220  block_len = len(entry_block.operations)
221  if block_len > 0:
222  last_op = entry_block.operations[block_len - 1]
223  if isinstance(last_op, hw.OutputOp):
224  # If it does, the return from body_builder must be None.
225  if bb_ret is not None and bb_ret != last_op:
226  raise support.ConnectionError(
227  f"In {cls_name}, cannot return value from body_builder and "
228  "create hw.OutputOp")
229  return
230 
231  # If builder didn't create an output op and didn't return anything, this op
232  # mustn't have any outputs.
233  if bb_ret is None:
234  if len(output_ports) == 0:
235  hw.OutputOp([])
236  return
237  raise support.ConnectionError(
238  f"In {cls_name}, must return module output values")
239 
240  # Now create the output op depending on the object type returned
241  outputs: list[Value] = list()
242 
243  # Only acceptable return is a dict of port, value mappings.
244  if not isinstance(bb_ret, dict):
245  raise support.ConnectionError(
246  f"In {cls_name}, can only return a dict of port, value mappings "
247  "from body_builder.")
248 
249  # A dict of `OutputPortName` -> ValueLike must be converted to a list in port
250  # order.
251  unconnected_ports = []
252  for (name, port_type) in output_ports:
253  if name not in bb_ret:
254  unconnected_ports.append(name)
255  outputs.append(None)
256  else:
257  val = support.get_value(bb_ret[name])
258  if val is None:
259  field_type = type(bb_ret[name])
260  raise TypeError(
261  f"In {cls_name}, body_builder return doesn't support type "
262  f"'{field_type}'")
263  if val.type != port_type:
264  if isinstance(port_type, hw.TypeAliasType) and \
265  port_type.inner_type == val.type:
266  val = hw.BitcastOp.create(port_type, val).result
267  else:
268  raise TypeError(
269  f"In {cls_name}, output port '{name}' type ({val.type}) doesn't "
270  f"match declared type ({port_type})")
271  outputs.append(val)
272  bb_ret.pop(name)
273  if len(unconnected_ports) > 0:
274  raise support.UnconnectedSignalError(cls_name, unconnected_ports)
275  if len(bb_ret) > 0:
276  raise support.ConnectionError(
277  f"Could not map the following to output ports in {cls_name}: " +
278  ",".join(bb_ret.keys()))
279 
280  hw.OutputOp(outputs)
