55 def var_to_attribute(obj, none_on_fail: bool = False) -> ir.Attribute:
56  """Create an MLIR attribute from a Python object for a few common cases."""
57  if isinstance(obj, ir.Attribute):
58  return obj
59  if isinstance(obj, bool):
60  return ir.BoolAttr.get(obj)
61  if isinstance(obj, int):
62  attrTy = ir.IntegerType.get_signless(64)
63  return ir.IntegerAttr.get(attrTy, obj)
64  if isinstance(obj, str):
65  return ir.StringAttr.get(obj)
66  if isinstance(obj, list):
67  arr = [var_to_attribute(x, none_on_fail) for x in obj]
68  if all(arr):
69  return ir.ArrayAttr.get(arr)
70  return None
71  if none_on_fail:
72  return None
73  raise TypeError(f"Cannot convert type '{type(obj)}' to MLIR attribute")

276 class NamedValueOpView:
277  """Helper class to incrementally construct an instance of an operation that
278  names its operands and results"""
279 
280  def __init__(self,
281  cls,
282  data_type=None,
283  input_port_mapping=None,
284  pre_args=None,
285  post_args=None,
286  needs_result_type=False,
287  **kwargs):
288  # Set defaults
289  if input_port_mapping is None:
290  input_port_mapping = {}
291  if pre_args is None:
292  pre_args = []
293  if post_args is None:
294  post_args = []
295 
296  # Set result_indices to name each result.
297  result_names = self.result_names()
298  result_indices = {}
299  for i in range(len(result_names)):
300  result_indices[result_names[i]] = i
301 
302  # Set operand_indices to name each operand. Give them an initial value,
303  # either from input_port_mapping or a default value.
304  backedges = {}
305  operand_indices = {}
306  operand_values = []
307  operand_names = self.operand_names()
308  for i in range(len(operand_names)):
309  arg_name = operand_names[i]
310  operand_indices[arg_name] = i
311  if arg_name in input_port_mapping:
312  value = get_value(input_port_mapping[arg_name])
313  operand = value
314  else:
315  backedge = self.create_default_value(i, data_type, arg_name)
316  backedges[i] = backedge
317  operand = backedge.result
318  operand_values.append(operand)
319 
320  # Some ops take a list of operand values rather than splatting them out.
321  if isinstance(data_type, list):
322  operand_values = [operand_values]
323 
324  # In many cases, result types are inferred, and we do not need to pass
325  # data_type to the underlying constructor. It must be provided to
326  # NamedValueOpView in cases where we need to build backedges, but should
327  # generally not be passed to the underlying constructor in this case. There
328  # are some oddball ops that must pass it, even when building backedges, and
329  # these set needs_result_type=True.
330  if data_type is not None and (needs_result_type or len(backedges) == 0):
331  pre_args.insert(0, data_type)
332 
333  self.opview = cls(*pre_args, *operand_values, *post_args, **kwargs)
334  self.operand_indices = operand_indices
335  self.result_indices = result_indices
336  self.backedges = backedges
337 
338  def __getattr__(self, name):
339  # Check for the attribute in the arg name set.
340  if "operand_indices" in dir(self) and name in self.operand_indices:
341  index = self.operand_indices[name]
342  value = self.opview.operands[index]
343  return OpOperand(self.opview.operation, index, value, self)
344 
345  # Check for the attribute in the result name set.
346  if "result_indices" in dir(self) and name in self.result_indices:
347  index = self.result_indices[name]
348  value = self.opview.results[index]
349  return OpOperand(self.opview.operation, index, value, self)
350 
351  # Forward "attributes" attribute from the operation.
352  if name == "attributes":
353  return self.opview.operation.attributes
354 
355  # If we fell through to here, the name isn't a result.
356  raise AttributeError(f"unknown port name {name}")
